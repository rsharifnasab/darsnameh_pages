ﻳﮏ ﺭﻭﻳﮑﺮﺩ ﺍﺳﺘﺎﻧﺪﺍﺭﺩ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﻓﻼﺳﻔﻪ ﺩﺭ ﺗﻤﺎﻳﺰ ﺑﻴﻦ ﺩﻭ ﺩﻳﺪﮔﺎﻩ ﻣﺨﺘﻠﻒ ﺩﺭ ﺍﺧﻼﻕ ﻣﯽ‌ﻳﺎﺑﻨﺪ: ﺍﺧﻼﻕ «ﻧﺘﻴﺠﻪ ﮔﺮﺍﻳﺎﻧﻪ» ﻭ ﺍﺧﻼﻕ «ﻓﻀﻴﻠﺖ‌ﮔﺮﺍﻳﺎﻧﻪ» ﮐﻪ ﺑﻪ ﺁﻥ ﺍﺧﻼﻕ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﻫﻢ ﻣﯽ‌ﮔﻮﻳﻨﺪ. ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺍﺯ ﺍﺳﻢ‌ﺍﺵ ﺑﺮ ﻣﯽ‌ﺁﻳﺪ، ﻧﺘﻴﺠﻪ‌ﮔﺮﺍﻳﯽ ﺍﺯ ﺍﻳﻦ ﺍﻳﺪﻩ ﻣﯽ‌ﺁﻳﺪ ﮐﻪ ﺩﺭﺳﺖ ﻭ ﻏﻠﻂ ﺑﻮﺩﻥ ﺍﻋﻤﺎﻝ ﺭﺍ ﺑﺎﻳﺪ ﺑﺮ ﺍﺳﺎﺱ ﻧﺘﺎﻳﺞ ﻭ ﭘﻴﺎﻣﺪﻫﺎﯼ ﺁﻧﻬﺎ ﻗﻀﺎﻭﺕ ﮐﺮﺩ. ﺍﮔﺮ ﺩﻭ ﻋﻤﻞ ﻣﻤﮑﻦ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺷﻤﺎ ﺑﺎﻳﺪ ﺁﻥ ﻋﻤﻠﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﺧﺮﻭﺟﯽ ﺑﻬﺘﺮﯼ ﺩﺍﺭﺩ. ﺑﻬﺘﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﻌﻨﺎﯼ: ﺍﻳﺠﺎﺩ ﺷﺎﺩﯼ ﺑﻴﺸﺘﺮ، ﮐﺎﻫﺶ ﺭﻧﺞ، ﺍﻣﮑﺎﻧﭙﺬﻳﺮ ﮐﺮﺩﻥ ﺍﻧﺘﺨﺎﺏ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮ ﻳﺎ ﺁﺯﺍﺩﯼ، ﺗﺮﻭﻳﺞ ﺧﻮﺩﻣﺨﺘﺎﺭﯼ ﺑﻴﺸﺘﺮ، ﭘﻴﺸﺮﻓﺖ ﮔﻮﻧﻪ‌ﻫﺎ ﺣﻴﺎﺗﯽ ﻳﺎ ﺑﺴﻴﺎﺭﯼ ﭘﻴﺎﻣﺪﻫﺎﯼ ﺩﻳﮕﺮ ﺑﺎﺷﺪ. ﻧﻈﺮﻳﻪ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺟﺰﺋﻴﺎﺕ «ﺑﻬﺘﺮ» ﺭﺍ ﺑﺎ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﭘﺮ ﻣﯽ‌ﮐﻨﻨﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺍﺻﻞ ﺯﻳﺮﻳﻦ ﻫﻤﻪ ﺁﻧﻬﺎ ﺁﻥ ﺍﺳﺖ ﮐﻪ «ﺩﺭﺳﺖ» ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﭼﻴﺰ ﺧﻮﺏ ﺭﺍ ﺑﻴﺸﺘﺮ ﻧﺘﻴﺠﻪ ﻣﯽ‌ﺩﻫﺪ، «ﻧﺎﺩﺭﺳﺖ» ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺑﻴﺸﺘﺮ ﻧﺘﻴﺠﻪ ﺑﺪ ﺗﻮﻟﻴﺪ ﻣﯽ‌ﮐﻨﺪ.

ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ ﺍﺧﻼﻕ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﻣﺪﻋﯽ ﺍﺳﺖ ﮐﻪ ﺍﺧﻼﻗﻴﺎﺕ ﺩﺭﺑﺎﺭﻩ ﺑﺮ ﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﻭﻇﻴﻔﻪ‌ﻫﺎ ﻭ ﺗﮑﺎﻟﻴﻒ ﺍﺳﺖ، ﻣﺴﺘﻘﻞ ﺍﺯ ﭘﻴﺎﻣﺪﻫﺎ ﻭ ﻧﺘﺎﻳﺠﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻣﺎ ﻭﻇﻴﻔﻪ ﺩﺍﺭﻳﻢ ﮐﻪ ﺍﻓﺮﺍﺩ ﺑﯽ‌ﮔﻨﺎﻩ ﺭﺍ ﻧﮑﺸﻴﻢ. ﺍﻳﻦ ﻭﻇﻴﻔﻪ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﺗﻤﺎﻳﻞ ﺑﻪ ﺣﻔﻆ ﺩﻳﮕﺮ ﺍﻓﺮﺍﺩ ﺑﯽ‌ﮔﻨﺎﻩ ﮐﻨﺎﺭ ﮔﺬﺍﺷﺘﻪ ﺷﻮﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺎ ﻣﺠﺒﻮﺭ ﺑﺎﺷﻴﻢ ﻣﺠﻤﻮﻋﻪ ﺍﻋﻤﺎﻟﯽ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﻧﺘﺎﻳﺞ ﺑﺪﯼ ﺷﻮﺩ ﻭ ﺍﻓﺮﺍﺩ ﺑﯽ‌ﮔﻨﺎﻩ ﺑﻴﺸﺘﺮﯼ ﺑﻤﻴﺮﻧﺪ، ﺍﻣﺎ ﺑﺎﻳﺪ ﮐﺎﺭ ﺩﺭﺳﺖ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ. ﭘﺲ ﺩﺭ ﺣﺎﻟﻴﮑﻪ ﺑﺮﺍﯼ ﻓﺮﺩ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﺧﻮﺏ ﻭ ﺑﺪ ﺑﻮﺩﻥ ﺍﻋﻤﺎﻝ ﺭﺍﺑﻄﻪ ﻧﺰﺩﻳﮑﯽ ﺑﺎ ﻫﻢ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﻫﻢ ﭘﻴﻮﻧﺪ ﺩﺍﺭﻧﺪ، ﺑﺮﺍﯼ ﻓﺮﺩ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍ ﺧﻮﺏ ﻭ ﺑﺪ ﺍﺯ ﻫﻢ ﻣﺴﺘﻘﻞ ﻫﺴﺘﻨﺪ. ﺍﻟﺒﺘﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﻧﻴﺎﻳﯽ ﮐﻪ ﻫﻤﻪ ﻣﺮﺩﻡ ﺩﺭ ﺁﻥ ﮐﺎﺭﻫﺎﯼ ﺩﺭﺳﺖ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻨﺪ ﺑﻴﺸﺘﺮ ﺍﺯ ﺟﻬﺎﻧﯽ ﮐﻪ ﮐﺎﺭ ﻧﺎﺩﺭﺳﺖ ﻣﯽ‌ﮐﻨﻨﺪ، ﺣﺎﻣﻞ ﭼﻴﺰﻫﺎﯼ ﺑﺪ ﺑﺎﺷﺪ.

 

ﻣﻌﻴﺎﺭﯼ ﺑﺮﺍﯼ ﻋﻤﻞ ﺍﺧﻼﻗﯽ

ﺧﺐ ﺍﻳﻦ ﻓﺮﻣﻮﻝ ﺻﺎﻑ ﻭ ﺳﺎﺩﻩ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺌﻠﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﮐﻨﺪ ﮐﻪ ﺩﺭ ﺁﺯﻣﻮﻥ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺍﺧﻼﻕ ﻧﻤﺮﻩ ﻗﺒﻮﻟﯽ ﺑﮕﻴﺮﻳﺪ، ﺍﻣﺎ ﻭﺍﻗﻌﺎ ﻣﺸﮑﻞ ﺍﺧﻼﻗﯽ ﺷﻤﺎ ﺭﺍ ﺣﻞ ﻧﻤﯽ‌ﮐﻨﺪ ﻭ ﺣﺘﯽ ﺩﺍﺭﺍﯼ ﻣﺸﮑﻼﺕ ﻧﻈﺮﯼ ﻫﻢ ﻫﺴﺖ. ﻣﺴﺌﻠﻪ ﺁﻧﺠﺎﺳﺖ ﮐﻪ ﺩﺭ ﺗﺼﻤﻴﻢ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺩﻧﻴﺎﯼ ﻭﺍﻗﻌﯽ ﻓﺮﺻﺖ ﺁﻥ ﺭﺍ ﻧﺪﺍﺭﻳﻢ ﮐﻪ ﻓﮑﺮ ﮐﻨﻴﻢ ﺁﻳﺎ ﻭﺍﻗﻌﺎ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﻫﺴﺘﻴﻢ ﻳﺎ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍ ﻭ ﺑﺎﻳﺪ ﺍﺯ ﻗﺒﻞ ﺩﻳﺪﮔﺎﻩ‌ ﺧﻮﺩ ﺭﺍ ﺭﻭﺷﻦ ﮐﺮﺩﻩ ﺑﺎﺷﻴﻢ. ﺍﻣﺎ ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ ﺗﺎ ﺑﺎ ﻣﺴﺌﻠﻪ ﻭﺍﻗﻌﯽ ﻣﻮﺍﺟﻪ ﻧﺸﻮﻳﻢ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﭘﺎﺳﺨﯽ ﺑﻪ ﺍﻳﻦ ﺑﺤﺚ ﺳﻄﺢ ﺑﺎﻻ ﺩﺭ ﺍﺧﻼﻕ ﺑﺪﻫﻴﻢ. ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻫﺰﺍﺭﺍﻥ ﺳﺎﻝ ﺍﺳﺖ ﻓﻼﺳﻔﻪ ﺍﺧﻼﻕ ﻫﻨﻮﺯ ﻧﺘﻮﺍﻧﺴﺘﻪ‌ﺍﻧﺪ ﺑﺮ ﺳﺮ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﺍﺟﻤﺎﻉ ﺑﺮﺳﻨﺪ ﮐﻪ ﮐﺪﺍﻡ ﭼﺎﺭﭼﻮﺏ ﻓﮑﺮﯼ ﺩﺭ ﺍﺧﻼﻕ ﺩﺭﺳﺖ ﺍﺳﺖ.

ﺣﺘﯽ ﺑﻪ ﻃﻮﺭ ﺑﻨﻴﺎﺩﯼ‌ﺗﺮ ﻣﯽ‌ﺗﻮﺍﻥ ﮔﻔﺖ ﺧﺎﺭﺝ ﺍﺯ ﺩﻧﻴﺎﯼ ﻧﻈﺮﻳﻪ‌ﻫﺎﯼ ﺁﮐﺎﺩﻣﻴﮏ ﻭ ﺧﺎﺭﺝ ﺍﺯ ﮐﻼﺱ ﺩﺭﺱ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﮐﺴﯽ ﺭﺍ ﺩﻳﺪ ﮐﻪ ﮐﺎﻣﻼ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﻳﺎ ﮐﺎﻣﻼ ﻓﻀﻴﻠﺖ‌ﮔﺮﺍ ﺑﺎﺷﺪ. ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎ ﻫﻤﻮﺍﺭﻩ ﺳﻌﯽ ﺩﺭ ﺍﻳﺠﺎﺩ ﺗﻌﺎﺩﻝ ﺑﻴﻦ ﻫﺪﻑ‌ﻫﺎ ﻭ ﻭﺳﻴﻠﻪ‌ﻫﺎ ﺩﺍﺭﻳﻢ ﻭ ﺑﺎ ﺑﺮﺭﺳﯽ ﺍﺩﻋﺎﻫﺎﯼ ﻫﺮ ﺩﻭ ﻣﻮﺿﻊ ﻫﻤﭽﻨﺎﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﺩﺭ ﺑﺮﻗﺮﺍﺭﯼ ﺗﻌﺎﺩﻝ ﺑﻴﻦ ﻫﺪﻑ ﻭ ﻭﺳﻴﻠﻪ ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﻢ. ﺑﻠﻪ ﻣﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﺎﺭﯼ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﮐﻪ ﺑﻬﺘﺮﻳﻦ ﻧﺘﻴﺠﻪ ﺭﺍ ﺩﺍﺭﺩ ﺍﻣﺎ ﻓﮑﺮ ﻧﻤﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﺻﺮﻓﺎ ﺑﻪ ﻣﺜﺎﺑﻪ ﻭﺳﻴﻠﻪ ﺭﻓﺘﺎﺭ ﮐﺮﺩ.

 

ﺩﻭﮔﺎﻧﻪ ﮐﺎﺫﺏ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍﻳﯽ ﻭ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﯽ
ﺍﻭﺿﺎﻉ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺍﻳﻦ ﻫﻢ ﺑﺪﺗﺮ ‌ﺷﻮﺩ. ﺣﺘﯽ ﺍﻳﻦ ﺭﻭﺷﻦ ﻧﻴﺴﺖ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﮐﺪﺍﻡ ﺩﻳﺪﮔﺎﻩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺴﺌﻠﻪ ﺭﺍ ﺣﻞ ﮐﻨﺪ. ﻣﺜﺎﻝ ﻫﻮﺍﭘﻴﻤﺎﯼ ﺩﺯﺩﻳﺪﻩ ﺷﺪﻩ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﻣﯽ‌ﺗﻮﺍﻥ ﺗﺼﻤﻴﻢ ﺷﻠﻴﮏ ﺑﻪ ﻫﻮﺍﭘﻴﻤﺎﯼ ﻣﺴﺎﻓﺮﺑﺮﯼ ﺭﺍ ﺑﺎ ﺍﺳﺘﺪﻻﻝ ﮐﺎﻣﻼ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﺗﻮﺟﻴﻪ ﮐﺮﺩ: ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﻭﻇﻴﻔﻪ ﭼﻨﯽ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﻌﺎﻭﻥ ﺭﻳﻴﺲ ﺟﻤﻬﻮﺭ ﺑﺮﺍﯼ ﺣﻔﻆ ﺟﺎﻥ ﺷﻬﺮﻭﻧﺪﺍﻥ ﺁﻣﺮﻳﮑﺎﻳﯽ ﺑﻪ ﻭﻇﻴﻔﻪ ﺍﻭ ﺑﺮ ﻧﮑﺸﺘﻦ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺑﯽ‌ﮔﻨﺎﻩ ﻣﻘﺪﻡ ﺍﺳﺖ. ﻳﺎ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﺳﺘﺪﻻﻝ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﺗﻮﺟﻴﻪ ﮐﺮﺩ ﮐﻪ ﺣﻤﻠﻪ ﺑﻪ ﻫﻮﺍﭘﻴﻤﺎ ﮐﺎﺭﯼ ﻏﻴﺮﺍﺧﻼﻗﯽ ﺍﺳﺖ:‌ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﮐﻪ،‌ ﺩﺭﺳﺖ ﺍﺳﺖ ﺩﺭ ﮐﻮﺗﺎﻩ ﻣﺪﺕ ﺳﻘﻮﻁ ﻫﻮﺍﭘﻴﻤﺎ ﺟﺎﻥ ﺍﻓﺮﺍﺩ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﻧﺠﺎﺕ ﻣﯽ‌ﺩﻫﺪ، ﺍﻣﺎ ﺍﻋﺘﻤﺎﺩ ﻣﺮﺩﻡ ﺭﺍ ﺑﻪ ﺩﻭﻟﺖ ﻣﻨﺘﺨﺐ ﺩﻣﻮﮐﺮﺍﺗﻴﮏ ﺍﺯ ﺑﻴﻦ ﻣﯽ‌ﺑﺮﺩ ﮐﻪ ﺑﺮ ﻋﻠﻴﻪ ﻣﺮﺩﻡ ﺧﻮﺩﺵ ﺍﺳﻠﺤﻪ ﮐﺸﻴﺪﻩ ﺍﺳﺖ.

ﭘﺲ ﺩﻳﺪﻳﻢ ﮐﻪ ﺑﻪ ﺳﺎﺩﮔﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﻳﮏ ﺗﺼﻤﻴﻢ ﺭﺍ ﺑﺎ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺩﻭ ﭼﺎﺭﭼﻮﺏ ﺍﺧﻼﻗﯽ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﻭ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍ ﺗﻮﺟﻴﻪ ﮐﺮﺩ. ﺩﻟﻴﻞ‌ﺍﺵ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺗﻤﺎﻳﺰ ﺑﻴﻦ ﺍﻳﻦ ﺩﻭ ﺩﻳﺪﮔﺎﻩ ﺍﺧﻼﻗﯽ ﺁﻧﻘﺪﺭﻫﺎ ﻋﻤﻴﻖ ﻧﻴﺴﺖ. ﺣﺘﯽ ﮐﻠﯽ‌ﺗﺮﻳﻦ ﺍﺻﻮﻝ ﺁﻧﻬﺎ ﺭﺍ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻧﻈﺮﻳﻪ ﺭﻗﻴﺐ ﺗﻮﺿﻴﺢ ﺩﺍﺩ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺍﮔﺮ ﺑﮕﻮﻳﻴﻢ ﻭﻇﻴﻔﻪ ﺩﺍﺭﻳﻢ ﺑﻬﺘﺮﻳﻦ ﻧﺘﻴﺠﻪ ﻣﻤﮑﻦ ﺭﺍ ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮﻳﻦ ﺗﻌﺪﺍﺩ ﺍﻓﺮﺍﺩ ﺗﻀﻤﻴﻦ ﮐﻨﻴﻢ، ﺍﺧﻼﻕ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﺭﺍ ﺑﺎ ﮐﻤﮏ ﺍﺻﻮﻝ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ‌ﺍﻳﻢ. ﻭ ﺍﮔﺮ ﺑﮕﻮﻳﻴﻢ ﻣﺎ ﺩﺭ ﺟﻬﺎﻥ ﺑﻬﺘﺮﯼ ﺯﻧﺪﮔﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ ﺍﮔﺮ ﺗﺼﻤﻴﻢ‌ﻫﺎﻳﻤﺎﻥ ﺑﺮ ﻣﺒﻨﺎﯼ ﭘﻴﺎﻣﺪﻫﺎﻳﯽ ﮐﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﺯ ﻳﮏ ﻋﻤﻞ ﺧﺎﺹ ﺣﺎﺻﻞ ﻣﯽ‌ﺷﻮﺩ، ﻧﺒﺎﺷﺪ. ﺍﻣﺎ ﺍﮔﺮ ﺑﻪ ﺍﺳﺘﻘﻼﻝ ﺍﻓﺮﺍﺩ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﻳﻢ، ﻣﺴﺌﻮﻟﻴﺖ‌ﻫﺎﻳﻤﺎﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﻭ ﺑﺎ ﺍﻓﺮﺍﺩ ﺑﻪ ﻋﻨﻮﺍﻥ ﻭﺳﻴﻠﻪ ﺭﻓﺘﺎﺭ ﻧﮑﻨﻴﻢ، ﺁﻥ ﮔﺎﻩ ﺩﺭ ﻭﺍﻗﻊ ﺍﺧﻼﻕ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﺭﺍ ﺑﺎ ﺗﻌﺎﺭﻳﻒ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ‌ﺍﻳﻢ.

ﺍﻣﺎ ﭘﺲ ﭼﺮﺍ ﺍﻳﻦ ﺩﻭ ﻧﻈﺮﻳﻪ ﺍﻳﻨﻘﺪﺭ ﻣﺘﻔﺎﻭﺕ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﻨﺪ؟ ﺷﺎﻳﺪ ﺑﻪ ﺩﻟﻴﻞ ﺗﺼﺎﺩﻑ ﺗﺎﺭﻳﺨﯽ ﺑﺎﺷﺪ. ﺳﻨﺖ ﺍﺧﻼﻗﯽ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﯽ ﺭﻳﺸﻪ ﺩﺭ ﺍﺧﻼﻕ ﺩﻳﻨﯽ ﺩﺍﺭﺩ، ﮐﻪ ﺩﺭ ﺁﻥ ﺧﺪﺍ ﻭﻇﻴﻔﻪ ﻭ ﺍﻭﻟﻮﻳﺖ ﺍﺻﻠﯽ ﺍﺳﺖ. ﺩﺭ ﻧﺴﺨﻪ‌ﻫﺎﯼ ﻣﺪﺭﻥ ﺍﻭﻟﻴﻪ ﺁﻥ ﻫﻢ، ﮐﻪ ﺗﻮﺳﻂ ﺍﻣﺎﻧﻮﺋﻞ ﮐﺎﻧﺖ ﺍﺭﺍﺋﻪ ﺷﺪ، ﺍﺻﻞ ﺭﺍﻫﻨﻤﺎ ﻭ ﺑﻨﻴﺎﺩﯼ ﻭﻇﻴﻔﻪ‌ﺍﯼ ﺟﺰ ﻋﻘﻞ ﻧﺒﻮﺩ.

ﺩﺭ ﻫﺮ ﺩﻭ ﺗﻔﺴﻴﺮ ﻭﻇﻴﻔﻪ ﺑﻪ ﺻﻮﺭﺕ ﭼﻴﺰﯼ ﻓﺮﺍﺗﺮ ﺍﺯ ﺭﻓﺎﻩ ﻭ ﺷﺎﺩﯼ ﺍﻧﺴﺎﻥ ﻓﺮﻣﻮﻝ‌ﺑﻨﺪﯼ ﺷﺪﻩ ﺑﻮﺩ. ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ، ﻧﺘﻴﺠﻪ‌ﮔﺮﺍﻳﯽ ﺑﺎ ﺑﺮﺁﻣﺪﻥ ﻓﺎﻳﺪﻩ‌ﮔﺮﺍﻳﯽ ﺗﻮﺳﻂ ﻓﻼﺳﻔﻪ‌ﺍﯼ ﭼﻮﻥ ﺟﺮﻣﯽ ﺑﻨﺘﺎﻡ ﻭ ﺁﺩﺍﻡ ﺍﺳﻤﻴﺖ ﺑﺮﺟﺴﺘﻪ ﺷﺪ، ﮐﻪ ﺑﻴﺸﻴﻨﻪ ﮐﺮﺩﻥ ﺷﺎﺩﯼ ﻭ ﮐﻤﻴﻨﻪ ﮐﺮﺩﻥ ﺭﻧﺞ ﺭﺍ ﺍﻟﺰﺍﻡ ﺍﺻﻠﯽ ﺍﺧﻼﻗﯽ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﺍﺩﻧﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻳﻦ ﮔﻮﻧﻪ ﺑﻪ ﻧﻈﺮ ﺭﺳﻴﺪ ﮐﻪ ﻳﮏ ﺗﻔﺎﻭﺕ ﺁﺷﮑﺎﺭ ﺑﻴﻦ ﺍﻳﻦ ﺩﻭ ﺭﻭﻳﮑﺮﺩ ﻭ ﭼﺎﺭﭼﻮﺏ ﻓﮑﺮﯼ ﺩﺭ ﻓﻠﺴﻔﻪ ﺍﺧﻼﻕ ﻭﺟﻮﺩ ﺩﺍﺭﺩ: ﺩﺭ ﻳﮏ ﻣﻮﺭﺩ ﺍﺧﻼﻗﻴﺎﺕ ﺑﻪ ﺩﻧﺒﺎﻝ ﺑﻴﺸﺘﺮ ﮐﺮﺩﻥ ﺭﻓﺎﻩ ﻭ ﺁﺳﺎﻳﺶ ﻭ ﺳﻌﺎﺩﺕ ﺍﻧﺴﺎﻥ ﺑﻮﺩ ﻭ ﺩﺭ ﺣﺎﻟﺖ ﺩﻳﮕﺮ ﻧﺒﻮﺩ.

 

ﺳﻌﺎﺩﺕ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﺳﺖ ﻳﺎ ﻭﻇﻴﻔﻪ؟
ﺍﻣﺎ ﺍﻳﻦ ﺗﻤﺎﻳﺰ ﺑﻴﻦ ﺍﺧﻼﻕ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﻭ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍ ﮔﻤﺮﺍﻩ ﮐﻨﻨﺪﻩ ﺍﺳﺖ. ﺭﻓﺎﻩ ﻭ ﺁﺳﺎﻳﺶ ﻭ ﺳﻌﺎﺩﺕ ﺍﻧﺴﺎﻥ ﻣﺴﻠﻤﺎ ﻫﻤﻴﺸﻪ ﺩﺭ ﻗﻠﺐ ﺍﺧﻼﻕ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﺑﻮﺩﻩ ﺍﺳﺖ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺧﺪﺍ ﺑﻬﺘﺮﻳﻦ ﻣﻨﻔﻌﺖ ﺭﺍ ﺑﺮﺍﯼ ﻣﺎ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻭ ﭘﻴﺮﻭﯼ ﺍﺯ ﺍﻭ ﺑﻪ ﻃﻮﺭ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﺑﻪ ﻣﻌﻨﺎﯼ ﺑﻴﺸﺘﺮ ﺷﺪﻥ ﺭﻓﺎﻩ ﺍﻧﺴﺎﻥ ﺍﺳﺖ. ﺑﻪ ﻃﻮﺭ ﻣﺸﺎﺑﻪ ﺩﺭ ﻣﻮﺭﺩ ﮐﺎﻧﺖ ﻭ ﻓﺮﻣﺎﻥ ﻗﺎﻃﻊ ﺍﻭ (ﻳﻌﻨﯽ ﺍﻳﻦ ﺍﻳﺪﻩ ﮐﻪ ﺑﺎﻳﺪ ﮐﺎﺭﯼ ﮐﻨﻴﻢ ﮐﻪ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻳﮏ ﻗﺎﻧﻮﻥ ﺟﻬﺎﻧﺸﻤﻮﻝ ﺑﺮﺍﯼ ﻫﻤﻪ ﻣﺎ ﺑﺎﺷﺪ) ﺩﺭ ﺍﺻﻞ ﻣﯽ‌ﮔﻮﻳﺪ ﮐﻪ ﭼﻪ ﭼﻴﺰﯼ ﺑﺮﺍﯼ ﻫﻤﻪ ﺑﻬﺘﺮ ﺍﺳﺖ.

ﻓﺎﻳﺪﻩ‌ﮔﺮﺍﻳﯽ ﺑﻪ ﻧﻈﺮ ﺑﺴﻴﺎﺭ ﻣﺘﻔﺎﻭﺕ ﻣﯽ‌ﺭﺳﻴﺪ، ﻧﻪ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺑﺎ ﺭﻓﺎﻩ ﺍﻧﺴﺎﻥ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺷﺖ، ﺑﻠﮑﻪ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺭﻓﺎﻩ ﺭﺍ ﺑﻪ ﺷﮑﻠﯽ ﻣﺤﺪﻭﺩ ﻭ ﻟﺬﺕ‌ﮔﺮﺍﻳﺎﻧﻪ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﮐﺮﺩ: ﺑﻴﺸﺘﺮﻳﻦ ﺷﺎﺩﯼ ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮﻳﻦ ﺍﻓﺮﺍﺩ. ﺍﻣﺎ ﺍﮔﺮ ﮐﺴﯽ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﻔﻬﻮﻡ ﻭﺳﻴﻊ‌ﺗﺮﯼ ﻣﺜﻞ ﺷﮑﻮﻓﺎﻳﯽ ﻭ ﺭﺷﺪ ﺍﻧﺴﺎﻥ ﺑﺎﺷﺪ، ﺑﻪ ﻧﺎﮔﺎﻩ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﯽ ﺑﻪ ﻧﻈﺮ ﭼﺎﺭﭼﻮﺏ ﺑﻬﺘﺮﯼ ﻣﯽ‌ﺭﺳﺪ. ﺩﺭ ﺁﻥ ﺻﻮﺭﺕ ﻣﺎ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﻪ ﺻﺮﻓﺎ ﺷﺎﺩ ﺑﺎﺷﻴﻢ، ﻣﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﻪ ﺍﺳﺘﻘﻼﻝ ﻭ ﺍﻧﺴﺎﻧﻴﺖ‌ﻣﺎﻥ ﻫﻢ ﻣﻮﺭﺩ ﺍﺣﺘﺮﺍﻡ ﻗﺮﺍﺭ ﮔﻴﺮﺩ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭﺩ ﺑﮑﺸﻴﻢ، ﻳﺎ ﺣﺘﯽ ﺑﻤﻴﺮﻳﻢ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﮐﺮﺍﻣﺖ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺍﻧﺴﺎﻥ ﺣﻔﻆ ﮐﻨﻴﻢ ﺗﺎ ﺍﻳﻨﮑﻪ ﺑﺨﻮﺍﻫﻴﻢ ﺁﻥ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﻳﮏ ﺷﺎﺩﯼ ﻣﺼﻨﻮﻋﯽ ﺍﺯ ﺩﺳﺖ ﺑﺪﻫﻴﻢ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﺧﻼﻕ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍﻳﺎﻧﻪ ﻫﻢ ﺑﻪ ﺩﻧﺒﺎﻝ ﺭﻓﺎﻩ ﺍﻧﺴﺎﻥ ﺍﺳﺖ ﺍﻣﺎ ﺁﻥ ﺭﺍ ﺑﻪ ﻣﻌﻨﺎﯼ ﻭﺳﻴﻊ‌ﺗﺮﯼ ﺍﺯ ﺁﻧﭽﻪ ﮐﻪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺩﺭ ﻧﻈﺮ ﻣﯽ‌ﮔﻴﺮﻳﻢ، ﻣﯽ‌ﺩﺍﻧﺪ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﮔﺮ ﺗﻤﺎﻳﺰ ﺑﻴﻦ ﻫﺪﻑ ﻭ ﻭﺳﻴﻠﻪ ﭼﻴﺰﯼ ﻧﻴﺴﺖ ﮐﻪ ﺍﺧﻼﻕ ﻧﺘﻴﺠﻪ‌ﮔﺮﺍ ﻭ ﻭﻇﻴﻔﻪ‌ﮔﺮﺍ ﺭﺍ ﺍﺯ ﻫﻢ ﺟﺪﺍ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻫﺮ ﺩﻭﯼ ﺍﻳﻦ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎ ﺑﻪ ﻃﻮﺭ ﻣﺴﺎﻭﯼ ﻧﮕﺮﺍﻥ ﭘﻴﺎﻣﺪﻫﺎﯼ ﺍﻋﻤﺎﻝ ﺑﺮ ﺭﻓﺎﻩ ﺍﻧﺴﺎﻥ ﻫﺴﺘﻨﺪ، ﺁﻧﮕﺎﻩ ﺷﺎﻳﺪ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﺑﺮﺍﯼ ﺍﺳﺘﺪﻻﻝ ﺍﺧﻼﻗﯽ ﮐﺎﺭﺁﻣﺪ ﻧﻴﺴﺘﻨﺪ. ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺳﻮﺍﻝ ﺩﺭ ﻣﻮﺭﺩ ﻫﺪﻑ ﻭ ﻭﻇﻴﻔﻪ ﺭﺍ ﺍﺯ ﻧﻈﺮ ﺍﺧﻼﻗﯽ ﺟﺬﺍﺏ ﻣﯽ‌ﮐﻨﺪ ﭘﺎﺭﺍﺩﻭﮐﺲ‌ﻫﺎ ﻳﺎ ﺗﻀﺎﺩﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺁﻥ ﺑﺮ ﻣﯽ‌ﺁﻳﺪ: ﺍﻣﮑﺎﻥ ﺍﻧﺠﺎﻡ ﮐﺎﺭ ﻧﺎﺩﺭﺳﺖ ﺑﺮﺍﯼ ﻳﮏ ﮐﺎﺭ ﺩﺭﺳﺖ، ﻳﺎ ﺍﻧﺠﺎﻡ ﮐﺎﺭ ﺑﺪ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻫﺪﻓﯽ ﺧﻮﺏ. ﺍﻣﺎ ﺑﺎ ﻧﮕﺎﻩ ﻧﺰﺩﻳﮑﺘﺮ ﻭ ﺩﻗﻴﻖ‌ﺗﺮ ﺍﻳﻦ ﺗﻨﺎﻗﺾ‌ﻫﺎﯼ ﺁﺷﮑﺎﺭ ﺣﻞ ﻣﯽ‌ﺷﻮﻧﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺗﺼﻤﻴﻢ‌ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﺍﺧﻼﻗﯽ ﭼﻨﻴﻦ ﺗﻘﺴﻴﻢ‌ﺑﻨﺪﯼ‌ﻫﺎﻳﯽ ﻧﻤﯽ‌ﮐﻨﻴﻢ ﻭ ﻭﻗﺘﯽ ﻋﻤﻞ ﺑﺪﯼ ﺭﺍ ﺗﻮﺟﻴﻪ ﮐﺮﺩﻳﻢ ﺩﻳﮕﺮ ﺁﻥ ﺭﺍ ﺑﺪ ﻧﻤﯽ‌ﺩﺍﻧﻴﻢ، ﻭﮔﺮﻧﻪ ﺍﻧﺠﺎﻣﺶ ﻧﻤﯽ‌ﺩﺍﺩﻳﻢ. ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺟﻬﺎﻥ ﻋﻤﻞ ﭘﺎﺭﺍﺩﻭﮐﺲ ﻧﻈﺮﯼ ﺭﺍ ﺣﻞ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺑﻪ ﻣﺤﺾ ﺍﻳﻨﮑﻪ ﻭﺍﻗﻌﺎ ﺩﺭﮔﻴﺮ ﺣﻞ ﻣﻌﻤﺎﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺷﻮﻳﻢ، ﻓﻮﺭﺍ ﺍﻳﻦ ﻣﺸﮑﻼﺕ ﻧﻈﺮﯼ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻣﯽ‌ﮐﻨﻴﻢ.
