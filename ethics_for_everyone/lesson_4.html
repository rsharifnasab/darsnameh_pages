ﺩﺭﺱ ﺳﻮﻡ- ﺩﺭ ﺟﺴﺘﺠﻮﯼ ﻣﻨﺒﻊ ﺍﺧﻼﻕ ﺩﺭ ﺟﻬﺎﻥ ﺑﺪﻭﻥ ﺍﺭﺯﺵ

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺩﻳﺪﻳﻢ، ﺍﺧﻼﻕ ﺑﺎ ﻫﻤﺎﻥ ﺳﻮﺍﻝ‌ﻫﺎﻳﯽ ﺩﺭﮔﻴﺮ ﺍﺳﺖ ﮐﻪ ﻣﺎ ﻫﺮ ﺭﻭﺯ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺑﺎ ﺁﻧﻬﺎ ﺩﺳﺖ ﺑﻪ ﮔﺮﻳﺒﺎﻧﻴﻢ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﻴﻢ ﺧﻮﺩﻣﺎﻥ ﻣﺴﺘﻘﻼﻧﻪ ﺑﻴﻨﺪﻳﺸﻴﻢ ﻭ ﻗﻀﺎﻭﺕ‌ﻫﺎ ﻭ ﻣﻮﺍﺿﻊ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ. ﺍﻣﺎ ﺍﺯ ﮐﺠﺎ ﺑﺪﺍﻧﻴﻢ ﺍﻳﻦ ﻗﻀﺎﻭﺕ‌ﻫﺎ ﻭ ﻣﻮﺍﺿﻊ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﻳﺎ ﻧﻪ؟ ﺍﺯ ﮐﺠﺎ ﺑﺪﺍﻧﻴﻢ ﻧﻈﺮﻣﺎﻥ ﺩﺭ ﻣﻮﺭﺩ ﻣﺴﺎﺋﻞ ﺍﺧﻼﻗﯽ ﺍﺷﺘﺒﺎﻩ ﻧﻴﺴﺖ؟ ﺍﺯ ﮐﺠﺎ ﺑﺪﺍﻧﻴﻢ ﺩﺭ ﻣﻮﺍﺟﻬﻪ ﺑﺎ ﺩﻭﺭﺍﻫﯽ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺗﺼﻤﻴﻢ‌ﻫﺎﯼ ﺩﺭﺳﺘﯽ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﻳﺎ ﻧﻪ؟

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺁﻳﺎ ﺩﺭﺳﺖ ﺍﺳﺖ ﮐﻪ ﻣﺠﺎﺯﺍﺕ ﺍﻋﺪﺍﻡ ﺭﺍ ﺩﺭ ﻫﺮ ﻣﻮﺭﺩ ﻭ ﺩﺭ ﻫﻤﻪ ﺷﺮﺍﻳﻂ ﻧﺎﺩﺭﺳﺖ ﻭ ﻏﻴﺮﺍﺧﻼﻗﯽ ﺑﺪﺍﻧﻴﻢ،‌ ﺣﺘﯽ ﺍﮔﺮ ﺑﺤﺚ ﺑﺮ ﺳﺮ ﺍﻋﺪﺍﻡ ﻋﺎﻣﻞ ﻳﮏ ﺟﻨﺎﻳﺖ ﺍﻧﺴﺎﻧﯽ ﻭ ﻧﺴﻞ‌ﮐﺸﯽ ﺑﺎﺷﺪ؟ ﺍﺻﻼ ﺩﺭ ﺣﻮﺯﻩ‌ﺍﯼ ﻣﺎﻧﻨﺪ ﺍﺧﻼﻕ ﮐﻪ ﺑﺎ ﻃﻴﻒ ﻭﺳﻴﻌﯽ ﺍﺯ ﻧﻈﺮﻫﺎ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﺩ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺩﺭﺳﺖ ﻳﺎ ﻏﻠﻂ ﺑﻮﺩﻥ ﻣﻮﺍﺿﻊ ﺣﺮﻑ ﺯﺩ؟ ﺁﻳﺎ ﻭﺍﻗﻌﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﻣﻮﺭﺩ ﻣﺠﺎﺯﺍﺕ ﺍﻋﺪﺍﻡ ﺑﻪ ﻫﻤﻴﻦ ﺭﺍﺣﺘﯽ ﮔﻔﺖ ﮐﻪ ﻣﻮﺿﻊ ﺩﺭﺳﺖ ﻣﻤﻨﻮﻋﻴﺖ ﻣﺠﺎﺯﺍﺕ ﺍﻋﺪﺍﻡ ﺍﺳﺖ؟ ﻭ ﺍﻳﻦ ﻣﻮﺿﻊ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﺗﻌﺮﻳﻒ ﺩﺭ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﺮﺟﻊ ﻧﻮﺷﺖ؟ ﻗﻄﻌﺎ ﺍﻳﻦ ﻃﻮﺭ ﻧﻴﺴﺖ ﻭ ﺩﻗﻴﻘﺎ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺳﻮﺍﻝ‌ﻫﺎ ﻭ ﺑﺤﺚ‌ﻫﺎﻳﯽ ﺷﺒﻴﻪ ﺑﻪ ﺍﻳﻦ ﻫﻤﭽﻨﺎﻥ ﺩﺭ ﻫﻤﻪ ﻓﺮﻫﻨﮓ‌ﻫﺎ ﻭ ﺟﻮﺍﻣﻊ ﺍﺩﺍﻣﻪ ﺩﺍﺭﻧﺪ.

ﺑﺮﺍﯼ ﺁﺷﻨﺎﻳﯽ ﺑﻴﺸﺘﺮ ﺑﺎ ﺍﻳﻦ ﻣﺴﺎﺋﻞ ﻭ ﺩﻭﺭﺍﻫﯽ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﻻﺯﻡ ﻧﻴﺴﺖ ﺭﺍﻩ ﺩﻭﺭﯼ ﺑﺮﻭﻳﻢ ﻭ ﮐﺎﻓﯽ ﺍﺳﺖ ﺑﻪ ﻧﮕﺎﻫﯽ ﺑﻪ ﻫﻤﻴﻦ ﺍﺧﺒﺎﺭ ﺭﻭﺯﻣﺮﻩ ﺑﻴﻨﺪﺍﺯﻳﻢ. ﻫﻤﻮﺍﺭﻩ ﻣﯽ‌ﺑﻴﻨﻴﻢ ﮐﻪ ﺑﺤﺚ‌ﻫﺎﻳﯽ ﺣﻮﻝ ﻭ ﺣﻮﺵ ﺧﺒﺮﻫﺎﯼ ﺭﻭﺯ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﺴﺘﻪ ﺑﻪ ﺟﻨﺠﺎﻝ‌ ﺑﺮﺍﻧﮕﻴﺰ ﺑﻮﺩﻥ ﺍﺧﺒﺎﺭ ﺩﺷﻮﺍﺭﺗﺮ ﻧﻴﺰ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺑﺤﺚ ﺑﺮ ﺳﺮ ﺩﻭﮔﺎﻧﻪ ﺳﺎﻧﺴﻮﺭ ﻭ ﺁﺯﺍﺩﯼ ﺑﻴﺎﻥ ﺍﮔﺮ ﺑﺎ ﺍﻓﺸﺎﯼ ﺍﺳﺮﺍﺭ ﺗﻮﻃﺌﻪ‌ﻫﺎﯼ ﺣﮑﻮﻣﺘﯽ ﻳﺎ ﺷﻨﻮﺩ ﻣﮑﺎﻟﻤﺎﺕ ﺧﺼﻮﺻﯽ ﺷﻬﺮﻭﻧﺪﺍﻥ ﻫﻤﺮﺍﻩ ﺑﺎﺷﺪ، ﺩﻳﮕﺮ ﺁﻧﻘﺪﺭﻫﺎ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﺳﺎﺩﻩ‌ﺍﯼ ﻧﻴﺴﺖ. ﻋﻼﻭﻩ ﺑﺮ ﺧﺒﺮﻫﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺷﺨﺼﯽ ﺧﻮﺩﻣﺎﻥ ﻧﻴﺰ ﻫﻤﻮﺍﺭﻩ ﺑﺮ ﺳﺮ ﺩﻭﺭﺍﻫﯽ ﻗﺮﺍﺭ ﺩﺍﺭﻳﻢ. ﺍﮔﺮ ﺩﮐﺘﺮ ﻣﻌﺎﻟﺞ ﻣﺎﺩﺭﻣﺎﻥ ﺑﮕﻮﻳﺪ ﮐﻪ ﺍﻭ ﺑﻴﻤﺎﺭﯼ ﺑﺴﻴﺎﺭ ﻭﺧﻴﻤﯽ ﺩﺍﺭﺩ ﮐﻪ ﺗﺎ ﭼﻨﺪ ﻣﺎﻩ ﺩﻳﮕﺮ ﺑﻴﺸﺘﺮ ﺩﻭﺍﻡ ﻧﻤﯽ‌ﺁﻭﺭﺩ، ﺑﺎﻳﺪ ﺑﻴﻦ ﺍﻳﻨﮑﻪ ﺁﻳﺎ ﺑﺎﻳﺪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﻢ ﻳﺎ ﻣﺨﻔﯽ ﮐﻨﻴﻢ ﺗﺎ ﺁﺯﺍﺭ ﻧﺒﻴﻨﺪ، ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﻢ. ﺩﺷﻮﺍﺭﯼ ﺍﻳﻦ ﺩﻭﺭﺍﻫﯽ ﺍﺧﻼﻗﯽ ﻭﻗﺘﯽ ﺁﺷﮑﺎﺭﺗﺮ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻓﺮﺽ ﮐﻨﻴﻢ ﻣﺎﺩﺭﻣﺎﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎ ﺩﺍﻧﺴﺘﻦ ﺍﻳﻨﮑﻪ ﺑﻴﻤﺎﺭﯼ‌‌ﺍﺵ ﺳﺨﺖ ﻭ ﻣﺮﮔﺒﺎﺭ ﺍﺳﺖ ﺭﻭﺣﻴﻪ ﺑﮕﻴﺮﺩ ﻭ ﺑﺨﻮﺍﻫﺪ ﻣﻘﺎﻭﻣﺖ ﮐﻨﺪ، ﺍﻣﺎ ﻫﻤﺰﻣﺎﻥ ﺧﻄﺮ ﻧﺎﺍﻣﻴﺪﯼ ﻭ ﺗﺴﻠﻴﻢ ﺷﺪﻥ ﺯﻭﺩ ﻫﻨﮕﺎﻡ ﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

 

ﺍﺧﻼﻕ ﺑﻪ ﻋﻨﻮﺍﻥ ﺷﺎﺧﻪ‌ﺍﯼ ﺍﺯ ﻓﻠﺴﻔﻪ
ﭘﺲ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻣﻄﻤﺌﻦ ﺷﻮﻳﻢ ﻧﻈﺮﻣﺎﻥ ﺩﺭﺳﺖ ﺍﺳﺖ ﻭ ﺩﺭﺳﺖ ﻋﻤﻞ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺗﺼﻤﻴﻢ‌ﻫﺎﯼ ﺩﺭﺳﺘﯽ ﺩﺭ ﻣﻮﺍﻗﻊ ﺣﺴﺎﺱ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﭼﻪ ﮐﺎﺭﯼ ﺑﺎﻳﺪ ﮐﻨﻴﻢ؟ ﺁﻳﺎ ﺍﺳﺘﺎﻧﺪﺍﺭﺩﯼ ﺑﺮﺍﯼ ﻋﻤﻠﮑﺮﺩ ﺩﺭﺳﺖ ﺑﺮﺍﯼ ﻣﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ؟ ﺁﻳﺎ ﮐﺘﺎﺑﯽ ﻫﺴﺖ ﮐﻪ ﺑﻪ ﻋﻨﻮﺍﻥ ﮐﻠﻴﺪ ﺣﻞ ﻫﻤﻪ ﻣﺸﮑﻼﺕ ﺩﺭ ﮐﺘﺎﺑﺨﺎﻧﻪ‌ﻣﺎﻥ ﺑﮕﺬﺍﺭﻳﻢ ﻭ ﺩﺭ ﻣﻮﺍﻗﻊ ﻟﺰﻭﻡ ﺑﻪ ﺁﻥ ﻣﺮﺍﺟﻌﻪ ﮐﻨﻴﻢ؟ ﭼﻪ ﺑﺨﻮﺍﻫﻴﻢ ﻭ ﭼﻪ ﻧﺨﻮﺍﻫﻴﻢ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺍﻝ‌‌ﻫﺎ ﻭ ﺍﻳﻦ ﻟﺤﻈﻪ‌ﻫﺎﯼ ﻣﻬﻢ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﻫﻤﻴﺸﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺑﺮ ﺳﺮ ﺭﺍﻩ ﻣﺎ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺩﻗﻴﻘﺎ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺍﺧﻼﻕ ﻫﻤﻴﺸﻪ ﻳﮑﯽ ﺍﺯ ﺷﺎﺧﻪ‌ﻫﺎﯼ ﺍﺻﻠﯽ ﻓﻠﺴﻔﻪ ﺑﻮﺩﻩ ﺍﺳﺖ. ﺍﻳﻦ ﺳﻮﺍﻻﺕ ﻣﺎﻧﻨﺪ ﻓﻠﺴﻔﻪ، ﺫﻫﻦ ﻣﺎ ﺭﺍ ﺑﻪ ﭼﺎﻟﺶ ﻣﯽ‌ﮐﺸﻨﺪ ﻭ ﻧﻴﺎﺯﻣﻨﺪ ﺍﻧﺪﻳﺸﻪ ﻋﻤﻴﻖ ﻭ ﺍﺳﺘﺪﻻﻝ ﻫﺴﺘﻨﺪ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﻭﺭﻩ‌ﯼ ﻓﻠﺴﻔﻪ ﺑﺮﺍﯼ ﻫﻤﻪ ﺩﺭ ﺩﺭﺳﻨﺎﻣﻪ ﻫﻢ ﺩﻳﺪﻳﻢ ﻓﻼﺳﻔﻪ ﺑﺰﺭﮒ ﺑﺮﺍﯼ ﻫﺮ ﺳﻮﺍﻟﯽ ﻣﺠﻤﻮﻋﻪ ﻣﻮﺍﺿﻊ ﻭ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎﯼ ﻣﺸﺨﺺ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺩﺍﺭﻧﺪ ﻭ ﺑﺴﺘﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﺩﺭ ﮐﺪﺍﻡ ﭼﺎﺭﭼﻮﺏ ﻓﮑﺮﯼ ﻭ ﻓﻠﺴﻔﯽ، ﮐﻪ ﺑﻪ ﺍﺻﻄﻼﺡ ﻣﮑﺘﺐ ﻓﻠﺴﻔﯽ ﻧﺎﻡ ﺩﺍﺭﺩ، ﻗﺮﺍﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﭘﺎﺳﺦ ﻣﺘﻔﺎﻭﺗﯽ ﺑﻪ ﺳﻮﺍﻝ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﻣﺠﺎﺯﺍﺕ ﺍﻋﺪﺍﻡ، ﮐﻤﮏ ﺑﻪ ﺧﻴﺮﻳﻪ ﻳﺎ ﺳﻮﺍﻻﺕ ﺩﻳﮕﺮ ﺑﺪﻫﻨﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺑﻪ ﺁﻥ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﺍﻳﻦ ﻓﻼﺳﻔﻪ ﺗﻌﺼﺐ‌ﻫﺎ ﻭ ﺩﺍﻭﺭﯼ‌ﻫﺎﯼ ﺧﺎﺹ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺩﺍﺭﻧﺪ ﻭ ﺟﻮﺍﺏ‌ﻫﺎﻳﺸﺎﻥ ﺩﺭ ﭼﺎﺭﭼﻮﺑﯽ ﺻﻠﺐ ﻭ ﻏﻴﺮﻗﺎﺑﻞ ﺗﻐﻴﻴﺮ ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﻗﺒﻼ ﻫﻢ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﻓﻼﺳﻔﻪ ﺍﺯ ﺳﻘﺮﺍﻁ ﮔﺮﻓﺘﻪ ﺗﺎ ﻫﻴﻮﻡ ﻭ ﮐﺎﻧﺖ ﻭ ﺍﺑﻦ ﺳﻴﻨﺎ، ﺑﺮﺍﯼ ﻣﻮﺍﺿﻊ ﺧﻮﺩ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﻗﻮﯼ ﺍﺯ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﻳﯽ ﺩﺍﺭﻧﺪ ﮐﻪ ﻧﻈﺮ ﺁﻧﻬﺎ ﺭﺍ ﭘﺸﺘﻴﺒﺎﻧﯽ ﻣﯽ‌ﮐﻨﺪ.

 

ﺍﺯ ﻫﺴﺖ ﺗﺎ ﺑﺎﻳﺪ
ﭘﺲ ﺍﮔﺮ ﺍﺧﻼﻕ ﺣﻮﺯﻩ ﺍﺻﻠﯽ ﻓﻠﺴﻔﻪ ﺍﺳﺖ ﻭ ﺑﺮﺍﯼ ﻫﺮ ﺳﻮﺍﻝ ﺍﺧﻼﻗﯽ ﺑﺎﻳﺪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺩﻟﻴﻞ ﻣﻮﺟﻪ ﻭ ﺑﺮﻫﺎﻥ ﻣﺒﺘﻨﯽ ﺑﺮ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻭ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ ﺑﺎﺷﻴﻢ، ﺑﺒﻴﻨﻴﻢ ﻓﻼﺳﻔﻪ ﭼﻄﻮﺭ ﺍﺯ ﭘﺲ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺑﺮﺁﻣﺪﻩ‌ﺍﻧﺪ ﻭ ﻣﯽ‌ﺁﻳﻨﺪ. ﺍﺻﻼ ﺍﺯ ﮐﺠﺎ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺑﻪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺑﺮﺳﻨﺪ ﻭ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎﻳﺪﻫﺎ ﻭ ﻧﺒﺎﻳﺪ‌ﻫﺎﯼ ﺍﺧﻼﻕ ﺭﺍ ﺑﺎ ﺩﻟﻴﻞ ﻭ ﺑﺮﻫﺎﻥ ﻣﻮﺟﻪ ﮐﺮﺩ؟

ﺍﻳﻦ ﻣﻬﻤﺘﺮﻳﻦ ﺳﻮﺍﻝ ﺩﻳﻮﻳﺪ ﻫﻴﻮﻡ، ﻓﻴﻠﺴﻮﻑ ﺍﺳﮑﺎﺗﻠﻨﺪﯼ ﻗﺮﻥ ۱۸، ﺑﻮﺩ. ﺍﻭ ﻣﯽ‌ﺧﻮﺍﺳﺖ ﺑﺪﺍﻧﺪ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺗﻮﺻﻴﻒ ﺩﺭ ﺟﻬﺎﻥ ﻣﺎﻧﻨﺪ ﺍﻳﻨﮑﻪ ﭼﻴﺰﻫﺎ ﭼﮕﻮﻧﻪ ﻫﺴﺘﻨﺪ ﺑﻪ ﺗﻮﺻﻴﻪ‌ﻫﺎ ﻭ ﺩﺳﺘﻮﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﻫﻨﺠﺎﺭﯼ ﺑﺮﺳﻴﻢ. ﺍﻳﻦ ﺳﻮﺍﻝ ﻫﻴﻮﻡ ﺑﻪ ﻣﺴﺌﻠﻪ ﺭﺳﻴﺪﻥ ﺍﺯ «ﻫﺴﺖ‌ﻫﺎ ﻭ ﻧﻴﺴﺖ‌ﻫﺎ» ﺑﻪ «ﺑﺎﻳﺪ‌ﻫﺎ ﻭ ﻧﺒﺎﻳﺪ‌ﻫﺎ» ﻣﻌﺮﻭﻑ ﺍﺳﺖ. ﺍﻭ ﺧﻮﺩ ﻣﻌﺘﻘﺪ ﺑﻮﺩ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺟﻬﺎﻥ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎ ﺑﻪ ﺟﻬﺎﻥ ﺍﺭﺯﺵ‌ﻫﺎ ﺭﺳﻴﺪ ﻭ ﺍﺯ ﻧﻈﺮ ﻓﻠﺴﻔﯽ ﭼﻨﻴﻦ ﺍﺳﺘﺪﻻﻟﯽ ﻏﻴﺮﻗﺎﺑﻞ ﻗﺒﻮﻝ ﺍﺳﺖ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻣﺎ ﺗﻨﻬﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﮕﻮﻳﻴﻢ ﺧﻮﺭﺷﻴﺪ ﻫﺮ ﺭﻭﺯ ﺍﺯ ﺷﺮﻕ ﻃﻠﻮﻉ ﻣﯽ‌ﮐﻨﺪ، ﺍﻣﺎ ﺍﻳﻦ ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﺧﻮﺭﺷﻴﺪ ﻫﺮ ﺭﻭﺯ «ﺑﺎﻳﺪ» ﺍﺯ ﺷﺮﻕ ﻃﻠﻮﻉ ﮐﻨﺪ. ﺍﺯ ﻧﻈﺮ ﻫﻴﻮﻡ ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻧﺪﻳﺸﻤﻨﺪﺍﻥ ﭘﺲ ﺍﺯ ﺍﻭ ﭼﻨﻴﻦ ﺍﺳﺘﻨﺘﺎﺟﯽ ﻏﻴﺮﻗﺎﺑﻞ ﻗﺒﻮﻝ ﺍﺳﺖ.

ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻳﮑﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﻣﺒﺎﺣﺚ ﻓﻠﺴﻔﻪ ﺍﺧﻼﻕ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺍﺳﺘﺪﻻﻝ ﻋﻘﻼﻧﯽ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺩﺳﺘﻮﺭﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺭﺳﻴﺪ. ﺁﻥ ﮔﻮﻧﻪ ﮐﻪ ﺩﺭ ﻓﻠﺴﻔﻪ ﺍﺧﻼﻕ ﻣﻌﺮﻭﻑ ﺍﺳﺖ، «ﻫﺴﺖ» ﺑﻪ «ﺑﺎﻳﺪ» ﻣﻨﺠﺮ ﻧﻤﯽ‌ﺷﻮﺩ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺍﺯ ﻭﺍﻗﻌﻴﺎﺕ ﺟﻬﺎﻥ ﻣﺎﻧﻨﺪ ﺍﻳﻨﮑﻪ «ﮐﻼﻍ ﺳﻴﺎﻩ ﺍﺳﺖ» ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺍﻳﻦ ﺣﮑﻢ ﺍﺧﻼﻗﯽ ﺭﺳﻴﺪ ﮐﻪ ﺑﺎﻳﺪ ﺑﺎ ﻣﻮﺟﻮﺩ ﺯﻧﺪﻩ‌ﺍﯼ ﭼﻮﻥ ﮐﻼﻍ ﺧﻮﺵ‌ﺭﻓﺘﺎﺭﯼ ﮐﺮﺩ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﺧﺮﺩ ﻭ ﻋﻘﻼﻧﻴﺖ ﻫﻴﭻ ﻧﻘﺸﯽ ﺩﺭ ﺍﺧﻼﻕ ﻋﻤﻠﯽ ﻳﺎ ﺗﻔﮑﺮ ﺍﺧﻼﻗﯽ ﻧﺪﺍﺭﺩ. ﺍﻣﺎ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﮐﻪ ﺩﺍﺷﺘﻦ ﻫﺪﻑ ﻭ ﺍﺭﺯﺵ‌ﻫﺎﯼ ﻣﺸﺨﺺ ﺍﺳﺖ ﮐﻪ ﻋﻘﻼﻧﻴﺖ ﻭ ﺧﺮﺩ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﺗﺎ ﺑﻪ ﺁﻧﭽﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﺮﺳﻴﻢ ﻭ ﺑﺎ ﮐﻤﮏ ﺗﻔﮑﺮ ﻋﻘﻼﻧﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﻢ ﺍﺭﺯﺵ‌ﻫﺎ ﻭ ﻫﺪﻑ‌ﻫﺎﻣﺎﻥ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﺳﺎﺯﮔﺎﺭ ﻭ ﻫﻤﺎﻫﻨﮓ ﻫﺴﺘﻨﺪ.

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺍﺯ ﺗﺤﻤﻴﻞ ﺩﺭﺩ ﻭ ﺭﻧﺞ ﻏﻴﺮﺿﺮﻭﺭﯼ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﭘﺮﻫﻴﺰ ﮐﻨﻴﺪ، ﺍﺳﺘﺪﻻﻝ ﻋﻘﻼﻧﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﺪ ﮐﻪ ﺑﻪ ﭼﻪ ﻧﻮﻉ ﺍﺯ ﻣﻮﺟﻮﺩﺍﺕ ﺯﻧﺪﻩ ﺑﺎﻳﺪ ﺣﻮﺍﺳﺘﺎﻥ ﺑﺎﺷﺪ ﻭ ﭼﮕﻮﻧﻪ ﻋﻤﻞ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺭﻭﯼ ﺁﻧﻬﺎ ﺗﺎﺛﻴﺮ ﺑﮕﺬﺍﺭﺩ. ﺍﻣﺎ ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﭼﺮﺍ ﺑﺎﻳﺪ ﻣﻮﺍﻇﺐ ﺩﺭﺩ ﻭ ﺭﻧﺞ ﺩﻳﮕﺮﺍﻥ ﺑﺎﺷﻴﺪ، ﻋﻘﻞ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺳﺎﮐﺖ ﺍﺳﺖ.

ﭘﺲ ﭼﮕﻮﻧﻪ ﺑﺎﻳﺪ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﻨﺒﻌﯽ ﻣﺤﮑﻢ ﺑﺮﺍﯼ ﺍﻃﻤﻴﻨﺎﻥ ﺍﺯ ﺩﺭﺳﺘﯽ ﻳﺎ ﺍﻋﺘﺒﺎﺭ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﻗﻀﺎﻭﺕ‌ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﻣﺎ ﻣﯽ‌ﺷﻮﻧﺪ ﺑﮕﺮﺩﻳﻢ؟

 

ﺍﺭﺯﺵ‌ﻫﺎ ﺩﺭ ﺟﻬﺎﻥ ﺧﺎﻟﯽ ﺍﺯ ﺍﺭﺯﺵ
ﻣﺸﮑﻠﯽ ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺍﺯ ﻗﻮﻝ ﻫﻴﻮﻡ ﻧﻘﻞ ﮐﺮﺩﻳﻢ، ﺑﺮ ﻣﻮﺿﻮﻋﯽ ﮐﻪ ﻫﻤﻪ ﻣﺎ ﺑﺎ ﺁﻥ ﺩﺭﮔﻴﺮ ﻫﺴﺘﻴﻢ ﺍﻧﮕﺸﺖ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺍﺯ ﻳﮏ ﻃﺮﻑ ﻫﻤﻪ ﻣﺎ ﻣﻌﺘﻘﺪ ﻫﺴﺘﻴﻢ ﮐﻪ ﺩﺭ ﻳﮏ ﺟﻬﺎﻥ ﻓﻴﺰﻳﮑﯽ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺗﻮﺳﻂ ﻗﻮﺍﻧﻴﻦ ﻗﺎﺑﻞ ﮐﺸﻒ ﺗﻮﺳﻂ ﻋﻠﻢ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺟﻬﺎﻧﯽ ﺍﺯ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﻋﻴﻨﯽ ﮐﻪ ﺍﺯ ﺍﺭﺯﺵ‌ﻫﺎ ﻣﺴﺘﻘﻞ ﺍﺳﺖ. ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ ﺍﻣﺎ، ﻣﺎ ﺣﺲ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺩﺭ ﻗﻀﺎﻭﺕ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﻣﺎﻧﻨﺪ ﺍﻳﻨﮑﻪ ﻧﺴﻞ‌ﮐﺸﯽ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ، ﭼﻴﺰ ﺩﺭﺳﺘﯽ ﺩﺭﺑﺎﺭﻩ ﺟﻬﺎﻥ ﺍﻋﻼﻡ ﻣﯽ‌ﮐﻨﻴﻢ؛ ﭼﻴﺰﯼ ﮐﻪ ﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺪﺍﻧﻴﻢ ﻭ ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺩﺭﺳﺖ ﺍﺳﺖ ﻣﺴﺘﻘﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﭼﻪ ﺣﺴﯽ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

ﺍﻣﺎ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ ﮐﻪ ﺍﻳﻦ ﺩﻭ ﺩﻳﺪﮔﺎﻩ ﺑﺎ ﻫﻢ ﺳﺎﺯﮔﺎﺭ ﻧﻴﺴﺘﻨﺪ. ﺍﮔﺮ ﻧﻈﺮ ﻫﻴﻮﻡ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﻴﻢ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﺭﺯﻳﺎﺑﯽ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺭﺍ ﻣﺒﺘﻨﯽ ﺑﺮ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﺟﻬﺎﻥ ﺑﺪﻭﻥ ﺍﺭﺯﺵ ﻗﺎﺑﻞ ﺗﻮﺻﻴﻒ ﺗﻮﺳﻂ ﻋﻠﻢ ﻗﺮﺍﺭ ﺩﺍﺩ، ﻇﺎﻫﺮﺍ ﻣﺎ ﻣﺠﺒﻮﺭﻳﻢ ﮐﻪ ﺑﻪ ﻗﻀﺎﻭﺕ‌ﻫﺎ، ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺗﻤﺎﻳﻼﺕ ﺷﺨﺼﯽ ﺧﻮﺩ ﺑﺮﮔﺮﺩﻳﻢ ﻭ ﺑﺮﺍﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻣﻨﺸﺎ ﺍﺻﻠﯽ ﻗﻀﺎﻭﺕ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺑﻪ ﺩﺭﻭﻥ ﺧﻮﺩ ﻧﮕﺎﻩ ﮐﻨﻴﻢ.

ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻳﮑﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﺳﻮﺍﻻﺕ ﻓﻠﺴﻔﻪ ﺍﺧﻼﻕ ﺍﺳﺖ. ﺍﻳﻨﮑﻪ ﻣﻨﺒﻊ ﻗﻀﺎﻭﺕ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﭼﻴﺴﺖ ﻭ ﺍﺧﻼﻕ ﺑﺮ ﭼﻪ ﻣﺒﻨﺎﻳﯽ ﺍﺳﺘﻮﺍﺭ ﺍﺳﺖ؟ ﺩﺭ ﺍﺩﺍﻣﻪ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﺍﺯ ﺟﻨﺒﻪ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.
