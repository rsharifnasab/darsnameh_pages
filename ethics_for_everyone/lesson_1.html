ﺩﺭﺱ ﺍﻭﻝ- ﺍﺧﻼﻕ ﭼﻴﺴﺖ؟

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎ ﻳﺎﺩ ﮔﺮﻓﺘﻪ‌ﺍﻳﻢ ﻭ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﻧﺴﺒﺖ ﺑﻪ ﻣﺤﻴﻂ ﻓﻴﺰﻳﮑﯽ ﭘﻴﺮﺍﻣﻮﻥ ﺧﻮﺩ ﺣﺴﺎﺱ ﺑﺎﺷﻴﻢ. ﺍﻣﺎ ﮐﻤﺘﺮ ﻧﺴﺒﺖ ﺑﻪ ﭼﻴﺰﯼ ﮐﻪ ﻣﺤﻴﻂ ﺍﺧﻼﻗﯽ ﺍﻃﺮﺍﻑ ﻣﺎ ﻧﺎﻣﻴﺪﻩ ﻣﯽ‌ﺷﻮﺩ ﺩﻏﺪﻏﻪ ﺩﺍﺭﻳﻢ. ﺍﻳﻦ ﻣﺤﻴﻂ ﺍﺧﻼﻗﯽ ﺷﺎﻣﻞ ﺍﻓﮑﺎﺭ، ﺍﻧﺪﻳﺸﻪ‌ﻫﺎ ﻭ ﺍﻳﺪﻩ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻣﺎ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﭼﮕﻮﻧﻪ ﺯﻧﺪﮔﯽ ﮐﻨﻴﻢ. ﺁﻥ ﭼﻪ ﺭﺍ ﮐﻪ ﻣﺎ ﺍﺯ ﻧﻈﺮ ﺍﺧﻼﻗﯽ ﻳﺎ ﻭﺟﺪﺍﻧﯽ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻳﺎ ﻏﻴﺮﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﻣﻌﻴﻦ ﻣﯽ‌ﮐﻨﺪ، ﺍﺩﺭﺍﮎ ﻣﺎ ﺭﺍ ﺍﺯ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﻪ ﻭﻗﺖ ﭼﻴﺰﻫﺎ ﺧﻮﺏ ﭘﻴﺶ ﻣﯽ‌ﺭﻭﻧﺪ ﻭ ﭼﻪ ﻭﻗﺖ ﺑﺪ ﻫﺴﺘﻨﺪ. ﺍﺩﺭﺍﮎ ﻣﺎ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺍﻳﻦ ﻧﮑﺘﻪ ﻣﻌﻴﻦ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺑﺮ ﮔﺮﺩﻥ ﻣﺎﺳﺖ ﻭ ﭼﻪ ﺣﻘﻮﻗﯽ ﺩﺍﺭﻳﻢ. ﺑﺎ ﺗﻌﻴﻴﻦ ﺍﻳﻨﮑﻪ ﭼﻪ ﭼﻴﺰﯼ ﻋﺎﻣﻞ ﻏﺮﻭﺭ، ﺷﺮﻡ ﻳﺎ ﺧﺸﻢ ﻭ ﺳﭙﺎﺱ ﺍﺳﺖ، ﻭﺍﮐﻨﺶ‌ﻫﺎﯼ ﻋﺎﻃﻔﯽ ﻣﺎ ﺭﺍ ﺷﮑﻞ ﻣﯽ‌ﺩﻫﺪ؛ ﻳﺎ ﺍﻳﻨﮑﻪ ﺑﻪ ﻣﺎ ﻣﯽ‌ﮔﻮﻳﺪ ﭼﻪ ﭼﻴﺰﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺨﺸﻴﺪﻩ ﺷﻮﺩ ﻭ ﭼﻪ ﭼﻴﺰﯼ ﻏﻴﺮﻗﺎﺑﻞ ﺑﺨﺸﺶ ﺍﺳﺖ.

ﺍﺧﻼﻕ ﺑﻪ ﻣﺎ ﺍﺳﺘﺎﻧﺪﺍﺭﺩ ﻭ ﻣﻌﻴﺎﺭ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ، ﺍﺳﺘﺎﻧﺪﺍﺭﺩﻫﺎﯼ ﺭﻓﺘﺎﺭﯼ. ﺍﺯ ﻧﻈﺮ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺘﻔﮑﺮﺍﻥ ﺍﺯ ﺟﻤﻠﻪ ﻫﮕﻞ، ﻓﻴﻠﺴﻮﻑ ﺁﻟﻤﺎﻧﯽ ﺑﺰﺭﮒ ﻗﺮﻥ ﻧﻮﺯﺩﻫﻢ ﻣﻴﻼﺩﯼ، ﺍﺧﻼﻕ ﻫﻮﻳﺖ ﻣﺎ ﺭﺍ ﺷﮑﻞ ﻣﯽ‌ﻫﺪ. ﺧﻮﺩﺁﮔﺎﻫﯽ ﻣﺎ ﺍﺯ ﺧﻮﺩﻣﺎﻥ ﺗﺎ ﺣﺪﻭﺩ ﺯﻳﺎﺩﯼ ﻳﺎ ﺣﺘﯽ ﺿﺮﻭﺭﺗﺎ ﺑﻪ ﻣﻌﻨﺎﯼ ﺁﮔﺎﻫﯽ ﺍﺯ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻧﻈﺮ ﺩﻳﮕﺮﺍﻥ ﭼﮕﻮﻧﻪ ﻫﺴﺘﻴﻢ. ﻣﺎ ﻧﻴﺎﺯ ﺑﻪ ﺭﻭﺍﻳﺖ‌ﻫﺎﻳﯽ ﺍﺯ ﺍﺭﺯﺵ‌ﻫﺎﯼ ﺧﻮﺩﻣﺎﻥ ﺍﺯ ﻧﮕﺎﻩ ﺩﻳﮕﺮﺍﻥ ﺩﺍﺭﻳﻢ، ﺍﺯ ﻧﮕﺎﻩ ﺟﻬﺎﻥ ﺑﻴﺮﻭﻥ ﺍﺯ ﻣﺎ.

 

ﺍﻧﺴﺎﻥ ﻭ ﺍﺧﻼﻕ
ﺍﻧﺴﺎﻥ ﻣﻮﺟﻮﺩﯼ ﺍﺧﻼﻗﯽ ﺍﺳﺖ. ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﻣﺎ ﺑﻪ ﻃﻮﺭ ﻃﺒﻴﻌﯽ ﺧﻮﺏ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﻧﻴﺰ ﻧﻴﺴﺖ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﺑﯽ‌ﻭﻗﻔﻪ‌ﺍﯼ ﺑﻪ ﻳﮑﺪﻳﮕﺮ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﮐﻪ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ. ﺍﻣﺎ ﻣﺎ ﺑﻪ ﻫﻢ ﻧﻤﺮﻩ ﻣﯽ‌ﺩﻫﻴﻢ ﻭ ﺍﺭﺯﻳﺎﺑﯽ ﻣﯽ‌ﮐﻨﻴﻢ، ﻣﻘﺎﻳﺴﻪ ﻭ ﺗﺤﺴﻴﻦ ﻣﯽ‌ﮐﻨﻴﻢ، ﺍﺩﻋﺎ ﻭ ﺗﻮﺟﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ. ﻣﺎ ﺻﺮﻓﺎ ﺍﻳﻦ ﻳﺎ ﺁﻥ ﭼﻴﺰ ﺭﺍ ﺩﺭ ﺍﻧﺰﻭﺍ ﻭ ﺧﻼ «ﺗﺮﺟﻴﺢ» ﻧﻤﯽ‌ﺩﻫﻴﻢ. ﻣﺎ ﺩﻭﺳﺖ ﺩﺍﺭﻳﻢ ﮐﻪ ﺗﺮﺟﻴﺢ‌ﻫﺎﻳﻤﺎﻥ ﻣﺸﺘﺮﮎ ﺑﺎﺷﻨﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺩﺭﺧﻮﺍﺳﺖ‌ﻫﺎﻳﯽ ﺍﺯ ﻳﮑﺪﻳﮕﺮ ﺗﺒﺪﻳﻞ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﻳﻦ ﺩﺭﺧﻮﺍﺳﺖ‌ﻫﺎ ﻭ ﺗﻘﺎﺿﺎﻫﺎ ﺑﻪ ﻃﻮﺭ ﺑﯽ‌ﭘﺎﻳﺎﻧﯽ ﺍﺣﺴﺎﺱ ﻣﺎ ﺍﺯ ﻣﺴﺌﻮﻟﻴﺖ، ﮔﻨﺎﻩ ﻭ ﺷﺮﻡ ﻭ ﻫﻤﭽﻨﻴﻦ ﺣﺲ ﻣﺎ ﺭﺍ ﺍﺯ ﺍﺭﺯﺵ ﺧﻮﺩ ﻭ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺗﻨﻈﻴﻢ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺍﻧﺪﻳﺸﻴﺪﻥ ﺩﺭﺑﺎﺭﻩ ﻣﺤﻴﻂ ﺍﺧﻼﻗﯽ ﺗﻨﻬﺎ ﻣﺨﺼﻮﺹ ﻣﻌﺪﻭﺩﯼ ﺍﺯ ﻧﻈﺮﻳﻪ ﭘﺮﺩﺍﺯﺍﻥ ﺩﺍﻧﺸﮕﺎﻫﯽ ﻧﻴﺴﺖ. ﺍﺯ ﮐﺎﺭﺗﻮﻧﻴﺴﺖ‌ﻫﺎ ﮔﺮﻓﺘﻪ ﺗﺎ ﺭﻣﺎﻥ‌ﻧﻮﻳﺲ‌ﻫﺎ ﻭ ﺭﻭﺯﻧﺎﻣﻪ‌ﻧﮕﺎﺭﺍﻥ ﻭ ﺳﻴﺎﺳﺘﻤﺪﺍﺭﺍﻥ ﺩﺭ ﻣﻮﺭﺩ ﻣﺤﻴﻂ ﺍﺧﻼﻗﯽ ﺟﺎﻣﻌﻪ ﻧﻈﺮ ﻣﯽ‌ﺩﻫﻨﺪ ﻭ ﺁﻥ ﺭﺍ ﻧﻘﺪ ﻣﯽ‌ﮐﻨﻨﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻓﻠﺴﻔﻪ ﺗﻨﻬﺎ ﺣﻮﺯﻩ‌ﯼ ﺩﺭﮔﻴﺮ ﺑﺎ ﺍﺧﻼﻕ ﻧﻴﺴﺖ، ﺍﻣﺎ ﻳﮏ ﺟﺎﻩ‌ﻃﻠﺒﯽ ﻭﻳﮋﻩ ﺩﺍﺭﺩ. ﺟﺎﻩ‌ﻃﻠﺒﯽ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﺍﻧﮕﻴﺰﻩ‌ﻫﺎ، ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﻣﺎ ﺭﺍ ﺑﻔﻬﻤﺪ ﻭ ﺗﺤﻠﻴﻞ ﮐﻨﺪ.

 

ﺧﺮﻣﮕﺲ ﺁﺗﻦ
ﺍﺧﻼﻕ ﺁﺯﺍﺭﺩﻫﻨﺪﻩ ﺍﺳﺖ. ﻣﺎ ﺍﻏﻠﺐ ﺑﻪ ﻃﻮﺭ ﻣﺒﻬﻤﯽ ﻧﺎﺭﺍﺣﺖ ﻣﯽ‌ﺷﻮﻳﻢ ﻭﻗﺘﯽ ﺩﺭﺑﺎﺭﻩ ﭼﻴﺰﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﺍﺳﺘﺜﻤﺎﺭ ﻣﻨﺎﺑﻊ ﺟﻬﺎﻥ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﻢ. ﻳﺎ ﻭﻗﺘﯽ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﺁﺳﺎﻳﺶ ﻣﺎ ﺣﺎﺻﻞ ﺷﺮﺍﻳﻂ ﺳﺨﺖ ﻭ ﺗﻴﺮﻩ‌ﺭﻭﺯﯼ ﮐﺎﺭﮔﺮﺍﻥ ﺩﺭ ﮐﺸﻮﺭﻫﺎﯼ ﻓﻘﻴﺮ ﺍﺳﺖ. ﻣﺎ ﺣﺘﯽ ﮔﺎﻫﯽ ﺑﻪ ﻃﻮﺭ ﺩﻓﺎﻋﯽ ﺍﺯ ﻣﻄﺮﺡ ﺷﺪﻥ ﭼﻨﻴﻦ ﻣﺴﺎﺋﻠﯽ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﻳﻢ. ﺍﻣﺎ ﻫﻤﻴﺸﻪ ﺑﺮﺍﯼ ﻫﻤﻪ ﭼﻴﺰ ﺗﻮﺟﻴﻪ ﺩﺍﺭﻳﻢ ﻭ ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺗﻮﺟﻴﻪ ﺍﺳﺘﺜﻤﺎﺭ ﮐﺎﺭﮔﺮﺍﻥ ﺍﺯ ﺑﺎﺯﺍﺭ ﺁﺯﺍﺩ ﺣﺮﻑ ﺑﺰﻧﻴﻢ!

ﺩﻗﻴﻘﺎ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺳﻘﺮﺍﻁ، ﻓﻴﻠﺴﻮﻑ ﻳﻮﻧﺎﻥ ﺑﺎﺳﺘﺎﻥ، ﺧﺮﻣﮕﺲ ﺁﺗﻦ ﻟﻘﺐ ﮔﺮﻓﺖ. ﺍﻭ ﺩﺭ ﮐﻮﭼﻪ ﭘﺲ ﮐﻮﭼﻪ‌ﻫﺎﯼ ﺁﺗﻦ ﺍﺯ ﻣﺮﺩﻡ ﺩﺭﺑﺎﺭﻩ ﺑﺪﻳﻬﯽ‌ﺗﺮﻳﻦ ﺑﺎﻭﺭﻫﺎﻳﺸﺎﻥ ﺳﻮﺍﻝ ﻣﯽ‌ﮐﺮﺩ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﭼﺎﻟﺶ ﻣﯽ‌ﮐﺸﻴﺪ. ﺳﻮﺍﻝ‌ﻫﺎﯼ ﺳﺎﺩﻩ‌ﺍﯼ ﮐﻪ ﻫﺮ ﮐﺴﯽ ﺑﺎ ﮐﻤﯽ ﺍﻧﺪﻳﺸﻪ ﺩﺭ ﻣﯽ‌ﻳﺎﺑﺪ ﮐﻪ ﭼﻘﺪﺭ ﻧﻤﯽ‌ﺩﺍﻧﺪ ﻳﺎ ﭼﻘﺪﺭ ﻋﻘﺎﻳﺪﺵ ﺑﺎ ﻫﻢ ﺩﺭ ﺗﻀﺎﺩ ﻫﺴﺘﻨﺪ. ﺳﻮﺍﻝ‌ﻫﺎﯼ ﺳﻘﺮﺍﻁ ﻋﻤﺪﺗﺎ ﻣﺮﺑﻮﻁ ﺑﻪ ﻣﺴﺎﺋﻞ ﺍﺧﻼﻗﯽ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺑﻮﺩﻧﺪ ﮐﻪ ﭘﺲ ﺍﺯ ﭘﻴﺶ ﺭﻓﺘﻦ ﺩﻳﺎﻟﻮﮒ ﺑﺎ ﻣﺮﺩﻡ ﺑﻪ ﻣﺴﺎﺋﻞ ﺍﻧﺘﺰﺍﻋﯽ‌ﺗﺮ ﻓﻠﺴﻔﻪ ﻭ ﺍﺧﻼﻕ ﻣﻨﺘﻬﯽ ﻣﯽ‌ﺷﺪﻧﺪ. ﺳﻮﺍﻝ‌ﻫﺎﻳﯽ ﺩﺭﺑﺎﺭﻩ ﻋﺪﺍﻟﺖ، ﺍﻧﺼﺎﻑ، ﺧﻮﺑﯽ، ﻫﻤﺪﻟﯽ، ﺻﺪﺍﻗﺖ ﻭ ... ﻫﺮ ﺁﻧﭽﻪ ﮐﻪ ﺍﻣﺮﻭﺯ ﻫﻢ ﭘﺲ ﺍﺯ ﻗﺮﻥ‌ﻫﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﻣﺪﺭﻥ ﺑﺎ ﺁﻧﻬﺎ ﻣﻮﺍﺟﻪ ﻫﺴﺘﻴﻢ.

 

ﺳﻮﺍﻝ‌ﻫﺎﯼ ﺩﺷﻮﺍﺭ ﺍﺧﻼﻕ
ﺑﺎ ﻫﻤﻪ‌ﯼ ﺩﺷﻮﺍﺭﯼ‌ﻫﺎ ﻭ ﺁﺯﺍﺭﺩﻫﻨﺪﮔﯽ ﺍﺧﻼﻕ، ﺍﻳﻦ ﺣﻮﺯﻩ ﻭ ﻣﻮﺿﻮﻋﺎﺕ ﻣﻮﺭﺩ ﺑﺤﺚ ﺁﻥ ﺑﺮﺍﯼ ﻫﻤﻪ ﻣﺎ ﺟﺎﻟﺐ ﺍﺳﺖ. ﻫﻤﻪ ﻣﺎ ﺍﻧﺪﻳﺸﻪ‌ﻫﺎ ﻭ ﺍﻳﺪﻩ‌ﻫﺎﯼ ﺧﺎﺹ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﺩﺍﺭﻳﻢ ﮐﻪ ﭼﻪ ﭼﻴﺰﯼ ﺩﺭﺳﺖ ﻭ ﭼﻪ ﭼﻴﺰﯼ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ ﻭ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺧﻮﺏ ﻭ ﺑﺪ ﺭﺍ ﺍﺯ ﻫﻢ ﺗﻤﻴﺰ ﺩﺍﺩ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﻳﻦ ﺩﺭﺱ ﺭﺍ ﺑﺎ ﺳﻮﺍﻝ‌ﻫﺎﻳﯽ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﻣﺠﺒﻮﺭﻳﻢ ﺑﻪ ﺁﻧﻬﺎ ﭘﺎﺳﺦ ﺩﻫﻴﻢ، ﻭﻗﺘﯽ ﺍﺧﻼﻕ ﺩﺭ ﺯﻧﺪﮔﯽ ﻣﺎ ﺩﺧﺎﻟﺖ ﻣﯽ‌ﮐﻨﺪ! ﺷﺎﻳﺪ ﺑﺘﻮﺍﻥ ﮔﻔﺖ ﺑﻴﺸﺘﺮ ﻣﺴﺎﺋﻞ ﺍﺧﻼﻗﯽ ﺑﻪ ﻧﻮﻋﯽ ﺩﺭ ﺳﻮﺍﻝ‌ﻫﺎﯼ ﻓﻬﺮﺳﺖ ﺯﻳﺮ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ:

ﺍﺧﻼﻗﻴﺎﺕ ﻳﺎ ﺍﺧﻼﻕ

ﺩﺭ ﺩﺭﺱ ﭘﻨﺠﻢ ﺟﺴﺘﺠﻮ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﺒﻨﺎﯼ ﺍﺧﻼﻕ ﺭﺍ ﺑﺎ ﺍﻳﻦ ﺳﻮﺍﻝ ﭘﯽ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﮐﻪ ﺁﻳﺎ ﺩﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻨﺒﻌﯽ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﺑﺎﺷﺪ؟ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﺑﺴﻴﺎﺭﯼ ﺩﻳﻦ ﺭﺍ ﺑﺎ ﺍﺧﻼﻕ ﻫﻢ ﺍﺭﺯ ﻣﯽ‌ﺩﺍﻧﻨﺪ ﺷﺎﻳﺪ ﻓﻬﻢ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﮐﻪ ﭘﺎﺭﺍﺩﻭﮐﺲ ﻭ ﺗﻨﺎﻗﻀﯽ ﺩﺭ ﭘﺸﺖ ﺍﻳﻦ ﺑﺎﻭﺭ ﻧﻬﻔﺘﻪ ﺍﺳﺖ ﺷﮕﻔﺖ ﺁﻭﺭ ﺑﺎﺷﺪ.

ﺩﺭﺱ ﺷﺸﻢ ﺟﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﭘﺲ ﺍﺯ ﺟﺴﺘﺠﻮﯼ ﻓﺮﺍﻭﺍﻥ ﺑﻪ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﻣﯽ‌ﺭﺳﻴﻢ ﮐﻪ ﺍﮔﺮ ﻫﻴﭻ ﮐﺪﺍﻡ ﺍﺯ ﻣﻨﺎﺑﻊ ﻣﻄﺮﺡ ﺷﺪﻩ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﭘﺎﻳﻪ‌ﺍﯼ ﻣﺴﺘﺤﮑﻢ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﻧﻴﺴﺘﻨﺪ، ﭘﺲ ﺷﺎﻳﺪ ﺍﺧﻼﻕ ﺍﺻﻼ ﻣﺒﻨﺎﻳﯽ ﻧﺪﺍﺭﺩ. ﺍﻳﻨﺠﺎﺳﺖ ﮐﻪ ﺑﺤﺚ ﻋﻴﻨﻴﺖ ﺩﺭ ﺍﺧﻼﻕ ﮐﻪ ﻣﻮﺿﻮﻉ ﺍﻳﻦ ﺩﺭﺱ ﺍﺳﺖ ﻣﻄﺮﺡ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﺩﺭﺱ ﻫﻔﺘﻢ ﺑﻪ ﺩﻧﺒﺎﻝ ﭼﻴﺰﯼ ﻣﯽ‌ﮔﺮﺩﻳﻢ ﮐﻪ ﻗﺎﻋﺪﻩ ﻃﻼﻳﯽ ﺍﺧﻼﻕ ﺷﻬﺮﺕ ﺩﺍﺭﺩ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﺪﺍﻧﻴﻢ ﺁﻳﺎ ﺍﺻﻼ ﭼﻨﻴﻦ ﺍﺻﻞ ﻭ ﻗﺎﻋﺪﻩ‌ﺍﯼ ﮐﻤﮑﯽ ﺑﻪ ﺣﻞ ﻣﺴﺎﺋﻞ ﺍﺧﻼﻗﯽ ﻣﺎ ﺧﻮﺍﻫﺪ ﮐﺮﺩ؟

ﺩﺭﺱ ﻫﺸﺘﻢ ﺑﻪ ﺩﻧﺒﺎﻝ ﺩﻏﺪﻏﻪ ﻣﻌﺮﻭﻓﯽ ﺍﺳﺖ ﮐﻪ ﻫﻤﻪ ﻣﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺑﺎ ﺁﻥ ﻣﻮﺍﺟﻪ ﺷﺪﻩ‌ﺍﻳﻢ ﻭ ﺍﻳﻦ ﺳﻮﺍﻝ ﺭﺍ ﺍﺯ ﺧﻮﺩ ﭘﺮﺳﻴﺪﻩ‌ﺍﻳﻢ ﮐﻪ ﺁﻳﺎ ﻫﺪﻑ ﻭﺳﻴﻠﻪ ﺭﺍ ﺗﻮﺟﻴﻪ ﻣﯽ‌ﮐﻨﺪ؟

ﺩﺭﺱ ﻧﻬﻢ ﺑﻪ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﺍﺧﻼﻕ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ ﻭ ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﺭﺍ ﺟﺴﺘﺠﻮ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺁﻳﺎ ﻫﻤﻪ ﻣﻌﻤﺎﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺟﻮﺍﺏ ﺩﺍﺭﻧﺪ؟ ﺍﺧﻼﻕ ﻫﻢ ﻣﺎﻧﻨﺪ ﻫﺮ ﺑﺨﺶ ﺩﻳﮕﺮﯼ ﺍﺯ ﻣﻌﺮﻓﺖ ﺑﺸﺮﯼ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﻳﯽ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺍﺷﺎﺭﻩ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

ﺍﮔﺮ ﺑﻪ ﺩﺭﺱ ﺩﻫﻢ ﻭ ﺁﺧﺮ ﺭﺳﻴﺪﻳﺪ، ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﮐﻪ ﻫﻤﭽﻨﺎﻥ ﺩﺭ ﺍﺑﺘﺪﺍﯼ ﺍﻳﻦ ﻣﺴﻴﺮ ﺩﺷﻮﺍﺭ ﻗﺮﺍﺭ ﺩﺍﺭﻳﺪ ﻭ ﺍﺯ ﺁﻥ ﺟﺎ ﺑﻪ ﺑﻌﺪ ﺑﺎﻳﺪ ﺧﻮﺩﺗﺎﻥ ﺑﻪ ﺩﻧﺒﺎﻝ ﺟﻮﺍﺏ ﺳﻮﺍﻝ‌ﻫﺎﻳﺘﺎﻥ ﺑﺎﺷﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻋﻼﻭﻩ ﺑﺮ ﺍﺭﺍﺋﻪ ﺑﻌﻀﯽ ﺍﺯ ﺗﻌﺮﻳﻒ‌ﻫﺎ ﺍﺯ ﻣﮑﺘﺐ‌ﻫﺎ ﻭ ﭼﺎﺭﭼﻮﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺧﻼﻗﯽ، ﻓﻬﺮﺳﺘﯽ ﺍﺯ ﻣﻨﺎﺑﻊ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺑﻴﺸﺘﺮ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺗﺎﻥ ﻗﺮﺍﺭ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ ﺗﺎ ﺧﻮﺩﺗﺎﻥ ﺑﻪ ﺩﻧﺒﺎﻝ ﺣﻞ ﻣﺴﺎﺋﻞ ﺑﺮﻭﻳﺪ ﻭ ﻣﺴﺘﻘﻼﻧﻪ ﺑﻴﻨﺪﻳﺸﻴﺪ.
