ﺩﺭﺱ ﺷﺸﻢ- ﺁﻳﺎ ﺍﺧﻼﻕ ﺩﺍﺭﺍﯼ ﻳﮏ ﺣﻘﻴﻘﺖ ﻋﻴﻨﯽ ﺍﺳﺖ؟

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﻭﺭﻩ‌ﯼ «ﻓﻠﺴﻔﻪ ﺑﺮﺍﯼ ﻫﻤﻪ» ﺩﺭ ﺩﺭﺳﻨﺎﻣﻪ ﺩﻳﺪﻳﺪ، ﺩﺭ ﻓﻠﺴﻔﻪ ﻣﺒﺤﺜﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﺗﻤﺎﻳﺰ ﺫﻫﻦ ﺑﺎ ﺟﻬﺎﻥ ﺑﻴﺮﻭﻥ ﺍﺯ ﺫﻫﻦ ﻣﺮﺗﺒﻂ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﺑﺤﺚ ﻫﺮ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﺳﺎﺧﺘﻪ ﺫﻫﻦ ﺍﻧﺴﺎﻥ ﺑﺎﺷﺪ، ﺫﻫﻨﯽ ﻭ ﻫﺮ ﺁﻥ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﻣﺒﻨﺎﻳﯽ ﺧﺎﺭﺝ ﺍﺯ ﺫﻫﻦ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻋﻴﻨﯽ ﻣﯽ‌ﻧﺎﻣﻨﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺑﺴﻴﺎﺭﯼ ﻣﻌﺘﻘﺪﻧﺪ ﮐﻪ ﺯﻳﺒﺎﻳﯽ ﺍﻣﺮﯼ ﺫﻫﻨﯽ ﺍﺳﺖ ﻭﻟﯽ ﺩﺭ ﻋﻠﻢ ﮐﻪ ﺭﻳﺸﻪ ﺩﺭ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﺟﻬﺎﻥ ﻃﺒﻴﻌﯽ ﺩﺍﺭﺩ ﻣﻔﻬﻮﻣﯽ ﻋﻴﻨﯽ ﺍﺳﺖ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺭﺍ ﺻﺮﻓﺎ ﻳﮏ ﻧﺎﻡ‌ﮔﺬﺍﺭﯼ ﺑﺪﺍﻧﻴﺪ، ﺍﻣﺎ ﻭﻗﺘﯽ ﺍﻫﻤﻴﺖ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺑﻪ ﻣﻮﺿﻮﻋﯽ ﻣﺎﻧﻨﺪ ﺍﺧﻼﻕ ﻣﯽ‌ﺭﺳﺪ.

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺗﺎ ﺍﻳﻨﺠﺎﯼ ﺩﺭﺳﻨﺎﻣﻪ ﺍﺧﻼﻕ ﺩﻳﺪﻳﻢ، ﻧﻤﯽ‌ﺗﻮﺍﻥ ﻣﺒﻨﺎﻳﯽ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﺩﺭ ﻋﻠﻢ ﻳﺎ ﺩﻳﻦ ﭘﻴﺪﺍ ﮐﺮﺩ. ﭘﺲ ﺍﺧﻼﻕ ﺑﺎ ﺍﻳﻦ ﺧﻄﺮ ﻣﻮﺍﺟﻪ ﺍﺳﺖ ﮐﻪ ﮐﺎﻣﻼ ﺫﻫﻨﯽ ﻭ ﺳﺎﺧﺘﻪ ﺫﻫﻦ ﺍﻧﺴﺎﻥ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺍﺭﺯﺵ‌ﻫﺎ ﻭ ﺩﺳﺘﻮﺭﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺻﺮﻓﺎ ﺳﺎﺧﺘﻪ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﻣﺎ ﺑﺎﺷﻨﺪ، ﺑﻨﺎﺑﺮﺍﻳﻦ ﮐﺎﻣﻼ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﮐﻪ ﺍﺯ ﻳﮏ ﻓﺮﺩ ﺑﻪ ﻓﺮﺩ ﺩﻳﮕﺮ ﻣﺘﻔﺎﻭﺕ ﺑﺎﺷﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺗﺎ ﺟﺎﻳﯽ ﭘﻴﺶ ﺧﻮﺍﻫﻴﻢ ﺭﻓﺖ ﮐﻪ ﺍﺧﻼﻕ ﭼﻴﺰﯼ ﮐﺎﻣﻼ ﺩﻟﺒﺨﻮﺍﻫﯽ ﻭ ﺍﺧﺘﻴﺎﺭﯼ ﺑﺎﺷﺪ. ﭘﺲ ﺑﺮﺭﺳﯽ ﺍﻳﻦ ﺳﻮﺍﻝ ﺍﻫﻤﻴﺖ ﺩﺍﺭﺩ ﮐﻪ ﺁﻳﺎ ﺍﺧﻼﻕ ﺍﻣﺮﯼ ﺫﻫﻨﯽ ﺍﺳﺖ ﻳﺎ ﻳﮏ ﺣﻘﻴﻘﺖ ﻋﻴﻨﯽ ﻭ ﺧﺎﺭﺝ ﺍﺯ ﺫﻫﻦ ﻣﺎ ﺩﺍﺭﺩ؟

 

ﺍﺳﺘﺪﻻﻝ ﺁﻧﺘﺮﻭﭘﺮﻭ
ﺍﮔﺮ ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﻣﻌﻤﺎﯼ ﺍﻭﺗﻴﻔﺮﻭﻥ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﻣﻄﺮﺡ ﺷﺪ، ﺗﻨﻬﺎ ﺩﺭﺑﺎﺭﻩ ﺧﺪﺍ ﺑﺤﺚ ﻧﮑﻨﻴﻢ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﻳﻦ ﺭﻭﺵ ﺭﺍ ﺑﻪ ﺍﻓﺮﺍﺩ ﺧﻮﺏ ﻫﻢ ﺗﻌﻤﻴﻢ ﺩﺍﺩ. ﺑﻴﺎﻳﻴﺪ ﺍﺳﻢ ﺁﻥ ﺭﺍ ﻣﻌﻤﺎﯼ ﺁﻧﺘﺮﻭﭘﺮﻭ ﺑﮕﺬﺍﺭﻳﻢ (ﮐﻪ ﺩﺭ ﺁﻥ ﺁﻧﺘﺮﻭ ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﻧﺴﺎﻥ ﻭ ﭘﺮﻭ ﻧﺸﺎﻧﻪ‌ﯼ ﺣﻤﺎﻳﺖ ﻳﺎ ﻣﻮﺍﻓﻘﺖ ﺍﺳﺖ. ﺷﺎﻳﺪ ﺑﺘﻮﺍﻥ ﻧﺎﻡ ﺁﻥ ﺭﺍ ﺍﺳﺘﺪﻻﻝ ﺍﻧﺴﺎﻥ‌ﺩﻭﺳﺘﺎﻧﻪ ﻧﺎﻣﻴﺪ.) ﺳﻮﺍﻝ ﺍﺻﻠﯽ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺁﻳﺎ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﻮﺏ ﺍﻋﻤﺎﻝ ﺧﻮﺏ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﺮﺍﯼ ﺁﻧﮑﻪ ﺁﻥ ﺍﻋﻤﺎﻝ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ، ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﻳﻦ ﺍﻋﻤﺎﻝ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﻮﺏ ﺁﻧﻬﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻨﺪ؟ ﺷﻖ ﺩﻭﻡ ﺍﻳﻦ ﻣﻌﻤﺎ ﺑﺴﻴﺎﺭ ﻋﺠﻴﺐ‌ﺗﺮ ﺍﺯ ﺍﻳﻦ ﺍﻳﺪﻩ ﺍﺳﺖ ﮐﻪ ﺍﻋﻤﺎﻝ ﺧﻮﺏ، ﺧﻮﺏ ﻫﺴﺘﻨﺪ ﺗﻨﻬﺎ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺧﺪﺍ ﺑﻪ ﺍﻧﺠﺎﻡ ﺁﻧﻬﺎ ﻓﺮﻣﺎﻥ ﺩﺍﺩﻩ ﺍﺳﺖ. ﻣﺴﻠﻤﺎ ﭘﺎﺳﺦ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﻮﺏ ﺍﻳﻦ ﺍﻋﻤﺎﻝ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻨﺪ ﺯﻳﺮﺍ ﺍﻋﻤﺎﻝ ﺧﻮﺑﯽ ﻫﺴﺘﻨﺪ. ﺍﻣﺎ ﺍﻳﻦ ﺑﻪ ﻧﻈﺮ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎﺳﺖ ﮐﻪ ﺣﻘﻴﻘﺖ ﻭ ﺩﺭﺳﺘﯽ ﮔﺰﺍﺭﻩ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﻣﺴﺘﻘﻞ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﻭ ﮐﻨﺶ ﻋﻤﻞ ﺍﻧﺴﺎﻥ ﺍﺳﺖ. ﻭ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺍﺧﻼﻕ ﻋﻴﻨﯽ ﺍﺳﺖ ﻭ ﺍﻣﺮﯼ ﺫﻫﻨﯽ ﻭ ﺳﺎﺧﺘﻪ ﺑﺸﺮ ﻧﻴﺴﺖ.

ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻳﮏ ﻋﻨﺼﺮ ﻓﺮﺿﯽ ﺩﺭ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻫﻤﻪ ﺍﻳﻦ ﺣﺮﻑ‌ﻫﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺍﮔﺮ ﭼﻴﺰﯼ ﺑﻪ ﺍﺳﻢ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﻮﺏ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﭘﺲ ﺍﺧﻼﻕ ﺑﺎﻳﺪ ﻋﻴﻨﯽ ﻭ ﻣﺴﺘﻘﻞ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﺍﻧﺴﺎﻥ ﺑﺎﺷﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻋﻼﻭﻩ ﺑﺮ ﺍﺛﺒﺎﺕ ﻋﻴﻨﻴﺖ ﺍﺧﻼﻕ، ﺍﻳﻦ ﻣﻌﻤﺎ ﺣﺎﻻ ﺑﻪ ﺍﺳﺘﺪﻻﻟﯽ ﺗﺒﺪﻳﻞ ﻣﯽ‌ﺷﻮﺩ ﺩﺭﺑﺎﺭﻩ ﭼﮕﻮﻧﮕﯽ ﻭ ﺭﻭﺵ ﺣﻤﻠﻪ ﺑﻪ ﻳﮑﯽ ﺍﺯ ﺑﻨﻴﺎﺩﯼ‌ﺗﺮﻳﻦ ﻣﺒﺎﺣﺚ ﺩﺭ ﻣﺎﻫﻴﺖ ﺍﺧﻼﻗﻴﺎﺕ: ﻳﺎ ﻭﺍﻗﻌﺎ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﻮﺏ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﻭ ﺍﻧﺘﺨﺎﺏ‌ﻫﺎﯼ ﺁﻧﻬﺎ ﺩﺭﺳﺖ ﺍﺳﺖ، ﮐﻪ ﺩﺭ ﺁﻥ ﺻﻮﺭﺕ ﺍﺧﻼﻗﻴﺎﺕ ﻋﻴﻨﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ؛ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﺧﻼﻗﻴﺎﺕ ﻋﻴﻨﯽ ﻧﻴﺴﺖ ﺑﻠﮑﻪ ﻣﺤﺼﻮﻝ ﺍﻧﺘﺨﺎﺏ‌ﻫﺎﯼ ﺍﺧﺘﻴﺎﺭﯼ ﻭ ﺩﻟﺒﺨﻮﺍﻩ ﺍﻧﺴﺎﻥ ﺍﺳﺖ. ﻭ ﺍﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﺘﻴﺠﻪ ﺩﻫﺪ ﮐﻪ ﺍﮔﺮ ﺍﺧﻼﻕ ﻋﻴﻨﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ، ﭘﺲ ﻫﺮ ﮐﺎﺭﯼ ﻣﺠﺎﺯ ﺍﺳﺖ.

 

ﻣﺒﻨﺎﻳﯽ ﻋﺎﻃﻔﯽ ﺑﺮﺍﯼ ﺍﺧﻼﻕ
ﺣﺎﻻ ﻣﻌﻤﺎﯼ ﺍﻭﺗﻴﻔﺮﻭﻥ ﺑﻪ ﭼﻴﺰﯼ ﺗﻐﻴﻴﺮ ﺷﮑﻞ ﺩﺍﺩﻩ ﺍﺳﺖ ﮐﻪ ﻋﻼﻭﻩ ﺑﺮ ﺍﻧﺪﻳﺸﻤﻨﺪﺍﻥ ﺩﻳﻨﺪﺍﺭ، ﻣﺘﻔﮑﺮﺍﻥ ﺳﮑﻮﻻﺭ ﺭﺍ ﻫﻢ ﺑﻪ ﻣﺨﻤﺼﻪ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻓﻴﻠﺴﻮﻓﺎﻥ ﺍﺧﻼﻕ ﺧﺪﺍﻧﺎﺑﺎﻭﺭ ﻗﺎﻧﻊ ﺷﺪﻩ‌ﺍﻧﺪ ﮐﻪ ﺍﺧﻼﻗﻴﺎﺕ ﻋﻴﻨﯽ ﻧﻴﺴﺖ. ﺭﻭﻳﮑﺮﺩ ﺁﻧﻬﺎ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﭼﻪ ﻣﻌﻨﺎﯼ ﻣﻤﮑﻨﯽ ﺍﺭﺯﺵ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻳﮏ ﻣﻮﺟﻮﺩﻳﺖ ﻣﺴﺘﻘﻞ ﻭ ﻋﻴﻨﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ؟ ﺩﺭ ﻗﻮﺍﻧﻴﻦ ﻓﻴﺰﻳﮏ ﻫﻴﭻ «ﺑﺎﻳﺪ» ﺍﯼ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺩﺭ ﻳﮏ ﺟﻬﺎﻥ ﻃﺒﻴﻌﺖ‌ﮔﺮﺍﻳﺎﻧﻪ، ﻫﻴﭻ ﻣﮑﺎﻧﯽ ﺩﺭ ﮐﻬﮑﺸﺎﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﻗﻮﺍﻋﺪ ﺍﺧﻼﻕ ﺑﺮ ﺭﻭﯼ ﻳﮏ ﺗﺨﺘﻪ ﺳﻨﮓ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺑﺎﺷﻨﺪ. ﺑﻪ ﻃﻮﺭ ﺧﻼﺻﻪ، ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ ﮐﻪ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ ﺍﮔﺮ ﺍﺻﻮﻝ ﺍﺧﻼﻗﯽ ﻭﺟﻮﺩﯼ ﻣﺴﺘﻘﻞ ﻭ ﻋﻴﻨﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ.

ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺍﮐﺜﺮ ﮐﺴﺎﻧﯽ ﮐﻪ ﺍﻳﺪﻩ ﺍﺧﻼﻕ ﻋﻴﻨﯽ ﺭﺍ ﺭﺩ ﻣﯽ‌ﮐﻨﻨﺪ ﻫﻤﭽﻨﺎﻥ ﺑﺎﻭﺭ ﺩﺍﺭﻧﺪ ﮐﻪ ﺍﺧﻼﻕ ﻭﺍﻗﻌﯽ ﺍﺳﺖ ﻭ ﺍﺩﻋﺎﻫﺎﻳﯽ ﻧﺴﺒﺖ ﺑﻪ ﻣﺎ ﺩﺍﺭﺩ. ﺧﺐ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﻃﻮﺭ ﺑﺎﺷﺪ؟ ﺟﻮﺍﺏ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ،‌ ﺍﻣﺎ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺍﺧﻼﻕ ﻳﮏ ﺑﻨﻴﺎﻥ ﻳﮑﺘﺎ ﻧﺪﺍﺭﺩ. ﺑﻠﮑﻪ ﻇﻬﻮﺭ ﻳﺎﻓﺘﻪ ﺍﺯ ﺣﺎﺻﻞ ﺗﻼﻗﯽ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎ، ﺍﻣﻴﺎﻝ، ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﻧﻴﺎﺯﻫﺎﯼ ﻣﺸﺨﺼﯽ ﺍﺳﺖ. ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﻳﯽ ﺷﺎﻣﻞ ﺍﻳﻨﮑﻪ ﺩﺭﺩ ﻧﺎﺧﻮﺷﺎﻳﻨﺪ ﺍﺳﺖ،‌ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺍﺯ ﺁﻥ ﺍﺟﺘﻨﺎﺏ ﮐﻨﻨﺪ ﻭ ﺣﻴﻮﺍﻧﺎﺕ ﻫﻢ ﺁﻥ ﺭﺍ ﺣﺲ ﻣﯽ‌ﮐﻨﻨﺪ؛ ﻳﺎ ﺍﻳﻨﮑﻪ ﻫﻴﭻ ﺗﻔﺎﻭﺕ ﻗﺎﺑﻞ ﺗﻮﺟﻪ ﺍﺧﻼﻗﯽ ﺑﻴﻦ ﻣﺮﺩﻡ ﺑﺎ ﺭﻧﮓ ﻭ ﭘﻮﺳﺖ ﻣﺘﻔﺎﻭﺕ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ؛ ﻳﺎ ﺍﻳﻨﮑﻪ ﭼﮕﻮﻧﻪ ﻳﮏ ﻓﺮﺩ ﺛﺮﻭﺗﻤﻨﺪ ﻳﺎ ﺑﺎ ﺍﺳﺘﻌﺪﺍﺩ ﺑﺎ ﺷﺎﻧﺲ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﺩ. ﺷﺎﻣﻞ ﺁﺭﺯﻭﯼ ﺯﻧﺪﮔﯽ ﺑﺪﻭﻥ ﺗﻮﻫﻢ، ﺑﺪﻭﻥ ﺩﻭﺭﻭﻳﯽ ﻳﺎ ﺗﻤﺠﻴﺪ ﺷﺪﻥ ﺑﻪ ﺩﻻﻳﻞ ﺧﻮﺏ ﺍﺳﺖ. ﺍﺣﺴﺎﺳﺎﺕ ﺷﺎﻣﻞ ﺍﻧﺪﻭﻩ ﻫﻤﺪﺭﺩﺍﻧﻪ ﻫﻨﮕﺎﻡ ﺭﻧﺞ ﺩﻳﮕﺮﺍﻥ ﻭ ﻟﺬﺕ ﺍﺯ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﺍﺳﺖ ﮐﻪ ﺩﻳﮕﺮﺍﻥ ﺍﺣﺴﺎﺱ ﺑﻬﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﻭ ﻧﻴﺎﺯﻫﺎ ﺷﺎﻣﻞ ﺿﺮﻭﺭﺕ ﺯﻧﺪﮔﯽ ﺩﺭ ﺻﻠﺢ ﺑﺎ ﺩﻳﮕﺮﺍﻥ، ﺗﻮﺍﻧﺎﻳﯽ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﺁﻧﻬﺎ ﻭ ﻫﻤﮑﺎﺭﯼ ﺍﺳﺖ.

ﺍﮔﺮ ﻫﻤﻪ ﺍﻳﻦ‌ﻫﺎ ﺭﺍ ﮐﻨﺎﺭ ﻫﻢ ﺑﮕﺬﺍﺭﻳﻢ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﭼﻄﻮﺭ ﻳﮏ ﺍﺧﻼﻗﻴﺎﺕ ﻧﺴﺒﺘﺎ ﺣﺎﺿﺮ ﻭ ﺁﻣﺎﺩﻩ ﻇﻬﻮﺭ ﻣﯽ‌ﮐﻨﺪ. ﺑﺎ ﻓﺮﺽ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﻋﺮﻳﺎﻥ ﻭ ﺍﻣﻴﺎﻝ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﻧﻴﺎﺯﻫﺎﯼ ﻣﺎ، ﺩﻟﻴﻞ ﺧﻮﺑﯽ ﺩﺍﺭﻳﻢ ﺑﺮﺍﯼ ﺣﻤﺎﻳﺖ ﺍﺯ ﻋﺪﺍﻟﺖ،‌ ﺑﺮﺍﯼ ﻣﻨﺼﻒ ﻭ ﺻﺎﺩﻕ ﺑﻮﺩﻥ ﺩﺭ ﻣﺮﺍﻭﺩﺍﺗﻤﺎﻥ ﺑﺎ ﻣﺮﺩﻡ، ﺑﺮﺍﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻫﻤﺪﺭﺩﯼ، ﻧﮑﺸﺘﻦ ﺑﯽ‌ﮔﻨﺎﻫﺎﻥ ﻭ ﺑﺴﻴﺎﺭﯼ ﺻﻔﺖ‌ﻫﺎ ﻭ ﺍﻋﻤﺎﻝ ﺍﺧﻼﻗﯽ ﺩﻳﮕﺮ.

ﺍﻳﻦ ﭼﻬﺎﺭ ﻓﺎﮐﺘﻮﺭ ﻫﻤﺮﺍﻩ ﻫﻢ ﻳﮏ ﺑﻨﻴﺎﺩ ﻭ ﭘﺎﻳﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﻣﯽ‌ﺳﺎﺯﻧﺪ ﮐﻪ ﺑﻴﺶ ﺍﺯ ﺳﻮﺩ ﺷﺨﺼﯽ ﺻﺮﻑ ﺍﺳﺖ. ﺍﮔﺮ ﭼﻪ ﺁﻥ ﻫﻢ ﺑﺨﺸﯽ ﺍﺯ ﻫﻤﻴﻦ ﺗﺮﮐﻴﺐ ﺍﺳﺖ ﮐﻪ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻥ ﮐﺎﺭﻫﺎﯼ ﺩﺭﺳﺖ ﺣﺪﺍﻗﻞ ﺩﺭ ﺩﺭﺍﺯ ﻣﺪﺕ ﺑﻬﺘﺮ ﺍﺯ ﺍﻧﺠﺎﻡ ﻧﺪﺍﺩﻥ ﮐﺎﺭ ﺩﺭﺳﺖ ﺍﺳﺖ.

 

ﺁﻳﺎ ﺍﺧﻼﻕ ﻣﺒﺘﻨﯽ ﺑﺮ ﻋﻮﺍﻃﻒ ﻋﻴﻨﯽ ﺍﺳﺖ؟
ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﻪ ﻧﻈﺮ ﻧﻤﯽ‌ﺭﺳﺪ ﮐﻪ ﺍﻳﻦ ﭼﻴﺰﻫﺎ ﺟﻨﺒﻪ‌ﺍﯼ «ﻋﻴﻨﯽ» ﺑﻪ ﺍﺧﻼﻗﻴﺎﺕ ﺍﺿﺎﻓﻪ ﮐﻨﻨﺪ. ﺗﻨﻬﺎ ﻣﻮﻟﻔﻪ‌ﻫﺎﻳﯽ ﺍﺯ ﺁﻥ ﺟﻨﺒﻪ ﻋﻴﻨﯽ ﺩﺍﺭﻧﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﻋﻴﻨﯽ ﻧﻘﺶ ﺍﻳﻔﺎ ﻣﯽ‌ﮐﻨﻨﺪ، ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻣﺎ ﺁﻥ ﺭﺍ ﺩﺭ ﺍﺭﺯﺵ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ ﺑﻮﺩﻥ ﻧﻴﺎﺯ ﺩﺍﺭﻳﻢ. ﺍﻣﺎ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺁﻥ ﻫﻤﭽﻨﻴﻦ ﺩﺭﮔﻴﺮ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻣﻴﺎﻝ ﺍﺳﺖ، ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻣﺒﻨﺎﻳﯽ ﮐﺎﻣﻼ ﻋﻴﻨﯽ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﮐﺴﯽ ﺑﮕﻮﻳﺪ ﮐﻪ ﻧﻤﯽ‌ﺧﻮﺍﻫﺪ ﺑﺎ ﻫﻤﻨﻮﻋﺎﻥ ﺧﻮﺩ ﮐﻨﺎﺭ ﺑﻴﺎﻳﺪ ﻳﺎ ﺍﻫﻤﻴﺘﯽ ﺑﻪ ﺭﻧﺞ ﺁﻥ‌‌ﻫﺎ ﺑﺪﻫﺪ، ﻫﻴﭻ ﺭﺍﻫﯽ ﺑﺮﺍﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻋﻴﻨﯽ ﺍﻳﻨﮑﻪ ﺑﺎﻳﺪ ﺗﻤﺎﻳﻞ ﺩﻳﮕﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺩﺍﺭﺍﯼ ﻧﻘﺺ ﺍﺳﺖ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ.

ﺑﺎ ﻓﺮﺽ ﺍﻳﻨﮑﻪ ﺣﺎﻻ ﺑﺎﻳﺪ ﻭﺍﺿﺢ ﺑﺎﺷﺪ ﮐﻪ ﻣﻌﻤﺎﯼ ﺁﻧﺘﺮﻭﭘﺮﻭ ﺑﺮ ﻣﺒﻨﺎﯼ ﺁﻧﭽﻪ ﻳﮏ ﺩﻭﮔﺎﻧﻪ ﻏﻠﻂ ﻳﺎ ﺣﺪﺍﻗﻞ ﮔﻤﺮﺍﻩ‌ﮐﻨﻨﺪﻩ ﺍﺳﺖ، ﻗﺮﺍﺭ ﺩﺍﺭﺩ: ﺍﻧﺘﺨﺎﺑﯽ ﺑﻴﻦ ﻳﮏ ﺍﺧﻼﻕ ﻋﻴﻨﯽ ﻭ ﻳﮏ ﺍﺧﻼﻕ ﮐﺎﻣﻼ ﺩﻟﺒﺨﻮﺍﻩ. ﮔﻔﺘﻦ ﺍﻳﻨﮑﻪ ﻳﮏ ﺍﻧﺘﺨﺎﺏ ﺩﻟﺨﻮﺍﻩ ﺍﺳﺖ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎﺳﺖ ﮐﻪ ﺑﺪﻭﻥ ﻫﻴﭽﮕﻮﻧﻪ ﻣﺤﺪﻭﺩﻳﺘﯽ ﺑﺮ ﻣﺒﻨﺎﯼ ﻫﻮﺱ ﻳﺎ ﺗﺮﺟﻴﺢ ﺷﺨﺼﯽ ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﺍﻳﻦ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺴﺖ ﮐﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﺑﺎﺷﺪ. ﺣﺎﻝ ﺑﻪ ﭼﻪ ﻣﻌﻨﺎ ﺁﻥ ﻧﻮﻉ ﺍﺧﻼﻗﻴﺎﺗﯽ ﮐﻪ ﺍﺯ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎ، ﺗﻤﺎﻳﻼﺕ،‌ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﻧﻴﺎﺯﻫﺎ ﻇﻬﻮﺭ ﻣﯽ‌ﻳﺎﺑﺪ ﺩﻟﺨﻮﺍﻩ ﺍﺳﺖ؟ ﺗﻨﻬﺎ ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﮐﻪ ﻫﻴﭻ ﻗﺪﺭﺕ ﻳﺎ ﺍﺻﻞ ﺑﻴﺮﻭﻧﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﻣﺎ ﺭﺍ ﺑﻪ ﺍﻧﺘﺨﺎﺏ ﺍﺭﺯﺵ‌ﻫﺎ ﻣﺠﺒﻮﺭ ﮐﻨﺪ. ﺍﮔﺮ ﻣﺎ ﻭﺍﻗﻌﺎ ﻋﻼﻗﻪ ﺩﺍﺷﺘﻴﻢ ﺣﮑﻢ ﺩﻫﻴﻢ ﮐﻪ «ﻧﻔﺮﺕ، ﺩﺯﺩﯼ، ﺧﻴﺎﻧﺖ، ...» ﺍﺯ ﻧﻈﺮ ﺍﺧﻼﻗﯽ ﻣﺠﺎﺯ ﺍﺳﺖ، ﺁﻧﮕﺎﻩ ﻣﺎﻧﻨﺪ ﺧﺪﺍﯼ ﺍﻭﺗﻴﻔﺮﻭﻥ، ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﺳﺘﻴﻢ ﻣﯽ‌ﺗﻮﺍﻧﺴﺘﻴﻢ ﺁﻥ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ. ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﺍﺧﻼﻕ ﺩﻟﺨﻮﺍﻩ ﺍﺳﺖ ﻭ ﺑﺪﻭﻥ ﺧﺪﺍ، ﻫﻤﻪ ﭼﻴﺰ ﻣﺠﺎﺯ ﺍﺳﺖ.

ﺍﻣﺎ ﺍﻳﻦ ﺭﻭﺷﯽ ﻧﻴﺴﺖ ﮐﻪ ﻣﺎ ﻭﺍﮊﻩ «ﺩﻟﺨﻮﺍﻩ» ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺁﻥ ﻣﯽ‌ﻓﻬﻤﻴﻢ. ﺍﺧﻼﻗﻴﺎﺕ ﻣﺎ ﺩﻣﺪﻣﯽ ﻣﺰﺍﺝ ﻧﻴﺴﺖ ﺑﻠﮑﻪ ﺑﺮ ﻣﺒﻨﺎﯼ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﻣﻌﻴﻨﯽ ﺩﺭﺑﺎﺭﻩ ﻃﺒﻴﻌﺖ ﺍﻧﺴﺎﻥ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﻭ ﻧﻴﺎﺯ ﻣﺎ ﺑﻪ ﭘﻴﺶ ﺭﻓﺘﻦ، ﺩﺭ ﮐﻨﺎﺭ ﺗﻤﺎﻳﻼﺕ ﻭ ﺍﺣﺴﺎﺳﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﺍﻋﺘﺒﺎﺭ ﺁﻧﻬﺎ ﻭﻗﺘﯽ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﺣﻔﻆ ﻣﯽ‌ﺷﻮﺩ. ﺣﺘﯽ ﺍﮔﺮ ﺍﺳﺘﺤﮑﺎﻡ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﺳﺎﺩﻩ ﺭﺍ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﮔﺮ ﻓﮑﺮ ﮐﻨﻴﻢ ﮐﻪ ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺁﻳﺎ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻦ ﺩﺭﺳﺖ ﺍﺳﺖ ﻳﺎ ﻧﻪ، ﺍﺳﺘﺪﻻﻝ ﻋﻴﻨﯽ ﻭ ﻓﻮﺭﯼ‌ﺍﯼ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﺑﮕﻮﻳﺪ ﺍﻳﻨﻄﻮﺭ ﺍﺳﺖ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ، ﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﻋﺸﻖ ﻓﻮﺍﻳﺪﯼ ﺩﺍﺭﺩ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺍﺷﮑﺎﻻﺕ ﺯﻳﺎﺩﯼ ﻫﻢ ﺩﺍﺭﺩ، ﻭﻟﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﺩﻻﻳﻞ ﺧﻮﺑﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﭼﺮﺍ ﻣﺮﺩﻡ ﺑﺎﻳﺪ ﺁﻥ ﺭﺍ ﺑﺨﻮﺍﻫﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﮐﺴﯽ ﮐﻪ ﻣﺎﻧﻊ ﻋﺸﻖ ﺩﻳﮕﺮﺍﻥ ﻣﯽ‌ﺷﻮﺩ ﺑﻪ ﻣﻌﻨﺎﯼ ﻣﻨﻄﻘﯽ ﮔﻨﺎﻫﮑﺎﺭ ﻧﻴﺴﺖ، ﺍﻣﺎ ﺩﻻﻳﻞ ﺧﻮﺑﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﭼﺮﺍ ﻣﺎ ﺑﺎﻳﺪ ﺳﻌﯽ ﮐﻨﻴﻢ ﮐﺎﺭ ﺍﻭ ﺭﺍ ﺧﻨﺜﯽ ﮐﻨﻴﻢ. ﻭ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﻣﺎ ﺩﺭ ﺣﻘﻴﻘﺖ ﭼﻨﻴﻦ ﻗﻮﺍﻋﺪﯼ ﺭﺍ ﺩﻧﺒﺎﻝ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﻗﺎﻧﻮﻥ ﻭ ﻓﺸﺎﺭ ﺍﺟﺘﻤﺎﻋﯽ ﺣﻤﺎﻳﺖ ﻣﯽ‌ﮐﻨﻴﻢ،‌ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭼﻴﺰﻫﺎ ﻣﺠﺎﺯ ﻧﻴﺴﺖ.

 

ﺩﻭﮔﺎﻧﻪ ﮔﻤﺮﺍﻩ‌ﮐﻨﻨﺪﻩ ﻋﻴﻨﯽ ﻭ ﺫﻫﻨﯽ
ﻫﻤﭽﻨﺎﻥ ﺑﺴﻴﺎﺭﯼ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺍﻳﻦ ﺑﺎﻭﺭ ﺍﺩﺍﻣﻪ ﺩﻫﻴﻢ ﮐﻪ ﺗﺎ ﻭﻗﺘﯽ ﺍﺧﻼﻕ ﻣﺒﻨﺎﯼ ﮐﺎﻣﻼ ﻋﻴﻨﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺑﺎﻳﺪ ﺫﻫﻨﯽ ﺑﺎﺷﺪ. ﺍﺷﮑﺎﻝ ﺩﺭ ﺁﻧﺠﺎﺳﺖ ﮐﻪ ﺩﻭﮔﺎﻧﻪ ﻋﻴﻨﯽ/ﺫﻫﻨﯽ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻧﺴﺒﺖ ﻣﺎﻧﻌﻪ‌ﺍﻟﺠﻤﻊ ﻣﯽ‌ﺑﻴﻨﻴﻢ. ﺍﮔﺮ ﺑﺎ ﺗﻮﺿﻴﺢ ﺗﺎﻣﺲ ﻧﻴﮕﻞ، ﻓﻴﻠﺴﻮﻑ ﺫﻫﻦ، ﺩﺭ ﮐﺘﺎﺏ «ﻧﻘﻄﻪ ﻧﻈﺮ ﺍﺯ ﻫﻴﭻ ﮐﺠﺎ» ﻧﮕﺎﻩ ﮐﻨﻴﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﮐﻨﺪ. ﺑﺮﺍﯼ ﻧﻴﮕﻞ ﻋﻴﻨﻴﺖ ﻣﻄﻠﻖ ﻧﻴﺴﺖ، ﺑﻠﮑﻪ ﺩﺍﺭﺍﯼ ﺩﺭﺟﺎﺗﯽ ﺍﺳﺖ. ﻧﻘﻄﻪ ﻧﻈﺮ ﻣﺎ ﺑﻴﺸﺘﺮ ﻋﻴﻨﯽ ﺍﺳﺖ ﺍﮔﺮ ﮐﻤﺘﺮ ﺑﺮ «ﻣﻮﺿﻊ ﺍﻓﺮﺍﺩ ﺩﺭ ﺟﻬﺎﻥ» ﻣﺒﺘﻨﯽ ﺑﺎﺷﺪ. ﺷﺎﻳﺪ ﻏﻴﺮﻣﻤﮑﻦ ﺑﺎﺷﺪ ﮐﻪ ﻃﻮﺭ ﺩﻳﮕﺮﯼ ﺑﻪ ﻋﻴﻨﻴﺖ ﻧﮕﺎﻩ ﮐﻨﻴﻢ، ﺯﻳﺮﺍ ﺑﻪ ﺳﺎﺩﮔﯽ ﻫﻴﭻ ﺭﺍﻩ ﺩﻳﮕﺮﯼ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ ﺩﺭ ﮐﺴﺐ ﺩﺍﻧﺶ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﮐﺎﻣﻼ ﻣﺴﺘﻘﻞ ﺍﺯ ﻣﻮﺿﻊ ﻣﺎ ﻧﺴﺒﺖ ﺑﻪ ﺟﻬﺎﻥ ﺑﺎﺷﺪ.

ﺑﺎ ﻓﻬﻢ ﺍﻳﻦ ﺭﻭﺵ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﮐﻪ ﺍﺧﻼﻗﻴﺎﺗﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﻣﺜﻼ ﺫﻫﻨﯽ‌ﺗﺮ ﺍﺯ ﺯﻳﺴﺖ‌ﺷﻨﺎﺳﯽ ﺳﻠﻮﻟﯽ،‌ ﮐﻪ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺍﺯ ﺳﻠﻴﻘﻪ ﻣﻮﺳﻴﻘﯽ ﻋﻴﻨﯽ‌ﺗﺮ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﺗﺮﺱ ﺍﺯ ﺫﻫﻨﻴﺖ ﺩﺭ ﺍﺧﻼﻕ ﺍﻏﺮﺍﻕ ﺷﺪﻩ ﺍﺳﺖ. ﻣﺸﮑﻠﯽ ﻧﻴﺴﺖ ﺍﮔﺮ ﻣﻮﻟﻔﻪ‌ﺍﯼ ﺫﻫﻨﯽ ﺩﺭ ﺗﻔﮑﺮ ﺍﺧﻼﻗﯽ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﺴﺖ ﺟﻮﺭ ﺩﻳﮕﺮﯼ ﺑﺎﺷﺪ؟ ﺗﻨﻬﺎ ﺯﻣﺎﻧﯽ ﻣﺸﮑﻞ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺍﺟﺎﺯﻩ ﺩﻫﻴﻢ ﺁﻥ ﻣﻮﻟﻔﻪ ﺫﻫﻨﯽ ﺑﻴﺶ ﺍﺯ ﺍﻧﺪﺍﺯﻩ ﭼﻴﺮﻩ ﺷﻮﺩ، ﻭ ﭼﻴﺰﻫﺎ ﺭﺍ ﺑﻪ ﻫﻮﺱ ﺍﺷﺨﺎﺹ ﻳﺎ ﺗﺼﺎﺩﻑ‌ﻫﺎﯼ ﺗﺎﺭﻳﺨﯽ ﺍﻋﻤﺎﻝ ﺟﺎﺭﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻭﺍﮔﺬﺍﺭﺩ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﻳﺪﻳﻢ ﺍﻧﻮﺍﻉ ﻣﺘﻔﺎﻭﺗﯽ ﺍﺯ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎ ﺩﺭ ﺟﻬﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﻋﻴﻨﯽ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ. ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭﺳﺘﯽ ﺁﻧﻬﺎ ﻭﺍﺑﺴﺘﻪ ﺑﻪ ﻣﺎ ﻧﻴﺴﺖ. ﺑﻌﻀﯽ ﺍﺯ ﺍﻳﻦ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎ ﻋﻠﻤﯽ ﻫﺴﺘﻨﺪ، ﮐﻪ ﺭﻭﻳﺪﺍﺩﻫﺎ، ﻓﺮﺁﻳﻨﺪﻫﺎ ﻭ ﺭﻭﺍﺑﻂ ﻓﻴﺰﻳﮑﯽ ﺭﺍ ﺗﻮﺻﻴﻒ ﻣﯽ‌ﮐﻨﻨﺪ. ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺍﺧﻼﻗﯽ ﻫﺴﺘﻨﺪ ﻭ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﺩﺭ ﺟﻬﺎﻥ ﺗﻮﺻﻴﻒ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺩﺭﺳﺖ ﻳﺎ ﻧﺎﺩﺭﺳﺖ، ﺧﻮﺏ ﻳﺎ ﺑﺪ ﻫﺴﺘﻨﺪ. ﭼﻨﻴﻦ ﺗﺼﻮﻳﺮﯼ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺮﺍﯼ ﻋﻘﻞ ﺳﻠﻴﻢ ﺟﺬﺍﺏ ﺑﺎﺷﺪ، ﺍﻣﺎ ﺑﺮﺍﯼ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻓﻼﺳﻔﻪ ﭼﻨﺪﺍﻥ ﺟﺎﻟﺐ ﻧﻴﺴﺖ.

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻳﮏ ﻭﺍﻗﻌﻴﺖ ﺍﺧﻼﻗﯽ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ: ﻗﺘﻞ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ. ﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻋﻤﻞ ﮐﺸﺘﻦ ﺭﺍ ﺑﺎ ﺟﺰﺋﻴﺎﺗﺶ ﺗﻮﺻﻴﻒ ﮐﻨﻴﻢ، ﺑﺎ ﺍﺭﺟﺎﻉ ﺑﻪ ﺍﻧﻮﺍﻉ ﻭﺍﻗﻌﻴﺖ‌ﻫﺎﯼ ﻓﻴﺰﻳﮑﯽ ﻭ ﺭﻭﺍﻧﺸﻨﺎﺧﺘﯽ ﺑﺮﺍﯼ ﺗﻮﺿﻴﺢ ﺍﻳﻨﮑﻪ ﭼﮕﻮﻧﻪ ﻭ ﭼﺮﺍ ﺍﺗﻔﺎﻕ ﺍﻓﺘﺎﺩﻩ ﺍﺳﺖ. ﺍﻣﺎ ﭼﻪ ﻭﻳﮋﮔﯽ ﻳﺎ ﺧﺎﺻﻴﺖ ﺍﺿﺎﻓﻪ‌ﺍﯼ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﺗﺼﻮﻳﺮ ﺍﺿﺎﻓﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﻭﻗﺘﯽ ﻧﺎﺩﺭﺳﺖ ﺑﻮﺩﻥ ﺭﺍ ﺑﻪ ﺁﻥ ﻧﺴﺒﺖ ﻣﯽ‌ﺩﻫﻴﻢ؟ ﺑﻪ ﻃﻮﺭ ﺳﺎﺩﻩ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﮐﺸﺘﻦ ﺍﺯ ﺁﻥ ﻧﻮﻉ ﮐﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻧﺒﺎﻳﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ. ﺩﺭ ﺑﻴﻦ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﯼ ﺩﺭﺳﺖ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﻗﺘﻞ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﮕﻮﻳﻴﻢ، ﺁﻥ ﻫﻤﭽﻨﻴﻦ ﻳﮏ ﺧﺎﺻﻴﺖ ﺫﺍﺗﯽ «ﻣﻤﻨﻮﻋﻴﺖ» ﻫﻢ ﺩﺍﺭﺩ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻓﻼﺳﻔﻪ ﮐﻪ ﺑﺎ ﻳﺎﻓﺘﻦ ﭼﻨﻴﻦ ﺧﺎﺻﻴﺘﯽ ﺩﺭ ﺟﻬﺎﻥ (ﻣﻤﻨﻮﻉ ﺑﻮﺩﻥ ﻗﺘﻞ) ﺷﮕﻔﺖ‌ﺯﺩﻩ ﺷﺪﻩ‌ﺍﻧﺪ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﻣﺎ ﻣﻔﻬﻮﻡ ﺧﻮﺍﺹ ﻋﻴﻨﯽ ﺍﺧﻼﻗﯽ ﻣﻮﺟﻮﺩ ﺩﺭ ﺟﻬﺎﻥ ﺭﺍ ﺑﺎ ﻧﻮﻋﯽ ﻭﺍﮐﻨﺶ ﺫﻫﻨﯽ ﺑﻪ ﭼﻴﺰﻫﺎﯼ ﻣﻮﺟﻮﺩ ﺩﺭ ﺟﻬﺎﻥ ﻋﻮﺽ ﮐﻨﻴﻢ.

ﻣﻄﺎﺑﻖ ﻳﮏ ﺩﻳﺪﮔﺎﻩ ﺧﺎﻡ ﺫﻫﻦ‌ﮔﺮﺍﻳﺎﻧﻪ، ﺍﺩﻋﺎﻫﺎﯼ ﺍﺧﻼﻗﯽ ﺑﻪ ﺳﺎﺩﮔﯽ ﻫﻤﺎﻥ ﺗﻮﺻﻴﻒ‌ﻫﺎ ﻳﺎ ﮔﺰﺍﺭﺵ‌ﻫﺎﯼ ﻣﺎ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺗﻤﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﭼﮕﻮﻧﮕﯽ ﭼﻴﺰﻫﺎ ﺩﺭ ﺟﻬﺎﻥ ﻫﺴﺘﻨﺪ. ﭘﺲ ﻭﻗﺘﯽ ﻣﻦ ﻣﯽ‌ﮔﻮﻳﻢ ﮐﻪ «ﻗﺘﻞ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ» ﺑﻪ ﺳﺎﺩﮔﯽ ﺍﻧﺰﺟﺎﺭ ﺧﻮﺩ (ﻳﺎ ﺟﺎﻣﻌﻪ‌ﺍﻡ) ﺭﺍ ﺍﺯ ﺁﻥ ﺑﻴﺎﻥ ﻣﯽ‌ﮐﻨﻢ. ﺍﻣﺎ ﺍﻳﻦ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ‌ﺍﻧﮕﺎﺭﺍﻧﻪ ﺍﺳﺖ ﺯﻳﺮﺍ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﻣﻮﺿﻌﯽ ﮐﺎﻣﻼ ﻋﮑﺲ ﺍﻳﻦ ﻧﻴﺰ ﺻﺪﻕ ﮐﻨﺪ. ﻳﻌﻨﯽ ﺍﮔﺮ ﻣﻦ ﺑﮕﻮﻳﻢ «ﻗﺘﻞ ﺩﺭﺳﺖ ﺍﺳﺖ» ﻭ ﺍﻳﻦ ﺗﻮﺻﻴﻒ ﺩﻗﻴﻘﯽ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﻣﻦ ﺑﺎﺷﺪ، ﭘﺲ ﺁﻥ ﻧﻴﺰ ﺑﺎﻳﺪ ﺩﺭﺳﺖ ﺑﺎﺷﺪ! ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻋﺪﻡ ﺗﻮﺍﻓﻖ ﺍﺧﻼﻗﯽ ﺑﻪ ﻭﺿﻮﺡ ﻏﻴﺮ ﻣﻤﮑﻦ ﺍﺳﺖ. ﭘﺲ ﺑﻪ ﭼﻴﺰﯼ ﭘﻴﭽﻴﺪﻩ‌ﺗﺮ ﻧﻴﺎﺯ ﺩﺍﺭﻳﻢ.

ﺣﺎﻝ ﮐﻪ ﻳﺎﻓﺘﻦ ﻳﮏ ﻣﺒﻨﺎﯼ ﻣﺤﮑﻢ ﻭ ﺍﺳﺘﻮﺍﺭ ﺑﺮﺍﯼ ﺍﺧﻼﻗﻴﺎﺕ ﺍﻳﻨﻘﺪﺭ ﺩﺷﻮﺍﺭ ﺍﺳﺖ، ﻭ ﺑﻪ ﻧﻮﻋﯽ ﻣﺒﻨﺎﻳﯽ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﻣﮕﺮ ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺧﻮﺩﻣﺎﻥ (ﺑﻪ ﺻﻮﺭﺕ ﺟﻤﻌﯽ) ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ، ﻣﯽ‌ﺗﻮﺍﻥ ﮔﻔﺖ ﮐﻪ ﭼﺎﺭﭼﻮﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺧﻼﻗﯽ ﺣﺎﺻﻞ ﻣﺒﺎﻧﯽ ﻣﺨﺘﻠﻒ ﺑﺮﺍﯼ ﺍﺧﻼﻕ ﻫﺴﺘﻨﺪ. ﺍﻣﺎ ﺩﺭ ﭼﺎﺭﭼﻮﺏ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﻫﻢ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻥ ﺑﻪ ﺩﻧﺒﺎﻝ ﻳﮏ ﻣﻌﻴﺎﺭ ﺣﺪﺍﻗﻠﯽ ﻭ ﺳﺮﺭﺍﺳﺖ ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﺳﺘﻮﺭ‌ﺍﻟﻌﻤﻞ‌ ﺭﻓﺘﺎﺭ ﻭ ﻗﻀﺎﻭﺕ ﺍﺧﻼﻗﯽ ﺑﻮﺩ. ﭼﻴﺰﯼ ﮐﻪ ﺍﻧﺪﻳﺸﻤﻨﺪﺍﻥ ﺍﺧﻼﻕ ﺑﻪ ﺁﻥ ﻗﺎﻋﺪﻩ ﻃﻼﻳﯽ ﺍﺧﻼﻕ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﺩﺭ ﻣﻮﺭﺩ ﻗﺎﻋﺪﻩ ﻃﻼﻳﯽ ﺑﻴﺸﺘﺮ ﺑﺤﺚ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
