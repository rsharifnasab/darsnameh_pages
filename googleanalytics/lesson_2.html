ﺳﻪ ﻧﮑﺘﻪ ﻃﻼﻳﯽ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ

ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﮔﻔﺘﻴﻢ، ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ ﻣﺎﻧﻨﺪ ﻳﮏ ﺩﺳﺘﮕﺎﻩ ﺭﺍﺩﻳﻮﻟﻮﮊﯼ ﺍﺳﺖ ﮐﻪ ﻫﺮ ﺁﻧﭽﻪ ﺩﺭ ﺯﻳﺮ ﭘﻮﺳﺖ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﻣﯽ ﮔﺬﺭﺩ ﺭﺍ ﺑﻪ ﻧﻤﺎﻳﺶ ﻣﯽ ﮔﺬﺍﺭﺩ. ﺍﻣﺎ ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﺑﻪ ﺳﻪ ﻧﮑﺘﻪ ﻃﻼﻳﯽ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ ﺑﭙﺮﺩﺍﺯﻳﻢ، ﺑﺪ ﻧﻴﺴﺖ ﻧﮕﺎﻫﯽ ﮔﺬﺍﺭ ﺑﻪ ﺍﻫﻤﻴﺖ ﻫﺪﻑ ﺩﺭ ﻣﺪﻳﺮﻳﺖ ﻳﮏ ﻭﺏ ﺳﺎﻳﺖ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺩﺭ ﺍﻳﺠﺎﺩ ﻳﮏ ﻭﺏ ﺳﺎﻳﺖ ﺗﻌﺮﻳﻒ ﻳﮏ ﺳﺮﯼ ﺍﻫﺪﺍﻑ ﺑﺮﺍﯼ ﺁﻥ ﺍﺳﺖ ﺗﺎ ﺑﺮﺍﺳﺎﺱ ﺁﻥ ﺑﺮﻧﺎﻣﻪ ﺍﯼ ﺑﺮﺍﯼ ﺟﺬﺏ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺭﻳﺨﺘﻪ ﺷﻮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺍﮔﺮ ﺷﻤﺎ ﻗﺼﺪ ﺭﺍﻩ ﺍﻧﺪﺍﺯﯼ ﻳﮏ ﻭﺑﻼﮒ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﺁﻧﭽﻪ ﺩﺭ ﻗﺪﻡ ﺍﻭﻝ ﺩﺍﺭﺍﯼ ﺍﻫﻤﻴﺖ ﺍﺳﺖ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻫﺪﻑ ﻭ ﺯﻣﻴﻨﻪ ﻓﻌﺎﻟﻴﺖ ﻭﺑﻼﮒ ﺷﻤﺎ ﭼﻴﺴﺖ؟ ﺑﺮﺍﺳﺎﺱ ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻣﺮﺍﺣﻞ ﺑﻌﺪﯼ ﻣﺎﻧﻨﺪ ﺟﺬﺏ ﺧﻮﺍﻧﻨﺪﻩ ﻭ ﻳﺎ ﮔﺮﻓﺘﻦ ﺁﮔﻬﯽ ﺭﺍ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﺑﻠﻨﺪ ﻣﺪﺕ ﻭﺑﻼﮒ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ.

ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺍﮔﺮ ﻭﺑﻼﮒ ﺷﻤﺎ ﺩﺭ ﺯﻣﻴﻨﻪ ﻣﻮﺳﻴﻘﯽ ﻓﻌﺎﻟﻴﺖ ﻣﯽ ﮐﻨﺪ، ﺷﻤﺎ ﺑﺎﻳﺪ ﺗﻤﺮﮐﺰ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻋﻼﻗﻪ ﻣﻨﺪﺍﻥ ﺑﻪ ﻣﻮﺳﻴﻘﯽ ﻭ ﺗﺒﻠﻴﻐﺎﺕ ﺧﻮﺩ ﺭﺍ ﻧﻴﺰ ﺩﺭ ﻭﺏ ﺳﺎﻳﺖ ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﻣﻮﺳﻴﻘﯽ ﻗﺮﺍﺭ ﺑﺪﻫﻴﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺗﻌﻴﻴﻦ ﻫﺪﻑ ﺍﻭﻟﻴﻪ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﺪ ﺗﺎﺛﻴﺮ ﻣﺴﺘﻘﻴﻤﯽ ﺑﺮ ﺭﻭﯼ ﻣﻴﺰﺍﻥ ﻣﺨﺎﻃﺐ ﺷﻤﺎ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺍﻣﺎ ﺁﻧﭽﻪ ﺭﺍ ﮐﻪ ﻧﺒﺎﻳﺪ ﻓﺮﺍﻣﻮﺵ ﮐﺮﺩ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺯﻳﺎﺩ ﺑﻪ ﻣﻌﻨﺎﯼ ﻣﻮﻓﻖ ﺑﻮﺩﻥ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﻧﻴﺴﺖ ﻭ ﻧﺒﺎﻳﺪ ﺑﻪ ﺩﻟﻴﻞ ﺩﺍﺷﺘﻦ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺑﺎﻻ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﻫﺪﻑ ﺧﻮﺩ ﺩﺳﺖ ﭘﻴﺪﺍ ﮐﺮﺩﻩ‌ﺍﻳﺪ، ﺯﻳﺮﺍ ﺁﻧﭽﻪ ﻣﻬﻢ ﺍﺳﺖ ﺭﻓﺘﺎﺭ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺩﺭ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﻫﻤﺎﻥ ﺑﺨﺶ ﺗﺠﺰﻳﻪ ﻭ ﺗﺤﻠﻴﻞ ﻭﺏ ﺍﺳﺖ ﻭ ﺑﻪ ﮐﻤﮏ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ ﺷﻤﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺧﻮﺑﯽ ﺑﻪ ﺍﻳﻦ ﻓﺎﮐﺘﻮﺭ ﺩﺳﺖ ﭘﻴﺪﺍ ﮐﻨﻴﺪ.

ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﻣﺎﻧﻨﺪ ﻳﮏ ﻓﺮﻭﺷﮕﺎﻩ ﺍﺳﺖ

ﺑﺮﺍﯼ ﺩﺭﮎ ﺍﻫﻤﻴﺖ ﺗﻌﺮﻳﻒ ﻫﺪﻑ ﺑﺮﺍﯼ ﻳﮏ ﻭﺏ ﺳﺎﻳﺖ، ﻣﯽ ﺗﻮﺍﻥ ﻳﮏ ﻭﺏ ﺳﺎﻳﺖ ﺭﺍ ﺑﻪ ﻣﻐﺎﺯﻩ ﺍﯼ ﺗﺸﺒﻴﻪ ﮐﺮﺩ ﮐﻪ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﺁﻥ ﻧﻘﺶ ﺧﺮﻳﺪﺍﺭﺍﻥ ﺭﺍ ﺩﺍﺭﻧﺪ. ﺍﮔﺮ ﺷﻤﺎ ﺻﺎﺣﺐ ﻳﮏ ﻣﻐﺎﺯﻩ ﺑﺎﺷﻴﺪ ﻭ ﺑﺨﻮﺍﻫﻴﺪ ﮐﻪ ﻣﻐﺎﺯﻩ ﺷﻤﺎ ﺍﺯ ﺩﻳﮕﺮ ﻣﻐﺎﺯﻩ ﻫﺎ ﻓﺮﻭﺵ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺗﻨﻬﺎ ﺯﻳﺒﺎﻳﯽ ﻭ ﺩﮐﻮﺭ ﻣﻐﺎﺯﻩ ﻧﺒﺎﻳﺪ ﺑﺮﺍﻳﺘﺎﻥ ﻣﻬﻢ ﺑﺎﺷﺪ. ﺑﺎﻳﺪ ﮐﻴﻔﻴﺖ ﺍﺟﻨﺎﺱ، ﺳﺮﻭﻳﺲ ﺩﻫﯽ ﻭ... ﻧﻴﺰ ﺑﺮﺍﯼ ﺷﻤﺎ ﻣﻬﻢ ﻭ ﺩﺭ ﺑﺎﻻﺗﺮﻳﻦ ﮐﻴﻔﻴﺖ ﻣﻤﮑﻦ ﺑﺎﺷﺪ. ﺑﺮﺍﯼ ﻫﻤﻴﻦ ﺑﺎﻳﺪ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﻧﻴﺰ ﺗﻨﻬﺎ ﺯﻳﺒﺎ ﻧﺒﺎﺷﺪ ﻭ ﺩﺍﺭﺍﯼ ﻣﺤﺘﻮﺍ ﻭ ﻣﻄﺎﻟﺐ ﺟﺎﻟﺐ ﻭ ﮔﻴﺮﺍ ﺑﺮﺍﯼ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺑﺎﺷﺪ ﺗﺎ ﺁﻧﻬﺎ ﺩﺭ ﺩﻓﻌﺎﺕ ﺑﻌﺪ ﻧﻴﺰ ﺑﻪ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﺳﺮ ﺑﺰﻧﻴﺪ.

ﺳﻪ ﻧﮑﺘﻪ ﻃﻼﻳﯽ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ

۱- ﺍﻃﻼﻋﺎﺕ ﺩﺭﺳﺖ، ﻣﻬﻤﺘﺮﻳﻦ ﻭﻳﮋﮔﯽ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ

ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﻣﺰﻳﺖ ﻫﺎﯼ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﻃﻼﻋﺎﺕ ﺩﻗﻴﻖ ﻭ ﺩﺭﺳﺘﯽ ﺍﺯ ﻭﺿﻌﻴﺖ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﻭﺏ ﺳﺎﻳﺖ ﺑﻪ ﺷﻤﺎ ﻣﯽ ﺩﻫﺪ. ﺑﻪ ﻃﻮﺭﯼ ﮐﻠﯽ ﺩﺍﺷﺘﻦ ﺍﻃﻼﻋﺎﺕ ﻧﺎﺩﺭﺳﺖ ﻭ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﺑﺮﺍﺳﺎﺱ ﺁﻧﻬﺎ ﺑﺴﻴﺎﺭ ﺧﻄﺮﻧﺎﮎ ﺗﺮ ﺍﺯ ﻧﺪﺍﺷﺘﻦ ﺍﻃﻼﻋﺎﺕ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺩﺭﮎ ﺑﻬﺘﺮ ﺑﻪ ﺍﻳﻦ ﻣﺜﺎﻝ ﺗﻮﺟﻪ ﮐﻨﻴﺪ:

ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﺭﻭﺯﺍﻧﻪ ۱۰۰ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺩﺍﺭﺩ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﺗﻌﺪﺍﺩ ۹۹ ﻧﻔﺮ ﺍﺯ ﺁﻧﻬﺎ ﻫﻴﭻ ﮐﻠﻴﮑﯽ ﺑﺮ ﺭﻭﯼ ﺗﺒﻠﻴﻐﺎﺕ ﻧﻤﺎﻳﺶ ﺩﺍﺩﻩ ﺷﺪﻩ ﺑﺮ ﺭﻭﯼ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﻧﻤﯽ ﮐﻨﻨﺪ ﻭ ﺗﻨﻬﺎ ۱ ﻧﻔﺮ ﺑﺮ ﺭﻭﯼ ﻫﺮ ۱۰۰ ﺗﺒﻠﻴﻎ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﮐﻠﻴﮏ ﻣﯽ ﮐﻨﺪ. ﺣﺎﻝ ﺍﮔﺮ ﺷﻤﺎ ﺗﻨﻬﺎ ﺑﻪ ﺍﻋﺪﺍﺩ ﻭ ﺍﺭﻗﺎﻡ ﻧﮕﺎﻩ ﮐﻨﻴﺪ ﻭ ﺑﻪ ﺗﺠﺰﻳﻪ ﻭ ﺗﺤﻠﻴﻞ ﺁﻧﻬﺎ ﻧﭙﺮﺩﺍﺯﻳﺪ، ﺧﻮﺷﺤﺎﻝ ﺧﻮﺍﻫﻴﺪ ﺑﻮﺩ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﻣﺘﻮﺳﻂ ﻫﺮ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺑﺮ ﺭﻭﯼ ﺗﺒﻠﻴﻐﺎﺕ ﻭﺏ ﺳﺎﻳﺖ ﺷﻤﺎ ﻳﮏ ﺑﺎﺭ ﮐﻠﻴﮏ ﮐﺮﺩﻩ ﺍﺳﺖ، ﺍﻣﺎ ﺁﻳﺎ ﺍﻳﻦ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﺩﺭﺳﺖ ﺍﺳﺖ؟

ﺑﻪ ﮐﻤﮏ ﮔﻮﮔﻞ ﺁﻧﺎﻟﻴﺘﻴﮑﺰ ﺷﻤﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺟﺰﻳﻴﺎﺕ ﻭ ﺭﻓﺘﺎﺭ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﺧﻮﺩ ﺩﺳﺖ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﻭ ﻣﺘﻮﺟﻪ ﻣﻮﺍﺭﺩﯼ ﻣﺎﻧﻨﺪ ﻣﺜﺎﻝ ﺑﺎﻻ ﺷﻮﻳﺪ ﺗﺎ ﺩﺭ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﻫﺎﯼ ﺧﻮﺩ ﺩﭼﺎﺭ ﺧﻄﺎ ﻧﺸﻮﻳﺪ ﻭ ﻣﻴﺰﺍﻥ ﺧﻄﺎ ﺭﺍ ﺑﻪ ﺣﺪﺍﻗﻞ ﻣﻴﺰﺍﻥ ﻣﻤﮑﻦ ﮐﺎﻫﺶ ﺩﻫﻴﺪ. ﭘﺲ ﻫﻤﻮﺍﺭﻩ ﺍﻳﻦ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻋﺪﺍﺩ ﺑﺪﻭﻥ ﺗﺠﺰﻳﻪ ﻭ ﺗﺤﻠﻴﻞ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﺭﺍﻩ ﮐﺞ ﻭ ﻧﺎﺩﺭﺳﺖ ﻫﺪﺍﻳﺖ ﮐﻨﻨﺪ.
