ﭼﮕﻮﻧﻪ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﻣﻮﺛﺮ ﺑﻨﻮﻳﺴﻴﻢ - ﺑﺨﺶ ﺍﻭﻝ

ﺑﻪ ﺩﻭﺭﻩ ﺁﻣﻮﺯﺷﯽ «ﺍﺳﺘﺎﺩ ﺑﺰﺭﮔﯽ ﺍﻳﻤﻴﻞ ﻭ ﺟﻴﻤﻴﻞ» ﺧﻮﺵ ﺁﻣﺪﻳﺪ. ﺍﺭﺳﺎﻝ ﻭ ﺩﺭﻳﺎﻓﺖ ﺍﻳﻤﻴﻞ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﮐﻪ ﮐﺎﺭﯼ ﺳﺎﺩﻩ ﻣﺤﺴﻮﺏ ﻣﯽ ﺷﻮﺩ ﺩﺍﺭﺍﯼ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﺯﻳﺎﺩﯼ ﺍﺳﺖ. ﺍﺯ ﺁﺩﺍﺏ ﻭ ﺭﺳﻮﻡ ﮔﺮﻓﺘﻪ ﺗﺎ ﻧﮑﺎﺕ ﻣﺨﺘﻠﻒ ﻓﻨﯽ ﻭ ﻣﻮﺍﺭﺩ ﺍﻣﻨﻴﺘﯽ. ﺍﻳﻦ ﺩﻭﺭﻩ ﯼ ﺁﻣﻮﺯﺷﯽ ﻃﺮﺍﺣﯽ ﺷﺪﻩ ﺗﺎ ﺑﻪ ﺯﺑﺎﻥ ﺳﺎﺩﻩ ﺷﻤﺎ ﺭﺍ ﺑﺎ ﺍﻳﻦ ﻗﻮﺍﻋﺪ ﺁﺷﻨﺎ ﮐﻨﺪ. ۲۳ ﺩﺭﺱ ﻭ ﺗﻌﺪﺍﺩﯼ ﺁﺯﻣﻮﻥ ﺩﺭ ﺍﻳﻦ ﻣﺴﻴﺮ ﻫﻤﺮﺍﻩ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺷﺎﻳﺪ ﺣﺠﻢ ﻣﺤﺘﻮﺍﯼ ﺁﻣﻮﺯﺷﯽ ﮐﻤﯽ ﺯﻳﺎﺩ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ ﺍﻣﺎ ﺗﻮﺻﻴﻪ ﻣﯽ ﮐﻨﻴﻢ ﺑﺎ ﺩﻗﺖ ﻭ ﺣﻮﺻﻠﻪ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺁﻧﻬﺎ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﺍﻳﻦ ﻧﻬﺎﻳﺖ ﺧﻮﺷﺤﺎﻟﯽ ﻣﺎ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﭘﺎﻳﺎﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﯼ ﺁﻣﻮﺯﺷﯽ، ﺍﺳﺘﺎﺩ ﺑﺰﺭﮒ ﺍﻳﻤﻴﻞ ﺑﻨﺎﻣﻴﻢ. ﭘﺲ ﺑﺎ ﺩﺭﺳﻨﺎﻣﻪ ﺑﺎﺷﻴﺪ.

 

ﺑﺎ ﺍﺻﻮﻝ ﻧﻮﺷﺘﻦ ﺍﻳﻤﻴﻞ ﻣﻮﺛﺮ ﺷﺮﻭﻉ ﻣﯽ ﮐﻨﻴﻢ! ﭼﻪ ﮐﻨﻴﻢ ﺗﺎ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﻣﺎﻥ ﺣﺘﻤﺎ ﺧﻮﺍﻧﺪﻩ ﺷﺪﻩ ﻭ ﺑﻪ ﺁﻧﻬﺎ ﻋﻤﻞ ﮔﺮﺩﺩ؟ ﺁﻳﺎ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺍﺭﺳﺎﻟﯽ ﺷﻤﺎ ﺑﻪ ﻫﻤﺎﻥ ﺷﮑﻠﯽ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺍﻫﻤﻴﺖ ﻣﯽ ﺩﻫﻨﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺳﺮﻋﺖ ﭘﺎﺳﺦ ﻣﯽ ﺩﻫﻨﺪ؟ ﻳﺎ ﺍﻳﻨﮑﻪ ﺁﻧﻬﺎ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺍﻃﻼﻋﺎﺕ ﻣﻬﻤﯽ ﺭﺍ ﮐﻪ ﺍﺭﺳﺎﻝ ﮐﺮﺩﻩ ﺍﻳﺪ ﺍﺯ ﺩﺳﺖ ﻣﯽ ﺩﻫﻨﺪ؟ ﻭ ﻣﻬﻤﺘﺮ ﺍﻳﻨﮑﻪ ﺁﻳﺎ ﺍﻃﻤﻴﻨﺎﻥ ﺩﺍﺭﻳﺪ ﮐﻪ ﺩﺭ ﻣﺘﻦ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺗﺎﻥ ﺣﺪﺍﮐﺜﺮ ﺗﺎﺛﻴﺮ ﻣﻤﮑﻦ ﺭﺍ ﺑﺮ ﻣﺨﺎﻃﺐ ﮔﺬﺍﺷﺘﻪ ﻭ ﺍﻧﮕﻴﺰﻩ ﻻﺯﻡ ﺭﺍ ﺩﺭ ﻭﯼ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺍﻳﺪ؟

 

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﺭﺍ ﻣﯽ ﻧﻮﻳﺴﻴﺪ، ﺗﻨﻬﺎ ﺑﺎ ﺭﻋﺎﻳﺖ ﭼﻨﺪ ﺍﺻﻞ ﺳﺎﺩﻩ ﻭ ﮐﺎﺭﺑﺮﺩﯼ، ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺗﺎﺛﻴﺮ ﻣﺜﺒﺘﯽ ﺑﺮ ﮔﻴﺮﻧﺪﻩ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺍﺭﺳﺎﻝ ﭘﺎﺳﺦ ﻣﻮﺭﺩ ﻧﻈﺮﺗﺎﻥ ﺩﺭ ﻣﺪﺕ ﺯﻣﺎﻥ ﮐﻮﺗﺎﻫﯽ ﺗﺮﻏﻴﺐ ﻧﻤﺎﻳﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﻣﻌﺮﻓﯽ ﺑﺮﺧﯽ ﺍﺯ ﺍﻳﻦ ﺍﺻﻮﻝ ﻣﻬﻢ ﺩﺭ ﻧﻮﺷﺘﻦ ﺍﻳﻤﻴﻞ ﻣﯽ ﭘﺮﺩﺍﺯﻳﻢ ﻭ ﺳﻌﯽ ﻣﯽ ﮐﻨﻴﻢ ﺁﻧﻬﺎ ﺭﺍ ﺑﺎ ﻣﺜﺎﻝ ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﺧﻮﺏ ﻭ ﺍﻳﻤﻴﻞ ﺑﺪ، ﮐﺎﻣﻼ ﺑﺮﺍﯼ ﺗﺎﻥ ﺭﻭﺷﻦ ﮐﻨﻴﻢ.

 

- ﻣﻮﺿﻮﻉ (ﺳﺎﺑﺠﮑﺖ- Subject) ﺍﻳﻤﻴﻞ ﻳﮏ ﺳﺮﺗﻴﺘﺮ ﺧﺒﺮﯼ ﺍﺳﺖ

ﺗﻴﺘﺮ ﻳﮏ ﺭﻭﺯﻧﺎﻣﻪ ﺩﻭ ﮐﺎﺭ ﻣﻬﻢ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ ﺩﻫﺪ: ﺗﻮﺟﻪ ﺷﻤﺎ ﺭﺍ ﺟﻠﺐ ﻣﯽ ﮐﻨﺪ ﻭ ﺑﻪ ﺷﻤﺎ ﻣﯽ ﮔﻮﻳﺪ ﮐﻪ ﻣﻘﺎﻟﻪ ﯼ ﺍﺻﻠﯽ ﺍﺵ ﺩﺭﺑﺎﺭﻩ ﯼ ﭼﻪ ﻣﻮﺿﻮﻋﯽ ﺍﺳﺖ. ﺁﻧﮕﺎﻩ ﺷﻤﺎ ﺗﺼﻤﻴﻢ ﻣﯽ ﮔﻴﺮﻳﺪ ﮐﻪ ﺁﻥ ﺭﻭﺯﻧﺎﻣﻪ ﺭﺍ ﺧﺮﻳﺪﺍﺭﯼ ﮐﺮﺩﻩ ﻭ ﻣﻄﺎﻟﻌﻪ ﮐﻨﻴﺪ، ﻳﺎ ﺍﻳﻨﮑﻪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺍﺯ ﮐﻨﺎﺭ ﺁﻥ ﺑﮕﺬﺭﻳﺪ. ﺳﺎﺑﺠﮑﺖ ﺍﻳﻤﻴﻞ ﻫﻢ ﺑﺎﻳﺪ ﺩﻗﻴﻘﺎ ﻫﻤﻴﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻣﻨﺎﺳﺐ ﺗﻌﺪﺍﺩ ﮐﻤﯽ ﮐﻠﻤﻪ، ﮔﻴﺮﻧﺪﻩ ﺩﺭ ﻳﮏ ﻧﮕﺎﻩ ﺧﻮﺍﻫﺪ ﻓﻬﻤﻴﺪ ﮐﻪ ﺍﻳﻦ ﺍﻳﻤﻴﻞ ﺩﺭﺑﺎﺭﻩ ﭼﻪ ﻣﻮﺿﻮﻋﯽ ﺍﺳﺖ ﻭ ﭼﻪ ﻫﺪﻓﯽ ﺩﺍﺭﺩ.

 

ﺍﮔﺮ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﻳﮑﯽ ﺍﺯ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﻋﻤﻮﻣﯽ ﻋﺎﺩﯼ ﺍﺳﺖ ﮐﻪ ﻣﺮﺗﺒﺎ ﺍﺭﺳﺎﻝ ﻣﯽ ﺷﻮﻧﺪ (ﻣﺎﻧﻨﺪ ﮔﺰﺍﺭﺵ ﮐﺎﺭ ﻫﻔﺘﮕﯽ)، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺗﺎﺭﻳﺦ ﺭﺍ ﺩﺭ ﺳﺎﺑﺠﮑﺖ ﺫﮐﺮ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻤﻴﻞ ﻫﺎﻳﯽ ﻫﻢ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﭘﺎﺳﺦ ﺩﺍﺭﻧﺪ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﻳﻦ ﻣﻮﺭﺩ ﺩﺭ ﺳﺎﺑﺠﮑﺖ ﺩﺭﺝ ﺷﻮﺩ. ﻣﺜﻼ: «ﻟﻄﻔﺎ ﺗﺎ ۱۵ ﺧﺮﺩﺍﺩ ﭘﺎﺳﺦ ﺩﻫﻴﺪ». 

 

ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﻤﻪ ﺗﻼﺵ ﻣﯽ ﮐﻨﻨﺪ ﺗﻌﺪﺍﺩ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺍﺳﭙﻢ ﺩﺭﻳﺎﻓﺘﯽ ﺷﺎﻥ ﺭﺍ ﮐﺎﻫﺶ ﺩﻫﻨﺪ.  ﺍﮔﺮ ﺷﻤﺎ ﺍﺯ ﺳﺎﺑﺠﮑﺖ ﺗﺎﻥ ﺑﻪ ﺧﻮﺑﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﺷﺎﻧﺲ ﺍﻳﻨﮑﻪ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﺧﻮﺍﻧﺪﻩ ﺷﻮﺩ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﺩﻫﻴﺪ، ﻭ ﺟﻠﻮﯼ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﮐﻪ ﺑﺎ ﺍﺳﭙﻢ ﺍﺷﺘﺒﺎﻩ ﮔﺮﻓﺘﻪ ﺷﺪﻩ ﻭ ﺣﺘﯽ ﺑﺪﻭﻥ ﻳﮏ ﻧﮕﺎﻩ ﺳﺮﺳﺮﯼ ﭘﺎﮎ ﺷﻮﺩ ﺭﺍ ﻣﯽ ﮔﻴﺮﻳﺪ.

 

ﻣﻄﻤﺌﻨﺎ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﻧﺘﺸﺎﺭ ﻳﮏ ﺭﻭﺯﻧﺎﻣﻪ ﺑﺪﻭﻥ ﺗﻴﺘﺮ ﻳﮏ ﮐﺎﺭ ﺍﺣﻤﻘﺎﻧﻪ ﺍﺳﺖ، ﺧﺎﻟﯽ ﮔﺬﺍﺷﺘﻦ ﺳﺎﺑﺠﮑﺖ ﺍﻳﻤﻴﻞ ﺍﺯ ﺁﻥ ﻫﻢ ﺑﺪﺗﺮ ﺍﺳﺖ. ﺯﻳﺮﺍ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺑﺎ ﺳﺎﺑﺠﮑﺖ ﺧﺎﻟﯽ ﻣﻌﻤﻮﻻ ﺑﻪ ﭘﻮﺷﻪ ﯼ ﺍﺳﭙﻢ ﻣﻨﺘﻘﻞ ﻣﯽ ﺷﻮﻧﺪ.

ﺧﺐ ﻣﺸﮑﻞ ﮐﺠﺎﺳﺖ؟ ﺍﻳﻦ ﺳﺎﺑﺠﮑﺖ ﻫﻴﭻ ﮔﻮﻧﻪ ﺍﻃﻼﻋﺎﺗﯽ ﺩﺭﺑﺎﺭﻩ ﻣﺘﻦ ﭘﻴﺎﻡ، ﺑﻪ ﮔﻴﺮﻧﺪﻩ ﻧﻤﯽ ﺩﻫﺪ. ﺍﮔﺮ ﺗﻘﻮﻳﻢ ﺷﻤﺎ ﭘﺮ ﺍﺯ ﺍﻳﻦ ﻗﺮﺍﺭ ﻣﻼﻗﺎﺕ ﻫﺎ ﺑﺎﺷﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﺮ ﺩﺭﮔﻢ ﺷﻮﻳﺪ ﮐﻪ ﺣﺎﻣﺪ ﺩﺭﺑﺎﺭﻩ ﮐﺪﺍﻡ ﻳﮏ ﺻﺤﺒﺖ ﻣﯽ ﮐﻨﺪ. ﺑﺪﺗﺮ ﺍﺯ ﺁﻥ ﻧﻪ ﺗﻨﻬﺎ ﺩﺭ ﺳﺎﺑﺠﮑﺖ ﮐﻪ ﺩﺭ ﻣﺘﻦ ﺍﻳﻤﻴﻞ ﻫﻢ ﺻﺤﺒﺘﯽ ﺍﺯ ﺯﻣﺎﻥ ﻭ ﻣﮑﺎﻥ ﻣﻼﻗﺎﺕ ﺑﻪ ﻣﻴﺎﻥ ﻧﻴﺎﻣﺪﻩ ﺍﺳﺖ. ﺑﻪ ﺩﻟﻴﻞ ﻋﺪﻡ ﻭﺟﻮﺩ ﺍﻃﻼﻋﺎﺕ ﻣﻨﺎﺳﺐ ﺩﺭ ﺳﺎﺑﺠﮑﺖ ﺍﻳﻦ ﺍﻳﻤﻴﻞ ﮐﺎﻣﻼ ﺷﺒﻴﻪ ﻳﮏ ﺍﺳﭙﻢ ﺍﺳﺖ ﻭ ﺍﺣﺘﻤﺎﻝ ﺁﻥ ﺯﻳﺎﺩ ﺍﺳﺖ ﮐﻪ ﺑﺪﻭﻥ ﺧﻮﺍﻧﺪﻩ ﺷﺪﻥ ﭘﺎﮎ ﺷﺪﻩ ﻭ ﻳﺎ ﺣﺘﯽ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﺍﺳﭙﻢ ﮔﺰﺍﺭﺵ ﺷﻮﺩ.

ﺳﻼﻡ ﻣﺤﻤﺪ

ﻣﯽ ﺧﻮﺍﺳﺘﻢ ﺟﻠﺴﻪ ﺍﯼ ﺭﺍ ﮐﻪ ﺑﺮﺍﯼ ﺭﻭﺯ ﺳﻪ ﺷﻨﺒﻪ ۱۵ ﻓﺮﻭﺭﺩﻳﻦ ﺳﺎﻋﺖ ۱۰ ﺻﺒﺢ  ﺗﻨﻈﻴﻢ ﮐﺮﺩﻩ ﺑﻮﺩﻳﻢ ﻳﺎﺩﺁﻭﺭﯼ ﮐﻨﻢ. ﺍﻳﻦ ﻣﻼﻗﺎﺕ ﺩﺭ ﺍﺗﺎﻕ ﮐﻨﻔﺮﺍﻧﺲ ﺷﺮﮐﺖ ﺳﺎﺳﺎﻥ ﺩﺭﺑﺎﺭﻩ ﻣﺮﺍﺣﻞ ﺩﺭﻳﺎﻓﺖ ﻭﻳﺰﺍ ﺑﺮﮔﺰﺍﺭ ﺧﻮﺍﻫﺪ ﺷﺪ.

 

ﺑﺎ ﻣﻦ ﺭﺍﺣﺖ ﺑﺎﺵ ﻭ ﺍﮔﺮ ﺳﻮﺍﻟﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺩﺍﺷﺘﯽ ﺣﺘﻤﺎ ﺑﺎ ﻣﻦ ﺗﻤﺎﺱ ﺑﮕﻴﺮ.

 

ﺑﺎ ﺍﺣﺘﺮﺍﻡ

ﺣﺎﻣﺪ

ﺧﺐ، ﻣﯽ ﺑﻴﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﺳﺎﺑﺠﮑﺖ ﻭ ﺍﻳﻤﻴﻞ ﭼﻘﺪﺭ ﻣﺸﺨﺺ، ﺭﻭﺷﻦ ﻭ ﮔﻮﻳﺎ ﺍﺳﺖ.  ﻣﻬﻤﺘﺮﻳﻦ ﻧﮑﺘﻪ ﺩﺭ ﻣﻮﺭﺩ ﺍﻳﻦ ﺳﺎﺑﺠﮑﺖ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﮔﻴﺮﻧﺪﻩ ﻧﺎﻣﻪ ﺣﺘﯽ ﺑﺪﻭﻥ ﺑﺎﺯ ﮐﺮﺩﻥ ﺍﻳﻤﻴﻞ، ﺗﻘﺮﻳﺒﺎ ﺗﻤﺎﻣﯽ ﺍﻃﻼﻋﺎﺕ ﻻﺯﻡ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﻣﯽ ﺁﻭﺭﺩ. ﻣﺎﻫﻴﺖ ﻭ ﻃﺒﻴﻌﺖ ﻳﮏ ﺳﺮ ﺗﻴﺘﺮ ﺧﺒﺮﯼ ﺧﻮﺏ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻫﻤﻮﺍﺭﻩ ﺍﻃﻼﻋﺎﺕ ﮐﺎﺭﺑﺮﺩﯼ ﺭﺍ ﺑﻪ ﺭﺍﺣﺘﯽ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﮔﻴﺮﻧﺪﻩ ﺑﮕﺬﺍﺭﺩ. ﻫﺮ ﮔﺎﻩ ﮐﻪ ﮔﻴﺮﻧﺪﻩ ﺍﻳﻦ ﺍﻳﻤﻴﻞ ﻧﮕﺎﻫﯽ ﺳﺮﺳﺮﯼ ﺑﻪ ﺍﻳﻨﺒﺎﮐﺲ ﺍﺵ ﺑﻴﺎﻧﺪﺍﺯﻧﺪ، ﻗﺮﺍﺭ ﻣﻼﻗﺎﺕ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﻣﯽ ﺁﻭﺭﺩ.

 

- ﺩﺭ ﻫﺮ ﺍﻳﻤﻴﻞ ﻓﻘﻂ ﺑﺮ ﺭﻭﯼ ﻳﮏ ﻣﻮﺿﻮﻉ ﺗﻤﺮﮐﺰ ﮐﻨﻴﺪ

ﻳﮑﯽ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻳﻦ ﻣﺰﻳﺖ ﻫﺎ ﻭ ﭘﻴﺸﺮﻓﺖ ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﻧﺴﺒﺖ ﺑﻪ ﻧﺎﻣﻪ ﻫﺎﯼ ﭘﺴﺘﯽ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺮ ﺧﻼﻑ ﺳﻴﺴﺘﻢ ﭘﺴﺘﯽ، ﺍﺭﺳﺎﻝ ﻳﮏ ﺍﻳﻤﻴﻞ ﻳﺎ ﭼﻨﺪﻳﻦ ﺍﻳﻤﻴﻞ، ﺍﺯ ﻧﻈﺮ ﻫﺰﻳﻨﻪ ﺗﻔﺎﻭﺗﯽ ﻧﺪﺍﺭﺩ! ﺧﺐ، ﭘﺲ ﺍﮔﺮ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺩﺭﺑﺎﺭﻩ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﺍﺯ ﻣﻮﺿﻮﻋﺎﺕ ﻣﺨﺘﻠﻒ ﺑﻪ ﺷﺨﺼﯽ ﺍﻳﻤﻴﻞ ﺑﺰﻧﻴﺪ، ﻟﻄﻔﺎ ﺩﺭ ﻫﺮ ﺍﻳﻤﻴﻞ ﻓﻘﻂ ﺑﻪ ﻳﮏ ﻣﻮﺿﻮﻉ ﻣﺠﺰﺍ ﺑﭙﺮﺩﺍﺯﻳﺪ ﻭ ﮐﺎﺭﻫﺎﯼ ﺗﺎﻥ ﺭﺍ ﺩﺭ ﭼﻨﺪ ﺍﻳﻤﻴﻞ ﺑﻪ ﻭﯼ ﺍﻃﻼﻉ ﺩﻫﻴﺪ. 

 

ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﻃﺮﻑ ﺣﺴﺎﺏ ﺷﻤﺎ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻫﺮ ﺳﻮﺍﻝ ﻭ ﻣﻮﺿﻮﻉ ﺑﻪ ﺻﻮﺭﺕ ﺟﺪﺍﮔﺎﻧﻪ ﻭ ﺩﺭ ﻭﻗﺖ ﻣﻨﺎﺳﺐ ﭘﺎﺳﺦ ﻻﺯﻡ ﺭﺍ ﺑﺪﻫﺪ. ﻳﮏ ﻣﻮﺿﻮﻉ ﻣﺸﺨﺺ ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﻘﻂ ﺑﻪ ﻳﮏ ﺟﻮﺍﺏ ﮐﻮﺗﺎﻩ ﻧﻴﺎﺯ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﮐﻪ ﮔﻴﺮﻧﺪﻩ ﺍﻳﻤﻴﻞ ﻫﻤﺎﻥ ﻟﺤﻈﻪ ﺁﻥ ﺭﺍ ﭘﺎﺳﺦ ﺧﻮﺍﻫﺪ ﺩﺍﺩ. ﺑﻘﻴﻪ ﻣﻮﺿﻮﻋﺎﺕ ﻫﻢ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﻣﻄﺎﻟﻌﻪ، ﺗﺤﻘﻴﻖ ﻭ ﺻﺮﻑ ﻭﻗﺖ ﺑﻴﺸﺘﺮﯼ ﻧﻴﺎﺯ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ، ﮐﻪ ﺩﺭ ﺟﺎﯼ ﺧﻮﺩ ﭘﺎﺳﺦ ﺩﺍﺩﻩ ﻣﯽ ﺷﻮﻧﺪ. ﺑﺎ ﺍﺭﺳﺎﻝ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺟﺪﺍﮔﺎﻧﻪ، ﺷﻤﺎ ﺟﻮﺍﺏ ﻫﺎﯼ ﺭﻭﺷﻦ ﺗﺮﯼ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﺧﻮﺍﻫﻴﺪ ﮐﺮﺩ. ﻫﻤﭽﻨﻴﻦ ﺑﻪ ﮔﻴﺮﻧﺪﻩ ﮐﻤﮏ ﻣﯽ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻨﺒﺎﮐﺲ ﺍﺵ ﺭﺍ ﺑﻬﺘﺮ ﻣﺪﻳﺮﻳﺖ ﮐﻨﺪ ﻭ ﻧﻈﻢ ﺑﺪﻫﺪ. 

 

ﺍﮔﺮ ﻫﻢ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﻴﺪ ﭼﻨﺪﻳﻦ ﻣﻮﺿﻮﻉ ﻣﺨﺘﻠﻒ ﺭﺍ ﺩﺭ ﻳﮏ ﺍﻳﻤﻴﻞ ﻗﺮﺍﺭ ﺩﻫﻴﺪ (ﻣﺜﻼ ﺑﺨﺎﻃﺮ ﺍﻳﻨﮑﻪ ﻫﻤﻪ ﺁﻧﻬﺎ ﻣﺮﺑﻮﻁ ﺑﻪ ﻳﮏ ﭘﺮﻭﮊﻩ ﺍﻧﺪ)، ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺮ ﻣﻮﺿﻮﻉ ﺩﺭ ﻳﮏ ﭘﺎﺭﺍﮔﺮﺍﻑ ﺟﺪﺍﮔﺎﻧﻪ ﻭ ﺷﻤﺎﺭﻩ ﮔﺬﺍﺭﯼ ﺷﺪﻩ ﻧﻮﺷﺘﻪ ﺷﻮﺩ. ﺍﻳﻦ ﮐﺎﺭ ﻫﺮ ﻣﻮﺿﻮﻉ ﺭﺍ ﺟﺪﺍ ﻭ ﺩﺍﺭﺍﯼ ﺷﺨﺼﻴﺖ ﻣﺴﺘﻘﻞ ﻣﯽ ﻧﻤﺎﻳﺪ ﻭ ﺑﻪ ﻃﻮﺭ ﻣﻌﻨﯽ ﺩﺍﺭﯼ ﺍﻳﻦ ﺍﺣﺘﻤﺎﻝ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﺩﻫﺪ ﮐﻪ ﻫﺮ ﻣﻮﺿﻮﻉ ﺑﺎ ﺍﺷﺎﺭﻩ ﺑﻪ ﺷﻤﺎﺭﻩ ﻭ ﺩﻗﻴﻖ ﺗﺮ ﺟﻮﺍﺏ ﺩﺍﺩﻩ ﺷﻮﺩ. 

 

ﻫﺮ ﺍﻳﻤﻴﻞ ﺑﻪ ﺻﻮﺭﺕ ﺟﺪﺍﮔﺎﻧﻪ ﺑﺎﻳﺪ ﻣﺨﺘﺼﺮ، ﻣﻮﺟﺰ ﻭ ﻣﺸﺨﺺ ﺑﺎﺷﺪ. ﺟﺰﺋﻴﺎﺕ ﻭ ﻫﺪﻑ ﺍﺯ ﻧﮕﺎﺭﺵ ﺁﻥ ﺩﺭ ﻫﻤﺎﻥ ﭘﺎﺭﺍﮔﺮﺍﻑ ﺍﻭﻝ ﺑﻪ ﺧﻮﺑﯽ ﮔﻨﺠﺎﻧﺪﻩ ﺷﻮﺩ. ﺟﻤﻼﺕ ﺑﺎﻳﺪ ﮐﻮﺗﺎﻩ ﻭ ﻫﺪﻓﻤﻨﺪ ﻭ ﻣﺮﺗﺒﻂ ﺑﺎ ﻣﻮﺿﻮﻉ ﺑﺎﺷﻨﺪ. ﻣﺘﻦ ﺍﻳﻤﻴﻞ ﺑﺎﻳﺪ ﺣﺎﻭﯼ ﺗﻤﺎﻣﯽ ﺍﻃﻼﻋﺎﺕ ﺿﺮﻭﺭﯼ ﻭ ﻣﻨﺎﺳﺐ ﺑﺎ ﻣﻮﺿﻮﻉ ﺑﺎﺷﺪ. ﺗﻤﺎﻡ ﻣﺘﻦ ﺑﺎﻳﺪ ﺳﺮ ﺭﺍﺳﺖ ﻭ ﺁﮔﺎﻫﯽ ﺑﺨﺶ ﺑﺎﺷﺪ.

ﻳﮏ ﺍﻳﻤﻴﻞ ﺑﺪ:

ﺳﺎﺑﺠﮑﺖ: ﻧﺴﺨﻪ ﺟﺪﻳﺪ ﮔﺰﺍﺭﺵ ﻓﺮﻭﺵ 

ﻣﺘﻦ ﻧﺎﻣﻪ:

ﺳﻼﻡ ﻣﺮﻳﻢ

ﺑﺎﺑﺖ ﮔﺰﺍﺭﺵ ﺍﺭﺳﺎﻟﯽ ﻫﻔﺘﻪ ﮔﺬﺷﺘﻪ ﻣﺘﺸﮑﺮﻡ. ﻣﻦ ﺁﻥ ﺭﺍ ﺩﻳﺮﻭﺯ ﻣﻄﺎﻟﻌﻪ ﮐﺮﺩﻡ ﻭ ﻓﮑﺮ ﻣﯽ ﮐﻨﻢ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﺑﺨﺶ ﺍﻋﺪﺍﺩ ﻭ ﺍﺭﻗﺎﻡ ﻓﺮﻭﺵ ﮐﻤﯽ ﺩﻗﻴﻖ ﺗﺮ ﻋﻤﻞ ﮐﻨﯽ. ﻫﻤﭽﻨﻴﻦ ﺑﻪ ﻧﻈﺮ ﻣﯽ ﺭﺳﺪ ﮐﻪ ﻟﺤﻦ ﮔﺰﺍﺭﺵ ﺑﺎﻳﺪ ﮐﻤﯽ ﺭﺳﻤﯽ ﺗﺮ ﺑﺎﺷﺪ. ﺍﻳﻦ ﮔﺰﺍﺭﺵ ﻗﺮﺍﺭ ﺍﺳﺖ ﺗﻮﺳﻂ ﻫﻴﺌﺖ ﺭﺋﻴﺴﻪ ﻣﻄﺎﻟﻌﻪ ﺷﻮﺩ ﻭ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﻧﺸﺎﻧﮕﺮ ﺣﺮﻓﻪ ﺍﯼ ﺑﻮﺩﻥ ﻣﺎ ﺑﺎﺷﺪ.

 

ﻫﻤﭽﻨﻴﻦ ﻣﻦ ﺑﺎ ﮔﺮﻭﻩ ﺧﺒﺮﯼ ﺷﺮﮐﺖ ﻫﻢ ﻗﺮﺍﺭ ﻣﻼﻗﺎﺗﯽ ﺑﺮﺍﯼ ﺭﻭﺯ ﭘﻨﺠﺸﻨﺒﻪ ﺟﻬﺖ ﺑﺮﺭﺳﯽ ﺑﺮﻧﺎﻣﻪ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺟﺪﻳﺪﻣﺎﻥ ﺗﻨﻈﻴﻢ ﮐﺮﺩﻡ. ﺍﻳﻦ ﺑﺮﻧﺎﻣﻪ ﺳﺎﻋﺖ ۱۱ ﺻﺒﺢ ﺩﺭ ﺍﺗﺎﻕ ﮐﻨﻔﺮﺍﻧﺲ ﺷﺮﮐﺖ ﺑﺮﮔﺰﺍﺭ ﺧﻮﺍﻫﺪ ﺷﺪ.

 

ﻟﻄﻔﺎ ﺩﺭ ﺧﺼﻮﺹ ﺍﻣﮑﺎﻥ ﺣﻀﻮﺭﺕ ﺩﺭ ﺁﻥ ﺯﻣﺎﻥ، ﺯﻭﺩﺗﺮ ﺑﻪ ﻣﻦ ﺧﺒﺮ ﺑﺪﻩ.

ﻣﺘﺸﮑﺮﻡ

ﻣﻴﻼﺩ

ﻣﻴﻼﺩ ﺳﺎﺑﺠﮑﺖ ﻧﺴﺒﺘﺎ ﻣﻨﺎﺳﺒﯽ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ ﻭ ﺩﺭ ﺧﺼﻮﺹ ﺗﻐﻴﻴﺮﺍﺕ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺩﺭ ﮔﺰﺍﺭﺵ ﺑﻪ ﺻﻮﺭﺕ ﮐﺎﻣﻼ ﺭﻭﺷﻦ ﻭ ﻗﺎﺑﻞ ﺩﺭﮐﯽ ﺳﺨﻦ ﮔﻔﺘﻪ ﺍﺳﺖ.  ﭘﺲ ﺍﺷﺘﺒﺎﻩ ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺍﻳﻤﻴﻞ ﭼﻪ ﺑﻮﺩﻩ ﺍﺳﺖ؟

ﺧﺐ، ﭘﺎﺭﺍﮔﺮﺍﻑ ﺩﻭﻡ ﺩﺭﺑﺎﺭﻩ ﻗﺮﺍﺭ ﻣﻼﻗﺎﺕ، ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ. ﺍﻣﺎ ﻣﻴﻼﺩ ﺁﻥ ﺭﺍ ﺑﺎ ﺗﻘﺎﺿﺎ ﺩﺭ ﺧﺼﻮﺹ ﻧﺴﺨﻪ ﺟﺪﻳﺪ ﮔﺰﺍﺭﺵ ﻳﮑﺠﺎ ﺭﻭﯼ ﻫﻢ ﺗﻠﻨﺒﺎﺭ ﮐﺮﺩﻩ ﺍﺳﺖ.  ﭘﺲ ﺍﮔﺮ ﻣﺮﻳﻢ ﺍﻳﻦ ﻗﺮﺍﺭ ﻣﻼﻗﺎﺕ ﺭﺍ ﺩﺭ ﺗﻘﻮﻳﻢ ﺍﺵ ﻳﺎﺩﺩﺍﺷﺖ ﻧﮑﺮﺩﻩ ﺑﺎﺷﺪ، ﺑﺎﻳﺪ ﺑﺎ ﺩﻳﺪﻥ ﻋﻨﻮﺍﻥ «ﻧﺴﺨﻪ ﺟﺪﻳﺪ ﮔﺰﺍﺭﺵ ﻓﺮﻭﺵ» ﺁﻥ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺑﻴﺎﻭﺭﺩ! ﺑﻪ ﻧﻈﺮ ﺷﻤﺎ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﻣﻨﻄﻘﯽ ﺍﺳﺖ؟

 

ﺗﺮﮐﻴﺐ ﮐﺮﺩﻥ ﺩﻭ ﻣﻮﺿﻮﻉ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺩﺭ ﻳﮏ ﺍﻳﻤﻴﻞ، ﺷﺎﻧﺲ ﺍﻳﻨﮑﻪ ﻳﮑﯽ ﺍﺯ ﺁﻥ ﺩﻭ ﻓﺮﺍﻣﻮﺵ ﺷﻮﺩ ﺭﺍ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩ ﻣﯽ ﮐﻨﺪ. ﺑﮕﺬﺍﺭﻳﺪ ﺑﺒﻴﻨﻴﻢ ﻣﻴﻼﺩ ﭼﮕﻮﻧﻪ ﺑﺎﻳﺪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﻪ ﺷﮑﻞ ﺑﻬﺘﺮﯼ ﺍﻧﺠﺎﻡ ﺩﻫﺪ.
