ﺍﺻﻮﻝ ﻣﺪﻳﺮﻳﺖ ﺍﻳﻤﻴﻞ - ﺑﺨﺶ ﺍﻭﻝ

ﻣﻄﻤﺌﻨﺎ ﺩﺍﺷﺘﻦ ﺍﻳﻨﺒﺎﮐﺴﯽ ﮐﻪ ﺷﺘﺮ ﻫﻢ ﺑﺎ ﺑﺎﺭﺵ ﺩﺭ ﺁﻥ ﮔﻢ ﺷﻮﺩ، ﺳﺎﻋﺖ ﻫﺎﯼ ﻣﺘﻤﺎﺩﯼ ﺍﺯ ﻭﻗﺖ ﻣﻔﻴﺪ ﺭﻭﺯﺍﻧﻪ ﺷﻤﺎ ﺭﺍ ﺗﻠﻒ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎﻋﺚ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻥ ﻧﺎﻣﻪ ﻫﺎﯼ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﺷﻮﺩ ﮐﻪ ﺯﻧﺪﮔﯽ ﺷﻤﺎ ﺭﺍ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﻗﺮﺍﺭ ﻣﯽ ﺩﻫﻨﺪ. ﺩﺭ ﺍﻳﻦ ﺩﻭ ﺩﺭﺱ ﭘﺎﻳﺎﻧﯽ ﺗﺼﻤﻴﻢ ﺩﺍﺭﻳﻢ ﺑﻪ ﺷﻴﻮﻩ ﻫﺎﯼ ﺻﺤﻴﺢ ﻭ ﮐﺎﺭﺑﺮﺩﯼ ﻣﺪﻳﺮﻳﺖ ﺍﻳﻤﻴﻞ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﺍﻳﻨﮑﻪ ﭼﮕﻮﻧﻪ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺗﺎﻥ ﺭﺍ ﺑﺨﻮﺍﻧﻴﺪ، ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﮐﻨﻴﺪ، ﭘﺎﺳﺦ ﺩﻫﻴﺪ ﻭ ﻫﻤﻴﺸﻪ "ﺻﻨﺪﻭﻕ ﻭﺭﻭﺩﯼ" ﺟﻤﻊ ﻭ ﺟﻮﺭﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻦ ﻧﮑﺎﺕ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﺎﺯﺩﻩ ﻭ ﮐﺎﺭﺍﻳﯽ ﺧﻮﺩ ﺭﺍ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺍﻓﺰﺍﻳﺶ ﺩﺍﺩﻩ ﻭ ﻭﻗﺖ ﺍﺿﺎﻓﻪ ﺍﯼ ﺭﺍ ﮐﻪ ﺑﻪ ﺩﺳﺖ ﻣﯽ ﺁﻭﺭﻳﺪ ﺑﻪ ﮐﺎﺭﻫﺎﯼ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺗﺎﻥ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﺷﻤﺎ ﺑﻪ «ﺍﺳﺘﺎﺩ ﺗﻤﺎﻣﯽ ﺍﻳﻤﻴﻞ» ﻧﺰﺩﻳﮏ ﺷﺪﻩ ﺍﻳﺪ.

- ﺑﻪ ﺑﺮﻧﺎﻣﻪ ﻣﺪﻳﺮﻳﺖ ﺍﻳﻤﻴﻞ ﻭ ﻳﺎ ﺳﺮﻭﻳﺲ ﺩﻫﻨﺪﻩ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﺭﺍ ﻣﺮﺗﺐ ﮐﻨﻨﺪ.
ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ ﻫﺎﯼ ﻣﻨﻈﻢ ﮐﺮﺩﻥ ﺍﻳﻨﺒﺎﮐﺲ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻓﻴﻠﺘﺮﻫﺎ ﺍﺳﺖ. ﺍﮔﺮ ﺍﺯ ﺑﺮﻧﺎﻣﻪ ﺍﯼ ﻣﺜﻞ ﺍﻭﺕ ﻟﻮﮎ ﻭ ﺗﺎﻧﺪﺭ ﺑﺮﺩ، ﻳﺎ ﺳﺮﻭﻳﺴﯽ ﻣﺎﻧﻨﺪ ﺟﻴﻤﻴﻞ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﮐﻪ ﺑﺎ ﺗﻌﺮﻳﻒ ﻓﻴﻠﺘﺮ ﻣﻨﺎﺳﺐ ﺍﺳﭙﻢ ﻫﺎ (ﻫﺮﺯﻧﺎﻣﻪ) ﺭﺍ ﻫﻤﻴﺸﻪ ﻣﺴﺘﻘﻴﻤﺎ ﺑﻪ "ﺣﺬﻑ ﺷﺪﻩ ﻫﺎ" (Trash) ﺳﺮﺍﺯﻳﺮ ﮐﻨﻴﺪ ﻭ ﻭﻗﺖ ﺗﺎﻥ ﺭﺍ ﺑﺎ ﺑﺎﺯ ﮐﺮﺩﻥ ﻭ ﭘﺎﮎ ﮐﺮﺩﻥ ﺁﻧﻬﺎ ﻫﺪﺭ ﻧﺪﻫﻴﺪ. ﺩﺭ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﺮﭼﺴﺐ ﻭ ﻓﻴﻠﺘﺮﻫﺎ ﺩﺭ ﺩﺭﺱ ﻫﺎﯼ ﮔﺬﺷﺘﻪ ﺑﻪ ﺻﻮﺭﺕ ﻣﻔﺼﻞ ﺻﺤﺒﺖ ﮐﺮﺩﻩ ﺍﻳﻢ ﻭ ﺍﺣﺘﻤﺎﻻ ﻫﻢ ﺍﮐﻨﻮﻥ ﻣﺸﻐﻮﻝ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻧﻬﺎ ﻫﺴﺘﻴﺪ.

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺩﺳﺘﻮﺭﺍﺕ ﺭﺳﻴﺪﮔﯽ ﺧﻮﺩﮐﺎﺭ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﻣﺸﺨﺺ، ﺗﺎ ﺣﺪ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺑﻬﺮﻩ ﻭﺭﯼ ﺷﻤﺎ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﺩﻫﺪ. ﻣﺜﻼ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﻫﺮ ﺍﻳﻤﻴﻠﯽ ﮐﻪ ﺍﺯ ﺩﺭﺳﻨﺎﻣﻪ ﺑﻪ ﺷﻤﺎ ﻣﯽ ﺭﺳﺪ، ﻟﻴﺒﻴﻞ ﺧﺎﺻﯽ ﭘﻴﺪﺍ ﻣﯽ ﮐﻨﺪ ﻭ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﻗﺮﺍﺭ ﻣﯽ ﮔﻴﺮﺩ.  ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﻳﻤﻴﻞ ﮔﺮﻭﻩ ﻫﺎﯼ ﻳﺎﻫﻮ ﺑﺎ ﻳﮏ ﺳﺎﺑﺠﮑﺖ ﺧﺎﺹ ﺭﺍ  ﮐﻪ ﺑﻪ ﮐﺎﺭﺗﺎﻥ ﻧﻤﯽ ﺁﻳﻨﺪ ﻭ ﺗﻨﻬﺎ ﺑﺮﺧﯽ ﺍﻭﻗﺎﺕ ﺁﻧﻬﺎ ﺭﺍ ﻣﻄﺎﻟﻌﻪ ﻣﯽ ﮐﻨﻴﺪ، ﻣﺴﺘﻘﻴﻤﺎ ﻭﺍﺭﺩ ﭘﻮﺷﻪ ﺧﺎﺹ ﺧﻮﺩﺷﺎﻥ ﻣﯽ ﺷﻮﻧﺪ ﻭ ﻳﺎ ﺑﻪ ﺁﺭﺷﻴﻮ ﻣﯽ ﺭﻭﻧﺪ. ﭘﺲ ﺑﺎﻋﺚ ﺷﻠﻮﻏﯽ ﺍﻳﻨﺒﺎﮐﺲ ﻭ ﺗﻠﻒ ﺷﺪﻥ ﻭﻗﺖ ﺗﺎﻥ ﻧﻤﯽ ﮔﺮﺩﻧﺪ.

- ﻻﺯﻡ ﻧﻴﺴﺖ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻟﺤﻈﻪ ﺍﯼ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺑﻪ ﻣﺤﺾ ﺩﺭﻳﺎﻓﺖ ﺍﻳﻤﻴﻞ ﺁﻥ ﺭﺍ ﭘﺎﺳﺦ ﺩﻫﻴﺪ.
ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ ﮐﻨﻢ ﺍﺻﻼ ﺍﺯ ﺑﺮﻧﺎﻣﻪ ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﻣﺤﺾ ﺩﺭﻳﺎﻓﺖ ﺍﻳﻤﻴﻞ ﺟﺪﻳﺪ، ﺑﺎ ﺑﺎﺯ ﮐﺮﺩﻥ ﭘﻨﺠﺮﻩ ﻳﺎ ﺳﺮﻭﺻﺪﺍ ﻭ ﻳﺎ ﻫﺮ ﺭﺍﻩ ﺩﻳﮕﺮﯼ ﺑﻪ ﺷﻤﺎ ﺍﺧﻄﺎﺭ ﻣﯽ ﺩﻫﻨﺪ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﺪ. ﺍﻳﻦ ﻋﺎﺩﺕ ﺭﺍ ﻫﻢ ﮐﻪ ﺩﺍﺋﻤﺎ ﻭ ﻟﺤﻈﻪ ﺍﯼ ﺑﻪ ﺍﻳﻤﻴﻞ ﺗﺎﻥ ﺳﺮ ﺑﺰﻧﻴﺪ ﮐﻨﺎﺭ ﺑﮕﺬﺍﺭﻳﺪ.

ﺯﻳﺮﺍ ﭼﮏ ﮐﺮﺩﻥ ﺍﻳﻤﻴﻞ ﺑﻪ ﻣﺤﺾ ﺩﺭﻳﺎﻓﺖ ﻫﺮ ﭘﻴﺎﻡ ﺑﻪ ﻃﻮﺭ ﺟﺪﯼ ﺩﺭ ﺗﻤﺮﮐﺰ ﺷﻤﺎ ﺑﺮ ﺭﻭﯼ ﮐﺎﺭ ﻭ ﻓﻌﺎﻟﻴﺘﯽ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﺍﻧﺠﺎﻡ ﺁﻥ ﻫﺴﺘﻴﺪ ﺗﺪﺍﺧﻞ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﺪ. ﺑﻬﺘﺮ ﺍﺳﺖ ﻓﺎﺻﻠﻪ ﻫﺎﯼ ﺯﻣﺎﻧﯽ ﻣﺸﺨﺼﯽ ﺩﺭ ﻃﻮﻝ ﺭﻭﺯ ﺭﺍ ﺑﻪ ﭼﮏ ﮐﺮﺩﻥ ﺍﻳﻤﻴﻞ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ. ﺍﻳﻦ ﺑﺎﺯﻩ ﺯﻣﺎﻧﯽ ﺑﻪ ﻧﺴﺒﺖ ﺣﺴﺎﺳﻴﺖ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺩﺭﻳﺎﻓﺘﯽ ﻭ ﺗﻌﺪﺍﺩ ﺁﻧﻬﺎ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺯ ۳۰ ﺩﻗﻴﻘﻪ ﻳﮏ ﺑﺎﺭ ﺗﺎ ﺭﻭﺯﯼ ﻳﮏ ﺑﺎﺭ ﻣﺘﻐﻴﺮ ﺑﺎﺷﺪ.

ﻣﻄﻤﺌﻨﺎ ﺩﺭ ﻫﺮ ﻟﺤﻈﻪ ﺍﻧﺠﺎﻡ ﻓﻘﻂ ﻳﮏ ﮐﺎﺭ ﻭ ﺩﻗﺖ ﺑﺮ ﺭﻭﯼ ﺁﻥ، ﺑﻬﺮﻩ ﻭﺭﯼ ﻭ ﮐﺎﺭﺍﻳﯽ ﺷﻤﺎ ﺭﺍ ﺗﺎ ﺣﺪ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺍﻓﺰﺍﻳﺶ ﺧﻮﺍﻫﺪ ﺩﺍﺩ.

- ﻫﻤﻪ ﺭﻭﺯﺗﺎﻥ ﺭﺍ ﺑﻪ ﺧﻮﺍﻧﺪﻥ ﻭ ﺟﻮﺍﺏ ﺩﺍﺩﻥ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎ ﻧﮕﺬﺭﺍﻧﻴﺪ.
ﺷﻤﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﻫﺮ ﺭﻭﺯ ﺩﻩ ﻫﺎ ﻳﺎ ﺻﺪﻫﺎ ﺍﻳﻤﻴﻞ ﺩﺭﻳﺎﻓﺖ ﮐﻨﻴﺪ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﭘﺎﺳﺦ ﺩﺍﺭﻧﺪ. ﺍﻣﺎ ﻣﻄﻤﺌﻨﺎ ﺍﮐﺜﺮ ﺁﻧﻬﺎ ﻓﻮﺭﯼ ﻧﻴﺴﺘﻨﺪ ﻭ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﻌﺪﺍ ﻫﻢ ﺑﻪ ﺁﻧﻬﺎ ﭘﺎﺳﺦ ﺩﻫﻴﺪ. ﺍﻳﻨﮕﻮﻧﻪ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎ ﻣﻄﻤﺌﻨﺎ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﻭﻗﻔﻪ ﺩﺭ ﺭﻭﺍﻝ ﻋﺎﺩﯼ ﮐﺎﺭﻫﺎﯼ ﺗﺎﻥ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺯﻣﺎﻧﯽ ﻣﺸﺨﺺ ﺭﻭﺯﺍﻧﻪ ﺭﺍ ﺑﺮﺍﯼ ﭘﺎﺳﺨﮕﻮﻳﯽ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺭﻭﺯﯼ ﻳﮏ ﻳﺎ ﺩﻭ ﺳﺎﻋﺖ ﻭ ﺣﺘﯽ ﺑﻴﺸﺘﺮ ﺯﻣﺎﻥ ﻧﻴﺎﺯ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﻣﺎ ﺗﻼﺵ ﮐﻨﻴﺪ ﻓﻘﻂ ﻃﯽ ﻫﻤﻴﻦ ﺯﻣﺎﻥ ﻣﺸﺨﺺ ﻭ ﺑﺎ ﺭﻋﺎﻳﺖ ﺍﻭﻟﻮﻳﺖ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎﯼ ﺩﺭﻳﺎﻓﺘﯽ ﭘﺎﺳﺦ ﺩﻫﻴﺪ.

- ﺳﺎﻋﺎﺕ ﭘﺮﺑﺎﺯﺩﻩ ﻭ ﻣﻔﻴﺪ ﺭﻭﺯﺗﺎﻥ ﺭﺍ ﺑﺎ ﭘﺎﺳﺨﮕﻮﻳﯽ ﺑﻪ ﺍﻳﻤﻴﻞ ﻫﺎ ﺗﻠﻒ ﻧﮑﻨﻴﺪ.
