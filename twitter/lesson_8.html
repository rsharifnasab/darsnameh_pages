ﺗﻮﻳﻴﺘﺮ ﺍﻣﮑﺎﻧﯽ ﺭﺍ ﻓﺮﺍﻫﻢ ﮐﺮﺩﻩ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﮐﻤﮏ ﺁﻥ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺣﺴﺎﺏ ﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﻧﻈﺮ ﺷﻤﺎ ﻣﺸﻐﻮﻝ ﺍﺭﺳﺎﻝ ﺍﺳﭙﻢ ﻳﺎ ﻫﻤﺎﻥ ﻫﺮﺯﻧﺎﻣﻪ ﻫﺴﺘﻨﺪ ﺭﺍ ﮔﺰﺍﺭﺵ ﮐﻨﻴﺪ ﺗﺎ ﺗﻮﻳﻴﺘﺮ ﺑﺮﺍﺳﺎﺱ ﺗﻌﺪﺍﺩ ﮔﺰﺍﺭﺵ ﻫﺎﯼ ﺩﺭﻳﺎﻓﺘﯽ ﻭ ﺑﺮﺭﺳﯽ ﻫﺎﻳﯽ ﮐﻪ ﺍﻧﺠﺎﻡ ﻣﯽ ﺩﻫﺪ، ﺁﻥ ﺣﺴﺎﺏ ﺭﺍ ﻣﺴﺪﻭﺩ ﮐﻨﺪ. ﺍﮔﺮ ﺯﻣﺎﻧﯽ ﺷﻤﺎ ﻧﻴﺰ ﺑﺎ ﻳﮏ ﺣﺴﺎﺏ ﺩﺭ ﺗﻮﻳﻴﺘﺮ ﻣﻮﺍﺟﻪ ﺷﺪﻳﺪ ﮐﻪ ﻣﺸﻐﻮﻝ ﺍﺳﭙﻢ ﭘﺮﺍﮐﻨﯽ ﺑﻮﺩ، ﺑﻪ ﭘﺮﻭﻓﺎﻳﻞ ﺁﻥ ﺣﺴﺎﺏ ﻣﺮﺍﺟﻌﻪ ﮐﻨﻴﺪ ﻭ ﺳﭙﺲ ﻣﺎﻧﻨﺪ ﺗﺼﻮﻳﺮ ﺯﻳﺮ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻣﺴﺪﻭﺩ ﮐﺮﺩﻥ ﻳﺎ ﮔﺰﺍﺭﺵ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

 

ﻋﻼﻭﻩ ﺑﺮ ﺁﻥ ﺍﮔﺮ ﺧﻮﺍﺳﺘﻴﺪ ﺣﺴﺎﺏ ﺗﻮﻳﻴﺘﺮﯼ ﺭﺍ ﺑﻼﮎ ﮐﻨﻴﺪ، ﺑﻪ ﭘﺮﻭﻓﺎﻳﻞ ﺣﺴﺎﺑﯽ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﻣﺴﺪﻭﺩ ﮐﻨﻴﺪ ﻣﺮﺍﺟﻌﻪ ﮐﻨﻴﺪ ﻭ ﺳﭙﺲ ﺍﺯ ﻃﺮﻳﻖ ﭼﺮﺥ‌ﺩﻧﺪﻩ ﺳﻤﺖ ﭼﭗ ﺻﻔﺤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﮐﺎﺭﺑﺮ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﻣﺴﺪﻭﺩ ﮐﻨﻴﺪ.

 

 

 

ﻧﮑﺘﻪ ۱: ﻭﻗﺘﯽ ﺷﻤﺎ ﺣﺴﺎﺑﯽ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺍﺳﭙﻤﺮ ﮔﺰﺍﺭﺵ ﮐﻨﻴﺪ، ﺗﻮﻳﻴﺘﺮ ﺁﻥ ﺣﺴﺎﺏ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﻼﮎ ﻣﯽ ﮐﻨﺪ.

 

ﻧﮑﺘﻪ ۲: ﺑﺮﺍﯼ ﺁﻧﮑﻪ ﺣﺴﺎﺑﯽ ﺭﺍ ﮐﻪ ﺑﻼﮎ ﮐﺮﺩﻩ ﺍﻳﺪ، ﺍﺯ ﺍﻳﻦ ﺣﺎﻟﺖ ﺧﺎﺭﺝ ﮐﻨﻴﺪ، ﺩﻗﻴﻘﺎ ﻣﺴﻴﺮﯼ ﮐﻪ ﺩﺭ ﺑﺎﻻ ﺑﺮﺍﯼ ﺑﻼﮎ ﮐﺮﺩﻥ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ ﺷﺪﻩ ﺍﺳﺖ ﺭﺍ ﺑﺎﻳﺪ ﻃﯽ ﮐﻨﻴﺪ.

 

ﺍﻳﺠﺎﺩ ﻭ ﺣﺬﻑ ﻳﮏ ﻟﻴﺴﺖ

 

ﻟﻴﺴﺖ (List) ﻳﮑﯽ ﺍﺯ ﻭﻳﮋﮔﯽ ﻫﺎﯼ ﺗﻮﻳﻴﺘﺮ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﮐﻤﮏ ﺁﻥ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺣﺴﺎﺏ ﻫﺎﯼ ﺗﻮﻳﻴﺘﺮﯼ ﮐﻪ ﺩﻧﺒﺎﻝ ﻣﯽ ﮐﻨﻴﺪ ﺭﺍ ﺑﺮﺍﺳﺎﺱ ﻋﻼﻗﻪ ﺧﻮﺩ ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺯ ﺩﻭﺳﺘﺎﻥ ﺻﻤﻴﻤﯽ ﺷﻤﺎ ﺷﺮﻭﻉ ﺷﻮﺩ ﻭ ﺑﻪ ﺧﺒﺮﮔﺰﺍﺭﯼ ﻫﺎ ﺧﺘﻢ ﺷﻮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺍﮔﺮ ﺷﻤﺎ ﭼﻨﺪﻳﻦ ﺣﺴﺎﺏ ﺗﻮﻳﻴﺘﺮ ﺧﺒﺮﮔﺰﺍﺭﯼ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﺭ ﺳﺮﺗﺎﺳﺮ ﺩﻧﻴﺎ ﺭﺍ ﺩﻧﺒﺎﻝ ﮐﻨﻴﺪ ﻭ ﺑﺨﻮﺍﻫﻴﺪ ﺗﻨﻬﺎ ﺗﻮﻳﻴﺖ ﻫﺎﯼ ﺍﺭﺳﺎﻝ ﺍﺯ ﺳﻮﯼ ﺍﻳﻦ ﺧﺒﺮﮔﺰﺍﺭﯼ ﻫﺎ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻴﺪ، ﺑﻬﺘﺮﻳﻦ ﮐﺎﺭ ﺍﻳﺠﺎﺩ ﻳﮏ ﻟﻴﺴﺖ ﻭ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﺍﻳﻦ ﺧﺒﺮﮔﺰﺍﺭﯼ ﻫﺎ ﺩﺭ ﺁﻥ ﻟﻴﺴﺖ ﺍﺳﺖ.

 

ﺑﺎ ﺍﻳﺠﺎﺩ ﻟﻴﺴﺖ، ﺷﻤﺎ ﺩﺭ ﺣﻘﻴﻘﺖ ﻳﮏ ﻓﻴﻠﺘﺮ ﺑﺮﺍﯼ ﺣﺴﺎﺏ ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﮐﻪ ﺩﻧﺒﺎﻝ ﻣﯽ ﮐﻨﻴﺪ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺍﻳﺪ ﺗﺎ ﻫﺮ ﺯﻣﺎﻥ ﮐﻪ ﺧﻮﺍﺳﺘﻴﺪ ﺗﻮﻳﻴﺖ ﻫﺎﯼ ﺧﺎﺻﯽ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻴﺪ.

 

ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﻳﮏ ﻟﻴﺴﺖ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ Lists(ﻓﻬﺮﺳﺖ‌ﻫﺎ) ﺍﺯ ﻃﺮﻳﻖ ﮔﺰﻳﻨﻪ ﺗﻨﻈﻴﻤﺎﺕ ﺩﺭ ﻧﻮﺍﺭﺍﺑﺰﺍﺭ ﮐﻪ ﺩﺭ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺗﻮﻳﻴﺘﺮ ﻭ ﻳﺎ ﭘﺮﻭﻓﺎﻳﻞ ﺷﻤﺎ ﺩﺭ ﺩﺳﺘﺮﺱ ﺍﺳﺖ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﺳﭙﺲ ﮔﺰﻳﻨﻪ Create a list(ﺍﻳﺠﺎﺯ ﻓﻬﺮﺳﺖ ﺟﺪﻳﺪ) ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.

 

 

ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﭘﻨﺠﺮﻩ ﺍﯼ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﺑﺎﺯ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﺷﻤﺎ ﺑﺎﻳﺪ ﻧﺎﻡ ﻭ ﺗﻮﺿﻴﺤﺎﺕ ﻟﻴﺴﺖ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ ﻭ ﺳﭙﺲ ﻋﻤﻮﻣﯽ (Public) ﻳﺎ ﺧﺼﻮﺻﯽ (Private) ﺑﻮﺩﻥ ﺁﻥ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﻧﻬﺎﻳﺖ ﺑﺮ ﺭﻭﯼ Save list ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﻟﻴﺴﺘﯽ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺍﻳﺠﺎﺩ ﺷﻮﺩ.

 

 

ﻧﮑﺘﻪ: ﺍﮔﺮ ﺷﻤﺎ ﻟﻴﺴﺘﯽ ﮐﻪ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﻴﺪ ﺑﻪ ﺻﻮﺭﺕ ﻋﻤﻮﻣﯽ (Public) ﺑﺎﺷﺪ، ﺩﻳﮕﺮ ﮐﺎﺭﺑﺮﺍﻥ ﺗﻮﻳﻴﺘﺮ ﻧﻴﺰ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺍﻳﻦ ﻟﻴﺴﺖ ﺭﺍ ﺩﻧﺒﺎﻝ ﮐﻨﻨﺪ ﺍﻣﺎ ﺩﺭ ﺣﺎﻟﺘﯽ ﮐﻪ ﻟﻴﺴﺖ ﺑﻪ ﺻﻮﺭﺕ ﺧﺼﻮﺻﯽ ﺑﺎﺷﺪ، ﺗﻨﻬﺎ ﺷﻤﺎ ﺍﻣﮑﺎﻥ ﺩﺳﺘﺮﺳﯽ ﻭ ﻣﺸﺎﻫﺪﻩ ﺁﻥ ﻟﻴﺴﺖ ﺭﺍ ﺩﺍﺭﻳﺪ.

 

ﺑﺮﺍﯼ ﭘﺎﮎ ﮐﺮﺩﻥ ﻳﮏ ﻟﻴﺴﺖ ﻧﻴﺰ ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﻭﺍﺭﺩ ﻟﻴﺴﺘﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﭘﺎﮎ ﮐﻨﻴﺪ ﺑﺸﻮﻳﺪ ﻭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﺣﺬﻑ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

 

 

ﺑﺮﺍﯼ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺗﻮﻳﻴﺘﺮ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﻟﻴﺴﺘﯽ ﮐﻪ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺍﻳﻢ ﻧﻴﺰ ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﺍﺳﻢ ﺁﻧﻬﺎ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺟﻌﺒﻪ ﺟﺴﺘﺠﻮ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﻭ ﻳﺎ ﺍﺯ ﻃﺮﻳﻖ ﺻﻔﺤﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺁﻧﻬﺎ ﺭﺍ ﺩﻧﺒﺎﻝ ﻣﯽ ﮐﻨﻴﻢ (Following)،‌ ﺍﻓﺮﺍﺩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﺑﻪ ﻟﻴﺴﺘﻤﺎﻥ ﺍﺿﺎﻓﻪ ﮐﻨﻴﻢ.

 

 

ﺗﺼﻮﻳﺮ ﺑﺎﻻ ﺑﻪ ﺷﻤﺎ ﺭﻭﺵ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﻳﮏ ﮐﺎﺭﺑﺮ ﺭﺍ ﺑﻪ ﻟﻴﺴﺘﯽ ﮐﻪ ﺩﺭ ﺗﻮﻳﻴﺘﺮ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺍﻳﺪ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ. ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﻳﻨﮑﺎﺭ ﺑﺎﻳﺪ ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ ﺑﻪ ﭘﺮﻭﻓﺎﻳﻠﯽ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﻟﻴﺴﺖ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ ﻣﺮﺍﺟﻌﻪ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺁﻧﮑﻪ ﺑﻪ ﺻﻔﺤﻪ ﺩﻧﺒﺎﻝ ﮐﻨﻨﺪﮔﺎﻥ (Following) ﺑﺮﻭﻳﺪ. ﺳﭙﺲ ﺑﺮ ﺭﻭﯼ ﺩﻭ ﮔﺰﻳﻨﻪ ﺍﯼ ﮐﻪ ﺩﺭ ﺑﺎﻻ ﻣﺸﺨﺺ ﺷﺪﻩ ﺍﺳﺖ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

 

ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻓﺰﻭﺩﻥ ﻳﺎ ﺑﺮﺩﺍﺷﺘﻦ ﺍﺯ ﻟﻴﺴﺖ، ﭘﻨﺠﺮﻩ‌ﺍﯼ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺗﻤﺎﻣﯽ ﻟﻴﺴﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺭﺍ ﻧﻤﺎﻳﺶ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺷﻤﺎ ﺑﺎ ﭼﮏ ﺯﺩﻥ ﺩﺭ ﮐﻨﺎﺭ ﻫﺮ ﻳﮏ ﺍﺯ ﻋﻨﻮﺍﻥ‌ﻫﺎﯼ ﻟﻴﺴﺖ‌ﻫﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﮐﺎﺭﺑﺮ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﺭﺍ ﺭﺍ ﺑﻪ ﺁﻥ ﻟﻴﺴﺖ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ.
