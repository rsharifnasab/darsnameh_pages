ﺩﺭﺱ ﺑﻴﺴﺖ ﻭ ﺳﻮﻡ- ﻋﺸﻖ ﻫﺎﯼ ﺧﺎﻧﻮﺍﺩﮔﯽ

ﺩﺭ ﺩﻭﺭﻩ‌ﺍﯼ ﮐﻪ ﻣﺎ ﺍﮐﻨﻮﻥ ﻣﺸﻐﻮﻝ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻥ ﺁﻥ ﻫﺴﺘﻴﻢ، ﻳﻌﻨﯽ ﺩﻭﺭﻩ‌ﺍﯼ ﮐﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻝ ﺩﺍﺭﺩ، ﮐﻮﺩﮎ ﺑﺎﻳﺪ ﻣﺮﺣﻠﻪ­‌ﺍﯼ ﻃﺒﻴﻌﯽ ﺭﺍ ﻃﯽ ﮐﻨﺪ ﮐﻪ ﻣﺎ ﺁﻥ ﺭﺍ «ﻋﺸﻖ‌ﻫﺎﯼ ﺧﺎﻧﻮﺍﺩﮔﯽ» ﻧﺎﻣﻴﺪﻩ‌ﺍﻳﻢ. ﮐﻮﺩﮎ ﺩﺭ ﺣﺪﻭﺩ ﺳﻪ ﺳﺎﻟﮕﯽ ﺑﻪ ﺗﺪﺭﻳﺞ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﺭﻭﺍﺑﻄﯽ ﺩﺍﺭﻧﺪ ﮐﻪ ﻣﺘﻔﺎﻭﺕ ﺍﺯ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺍﻭ ﺩﺍﺭﻧﺪ. ﺗﺎ ﭘﻴﺶ ﺍﺯ ﺁﻥ، ﻋﻘﻞ ﺍﻭ ﺭﺷﺪ ﻻﺯﻡ ﺭﺍ ﻧﮑﺮﺩﻩ ﺑﻮﺩ ﮐﻪ ﺍﻳﻦ ﺗﻔﺎﻭﺕ ﺭﺍ ﺩﺭﮎ ﮐﻨﺪ. ﺍﺣﺴﺎﺳﺎﺕ ﺟﺪﻳﺪﯼ ﺩﺭ ﮐﻮﺩﮎ ﺑﻴﺪﺍﺭ ﻣﯽ‌ﺷﻮﺩ. ﭘﺴﺮﻫﺎ ﻋﺎﺷﻖ ﻣﺎﺩﺭ ﻭ ﺩﺧﺘﺮﻫﺎ ﻋﺎﺷﻖ ﭘﺪﺭﺷﺎﻥ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﻳﻦ ﭘﺪﻳﺪﻩ ﺩﺭ ﻧﺰﺩ ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ ﻭ ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺁﻧﻬﺎ ﺭﺍ ﺟﺪﺍﮔﺎﻧﻪ ﺗﻮﺿﻴﺢ ﺩﻫﻢ.

ﻋﺸﻖ‌ﻫﺎﯼ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺑﺮﺍﯼ ﭘﺴﺮﺍﻥ
ﺍﻭﻝ ﻣﺴﺌﻠﻪ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﭘﺴﺮﻫﺎ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ. ﻣﺎﺩﺭ، ﺩﺭ ﺗﻤﺎﻡ ﺩﻭﺭﺍﻥ ﺯﻧﺪﮔﯽ ﮐﻮﺗﺎﻩ ﮐﻮﺩﮎ ﻣﺮﮐﺰ ﺛﻘﻞ ﺟﻬﺎﻥ ﺑﻮﺩﻩ ﺍﺳﺖ. ﺍﻭ ﮐﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﺧﻮﺩ ﺭﺍ ﻧﻮﺯﺍﺩﯼ ﻭﺍﺑﺴﺘﻪ ﺑﻪ ﺍﻭ ﺣﺲ ﻣﯽ‌ﮐﺮﺩ، ﺍﮐﻨﻮﻥ ﺍﺣﺴﺎﺳﯽ ﺩﻳﮕﺮ ﻧﺴﺒﺖ ﺑﻪ ﺍﻭ ﺩﺍﺭﺩ. ﺣﻘﻴﻘﺖ ﺟﺪﻳﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﻭ ﻋﺎﺷﻖ ﻣﺎﺩﺭﺵ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻋﺎﻃﻔﻪ ﺧﺎﺻﯽ ﻧﺴﺒﺖ ﺑﻪ ﺍﻭ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ. ﺍﻳﻦ ﻭﺍﻗﻌﻴﺘﯽ ﺍﺳﺖ ﻃﺒﻴﻌﯽ، ﻭ ﺗﻤﺎﻡ ﭘﺴﺮﻫﺎﯼ ﺧﺮﺩﺳﺎﻝ ﺁﻥ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ. ﺑﻌﻀﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺁﻥ ﺭﺍ ﭘﻨﻬﺎﻥ ﮐﻨﻨﺪ ﻭ ﺑﻌﻀﯽ ﺩﻳﮕﺮ، ﺧﺼﻮﺻﺎ ﺍﮔﺮ ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ‌ﺍﯼ ﺑﺎﺷﻨﺪ ﮐﻪ ﺑﺘﻮﺍﻧﻨﺪ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺑﻪ ﺁﺯﺍﺩﯼ ﺑﻴﺎﻥ ﮐﻨﻨﺪ، ﺁﻥ ﺭﺍ ﺑﻪ ﺯﺑﺎﻥ ﻣﯽ‌ﺁﻭﺭﻧﺪ.

ﭘﺴﺮ ﮐﻮﭼﮏ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﮐﻪ ﻣﺎﺩﺭﺵ ﻓﻘﻂ ﻣﺎﻝ ﺍﻭ ﺑﺎﺷﺪ ﻭ ﮐﻢ‌ﮐﻢ ﺍﺣﺴﺎﺱ ﺿﺪﻳﺘﯽ ﺑﺮ ﻋﻠﻴﻪ ﭘﺪﺭﺵ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ. ﺯﻳﺮﺍ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﮐﻪ ﻣﺎﺩﺭﺵ ﻓﻘﻂ ﻣﺎﻝ ﺍﻭ ﺑﺎﺷﺪ، ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻣﺠﺒﻮﺭ ﺍﺳﺖ ﺍﻭ ﺭﺍ ﺑﺎ ﭘﺪﺭﺵ ﺗﻘﺴﻴﻢ ﮐﻨﺪ. ﻣﺴﺌﻠﻪ ﻧﺎﺭﺍﺣﺖ‌ﮐﻨﻨﺪﻩ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ«ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ»، ﺧﺎﻧﻮﺍﺩﻩ ﺳﻪ ﻧﻔﺮﻩ ﺍﺳﺖ. ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﻧﻪ ﺗﻨﻬﺎ ﻋﺸﻖ ﻭ ﻣﺤﺒﺘﯽ ﻋﻤﻴﻖ ﻧﺴﺒﺖ ﺑﻪ ﻣﺎﺩﺭﺵ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ، ﺑﻠﮑﻪ ﺣﺴﺎﺩﺕ ﻭ ﺿﺪﻳﺘﯽ ﻧﺴﺒﺖ ﺑﻪ ﭘﺪﺭﺵ ﺩﺭ ﺍﻭ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻳﺪ. ﺍﮔﺮ ﭘﺪﺭ ﻣﺘﻮﺟﻪ ﺍﻳﻦ ﺍﻣﺮ ﻧﺒﺎﺷﺪ، ﻭ ﻣﺘﺄﺳﻔﺎﻧﻪ ﺍﻳﻦ ﻭﺿﻌﯽ
 ﺍﺳﺖ ﮐﻪ ﺍﮐﺜﺮ ﺍﻭﻗﺎﺕ ﭘﻴﺶ ﻣﯽ­‌ﺁﻳﺪ، ﺍﻳﻦ ﻭﺟﻪ ﺍﺯ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺵ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻭ ﺭﺍ ﺍﺯ ﮐﻮﺭﻩ ﺑﻪ ﺩﺭ ﺑﺮﺩ.

ﻣﻌﻤﻮﻻ ﻋﮑﺲ‌­ﺍﻟﻌﻤﻞ ﺩﻓﺎﻋﯽ ﻣﺎ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺳﺖ ﮐﻪ، ﺁﻥ ﺍﺣﺴﺎﺳﺎﺗﯽ ﺭﺍ ﮐﻪ ﺧﻮﺩ ﻧﺴﺒﺖ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺩﺍﺭﻳﻢ، ﺑﻪ ﺁﻧﻬﺎ ﻧﺴﺒﺖ ﻣﯽ‌ﺩﻫﻴﻢ. ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﻧﻴﺰ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﭘﺪﺭﺵ ﻧﺴﺒﺖ ﺑﻪ ﺍﻭ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺭﻗﺎﺑﺖ ﻭ ﺿﺪﻳﺖ ﺭﺍ ﺩﺍﺭﺩ. ﺍﻭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭘﺪﺭﺵ ﻋﻠﻴﻪ ﺍﻭﺳﺖ. ﻭ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﭘﺪﺭﺵ ﺧﻴﻠﯽ ﺍﺯ ﺍﻭ ﺑﺰﺭﮒ‌ﺗﺮ ﻭ ﻗﻮﯼ‌ﺗﺮ ﺍﺳﺖ، ﺗﺮﺱ ﺍﺯ ﺍﻧﺘﻘﺎﻡ ﺩﺍﺭﺩ، ﻣﯽ‌­ﺗﺮﺳﺪ ﮐﻪ ﻣﺒﺎﺩﺍ ﭘﺪﺭﺵ ﺍﻭ ﺭﺍ ﺗﻨﺒﻴﻪ ﮐﻨﺪ. ﻋﻠﺖ ﮐﺎﺑﻮﺱ‌­ﻫﺎﻳﯽ ﮐﻪ ﮐﻮﺩﮐﺎﻥ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﻪ ﻣﯽ‌ﺑﻴﻨﻨﺪ ﻧﻴﺰ ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺍﺳﺖ. ﺍﻳﻦ ﺗﺮﺱِ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﺑﻪ ﺻﻮﺭﺕ ﮐﺎﺑﻮﺱ‌ﻫﺎﻳﯽ ﻇﺎﻫﺮ ﻣﯽ‌­ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﭘﺪﺭﺵ ﺍﻭ ﺭﺍ ﮐﺘﮏ ﻣﯽ‌ﺯﻧﺪ ﻭ ﻳﺎ ﺑﺒﺮ، ﺷﻴﺮ ﻭ ﻳﺎ ﻫﻴﻮﻻﺋﯽ ﺍﻭ ﺭﺍ ﺩﺭ ﺧﻮﺍﺏ ﺷﮑﺎﺭ ﻣﯽ‌­ﮐﻨﺪ.

ﺍﮔﺮ ﻣﺤﻴﻂ ﺧﺎﻧﻮﺍﺩﮔﯽ ﮐﻮﺩﮎ ﺳﺎﻟﻢ ﺑﺎﺷﺪ، ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﮐﻢ‌­ﮐﻢ ﺍﻳﻦ ﻋﺸﻖ ﺑﺮﺍﯼ ﻣﺎﺩﺭﺵ ﻭ ﺁﺭﺯﻭﯼ ﺟﺎﻧﺸﻴﻦِ ﭘﺪﺭ ﺷﺪﻥ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻣﯽ­‌ﮐﻨﺪ ﻭ ﺑﺎﻻﺧﺮﻩ ﻣﯽ‌ﻓﻬﻤﺪ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺟﺎﯼ ﭘﺪﺭﺵ ﺭﺍ ﺑﮕﻴﺮﺩ ﻭ ﺗﺼﻤﻴﻢ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﺷﺒﻴﻪ ﺍﻭ ﺷﻮﺩ.

ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺭﻭﻧﺪ ﻳﮑﺴﺎﻥ ﺳﺎﺯﯼ ﺧﻮﺩ ﺑﺎ ﭘﺪﺭﺵ ﺭﺍ ﺷﺮﻭﻉ ﻣﯽ­‌ﮐﻨﺪ. ﭘﺪﺭ ﺑﺮﺍﻳﺶ ﻣﺪﻝ ﻣﯽ­‌ﺷﻮﺩ ﻭ ﺑﻪ ﺗﻤﺎﻡ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﻤﮑﻦ ﺍﺯ ﺍﻭ ﺗﻘﻠﻴﺪ ﻣﯽ­‌ﮐﻨﺪ. ﺍﻳﻦ ﺩﻭ ﺭﻭﻧﺪ (ﺭﻫﺎ ﮐﺮﺩﻥ ﺗﺪﺭﻳﺠﯽ ﻋﺸﻖ ﺭﻭﻳﺎﻳﯽ ﺑﺮﺍﯼ ﻣﺎﺩﺭﺵ ﻭ ﺭﻫﺎ ﮐﺮﺩﻥ ﺗﺪﺭﻳﺠﯽ ﺭﻗﺎﺑﺖ ﺑﺎ ﭘﺪﺭ ﻭ ﺟﺎﻧﺸﻴﻦ ﮐﺮﺩﻥ ﺍﻭ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺪﻝ) ﺗﻤﺎﻣﯽ ﺩﻭﺭﺍﻥ ﺳﻪ ﺳﺎﻟﻪ ﺭﺷﺪ ﺍﻭ ﺭﺍ ﺩﺭ ﺑﺮﻣﯽ‌­ﮔﻴﺮﺩ.

ﺍﮔﺮ ﭘﺴﺮﯼ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﺩﺍﺭﻳﺪ ﻭ ﻃﻼﻕ ﻣﯽ­‌ﮔﻴﺮﻳﺪ، ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻭ ﺧﻮﺩ ﺭﺍ ﻣﺴﺌﻮﻝ ﺍﻳﻦ ﺟﺪﺍﻳﯽ ﻧﺪﺍﻧﺪ. ﺍﺯ ﻓﻦ ﺍﻧﻌﮑﺎﺱ ﺍﺣﺴﺎﺳﺎﺕ ﮐﻪ ﻗﺒﻼ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﺗﺎ ﺍﻭ ﺑﺘﻮﺍﻧﺪ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ، ﻫﺮ ﻗﺪﺭ ﻫﻢ ﮐﻪ ﻏﻴﺮﻣﻨﻄﻘﯽ ﺑﺎﺷﺪ، ﺑﻴﺎﻥ ﮐﻨﺪ. ﺳﭙﺲ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﺗﻮﺿﻴﺢ ﺩﻫﻴﺪ ﮐﻪ ﺍﻭ ﻫﻴﭻ ﻣﺴﺌﻮﻟﻴﺘﯽ ﺩﺭ ﺍﻳﻦ ﺟﺪﺍﻳﯽ ﻧﺪﺍﺭﺩ.

ﻋﺸﻖ‌ﻫﺎﯼ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺑﺮﺍﯼ ﺩﺧﺘﺮﺍﻥ
ﺣﺎﻻ ﻣﺴﺌﻠﻪ ﺭﺍ ﺑﺮﺍﯼ ﺩﺧﺘﺮ ﮐﻮﭼﮏ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ. ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺑﺮﺍﯼ ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﮐﻤﯽ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻭﻟﻴﻦ ﻋﺸﻖ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﻣﺎﺩﺭﺵ ﺍﺳﺖ، ﺗﺼﻮﻳﺮﯼ ﺍﺯ ﻣﻬﺮ ﻭ ﻣﺤﺒﺖ ﮐﻪ ﺍﻭ ﮐﻪ ﺩﺭ ﺗﻤﺎﻡ ﺩﻭﺭﺍﻥ ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺣﻔﻆ ﻣﯽ‌ﻛﻨﺪ. ﺩﺧﺘﺮ ﻧﻴﺰ ﺍﻭﻝ ﺑﺎ ﻫﻤﻴﻦ ﻋﺸﻖ ﺁﺷﻨﺎ ﻣﯽ‌ﺷﻮﺩ، ﻋﺸﻖ ﺑﺮﺍ‌ﯼ ﻣﺎﺩﺭﺵ. ﻭﻟﯽ ﺑﺮﺧﻼﻑ ﭘﺴﺮ، ﺍﻭ ﺑﺎﻳﺪ ﺗﺼﻮﻳﺮ ﻋﺸﻘﺶ ﺭﺍ ﺑﻪ ﭘﺪﺭﺵ ﻣﻨﺘﻘﻞ ﮐﻨﺪ. ﺍﻳﻦ ﻛﺎﺭ ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺭﺍ ﺑﺮﺍﯼ ﺍﻭ ﭘﻴﭽﻴﺪﻩ‌ﺗﺮ ﻣﯽ‌ﻛﻨﺪ.

ﻭﻗﺘﯽ ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﺷﺮﻭﻉ ﺑﻪ ﻛﺴﺐ ﺍﺳﺘﻘﻼﻝ ﻣﯽ‌ﻛﻨﺪ ﻭ ﺑﻪ ﺗﻔﺎﻭﺕ ﺑﻴﻦ ﺧﻮﺩ ﻭ ﻣﺎﺩﺭﺵ ﭘﯽ ﻣﯽ‌ﺑﺮﺩ، ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ ﻣﺘﻮﺟﻪ ﻭﺟﻮﺩ ﻣﻬﺮ ﻭ ﻣﺤﺒﺘﯽ ﺍﺯ ﻧﻮﻉ ﺩﻳﮕﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻋﺎﺷﻖ ﭘﺪﺭﺵ ﻣﯽ‌ﺷﻮﺩ. ﺑﺮﺧﯽ ﺩﺧﺘﺮﻫﺎ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﭘﺴﺮﻫﺎ، ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﭘﻨﻬﺎﻥ ﻣﯽ‌ﻛﻨﻨﺪ ﻭﻟﯽ ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﺁﻧﻬﺎ ﺭﺍ ﺁﺯﺍﺩﺍﻧﻪ ﺑﻴﺎﻥ ﻣﯽ‌ﻛﻨﻨﺪ.

ﺩﺧﺘﺮﺍﻥ ﺧﺮﺩﺳﺎﻝ، ﺩﺭ ﺩﻭﺭﺍﻥ ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ، ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺧﻴﻠﯽ ﺯﻳﺮﻛﺎﻧﻪ‌ﺗﺮ ﺍﺯ ﭘﺴﺮﻫﺎ ﺑﻪ ﺍﻫﺪﺍﻓﺸﺎﻥ ﺑﺮﺳﻨﺪ. ﺍﻳﻦ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻧﺎﺷﯽ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺯﻥ‌ﻫﺎ ﺩﺭ ﺭﻭﺍﺑﻂ ﺍﻧﺴﺎﻧﯽ ﻇﺮﺍﻓﺖ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﻧﺪ. ﺩﺭ ﻭﺍﻗﻊ ﻣﻤﻜﻦ ﺍﺳﺖ ﻛﻪ ﻣﺎﺩﺭ ﺣﺘﯽ ﻣﺘﻮﺟﻪ ﻣﺒﺎﺭﺯﻩ ﺩﺭﻭﻧﯽ ﺩﺧﺘﺮﺵ ﺑﺮﺍﯼ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻥ ﻋﺸﻖ ﭘﺪﺭﺵ ﻧﺸﻮﺩ.  ﺯﻣﺎﻧﯽ ﻛﻪ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺁﺷﭙﺰﯼ ﻳﺎﺩ ﺑﮕﻴﺮﺩ ﻭ ﻳﺎ ﺧﺎﻧﻪ ﺭﺍ ﺗﻤﻴﺰ ﻛﻨﺪ ﺷﻤﺎ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻴﺪ ﮐﻪ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺍﺯ ﺷﻤﺎ ﺗﻘﻠﻴﺪ ﻛﻨﺪ، ﺩﺭ ﺻﻮﺭﺗﯽ ﻛﻪ ﺩﺭﻭﻥ ﻭﺟﻮﺩﺵ، ﺍﻭ ﺍﻳﻦ ﻛﺎﺭﻫﺎ ﺭﺍ ﻣﯽ‌ﻛﻨﺪ ﺗﺎ ﺑﻪ ﭘﺪﺭﺵ ﻧﺸﺎﻥ ﺩﻫﺪ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﺍﻭ ﻫﻤﺴﺮﯼ ﺑﻬﺘﺮ ﺍﺯ ﺷﻤﺎ ﺑﺎﺷﺪ.

ﺩﺧﺘﺮﻫﺎﯼ ﻛﻮﭼﻚ ﺍﺯ ﺍﻳﻦ ﻧﻈﺮ ﻛﻪ ﻭﻗﺖ ﻛﻤﺘﺮﯼ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺎ ﺍﻳﻦ ﭘﺪﺭ ﻣﺤﺒﻮﺏ ﺑﮕﺬﺭﺍﻧﻨﺪ ﻧﻴﺰ ﺑﺎ ﭘﺴﺮﻫﺎ ﺗﻔﺎﻭﺕ ﺩﺍﺭﻧﺪ. ﺩﺭ ﺟﺎﻣﻌﻪ ﻣﺎ، ﭘﺪﺭﻫﺎ ﺍﻛﺜﺮﺍ ﺧﺎﺭﺝ ﺍﺯ ﺧﺎﻧﻪ ﺑﻪ ﺳﺮ ﻣﯽ‌ﺑﺮﻧﺪ. ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﻧﺴﺒﺖ ﺑﻪ ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﻭﻗﺖ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺑﺎ ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ‌ﺍﺵ ﻣﯽ‌ﮔﺬﺭﺍﻧﺪ. ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﺑﻴﺸﺘﺮ ﺍﻳﻦ ﻋﺸﻖ ﺭﺍ ﺩﺭ ﺗﺼﻮﺭﺍﺗﺶ ﺯﻧﺪﮔﯽ ﻛﻨﺪ
.
ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﻧﺴﺒﺖ ﺑﻪ ﻣﺎﺩﺭﺵ ﺍﺣﺴﺎﺱ ﺣﺴﺎﺩﺕ ﻭ ﺿﺪﻳﺖ ﭘﻴﺪﺍ ﻣﯽ‌ﻛﻨﺪ ﻭ ﺍﻭ ﻧﻴﺰ ﺑﻪ ﻣﺎﻧﻨﺪ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﺩﭼﺎﺭ ﻣﺸﻜﻼﺗﯽ ﻣﯽ‌ﺷﻮﺩ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻛﻪ ﻣﺎﺩﺭﺵ ﻫﻤﻴﺸﻪ ﺩﺭ ﺣﺎﻝ ﺭﺳﻴﺪﮔﯽ ﺑﻪ ﺍﻭﺳﺖ ، ﻣﺎﺩﺭﺵ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﺍﻭ ﻭﺍﺑﺴﺘﻪ ﺍﺳﺖ. ﻓﻜﺮ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﺍﮔﺮ ﻣﺎﺩﺭﺵ ﺑﺮﻭﺩ ﻭ ﺩﻳﮕﺮ ﺑﺮﻧﮕﺮﺩﺩ ﺧﻴﻠﯽ ﻭﺣﺸﺘﻨﺎﻙ ﺍﺳﺖ ﻭ ﻫﻤﺎﻥ ﻃﻮﺭ ﻛﻪ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﺍﺣﺴﺎﺱ ﺿﺪﻳﺖ ﻭ ﺭﻗﺎﺑﺘﺶ ﺭﺍ ﺩﺭ ﭘﺪﺭﺵ ﻣﯽ‌ﺩﻳﺪ، ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﺩﺭ ﻣﺎﺩﺭﺵ ﻣﯽ‌ﺑﻴﻨﺪ. ﺍﻭ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﻣﺎﺩﺭﺵ ﻣﯽ‌ﺩﺍﻧﺪ ﻛﻪ ﻭﯼ ﺿﺪ ﺍﻭﺳﺖ ﻭ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻛﻪ ﺍﺯ ﺩﺳﺖ ﺍﻭ- ﻣﺎﺩﺭﺵ-  ﺧﻼﺹ ﺷﻮﺩ.

ﺩﺭ ﺗﻤﺎﻡ ﺍﻳﻦ ﺩﻭﺭﺍﻥ، ﺧﻮﺍﺏ‌ﻫﺎﯼ ﺍﻭ ﺑﺎ ﺩﻳﺪﻥ ﻛﺎﺑﻮﺱ ﭘﺮﻳﺸﺎﻥ ﻣﯽ‌ﺷﻮﺩ. ﺍﻭ ﻣﻌﻤﻮﻻ ﺩﺭ ﻛﺎﺑﻮﺱ‌ﻫﺎﻳﺶ ﺟﺎﺩﻭﮔﺮﯼ ﻭﺣﺸﺘﻨﺎﻙ ﻳﺎ ﻏﻮﻟﯽ ﺑﺪﺟﻨﺲ ﺭﺍ ﻣﯽ‌ﺑﻴﻨﺪ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺍﻭ ﺭﺍ ﺗﻨﺒﻴﻪ ﻛﻨﺪ (ﺍﻳﻦ ﺟﺎﺩﻭﮔﺮ ﻭ ﻏﻮﻝ ﺩﺭ ﻭﺍﻗﻊ ﺗﺼﻮﻳﺮﯼ ﺍﺳﺖ ﻛﻪ ﺍﻭ ﺑﻪ ﻃﻮﺭ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﺯ ﻣﺎﺩﺭﺵ ﺩﺍﺭﺩ.)

ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ‌ﻫﺎﯼ ﻃﺒﻴﻌﯽ، ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻝ ﻧﻴﺰ ﺩﺭ ﻣﯽ‌ﻳﺎﺑﺪ ﻛﻪ ﭘﺪﺭﺵ ﻣﺘﻌﻠﻖ ﺑﻪ ﻣﺎﺩﺭﺵ ﺍﺳﺖ ﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺷﻮﻫﺮ ﺍﻭ ﺷﻮﺩ. ﻛﺸﻒ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺩﺭ ﺗﻤﺎﻡ ﻣﺪﺕ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ ﻛﻮﺩﻙ ﺑﻪ ﻃﻮﻝ ﻣﯽ‌ﺍﻧﺠﺎﻣﺪ. ﺍﻭ ﺑﻪ ﺗﺪﺭﻳﺞ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﻋﺎﺷﻘﺎﻧﻪ‌ﺍﯼ ﻛﻪ ﻧﺴﺒﺖ ﺑﻪ ﭘﺪﺭﺵ ﺩﺍﺭﺩ، ‌ﺩﺳﺖ ﻣﯽ‌ﻛﺸﺪ ﻭ ﺟﺎﯼ ﺁﻥ ﺭﺍ ﺗﺼﻮﻳﺮﯼ ﺍﺯ ﻣﺮﺩﯼ ﻛﻪ ﺩﺭ ﺁﻳﻨﺪﻩ ﺩﻭﺳﺖ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ ﻭ ﺑﺎ ﺍﻭ ﺍﺯﺩﻭﺍﺝ ﺧﻮﺍﻫﺪ ﻛﺮﺩ، ﺧﻮﺍﻫﺪ ﮔﺮﻓﺖ.

ﺩﺭ ﺍﻳﻦ ﺑﺤﺮﺍﻥ، ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﮑﺘﻪ ﻛﻴﻔﻴﺖ ﺭﻭﺍﺑﻂ ﺷﻤﺎ ﻭ ﺷﻮﻫﺮﺗﺎﻥ ﺍﺳﺖ. ﺍﮔﺮ ﺩﺭ ﺭﺍﺑﻄﻪ ﺯﻧﺎﺷﻮﻳﻲ ﺷﻤﺎ ﻋﺸﻖ ﺣﻜﻤﻔﺮﻣﺎ ﺑﺎﺷﺪ ﻭ ﺯﻧﺪﮔﯽ ﺑﺎﺛﺒﺎﺗﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺑﻪ ﺗﺪﺭﻳﺞ ﻣﯽ‌ﻓﻬﻤﻨﺪ ﻛﻪ ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺁﻧﻬﺎ ﻏﻴﺮﻣﻤﻜﻦ ﺍﺳﺖ ﻭ ﺭﺍﻩ‌ﺣﻠﯽ ﺑﺮﺍﯼ ﺁﻥ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﻨﺪ. ﻭﻟﯽ ﺍﮔﺮ ﺯﻧﺪﮔﯽ ﺯﻧﺎﺷﻮﻳﯽ‌ﺗﺎﻥ ﺗﻮﺍﻡ ﺑﺎ ﻋﺪﻡ‌ﺗﻮﺍﻓﻖ‌ﻫﺎﯼ ﻋﻤﻴﻖ ﺑﺎﺷﺪ، ﺣﻞ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﻣﺸﻜﻞ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺍﮔﺮ ﺩﺭ ﺣﺎﻝ ﻣﺘﺎﺭﻛﻪ ﺑﺎﺷﻴﺪ، ﺑﻬﺘﺮﻳﻦ ﻛﻤﻚ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﺮﺍﺟﻌﻪ ﺑﻪ ﻳﻚ ﻣﺸﺎﻭﺭ ﺍﻣﻮﺭ ﺯﻧﺎﻧﺸﻮﻳﯽ ﺍﺳﺖ. ﺍﻳﻦ ﭼﻴﺰﯼ ﺭﺍ ﻛﻪ ﻣﻴﮕﻮﻳﻢ ﺟﺪﯼ ﺑﮕﻴﺮﻳﺪ.

ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﺴﻠﻤﺎ ﺳﻌﯽ ﻣﯽ‌ﻛﻨﺪ ﺑﻴﻦ ﺷﻤﺎ ﺩﻋﻮﺍ ﺑﻴﻨﺪﺍﺯﺩ ﺗﺎ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﺪ. ﺑﻪ ﺻﻮﺭﺗﯽ ﺑﭽﻪ‌ﮔﺎﻧﻪ، ﺳﻌﯽ ﻣﯽ‌ﻛﻨﺪ ﺳﺪﻫﺎﻳﯽ ﺑﻴﻦ ﺷﻤﺎ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﺩ. ﺍﮔﺮ ﻣﺴﺎﺋﻞ ﻋﻤﺪﻩ‌ﺍﯼ ﺑﻴﻦ ﺷﻤﺎ ﻭ ﺷﻮﻫﺮﺗﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ،‌ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﻣﻤﻜﻦ ﺍﺳﺖ ﻣﻮﻓﻖ ﺷﻮﺩ ﻣﺎﺩﺭﺵ ﺭﺍ ﻭﺍ ﺩﺍﺭﺩ ﺗﺎ ﺑﺎ ﺍﻭ ﺑﻪ ﻣﺜﺎﺑﻪ «ﻳﻚ ﻋﺎﺷﻖ ﻛﻮﭼﻮﻟﻮ»، ﺗﺎ ﻳﻚ ﺑﭽﻪ، ﺑﺮﺧﻮﺭﺩ ﻛﻨﺪ. ﺍﮔﺮ ﻣﺎﺩﺭﯼ ﺍﺯ ﺗﻮﺟﻪ ﻓﺮﺍﻭﺍﻧﯽ ﻛﻪ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻟﺶ ﺑﻪ ﺍﻭ ﻣﻴﻜﻨﺪ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﺧﺸﻨﻮﺩ ﺷﻮﺩ، ﭼﺮﺍ ﻛﻪ ﺷﻮﻫﺮﺵ ﺑﻪ ﺍﻭ ﺗﻮﺟﻪ ﺯﻳﺎﺩﯼ ﻧﻤﯽ‌ﻛﻨﺪ، ﺍﻳﻦ ﺑﺮﺍﯼ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ ﺯﻳﺎﺩ ﺳﺎﻟﻢ ﻧﻴﺴﺖ. ﻣﺎﺩﺭ ﺑﻪ ﻋﻮﺽ ﻣﺴﺘﺤﻜﻢ‌ﺗﺮ ﻛﺮﺩﻥ ﺗﺼﻮﻳﺮ ﭘﺪﺭ، ﺍﻭ ﺭﺍ ﻧﺰﺩ ﻓﺮﺯﻧﺪﺵ ﺗﺤﻘﻴﺮ ﻣﯽ‌ﻛﻨﺪ. ﻣﺴﻠﻤﺎ ﺩﺧﺘﺮﺍﻥ ﺧﺮﺩﺳﺎﻝ ﻧﻴﺰ ﺍﺯ ﻫﻤﻴﻦ ﺳﻴﺎﺳﺖ «ﺗﻔﺮﻗﻪ ﺑﻴﻨﺪﺍﺯ ﻭ ﺣﻜﻮﻣﺖ ﻛﻦ» ﺍﺳﺘﻔﺎﺩﻩ ﻛﺮﺩﻩ ﻭ ﺳﻌﯽ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ ﺑﻴﻦ ﻣﺎﺩﺭ ﻭ ﭘﺪﺭﺷﺎﻥ ﺍﺧﺘﻼﻑ ﺑﻴﻨﺪﺍﺯﻧﺪ.

ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺎﻳﺪ ﻣﺮﺍﻗﺐ ﺑﺎﺷﻨﺪ ﻛﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺍﺯ ﺍﻳﻦ ﻧﻘﻄﻪ‌ ﺿﻌﻒ ﺑﺮﺍﯼ «ﺗﻔﺮﻗﻪ ﺍﻧﺪﺍﺧﺘﻦ ﻭ ﺣﻜﻮﻣﺖ ﻛﺮﺩﻥ» ﺍﺳﺘﻔﺎﺩﻩ ﻧﻜﻨﻨﺪ. ﺁﻧﻬﺎ ﺑﺎﻳﺪ ﺩﺭ ﻣﻘﺎﺑﻞ ﻭﺳﻮﺳﻪ ﺟﻮﺍﺑﮕﻮﻳﯽ ﺑﻪ ﺩﻟﺒﺮﯼ ﺭﻭﻣﺎﻧﺘﻴﻚ ﻛﻮﺩﻙ ﻣﻘﺎﻭﻣﺖ ﻛﻨﻨﺪ. ﺑﺎﻳﺪ ﻛﻮﺩﻙ ﺭﺍ ﺑﺎ «ﻣﻬﺮﺑﺎﻧﯽ ﻃﺮﺩ» ﻛﻨﻨﺪ ﻭ ﺍﻳﻦ ﺭﻭﺍﺑﻂ ﺭﻭﻣﺎﻧﺘﻴﻜﯽ ﻛﻪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺑﻪ ﺁﻧﻬﺎ ﻋﺮﺿﻪ ﻣﯽ‌ﻛﻨﺪ، ﺭﺩ ﻛﻨﻨﺪ. ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺑﻪ ﭘﺴﺮﺵ ﺑﻔﻬﻤﺎﻧﺪ ﻛﻪ ﭘﺪﺭﺵ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﺩ، ﻛﻪ ﻫﻤﺴﺮ ﭘﺪﺭﺵ ﺍﺳﺖ ﻭ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺍﻭ ﻧﻴﺰ ﭘﺪﺭﺵ ﺭﺍ ﺩﻭﺳﺖ ﺑﺪﺍﺭﺩ، ﺑﺎﻳﺪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﺪ ﻛﻪ ﻭﻗﺘﯽ ﺑﺰﺭﮒ ﺷﺪ، ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺯﻥ ﺍﻭ ﺑﺸﻮﺩ، ﭼﻮﻥ ﺯﻥ ﭘﺪﺭﺵ ﺍﺳﺖ ﻭ ﺧﻴﻠﯽ ﻫﻢ ﺍﺯ ﺍﻳﻦ ﺑﺎﺑﺖ ﺧﻮﺷﺤﺎﻝ ﻭ ﺧﻮﺷﺒﺨﺖ ﺍﺳﺖ. ﺍﻭ ﻧﻴﺰ ﺭﻭﺯﯼ ﺩﺧﺘﺮﯼ ﺭﺍ ﭘﻴﺪﺍ ﺧﻮﺍﻫﺪ ﻛﺮﺩ ﻭ ﺑﺎ ﺍﻭ ﺍﺯﺩﻭﺍﺝ ﺧﻮﺍﻫﺪ ﻛﺮﺩ. ﻭﻟﻲ ﻓﻌﻼ ﭘﺴﺮ ﻛﻮﭼﻮﻟﻮﯼ ﺑﺎﺑﺎ ﺍﺳﺖ.

ﭘﺪﺭ ﻧﻴﺰ ﺑﺎﻳﺪ ﺧﻴﻠﯽ ﺭﻭﺷﻦ‌ ﺑﻪ ﺩﺧﺘﺮ ﺧﺮﺩﺳﺎﻟﺶ ﺗﻮﺿﻴﺢ ﺩﻫﺪ ﻛﻪ ﺍﻭ ﺭﺍ ﺧﻴﻠﻲ ﺩﻭﺳﺖ ﺩﺍﺭﺩ، ﻭﻟﯽ ﻣﺎﻣﺎﻥ ﺯﻥ ﺍﻭ ﺍﺳﺖ. ﺍﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻭﻗﺘﯽ ﺑﺰﺭﮒ ﺷﺪ، ﺯﻥ ﺍﻭ ﺑﺸﻮﺩ. ﺍﻭ ﻗﺒﻼ ﺑﺎ ﻣﺎﺩﺭﺵ ﺍﺯﺩﻭﺍﺝ ﻛﺮﺩﻩ ﻭ ﺧﻴﻠﯽ ﻫﻢ ﺧﻮﺷﺒﺨﺖ ﺍﺳﺖ. ﺍﻭ ﺩﺧﺘﺮ ﻛﻮﭼﻮﻟﻮﯼ ﺑﺎﺑﺎﻳﺶ ﺍﺳﺖ ﻭ ﻫﻴﭻ ﻛﺲ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺟﺎﯼ ﺍﻭ ﺭﺍ ﺩﺭ ﭘﻴﺶ ﭘﺪﺭﺵ ﺑﮕﻴﺮﺩ. ﺍﻭ ﻫﻢ ﺷﻮﻫﺮﯼ ﭘﻴﺪﺍ ﺧﻮﺍﻫﺪ ﻛﺮﺩ ﻭ ﺑﺎ ﺍﻭ ﺍﺯﺩﻭﺍﺝ ﺧﻮﺍﻫﺪ ﻛﺮﺩ.

ﺣﺴﺎﺳﻴﺖ ﺑﻪ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎﯼ ﺫﻫﻨﯽ
ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ، ﺍﻭ ﺩﻭﺭﺍﻧﯽ ﺭﺍ ﻃﯽ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﻧﺴﺒﺖ ﺑﻪ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎﯼ ﺫﻫﻨﯽ ﺣﺴﺎﺳﻴﺖ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺭﺩ، ﺑﺎ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎﯼ ﻭﻳﮋﻩ، ﺍﻭ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎ ﻭ ﻗﻮﻩ ﺗﻔﻜﺮﯼ ﺑﻨﻴﺎﺩﻳﻨﯽ ﮐﺴﺐ ﻣﯽ­‌ﮐﻨﺪ ﻛﻪ ﺗﺎ ﭘﺎﻳﺎﻥ ﻋﻤﺮﺵ ﺍﺯ ﺁﻥ­ﻫﺎ ﺑﻬﺮﻩ ﺧﻮﺍﻫﺪ ﺑﺮﺩ. ﻫﻮﺵ ﻳﻚ ﻛﻮﺩﻙ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﺠﻤﻮﻉ ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﻳﯽ ﻛﻪ ﺑﺎﻳﺪ ﻓﺮﺍﮔﻴﺮﺩ ﺗﻌﺮﻳﻒ ﻛﺮﺩ. ﻫﺮ ﺑﺎﺭ ﻛﻪ ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﻫﻴﺪ، ﻫﻮﺵ ﺍﻭ ﺭﺍ ﺯﻳﺎﺩﺗﺮ ﻛﺮﺩﻩ‌ﺍﻳﺪ.

ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﻄﻤﺌﻦ ﺷﺪ ﻛﻪ ﺍﻭ ﺑﻪ ﻃﺮﺯ ﻣﻄﻠﻮﺑﯽ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎﯼ ﺫﻫﻨﯽ ﻭ ﻋﺎﻃﻔﻲ ﻻﺯﻡ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﻣﯽ‌ﻛﻨﺪ؟ ﻳﻜﯽ ﺍﺯ ﻣﻄﻤﺌﻦ‌ﺗﺮﻳﻦ ﺭﻭﺵ‌ﻫﺎ ﻓﺮﺳﺘﺎﺩﻥ ﺍﻭ ﺑﻪ ﻳﻚ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺧﻮﺏ ﺍﺳﺖ.

ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﻣﻨﻈﻮﺭ  ﺍﺯ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺭﺍ ﺷﺮﺡ ﺩﻫﻴﻢ. ﺑﻌﻀﯽ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﻣﻬﺪﻛﻮﺩﻙ ﻭ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺭﺍ ﻳﻜﯽ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﻣﻬﺪﻛﻮﺩﻙ ﺟﺎﻳﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﻧﮕﻬﺪﺍﺭﯼ ﺍﻃﻔﺎﻝ ﺗﺎ ﭘﻨﺞ ﺳﺎﻝ ﻛﻪ ﺍﺯ ﺳﺎﻋﺖ ﺷﺶ ﻭ ﻧﻴﻢ ﻳﺎ ﻫﻔﺖ ﺻﺒﺢ ﺗﺎ ﭼﻬﺎﺭ ﺑﻌﺪ ﺍﺯ ﻇﻬﺮ ﺑﺎﺯ ﺍﺳﺖ ﻭ ﻣﺎﺩﺭﺍﻧﯽ ﻛﻪ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﻨﺪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺭﺍ ﺑﻪ ﺁﻧﺠﺎ ﻣﯽ‌ﺳﭙﺎﺭﻧﺪ. ﺑﻌﻀﯽ ﺍﺯ ﺁﻧﻬﺎ ﺧﻴﻠﯽ ﺧﻮﺏ ﻫﺴﺘﻨﺪ ﻭ ﻣﻌﻠﻤﺎﻥ ﺧﻴﻠﯽ ﺧﻮﺑﯽ ﺩﺍﺭﻧﺪ، ﺗﺎﺳﻴﺴﺎﺕ ﺧﻮﺑﯽ ﺩﺍﺭﻧﺪ ﻭ ﺑﺮﻧﺎﻣﻪ ﺗﺮﺑﻴﺘﯽ ﺧﻮﺑﯽ ﻧﻴﺰ ﺩﺍﺭﻧﺪ. ﻭﻟﯽ ﺍﻛﺜﺮﻳﺖ ﺁﻧﻬﺎ ﻭﺣﺸﺘﻨﺎﻛﻨﺪ ﻭ ﺗﻮﺳﻂ ﻛﺴﺎﻧﯽ ﺍﺩﺍﺭﻩ ﻣﯽ‌ﺷﻮﻧﺪ ﻛﻪ ﺗﻨﻬﺎ ﻫﺪﻓﺸﺎﻥ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻥ ﻫﺮﭼﻪ ﺑﻴﺸﺘﺮ ﭘﻮﻝ ﺩﺭ ﻣﺪﺕ ﺑﺴﻴﺎﺭ ﻛﻢ ﺍﺳﺖ. ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﻦ ﻣﺮﺍﻛﺰ، ﺍﺷﺘﺒﺎﻫﺎﺕ ﺭﻭﺍﻧﺸﻨﺎﺧﺘﯽ ﻧﮕﺮﺍﻥ‌ﻛﻨﻨﺪﻩ‌ﺍﯼ ﺩﺭ ﻣﻮﺭﺩ ﺑﭽﻪ‌ﻫﺎ ﻣﺮﺗﻜﺐ ﻣﯽ‌ﺷﻮﻧﺪ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎﯼ ﺳﺎﺩﻩ‌ﻟﻮﺡ ﺷﺎﻳﺪ ﻗﺎﺩﺭ ﺑﻪ ﺗﺸﺨﻴﺺ ﺁﻧﻬﺎ ﻧﺒﺎﺷﻨﺪ.

ﻛﻮﺩﻛﺴﺘﺎﻥ ﻛﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺁﻧﻬﺎ ﺍﻟﺰﺍﻣﺎ ﺑﺮﺍﯼ ﻣﺎﺩﺭﺍﻧﻲ ﻛﻪ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﻨﺪ ﻃﺮﺍﺣﯽ ﻧﺸﺪﻩ‌ﺍﻧﺪ. ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﻣﻌﻤﻮﻻ ﺍﻃﻔﺎﻝ ﻛﻤﺘﺮ ﺍﺯ ﺳﻪ ﺳﺎﻝ ﺭﺍ ﻗﺒﻮﻝ ﻧﻤﯽ‌ﻛﻨﻨﺪ.

ﺣﺎﻻ ﭼﮕﻮﻧﻪ ﻳﻚ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺧﻮﺏ ﭘﻴﺪﺍ ﻛﻨﻴﻢ. ﺑﻪ ﻋﻨﻮﺍﻥ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻛﻪ ﺭﻭﺍﺑﻂ ﻣﺘﻌﺪﺩﯼ ﺑﺎ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﺩﺍﺭﺩ ﻭ ﺑﺎ ﺁﻣﻮﺯﺵ‌ﻫﺎﻳﯽ ﻛﻪ ﺩﺭ ﺁﻧﻬﺎ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﺳﺮ ﻭ ﻛﺎﺭ ﺩﺍﺭﺩ، ﻣﺘﺎﺳﻔﺎﻧﻪ ﺑﺎﻳﺪ ﺑﮕﻮﻳﻢ ﻛﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﺩﺭ ﺍﻧﺘﺨﺎﺏ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺭﺍﺣﺘﯽ‌ﺍﯼ ﻛﻪ ﺍﻳﻦ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﺩﺍﺭﺩ ﺩﺭ ﻧﻈﺮ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﻣﺎﺩﺭ ﺍﺯ ﺍﻳﻦ ﺟﻬﺖ ﺍﻳﻦ ﻳﺎ ﺁﻥ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﻧﺰﺩﻳﻚ ﺧﺎﻧﻪ‌ﺍﺵ ﺍﺳﺖ ﻭ ﻳﺎ ﺑﻪ ﻋﻠﺖ ﺍﻳﻨﻜﻪ ﺳﺮﻭﻳﺲ ﺭﻓﺖ‌ ﻭ‌ ﺁﻣﺪ ﺩﺍﺭﺩ. ﻣﻦ ﻭﺍﻗﻌﺎ ﺩﺭ ﻣﻘﺎﺑﻞ ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﻣﺎﺕ ﻭ ﻣﺘﺤﻴﺮﻡ.

ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺯ ﺳﻪ ﻧﻈﺮ ﺑﺎ ﻫﻢ ﺗﻔﺎﻭﺕ ﺩﺍﺭﻧﺪ: ﺍﻭﻻ ﺍﺯ ﻧﻈﺮ ﻣﻌﻠﻢ‌ﻫﺎ، ﺛﺎﻧﻴﺎ ﺍﺯ ﻧﻈﺮ ﻭﺳﺎﻳﻠﯽ ﻛﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻧﺪ ﻭ ﺳﻮﻣﺎ ﺍﺯ ﻧﻈﺮ ﺑﺮﻧﺎﻣﻪ ﺁﻣﻮﺯﺷﯽ‌ﺷﺎﻥ. ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺍﻳﻨﻬﺎ ﻭﺳﺎﻳﻞ ﻭ ﻣﻌﻠﻢ‌ﻫﺎ ﻭ ﺑﻘﻴﻪ ﻛﺎﺭﻛﻨﺎﻥ ﻛﻮﺩﻛﺴﺘﺎﻥ ﻫﺴﺘﻨﺪ. ﻣﻌﻠﻤﯽ ﻣﺠﺮﺏ، ﻣﻬﺮﺑﺎﻥ ﻭ ﺁﺭﺍﻡ ﻛﻪ ﺍﺯ ﺧﻮﺩﺵ ﻣﺎﻳﻪ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﻧﻘﺶ ﺑﺴﻴﺎﺭ ﻣﻬﻤﯽ ﺩﺍﺭﺩ.

ﺑﺎ ﻣﻌﻠﻢ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺻﺤﺒﺖ ﻛﻨﻴﺪ ﻭ ﺍﺯ ﺍﻭ ﺩﺭ ﻣﻮﺭﺩ ﺍﻃﻼﻋﺎﺗﯽ ﻛﻪ ﺩﺭ ﻣﻮﺭﺩ ﻛﻮﺩﻛﺎﻥ ﺧﺮﺩﺳﺎﻝ ﺩﺍﺭﺩ ﺳﻮﺍﻝ ﻛﻨﻴﺪ. ﺍﻳﻦ ﺣﻖ ﺷﻤﺎﺳﺖ ﺑﺪﺍﻧﻴﺪ ﺁﻳﺎ ﺍﻭ ﻛﻮﺩﻛﺎﻥ ﺧﺮﺩﺳﺎﻝ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﺪ ﻳﺎ ﻧﻪ. ﻫﺮﭼﻪ ﺑﺎﺷﺪ ﺷﻤﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺍﻭ ﻣﯽ‌ﺳﭙﺎﺭﻳﺪ. ﺍﻛﺜﺮ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﻣﻮﺭﺩ ﻭﺳﺎﻳﻠﯽ ﻛﻪ ﺩﺭ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﻫﺴﺘﻨﺪ ﻭ ﻧﻴﺰ ﺑﺮﻧﺎﻣﻪ ﺁﻣﻮﺯﺷﯽ ﺁﻧﻬﺎ ﭼﻨﺪﺍﻥ ﭼﻴﺰﯼ ﻧﻤﯽ‌ﺩﺍﻧﻨﺪ.

ﺍﮔﺮ ﺍﻃﻼﻋﺎﺗﯽ ﺩﺭ ﻣﻮﺭﺩ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﺩﺍﺷﺘﻴﺪ، ﺍﮔﺮ ﻣﯽ‌ﺩﺍﻧﺴﺘﻴﺪ ﻛﻪ ﻭﺳﺎﻳﻠﯽ ﻛﻪ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻧﺪ، (ﭼﻪ ﺁﻧﻬﺎﻳﻲ ﻛﻪ ﺩﺭ ﻓﻀﺎﯼ ﺑﺴﺘﻪ ﻗﺎﺑﻞ‌ ﺍﺳﺘﻔﺎﺩﻩ ﻫﺴﺘﻨﺪ ﻭ ﭼﻪ ﺁﻧﻬﺎﻳﯽ ﻛﻪ ﺩﺭ ﻓﻀﺎﯼ ﺑﺎﺯ ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ) ﭼﻪ ﺗﺎﺛﻴﺮﯼ ﺩﺭ ﺭﺷﺪ ﻋﺎﻃﻔﯽ ﻭ ﺫﻫﻨﯽ ﻛﻮﺩﻛﺎﻥ ﺧﺮﺩﺳﺎﻝ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﻨﺪ، ﺣﺘﻤﺎ ﺑﻪ ﻫﻨﮕﺎﻡ ﺑﺎﺯﺩﻳﺪ ﺍﺯ ﮐﻮﺩﮐﺴﺘﺎﻥ‌ﻫﺎ ﺗﻮﺟﻪ ﺑﻴﺸﺘﺮﯼ ﺑﻪ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﮐﺮﺩﻳﺪ.

ﺩﺭ ﻣﻮﺭﺩ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﻛﻪ ﺩﺭ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﻪ ﻛﺎﺭ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﺩ، ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﺍﺧﺘﻼﻓﺎﺕ ﺑﺴﻴﺎﺭ ﻓﺎﺣﺸﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﮔﺮﻭﻫﻲ ﻃﺮﻓﺪﺍﺭ ﺁﻣﻮﺯﺵ ﻫﻨﺮ ﺳﻨﺘﯽ ﻭ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻳﺪﻧﯽ ﻫﺴﺘﻨﺪ، ﺍﻳﻦ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎ ﺑﻪ ﺭﺷﺪ ﻋﺎﻃﻔﻲ ﻛﻤﻚ ﻛﻮﺩﻙ ﻣﯽ‌ﻛﻨﻨﺪ. ﻛﻮﺩﻙ ﺩﺭ ﺍﻳﻦ ﮔﻮﻧﻪ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﺍﻣﻜﺎﻥ ﺭﺷﺪ ﻋﻀﻼﺗﺶ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻭﺳﺎﻳﻠﯽ ﻛﻪ ﺩﺭ ﻓﻀﺎﯼ ﺑﺎﺯ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺁﻭﺭﺩ (ﺳﻪ‌ﭼﺮﺧﻪ ﺑﺎﺯﯼ ﻛﺮﺩﻥ، ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﺭﻓﺘﻦ ﺍﺯ ﻣﻜﻌﺐ‌ﻫﺎﯼ ﺑﺰﺭﮒ ﻛﻪ ﺩﺭ ﻓﻀﺎﯼ ﺑﺎﺯ ﻛﺎﺭ ﮔﺬﺍﺷﺘﻪ‌ﺍﻧﺪ). ﺩﺭ ﻓﻀﺎﯼ ﺑﺴﺘﻪ، ﻛﻮﺩﻙ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﮔﻞ ﺭﺱ ﺑﺎﺯﯼ ﻛﻨﺪ، ﺑﺎ ﺍﻧﮕﺸﺘﺎﻧﺶ، ﺑﺎ ﺁﺑﺮﻧﮓ ﻧﻘﺎﺷﯽ ﻛﻨﺪ، ﺍﺯ ﻭﺳﺎﻳﻞ ﻣﺨﺘﻠﻔﯽ ﻛﻪ ﺩﺭ ﺩﺍﺧﻞ ﻛﻮﺩﻛﺴﺘﺎﻥ ﻫﺴﺖ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﺪ ﺗﺎ ﻋﻀﻼﺕ ﻛﻮﺗﺎﻩ، ﻗﺪﺭﺕ ﺍﺑﺘﻜﺎﺭ ﻭ ﻋﻮﺍﻃﻔﺶ ﺭﺍ ﺭﺷﺪ ﺩﻫﺪ.

ﺭﻭﺵ ﺩﻳﮕﺮ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺭﻭﺵ «ﺷﻨﺎﺧﺘﯽ» ﺑﻨﺎﻣﻴﻢ. ﺍﻳﻦ ﺭﻭﺵ ﻣﺪﻋﯽ ﺍﺳﺖ ﻛﻪ ﺭﻭﺵ ﻗﺒﻠﯽ ﻫﻴﭻ ﺑﺪ ﻧﻴﺴﺖ، ﻭﻟﯽ ﺑﺎﻳﺪ ﺗﻮﺳﻌﻪ ﺩﺍﺩﻩ ﺷﻮﺩ. ﻃﺮﻓﺪﺍﺭﺍﻥ ﺍﻳﻦ ﺭﺍﻩ، ﻣﻌﺘﻘﺪﻧﺪ ﻛﻪ ﻛﻮﺩﻙ ﺩﺭ ﺍﻳﻦ ﺍﻭﻟﻴﻦ ﺳﺎﻝ‌ﻫﺎﯼ ﺯﻧﺪﮔﻴﺶ ﺑﺎﻳﺪ ﺗﺤﺖ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺣﺴﯽ ﻗﺮﺍﺭ ﮔﻴﺮﺩ، ﻣﻨﺘﻬﺎ ﺍﻳﻦ ﻛﺎﺭ ﺑﺎﻳﺪ ﺑﺪﻭﻥ ﻓﺸﺎﺭ ﺁﻭﺭﺩﻥ ﺑﺮ ﺍﻭ ﺑﺎﺷﺪ. ﺁﻧﻬﺎ ﻣﻌﺘﻘﺪﻧﺪ ﻛﻪ ﺑﺎ ﻛﻼﺱ‌ﻫﺎﯼ ﺭﺳﻤﻲ ﻛﻪ ﺍﺻﻼ ﻣﻨﺎﺳﺐ ﺍﻳﻦ ﺳﻦ ﻧﻴﺴﺖ، ﻧﺒﺎﻳﺪ ﺟﻮ ﺑﺪﻭﻥ ﺗﻨﺶ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﻛﻮﺩﻛﯽ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﺍﺩ. ﻣﻨﻈﻮﺭ ﻣﻦ ﻭﻳﺪﻳﻮ، ﻭﺳﺎﻳﻞ ﺁﻣﻮﺯﺷﯽ‌ﺍﯼ ﻛﻪ ﻛﻮﺩﻙ ﺑﺎ ﺁﻧﻬﺎ ﭼﺎﭖ ﻛﺮﺩﻥ ﺭﺍ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ، ﻭﺳﺎﺋﻠﯽ ﻛﻪ ﺑﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﻛﻮﺩﻙ ﻛﻤﻚ ﻣﯽ‌ﻛﻨﻨﺪ، ﻣﺜﻞ ﻋﺮﻭﺳﻚ‌ﻫﺎﯼ ﺧﻴﻤﻪ‌ﺷﺐ‌ﺑﺎﺯﯼ، ﻭ ﻭﺳﺎﻳﻠﯽ ﺍﺳﺖ ﻛﻪ ﺑﺮﺍﯼ ﺁﻣﻮﺯﺵ ﺧﻮﺍﻧﺪﻥ ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ.

‌ﺁﺧﺮﻳﻦ ﺳﺎﻝ ﻛﻮﺩﻛﺴﺘﺎﻧﯽ (ﺁﻣﺎﺩﮔﯽ) ﺑﺮﺍﻱ ﻛﻮﺩﻙ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺩﺍﺭﺩ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻛﻪ ﻛﻮﺩﻙ ﭘﻨﺞ ﺗﺎ ﺷﺶ ﺳﺎﻟﻪ ﺁﻣﺎﺩﮔﯽ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﻱ ﻳﺎﺩﮔﻴﺮﯼ ﺩﺍﺭﺩ ﻭ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻛﻪ ﺍﻣﻜﺎﻧﺎﺕ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﯼ ﻳﺎﺩﮔﻴﺮﯼ ﺩﺭ ﻛﻮﺩﻛﺴﺘﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﺧﻴﻠﯽ ﺗﺎﺳﻒ‌ﺁﻭﺭ ﺍﺳﺖ ﻛﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻛﻮﺩﻛﺎﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺯ ﺍﻳﻦ ﺍﻣﻜﺎﻧﺎﺕ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻨﺪ. ﺍﻳﻦ ﻭﺿﻌﻴﺖ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻧﺎﺷﯽ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺩﺭ ﺑﻌﻀﯽ ﺍﺯ ﺟﺎﻫﺎ ﻛﻮﺩﻛﺴﺘﺎﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﻭ ﻳﺎ ﺁﻥ ﻛﻪ ﺑﻪ ﻋﻠﺖ ﻛﻤﺒﻮﺩﻫﺎ، ﺁﻥ ﺭﺍ ﺣﺬﻑ ﻛﺮﺩﻩ‌ﺍﻧﺪ.

ﺍﮔﺮ ﻛﻮﺩﻛﺴﺘﺎﻥ ﺧﻮﺏ ﺩﻭﻟﺘﯽ ﺩﺭ ﻣﺤﻠﻪ ﺷﻤﺎ ﻧﺒﺎﺷﺪ، ﭼﻪ ﺭﺍﻩ‌ﺣﻠﯽ ﺑﺎﻳﺪ ﭘﻴﺪﺍ ﻛﺮﺩ؟ ﻳﻚ ﻛﻼﺱ ﺁﻣﺎﺩﮔﯽ ﺧﻮﺏ ﺧﺼﻮﺻﯽ ﺑﺮﺍﯼ ﺍﻭ ﭘﻴﺪﺍ ﻛﻨﻴﺪ. ﻣﻦ ﺧﻮﺩﻡ، ﺍﮔﺮ ﻻﺯﻡ ﺑﺎﺷﺪ، ﭘﻮﻝ ﻻﺯﻡ ﺭﺍ ﻗﺮﺽ ‌ﻣﯽ‌ﮐﻨﻢ.


ﺟﻤﻊ‌ﺑﻨﺪﯼ
ﺩﺭ ﭼﻬﺎﺭ ﺩﺭﺱ ﺍﺧﻴﺮ ﻣﺎ ﺭﺍﻩ ﺩﺭﺍﺯﯼ ﺭﺍ ﭘﻴﻤﻮﺩﻳﻢ ﻛﻪ ﺍﻟﺒﺘﻪ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺭﺷﺪ ﻋﻈﻴﻤﯽ ﻛﻪ ﻛﻮﺩﻙ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﻪ ﻣﯽ‌ﻛﻨﺪ، ﺟﺎﯼ ﻫﻴﭻ ﺗﻌﺠﺒﯽ ﻫﻢ ﻧﺪﺍﺭﺩ. ﺑﻴﺎﻳﻴﺪ ﺁﻥ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﺩﺭ ﺍﻳﻦ ﺳﻪ ﺳﺎﻝ ﻣﯽ‌ﮔﺬﺭﺩ ﺧﻼﺻﻪ ﻛﻨﻴﻢ ﻭ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪﺵ ﺗﻮﺍﻧﺴﺘﻪ ﻳﺎﺩ ﺑﮕﻴﺮﺩ،‌ ﺑﺮﺭﺳﯽ ﻛﻨﻴﻢ:


ﺍﻭ ﻧﻴﺎﺯﻫﺎﯼ ﺯﻳﺴﺘﯽ ﻻﺯﻡ ﺑﺮﺍﯼ ﺭﺷﺪ ﻋﻀﻼﺗﺶ، ﭼﻪ ﻋﻀﻼﺕ ﻛﻮﺗﺎﻩ ﻭ ﭼﻪ ﻋﻀﻼﺕ ﺑﻠﻨﺪ ﺭﺍ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻩ ﺍﺳﺖ.
ﻛﻮﺩﻙ ﻛﻨﺘﺮﻝ ﺟﻬﺶ‌ﻫﺎﯼ ﻏﺮﻳﺰﯼ‌ﺍﺵ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻩ ﺍﺳﺖ.
ﻛﻮﺩﻙ ﺍﺯ ﻣﺎﺩﺭﺵ ﺟﺪﺍ ﺷﺪﻩ ﺍﺳﺖ.
ﻛﻮﺩﻙ ﻳﺎﺩ ﮔﺮﻓﺘﻪ ﺍﺳﺖ ﻛﻪ ﭼﮕﻮﻧﻪ ﺑﺎ ﻛﻮﺩﻛﺎﻥ ﻫﻢ‌ﺳﻦ‌ ﻭ‌ ﺳﺎﻟﺶ ﺭﺍﺑﻄﻪ ﺑﺮﻗﺮﺍﺭ ﻛﻨﺪ.
ﻛﻮﺩﻙ ﻳﺎﺩ ﮔﺮﻓﺘﻪ ﺍﺳﺖ ﻛﻪ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ ﻳﺎ ﺑﻴﺎﻥ ﻛﻨﺪ ﻭ ﻳﺎ ﺍﻳﻦ ﻛﻪ ﺁﻥ ﺭﺍ ﭘﻴﺶ ﺧﻮﺩﺵ ﻧﮕﺎﻩ ﺩﺍﺭﺩ.
ﺗﻌﻠﻖ ﺍﻭ ﺑﻪ ﻳﻚ ﺟﻨﺲ (ﻣﺬﻛﺮ ﻳﺎ ﻣﻮﻧﺚ) ﻛﺎﻣﻼ ﻣﺸﺨﺺ ﺷﺪﻩ ﺍﺳﺖ.
ﻛﻮﺩﻙ ﺭﻭﺵ‌ ﺍﺳﺎﺳﯽ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﺴﺎﺋﻠﯽ ﺭﺍ ﻛﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﺗﻮﻟﺪ ﻭ ﺗﻨﺎﺳﻞ ﻣﯽ‌ﺷﻮﺩ ﻓﺮﺍ ﮔﺮﻓﺘﻪ ﺍﺳﺖ.
ﺍﻭ ﻣﺴﺎﻟﻪ ﻋﺸﻖ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺭﺍ ﺣﻞ ﻛﺮﺩﻩ ﺍﺳﺖ.

ﺧﺐ،‌ ﺍﻳﻦ ﻓﺮﺯﻧﺪ ﺷﻤﺎ، ﺳﻔﺮﯼ ﭘﻨﺞ ﺳﺎﻟﻪ ﺩﺭ ﺳﻔﻴﻨﻪ ﻓﻀﺎﻳﯽ ﻣﺎ، ﺯﻣﻴﻦ، ﮔﺬﺭﺍﻧﺪﻩ ﺍﺳﺖ ﻭ ﺍﮔﺮ ﺑﻪ ﺗﻮﺻﻴﻪ‌ﻫﺎﯼ ﻣﺎ ﺩﺭ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﻋﻤﻞ ﻛﺮﺩﻩ ﺑﺎﺷﻴﺪ، ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺍﺭﺍﯼ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌‌ﺍﯼ ﻗﻮﯼ ﺍﺳﺖ ﻭ ﭘﺎﻳﻪ‌ﻫﺎﯼ ﺷﺨﺼﻴﺘﯽ ﺳﺎﻟﻢ ﻭ ﻣﺴﺘﺤﻜﻤﯽ ﺩﺍﺭﺩ. ﺍﮔﺮ ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﺧﻮﺑﯽ ﭘﻴﺶ ﺭﻓﺘﻪ ﺑﺎﺷﺪ، ﺍﻭ ﺑﺎﻳﺪ ﺍﺣﺴﺎﺱ ﺍﻣﻨﻴﺖ ﻋﻤﻴﻘﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺑﻪ ﺧﻮﺩﺵ ﺍﻃﻤﻴﻨﺎﻥ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﺷﺨﺼﻴﺖ ﺧﻮﺩ ﺁﺷﻨﺎﻳﯽ ﺩﺍﺭﺩ. ﻛﺎﺭ ﺷﻤﺎ ﺑﺮﺍﯼ ﻛﻤﻚ ﺑﻪ ﺍﻭ ﺑﺮﺍﯼ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻥ ﺍﻳﻦ ﭘﺎﻳﻪ‌ﻫﺎﯼ ﻣﺴﺘﺤﻜﻢ ﺑﻪ ﻃﻮﺭ ﻋﻤﺪﻩ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﻴﺪﻩ ﺍﺳﺖ.


ﺣﺎﻝ ﻛﻪ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﻛﻮﺩﻙ ﺭﺍ ﺑﻪ ﺗﺮﺗﻴﺐ ﺗﺎﺭﻳﺦ ﺑﺮﺭﺳﯽ ﻛﺮﺩﻳﻢ، ﺩﺭ ﺩﻭﺭﻩ ﺩﻭﻡ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﮐﻮﺩﮎ ﺩﺭﺳﻨﺎﻣﻪ، ﺑﻪ ﻃﻮﺭ ﻣﻔﺼﻞ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺣﻴﺎﺗﯽ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﻧﻪ ﺗﻨﻬﺎ ﺩﺭ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ، ﺑﻠﻜﻪ ﺩﺭ ﺗﻤﺎﻡ ﻣﺪﺗﯽ ﻛﻪ ﻣﺴﺌﻮﻟﻴﺖ ﺍﻭ ﺑﺎ ﺷﻤﺎ ﺍﺳﺖ، ﺍﻫﻤﻴﺖ ﺧﻮﺍﻫﻨﺪ ﺩﺍﺷﺖ.
