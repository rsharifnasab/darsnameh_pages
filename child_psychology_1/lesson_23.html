ﺩﺭﺱ ﻫﺠﺪﻫﻢ- ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ

ﺗﺎ ﺑﻪ ﺣﺎﻝ، ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺁﻣﻮﺯﺷﯽ ﺭﺷﺪ ﮐﻮﺩﮎ ﺭﺍ ﺩﺭ ﺳﻪ ﻣﺮﺣﻠﻪ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ: ﻳﮏ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ، ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﻭ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ. ﺣﺎﻝ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺩﻳﮕﺮ ﺍﺯ ﺭﺷﺪ ﮐﻮﺩﮎ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ: ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ.
 
ﺍﮔﺮ ﻫﻨﻮﺯ ﺩﺭ ﺣﺎﻝ ﺩﺳﺖ ﻭ ﭘﻨﺠﻪ ﻧﺮﻡ ﮐﺮﺩﻥ ﺑﺎ ﮐﻮﺩﮐﯽ ﺩﺭ ﻣﺮﺣﻠﻪ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﻫﺴﺘﻴﺪ، ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﺸﺎﻫﺪﻩ ﺧﻮﺍﻫﻴﺪ ﮐﺮﺩ ﺩﻭﺭﻩ ﺑﻌﺪﯼ ﺭﺷﺪ ﮐﻮﺩﮎ ﺑﺴﻴﺎﺭ ﺭﺍﺣﺖ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﻃﻔﻞ ﺣﺪﺍﮐﺜﺮ ﺳﻌﯽ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺟﻠﺐ ﺭﺿﺎﻳﺖ ﻭﺍﻟﺪﻳﻨﺶ ﻣﯽ‌­ﮐﻨﺪ. ﺍﻭ ﺍﺯ ﻫﻤﮑﺎﺭﯼ ﮐﺮﺩﻥ ﺑﺎ ﺷﻤﺎ ﻭﺍﻗﻌﺎ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ، ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺩﺭ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﺍﻧﺴﺎﻥ ﻓﮑﺮ ﻣﯽ­‌ﮐﻨﺪ ﺗﻨﻬﺎ ﻟﺬﺕ ﺍﻭ ﭼﻮﺏ ﮔﺬﺍﺷﺘﻦ ﻻﯼ ﭼﺮﺥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺍﺳﺖ.

ﭘﻴﺶ ﺍﺯ ﻣﺪﺭﺳﻪ ﺗﻘﺮﻳﺒﺎ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ ﺍﺳﺖ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺎ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﻮﺛﺮ ﺩﺭ ﺭﺷﺪﺵ ﺭﻭ ﺩﺭ ﺭﻭ ﺷﻮﺩ ﻭ ﭼﮕﻮﻧﮕﯽ ﺍﻧﺠﺎﻡ ﻭ ﺗﺴﻠﻂ ﺑﺮ ﺍﻳﻦ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎ، ﻧﻘﺶ ﺗﻌﻴﻴﻦ‌ﮐﻨﻨﺪﻩ‌­ﺍﯼ ﺩﺭ ﺷﮑﻞ‌­ﮔﻴﺮﯼ ﺧﻮﺩ­ ﺍﻧﮕﺎﺭﻩ ﻭ ﺳﺎﺧﺘﺎﺭ ﺷﺨﺼﻴﺘﯽ ﺍﻭ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺷﺶ ﺳﺎﻟﮕﯽ ﺷﮑﻞ ﻗﻄﻌﯽ‌ﺍﺵ ﺭﺍ ﭘﻴﺪﺍ ﻣﯽ­‌ﮐﻨﺪ.

ﭘﻴﺶ ﺍﺯ ﺑﺮﺭﺳﯽ ﻭﻇﺎﻳﻔﯽ ﮐﻪ ﺍﻭ ﺑﺮﺍﯼ ﺭﺷﺪﺵ ﺑﻪ ﻋﻬﺪﻩ ﺩﺍﺭﺩ، ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺭﺍ ﺑﻪ ﺳﺮﻋﺖ، ﻭ ﺳﺎﻝ ﺑﻪ ﺳﺎﻝ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ ﺗﺎ ﺩﻳﺪﯼ ﮐﻠﯽ ﺩﺭ ﻣﻮﺭﺩ ﮐﻮﺩﮎ ﺳﻪ، ﭼﻬﺎﺭ ﻭ ﭘﻨﺞ ﺳﺎﻟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ
ﺍﻭﻝ ﺧﺼﻮﺻﻴﺎﺕ ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ ﺭﺍ ﺑﮕﻮﻳﻴﻢ. ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﭼﻪ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺧﻮﺩﺵ ﻭ ﭼﻪ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺟﻬﺎﻥ ﺩﻭﺭﻭﺑﺮﺵ ﺗﻌﺎﺩﻟﯽ ﻣﺴﺘﺤﮑﻢ‌ﺗﺮ ﺍﺯ ﭘﻴﺶ ﺩﺍﺭﺩ. ﺍﻭ ﺩﻭﺭﻩ ﮔﺬﺍﺭ ﺍﺯ ﻧﻮﺯﺍﺩ ﺑﻪ ﮐﻮﺩﮎ ﺭﺍ ﻃﯽ ﮐﺮﺩﻩ ﻭ ﺩﻳﮕﺮ ﺁﻥ ﭘﺮﻳﺸﺎﻧﯽ ﺩﻭﺭ‌ﻩ‌ﯼ ﮔﺬﺍﺭ ﺭﺍ ﻧﺪﺍﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﻳﮕﺮ ﺍﺣﺘﻴﺎﺟﯽ ﺑﻪ ﺗﮑﺮﺍﺭ ﻋﺎﺩﺍﺗﯽ ﺧﺎﺹ ﺟﻬﺖ ﺣﻤﺎﻳﺘﺶ ﺩﺭ ﻣﻘﺎﺑﻞ ﻋﻮﺍﻣﻞ ﺑﻴﺮﻭﻧﯽ ﻧﺪﺍﺭﺩ. ﺍﻭ ﺩﻳﮕﺮ ﺍﺣﺘﻴﺎﺟﯽ ﻧﻤﯽ‌ﺑﻴﻨﺪ ﻫﻤﻪ ﮐﺎﺭﻫﺎ ﺭﺍ ﺑﻪ ﺻﻮﺭﺗﯽ ﻏﻴﺮ ﻗﺎﺑﻞ ﺗﻐﻴﻴﺮ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. ﺭﻭﺣﻴﻪ ﺍﻭ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﻭ ﺣﺘﯽ ﺧﻮﺍﻫﺮ ﻭ ﺑﺮﺍﺩﺭﻫﺎﯼ ﺑﺰﺭﮔﺘﺮﺵ ﺍﻭ ﺭﺍ ﺗﺎﻳﻴﺪ ﮐﻨﻨﺪ.

ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺩﺭ ﺩﻭﺭﻩ ﻗﺒﻞ ﺑﺎ ﺗﻤﺎﻡ ﻗﻮﺍﻧﻴﻦ ﻣﻮﺟﻮﺩ ﻣﺨﺎﻟﻒ ﺑﻮﺩ، ﺍﻳﻨﮏ ﺍﺯ ﺍﻃﺎﻋﺖ ﻭ ﻟﻄﻒ ﮐﺮﺩﻥ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ. ﺩﻭﺭﻩ ﺧﺸﻢ‌ﻫﺎﯼ ﻃﻮﻓﺎﻧﯽ ﺑﻪ ﺳﺮ ﺭﺳﻴﺪﻩ ﻭ ﻭﺍﻟﺪﻳﻦ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺎ ﺍﻭ ﺑﻪ ﻃﻮﺭ ﻣﻨﻄﻘﯽ ﺻﺤﺒﺖ ﮐﻨﻨﺪ (ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﭘﻴﺶ ﺍﺯ ﺁﻥ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺁﻣﺪ ﺍﻳﻦ ﮐﺎﺭ ﻏﻴﺮ ﻣﻤﮑﻦ ﺑﺎﺷﺪ.)

ﺍﻭ ﺩﻳﮕﺮ ﺣﺎﮐﻢ ﻣﻄﻠﻖ ﻭ ﻣﺴﺘﺒﺪ ﻧﻴﺴﺖ. ﺩﻳﮕﺮ ﻣﺠﺒﻮﺭ ﻧﻴﺴﺘﻴﻢ ﺩﺭ ﻣﻘﺎﺑﻞ ﺗﺼﻤﻴﻤﺎﺕ ﺍﻭ ﺳﺮ ﺗﻌﻈﻴﻢ ﻓﺮﻭﺩ ﺁﻭﺭﻳﻢ. ﺷﺎﻩ ﺷﺎﻫﺎﻥ، ﻣﺴﺘﺒﺪ ﮐﻮﭼﻮﻟﻮ ﺍﺯ ﺗﺨﺖ ﭘﺎﻳﻴﻦ ﮐﺸﻴﺪﻩ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻭ ﺑﻪ ﺗﺪﺭﻳﺞ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﻣﺸﺘﺮﮐﺎ ﺍﺯ ﭼﻴﺰﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ ﻭ ﻣﻨﺘﻈﺮ ﻧﻮﺑﺘﺶ ﺑﺸﻮﺩ. ﺑﺮ ﺧﻼﻑ ﺳﻦ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﮕﯽ ﮐﻪ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﻣﻐﺸﻮﺵ ﻣﯽ‌ﮐﺮﺩ، ﺍﮐﻨﻮﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﺣﻮﺻﻠﻪ ﺑﻪ ﮐﺎﺭﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﭙﺮﺩﺍﺯﺩ.

ﺍﻳﻦ ﻫﺎ ﻫﻤﻪ ﻧﺎﺷﯽ ﺍﺯ ﺍﺯ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺟﺪﻳﺪﯼ ﺍﺳﺖ ﮐﻪ ﮐﺴﺐ ﮐﺮﺩﻩ ﻭ ﺑﺨﺸﯽ ﻫﻢ ﻧﺎﺷﯽ ﺍﺯ ﮐﺎﻫﺶ ﻳﺎﻓﺘﻦ ﺍﺿﻄﺮﺍﺏ‌ﻫﺎﯼ ﺩﻭﺭﺍﻥ ﮔﺬﺍﺭ ﺍﺳﺖ. ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻋﻀﻼﻧﯽ ﺍﻭ ﻣﻄﻤﺌﻦ‌ﺗﺮ ﺷﺪﻩ‌ﺍﻧﺪ. ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺻﺒﻮﺭﺍﻧﻪ‌ﺗﺮ ﻟﺒﺎﺱ‌ﻫﺎﻳﺶ ﺭﺍ ﺑﭙﻮﺷﺪ ﻳﺎ ﺑﺎ ﻣﮑﻌﺐ‌ﻫﺎﯼ ﻣﺨﺼﻮﺹ ﺧﺎﻧﻪ‌ﺳﺎﺯﯼ ﮐﻨﺪ. ﭘﻴﺸﺮﻓﺖ‌ ﺯﺑﺎﻧﺶ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﻬﺘﺮ ﺩﺭﮎ ﮐﻨﺪ ﻭ ﺩﺭ ﺿﻤﻦ ﺑﺘﻮﺍﻧﺪ ﺟﻮﺷﺶ‌ﻫﺎﯼ ﺩﺭﻭﻧﯽ ﺧﻮﺩ ﺭﺍ ﺑﻬﺘﺮ ﮐﻨﺘﺮﻝ ﮐﻨﺪ. ﺍﻭ ﻋﺎﺷﻖ ﮐﻠﻤﺎﺕ ﺟﺪﻳﺪ ﺍﺳﺖ. ﺑﻪ ﺗﺪﺭﻳﺞ ﮐﻪ ﺫﻫﻨﺶ ﺭﺷﺪ ﺑﻴﺸﺘﺮﯼ ﻣﯽ‌ﮐﻨﺪ، ﺩﺭﻫﺎﯼ ﺩﻧﻴﺎﯼ ﺟﺪﻳﺪﯼ ﺍﺯ ﺗﺼﻮﺭﺍﺕ ﻭ ﺧﻴﺎﻝ ﺑﻪ ﺭﻭﯼ ﺍﻭ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﺳﻦ ﮐﻮﺩﮎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺮﺍﯼ ﺧﻮﺩ ﻫﻤﺒﺎﺯﯼ ﺧﻴﺎﻟﯽ‌­ﺍﯼ - ﮐﻮﺩﮎ ﻳﺎ ﺣﻴﻮﺍﻥ - ﺩﺭﺳﺖ ﮐﻨﺪ ﮐﻪ ﻏﻴﺮ ﺍﺯ ﺧﻮﺩ ﺍﻭ ﮐﺲ ﺩﻳﮕﺮﯼ ﻗﺎﺩﺭ ﺑﻪ ﺩﻳﺪﻧﺶ ﻧﻴﺴﺖ. ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺷﺪﻳﺪ ﺑﻪ ﻳﮏ ﻫﻤﺒﺎﺯﯼ ﺩﺍﺭﺩ. ﺍﮔﺮ ﮐﻮﺩﮐﯽ ﻣﺠﺒﻮﺭ ﺑﺎﺷﺪ ﺍﮐﺜﺮ ﺍﻭﻗﺎﺕ ﺭﺍ ﺗﻨﻬﺎﻳﯽ ﺑﺎﺯﯼ ﮐﻨﺪ، ﺍﺣﺘﻤﺎﻝ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﺩ ﻳﮏ ﻫﻤﺒﺎﺯﯼ ﺧﻴﺎﻟﯽ ﺑﺮﺍﯼ ﺧﻮﺩﺵ ﺩﺭﺳﺖ ﮐﻨﺪ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻧﺒﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻧﮕﺮﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺍﻳﻦ ﻫﻤﺒﺎﺯﯼ ﺧﻴﺎﻟﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﭼﻨﺪﻳﻦ ﺳﺎﻝ ﺑﻪ ﺣﻴﺎﺗﺶ ﺍﺩﺍﻣﻪ ﺩﻫﺪ ﻭ ﺑﻌﺪﺍ ﺍﺯ ﺑﻴﻦ ﺑﺮﻭﺩ. ﻭﻟﯽ ﺩﺭ ﺯﻣﺎﻥ ﺣﻴﺎﺗﺶ، ﺍﻳﻦ ﻫﻤﺒﺎﺯﯼ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﺣﺴﺎﺱ ﺍﻣﻨﻴﺖ ﮐﺮﺩﻥ ﮐﻮﺩﮐﯽ ﺍﺳﺖ ﮐﻪ ﺁﻥ ﺭﺍ ﺍﺧﺘﺮﺍﻉ ﮐﺮﺩﻩ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ ﺭﺍﺑﻄﻪ ﺑﺎ ﮐﻮﺩﮐﺎﻥ ﻫﻢ ﺳﻦ ﻭ ﺳﺎﻟﺶ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ. ﺩﺭ ﺳﻦ ﺩﻭ ﺳﺎﻟﮕﯽ ﺍﻭ ﺩﺭ ﻣﺮﺣﻠﻪ ﺑﺎﺯﯼ ﺩﺭ ﮐﻨﺎﺭ ﺩﻳﮕﺮﯼ ﺑﻮﺩ . ﺍﮐﻨﻮﻥ ﻗﺪﺭﺕ ﺯﻧﺪﮔﯽ ﺍﺟﺘﻤﺎﻋﯽ ﺩﺍﺭﺩ، ﻗﺎﺩﺭ ﺍﺳﺖ ﻣﻨﺘﻈﺮ ﺷﻮﺩ، ﺑﻪ ﺻﻮﺭﺕ ﺷﺮﻳﮑﯽ ﺍﺯ ﻭﺳﺎﻳﻞ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ، ﻧﻮﺑﺘﺶ ﮐﻪ ﺭﺳﻴﺪ ﺑﺎﺯﯼ ﮐﻨﺪ ﻭ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻳﺶ ﺭﺍ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻋﻮﺽ ﮐﻨﺪ.

ﺳﻪ ﺳﺎﻟﮕﯽ ﭼﻪ ﺑﺮﺍﯼ ﻭﺍﻟﺪﻳﻦ ﻭ ﭼﻪ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﺎ ﺍﻃﺮﺍﻓﻴﺎﻧﺶ ﺩﺭ ﺻﻠﺢ ﻭ ﺻﻔﺎ ﺑﻪ ﺳﺮ ﺑﺮﺩ، «ﺳﻨﯽ ﻃﻼﻳﯽ» ﺍﺳﺖ. ﺍﻭ ﺯﻧﺪﮔﯽ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﺩ، ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺩﺭ ﻣﻮﺭﺩ ﺍﻭ ﻧﻈﺮ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﺩﺍﺭﻧﺪ. ﺳﻪ ﺳﺎﻟﮕﯽ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﻣﺘﻌﺎﺩﻝ ﺍﺳﺖ ﻭ ﻭﺍﻟﺪﻳﻦ ﺑﺎﻳﺪ ﺍﺯ ﺍﻳﻦ ﺳﻦ ﮐﻮﺩﮎ ﺣﺪﺍﮐﺜﺮ ﺍﺳﺘﻔﺎﺩﻩ ﺭﺍ ﺑﮑﻨﻨﺪ، ﺯﻳﺮﺍ ﻣﺮﺣﻠﻪ ﺑﻌﺪﯼ، ﻳﻌﻨﯽ ﺳﻦ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ، ﺳﻨﯽ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭ ﺁﻥ ﻧﺎ‌ﻣﺘﻌﺎﺩﻝ ﺍﺳﺖ.
 

ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ
 ﻃﺒﻴﻌﺖ ﻳﮏ ﺩﻭﺭﻩ ﻧﺎﻣﺘﻌﺎﺩﻝ ﺩﻳﮕﺮ ﺭﺍ ﺑﺮﺍﯼ ﻣﺎ ﺗﺪﺍﺭﮎ ﺩﻳﺪﻩ ﺍﺳﺖ: ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ. ﺩﺭ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﮐﻮﺩﮎ ﺑﺎﺭ ﺩﻳﮕﺮ ﺍﻓﺴﺎﺭ ﮔﺴﻴﺨﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺩﺭ ﭘﺎﻳﺎﻥ، ﺑﺎﺯ ﺑﻪ ﺩﻭﺭﻩ‌ﺍﯼ ﻣﺘﻌﺎﺩﻝ ﻣﯽ‌ﺭﺳﺪ. ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﺳﻨﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺁﻥ ﺗﻌﺎﺩﻝ ﺳﻪ ﺳﺎﻟﮕﯽ ﺍﺯ ﺑﻴﻦ ﻣﯽ‌ﺭﻭﺩ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺗﻌﺮﻳﻔﯽ ﺍﺯ ﺑﭽﻪ‌ﻫﺎﯼ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺩﺍﺩﻩ ﺑﺎﺷﻴﻢ‌ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﮐﻪ ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺑﻪ ﻣﺎﻧﻨﺪ ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﻪ ﺍﺳﺖ ﺑﺎ ﺍﻳﻦ ﺗﻔﺎﻭﺕ ﮐﻪ ﭘﺨﺘﻪ‌ﺗﺮ ﻭ ﮐﻨﺘﺮﻟﺶ ﮐﻤﯽ ﺭﺍﺣﺖ‌ﺗﺮ‌ ﺍﺳﺖ.

ﺍﻳﻦ ﺳﻦ ﺑﺎ ﻋﺪﻡ ﺗﻌﺎﺩﻝ، ﻧﺪﺍﺷﺘﻦ ﺍﺣﺴﺎﺱ ﺍﻣﻨﻴﺖ ﻭ ﻧﺎ ﻫﻤﺎﻫﻨﮕﯽ ﺩﺭﺗﻘﺮﻳﺒﺎ ﺗﻤﺎﻡ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻃﻔﻞ ﻣﺸﺨﺺ ﻣﯽ‌ﺷﻮﺩ. ﮐﻮﺩﮎ ﮐﻪ ﺩﺭ ﺳﻪ ﺳﺎﻟﮕﯽ ﺑﻪ ﻧﻈﺮ ﺧﻴﻠﯽ ﻣﺘﻌﺎﺩﻝ ﻣﯽ‌ﺭﺳﻴﺪ، ﺍﮐﻨﻮﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﮑﻨﺪﺭﯼ ﺑﺨﻮﺭﺩ، ﺑﻴﻔﺘﺪ ﻭ ﻳﺎ ﺗﺮﺱ ﺍﺯ ﺍﻓﺘﺎﺩﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻋﻼﺋﻢ ﻓﻴﺰﻳﮑﯽ ﺍﺿﻄﺮﺍﺏ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺑﻴﺸﺘﺮ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﺷﻮﺩ: ﭼﺸﻤﮏ ﺯﺩﻥ، ﻧﺎﺧﻦ ﺧﻮﺭﺩﻥ، ﺍﻧﮕﺸﺖ ﺩﺭ ﺑﻴﻨﯽ ﮐﺮﺩﻥ، ﺩﺳﺖ ﺯﺩﻥ ﺑﻪ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ، ﻣﮑﻴﺪﻥ ﺍﻧﮕﺸﺖ‌ﻫﺎ. ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻮﺩﮎ ﺗﻴﮏ‌ﻫﺎﻳﯽ ﺩﺭ ﺻﻮﺭﺗﺶ ﭘﻴﺪﺍ ﮐﻨﺪ.

ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺩﺭ ﺍﻳﻦ ﺳﻦ، ﮐﻮﺩﮎ ﺑﻪ ﻣﻌﻨﯽ ﻭﺍﻗﻌﯽ ﺗﺒﺪﻳﻞ ﺑﻪ ﻓﺮﺩﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﯽ‌ﺷﻮﺩ. ﺑﻪ ﺭﻏﻢ ﻣﺸﮑﻼﺗﯽ ﮐﻪ ﺑﺮﺍﯼ ﺳﺎﺯﺵ ﺑﺎ ﺩﻭﺳﺘﺎﻧﺶ ﺩﺍﺭﺩ، ﺩﻭﺳﺘﯽ ﺑﺮﺍﻳﺶ ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﭘﻴﺪﺍ ﻣﯽ­‌ﮐﻨﺪ. ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﭽﻪ‌ﻫﺎﯼ ﺩﻳﮕﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ، ﻣﺴﺎﺋﻠﯽ ﮐﻪ ﺍﻳﺠﺎﺩ ﻣﯽ‌­ﮐﻨﺪ ﺍﻧﺴﺎﻥ ﺭﺍ ﻳﺎﺩ ﺯﻣﺎﻧﯽ ﮐﻪ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻢ ﺩﺍﺷﺖ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ. ﺍﻭ ﻣﺴﺘﺒﺪ ﻭ ﭘﺮﺧﺎﺷﮕﺮ ﺍﺳﺖ ﻭ ﺩﻧﺒﺎﻝ ﺩﻋﻮﺍﺳﺖ. ﻣﻤﮑﻦ ﺍﺳﺖ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻓﮑﺮ ﮐﻨﻨﺪ ﮐﻮﺩﮎ ﺩﺍﺭﺩ ﺑﻪ ﻋﻘﺐ ﺑﺮﻣﯽ‌ﮔﺮﺩﺩ، ﺍﻣﺎ ﺍﻳﻦ ﻃﻮﺭ ﻧﻴﺴﺖ ﻭ ﮐﻮﺩﮎ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺳﻦ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﮐﻪ ﺳﻨﯽ ﻣﺘﻌﺎﺩﻝ ﺍﺳﺖ ﺑﺎﻳﺪ ﺍﺯ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺑﮕﺬﺭﺩ. ﻭﻟﯽ ﺍﻳﻦ ﺩﺭﺳﺖ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺍﻧﺴﺎﻥ ﺭﺍ ﺍﻏﻠﺐ ﺑﻪ ﻳﺎﺩ ﺯﻣﺎﻧﯽ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ ﮐﻪ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻢ ﺩﺍﺷﺖ. ﻫﻤﺎﻥ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﺍﻓﺮﺍﻃﯽ ﺭﺍ ﺩﺍﺭﺩ. ﮔﺎﻫﯽ ﺧﺠﻮﻝ ﺍﺳﺖ ﮔﺎﻫﯽ ﺑﺴﻴﺎﺭ ﭘﺮ ﺭﻭ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﻮﺩﮐﺎﻥ ﺑﺎﺯ ﺑﻪ ﻋﺎﺩﺕ‌ﻫﺎﯼ ﻣﺸﺨﺺ ﻭﺍﺑﺴﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ. ﺁﻧﻬﺎ ﺧﻮﺩ ﺭﺍ ﺯﻧﺪﺍﻧﯽ ﻗﻮﺍﻧﻴﻨﯽ ﺧﺸﮏ ﺑﺮﺍﯼ ﺧﻮﺭﺩﻥ، ﻟﺒﺎﺱ ﭘﻮﺷﻴﺪﻥ ﻭ ﻳﺎ ﺧﻮﺍﺑﻴﺪﻥ ﻣﯽ‌ﺑﻴﻨﻨﺪ. ﻗﺒﻮﻻﻧﺪﻥ ﮐﻮﭼﮑﺘﺮﻳﻦ ﺗﻐﻴﻴﺮﯼ ﺑﻪ ﺁﻧﻬﺎ ﺑﺴﻴﺎﺭ ﻣﺸﮑﻞ ﺍﺳﺖ. ﻧﮕﺮﺍﻧﯽ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﺑﺎﺑﺖ ﻣﺤﺒﺖ ﺩﺍﺭﻧﺪ ﺑﻪ ﺻﻮﺭﺕ ﮔﺮﻳﻪ، ﻧﻖ ﺯﺩﻥ ﻭ ﺳﻮﺍﻻﺕ ﺑﯽ‌ﺷﻤﺎﺭ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ.

ﭼﻬﺎﺭ ﺳﺎﻟﻪ‌ﻫﺎ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺑﺎ ﻫﻢ ﺑﺎﺯﯼ ﮐﻨﻨﺪ. ﺍﻣﺎ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﻫﻴﭻ ﺭﺑﻄﯽ ﺑﻪ ﮔﺮﺩﻫﻢ‌ﺁﻳﯽ‌ﻫﺎﯼ ﺗﻔﺮﻳﺤﯽ ﻧﺪﺍﺭﺩ. ﺍﻳﻦ ﺭﻭﺍﺑﻂ ﺑﺴﻴﺎﺭ ﺧﺸﻦ ﻭ ﻃﻮﻓﺎﻧﯽ ﺍﺳﺖ. ﺑﻪ ﻣﺤﺾ ﺍﻳﻨﮑﻪ ﮔﺮﻭﻫﯽ ﺗﺸﮑﻴﻞ ﺷﺪ، ﺑﻪ ﻫﻴﭻ ﮐﺴﯽ ﺍﺟﺎﺯﻩ ﻭﺭﻭﺩ ﺑﻪ ﺁﻥ ﺭﺍ ﻧﻤﯽ‌ﺩﻫﻨﺪ. ﮐﻮﺩﮐﺎﻥ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺩﺳﺘﻮﺭ ﺯﻳﺎﺩ ﻣﯽ‌ﺩﻫﻨﺪ، ﻫﻞ ﻣﯽ‌ﺩﻫﻨﺪ، ﮐﺘﮏ ﻣﯽ‌ﺯﻧﻨﺪ ﻭ ﺑﺎ ﮐﻤﺎﻝ ﻣﻴﻞ ﻓﺤﺶ ﺭﺩ ﻭ ﺑﺪﻝ ﻣﯽ‌ﮐﻨﻨﺪ. ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺧﺸﻦ ﺍﺳﺖ ﻭ ﻫﻴﭻ ﺍﻫﻤﻴﺘﯽ ﺑﻪ ﻋﻘﻴﺪﻩ ﺩﻳﮕﺮﺍﻥ ﻧﻤﯽ‌ﺩﻫﺪ.

ﺍﻭ «ﺍﻓﺴﺎﺭ ﮔﺴﻴﺨﺘﻪ» ﺍﺳﺖ. ﺁﻧﭽﻪ ﺭﻓﺘﺎﺭ ﮐﻮﺩﮎ ﺭﺍ ﮐﻨﺘﺮﻝ ﻣﯽ‌ﮐﺮﺩ ﺍﺯ ﺩﺳﺖ ﺭﻓﺘﻪ ﺍﺳﺖ. ﺍﻭ ﻣﺸﺖ ﻣﯽ‌ﺯﻧﺪ، ﻟﮕﺪ ﻣﯽ‌ﺯﻧﺪ ﻭ ﺩﭼﺎﺭ ﻋﺼﺒﺎﻧﻴﺖ‌ﻫﺎﯼ ﺷﺪﻳﺪ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﻧﻴﺰ ﺍﻓﺴﺎﺭ ﮔﺴﻴﺨﺘﻪ ﺍﺳﺖ. ﻓﺮﻳﻔﺘﻪ ﮐﻠﻤﺎﺕ ﻭ ﺻﺪﺍﯼ ﺁﻧﻬﺎﺳﺖ. ﺑﺮﺍﯼ ﺍﻭﻟﻴﻦ ﺑﺎﺭ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﺩ ﻳﮏ ﺳﺮﯼ ﮐﻠﻤﺎﺕ ﻣﻄﺎﺑﻖ ﻣﻴﻞ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﻧﻴﺴﺘﻨﺪ.  ﺍﻭ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺑﺎ ﺭﻭ ﮐﺮﺩﻥ ﺍﻳﻦ ﺩُﺭ ﻭ ﮔﻮﻫﺮﻫﺎ، ﺧﺼﻮﺻﺎ ﺍﮔﺮ ﺗﻤﺎﺷﺎﭼﯽ ﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻭﺍﻟﺪﻳﻨﺶ ﺭﺍ ﺑﻪ ﺳﺘﻮﻩ ﺁﻭﺭﺩ. ﺷﻮﺧﯽ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻣﺪﻓﻮﻉ ﺍﻧﺴﺎﻥ ﺑﺎﺷﺪ ﺍﻭ ﺭﺍ ﺑﺴﻴﺎﺭ ﺧﻮﺷﻨﻮﺩ ﻣﯽ‌ﮐﻨﺪ، ﻣﺜﻼ ﻣﯽ‌ﮔﻮﻳﺪ: «ﻣﺎﻣﺎﻥ ﻣﯽ‌ﺩﻭﻧﯽ ﺍﻣﺮﻭﺯ ﻧﺎﻫﺎﺭ ﭼﯽ ﻣﯽ‌ﺧﻮﺍﻡ؟ ﮔﻮﺷﺖ، ﻫﻮﻳﺞ، ﺑﺴﺘﻨﯽ ﻭ ﺍَﻳﯽ!» ﺁﻥ ﻭﻗﺖ، ﺍﺯ ﺑﺬﻟﻪ ﮔﻮﻳﯽ ﺧﻮﺩ ﻗﺎﻩ‌ﻗﺎﻩ ﺑﻪ ﺧﻨﺪﻩ ﻣﯽ‌ﺍﻓﺘﺪ. ﻣﺎﺩﺭﯼ ﮐﻪ ﺍﺯ ﺍﺻﻞ «ﺗﻘﻮﻳﺖ ﮐﺮﺩﻥ» ﻣﻄﻠﻊ ﺑﺎﺷﺪ، ﺍﻫﻤﻴﺘﯽ ﺑﻪ ﺍﻳﻦ ﺣﺮﻑ‌ﻫﺎﯼ ﺍﻭ ﻧﻤﯽ‌ﺩﻫﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﻧﺸﻨﻴﺪﻩ ﻣﯽ‌ﮔﻴﺮﺩ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ، ﺑﻪ ﺗﺪﺭﻳﺞ، ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﻪ ﻣﯽ‌ﺷﻮﺩ ﺍﻳﻦ ﮐﻠﻤﺎﺕ ﻧﻴﺰ ﻓﺮﺍﻣﻮﺵ ﻣﯽ‌ﺷﻮﻧﺪ.

ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺍﻓﺮﺍﺩ ﻧﻴﺰ ﺍﻓﺴﺎﺭ ﮔﺴﻴﺨﺘﻪ ﺍﺳﺖ. ﺍﻭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺩﺭ ﻣﻘﺎﺑﻞ ﺩﺳﺘﻮﺭﻫﺎﯼ ﺻﺮﻳﺢ ﻣﻘﺎﻭﻣﺖ ﮐﻨﺪ، ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﺍﻭ ﺭﺍ ﺗﺤﺮﻳﮏ ﻣﯽ‌ﮐﻨﻨﺪ، ﻻﻑ ﻣﯽ‌ﺯﻧﺪ ﻭ ﻗﺴﻢ ﻣﯽ‌ﺧﻮﺭﺩ. «ﺍﻟﺂﻥ ﻳﻪ ﻣﺸﺖ ﻣﯽ‌ﺯﻧﻢ ﺗﻮ ﺷﻴﮑﻤﺖ» ﻳﮑﯽ ﺍﺯ ﺗﻬﺪﻳﺪﻫﺎﯼ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺍﻳﻦ ﺳﻦ ﺍﺳﺖ. ﺗﺼﻮﺭﺍﺗﺶ ﻧﻴﺰ ﺍﻓﺴﺎﺭ ﮔﺴﻴﺨﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺳﺖ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﻓﺮﻕ ﻭﺍﻗﻌﯽ ﻭ ﻏﻴﺮ ﻭﺍﻗﻌﯽ ﺭﺍ ﺑﻪ ﺍﻭ ﻓﻬﻤﺎﻧﺪ. ﺟﺪﺍ ﮐﺮﺩﻥ ﻭﺍﻗﻌﯽ ﻭ ﻏﻴﺮ ﻭﺍﻗﻌﯽ ﺑﻪ ﻧﻈﺮ ﺍﻭ ﭼﻨﺪﺍﻥ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﻧﻴﺴﺖ. ﺍﻭ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺧﻢ ﺑﻪ ﺍﺑﺮﻭ ﺑﻴﺎﻭﺭﺩ ﺑﺮﺍﻳﺘﺎﻥ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﻏﻴﺮ‌ﻗﺎﺑﻞ ﺑﺎﻭﺭ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﮐﻨﺪ. ﻭﻟﯽ ﻭﺍﻟﺪﻳﻦ ﻧﺒﺎﻳﺪ ﻣﺮﺗﮑﺐ ﺍﻳﻦ ﺍﺷﺘﺒﺎﻩ ﺷﻮﻧﺪ ﮐﻪ ﺍﻭ ﺭﺍ «ﺩﺭﻭﻏﮕﻮ» ﺑﺨﻮﺍﻧﻨﺪ. ﺍﻳﻦ ﺑﻪ ﻣﻌﻨﯽ ﻧﺎﺁﮔﺎﻫﯽ ﺁﻧﻬﺎﺳﺖ ﺍﺯ ﻗﺼﺪ ﮐﻮﺩﮎ ﺑﺮﺍﯼ ﺗﻤﻴﺰ ﻭﺍﻗﻌﯽ ﺍﺯ ﻏﻴﺮ ﻭﺍﻗﻌﯽ.

ﺍﻭ ﻣﮕﺮ ﺩﺭ ﻣﻮﺭﺩ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﺪ ﻭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﻣﺎﻝ ﺍﻭﺳﺖ، ﻣﻌﻨﯽ ﻣﺎﻟﮑﻴﺖ ﺭﺍ ﻧﻤﯽ‌ﻓﻬﻤﺪ. ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﻫﻤﺎﻥ ﻗﺪﺭ «ﺩﺯﺩ» ﺍﺳﺖ ﮐﻪ «ﺩﺭﻭﻏﮕﻮ». ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﻭ ﻓﺮﻗﯽ ﺑﻴﻦ  ﭼﻴﺰﯼ ﮐﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﺎﺷﺪ ﻭ ﭼﻴﺰﯼ ﮐﻪ ﻣﺘﻌﻠﻖ ﺑﻪ ﺍﻭﺳﺖ ﻧﻤﯽ‌ﮔﺬﺍﺭﺩ. ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺑﭽﻪ ﻫﻤﺴﺎﻳﻪ «ﻣﺎﻝ» ﺍﻭﺳﺖ، ﭼﻮﻥ ﺑﺎ ﺁﻥ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﺮﺩ، ﺩﺭ ﺟﻴﺒﺶ ﮔﺬﺍﺷﺘﻪ ﻭ ﺑﻪ ﺧﺎﻧﻪ ﺁﻭﺭﺩﻩ ﺍﺳﺖ. ﻧﻈﺮﻳﻪ ﻣﺎﻟﮑﻴﺖ ﺩﺭ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺳﺖ.

ﺗﺤﺮﮎ ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻓﻮﻕ‌ﺍﻟﻌﺎﺩﻩ ﺍﺳﺖ ﻭ ﺩﺭ ﺣﺎﻝ ﺩﻭ ﺍﺯ ﭘﻠﻪ‌ﻫﺎ ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﻣﯽ‌ﺭﻭﺩ، ﺩﺭﺧﺎﻧﻪ ﻳﺎ ﺁﭘﺎﺭﺗﻤﺎﻥ ﻣﺜﻞ ﻓﺸﻨﮓ ﻣﯽ‌ﺩﻭﺩ ﻭ ﺩﺭﻫﺎ ﺭﺍ ﺑﺎ ﺳﺮ ﻭ ﺻﺪﺍ ﺑﻪ ﻫﻢ ﻣﯽ‌ﮐﻮﺑﺪ. ﻭﺭﺍﺟﯽ ﺍﺳﺖ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺩﺭﺑﺎﺭﻩ ﻫﻤﻪ ﭼﻴﺰ ﺻﺤﺒﺖ ﮐﻨﺪ. ﺧﻮﺩ ﻣﻔﺴﺮ ﺩﻧﻴﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺍﻭ ﺭﺍ ﺩﺭ ﺑﺮ ﮔﺮﻓﺘﻪ ﻭ ﮔﺎﻫﯽ، ﻣﺨﺎﻃﺐ ﺍﻳﻦ «ﻣﻔﺴﺮ» ﻧﻴﺰ ﻫﺴﺖ.
ﮐﻮﺩﮎ ﺷﻴﻔﺘﻪ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﮐﻠﻤﺎﺕ ﺑﯽ‌ﻣﻌﻨﯽ ﻭ ﻗﺎﻓﻴﻪ ﺩﺭﺳﺖ ﮐﺮﺩﻥ ﺍﺳﺖ . ﻭﺍﻟﺪﻳﻦ ﺁﮔﺎﻩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺯ ﺍﻳﻦ ﻋﻼﻗﻪ ﺟﺪﻳﺪ ﺑﻪ ﮐﻠﻤﺎﺕ ﻭ ﻗﺎﻓﻴﻪ ﮐﺮﺩﻥ ﺁﻧﻬﺎ، ﺍﺯ ﻃﺮﻳﻖ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺎ ﮐﻠﻤﺎﺕ، ﺑﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﻓﺮﺯﻧﺪﺷﺎﻥ ﮐﻤﮏ ﮐﻨﻨﺪ. ﮐﻮﺩﮎ ﻋﻼﻗﻪ ﻭﺍﻓﺮﯼ ﺑﻪ ﭼﻴﺰﻫﺎﯼ ﻓﮑﺎﻫﯽ، ﺍﻏﺮﺍﻕ ﻭ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎ ﻭ ﺍﻓﺴﺎﻧﻪ‌ﻫﺎﯼ ﮐﻮﺩﮐﺎﻧﻪ ﺩﺍﺭﺩ.  ﺍﺯ ﺳﻮﺍﻝ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ: «ﺑﮕﻮ ﺑﺒﻴﻨﻢ، ﺗﻮﯼ ﺟﻴﺒﺖ ﻳﻪ ﻓﻴﻞ ﻗﺎﻳﻢ ﻧﮑﺮﺩﯼ؟» ﺧﻴﻠﯽ ﺧﻮﺷﺶ ﻣﯽ‌ﺁﻳﺪ.

ﺍﻭ ﺍﺯ ﺍﻏﺮﺍﻕ ﮐﺮﺩﻥ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ، ﺑﻪ ﻧﻤﺎﻳﺶ‌ﻫﺎﯼ ﻋﺮﻭﺳﮑﯽ ﻋﻼﻗﻤﻨﺪ ﺍﺳﺖ ﻭ ﺧﻴﻠﯽ ﺧﻮﺏ ﻗﺎﺩﺭ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻋﺮﻭﺳﮏ‌ﻫﺎﯼ ﺧﻴﻤﻪ‌ﺷﺐ‌ﺑﺎﺯﯼ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺩﺳﺖ ﺁﻧﻬﺎ ﺭﺍ ﺣﺮﮐﺖ ﻣﯽ‌ﺩﻫﻨﺪ. ﺩﺭ ﺍﻃﺮﺍﻓﺶ، ﭼﻪ ﺩﺭ ﺧﺎﻧﻪ ﻭ ﭼﻪ ﺩﺭ ﺧﺎﺭﺝ ﺍﺯ ﺁﻥ، ﺻﺤﻨﻪ‌ﻫﺎﯼ ﻧﻤﺎﻳﺸﯽ ﺑﯽ‌ﭘﺎﻳﺎﻧﯽ ﺍﺯ ﻣﮑﻌﺐ‌ﻫﺎﯼ ﺧﺎﻧﻪ‌ﺳﺎﺯﯼ، ﻣﺎﺷﻴﻦ، ﮐﺎﻣﻴﻮﻥ، ﻗﻄﺎﺭ، ﮐﺸﺘﯽ ﻭ ﻋﺮﻭﺳﮏ ﻭ ﺁﺩﻣﮏ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﺩﺭﺳﺖ ﻣﯽ‌ﮐﻨﺪ.

 ﺍﻭ ﻫﻴﭻ‌ﻭﻗﺖ ﻧﻤﯽ‌ﺩﺍﻧﺪ ﺑﻪ ﮐﺠﺎ ﻣﯽ‌ﺭﻭﺩ. ﺳﺮﻋﺖ ﺣﺮﮐﺖ ﻭ ﺳﺎﺧﺘﺎﺭ ﺫﻫﻨﯽ ﺩﺭ ﻧﻮﺳﺎﻧﺶ ﺍﻭ ﺭﺍ ﺑﻪ ﺭﺍﻩ‌ﻫﺎﯼ ﺩﻭﺭ ﺍﻓﺘﺎﺩﻩ ﻭ ﻏﻴﺮ ﻗﺎﺑﻞ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﻣﯽ‌ﺑﺮﺩ. ﺭﻭﺯﯼ ﻣﺎﺩﺭﯼ ﺍﺯ ﻓﺮﺯﻧﺪ ﭼﻬﺎﺭ ﺳﺎﻟﻪ‌ﺍﺵ ﭘﺮﺳﻴﺪ ﮐﻪ ﭼﻪ ﭼﻴﺰﯼ ﻧﻘﺎﺷﯽ ﻣﯽ‌ﮐﻨﯽ ﻭ ﺍﻭ ﺩﺭ ﺟﻮﺍﺏ ﮔﻔﺖ: «ﭼﻪ ﻣﯽ‌ﺩﻭﻧﻢ، ﻣﻦ ﮐﻪ ﻫﻨﻮﺯ ﺗﻤﻮﻣﺶ ﻧﮑﺮﺩﻡ!» ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﺮﻭﻉ ﺑﻪ ﻧﻘﺎﺷﯽ ﻳﮏ ﺟﺎﺩﻩ ﮐﻨﺪ ﻭ ﺁﺧﺮ ﮐﺎﺭ ﻧﻘﺎﺷﯽ‌ﺍﺵ ﺩﺍﻳﻨﺎﺳﻮﺭ ﻳﺎ ﮐﺎﻣﻴﻮﻧﯽ ﺍﺯ ﮐﺎﺭ ﺩﺭ ﺁﻳﺪ. ﺑﻪ ﻫﻨﮕﺎﻡ ﺗﻌﺮﻳﻒ ﭼﻴﺰﯼ ﮐﻪ ﺩﻳﺪﻩ ﺍﺳﺖ ﻧﻴﺰ ﺩﭼﺎﺭ ﻫﻤﻴﻦ ﮔﻢ­ﮔﺸﺘﮕﯽ‌ﻫﺎﯼ ﻏﻴﺮ ﻗﺎﺑﻞ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﻣﯽ‌ﺷﻮﺩ.

ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﺎﮐﺖ، ﭘﺮ ﺳﺮ ﻭ ﺻﺪﺍ، ﺁﺭﺍﻡ، ﺭﺍﺣﺖ، ﺯﻭﺭﮔﻮ، ﻣﺘﻠﮏ‌ﮔﻮ، ﻣﺴﺘﻘﻞ، ﺍﺟﺘﻤﺎﻋﯽ، ﻗﻬﺮﻣﺎﻥ ﺩﻭ، ﻫﻨﺮﭘﻴﺸﻪ، ﻣﻄﻴﻊ، ﺑﯽ‌ﻗﻴﺪ، ﮐﻨﺠﮑﺎﻭ، ﺭﮎ، ﺭﻭﺩﻩ ﺩﺭﺍﺯ، ﺷﻮﺥ، ﺟﺰﻡ ﺍﻧﺪﻳﺶ، ﺍﺣﻤﻖ ﻭ ﭘﺮﺧﺎﺷﮕﺮ ﺑﺎﺷﺪ.
ﺑﻪ ﻋﻠﺖ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﺧﺎﺹ ﺍﻳﻦ ﺳﻦ، ﺑﺎﻳﺪ ﺧﻴﻠﯽ ﺟﺪﯼ ﺑﺎ ﺍﻭ ﺑﺮﺧﻮﺭﺩ ﮐﺮﺩ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎﯼ ﺿﻌﻴﻒ ﻳﺎ ﻣﺮﺩﺩ ﮐﺎﺭﺷﺎﻥ ﺧﻴﻠﯽ ﺳﺨﺖ ﺍﺳﺖ. ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺍﺯ ﺗﻨﻮﻉ ﺧﻮﺷﺶ ﻣﯽ‌ﺁﻳﺪ. ﺍﻭ ﺍﻏﻠﺐ ﺑﺎﻳﺪ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﻳﺶ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﺪ. ﻣﺎﺩﺭ ﻣﻄﻠﻊ ﻫﻤﻴﺸﻪ ﭼﻨﺪ ﻓﻌﺎﻟﻴﺖ ﺗﺎﺯﻩ ﺩﺭ ﭼﻨﺘﻪ‌ﺍﺵ ﺩﺍﺭﺩ ﺗﺎ ﺑﻪ ﻣﺤﺾ ﺍﻳﻦ ﮐﻪ ﻓﺮﺯﻧﺪ ﭼﻬﺎﺭ ﺳﺎﻟﻪ‌ﺍﺵ ﺩﺭ ﻭﺿﻌﻴﺘﯽ ﻗﺮﺍﺭ ﮔﻴﺮﺩ ﮐﻪ ﻣﻤﮑﻦ ﺑﺎﺷﺪ ﮐﺎﺭ ﺑﻪ ﺟﺎﻫﺎﯼ ﺑﺎﺭﻳﮏ ﺑﮑﺸﺪ، ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺍﻭ ﭘﻴﺸﻨﻬﺎﺩ ﮐﻨﺪ. ﺑﺎﺯﯼ‌ﻫﺎ ﻭ ﺭﻓﺘﺎﺭ ﺍﻭ ﺍﮔﺮ ﮐﻨﺘﺮﻝ ﻧﺸﻮﻧﺪ، ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺗﺒﺪﻳﻞ ﺑﻪ ﺧﺮﺍﺑﮑﺎﺭﯼ ﻣﯽ‌ﺷﻮﻧﺪ. ﭘﺪﺭ ﻳﺎ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﭼﻨﻴﻦ ﻟﺤﻈﺎﺗﯽ ﺭﺍ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﮐﺮﺩﻩ ﻭ ﻓﻌﺎﻟﻴﺖ ﺟﺪﻳﺪﯼ ﺑﻪ ﻭﯼ ﭘﻴﺸﻨﻬﺎﺩ ﮐﻨﻨﺪ.

ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺩﺭ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﻭﺟﻬﻪ ﺍﺟﺘﻤﺎﻋﯽ ﮐﻮﺩﮎ ﺭﺷﺪ ﺑﻴﺸﺘﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﺳﻪ ﺳﺎﻟﮕﯽ ﻳﺎﻓﺘﻪ ﻭ ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﺩﻭﺳﺘﯽ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺩﺍﺭﺩ، ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﻨﺒﻴﻬﯽ ﻣﻮﺛﺮ، ﺍﺯ ﺟﺪﺍ ﮐﺮﺩﻥ ﺍﻭ ﺍﺯ ﺟﻤﻊ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪ. ﻣﺜﻼ ﻣﺎﺩﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﮕﻮﻳﺪ: «ﭼﻮﻥ ﻧﻤﯽ‌ﺗﻮﻧﯽ ﺑﺎ ﺑﭽﻪ‌­ﻫﺎ ﺑﺎﺯﯼ ﮐﻨﯽ، ﭘﺲ ﺗﻨﻬﺎ ﺑﺎﺯﯼ ﮐﻦ. ﺷﺎﻳﺪ ﻳﻪ ﺩﻗﻴﻘﻪ ﺩﻳﮕﻪ ﺑﺘﻮﻧﯽ ﺩﻭﺑﺎﺭﻩ ﺑﺎ ﺍﻭﻧﺎ ﺑﺎﺯﯼ ﮐﻨﯽ. ﺣﺎﻻ ﺑﺒﻴﻨﻴﻢ!» ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﺎﺩﺭ ﺑﻪ ﺍﻭ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ «ﺧﻮﺩ ﺭﺍ ﺑﺎﺯﻧﺪﻩ ﺍﺣﺴﺎﺱ ﻧﮑﻨﺪ». ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺍﻭ ﺭﺍ ﺑﻪ ﺑﻬﺘﺮ ﮐﺮﺩﻥ ﺭﻓﺘﺎﺭﺵ ﺟﻬﺖ ﻭﺍﺭﺩ ﺷﺪﻥ ﺩﻭﺑﺎﺭﻩ ﺑﻪ ﮔﺮﻭﻩ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﺪ.

ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺍﻳﻦ‌ﮔﻮﻧﻪ ﺍﺳﺖ. ﻭ ﺩﺭﺳﺖ ﺯﻣﺎﻧﯽ ﮐﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺩﻳﮕﺮ ﺯﻧﺪﮔﯽ ﮐﺮﺩﻥ ﺩﺭ ﮐﻨﺎﺭ ﺍﻳﻦ ﻏﻮﻝ ﮐﻮﭼﻮﻟﻮ ﺩﻳﮕﺮ ﺍﺭﺯﺷﯽ ﻧﺪﺍﺭﺩ، ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﺶ ﻣﯽ‌ﺷﻮﺩ. ﻧﺎﮔﻬﺎﻥ ﻫﻤﻪ ﭼﻴﺰ ﺗﻐﻴﻴﺮ ﻣﯽ‌ﮐﻨﺪ. ﺗﻌﺎﺩﻝ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺟﺎﯼ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﺭﺍ ﻣﯽ‌ﮔﻴﺮﺩ.

ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﻪ
ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺳﻦ ﺷﻴﺮﻳﻨﯽ ﺍﺳﺖ. ﺭﻓﺘﺎﺭ ﻟﺠﺎﻡ ﮔﺴﻴﺨﺘﻪ ﺩﻳﮕﺮ ﺗﻤﺎﻡ ﺷﺪﻩ ﺍﺳﺖ. ﺍﮐﻨﻮﻥ ﮐﻮﺩﮎ ﺑﺎ ﮐﻤﺎﻝ ﻣﻴﻞ ﻋﺎﻗﻞ، ﺟﺪﯼ، ﺛﺎﺑﺖ ﻗﺪﻡ ﻭ  ﻣﺘﻌﺎﺩﻝ ﺍﺳﺖ. ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺩﺭ ﻛﻨﻪ ﻭﺟﻮﺩﺵ ﺑﻪ ﺧﻮﺩ ﺍﻋﺘﻤﺎﺩ ﭘﻴﺪﺍ ﮐﺮﺩﻩ، ﺁﺭﺍﻡ ﻭ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻨﯽ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺧﻴﻠﯽ ﻣﺸﮑﻞ ﭘﺴﻨﺪ ﻭ ﭘﺮﻣﺪﻋﺎ ﻧﻴﺴﺖ. ﻓﻘﻂ ﺩﺳﺖ ﺑﻪ ﮐﺎﺭﻫﺎﻳﯽ ﻣﯽ‌ﺯﻧﺪ ﮐﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. ﻭ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﺍﮐﺜﺮﺍ ﺩﺭ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﺪ ﻣﻮﻓﻖ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﺳﻦ ﻧﻮﺳﺎﻧﺎﺕ ﺍﺳﺖ، ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺳﻨﯽ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﺑﺴﻴﺎﺭ ﺛﺎﺑﺖ ﻗﺪﻡ ﻣﯽ‌ﺷﻮﺩ. ﺑﺮ ﺧﻼﻑ ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﮐﻪ ﻫﻴﭻ ﻭﻗﺖ ﻧﻤﯽ‌ﺩﺍﻧﺪ ﭼﻪ ﻧﻘﺎﺷﯽ‌ﺍﯼ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﮑﺸﺪ، ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﻪ ﺍﺯ ﻗﺒﻞ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﮑﺸﺪ. ﺩﺭ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﺑﺮﺍﻳﺶ ﺍﻫﻤﻴﺘﯽ ﻧﺪﺍﺷﺖ ﮐﺎﺭﯼ ﺭﺍ ﮐﻪ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﺮﺩ ﺑﻪ ﺍﺗﻤﺎﻡ ﺑﺮﺳﺎﻧﺪ ﻭ ﻳﺎ ﺩﺭ ﻧﻴﻤﻪ ﺭﺍﻩ ﺁﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﻣﯽ‌ﺩﺍﺩ. ﺍﮐﻨﻮﻥ، ﺩﺭ ﭘﻨﺞ ﺳﺎﻟﮕﯽ، ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﮐﺎﺭﯼ ﺭﺍ ﮐﻪ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﺪ ﺗﻤﺎﻡ ﮐﻨﺪ. ﺩﺭ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﮐﻮﺩﮎ ﺗﻦ ﺑﻪ ﻣﺎﺟﺮﺍﺟﻮﻳﯽ ﻣﯽ‌ﺩﻫﺪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺩﺭ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﻣﯽ‌ﺩﺍﻧﺪ ﮐﻪ ﺑﻪ ﮐﺠﺎ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﺮﻭﺩ. ﺑﺮ ﺧﻼﻑ ﭼﻬﺎﺭﺳﺎﻟﮕﯽ ﮐﻪ ﺭﻓﺘﺎﺭﯼ ﺯﻳﺎﺩﻩ ﺍﺯ ﺣﺪ ﺍﻓﺴﺎﺭ ﮔﺴﻴﺨﺘﻪ ﺩﺍﺷﺖ، ﺩﺭ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺑﺮ ﺭﻓﺘﺎﺭ ﻭ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﻳﺶ ﮐﻨﺘﺮﻝ ﺯﻳﺎﺩﯼ ﺩﺍﺭﺩ.

ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﻪ ﺍﺯ ﺯﻧﺪﮔﯽ ﮐﺮﺩﻥ ﺩﺭ ﺍﻳﻦ ﺟﻬﺎﻥ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ. ﺍﻭ ﺗﻌﺮﻳﻒ‌ﻫﺎﻳﯽ ﺑﺴﻴﺎﺭ ﻋﻤﻠﯽ ﻣﯽ‌ﺩﻫﺪ: «ﺳﻮﺭﺍﺥ ﺑﺮﺍﯼ ﮐﻨﺪﻥ ﺍﺳﺖ، ﺑﺴﺘﻨﯽ ﺑﺮﺍﯼ ﺧﻮﺭﺩﻥ.» ﺍﻭ ﺑﺎ ﺧﻮﺩ ﻭ ﺍﻃﺮﺍﻓﻴﺎﻧﺶ ﺩﺭ ﺳﺘﻴﺰ ﻧﻴﺴﺖ. ﺍﺯ ﺧﻮﺩﺵ ﺭﺿﺎﻳﺖ ﺩﺍﺭﺩ ﻭ ﺩﻳﮕﺮﺍﻥ ﻧﻴﺰ ﺍﺯ ﺍﻭ ﺭﺍﺿﯽ ﻫﺴﺘﻨﺪ. ﺭﻭﺣﻴﻪ ﻫﻤﮑﺎﺭﯼ ﻭ ﺗﻤﺎﻳﻞ ﺑﻪ ﺗﺎﻳﻴﺪ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﻣﺎﻧﻨﺪ ﺳﻪ ﺳﺎﻟﮕﯽ ﻭﻟﯽ ﺑﺎ ﺩﺭﺟﻪ‌ﺍﯼ ﺑﺎﻻﺗﺮ، ﺩﻭﺑﺎﺭﻩ ﺩﺭ ﺍﻭ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ. ﻣﺎﺩﺭ ﻫﻨﻮﺯ ﺑﺮﺍﯼ ﺍﻭ ﻣﺮﮐﺰ ﺟﻬﺎﻥ ﺍﺳﺖ، ﺍﻭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﻧﺰﺩ ﻣﺎﺩﺭﺵ ﺑﻤﺎﻧﺪ، ﮐﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺑﺎ ﺍﻭ ﻭ ﺩﺭ ﮐﻨﺎﺭ ﺍﻭ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺍﺯ ﻣﺎﺩﺭﺵ ﺍﻃﺎﻋﺖ ﮐﻨﺪ، ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺗﺎ ﭼﻨﺪ ﻣﺎﻩ ﭘﻴﺶ ﺑﺎ ﻟﺠﺎﺟﺖ ﺍﺯ ﺍﻭﺍﻣﺮ ﺍﻭ ﺳﺮﭘﻴﭽﯽ ﻣﯽ‌ﮐﺮﺩ. ﺑﻪ ﻣﺎﻧﻨﺪ ﺳﻪ ﺳﺎﻟﮕﯽ، ﺑﺎﺯ ﻋﻼﻗﻤﻨﺪ ﻣﯽ‌ﺷﻮﺩ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻨﺪ ﭼﻪ ﮐﺎﺭﯼ ﺑﺎﻳﺪ ﺑﮑﻨﺪ ﻭ ﭼﻪ ﮐﺎﺭﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﮑﻨﺪ.

ﺑﻪ ﺭﻏﻢ ﻋﺸﻘﺶ ﺑﻪ ﻣﺎﺩﺭ ﻭ ﺧﺎﻧﻪ، ﺧﺎﻧﻮﺍﺩﻩ ﺩﻳﮕﺮ ﮐﺎﻣﻼ ﺑﺮﺍﯼ ﺍﻭ ﮐﺎﻓﯽ ﻧﻴﺴﺖ. ﺍﻭ ﭘﺨﺘﮕﯽ ﻻﺯﻡ ﺭﺍ ﺑﺮﺍﯼ ﺗﺠﺮﺑﻪ ﺟﻤﻌﯽِ ﻭﺳﻴﻊ‌ﺗﺮﯼ ﺩﺍﺭﺩ. ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺑﺎ ﺩﻭﺳﺘﺎﻥ ﻭ ﻫﻤﺴﺎﻳﻪ‌ﻫﺎﻳﺶ ﺑﺎﺯﯼ ﮐﻨﺪ. ﺑﺎ ﺑﯽ ﺻﺒﺮﯼ ﻣﻨﺘﻈﺮ «ﻣﺪﺭﺳﻪ» ﺭﻓﺘﻦ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺑﮕﺬﺍﺭﻳﺪ. ﮐﻮﺩﮐﺴﺘﺎﻥ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﻪ، ﺑﻪ ﺷﺮﻁ ﺍﻳﻨﮑﻪ ﻣﻌﻠﻢ‌ﻫﺎﯼ ﮐﺎﺭﺩﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺑﻬﺘﺮﻳﻦ ﺟﺎ ﺑﺮﺍﯼ ﺷﮑﻔﺘﮕﯽ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﻭﺳﻴﻊ ﺫﻫﻨﯽ ﺍﻭﺳﺖ. ﺍﮔﺮ ﮐﻮﺩﮎ ﺭﺍ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﻧﻤﯽ‌ﮔﺬﺍﺭﻳﺪ، ﺑﺎﻳﺪ ﺑﻪ ﺑﺎﺯﯼ‌ﻫﺎﻳﺶ ﺗﻮﺟﻪ ﺑﺴﻴﺎﺭﯼ ﺑﮑﻨﻴﺪ.

ﻛﻮﺩﻙ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺗﻌﺎﺩﻟﯽ ﺷﺎﻳﺎﻥ ﺗﻮﺟﻪ ﺩﺍﺭﺩ، ﺑﻪ ﺷﺪﺕ ﺍﺟﺘﻤﺎﻋﯽ ﻭ ﻣﺴﺘﻘﻞ ﺍﺳﺖ، ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺍﺭﺩ ﻭ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻗﻮﺍﻧﻴﻦ ﺍﺟﺘﻤﺎﻋﯽ ﺧﻴﻠﯽ ﺧﻮﺏ ﺗﻄﺒﻴﻖ ﻣﯽ‌ﺩﻫﺪ. ﭘﺎﮐﯽ ﻭ ﺻﻔﺎ ﻭ ﺟﺪﻳﺖ ﻭ ﺍﺣﺘﻴﺎﻁ، ﻣﻨﻄﻖ ﻭ ﺍﺩﺏ ﻭ ﺟﺬﺍﺑﻴﺖ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﮐﻮﺩﮎ ﭘﻨﺞ ﺳﺎﻟﻪ ﺍﺳﺖ.

ﺍﻭ ﺩﺭ ﺁﻥ ﻭﺍﺣﺪ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﺧﻮﺏ ﺷﺎﻳﺎﻥ ﺗﻮﺟﻬﯽ ﺩﺍﺭﺩ. ﺍﺯ ﺯﻧﺪﮔﯽ ﺩﺭ ﮐﻨﺎﺭ ﺍﻭ ﻭﺍﻗﻌﺎ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﻳﺪ. ﺍﺯ ﻧﻈﺮ ﺟﺴﻤﺎﻧﯽ، ﻣﻬﺎﺭﺕ ﻭ ﺗﻌﺎﺩﻝ ﻋﻀﻼﻧﯽ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺍﺯ ﻧﻈﺮ ﻋﺎﻃﻔﯽ، ﮐﺎﻣﻼ ﻣﺘﻌﺎﺩﻝ ﺍﺳﺖ. ﺍﺯ ﻧﻈﺮ ﺫﻫﻨﯽ، ﻣﻤﻠﻮ ﺍﺯ ﮐﻨﺠﮑﺎﻭﯼ ﺑﺮﺍﯼ ﻳﺎﺩﮔﻴﺮﯼ ﺍﺳﺖ. ﺩﺭ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﮐﻮﺩﮎ ﺯﻧﺪﮔﯽ ﺭﺍ ﺁﻥ‌ﻃﻮﺭ ﮐﻪ ﻫﺴﺖ ﺩﺭﻣﯽ‌ﻳﺎﺑﺪ ﻭ ﺑﻪ ﻧﻈﺮﺵ ﺯﻧﺪﮔﯽ ﺯﻳﺒﺎﺳﺖ. ﺍﺟﺎﺯﻩ ﺩﻫﻴﻢ ﮐﻪ ﮐﻮﺩﮐﯽ ﭘﻨﺞ ﺳﺎﻟﻪ، ﺧﻮﺩ ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﺳﺨﻦ ﮔﻮﻳﺪ. ﻭﻗﺘﯽ ﺍﺯ ﺍﻭ ﭘﺮﺳﻴﺪﻧﺪ ﮐﻪ «ﭼﻪ ﭼﻴﺰﯼ ﺭﺍ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﯽ؟» ﺟﻮﺍﺏ ﺩﺍﺩ: «ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺭﺍ».
 

ﻫﻤﭽﻨﺎﻥ ﮐﻪ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ، ﺍﺧﺘﻼﻓﺎﺕ ﻓﺎﺣﺸﯽ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ، ﺑﻴﻦ ﮐﻮﺩﮐﺎﻥ ﺳﻪ، ﭼﻬﺎﺭ ﻭ ﭘﻨﺞ ﺳﺎﻟﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﮐﺴﯽ ﮐﻪ ﺑﻪ ﺗﺮﺑﻴﺖ ﺁﻧﻬﺎ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ ﺑﺎﻳﺪ ﺍﻳﻦ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﺯ ﻳﮏ ﮐﻮﺩﮎ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺍﻧﺘﻈﺎﺭ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻣﺎﻧﻨﺪ ﮐﻮﺩﮐﯽ ﭘﻨﺞ ﺳﺎﻟﻪ ﺍﺯ ﺷﻤﺎ ﺍﻃﺎﻋﺖ ﮐﻨﺪ.
 
ﺍﻣﺎ ﺑﻪ ﺭﻏﻢ ﺍﻳﻦ ﺍﺧﺘﻼﻓﺎﺕ، ﺍﻳﻦ ﺳﻪ ﺳﺎﻝ ﻣﺸﺘﺮﮐﺎﺕ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺳﺖ ﮐﻪ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﻳﮏ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪ ﻃﻔﻞ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﻴﻢ.

ﺑﺮﺍﯼ ﻣﺮﺑﯽ ﺩﺭ ﺗﻤﺎﻡ ﺍﻳﻦ ﺩﻭﺭﻩ ﺳﻪ ﺳﺎﻟﻪ ﻭﻇﺎﻳﻒ ﻋﺎﺟﻠﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺩﺭﺳﺖ ﻣﺜﻞ ﺿﺮﺏ ﻳﮏ ﻗﻄﻌﻪ ﻣﻮﺳﻴﻘﯽ، ﺿﺮﺏ ﺍﻳﻦ ﺳﻪ ﺳﺎﻝ ﺗﻐﻴﻴﺮ ﻣﯽ‌ﮐﻨﺪ، ﻭﻟﯽ ﻣﺎﻳﻪ ﺍﺻﻠﯽ ﻫﻤﻴﺸﻪ ﻳﮑﯽ ﺍﺳﺖ. ﺣﺎﻝ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﺍﻳﻦ ﻭﻇﺎﻳﻒ ﻓﻮﺭﯼ ﺩﺭ ﺩﻭﺭﺍﻥ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﭼﻴﺴﺖ؟ ﮐﻮﺩﮎ ﺩﺭ ﻃﻮﻝ ﺍﻳﻦ ﺳﻪ ﺳﺎﻝ، ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺑﺎﻳﺪ ﺑﻴﺎﻣﻮﺯﺩ ﺗﺎ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻭ ﺳﺎﺧﺘﺎﺭ ﺍﺳﺎﺳﯽ ﺷﺨﺼﻴﺘﺶ ﺭﺍ ﮐﺎﻣﻞ ﮐﻨﺪ؟ ﺍﻳﻦ ﻭﻇﺎﻳﻒ ﺭﺍ ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﺑﺮﺭﺳﯽ ﺧﻮﺍﻫﻴﻢ ﻛﺮﺩ.
