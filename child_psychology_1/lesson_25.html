ﺩﺭﺱ ﺑﻴﺴﺘﻢ- ﺟﺪﺍ ﺷﺪﻥ ﺍﺯ ﻣﺎﺩﺭ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺩﻳﺪﻳﻢ ﮐﻪ ﮐﻮﺩﮎ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﭘﻨﺞ ﺳﺎﻝ ﺑﺎ ﺟﺬﺏ ﻣﻮﺍﺩ ﻏﺬﺍﻳﯽ ﻣﻘﺪﺍﺭ ﺯﻳﺎﺩﯼ ﺍﻧﺮﮊﯼ ﺗﻮﻟﻴﺪ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺁﻥ ﺭﺍ ﺻﺮﻑ ﺩﻭﻳﺪﻥ، ﺭﺍﻩ ﺭﻓﺘﻦ، ﻓﺮﻳﺎﺩ ﮐﺸﻴﺪﻥ، ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﭘﺮﻳﺪﻥ ﻭ ﻏﻴﺮﻩ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﺭﺍﻩ ﺗﻌﺎﺩﻝ ﻻﺯﻡ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺣﺮﮐﺎﺗﺶ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺁﻭﺭﺩ. ﻫﻤﻴﻦ ﻃﻮﺭ ﺩﻳﺪﻳﻢ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺳﺖ ﮐﻪ ﮐﻨﺘﺮﻝ ﺣﺮﮐﺎﺕ ﺧﻮﺩﺟﻮﺵ ﻭ ﻏﺮﻳﺰﯼ‌ﺍﺵ ﺭﺍ ﻓﺮﺍ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﺩﻳﺪﻳﻢ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﻧﺒﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﺁﻣﻮﺯﺵ ﮐﻮﺩﮎ ﺭﺍ ﺗﺤﺖ ﻓﺸﺎﺭ ﻗﺮﺍﺭ ﺩﻫﻨﺪ. ﺍﮐﻨﻮﻥ ﺟﺪﺍﻳﯽ ﮐﻮﺩﮎ ﺍﺯ ﻣﺎﺩﺭﺵ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ.

ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺭ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ، ﺟﺪﺍﺷﺪﻥ ﺍﺯ ﻣﺎﺩﺭ ﺭﺍ ﻓﺮﺍ ﻣﯽ­‌ﮔﻴﺮﺩ. ﺩﺭ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﺍﻭ ﻫﻨﻮﺯ ﮐﺎﻣﻼ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺁﻣﺎﺩﮔﯽ ﻧﺪﺍﺭﺩ. ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻟﻪ ﻣﺎﺩﺭ ﻣﺮﮐﺰ ﺟﻬﺎﻥ ﺍﺳﺖ ﻭ ﺑﻪ ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﺩ. ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺯﻳﺎﺩ ﺧﻮﺏ ﻧﻴﺴﺖ ﮐﻪ ﮐﻮﺩﮎ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﺭﺍ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﻓﺮﺳﺘﺎﺩ. ﻣﻌﻤﻮﻻ، ﺍﻭ ﺣﺎﺿﺮ ﻧﻴﺴﺖ ﻣﺎﺩﺭﺵ ﺭﺍ ﺳﻪ ﺳﺎﻋﺖ ﺗﺮﮎ ﮐﻨﺪ ﺗﺎ ﺑﺮﻭﺩ ﻭ ﺯﻳﺮ ﻧﻈﺮ ﻳﮏ ﻏﺮﻳﺒﻪ (ﻣﻌﻤﻠﺶ) ﺑﺎ ﺑﭽﻪ­‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺑﺎﺯﯼ ﮐﻨﺪ.

ﻣﺴﻠﻤﺎ ﺍﮔﺮ ﻣﺎﺩﺭﯼ ﺑﻪ ﺧﺎﻃﺮ ﻣﺴﺎﺋﻞ ﻣﺎﻟﯽ، ﻣﺠﺒﻮﺭ ﺑﺎﺷﺪ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﺑﻪ ﻣﻬﺪ ﮐﻮﺩﮎ ﺑﺴﭙﺎﺭﺩ ﺩﻳﮕﺮ ﭼﺎﺭﻩ‌­ﺍﯼ ﻧﻴﺴﺖ. ﻭﻟﯽ ﺍﻳﺪﻩ‌­ﺁﻝ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺳﻪ ﺳﺎﻟﮕﯽ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﻧﺮﻭﺩ.

ﺩﺭ ﺍﻳﻦ ﺳﻦ ﮐﻮﺩﮎ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻫﻤﺒﺎﺯﯼ ﺩﺍﺭﺩ. ﺍﻭ ﺩﻟﺶ ﻣﯽ­‌ﺧﻮﺍﻫﺪ ﺍﺳﺘﻘﻼﻝ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺍﺯ ﻣﺎﺩﺭﺵ ﺟﺪﺍ ﺷﻮﺩ. ﺑﻬﺘﺮﻳﻦ ﻛﺎﺭ ﻓﺮﺳﺘﺎﺩﻧﺶ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻧﯽ ﺧﻮﺏ ﺍﺳﺖ. ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺍﺯ ﻣﺎﺩﺭﺵ ﺟﺪﺍ ﺷﻮﺩ ﻭ ﺑﻪ ﻫﻤﺒﺎﺯﯼ‌­ﻫﺎﻳﺶ ﺑﭙﻴﻮﻧﺪﺩ، ﺑﻪ ﻫﻨﮕﺎﻡ ﺟﺪﺍﻳﯽ ﺍﺯ ﻣﺎﺩﺭﺵ ﺩﭼﺎﺭ ﺩﻭﮔﺎﻧﮕﯽ ﻣﯽ‌­ﺷﻮﺩ. ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ «ﻧﮕﺮﺍﻥ ﺍﻳﻦ ﺟﺪﺍﺷﺪﻥ» ﺑﺎﺷﺪ- ﮐﻪ ﺩﺭ ﺑﺮﺧﯽ ﺍﻳﻦ ﻧﮕﺮﺍﻧﯽ ﺑﻴﺸﺘﺮ ﺍﺯ ﺑﺮﺧﯽ ﺩﻳﮕﺮ ﺍﺳﺖ- ﭼﺮﺍ ﮐﻪ ﻣﺎﺩﺭﺵ ﺑﻪ ﻣﺪﺕ ﺳﻪ ﺳﺎﻝ ﺗﻤﺎﻡ «ﻧﻘﻄﻪ ﺍﺗﮑﺎﯼ» ﺍﻭ ﺑﻮﺩﻩ ﺍﺳﺖ.

ﻫﻤﻪ ﺍﻃﻔﺎﻝ، ﺭﻭﯼ ﻫﻤﻪ ﺗﺄﮐﻴﺪ ﻣﯽ‌ﮐﻨﻢ، ﮐﻤﺎ ﺑﻴﺶ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺭﺍ ﺩﺍﺭﻧﺪ. ﺍﮔﺮ ﮐﻮﺩﮎ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻣﺘﻌﺎﺩﻝ ﺑﺎﺷﺪ، ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺑﺴﻴﺎﺭ ﻣﻮﻗﺘﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺳﭙﺲ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻗﺎﻃﯽ ﻣﯽ­‌ﺷﻮﺩ ﻭ ﺑﺎ ﺁﻥ­ﻫﺎ ﺑﺎﺯﯼ ﻣﯽ­‌ﮐﻨﺪ ﻭﻟﯽ ﺣﺘﯽ ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻮﺩﮎ ﺑﺪﻭﻥ ﮔﺮﻳﻪ ﻭ ﺍﺩﺍ ﺩﺭ­­ﺁﻭﺭﺩﻥ ﺑﻪ ﺑﺎﺯﯼ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺑﭙﺮﺩﺍﺯﺩ، ﺯﻳﺎﺩ ﻣﺎﻳﻞ ﻧﻴﺴﺖ ﮐﻪ ﻣﺎﺩﺭﺵ ﺍﺯ ﺍﻭ ﺩﻭﺭ ﺷﻮﺩ ﻭ ﺍﻭ ﺭﺍ ﺩﺭ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺗﻨﻬﺎ ﺑﮕﺬﺍﺭﺩ.

ﮐﻮﺩﮎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺣﺴﺎﺳﯽ ﺭﺍ ﮐﻪ ﺍﺯ ﺟﺪﺍﻳﯽ ﻣﺎﺩﺭﺵ ﻣﯽ­‌ﮐﻨﺪ ﺑﺎ ﮐﻠﻤﺎﺕ ﺑﻴﺎﻥ ﻧﮑﻨﺪ. ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎ ﻋﻼﻳﻢ ﺟﺴﻤﺎﻧﯽ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﻨﺪ. ﺻﺒﺢ ﻭﻗﺖ ﺭﻓﺘﻦ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﮐﻪ ﻣﯽ‌ﺭﺳﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﭘﺎ ﺩﺭﺩ ﺷﺪﻳﺪ ﺑﮕﻴﺮﺩ، ﻳﺎ ﺣﺎﻟﺖ ﺗﻬﻮﻉ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﻳﺎ ﺣﺘﯽ ﺍﺳﺘﻔﺮﺍﻍ ﮐﻨﺪ.

ﻋﮑﺲ‌­ﺍﻟﻌﻤﻞ ﮐﻮﺩﮎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎ ﺗﺄﺧﻴﺮ ﺍﻧﺠﺎﻡ ﺷﻮﺩ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﮐﻪ ﻫﻔﺘﻪ ﺍﻭﻝ ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﺧﻮﺑﯽ ﭘﻴﺶ ﻣﯽ­‌ﺭﻭﺩ. ﺑﻪ ﻧﻈﺮ ﻣﯽ‌­ﺭﺳﺪ ﻛﻪ ﺑﺪﻭﻥ ﻫﻴﭻ ﺍﻋﺘﺮﺍﺿﯽ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﻣﯽ­‌ﺭﻭﺩ. ﺍﻭ ﻧﮕﺮﺍﻧﯽ ﺧﻮﺩ ﺭﺍ ﺑﺪﻭﻥ ﺍﻳﻦ ﮐﻪ ﻣﺎ ﻣﺘﻮﺟﻪ ﺁﻥ ﺷﻮﻳﻢ ﭘﻨﻬﺎﻥ ﻣﯽ‌­ﮐﻨﺪ. ﻭﻟﯽ ﺍﻳﻦ ﻧﮕﺮﺍﻧﯽ ﺑﺎﻻﺧﺮﻩ ﺑﻌﺪ ﺍﺯ ﺩﻭ ﻳﺎ ﺳﻪ ﻫﻔﺘﻪ ﺧﻮﺩ ﺭﺍ ﻇﺎﻫﺮ ﻣﯽ­‌ﮐﻨﺪ ﻭ ﮐﻮﺩﮎ ﺑﺎ ﺗﻌﺠﺐ ﻓﺮﺍﻭﺍﻥ ﻣﺎﺩﺭﺵ ﺍﻋﻼﻡ ﻣﯽ­‌ﮐﻨﺪ ﮐﻪ ﺍﺯ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺭﻓﺘﻦ ﻣﯽ‌­ﺗﺮﺳﺪ، ﻣﺎﺩﺭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﻄﻮﺭ ﭼﻨﻴﻦ ﭼﻴﺰﯼ ﻣﻤﮑﻦ ﺍﺳﺖ، ﺍﻭ ﮐﻪ ﻫﻴﭻ ﻣﺴﺌﻠﻪ­‌ﺍﯼ ﻧﺪﺍﺷﺖ! ﻭﻟﯽ ﭼﺮﺍ، ﺍﻭ ﻣﯽ­‌ﺗﺮﺳﻴﺪ. ﻣﺎﺩﺭﺵ ﻣﺘﻮﺟﻪ ﻧﺸﺪﻩ ﺑﻮﺩ، ﺯﻳﺮﺍ ﮐﻮﺩﮎ ﺁﻥ ﺭﺍ ﺑﺮﻭﺯ ﻧﻤﯽ‌­ﺩﺍﺩ. ﺩﺭ ﺩﺭﺍﺯ ﻣﺪﺕ ﺩﻳﮕﺮ ﻗﺎﺩﺭ ﺑﻪ ﭘﻨﻬﺎﻥ ﮐﺮﺩﻥ ﺁﻥ ﻧﺒﻮﺩﻩ ﺑﺎﻻﺧﺮﻩ ﺍﻳﻦ ﺗﺮﺱ ﻭ ﻧﮕﺮﺍﻧﯽ ﺧﻮﺩ ﺭﺍ ﺑﺮﻭﺯ ﺩﺍﺩﻩ ﺍﺳﺖ.

ﺍﮔﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﺎ ﻳﮑﯽ ﺍﺯ ﻫﻤﺒﺎﺯﯼ‌ﻫﺎﻳﺶ ﺗﻨﻬﺎ ﺑﮕﺬﺍﺭﻳﺪ ﺗﺎ ﺑﺎﺯﯼ ﮐﻨﻨﺪ، ﮐﻤﮏ ﺑﺴﻴﺎﺭﯼ ﺑﺮﺍﯼ ﺟﺪﺍ ﺷﺪﻥ ﺍﺯ ﺷﻤﺎ ﺑﻪ ﻫﻨﮕﺎﻡ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺭﻓﺘﻦ ﻣﯽ­‌ﮐﻨﺪ. ﺍﮔﺮ ﻧﻤﯽ­‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺑﻔﺮﺳﺘﻴﺪ، ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺭﻭﻧﺪ ﺟﺪﺍﻳﯽ ﺍﺯ ﺷﻤﺎ ﺭﺍ ﭘﺸﺖ ﺳﺮﮔﺬﺍﺭﺩ، ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺑﻪ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺩﺭ ﮔﺮﻭﻩ ﺑﭽﻪ­­‌ﻫﺎﯼ ﻫﻤﺴﺎﻳﻪ ﺗﺸﻮﻳﻖ ﮐﻨﻴﺪ، ﮐﻪ ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﮐﻮﺩﮎ ﺧﺠﻮﻟﯽ ﺑﺎﺷﺪ ﮐﺎﺭ ﭼﻨﺪﺍﻥ ﺁﺳﺎﻧﯽ ﻧﻴﺴﺖ، ﺯﻳﺮﺍ ﻭﻗﺘﯽ ﮐﻪ ﮐﻮﺩﮎ ﮐﻮﺩﮐﺴﺘﺎﻥ ﻣﯽ‌ﺭﻭﺩ، ﻣﻌﻠﻢ ﮐﻤﮏ ﺷﺎﻳﺎﻧﯽ ﺑﻪ ﺍﻭ ﺩﺭ ﭘﺸﺖ ﺳﺮﮔﺬﺍﺷﺘﻦ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺟﺪﺍﻳﯽ ﻣﯽ­‌ﮐﻨﺪ. ﺍﻣﺎ ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﺷﻤﺎ ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻳﻦ ﺟﺪﺍﻳﯽ ﻧﺎﺋﻞ ﺷﻮﻳﺪ.


ﺩﻧﻴﺎﯼ ﺟﺪﻳﺪ ﮐﻮﺩﮐﺎﻥ ﺩﻳﮕﺮ
ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺭ ﺣﺎﻝ ﻳﺎﺩﮔﻴﺮﯼ ﺭﺍﺑﻄﻪ«ﺑﺪﻩ ﻭ ﺑﺴﺘﺎﻥ» ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺖ. ﺍﻳﻦ ﺩﻳﮕﺮﺍﻥ، ﺍﻓﺮﺍﺩ ﺷﺒﻴﻪ ﺑﻪ ﺧﻮﺩ ﺍﻭ ﻫﺴﺘﻨﺪ. ﺩﻧﻴﺎﯼ ﺭﻭﺍﺑﻂ ﺑﺮﺍﺑﺮ، ﺑﻪ ﻃﺮﺯ ﻭﺣﺸﺘﻨﺎﮐﯽ ﺑﺎ ﺩﻧﻴﺎﯼ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺗﻔﺎﻭﺕ ﺩﺍﺭﺩ. ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺭﻭﺯﯼ ﺩﺭ ﮐﻮﺩﮐﺴﺘﺎﻧﯽ، ﭘﺴﺮ ﺑﭽﻪ‌­ﺍﯼ ﮐﺎﻣﻴﻮﻥ ﮐﻮﭼﮑﯽ ﺭﺍ، ﮐﻪ ﭘﺴﺮ ﮐﻮﭼﮏ ﺩﻳﮕﺮﯼ ﺩﺭ ﺣﺎﻝ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺑﺎ ﺁﻥ ﺑﻮﺩ، ﻣﯽ‌ﺧﻮﺍﺳﺖ.

ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ، ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ  ﺑﺎ ﺩﺷﻮﺍﺭﯼ ﺩﺭ ﻣﯽ­‌ﻳﺎﺑﺪ ﮐﻪ ﺩﻧﻴﺎﯼ ﺭﻭﺍﺑﻂ ﺑﺎ ﺍﻓﺮﺍﺩ ﻣﺸﺎﺑﻪ ﺧﻮﺩﺵ، ﺩﻧﻴﺎﯼ ﺟﺪﻳﺪﯼ ﺍﺳﺖ ﮐﻪ ﻗﻮﺍﻧﻴﻦ ﻭ ﺍﻧﺘﻈﺎﺭﺍﺕ ﺩﻳﮕﺮﯼ ﺩﺭ ﺁﻥ ﺣﮑﻢ ﻓﺮﻣﺎﺳﺖ. ﻫﻢ ﻧﻘﺎﻁ ﺿﻌﻒ ﻭ ﻫﻢ ﻧﻘﺎﻁ ﻗﻮﺕ ﺷﺨﺼﻴﺖ ﮐﻮﺩﮎ ﺩﺭ ﺍﻭﻟﻴﻦ ﺗﺠﺮﺑﻪ ﮔﺮﻭﻫﻴﺶ ﺑﺮﻭﺯ ﻣﯽ­‌ﮐﻨﺪ. ﺍﮔﺮ ﺩﻳﮕﺮﺍﻥ ﺍﻭ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﻨﺪ ﺧﻮﺷﻨﻮﺩ ﻣﯽ‌­ﺷﻮﺩ ﻭ ﺍﮔﺮ ﺍﻭ ﺭﺍ ﻧﭙﺬﻳﺮﻧﺪ ﺍﺯ ﺁﻥ ﺭﻧﺞ ﻣﯽ‌ﺑﺮﺩ. ﺍﻭ ﺑﺪﻩ ﺑﺴﺘﺎﻥ ﺭﺍ ﻳﺎﺩ ﻣﯽ‌­ﮔﻴﺮﺩ.

ﮐﻮﺩﮎ ﺩﺭ ﻣﻴﺎﻥ ﻫﻤﺴﺎﻧﺎﻥ ﺧﻮﺩ، ﻋﺎﺩﺍﺗﯽ ﺍﺟﺘﻤﺎﻋﯽ ﮐﺴﺐ ﻣﯽ­‌ﮐﻨﺪ، ﺑﺎﻳﺪ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﻢ ﮐﻪ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺷﺮﻳﮏ ﺷﻮﺩ، ﻣﻨﺘﻈﺮ ﻧﻮﺑﺘﺶ ﺷﻮﺩ، ﭼﮕﻮﻧﻪ ﭼﻴﺰﯼ ﺭﺍ ﺍﺯ ﮐﺴﯽ ﺑﺨﻮﺍﻫﺪ ﻭ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ ﺑﺎ ﮐﻠﻤﺎﺕ ﺑﻴﺎﻥ ﮐﻨﺪ. ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻣﺒﺎﺭﺯﻩ ﺑﺮﺍﯼ ﺩﻓﺎﻉ ﺍﺯ ﺣﻘﻮﻕ ﺧﻮﺩ ﺩﺍﺭﺩ. ﮐﻮﺩﮎ ﺑﺎﻳﺪ ﻳﺎﺩ ﺑﮕﻴﺮﺩ ﺍﻓﮑﺎﺭﺵ ﺭﺍ ﺑﺪﻭﻥ ﻣﺸﺖ ﻭ ﻟﮕﺪ ﺑﻴﺎﻥ ﮐﻨﺪ. ﺍﻭ ﻣﺤﺘﺎﺝ ﺁﻣﻮﺧﺘﻦ ﺷﺮﺍﮐﺖ ﻭ ﻧﻈﺎﺭﻩ ﮐﺮﺩﻥ ﺍﺳﺖ ﻭ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﺪ ﺧﻮﺩ­ﺍﻧﮕﺎﺭﻩ‌­ﺍﺵ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺭﻭﺍﺑﻄﯽ ﮐﻪ ﺑﺎ ﮐﻮﺩﮐﺎﻥ ﺩﻳﮕﺮ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌­ﮐﻨﺪ، ﺭﺷﺪ ﺩﻫﺪ. ﻫﻴﭻ ﮐﻮﺩﮐﯽ ﺑﺎ ﺍﻳﻦ ﺧﺼﻠﺖ‌­ﻫﺎﯼ ﺑﻨﻴﺎﺩﻳﻦ ﺩﺭ ﺯﻣﻴﻨﻪ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﻭ ﻋﺎﻃﻔﯽ ﻣﺘﻮﻟﺪ ﻧﻤﯽ­‌ﺷﻮﺩ.

ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺭ ﺳﻨﻴﻦ ﺳﻪ، ﭼﻬﺎﺭ ﻭ ﭘﻨﭻ ﺳﺎﻟﮕﯽ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺑﺎ ﮐﻮﺩﮐﺎﻥ ﺩﻳﮕﺮ ﺩﺍﺭﺩ ﺗﺎ ﺑﻪ ﺍﻳﻦ ﻃﺮﻳﻖ ﻋﺎﺩﺍﺕ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﻓﺮﺍ ﮔﻴﺮﺩ. ﮐﻮﺩﮐﺴﺘﺎﻥ ﺑﻬﺘﺮﻳﻦ ﻣﺤﻞ ﺑﺮﺍﯼ ﺍﻳﻦ ﻳﺎﺩﮔﻴﺮﯼ ﺍﺳﺖ ﭼﺮﺍ ﮐﻪ ﻣﻌﻠﻤﯽ ﺑﺎ ﺗﺠﺮﺑﻪ ﺍﻳﻦ ﺭﻭﺍﺑﻂ ﺭﺍ ﺗﺤﺖ ﻧﻈﺮ ﺩﺍﺭﺩ.

ﻳﺎﺩﮔﻴﺮﯼ ﺍﺑﺮﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ
ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ ﻳﺎﺩ ﻣﯽ­‌ﮔﻴﺮﺩ ﮐﻪ ﻳﺎ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ ﺑﻴﺎﻥ ﮐﻨﺪ ﻭ ﻳﺎ ﺑﺮﻋﮑﺲ ﺁﻧﻬﺎ ﺭﺍ ﺳﺮﮐﻮﺏ ﮐﻨﺪ. ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﭘﻴﺶ ﺑﻪ ﺍﻫﻤﻴﺖ ﻓﺮﺍﻭﺍﻥ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺕ ﮐﻮﺩﮎ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ ﻭ ﮔﻔﺘﻴﻢ ﮐﻪ ﺳﺮﮐﻮﺏ ﺍﺣﺴﺎﺳﺎﺕ ﻣﯽ‌­ﺗﻮﺍﻧﺪ ﺑﻪ ﻃﺮﺯﯼ ﺧﻄﺮﻧﺎﮎ ﻭ ﺗﻐﻴﻴﺮ ﺷﮑﻞ ﻳﺎﻓﺘﻪ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﺪ. ﺩﺭ ﺍﻳﻦ ﺳﻨﻴﻦ ﮐﻮﺩﮎ ﺑﻪ ﭘﺎﻳﻪ­‌ﺭﻳﺰﺭﯼ ﺭﻭﺵ ﺑﺮﺧﻮﺭﺩ ﺍﺣﺴﺎﺳﯽ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺣﺴﺎﺳﺎﺕ ﺩﺭﻭﻧﻴﺶ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ. ﺍﻭ ﻳﺎ ﻓﮑﺮ ﻣﯽ‌­ﮐﻨﺪ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺗﺄﺛﺮﺍﺕ ﺧﻄﺮﻧﺎﮐﻨﺪ ﻭ ﺑﻬﺘﺮ ﺍﺳﺖ ﺁﻧﻬﺎ ﺭﺍ ﺳﺮﮐﻮﺏ ﮐﺮﺩ ﻭ ﻳﺎ ﻳﺎﺩ ﻣﯽ‌­ﮔﻴﺮﺩ ﮐﻪ ﺑﺎ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﭼﻪ ﻣﺜﺒﺖ ﭼﻪ ﻣﻨﻔﯽ، ﺭﺍﺣﺖ ﺑﺎﺷﺪ. ﺑﻪ ﻛﺎﺭ ﻧﺒﺮﺩﻥ «ﺑﺪ» ﻭ «ﺧﻮﺏ» ﺩﺭ ﻣﻮﺭﺩ ﺍﺣﺴﺎﺳﺎﺕ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ. ﮐﻮﺩﮐﺎﻥ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﺑﺪ ﻭ ﺧﻮﺏ ﺭﺍ ﻓﻘﻂ ﺑﺮﺍﯼ ﺍﻋﻤﺎﻝ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺣﻔﻆ ﮐﻨﻨﺪ.

ﻣﺜﺎﻟﯽ ﺑﻴﺎﻭﺭﻳﻢ. ﻣﺎﺩﺭﯼ ﻓﺮﺯﻧﺪ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﻪ­‌ﺍﺵ ﺭﺍ ﺑﺎ ﺧﻮﺩﺵ ﺑﺮﺍﯼ ﺧﺮﻳﺪ ﻣﯽ­‌ﺑﺮﺩ. ﻣﺎﺩﺭ ﺧﺮﻳﺪﻫﺎﻳﺶ ﺭﺍ ﺩﺭ ﭼﺮﺥ ﺧﺮﻳﺪ ﮔﺬﺍﺷﺘﻪ ﻭ ﻓﺮﺯﻧﺪﺵ ﻧﻴﺰ ﺩﺭ ﻫﻤﺎﻥ ﭼﺮﺥ ﺧﺮﻳﺪ ﺍﺳﺖ. ﺧﺎﻧﻢ ﻣﺴﻨﯽ ﺳﺮ ﻣﯽ‌­ﺭﺳﺪ ﻭ ﺑﺎ ﻟﺒﺨﻨﺪﯼ ﺑﻪ ﺩﺧﺘﺮ ﮐﻮﭼﮏ ﻣﯽ‌­ﮔﻮﻳﺪ«ﻭﺍﯼ، ﭼﻪ ﺑﭽﻪ ﻧﺎﺯﯼ». ﺩﺧﺘﺮ ﮐﻮﭼﻮﻟﻮ ﺳﺮﺵ ﺭﺍ ﺑﺎﻻ ﻣﯽ­‌ﮐﻨﺪ ﻭ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﻓﺮﻳﺎﺩ ﻣﯽ­‌ﮐﺸﺪ:«ﺑﺮﻭ! ﺗﻮﺭﻭ ﺩﻭﺳﺖ ﻧﺪﺍﺭﻡ!»

ﺍﮔﺮ ﺷﻤﺎ ﺟﺎﯼ ﺁﻥ ﻣﺎﺩﺭ ﺑﻮﺩﻳﺪ ﭼﻪ ﻋﮑﺲ­‌ﺍﻟﻌﻤﻠﯽ ﺩﺍﺷﺘﻴﺪ؟ ﻣﺘﺄﺳﻔﺎﻧﻪ ﻓﮑﺮ ﻣﯽ‌­ﮐﻨﻢ ﮐﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﭼﻴﺰﯼ ﺷﺒﻴﻪ ﺟﻤﻠﻪ ﺯﻳﺮ ﺭﺍ ﻣﯽ­‌ﮔﻮﻳﻨﺪ:«ﺁﺩﻡ ﺑﺎ ﺍﻳﻦ ﺧﺎﻧﻢ ﻣﻬﺮﺑﻮﻥ ﺍﻳﻦ ﻃﻮﺭﯼ ﺣﺮﻑ ﻧﻤﯽ­‌ﺯﻧﻪ، ﻣﻌﺬﺭﺕ ﺑﺨﻮﺍﻩ.» ﻭ ﻳﺎ ﺍﺯ ﺁﻥ ﺧﺎﻧﻢ ﻣﻌﺬﺭﺕ ﻣﯽ­‌ﺧﻮﺍﻫﻴﺪ ﻭ ﻣﯽ­‌ﮔﻮﻳﻴﺪ:«ﺍﻭ ﻓﻘﻂ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﺶ ﺍﺳﺖ ﻭ ﻫﻨﻮﺯ ﺧﻮﺏ ﻧﻤﯽ‌ﻓﻬﻤﺪ.» ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﺩﻳﮕﺮ، ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺑﮕﻮﻳﻨﺪ ﮐﻪ ﺍﺯ ﺍﻳﻨﮑﻪ ﺁﻥ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﺍﺣﺴﺎﺱ ﻣﯽ‌­ﮐﺮﺩ ﺑﻴﺎﻥ ﮐﺮﺩﻩ، ﮐﺎﺭ ﺯﺷﺘﯽ ﮐﺮﺩﻩ ﺍﺳﺖ.

ﻣﻦ ﻓﮑﺮ ﻣﯽ‌­ﮐﻨﻢ ﮐﻪ ﺁﻧﻬﺎ ﮐﺎﺭ ﺧﻮﺑﯽ ­ﻧﻤﯽ‌­ﮐﻨﻨﺪ. ﻓﮑﺮ ﻣﯽ‌­ﮐﻨﻢ ﮐﻪ ﺑﭽﻪ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺣﻖ ﺩﺍﺭﻧﺪ ﻫﺮﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﺍﺣﺴﺎﺱ ﻣﯽ­‌ﮐﻨﻨﺪ ﺑﻴﺎﻥ ﮐﻨﻨﺪ. ﺑﺮﺍﯼ ﺭﺷﺪ ﺷﺨﺼﻴﺖ، ﻓﯽ‌­ﺍﻟﺒﺪﺍﻫﮕﯽ ﻭ ﺣﻖ ﻗﺎﻧﻮﻧﯽِ ﺑﭽﮕﯽ‌ﺷﺎﻥ ﻣﻀﺮ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺁﻧﻬﺎ ﭘﻨﻬﺎﻥ ﮐﺮﺩﻥ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺍﻭﻟﻴﻦ ﺳﻨﻴﻦ ﺯﻧﺪﮔﯽ ﺑﻴﺎﻣﻮﺯﻳﻢ.
ﺍﻟﺒﺘﻪ ﻣﻦ ﻣﺪﺍﻓﻊ ﺍﻳﻦ ﻧﻴﺴﺘﻢ ﮐﻪ ﮐﻮﺩﮐﺎﻥ ﺩﺭ ﻫﺮ ﺳﻨﯽ ﻭ ﻫﺮ ﻭﻗﺘﯽ ﻭ ﻫﺮ ﺟﺎﺋﯽ ﻫﺮﭼﻪ ﺩﻟﺸﺎﻥ ﺧﻮﺍﺳﺖ ﺑﮕﻮﻳﻨﺪ. ﺁﻧﻬﺎ ﺑﺎﻳﺪ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺩﻳﮕﺮﺍﻥ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﻧﺪ ﻭ ﻣﻮﺩﺏ ﺑﺎﺷﻨﺪ. ﻭﻟﯽ ﺩﺭ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﻧﻤﯽ‌­ﺗﻮﺍﻧﻨﺪ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺑﻪ ﻗﺮﻳﺤﻪ ﻭ ﻓﯽ‌­ﺍﻟﺒﺪﺍﻫﮕﯽ­‌ﺷﺎﻥ ﺻﺪﻣﻪ‌­ﺍﯼ ﺑﺨﻮﺭﺩ، ﺑﺎ ﻇﺮﺍﻓﺖ ﺭﻓﺘﺎﺭﮐﺮﺩﻥ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻓﺮﺍ ﮔﻴﺮﻧﺪ. ﺁﻥ ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﻪ ﻫﻨﻮﺯ ﻳﺎﺩ ﻧﮕﺮﻓﺘﻪ ﺑﻮﺩ ﮐﻪ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺕ ﺁﻥ ﺧﺎﻧﻢ ﺑﻪ ﻃﺮﺯ ﻣﻮﺩﺑﺎﻧﻪ‌­ﺍﯼ ﺑﺮﺧﻮﺭﺩ ﮐﻨﺪ.
ﺩﺭ ﺣﺪﻭﺩ ﺳﻦ ﺷﺶ ﺳﺎﻟﮕﯽ ﺑﺎﻳﺪ ﺑﻪ ﮐﻮﺩﮎ ﺁﻣﻮﺧﺖ ﮐﻪ ﻣﻼﺣﻈﻪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﮑﻨﺪ؛ ﺩﺭ ﺍﻳﻦ ﺯﻣﺎﻥ ﻓﺮﺻﺖ ﮐﺎﻓﯽ ﺩﺍﺭﻳﺪ ﺗﺎ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ ﮐﻪ ﺩﺭ ﺑﻌﻀﯽ ﺷﺮﺍﻳﻂ ﻭ ﺍﻣﺎﮐﻦ ﺍﻧﺴﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺁﻥ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺍﺣﺴﺎﺱ ﻣﯽ­‌ﮐﻨﺪ ﺑﻴﺎﻥ ﮐﻨﺪ ﻭﻟﯽ ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﭘﻴﺶ ﺁﻣﺪﻥ ﺩﺭﺩﺳﺮ، ﺁﻥ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﭘﻴﺶ ﺧﻮﺩﺵ ﻧﮕﺎﻩ ﺩﺍﺭﺩ. ﻭﻗﺘﯽ ﮐﻮﺩﮎ ﺑﻪ ﺩﺑﺴﺘﺎﻥ ﻣﯽ‌­ﺭﻭﺩ، ﻣﯽ‌­ﺗﻮﺍﻥ ﺑﻪ ﺍﻭ ﺁﻣﻮﺧﺖ ﮐﻪ ﭼﮕﻮﻧﻪ ﺷﺮﺍﻳﻄﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﯽ­‌ﺗﻮﺍﻥ ﺑﻌﻀﯽ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﺁﺯﺍﺩﺍﻧﻪ ﺑﻴﺎﻥ ﮐﺮﺩ ﻭ ﺍﺯ ﺷﺮﺍﻳﻄﯽ ﮐﻪ ﺑﻬﺘﺮ ﺍﺳﺖ ﺁﻧﻬﺎ ﺭﺍ ﺑﻴﺎﻥ ﻧﮑﺮﺩ، ﺗﺸﺨﻴﺺ ﺩﻫﺪ. ﻭﻟﯽ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺭ ﺩﺑﺴﺘﺎﻥ، ﺯﻣﺎﻧﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺍﻭ ﺭﺍ ﺑﻪ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺗﺸﻮﻳﻖ ﮐﺮﺩ.
