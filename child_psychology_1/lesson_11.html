ﺩﺭﺱ ﻧﻬﻢ- ﺗﻐﺬﻳﻪ

ﻏﺬﺍ ﺩﺍﺩﻥ ﺑﻪ ﮐﻮﺩﮎ ﻳﮑﯽ ﺍﺯ ﻣﺴﺎﻳﻞ ﻣﻬﻢ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺴﺎﻟﻪ ﺗﻐﺬﻳﻪ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ. ﻣﮕﺮ ﺩﺭ ﻣﻮﺍﺭﺩ ﺑﻴﻤﺎﺭﯼ‌ﻫﺎﯼ ﺟﺪﯼ، ﻗﺎﻋﺪﺗﺎ ﻧﺒﺎﻳﺪ ﻣﺴﺎﻟﻪ ﻣﻬﻤﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻳﺪ. ﺗﻨﻬﺎ ﻋﻠﺖ ﺭﻭﺍﻧﺸﻨﺎﺧﺘﯽ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻥ ﻣﺴﺎﻟﻪ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻧﺪﺍﺷﺘﻦ ﺍﻃﻼﻋﺖ ﺩﻗﻴﻖ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﺍﺳﺖ.

ﺑﺒﻴﻨﻴﻢ ﻣﻌﻤﻮﻻ ﭼﻪ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ. ﺣﺪﻭﺩ ﻳﮏ‌ ﺳﺎﻟﮕﯽ ﻧﻮﺯﺍﺩ ﻣﻌﻤﻮﻻ ﮐﻢ‌ﺗﺮ ﻭ ﺳﺨﺖ‌ﺗﺮ ﻏﺬﺍ ﻣﯽ‌ﺧﻮﺭﺩ. ﻫﻴﭻ ﺗﻌﺠﺐ ﻧﮑﻨﻴﺪ ﺯﻳﺮﺍ ﺍﮔﺮ ﺍﻭ ﺑﻪ ﻣﺎﻧﻨﺪ ﻳﮏ ﺳﺎﻝ ﺍﻭﻝ ﻏﺬﺍ ﺑﺨﻮﺭﺩ ﺑﻪ ﺯﻭﺩﯼ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﻭ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﭘﺪﻳﺪﻩ ﻋﺠﻴﺐ ﺩﺭ ﺳﻴﺮﮎ ﺑﻪ ﻣﻌﺮﺽ ﻧﻤﺎﻳﺶ ﺑﮕﺬﺍﺭﻳﻢ. ﮐﻮﺩﮎ ﻓﻘﻂ ﻓﺮﺩﻳﺖ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﻏﺬﺍﻫﺎﻳﯽ ﮐﻪ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﺪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺑﻪ ﻋﻼﻭﻩ ﺍﻭ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺍﺷﺘﻬﺎﻳﺶ ﺍﺯ ﺭﻭﺯﯼ ﺑﻪ ﺭﻭﺯ ﺩﻳﮕﺮ ﻭ ﺍﺯ ﻫﻔﺘﻪ‌ﺍﯼ ﺑﻪ ﻫﻔﺘﻪ ﺩﻳﮕﺮ ﻓﺮﻕ ﻣﯽ‌ﮐﻨﺪ. ﺍﻣﺎ ﺍﮐﺜﺮ ﺍﻭﻗﺎﺕ ﻣﺎﺩﺭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻣﯽ¬‌ﻳﻨﺪ ﻓﺮﺯﻧﺪﺵ ﮐﻢ ﻏﺬﺍ ﻣﯽ‌ﺧﻮﺭﺩ ﻧﺎﺭﺍﺣﺖ ﻣﯽ‌ﺷﻮﺩ. ﺍﻭ ﮐﻪ ﻧﺎﺭﺍﺣﺖ ﻭ ﻧﮕﺮﺍﻥ ﺍﺳﺖ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﺑﻪ ﺯﻭﺭ ﺑﻪ ﮐﻮﺩﮎ ﻏﺬﺍ ﺑﺪﻫﺪ. «ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﯼ ﺑﺰﺭﮒ ﻭ ﻗﻮﯼ ﺑﺸﻮﯼ، ﻫﻮﻳﺞ‌ﻫﺎ‌ﺗﻮ ﺑﺨﻮﺭ.» ﺍﺯ ﺍﻳﻦ ﻟﺤﻈﻪ ﺑﻪ ﺑﻌﺪ ﻭﺍﺭﺩ ﻳﮏ ﺩﻭﺭ ﺑﺎﻃﻞ ﻣﯽ‌ﺷﻮﻳﻢ. ﻫﺮ ﭼﻘﺪﺭ ﻣﺎﺩﺭ ﺍﺻﺮﺍﺭ ﻣﯽ‌ﮐﻨﺪ، ﮐﻮﺩﮎ ﺑﻴﺸﺘﺮ ﺍﺩﺍ ﺩﺭ ﻣﯽ‌ﺁﻭﺭﺩ ﻭ ﻣﺨﺎﻟﻔﺖ ﻣﯽ‌ﮐﻨﺪ. ﻫﺮﭼﻘﺪﺭ ﮐﻮﺩﮎ ﮐﻤﺘﺮ ﻏﺬﺍ ﻣﯽ‌ﺧﻮﺭﺩ، ﻣﺎﺩﺭ ﭘﺮﻳﺸﺎﻥ‌ﺗﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻓﺸﺎﺭ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﺧﻮﺭﺩﻥ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻭﺭﺩ ﻭ ﺍﻭ ﺭﺍ ﻟﻮﺱ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﻏﺬﺍ ﺑﺨﻮﺭﺩ. ﺩﺭ ﻣﺪﺕ ﮐﻮﺗﺎﻫﯽ ﻣﺎﺩﺭ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﺴﺎﻟﻪ ﺟﺪﻳﺪﯼ ﮐﻪ ﻗﺒﻼ ﻭﺟﻮﺩ ﻧﺪﺍﺷﺖ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ.

ﺗﻤﺎﻡ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺑﯽ‌ﻓﺎﻳﺪﻩ ﺍﺳﺖ. ﭼﺮﺍ؟ ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺳﺎﺩﻩ ﮐﻪ ﻣﺎ، ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ، ﻳﮏ ﻫﻢ‌ﺩﺳﺖ ﻧﻴﺮﻭﻣﻨﺪ ﺩﺍﺭﻳﻢ: ﮔﺮﺳﻨﮕﯽ ﻃﺒﻴﻌﯽ ﮐﻮﺩﮎ. ﺍﮔﺮ ﻏﺬﺍﻫﺎﻳﯽ ﻣﺘﻨﻮﻉ، ﻣﺘﻌﺎﺩﻝ ﻭ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﻌﻘﻮﻝ ﺑﺮﺍﻳﺶ ﺁﻣﺎﺩﻩ ﮐﻨﻴﻢ ﻭ ﺑﮕﺬﺍﺭﻳﻢ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻏﺬﺍﻳﺶ ﺭﺍ ﺑﺨﻮﺭﺩ، ﻭﯼ ﺑﻪ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﮐﻪ ﺳﺎﻟﻢ ﻭ ﻗﻮﯼ ﺑﻤﺎﻧﺪ ﺍﺯ ﻏﺬﺍﻳﯽ ﮐﻪ ﺟﻠﻮﻳﺶ ﮔﺬﺍﺷﺘﻪ‌ﺍﻳﺪ ﺧﻮﺍﻫﺪ ﺧﻮﺭﺩ. ﺍﻣﺎ ﺑﺎﻳﺪ ﺑﻪ ﺧﻮﺩ ﮐﻮﺩﮎ ﺩﺭ ﻣﻮﺭﺩ ﻋﺎﺩﺍﺕ ﻏﺬﺍ ﺧﻮﺭﺩﻧﺶ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﻳﻢ. ﺍﻳﻦ ﺣﻖ ﺭﺍ ﺑﺮﺍﯼ ﺍﻭ ﻗﺎﺋﻞ ﺑﺎﺷﻴﺪ ﮐﻪ ﺫﺍﺋﻘﻪ‌ﺍﺵ ﺍﺯ ﻣﺎﻫﯽ ﺑﻪ ﻣﺎﻩ ﺩﻳﮕﺮ ﻭ ﺍﺷﺘﻬﺎﻳﺶ ﺍﺯ ﺭﻭﺯﯼ ﺑﻪ ﺭﻭﺯ ﺩﻳﮕﺮ ﻓﺮﻕ ﮐﻨﺪ. ﺍﮔﺮ ﺍﻭ ﻧﻮﻋﯽ ﺳﺒﺰﯼ ﻭ ﻳﺎ ﻣﻴﻮﻩ ﺭﺍ ﮐﻪ ﺗﺎ ﺩﻳﺮﻭﺯ ﺧﻴﻠﯽ ﺩﻭﺳﺖ ﺩﺍﺷﺖ ﻧﺎﮔﻬﺎﻥ ﺩﻳﮕﺮ ﻧﺨﻮﺭﺩ ﭼﻪ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻴﺪ؟ ﺑﻪ ﺍﻭ ﺍﻳﻦ ﺁﺯﺍﺩﯼ ﺭﺍ ﺑﺪﻫﻴﺪ ﮐﻪ ﺁﻥ ﺭﺍ ﻧﺨﻮﺭﺩ ﻭ ﺍﻭ ﺭﺍ ﺑﺎ ﻧﺎﺯ ﻭ ﻧﻮﺍﺯﺵ ﻣﺠﺒﻮﺭ ﺑﻪ ﺧﻮﺭﺩﻥ ﺁﻥ ﻧﮑﻨﻴﺪ. ﺍﮔﺮ ﻏﺬﺍﻳﯽ ﮐﺎﻣﻞ، ﻣﺘﻨﻮﻉ ﻭ ﻣﺘﻌﺎﺩﻝ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﺪ ﻭ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺑﺮﺍﺳﺎﺱ ﺍﺣﺘﻴﺎﺟﺎﺗﺶ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺑﺨﻮﺭﺩ، ﻫﻴﭻ ﮔﺎﻩ ﻣﺴﺎﻟﻪ‌ﺍﯼ ﺩﺭ ﻣﻮﺭﺩ ﺗﻐﺬﻳﻪ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ.

ﺩﻭﺭﻩ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ، ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﻫﻨﮕﺎﻡ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻭﻗﺖ ﺯﻳﺎﺩﯼ ﺗﻠﻒ ﻣﯽ‌ﮐﻨﺪ. ﻋﻠﺖ ﺍﻳﻦ ﺍﻣﺮ ﺗﺎ ﺣﺪﯼ ﻧﺎﺷﯽ ﺍﺯ ﻛﻢ ﺷﺪﻥ ﺍﺷﺘﻬﺎﯼ ﻃﻔﻞ ﻧﺴﺒﺖ ﺑﻪ ﻗﺒﻞ ﺍﺳﺖ، ﺑﻪ ﻋﻼﻭﻩ ﻋﻨﺎﺻﺮ ﺑﺴﻴﺎﺭ ﺟﺎﻟﺐ‌ﺗﺮﯼ ﺍﺯ ﻏﺬﺍ ﺑﻪ ﻫﻨﮕﺎﻡ ﻏﺬﺍﺧﻮﺭﺩﻥ ﻭﺍﺭﺩ ﻋﻤﻞ ﻣﯽ‌ﺷﻮﻧﺪ. ﻣﻨﻈﻮﺭ ﺍﺯ ﻋﻨﺎﺻﺮ ﺩﻳﮕﺮ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺍﻓﺴﻮﻥ‌ﮐﻨﻨﺪﻩ‌ﺍﯼ ﭼﻮﻥ ﺗﺒﺪﻳﻞ ﻏﺬﺍ ﺑﻪ ﺧﻤﻴﺮ ﻳﺎ ﺿﺮﺑﻪ ﮔﺮﻓﺘﻦ ﺑﺎ ﻗﺎﺷﻖ ﺭﻭﯼ ﻣﻴﺰ ﻳﺎ ﭘﺮﺕ ﮐﺮﺩﻥ ﻫﻤﻪ ﭼﻴﺰ ﺭﻭﯼ ﺯﻣﻴﻦ ﺍﺳﺖ. ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﺑﺮﺍﯼ ﻣﺎﺩﺭ ﺑﺴﻴﺎﺭ ﺍﺫﻳﺖ ﮐﻨﻨﺪﻩ ﺍﺳﺖ. ﻭﻟﯽ ﺑﺪﺍﻧﻴﻢ ﮐﻪ ﺍﻳﻦ ﺣﺮﮐﺎﺕ ﺑﺮﺍﯼ ﮐﻮﺩﮐﯽ ﺧﺮﺩﺳﺎﻝ ﻃﺒﻴﻌﯽ ﺍﺳﺖ. ﺍﻭ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺭﺍ ﺍﺯ ﺭﻭﯼ ﻗﺼﺪ ﻭ ﺑﻪ ﻣﻨﻈﻮﺭ ﻣﺨﺎﻟﻔﺖ ﻳﺎ ﺗﺤﺮﻳﮏ ﻣﺎﺩﺭﺵ ﺍﻧﺠﺎﻡ ﻧﻤﯽ‌ﺩﻫﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﭼﮕﻮﻧﻪ ﻋﮑﺲ‌ﺍﻟﻌﻤﻠﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ؟ ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ، ﺑﻴﻦ ﺩﻭ ﺩﻭﺭﻩ ﮐﻮﺗﺎﻩ ﺑﺎﺯﯼ ﻭ ﺣﻮﺍﺱ ﭘﺮﺕ ﮐﺮﺩﻥ ﭼﻨﺪ ﻟﻘﻤﻪ‌ﺍﯼ ﺑﻪ ﺍﻭ ﻏﺬﺍ ﺑﺨﻮﺭﺍﻧﻴﺪ. ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﻫﻢ ﻭﻗﺘﯽ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﮐﻪ ﺩﻳﮕﺮ ﻋﻼﻗﻪ‌ﺍﯼ ﺑﻪ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻧﺸﺎﻥ ﻧﻤﯽ‌ﺩﻫﺪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺩﻳﮕﺮ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﻏﺬﺍ ﺧﻮﺭﺩﻩ ﻭ ﺍﺯ ﺻﻨﺪﻟﻴﺶ ﭘﺎﻳﻴﻦ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﻏﺬﺍﻳﺶ ﺭﺍ ﺟﻤﻊ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺩﻳﺪﻳﺪ ﺑﻼﻓﺎﺻﻠﻪ ﺷﺮﻭﻉ ﺑﻪ ﮔﺮﻳﻪ ﻭ ﺁﻩ ﻭ ﻧﺎﻟﻪ ﻣﯽ‌ﮐﻨﺪ، ﺩﻭﺑﺎﺭﻩ ﻏﺬﺍ ﺭﺍ ﺟﻠﻮﯼ ﺍﻭ ﺑﮕﺬﺍﺭﻳﺪ.

ﺣﺎﻝ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﻣﻬﻢ ﻣﯽ‌ﺭﺳﻴﻢ: ﭼﮕﻮﻧﻪ ﺑﻪ ﻳﮏ ﮐﻮﺩﮎ ﺑﻴﺎﻣﻮﺯﻳﻢ ﮐﻪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻏﺬﺍ ﺑﺨﻮﺭﺩ؟ ﺑﻪ ﻧﻈﺮ ﻣﻦ ﺟﻮﺍﺏ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﺍﺳﺖ: ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﺪ. ﺍﮔﺮ ﺩﺭ ﺳﻦ ﺷﺶ ﻣﺎﻫﮕﯽ ﺍﺟﺎﺯﻩ ﺩﺍﺩﻩ ﺑﺎﺷﻴﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﻴﺴﮑﻮﻳﺘﺶ ﺭﺍ ﺑﮕﻴﺮﺩ،  ﺍﻭ ﺭﺍ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﺎ ﻗﺎﺷﻖ ﻏﺬﺍ ﺑﺨﻮﺭﺩ ﺁﻣﺎﺩﻩ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﻧﻮﺯﺍﺩﯼ ﮐﻪ ﺑﻴﻦ ﺷﺶ ﻣﺎﻫﮕﯽ ﺗﺎ ﻳﮏ ﺳﺎﻟﮕﯽ ﻫﻴﭻ ﻭﻗﺖ ﺍﺟﺎﺯﻩ ﻧﺪﺍﺷﺘﻪ ﺑﺎ ﺍﻧﮕﺸﺘﺎﻧﺶ ﻏﺬﺍ ﺑﺨﻮﺭﺩ، ﺍﺣﺘﻤﺎﻻ ﺑﺮﺍﯼ ﺍﻳﻦ ﺁﻣﻮﺯﺵ ﮐﻤﯽ ﻋﻘﺐ ﻣﺎﻧﺪﮔﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﮐﻮﭼﻮﻟﻮﯼ ﺷﻤﺎ ﺍﺣﺘﻤﺎﻻ ﺯﻣﺎﻧﯽ ﮐﻪ ﺣﺪﻭﺩ ﻳﮏ ﺳﺎﻝ ﺩﺍﺭﺩ، ﻗﺎﺷﻖ ﺭﺍ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﻳﺎ ﺑﻪ ﻃﺮﻗﯽ ﺩﻳﮕﺮ ﺳﻌﯽ ﺑﻪ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺁﻣﺎﺩﮔﯽ ﺧﻮﺩ ﺑﺮﺍﯼ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻣﯽ‌ﮐﻨﺪ. ﺑﮕﺬﺍﺭﻳﺪ ﺍﻣﺘﺤﺎﻥ ﮐﻨﺪ. ﻣﺴﻠﻤﺎ ﺍﮔﺮ ﺧﻮﺩﺗﺎﻥ ﺑﻪ ﺍﻭ ﻏﺬﺍ ﺑﺪﻫﻴﺪ، ﺳﺮﻳﻊ‌ﺗﺮ ﻭ ﺑﻬﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭﻟﯽ ﺟﻠﻮ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﮕﻴﺮﻳﺪ. ﺑﺎ ﺍﺩﺍﻣﻪ ﺍﻳﻦ ﮐﺎﺭ- ﻏﺬﺍ ﺩﺍﺩﻥ ﺑﻪ ﺍﻭ- ﮐﻤﮑﯽ ﺑﻪ ﺭﺷﺪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻭ ﻗﻮﻩ ﺍﺑﺘﮑﺎﺭ ﺍﻭ ﻧﺨﻮﺍﻫﻴﺪ ﮐﺮﺩ.  

ﺑﻴﺸﺘﺮ ﮐﻮﺩﮐﺎﻥ ﺩﺭ ﺁﻏﺎﺯ ﻣﺮﺣﻠﻪ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ، ﺑﻴﻦ ۱۲ ﺗﺎ ۱۶ ﻣﺎﻫﮕﯽ، ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﺎ ﻗﺎﺷﻖ ﻏﺬﺍ ﺑﺨﻮﺭﻧﺪ. ﺍﮔﺮ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻳﺎ ﮐﻤﯽ ﺑﻌﺪ- ﺣﺪﻭﺩ ﺩﻭ ﺳﺎﻟﮕﯽ- ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺪﻫﻴﺪ، ﺩﻳﮕﺮ ﺍﺷﺘﻴﺎﻗﯽ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ  ﻭ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺎﻳﺪ ﻫﻤﭽﻨﺎﻥ ﺧﻮﺩﺗﺎﻥ ﻏﺬﺍ ﺩﺍﺩﻥ ﺑﻪ ﺍﻭ ﺭﺍ ﺍﺩﺍﻣﻪ ﺩﻫﻴﺪ.

ﺁﻣﻮﺯﺵ ﮐﻨﺘﺮﻝ ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ
ﺁﻣﻮﺯﺵ ﮐﻨﺘﺮﻝ ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ ﻧﻴﺰ ﺯﻣﻴﻨﻪ ﺩﻳﮕﺮﯼ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺸﮑﻼﺕ ﺑﻴﻬﻮﺩﻩ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻓﺮﺍﻫﻢ ﺁﻭﺭﺩ. ﺑﺴﻴﺎﯼ ﺍﺯ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺗﻴﺐ ﺍﻃﻔﺎﻝ ﺯﻳﺮ ﺳﻪ ﺳﺎﻝ ﻣﯽ‌ﭘﺮﺩﺍﺯﻧﺪ، ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺑﻪ ﺻﻮﺭﺕ ﺑﺴﻴﺎﺭ ﻣﺒﻬﻤﯽ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﺎﺩﺭ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﮐﻨﺘﺮﻝ ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ ﺭﺍ ﺑﻪ ﺍﻭﻟﻴﻦ ﻓﺮﺯﻧﺪﺵ ﺑﻴﺎﻣﻮﺯﺩ ﺑﻪ ﺣﺎﻝ ﺧﻮﺩ ﺭﻫﺎ ﺷﺪﻩ ﺍﺳﺖ. ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﮑﺎﺕ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺪﺍﻧﻴﻢ ﺍﻳﻦ ﺁﻣﻮﺯﺵ ﺭﺍ ﭼﻪ ﺯﻣﺎﻧﯽ ﺷﺮﻭﻉ ﮐﻨﻴﻢ . ﺁﻥ ﺭﻭﺯﯼ ﮐﻪ ﮐﻮﺩﮎ ﺑﺘﻮﺍﻧﺪ ﺑﺮ ﻗﻀﺎﯼ ﺣﺎﺟﺘﺶ ﮐﻨﺘﺮﻝ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺭﻭﺯ ﻣﻴﻤﻮﻧﯽ ﺍﺳﺖ. ﻭﻟﯽ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﮐﻪ ﺩﺭ ﭼﻪ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺍﺯ ﺭﺷﺪ ﮐﻮﺩﮎ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﻌﻘﻮﻝ ﺑﺮ ﺍﻳﻦ ﺍﺣﺘﻴﺎﺟﺶ ﻓﺎﺋﻖ ﺁﻳﺪ.

ﮐﻮﺩﮎ ﻗﺒﻞ ﺍﺯ ﺳﻦ ﺣﺪﻭﺩ ﺩﻭ ﺳﺎﻝ ﺍﺯ ﻧﻈﺮ ﻋﺼﺒﯽ- ﻋﻀﻼﻧﯽ ﻗﺪﺭﺕ ﻻﺯﻡ ﺟﻬﺖ ﮐﻨﺘﺮﻝ ﺣﺮﮐﺎﺕ ﺭﻭﺩﻩ ﻭ ﻣﺜﺎﻧﻪ ﺭﺍ ﻧﺪﺍﺭﺩ. ﺍﮔﺮ ﻗﺒﻞ ﺍﺯ ﺍﻳﻦ ﺳﻦ ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﻭ ﺭﺍ ﻣﺠﺒﻮﺭ ﺑﻪ ﺍﻳﻦ ﮐﺎﺭ ﮐﻨﻴﺪ، ﻳﺎ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻣﻨﺠﺮ ﺑﻪ ﻳﮏ ﻓﺎﺟﻌﻪ ﺧﻮﺍﻫﺪ ﺷﺪ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﻭﻗﺖ ﺗﻠﻒ ﮐﺮﺩﻥ ﻭ ﻓﺸﺎﺭ ﺑﻴﻬﻮﺩﻩ‌ﺍﯼ ﺑﺮﺍﯼ ﻣﺎﺩﺭ ﻭ ﮐﻮﺩﮎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﭘﺲ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺩﻭ ﺳﺎﻟﮕﯽ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺭﺍ ﺭﻫﺎ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﺁﻳﻨﺪﻩ،  ﺑﻴﺸﺘﺮ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ. ﻣﻤﮑﻦ ﺍﺳﺖ ﻫﻤﺴﺎﻳﻪ‌ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺎ ﺍﻓﺘﺨﺎﺭ ﺑﮕﻮﻳﺪ ﮐﻪ ﮐﻨﺘﺮﻝ ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﺩﺭ ﺳﻦ ﭘﺎﻧﺰﺩﻩ ﻣﺎﻫﮕﯽ ﺑﺪﻭﻥ ﻫﻴﭻ ﺍﺷﮑﺎﻟﯽ ﺁﻣﻮﺧﺘﻪ ﺍﺳﺖ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺣﺮﻑ ﺷﻤﺎ ﺭﺍ ﺑﺮﺍﻧﮕﻴﺰﺩ ﮐﻪ ﺷﻤﺎ ﻧﻴﺰ ﺑﻪ ﻓﺮﺯﻧﺪ ﺧﻮﺩ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﻳﻦ ﺁﻣﻮﺯﺵ ﺭﺍ ﺑﺪﻫﻴﺪ. ﻭﻟﯽ ﺷﺎﻳﺪ ﺷﻤﺎ ﻭ ﻫﻤﺴﺎﻳﻪ‌ﺗﺎﻥ ﻧﺪﺍﻧﻴﺪ ﮐﻪ ﻓﺮﺯﻧﺪ ﺍﻭ ﺍﺣﺘﻤﺎﻻ ﺩﺭ ﺳﻦ ﭼﻬﺎﺭ ﻳﺎ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺭﺧﺘﺨﻮﺍﺑﺶ ﺭﺍ ﺧﻴﺲ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﻭ ﺍﻳﻦ ﺍﻣﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﺗﺎ ﺳﺎﻝ‌ﻫﺎ ﺍﺩﺍﻣﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﭘﺲ ﮐﻤﯽ ﺻﺒﻮﺭ ﺑﺎﺷﻴﺪ. ﺩﺭﺑﺎﺭﻩ ﺍﻳﻦ ﻧﺎﺭﺍﺣﺘﯽ ﮐﻮﭼﮏ- ﭘﻮﺷﮏ- ﺯﻳﺎﺩ ﻓﮑﺮ ﻧﮑﻨﻴﺪ.

ﻓﺮﺽ ﮐﻨﻴﻢ ﮐﻪ ﮐﺴﯽ ﻧﻮﺯﺍﺩ ﺩﻭﺍﺯﺩﻩ ﻣﺎﻫﻪ‌ﺍﺵ ﺭﺍﺣﻤﺎﻡ ﻣﯽ‌ﮐﻨﺪ. ﮐﻮﺩﮎ ﺑﻪ ﺑﺮﺭﺳﯽ ﺑﺪﻥ ﺧﻮﺩ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ ﻭ ﺍﮐﺘﺸﺎﻓﺎﺕ ﻣﺠﺬﻭﺏ‌ﮐﻨﻨﺪﻩ‌ﺍﯼ ﻣﯽ‌ﮐﻨﺪ. ﻃﻔﻞ ﺑﺎ ﮔﻮﺵ‌ﻫﺎﻳﺶ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻧﺮﻣﻪ ﮔﻮﺷﺶ ﺭﺍ ﻣﯽ‌ﮐﺸﺪ. ﻣﺎﻣﺎﻥ ﺗﻔﺴﻴﺮ ﻣﯽ‌ﮐﻨﺪ:«ﻧﮕﺎﻩ ﮐﻦ ﭼﻪ‌ﻃﻮﺭﯼ ﺑﺎ ﮔﻮﺷﻬﺎﺵ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﻪ، ﺍﻳﻦ ﮐﺎﺭﺵ ﻧﺸﻮﻥ ﻣﯽ‌ﺩﻩ ﮐﻪ ﭼﻪ ﺑﭽﻪ ﺑﺎﻫﻮﺷﻴﻪ.» ﺑﻌﺪ ﮐﻮﺩﮎ ﺷﺮﻭﻉ ﺑﻪ ﺑﺎﺯﯼ ﺑﺎ ﺍﻧﮕﺸﺖ ﺷﺼﺖ ﭘﺎﻳﺶ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻣﺎﻣﺎﻥ ﻓﺮﻳﺎﺩ ﺗﺤﺴﻴﻦ ﺑﺮ ﻣﯽ‌ﺁﻭﺭﺩ ﮐﻪ:«ﺑﺒﻴﻦ ﭼﻪ ﺟﻮﺭ ﻋﺠﻴﺒﯽ ﺑﺎ ﺷﺼﺘﺶ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﻪ.» ﺳﭙﺲ ﮐﻮﺩﮎ ﺑﺎ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ. ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻣﺎﺩﺭ ﺧﻮﺍﻫﺪ ﮔﻔﺖ: «ﺁﻩ ﻣﺤﺸﺮﻩ، ﺩﺍﺭﻩ ﺑﺎ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﻪ.» ﻣﺴﻠﻤﺎ ﻧﻪ. ﺷﺎﻳﺪ ﻫﻢ ﺑﺰﻧﺪ ﺭﻭﯼ ﺩﺳﺘﺶ ﺗﺎ ﺑﻪ ﺍﻭ ﺑﻔﻬﻤﺎﻧﺪ ﮐﻪ ﮐﺎﺭ ﺑﺪﯼ ﻣﯽ‌ﮐﻨﺪ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﭼﻨﻴﻦ ﺻﺤﻨﻪ ﻏﻢ‌ﺍﻧﮕﻴﺰﯼ ﺩﺭ ﺧﺎﻧﻪ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﺍﺗﻔﺎﻕ ﻣﻴﻔﺘﺪ.

ﻣﺎﺩﺭ ﺑﺎ ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﭼﻪ ﭼﻴﺰﯼ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﻣﯽ‌ﺁﻣﻮﺯﺩ؟ ﺍﻭﻻ ﻣﺴﺎﻟﻪ ﺟﻨﺴﯽ ﻣﻮﺭﺩ ﺗﺼﻮﺭ، ﺑﺮﺍﯼ ﮐﻮﺩﮐﯽ ﺧﺮﺩﺳﺎﻝ ﻣﻄﺮﺡ ﻧﻴﺴﺖ، ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﻓﻘﻂ ﺩﺭ ﺫﻫﻦ ﻣﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﺮﺍﯼ ﮐﻮﺩﮎ، ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﺩﺭﺳﺖ ﻣﺜﻞ ﺷﺼﺖ ﭘﺎ ﻳﺎ ﮔﻮﺵ‌ﻫﺎﻳﺶ ﺍﺳﺖ. ﺍﻳﻦ ﻣﺎ ﻫﺴﺘﻴﻢ ﮐﻪ ﺑﭽﻪ ﺭﺍ ﻭﺍ ﻣﯽ‌ﺩﺍﺭﻳﻢ ﺑﻪ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﺗﻮﺟﻬﯽ ﺧﺎﺹ ﻭ ﻏﻴﺮﺳﺎﻟﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺑﺎ ﺍﻳﻦ ﮐﺎﺭﻣﺎﻥ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﻳﻢ ﮐﻪ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﻣﻨﻄﻘﻪ ﻣﻤﻨﻮﻉ ﺍﺯ ﺑﺪﻥ ﺍﻭﺳﺖ.

ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﺎﺩﺭﯼ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﻨﺪ، ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﻣﯽ‌ﮔﻮﻳﺪ: «ﺑﺎ ﮔﻮﺵ‌ﻫﺎﺕ ﺑﺎﺯﯼ ﮐﻦ ﭼﻮﻥ ﺧﻴﻠﯽ ﻗﺸﻨﮕﻦ، ﺍﻧﮕﺸﺘﻬﺎﺕ ﺧﻮﺷﮕﻠﻦ ﻭ ﺑﺎﻳﺪ ﺍﺯ ﺍﻭﻧﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﯽ، ﺍﻧﮕﺸﺖ ﭘﺎﻫﺎﺕ ﻗﺸﻨﮕﻦ ﺑﺎ ﺍﻭﻧﺎ ﺑﺎﺯﯼ ﮐﻦ. ﻭﻟﯽ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺑﺎ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ ﺑﺪ ﻭ ﺯﺷﺘﻪ. ﺍﻳﻦ ﺑﺨﺶ ﺍﺯ ﺑﺪﻥ ﺯﺷﺘﻪ.»

ﻭﻗﺘﯽ ﮐﻮﺩﮐﯽ ﮐﻤﺘﺮ ﺍﺯ ﺩﻭ ﺳﺎﻝ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﺭﺍ ﮐﺸﻒ ﻣﯽ‌ﮐﻨﺪ ﭼﮕﻮﻧﻪ ﺭﻓﺘﺎﺭﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ؟ ﺩﺭﺳﺖ ﻫﻤﺎﻥ ﮔﻮﻧﻪ ﮐﻪ ﻫﺮ ﻧﻘﻄﻪ ﺩﻳﮕﺮ ﺍﺯ ﺑﺪﻧﺶ ﺭﺍ ﮐﺸﻒ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﺗﺒﺪﻳﻞ ﺑﻪ «ﻣﺤﻞ ﺗﺎﺑﻮ»‌ﻳﯽ ﺩﺭ ﺫﻫﻨﺶ ﻧﻤﯽ‌ﺷﻮﺩ ﮐﻪ ﻫﻤﻴﺸﻪ ﺑﻪ ﻫﻤﺮﺍﻩ ﺍﺣﺴﺎﺱ ﺧﺠﺎﻟﺖ ﻭ ﺑﺪﯼ ﺑﺎﺷﺪ. ﺑﺎ ﺍﻳﻦ ﻛﺎﺭ ﻗﺪﻡ ﻣﺜﺒﺘﯽ ﺟﻬﺖ ﮐﻤﮏ ﺑﻪ ﺍﻭ ﺩﺭﺑﺎﺭﻩ ﺩﺭﮎ ﺟﻨﺴﻴﺖ ﻭ ﺑﺪﻧﺶ ﻭ ﺩﺍﺷﺘﻦ ﺭﻓﺘﺎﺭﯼ ﺻﺤﻴﺢ ﺑﻪ ﻫﻨﮕﺎﻡ ﺑﺰﺭﮒ ﺷﺪﻥ ﺑﺮﺩﺍﺷﺘﻪ‌ﺍﻳﺪ. ﺑﺎﻳﺪ ﺑﻪ ﮐﻮﺩﮎ ﻧﺎﻡ ﺁﻻﺕ ﺗﻨﺎﺳﻠﯽ ﻭ ﺩﻓﻊ ﺭﺍ ﻣﺎﻧﻨﺪ ﺗﻤﺎﻡ ﺍﻋﻀﺎﯼ ﺩﻳﮕﺮ ﺑﺪﻧﺶ ﺑﻴﺎﻣﻮﺯﻳﻢ. ﻣﻌﻤﻮﻻ ﺩﺭ ﻓﺮﻫﻨﮓ ﻣﺎ ﺍﺳﻢ ﺍﻳﻦ ﺁﻻﺕ ﺭﺍ ﺑﻪ ﮐﻮﺩﮎ ﻧﻤﯽ‌ﺁﻣﻮﺯﻧﺪ.

ﺍﮔﺮ ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﺍﻳﻦ ﻗﺴﻤﺖ ﻫﺎﯼ ﺑﺪﻧﺶ ﺭﺍ ﻗﺒﻮﻝ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺍﺣﺴﺎﺱ ﻣﺜﺒﺘﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﮐﺴﺐ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺍﻳﻦ ﮐﺎﺭ ﮐﻤﮑﯽ ﺑﻪ ﮐﻮﺩﮎ ﺍﺳﺖ ﻭ ﺍﺯ ﺍﻳﻦ ﻃﺮﻳﻖ ﺍﺯ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻥ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﺩﺭ ﺁﻳﻨﺪﻩ ﺑﺮﺍﯼ ﺍﻭ ﺟﻠﻮﮔﻴﺮﯼ ﻣﯽ‌ﮐﻨﻴﻢ.

ﻣﺮﺣﻠﻪ ﺭﺷﺪ ﮐﻮﺩﮐﯽ ﮐﻪ ﺷﺮﻭﻉ ﺑﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻳﮏ ﺟﻤﻠﻪ ﺧﻼﺻﻪ ﺷﻮﺩ: ﺍﻳﻦ ﺩﻭﺭﻩ ﺳﻦ ﺍﮐﺘﺸﺎﻓﺎﺕ ﺍﺳﺖ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻣﺤﻘﻖ ﺟﻮﺍﻧﯽ ﺍﺳﺖ ﮐﻪ ﺑﺪﻭﻥ ﺍﻧﻘﻄﺎﻉ ﺩﻧﻴﺎﻳﺶ ﻭ ﺍﺯ ﺟﻤﻠﻪ ﺑﺪﻧﺶ ﺭﺍ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ. ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺗﻘﻮﻳﺖ ﻋﻀﻼﺗﺶ ﺩﺍﺭﺩ ﻭ ﺩﺭ ﺿﻤﻦ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﺪ ﺍﻳﻦ ﺍﻧﺮﮊﯼ ﻋﻈﻴﻤﯽ ﮐﻪ ﻃﺒﻴﻌﺖ ﺑﻪ ﺍﻭ ﺩﺍﺩﻩ ﺍﺳﺖ ﻣﺼﺮﻑ ﮐﻨﺪ. ﺍﻣﺎ ﺍﺻﻞ ﺍﺳﺎﺳﯽ ﮐﻪ ﺗﻤﺎﻡ ﺗﻌﻠﻴﻤﺎﺕ ﺍﺯ ﺁﻥ ﻧﺎﺷﯽ ﻣﯽ‌ﺷﻮﻧﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺗﻤﺎﻡ ﺍﮐﺘﺸﺎﻓﺘﺶ ﺑﺎﻳﺪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺴﺶ ﺭﺍ ﭘﺎﻳﻪ‌ﺭﻳﺰﯼ ﮐﻨﺪ. ﺍﻭ ﺑﺎﻳﺪ ﺑﻪ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﺭﻭ ﺑﻪ ﺭﺷﺪﺵ ﺍﻋﺘﻤﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺑﺎﻳﺪ ﺑﺪﻭﺩ، ﺑﭙﺮﺩ، ﺑﺎﻻﺑﺮﻭﺩ، ﺑﺎ ﻣﮑﻌﺐ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﭼﻴﺰﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺴﺎﺯﺩ، ﺑﺎ ﻣﺎﺷﻴﻦ ﻭ ﮐﺎﻣﻴﻮﻥ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﺑﺎﺯﯼ ﮐﻨﺪ، ﺷﻦ ﻭ ﺧﺎﮎ ﺑﺎﺯﯼ ﮐﻨﺪ، ﺑﺎ ﻋﺮﻭﺳﮏ ﻭ ﺣﻴﻮﺍﻧﺎﺕ ﻋﺮﻭﺳﮑﯽ ﺑﺎﺯﯼ ﮐﻨﺪ، ﺁﻫﻨﮓ ﻭ ﺻﺪﺍ ﺍﺯ ﺧﻮﺩﺵ ﺩﺭ ﺑﻴﺎﻭﺭﺩ، ﺑﺎ ﻣﺎﺩﺭﺵ ﺑﺎﺯﯼ ﻭ ﻫﻤﮑﺎﺭﯼ ﮐﻨﺪ، ﮐﻠﻤﺎﺕ ﻭ ﺟﻤﻠﻪ‌ﻫﺎﻳﯽ ﺑﻪ ﺗﺪﺭﻳﺞ ﮐﻪ ﺯﺑﺎﻧﺶ ﺭﺷﺪ ﻣﯽ‌ﻳﺎﺑﺪ ﺑﻴﺎﻥ ﮐﻨﺪ، ﺑﺎ ﮐﺘﺎﺏ ﺑﺎﺯﯼ ﮐﻨﺪ ﻭ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻫﺎ ﺑﺨﻮﺍﻫﺪ ﮐﻪ ﺑﺮﺍﻳﺶ ﮐﺘﺎﺏ ﺑﺨﻮﺍﻧﻨﺪ.
