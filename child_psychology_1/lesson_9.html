ﺩﺭﺱ ﻫﻔﺘﻢ- ﺭﺍﻩ ﺍﻓﺘﺎﺩﻥ ﮐﻮﺩﮎ

ﺑﻪ ﻣﺤﺾ ﺍﻳﻦ‌ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍﻩ ﺭﻓﺘﻦ ﺭﺍ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ ﻭﺍﺭﺩ ﻣﺮﺣﻠﻪ ﺟﺪﻳﺪﯼ ﺍﺯ ﺭﺷﺪ ﻣﯽ‌ﺷﻮﺩ، ﻣﺮﺣﻠﻪ‌ﺍﯼ ﮐﻪ ﻧﺎﻡ ﺁﻥ ﺭﺍ «ﻣﺮﺣﻠﻪ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ» ﮔﺬﺍﺷﺘﻪ‌ﺍﻳﻢ. ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺩﻭﺭﻩ ﻳﺎﺩﮔﻴﺮﯼ ﺟﺪﻳﺪﯼ ﺭﺍ ﭘﻴﺶ ﺭﻭﯼ ﻭﯼ ﻣﯽ‌ﮔﺬﺍﺭﺩ: ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻓﻌﺎﻻﻧﻪ ﺩﻧﻴﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺍﺳﺖ ﮐﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺍﻭ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﺩ. ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭ ﻭﺿﻌﻴﺖ ﺧﺎﺻﯽ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﺑﻪ ﺷﺪﺕ ﺍﻭ ﺭﺍ ﺁﺳﻴﺐ‌ﭘﺬﻳﺮ ﻣﯽ‌ﺳﺎﺯﺩ: ﭼﻨﺎﻧﭽﻪ ﺍﻭ ﺭﺍ ﺯﻳﺎﺩﯼ ﺗﻨﺒﻴﻪ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺑﻪ ﺍﻭ ﺍﻟﻘﺎﺀ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﻋﻠﺖ ﺳﻌﯽ‌ﺍﺵ ﺩﺭ ﮐﺸﻒ ﻭ ﺑﺮﺭﺳﯽ ﺟﻬﺎﻧﺶ ﺑﭽﻪ ﺑﺪﯼ ﺍﺳﺖ، ﺑﻪ ﺧﻮﺩﺵ ﺷﮏ ﻣﯽ‌ﮐﻨﺪ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﭘﻴﺶ ﻧﻴﺰ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ، ﭘﺎﻳﻪ ﻭ ﺍﺳﺎﺱ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﮐﻮﺩﮐﯽ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﮐﻮﺩﮎ ﻧﺴﺒﺖ ﺑﻪ ﺟﻬﺎﻥ ﺑﺪﺑﻴﻦ ﻭ ﻳﺎ ﺧﻮﺵ‌ﺑﻴﻦ ﻣﯽ‌ﺷﻮﺩ. ﺯﻣﺎﻧﯽ ﮐﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﺁﻣﻮﺯﺩ ﺍﻳﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﻌﺪ ﺗﺎﺯﻩ‌ﺍﯼ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ: ﮐﻮﺩﮎ ﻳﺎ ﺑﻪ ﺧﻮﺩﺵ ﺍﻋﺘﻤﺎﺩ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﺷﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻧﻮﺯﺍﺩ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﺪ، ﻣﺎﺩﺭ ﻣﯽ‌ﻓﻬﻤﺪ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﻧﺒﺎﻳﺪ ﺩﺭ ﺩﺳﺘﺮﺱ ﻃﻔﻞ ﮔﺬﺍﺷﺖ، ﭼﻮﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﻣﺜﻼ ﺁﻥ ﺭﺍ ﺑﻪ ﺩﻫﺎﻥ ﺑﮕﺬﺍﺭﺩ ﻭ ﺑﺨﻮﺭﺩ.

ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪ، ﺩﻭﺭﺍﻥ ﺍﻭﺝ ﺑﺮﺭﺳﯽ‌ﻫﺎﯼ ﮐﻮﺩﮎ ﺍﺳﺖ ﻭ ﺗﻤﺎﻡ ﻣﺎﺩﺭﺍﻥ ﺑﺎﻳﺪ ﻳﺎﺩ ﺑﮕﻴﺮﻧﺪ ﺧﺎﻧﻪ‌ﺷﺎﻥ ﺭﺍ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﭘﺲ ﺍﻳﻦ ﺑﺎﺯﺭﺱ ﮐﻮﭼﻮﻟﻮ ﺩﺭ ﺁﻥ ﺁﺷﻴﺎﻧﻪ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﭼﮕﻮﻧﻪ ﺩﺭﺳﺖ ﮐﻨﻨﺪ. ﻳﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﺧﺎﻧﻪ‌ﺍﯼ ﻣﺨﺼﻮﺹ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺩﺍﺷﺖ ﻭ ﻳﺎ ﺧﺎﻧﻪ‌ﺍﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﺧﻄﺮﯼ ﺑﺮﺍﯼ ﺑﭽﻪ ﻭﺟﻮﺩ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺭﺍﻩ ﺣﻞ ﺍﻭﻝ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ، ﺑﺎﻳﺪ ﭼﻪ ﺑﻪ ﻃﻮﺭ ﺷﻔﺎﻫﯽ ﻭ ﭼﻪ ﺑﻪ ﺻﻮﺭﺕ ﻋﻤﻠﯽ ﺍﻧﺮﮊﯼ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﺗﺮﻣﺰ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻪ ﮐﺎﺭ ﺑﺮﻳﺪ. ﻣﺮﺗﺐ ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﺪ«ﻧﻪ» ﻭ ﺭﻭﯼ ﺩﺳﺘﺶ ﺑﺰﻧﻴﺪ ﮐﻪ ﺑﻪ ﭼﻴﺰﻫﺎ ﺩﺳﺖ ﻧﺰﻧﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﺍﻳﻨﮕﻮﻧﻪ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﻨﻨﺪ. ﺁﻧﻬﺎ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻨﺪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ‌ﺍﯼ ﺑﺰﺭﮒ ﮐﻨﻨﺪ ﮐﻪ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺩﺭﺳﺖ ﺷﺪﻩ ﻧﻪ ﺑﺮﺍﯼ ﺑﭽﻪ‌ﻫﺎ. ﺍﺯ ﻧﻈﺮ ﻋﻠﻤﯽ، ﺍﻳﻦ ﮐﺎﺭ ﺍﺷﺘﺒﺎﻩ ﻋﻈﻴﻤﯽ ﺩﺭ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺖ ﺍﺳﺖ. ﺧﻴﻠﯽ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﺑﻪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻳﺎﺩ ﻣﯽ‌ﺩﻫﻨﺪ ﺑﻪ ﺍﺷﻴﺎﻳﯽ ﮐﻪ ﻣﺎﻝ ﺑﺰﺭﮔﺘﺮ ﻫﺎﺳﺖ ﺩﺳﺖ ﻧﺰﻧﻨﺪ، ﻣﺎﺩﺭ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﻌﻠﻢ ﺧﺎﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﭼﻨﻴﻦ ﮐﻨﺪ، ﻭﻟﯽ ﺍﻳﻦ ﮐﺎﺭ ﻣﺎﻧﻊ ﺭﺷﺪ ﮐﻨﺠﮑﺎﻭﯼ ﻭ ﺍﺣﺘﻴﺎﺝ ﺍﺳﺎﺳﯽ ﮐﻮﺩﮎ ﺑﻪ ﺷﻨﺎﺳﺎﻳﯽ ﺍﺷﻴﺎﯼ ﭘﻴﺮﺍﻣﻮﻧﺶ ﻣﯽ‌ﺷﻮﺩ.

ﺑﻪ ﻣﺤﺾ ﺍﻳﻨﮑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺷﺮﻭﻉ ﺑﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ ﺑﺎﻳﺪ ﺩﺍﺧﻞ ﺧﺎﻧﻪ ﺭﺍ ﺑﺎ ﭼﺸﻢ ﺩﻳﮕﺮﯼ ﺑﻨﮕﺮﻳﺪ- ﺑﺎ ﭼﺸﻢ ﮐﺎﺭﺁﻣﻮﺯﯼ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﻳﺎﺩﮔﻴﺮﯼ ﺭﺍﻩ ﺭﻓﺘﻦ ﺍﺳﺖ. ﺑﺎﻳﺪ ﺍﺷﻴﺎﯼ ﺷﮑﺴﺘﻨﯽ ﻭ ﺧﻄﺮﻧﺎﮎ ﺭﺍ ﺍﺯ ﺩﺳﺘﺮﺱ ﺍﻭ ﺧﺎﺭﺝ ﮐﻨﻴﺪ. ﺍﻭ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﺪ ﺑﺪﻭﻥ ﻫﻴﭻ ﺧﻄﺮِ ﺻﺪﻣﻪ ﺯﺩﻥ ﺑﻪ ﺧﻮﺩﺵ ﻭ ﻳﺎ ﺷﮑﺴﺘﻦ ﺍﺷﻴﺎﯼ ﺗﺠﻤﻠﯽ ﺩﺭ ﺧﺎﻧﻪ ﮔﺮﺩﺵ ﮐﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﮔﻠﺪﺍﻥ‌ﻫﺎ ﻭ ﻇﺮﻭﻑ ﭼﻴﻨﯽ ﺭﺍ ﺍﺯ ﺩﺳﺘﺮﺱ ﺍﻭ ﺩﻭﺭ ﮐﻨﻴﺪ. ﺧﺎﻧﻪ‌ﺍﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﻫﺰﺍﺭ ﻧﻮﻉ ﻣﻤﻨﻮﻋﻴﺖ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻣﺴﻠﻤﺎ ﻣﺤﻴﻂ ﺩﻟﭙﺬﻳﺮﯼ ﺑﺮﺍﯼ ﺗﺠﺮﺑﻪ ﺍﻧﺪﻭﺯﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ.

ﺑﺎﻳﺪ ﻫﻮﺷﻴﺎﺭ ﺑﺎﺷﻴﺪ ﻭ ﺍﻭ ﺭﺍ ﺩﺭ ﻣﻘﺎﺑﻞ ﺧﻄﺮﺍﺕ ﭘﻨﻬﺎﻧﯽ ﻣﺤﺎﻓﻈﺖ ﮐﻨﻴﺪ. ﺯﻳﺮﺍ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻫﻨﻮﺯ ﻧﻮﺯﺍﺩ ﺍﺳﺖ. ﺍﻧﮕﻠﻴﺴﯽ‌ﻫﺎ ﻟﻘﺐ «ﻭﻟﮕﺮﺩ ﮐﻮﭼﻮﻟﻮ» ﺭﺍ ﺑﻪ ﮐﻮﺩﮎ ﺍﻳﻦ ﺳﻦ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﮐﻪ ﻟﻘﺐ ﺑﺴﻴﺎﺭ ﻣﻨﺎﺳﺒﯽ ﺍﺳﺖ. ﻗﺪﺭﺕ ﻣﺸﺎﻫﺪﻩ ﻭ ﻗﻀﺎﻭﺕ ﺍﻭ ﺑﺴﻴﺎﺭ ﻣﺤﺪﻭﺩ ﺍﺳﺖ. ﺍﻭ ﻓﺮﻗﯽ ﻣﻴﺎﻥ ﭼﻴﺰﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ ﻭ ﻏﻴﺮﺧﻄﺮﻧﺎﮎ ﻧﻤﯽ‌ﮔﺬﺍﺭﺩ ﻭ ﺗﺤﻘﻴﻘﺎﺗﺶ ﻫﻤﻴﺸﻪ ﺑﻪ ﺩﻫﺎﻧﺶ ﺧﺘﻢ ﻣﯽ‌ﺷﻮﺩ. ﻫﺮ ﭼﻨﺪ ﮐﻪ ﻫﻨﻮﺯ ﻧﻮﺯﺍﺩ ﺍﺳﺖ ﻭﻟﯽ ﻗﺎﺩﺭ ﺑﻪ ﺭﻓﺖ ﻭ ﺁﻣﺪ ﺍﺳﺖ. ﺑﻪ ﻋﻘﻴﺪﻩ ﺻﺎﺣﺐ ﻧﻈﺮﺍﻥ ﺍﮔﺮ ﻭﺍﻟﺪﻳﻦ ﺍﺣﺘﻴﺎﻁ‌ﻫﺎﯼ ﻻﺯﻡ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﺑﻪ ﻋﻤﻞ ﺑﻴﺎﻭﺭﻧﺪ، ﭘﻨﺠﺎﻩ ﺗﺎ ﻧﻮﺩ ﺩﺭﺻﺪ ﺣﻮﺍﺩﺛﯽ ﮐﻪ ﺩﺭﺧﺎﻧﻪ‌ﻫﺎ ﻣﻨﺠﺮ ﺑﻪ ﮐﺸﺘﻪ ﻭ ﻳﺎ ﺯﺧﻤﯽ ﺷﺪﻥ ﺷﺪﻳﺪ ﺍﻃﻔﺎﻝ ﻣﯽ‌ﺷﻮﺩ ﮐﺎﻫﺶ ﻣﯽ‌ﻳﺎﺑﺪ.


 ﭼﮕﻮﻧﻪ ﺧﻄﺮﺍﺕ ﺧﺎﻧﻪ ﺭﺍ ﺍﺯ ﺑﻴﻦ ﺑﺒﺮﻳﻢ
ﺗﻮﺻﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﺑﻪ ﻋﻮﺽ ﺗﻬﻴﻪ ﻓﻬﺮﺳﺘﯽ ﺍﺯ ﮐﺎﺭﻫﺎﯼ ﻻﺯﻡ  ﺑﺮﺍﯼ ﮐﻨﺘﺮﻝ،  ﺑﮑﻮﺷﻴﺪ ﻋﺎﺩﺕ ﮐﻨﻴﺪ ﺍﺯ ﺩﻳﺪ ﻳﮏ ﮐﻮﭼﻮﻟﻮ ﺧﺎﻧﻪ‌ﺗﺎﻥ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ ﻧﻪ ﺑﺎ ﻓﮑﺮ ﻳﮏ ﺁﺩﻡ ﺑﺎﻟﻎ. ﺑﻪ ﻧﻈﺮ ﺷﻤﺎ ﻗﺮﺹ‌ﻫﺎﯼ ﺁﺳﭙﻴﺮﻳﻦ ﮐﻪ ﺩﺭ ﺟﻌﺒﻪ ﺩﺍﺭﻭﻫﺎﺳﺖ، ﻣﺴﮑﻨﯽ ﺍﺳﺖ ﮐﻪ ﻫﻨﮕﺎﻡ ﺳﺮﺩﺭﺩ ﻣﯽ‌ﺧﻮﺭﻳﺪ. ﻭﻟﯽ ﺑﻪ ﻧﻈﺮ ﻧﻮﺯﺍﺩﺗﺎﻥ ﺁﻧﻬﺎ ﺁﺏ‌ﻧﺒﺎﺕ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﻫﺴﺘﻨﺪ. ﺍﻭ ﺑﺎ ﺧﻮﺭﺩﻥ ﻳﮏ ﺑﺴﺘﻪ ﺍﺯ ﺍﻳﻦ «ﺁﺏ‌ﻧﺒﺎﺕ‌ﻫﺎ» ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﮐﺸﺘﻦ ﺑﺪﻫﺪ.

ﺑﺮﺭﺳﯽ ﺩﻗﻴﻖ ﻭ ﻣﻨﻈﻤﯽ ﺍﺯ ﺗﻤﺎﻡ ﺧﺎﻧﻪ‌ﺗﺎﻥ ﺑﮑﻨﻴﺪ (ﺣﻴﺎﻁ ﻭ ﺯﻳﺮﺯﻣﻴﻦ ﻭ ﮔﺎﺭﺍﮊ ﻫﻢ ﺍﺯ ﺁﻥ ﺟﻤﻠﻪ)، ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﺯ ﺩﻳﺪ ﻳﮏ ﮐﻮﺩﮎ ﺁﻥ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ. ﻓﮑﺮ ﮐﻨﻴﺪ «ﭼﻴﺰﯼ ﺍﻳﻦ ﺟﺎ ﻧﻴﺴﺖ ﮐﻪ ﺍﻭﻥ ﺑﺘﻮﻧﻪ ﺑﺮﺩﺍﺭﻩ؟ ﭼﻴﺰﯼ ﮐﻪ ﺑﺘﻮﻧﻪ ﺑﺮ ﺩﺍﺭﻩ ﻭ ﺑﺬﺍﺭﻩ ﺩﻫﻨﺶ ﻭ ﺑﻪ ﺧﻮﺩﺵ ﺻﺪﻣﻪ ﺑﺰﻧﻪ» ﻭ ﺗﻤﺎﻡ ﺁﻧﻬﺎ ﺭﺍ ﺍﺯ ﺩﺳﺘﺮﺱ ﺍﻭ ﺩﻭﺭ ﮐﻨﻴﺪ. ﺩﺭ ﺯﻳﺮ ﻣﺜﺎﻝ‌ﻫﺎﻳﯽ ﺍﺯ ﺁﻧﭽﻪ ﺯﻣﺎﻥ ﺗﻤﻴﺰ ﮐﺮﺩﻥ ﺧﺎﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﻣﯽ‌ﺩﻫﻴﻢ.

ﻫﺮ ﭼﻴﺰ ﺳﻤﯽ‌ﺍﯼ ﺭﺍ ، ﻳﻌﻨﯽ ﺗﻤﺎﻡ ﺩﺍﺭﻭﻫﺎ ﻭ ﻣﺤﺼﻮﻻﺕ ﺳﻤﯽ، ﺩﻭﺭ ﺍﺯ ﺩﺳﺘﺮﺱ ﮐﻮﺩﮎ ﺑﮕﺬﺍﺭﻳﺪ. ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺯﻳﺮ ﺩﺳﺘﺸﻮﻳﯽ، ﻇﺮﻑ‌ﺷﻮﻳﯽ ﺁﺷﭙﺰﺧﺎﻧﻪ، ﺭﻭﯼ ﺯﻣﻴﻦ ﮔﺎﺭﺍﮊ ﻭ ﻏﻴﺮﻩ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ. ﺁﻳﺎ ﻧﻮﺯﺍﺩﺗﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻳﮑﯽ ﺍﺯ ﻣﺤﺼﻮﻻﺕ ﺯﻳﺮ ﺩﺳﺘﺮﺳﯽ ﭘﻴﺪﺍ ﮐﻨﺪ: ﺁﻣﻮﻧﻴﺎﮎ ﺟﻬﺖ ﺗﻤﻴﺰ ﮐﺮﺩﻥ ﺧﺎﻧﻪ، ﮐﻠﺮ، ﺳﻢ ﺿﺪﻣﻮﺭﭼﻪ، ﭘﻮﻟﻴﺶ ﺍﺗﻮﻣﻮﺑﻴﻞ، ﺳﻴﮕﺎﺭ، ﺿﺪﻳﺦ، ﻭﺍﻳﺘﮑﺲ، ﺭﻧﮓ ﺳﺮﺑﯽ، ﻭﺍﮐﺲ ﭘﺎﺭﮐﺖ، ﺳﻢ ﮔﻴﺎﻫﯽ، ﻭﺍﮐﺲ ﻣﺒﻞ، ﺩﺍﺭﻭﻫﺎﯼ ﺿﺪﺣﺸﺮﺍﺕ ﻭ ﺿﺪﺑﻴﺪ؟

ﺟﻌﺒﻪ ﺩﺍﺭﻭﺧﺎﻧﻪ ﺍﺯ ﻧﻈﺮ ﻣﻮﺍﺩ ﺳﻤﯽ ﺑﺴﻴﺎﺭ ﻏﻨﯽ ﺍﺳﺖ. ﺩﺍﺭﻭﻫﺎﻳﯽ ﮐﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺑﻪ ﻣﻘﺪﺍﺭ ﮐﻢ ﺑﯽ‌ﺧﻄﺮ ﺍﺳﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﮐﻮﺩﮐﺎﻥ ﺧﺮﺩﺳﺎﻝ ﻣﻬﻠﮏ ﺑﺎﺷﺪ. ﺩﺭ ﺩﺍﺭﻭﺧﺎﻧﻪ ﺷﻤﺎ ﺣﺘﻤﺎ ﺁﺳﭙﻴﺮﻳﻦ ﻳﺎ ﺩﺍﺭﻭﻫﺎﯼ ﺧﻮﺍﺏ‌ﺁﻭﺭ، ﺷﺮﺑﺖ ﻫﺎﯼ ﮐﺪﺋﻴﻦ‌ﺩﺍﺭ، ﺩﺍﺭﻭﻫﺎﯼ ﺁﺭﺍﻣﺶ‌ﺑﺨﺶ ﻭ ﻫﻤﻴﻦ ﻃﻮﺭ ﺩﺍﺭﻭﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﻧﺴﺨﻪ‌ﻫﺎﯼ ﻗﺪﻳﻤﯽ ﺑﺎﻗﯽ ﻣﺎﻧﺪﻩ‌ﺍﻧﺪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺗﻨﻬﺎ ﺭﺍﻩ ﺣﻔﺎﻇﺖ ﻣﻄﻤﺌﻦ ﺍﺯ ﺩﺍﺭﻭﺧﺎﻧﻪ (ﻳﺎ ﻫﺮ ﮐﻤﺪ ﻭ ﮔﻨﺠﻪ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻮﺍﺩ ﺳﻤﯽ ﺭﺍ ﻧﮕﻬﺪﺍﺭﯼ ﻣﯽ‌ﮐﻨﻴﺪ) ﻗﻔﻞ ﮐﺮﺩﻥ ﺁﻥ ﺍﺳﺖ.

ﺑﻪ ﻋﻨﻮﺍﻥ ﻭﺍﻟﺪﻳﻦ ﮐﻮﺩﮐﯽ ﺧﺮﺩﺳﺎﻝ ﺑﺎﻳﺪ ﭼﺸﻤﺎﻥ ﺗﻴﺰﺑﻴﻨﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﻫﺮ ﭼﻨﺪ ﮔﺎﻩ ﻳﮑﺒﺎﺭ ﺭﻭﯼ ﺯﻣﻴﻦ ﺍﺗﺎﻕ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺩﻗﺖ ﻧﮕﺎﻩ ﮐﻨﻴﺪ ﻭ ﺑﺒﻴﻨﻴﺪ ﮐﻪ ﻣﻴﺦ ﻳﺎ ﺗﻴﻎ ﺭﻳﺶ ﺗﺮﺍﺵ ﻳﺎ ﺗﮑﻪ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺷﮑﺴﺘﻪ ﻳﺎ ﺳﻨﺠﺎﻕ ﻗﻔﻠﯽ ﺑﺎﺯ ﻳﺎ ﻫﺮ ﺷﻴﺊ ﺧﻄﺮﻧﺎﮎ ﺩﻳﮕﺮﯼ ﺭﻭﯼ ﺯﻣﻴﻦ ﻧﻴﻔﺘﺎﺩﻩ ﺑﺎﺷﺪ.

ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﮐﻪ ﮐﻮﺩﮎ ﺑﺎ ﺁﻧﻬﺎ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ ﻧﻴﺰ ﺑﺎﻳﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺑﺰﺭﮒ ﺑﺎﺷﻨﺪ ﺗﺎ ﻧﺘﻮﺍﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﺑﺒﻠﻌﺪ (ﻣﻮﺍﻇﺐ ﺁﺏ ﻧﺒﺎﺕ‌ﻫﺎ، ﺳﻨﺠﺎﻕ ﻳﺎ ﺳﻮﺯﻥ، ﺩﺍﻧﻪ‌ﻫﺎﯼ ﻣﺮﻭﺍﺭﻳﺪ ﻳﺎ ﺍﺷﻴﺎﻳﯽ ﻧﻈﻴﺮ ﺍﻳﻦ‌ﻫﺎ ﺑﺎﺷﻴﺪ.) ﻣﻮﺍﺩ ﺧﻮﺭﺍﮐﯽ ﻧﻈﻴﺮ ﻓﻨﺪﻕ، ﭘﺴﺘﻪ، ﺫﺭﺕ ﺑﻮ ﺩﺍﺩﻩ ﻳﺎ ﭼﻴﺰﻫﺎﯼ ﻧﻈﻴﺮ ﺁﻥ ﺭﺍ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﻮﺟﺐ ﺧﻔﮕﯽ ﮐﻮﺩﮎ  ﺷﻮﻧﺪ ﻧﻴﺰ ﺑﻪ ﺍﻭ ﻧﺪﻫﻴﺪ.

ﺑﻪ ﻫﻨﮕﺎﻡ ﺁﺷﭙﺰﯼ ﺩﺳﺘﻪ ﺩﻳﮓ ﻭ ﻣﺎﻫﯽ‌ﺗﺎﺑﻪ ﺭﺍ ﺑﻪ ﻃﺮﻓﯽ ﺑﮕﺬﺍﺭﻳﺪ ﮐﻪ ﺍﻭ ﻧﺘﻮﺍﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﺑﮕﻴﺮﺩ ﻭ ﺑﮑﺸﺪ ﻭ ﺭﻭﯼ ﺧﻮﺩﺵ ﺑﺮﻳﺰﺩ ﻭ ﺑﺴﻮﺯﺍﻧﺪ. ﻇﺮﻑ ﺁﺷﻐﺎﻝ ﺯﻳﺮ ﺩﺳﺘﺸﻮﻳﯽ ﻳﺎ ﮐﺎﻏﺬﻫﺎﯼ ﺑﺎﻃﻠﻪ ﺭﺍ ﻧﻴﺰ ﺑﺎﻳﺪ ﺧﻴﻠﯽ ﻣﻮﺍﻇﺒﺸﺎﻥ ﺑﻮﺩ، ﺯﻳﺮﺍ ﻣﺎ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﻓﮑﺮ ﮐﻨﻴﻢ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﺩﺭ ﺁﻧﻬﺎ ﻣﯽ‌ﺭﻳﺰﻳﻢ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺑﺴﻴﺎﺭ ﺧﻄﺮﻧﺎﮎ ﺑﺎﺷﻨﺪ.

ﺑﺮﻕ ﻣﺴﻠﻤﺎ ﻣﻨﺒﻊ ﺩﻳﮕﺮ ﺧﻄﺮ ﺍﺳﺖ. ﭘﺮﻳﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﻫﻤﻴﺸﻪ ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﯽ‌ﮐﻨﻴﺪ ﺑﺎ ﺳﺮﭘﻮﺵ ﻣﺨﺼﻮﺹ ﺑﭙﻮﺷﺎﻧﻴﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﺎﻧﻊ ﺍﺯ ﺍﻳﻦ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﮐﻮﺩﮎ ﮐﺸﻒ ﮐﻨﺪ ﺍﮔﺮ ﺳﻨﺠﺎﻕ ﺭﺍ ﺩﺍﺧﻞ ﺳﻮﺭﺍﺥ ﻫﺎﯼ ﻳﮏ ﭘﺮﻳﺰ ﺑﺮﻕ ﮐﻨﺪ ﭼﻪ ﺍﺗﻔﺎﻗﯽ ﻣﯽ‌ﺍﻓﺘﺪ. ﺭﺍﺩﻳﻮ ﻳﺎ ﺷﻮﻓﺎﮊ ﺑﺮﻗﯽ ﻳﺎ ﻫﺮ ﺩﺳﺘﮕﺎﻩ ﺑﺮﻗﯽ ﺩﻳﮕﺮﯼ ﺭﺍ ﻫﻢ ﻫﻴﭽﮕﺎﻩ ﻧﺰﺩﻳﮏ ﻭﺍﻥ ﺣﻤﺎﻡ ﻧﮕﺬﺍﺭﻳﺪ.

ﺍﻧﺒﺎﺭ ﻣﺤﻠﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺁﻥ ﺧﻄﺮﺍﺕ ﻣﺮﮒ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭِ ﺁﻥ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﻗﻔﻞ ﺑﺎﺷﺪ. ﺍﮔﺮ ﻧﻪ ﮐﻮﺩﮎ ﺑﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﺧﻄﺮﻧﺎﮐﯽ ﭼﻮﻥ ﺍﺭﻩ، ﺭﻧﮓ، ﭼﺎﻗﻮ ﻭ ﻏﻴﺮﻩ ﺑﺎﺯﯼ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺍﮔﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺮﺍﯼ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺑﻪ ﺣﻴﺎﻁ ﺧﺎﻧﻪ ﻣﯽ‌ﺭﻭﺩ ﺩﺭ ﺣﻴﺎﻁ ﺧﺎﻧﻪ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﺑﺴﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﺍﻭ ﻧﺘﻮﺍﻧﺪ ﺑﺎ ﺑﻴﺮﻭﻥ ﺭﻓﺘﻦ ﺍﺯ ﺣﻴﺎﻁ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﻌﺮﺽ ﺧﻄﺮﺍﺕ ﺍﺣﺘﻤﺎﻟﯽ ﻗﺮﺍﺭ ﺩﻫﺪ.

ﻫﻴﭻ‌ﮔﺎﻩ ﻭ ﻫﻴﭻ‌ﮔﺎﻩ ﻓﺮﺯﻧﺪ ﮐﻮﭼﮑﺘﺎﻥ ﺭﺍ ﺑﺎ ﻓﺮﺯﻧﺪ ﺑﺰﺭﮒ‌ﺗﺮ ﺩﺭ ﻣﺎﺷﻴﻦ ﻧﮕﺬﺍﺭﻳﺪ ﮐﻪ «ﻓﻘﻂ ﻳﮏ ﺩﻗﻴﻘﻪ» ﻣﯽ‌ﺭﻭﻡ ﺩﺭ ﺍﻳﻦ ﻣﻐﺎﺯﻩ ﺧﺮﻳﺪﯼ ﺑﮑﻨﻢ، ﻫﻴﭻ‌ﮔﺎﻩ!.... ﻫﻤﻴﺸﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﻫﻤﺮﺍﻫﺘﺎﻥ ﺑﺒﺮﻳﺪ.

ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺷﻮﻫﺮﺗﺎﻥ ﻳﮏ ﻭﺿﻌﻴﺖ ﺍﻭﺭﮊﺍﻧﺲ ﺍﺣﺘﻤﺎﻟﯽ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ. ﺣﻘﻴﻘﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻭﺿﻌﻴﺖ ﺍﻭﺭﮊﺍﻧﺲ ﻫﻤﻴﺸﻪ ﺯﻣﺎﻧﯽ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ ﮐﻪ ﺍﻧﺴﺎﻥ ﺍﻧﺘﻈﺎﺭﺵ ﺭﺍ ﻧﺪﺍﺭﺩ. ﺍﮐﺜﺮ ﻣﺎ ﺩﺭﺑﺎﺭﻩ ﺍﻳﻨﮑﻪ ﺍﮔﺮ ﻭﺿﻌﻴﺖ ﺍﺿﻄﺮﺍﺭﯼ ﭘﻴﺶ ﺑﻴﺎﻳﺪ ﭼﻪ ﮐﺎﺭﻫﺎﻳﯽ ﺑﺎﻳﺪ ﺑﮑﻨﻴﻢ ﻓﮑﺮ ﻧﮑﺮﺩﻩ‌ﺍﻳﻢ. ﻣﺜﻼ ﺍﮔﺮ ﺣﺎﺩﺛﻪ‌ﺍﯼ ﭘﻴﺶ ﺑﻴﺎﻳﺪ ﺑﻪ ﺩﮐﺘﺮﺗﺎﻥ ﺗﻠﻔﻦ ﻣﯽ‌ﮐﻨﻴﺪ ﻳﺎ ﺑﻪ ﻗﺴﻤﺖ ﺍﻭﺭﮊﺍﻧﺲ ﻧﺰﺩﻳﮏ‌ﺗﺮﻳﻦ ﺑﻴﻤﺎﺭﺳﺘﺎﻥ ﻣﯽ‌ﺭﻭﻳﺪ؟ ﻫﻤﻴﺸﻪ ﺷﻤﺎﺭﻩ ﺗﻠﻔﻦ ﺩﮐﺘﺮ، ﺑﻴﻤﺎﺭﺳﺘﺎﻥ، ﺁﻣﺒﻮﻻﻧﺲ ﻭ ﺁﺗﺶ‌ﻧﺸﺎﻧﯽ ﻭ ﺩﻳﮕﺮ ﺷﻤﺎﺭﻩ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﺿﺮﻭﺭﯼ ﺭﺍ ﻧﺰﺩﻳﮏ ﺗﻠﻔﻦ ﻭ ﺩﺭ ﺩﺍﺧﻞ ﺍﺗﻮﻣﺒﻴﻞ‌ﺗﺎﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺗﻤﺎﻡ ﺍﻓﺮﺍﺩ ﺧﺎﻧﻮﺍﺩﻩ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻨﺪ ﭼﻪ ﮐﺎﺭﯼ ﺑﺎﻳﺪ ﺑﮑﻨﻨﺪ. ﻻﺯﻡ ﻧﻴﺴﺖ ﺍﻳﻦ ﻭﺿﻌﻴﺖ ﺭﺍ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻥ ﺑﺰﺭﮔﺘﺮ‌ﺗﺎﻥ ﻧﮕﺮﺍﻥ‌ﮐﻨﻨﺪﻩ ﺟﻠﻮﻩ ﺩﻫﻴﺪ: ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﺎ ﻭﺍﻗﻊ‌ﺑﻴﻨﯽ ﺑﺮﺍﯼ ﺁﻥ‌ﻫﺎ ﺗﺸﺮﻳﺢ ﮐﻨﻴﺪ.

ﺟﻌﺒﻪ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺭﺍﻫﻤﻴﺸﻪ ﺩﺭ ﺩﺳﺘﺮﺱ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺟﻌﺒﻪ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﻳﮏ ﺩﺍﺭﻭﯼ ﺿﺪﺳﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺍﺯ ﺩﮐﺘﺮﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﺟﻌﺒﻪ ﻧﮕﻬﺪﺍﺭﯼ ﮐﻨﻴﺪ.
