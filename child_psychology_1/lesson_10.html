ﺩﺭﺱ ﻫﺸﺘﻢ- ﮐﺘﺎﺏ ﻭ ﺑﺎﺯﯼ ﻧﺎﻡ ﮔﺬﺍﺭﯼ ﺍﺷﻴﺎ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺩﻳﺪﻳﻢ ﭼﻪ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﮑﻨﻴﻢ ﮐﻪ ﮐﻮﺩﮐﯽ ﻧﻮ ﭘﺎ ﺩﺭ ﺧﺎﻧﻪ ﺩﺭ ﺍﻣﻨﻴﺖ ﺑﻪ ﺍﮐﺘﺸﺎﻑ ﺩﻧﻴﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺑﭙﺮﺩﺍﺯﺩ. ﺍﮐﻨﻮﻥ ﺑﺎﻳﺪ ﻗﺪﻣﯽ ﻓﺮﺍﺗﺮ ﺑﺮﺩﺍﺭﻳﺪ ﻭ ﮐﺎﺭﻫﺎﯼ ﺟﺪﻳﺪﯼ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ، ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﻣﻮﺟﺐ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﺷﻮﻧﺪ. ﺩﺭ ﺍﺩﺍﻣﻪ ﺑﻪ ﺑﺮﺧﯽ ﺍﺯ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﻴﻢ.
 

ﮐﺘﺎﺏ ﺍﺯ ﺍﻳﻦ ﭘﺲ ﻳﮑﯽ ﺍﺯ ﻭﺳﺎﻳﻞ ﺑﺎﺯﯼ ﺿﺮﻭﺭﯼ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻣﻄﺎﻟﻌﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺷﺎﻣﻞ ﻧﮕﺎﻩ ﮐﺮﺩﻥ ﻣﺠﻠﻪ­‌ﻫﺎﯼ ﮐﻬﻨﻪ ﻭ ﮐﺎﺗﺎﻟﻮﮒ­‌ﻫﺎﻳﯽ ﺑﺎﺷﺪ ﮐﻪ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺁﻧﻬﺎ ﺭﺍ ﮐﺘﺎﺏ ﻧﻤﯽ­‌ﻧﺎﻣﻨﺪ.

ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻣﯽ­ﺗﻮﺍﻧﻴﺪ ﺑﺎﺯﯼ ﻧﺎﻡ ﮔﺬﺍﺭﯼ ﺭﻭﯼ ﺍﺷﺨﺎﺹ ﻭ ﺍﺷﻴﺎﺀ ﺭﺍ ﺩﻧﺒﺎﻝ ﮐﻨﻴﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﮐﻪ ﺑﺎ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻳﮏ ﺗﺼﻮﻳﺮ ﻧﺎﻡ ﺁﻥ ﺭﺍ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﺑﺮﺍﻳﺶ ﺗﻠﻔﻆ ﮐﻨﻴﺪ.

ﺍﮐﻨﻮﻥ ﻣﯽ‌­ﺗﻮﺍﻧﻴﺪ ﺷﺮﻭﻉ ﺑﻪ ﺧﻮﺍﻧﺪﻥ ﺩﺍﺳﺘﺎﻥ ﮐﻨﻴﺪ. ﺍﻭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺑﺮﺍﻳﺶ ﻗﺼﻪ ﺑﮕﻮﻳﻨﺪ.
ﮐﺘﺎﺏ­‌ﻫﺎ ﺑﺎﻳﺪ ﺍﺯ ﺟﻨﺲ ﭘﺎﺭﭼﻪ ﻳﺎ ﻣﻘﻮﺍ ﺑﺎﺷﻨﺪ ﺯﻳﺮﺍ ﮐﻮﺩﮎ ﻫﻨﻮﺯ ﮔﺮﺍﻳﺶ ﺑﻪ ﭘﺎﺭﻩ ﮐﺮﺩﻥ ﺻﻔﺤﺎﺕ ﺁﻥ ﺩﺍﺭﺩ. ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﮐﺘﺎﺏ ﺩﺍﺳﺘﺎﻥ­‌ﻫﺎﯼ  ﻣﺼﻮﺭ ﺩﺭ ﺳﻄﺢ ﺧﻮﺩﺵ ﺭﺍ ﺑﺴﻴﺎﺭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ. ﺍﻳﻦ ﮔﻮﻧﻪ ﮐﺘﺎﺏ­‌ﻫﺎ ﺩﺭ ﺗﻤﺎﻡ ﺳﺎﻝ­‌ﻫﺎﯼ ﻗﺒﻞ ﺍﺯ ﻣﺪﺭﺳﻪ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ­‌ﮔﻴﺮﻧﺪ. ﻭﺍﻟﺪﻳﻦ ﻣﯽ‌­ﺗﻮﺍﻧﻨﺪ ﺍﻭﻝ ﺗﺼﻮﻳﺮﻫﺎ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﻨﺪ ﻭ ﺑﻌﺪ ﺍﺳﻢ ﺁﻧﻬﺎ ﺭﺍ ﺑﮕﻮﻳﻨﺪ. ﺑﻌﺪ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﻮﺩﮎ ﺑﺰﺭﮔﺘﺮ ﺷﺪ ﻣﯽ­‌ﺗﻮﺍﻥ ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﻗﺒﻴﻞ ﮐﺮﺩ، ﺳﮓ، ﺗﺮﺍﮐﺘﻮﺭ، ﭘﻴﺮﺍﻫﻦ ﻭ ﻏﻴﺮﻩ ﺭﺍ ﭘﻴﺪﺍ ﮐﻦ. ﮐﻤﯽ ﺑﻌﺪ ﺗﺼﻮﻳﺮﻫﺎ ﺭﺍ ﺑﻪ ﮐﻮﺩﮎ ﻧﺸﺎﻥ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺩ ﻭ ﺍﻭ ﻧﺎﻡ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺧﻮﺍﻫﺪ ﮔﻔﺖ.

ﺑﺮﺍﯼ ﺭﺷﺪ ﺯﺑﺎﻥ ﮐﻮﺩﮎ، ﻧﻘﺶ ﻣﺎﺩﺭ ﺑﺴﻴﺎﺭ ﻣﻬﻢ‌ﺗﺮ ﺍﺯ ﻧﻘﺶ ﮐﺘﺎﺏ ﺍﺳﺖ. ﺑﺎﺯﯼ ﻧﺎﻡ‌ﮔﺬﺍﺭﯼ ﺭﻭﯼ ﺍﺷﻴﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﺗﺎﻥ ﺭﺍ ﮐﻪ ﺩﺭ ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﮐﻮﺩﮐﯽ ﺷﺮﻭﻉ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺍﺩﺍﻣﻪ ﺩﻫﻴﺪ. ﺍﻳﻦ ﺑﺎﺯﯼ ﺭﺍ ﻫﻤﻪ ﺟﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮑﻨﻴﺪ. ﻣﺜﻼ ﻭﻗﺘﯽ ﺩﺭ ﻣﺎﺷﻴﻦ ﻫﺴﺘﻴﺪ ﺍﺳﻢ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺍﺯ ﮐﻨﺎﺭﺵ ﻋﺒﻮﺭ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﮕﻮﻳﻴﺪ: ﮐﺎﻣﻴﻮﻥ، ﺗﺮﺍﮐﺘﻮﺭ، ﺧﺎﻧﻪ، ﺍﺗﻮﺑﻮﺱ. ﻭﻗﺘﯽ ﺑﺎ ﺍﻭ ﺑﻪ ﺧﺮﻳﺪ ﻣﯽ‌ﺭﻭﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮕﻮﻳﻴﺪ: ﺷﻴﺮ، ﭘﺮﺗﻘﺎﻝ، ﺳﻴﺐ، ﻣﻮﺯ، ﻣﺎﻫﯽ، ﺑﻴﺴﮑﻮﻳﺖ...

ﺭﺷﺪ ﺯﺑﺎﻥ ﺩﺭ ﮐﻮﺩﮎ ﺷﺎﻣﻞ ﺩﻭ ﻋﻨﺼﺮ ﺍﺳﺎﺳﯽ ﺍﺳﺖ: ﺯﺑﺎﻥ ﺍﻧﻔﻌﺎﻟﯽ (ﻳﻌﻨﯽ ﻓﻬﻤﻴﺪﻥ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺍﻭ ﻣﯽ‌­ﮔﻮﻳﻴﺪ) ﻭ ﺯﺑﺎﻥ ﻓﻌﺎﻝ (ﻳﻌﻨﯽ ﺻﺤﺒﺖ ﮐﺮﺩﻥ.)

ﺩﺭ ﺳﻦ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺯﺑﺎﻥ ﺍﻧﻔﻌﺎﻟﯽ ﺗﺴﻠﻂ ﺩﺍﺭﺩ. ﺩﺭ ﻭﺍﻗﻊ ﺍﺭﺯﺷﻴﺎﺑﯽ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺑﻪ ﻋﻨﻮﺍﻥ ﺍﻧﻔﻌﺎﻟﯽ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ ﺯﻳﺮﺍ ﺫﻫﻦ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﺯﺑﺎﻥ ﻣﺎﺩﺭﯼ‌­ﺍﺵ ﺭﺍ ﮐﻪ ﺍﻃﺮﺍﻓﻴﺎﻥ ﺑﻪ ﺁﻥ ﺻﺤﺒﺖ ﻣﯽ­‌ﮐﻨﻨﺪ ﻓﻬﻤﻴﺪﻩ ﻭ ﺁﻥ ﺭﺍ ﺗﻘﻠﻴﺪ ﮐﻨﺪ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ ﺭﺷﺪ ﺯﺑﺎﻥ ﮐﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﺑﺎ ﺻﺪﺍ ﺩﺭ ﺁﻭﺭﺩﻥ ﺍﺯ ﺧﻮﺩﺵ ﺷﺮﻭﻉ ﻣﯽ‌­ﺷﻮﺩ. ﺑﻌﺪﻫﺎ، ﺍﻭ ﺷﺮﻭﻉ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺯﺑﺎﻥ ﺧﺎﺹ ﺧﻮﺩ ﻣﯽ­‌ﮐﻨﺪ ﮐﻪ ﻋﺒﺎﺭﺕ ﺍﺳﺖ ﺍﺯ ﺗﻘﻠﻴﺪ ﮐﺮﺩﻥ ﻟﺤﻦ ﻭ ﺁﻫﻨﮓ ﺻﺪﺍﯼ ﺑﺰﺭﮔﺘﺮﻫﺎ. ﺍﺯ ﺍﻳﻦ ﺯﻣﺎﻥ ﺑﻪ ﻣﺎﻧﻨﺪ ﺍﻭ ﺑﺎ ﻫﻤﺎﻥ ﻟﺤﻦ ﺧﻮﺩﺵ ﺻﺤﺒﺖ ﮐﻨﻴﺪ، ﺩﺭﺳﺖ ﻣﺜﻞ ﺍﻳﻨﮑﻪ ﻳﮏ ﻣﺤﺎﻭﺭﻩ ﻭﺍﻗﻌﯽ ﺑﺎ ﺍﻭ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌­ﮐﻨﻴﺪ. ﺍﻳﻦ ﻣﮑﺎﻟﻤﻪ ﻓﻘﻂ ﺗﻔﺮﻳﺢ ﻧﻴﺴﺖ، ﺑﺎ ﺍﻳﻦ ﮐﺎﺭﺗﺎﻥ ﺑﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﺍﻭ ﮐﻤﮏ ﻣﯽ­‌ﮐﻨﻴﺪ.

ﺗﻘﺮﻳﺒﺎ ﺑﺎ ﺷﺮﻭﻉ ﺍﻳﻦ ﻣﺮﺣﻠﻪ، ﺍﻭﻟﻴﻦ ﺟﻤﻼﺕ ﻳﮏ ﮐﻠﻤﻪ‌­ﺍﯼ ﺑﺮ ﺯﺑﺎﻥ ﮐﻮﺩﮎ ﺟﺎﺭﯼ ﻣﯽ‌­ﺷﻮﻧﺪ. ﺍﻳﻦ ﻟﻐﺎﺕِ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﻣﻌﻨﺎﯼ ﻳﮏ ﺟﻤﻠﻪ ﮐﺎﻣﻞ ﺭﺍ ﺩﺍﺭﻧﺪ: «ﺑﻴﺴﮑﻮﻳﺖ (ﻣﻦ ﻳﮏ ﺑﻴﺴﮑﻮﻳﺖ ﻣﯽ‌­ﺧﻮﺍﻡ» ﻳﺎ «ﺩَﺩَ (ﻣﻦ ﻣﯽ‌ﺧﻮﺍﻡ ﺑﺮﻡ ﺑﻴﺮﻭﻥ » ﻳﺎ « ﺑﻐﻞ (ﺑﻐﻠﻢ ﮐﻦ)». ﺧﻴﻠﯽ ﮐﻢ ﻫﺴﺘﻨﺪ ﭘﺪﺭ  ﻣﺎﺩﺭﻫﺎﻳﯽ ﮐﻪ ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﮐﻪ ﺑﺎ ﻳﮏ ﮐﻠﻤﻪ ﺑﺮﻗﺮﺍﺭ ﻣﯽ­‌ﺷﻮﺩ ﺣﺴﺎﺱ ﻧﺒﺎﺷﻨﺪ. ﺩﺭ ﺳﻦ ﻫﺠﺪﻩ ﻣﺎﻫﮕﯽ ﮐﻮﺩﮎ ﮐﻠﻤﺎﺕ ﺭﺍ ﺟﺎﻳﮕﺰﻳﻦ ﺯﺑﺎﻥ ﺧﺎﺹ ﺧﻮﺩ ﻣﯽ‌­ﮐﻨﺪ. ﺍﻭ ﺍﺯ ﺍﻳﻦ ﭘﺲ ﻣﯽ­‌ﺗﻮﺍﻧﺪ ﺑﺎ ﺯﺑﺎﻧﺶ ﺭﺍﺑﻄﻪ­‌ﺍﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺮﻗﺮﺍﺭ ﮐﻨﺪ.

ﮐﻠﻤﺎﺗﯽ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻣﯽ‌­ﺩﺍﻧﺪ ﺑﻴﻦ ۳-۴ ﺗﺎ ۱۰۰ ﮐﻠﻤﻪ ﻣﺘﻐﻴﺮ ﺍﺳﺖ. ﺍﻭ ﻧﺎﮔﻬﺎﻥ ﺷﺮﻭﻉ ﺑﻪ ﺩﺭﺳﺖ ﮐﺮﺩﻥ ﺟﻤﻼﺕ ﺩﻭ ﮐﻠﻤﻪ‌­ﺍﯼ ﻣﯽ‌ﮐﻨﺪ، ﻣﺜﻞ «ﻧﮕﺎﻩ ﮐﻦ» ﻳﺎ «ﺑﻴﺎ ﺍﻳﻨﺠﺎ» ﻳﺎ «ﺑﺮﻳﻢ ﭘﺎﺭﻙ » ﻳﺎ «ﺑﻴﺴﮑﻮﻳﺖ ﺑﺪﻩ».‌

ﺗﻌﺪﺍﺩ ﮐﻠﻤﺎﺕ ﻭ ﭼﮕﻮﻧﮕﯽ ﺭﺷﺪ ﺯﺑﺎﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩ ﺑﺴﺘﮕﯽ ﺑﻪ ﺍﻳﻦ ﺩﺍﺭﺩ ﮐﻪ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﻭ ﺻﺤﺒﺖ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ ﻭ ﭼﻘﺪﺭ ﺑﺎ ﺍﻭ ﺑﺎﺯﯼ ﮐﻠﻤﺎﺕ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ. ﻫﻴﭻ ﮐﺲ ﺑﻬﺘﺮ ﺍﺯ ﺷﻤﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﺍﻭ ﮐﻤﮏ ﮐﻨﺪ ﻭ ﻫﻴﭻ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻳﺎ ﮐﺎﻣﭙﻴﻮﺗﺮﯼ ﻧﻴﺰ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﺧﺘﺮﺍﻉ ﮐﺮﺩ ﮐﻪ ﺑﺘﻮﺍﻧﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻣﻮﺛﺮ ﺑﺎﺷﺪ.
 

ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﻭ ﺗﻨﺒﻴﻪ

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ، ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﺭﺍ ﻣﺠﺒﻮﺭ ﻧﮑﻨﻴﺪ ﺩﺭ ﻣﺤﻴﻄﯽ ﺯﻧﺪﮔﯽ ﮐﻨﺪ ﮐﻪ ﻣﺨﺼﻮﺹ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﮐﻮﺩﮎ ﻫﻤﻴﺸﻪ ﺑﺎ ﻳﮏ ﺳﺮﯼ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﻭ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﺭﻭﺑﺮﻭ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺍﻣﺎ ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﻧﺒﺎﻳﺪ ﻫﻴﭻ ﻧﻮﻉ ﻣﺤﺪﻭﺩﻳﺖ ﻭ ﻣﻤﻨﻮﻋﻴﺘﯽ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩ. ﻣﺴﻠﻤﺎ ﺑﺎﻳﺪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﮐﺮﺩ. ﻭﻟﯽ ﺍﻳﻦ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﺑﺎﻳﺪ ﺑﻪ ﺁﺗﺶ ﺑﺨﺎﺭﯼ ﻭ ﺍﺟﺎﻕ ﻏﺬﺍﭘﺰﯼ ﻣﺤﺪﻭﺩ ﺷﻮﺩ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺍﺯ ﻣﺤﻴﻂ ﺣﺬﻑ ﮐﺮﺩ. ﻭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻣﻤﻨﻮﻋﻴﺘﯽ ﺑﮕﺬﺍﺭﻳﺪ ﺑﻪ «ﻧﻪ ﻧﻪ»، «ﺩﺳﺖ ﻧﺰﻥ»  ﮔﻔﺘﻦ ﻗﻨﺎﻋﺖ ﻧﮑﻨﻴﺪ. ﺍﻭ ﻧﻤﯽ‌ﻓﻬﻤﺪ ﮐﻪ ﺍﻳﻦ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﻓﻘﻂ ﻣﺮﺑﻮﻁ ﺑﻪ ﺑﻌﻀﯽ ﭼﻴﺰﻫﺎﺳﺖ.

ﺑﺮﻋﮑﺲ، ﻟﻐﺎﺗﯽ ﺑﻪ ﮐﺎﺭ ﺑﺮﻳﺪ ﮐﻪ ﺗﻮﺿﻴﺢ ﺩﻫﻨﺪﻩ ﺑﺎﺷﻨﺪ ﻣﺜﻞ «ﻧﻪ، ﺑﺨﺎﺭﯼ ﺩﺳﺘﺖ ﺭﻭ ﻣﯽ ﺳﻮﺯﻭﻧﻪ» ﻳﺎ «ﻧﻪ، ﺁﺗﺶ ﺑﭽﻪ ﺭﻭ ﻣﯽ ﺳﻮﺯﻭﻧﻪ» ﻳﺎ «ﺗﻮ ﺧﻴﺎﺑﻮﻥ ﻧﺪﻭ، ﻣﯽ‌ﺧﻮﺭﯼ ﺯﻣﻴﻦ ﺩﺭﺩﺕ ﻣﯽ‌ﮔﻴﺮﻩ». ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺑﻪ ﺍﻭ ﻣﯽ‌ﻓﻬﻤﺎﻧﻴﺪ ﮐﻪ ﺍﻳﻦ ﭼﻴﺰ ﺧﺎﺹ (ﺑﺨﺎﺭﯼ، ﺁﺗﺶ، ﭼﻴﺰﯼ ﺩﻳﮕﺮ) ﺧﻄﺮﻧﺎﮎ ﺍﺳﺖ ﻭ ﺑﻪ ﺍﻭ ﺻﺪﻣﻪ ﻣﯽ‌ﺯﻧﺪ، ﻭ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﻪ ﺍﻭ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺳﯽ ﺭﺍ ﺍﻟﻘﺎ ﻧﻤﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻫﻤﻪ ﭼﻴﺰﻫﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺧﻄﺮﻧﺎﮎ ﻫﺴﺘﻨﺪ ﻭ ﻧﺒﺎﻳﺪ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺩﺳﺖ ﺯﺩ.

ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ‌ﻫﺎﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺩﺳﺖ ﺯﺩﻥ ﺑﻪ ﭼﻴﺰﯼ ﭘﺮﺕ ﮐﺮﺩﻥ ﺣﻮﺍﺱ ﮐﻮﺩﮎ ﺍﺯ ﺁﻥ ﺷﻴﺊ ﺍﺳﺖ. ﺧﻮﺷﺒﺨﺘﺎﻧﻪ ﮐﻮﺩﮎ ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺗﻮﺟﻬﺶ ﺭﺍ ﺍﺯ ﭼﻴﺰﯼ ﺑﻪ ﭼﻴﺰ ﺩﻳﮕﺮ ﻣﻌﻄﻮﻑ ﻣﯽ‌ﮐﻨﺪ. ﻭﻗﺘﯽ ﺍﺯ ﺍﻳﻦ ﺭﻭﺵ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ ﮔﻮﻳﯽ ﺟﺎﺩﻭﮔﺮﯼ ﻫﺴﺘﻴﻢ ﮐﻪ ﺑﻪ ﮐﻮﺩﮎ ﻣﯽ‌ﮔﻮﻳﻴﻢ «ﺍﻭﻩ ﺍﻭﻩ، ﻧﮕﺎﻩ ﮐﻦ، ﺑﺒﻴﻦ ﭼﻪ ﭼﻴﺰ ﺟﺎﻟﺐ، ﻋﺠﻴﺐ ﻭ ﻫﻴﺠﺎﻥ‌ﺍﻧﮕﻴﺰﯼ ﺩﺍﺭﻡ، ﺍﻻﻥ ﺍﻭﻧﻮ ﻣﯽ‌ﺩﻡ ﺑﻪ ﺗﻮ ﮐﻪ ﺑﺎﻫﺎﺵ ﺑﺎﺯﯼ ﮐﻨﯽ!» (ﺍﺯ ﮐﻨﺎﺭ ﺑﺨﺎﺭﯼ ﺩﺍﻍ ﮐﻪ ﺩﺍﺷﺘﯽ ﺑﻬﺶ ﻣﯽ‌ﺧﻮﺭﺩﯼ ﺑﺮﻭ ﮐﻨﺎﺭ!)

ﮐﻮﺩﮐﯽ ﮐﻪ ﺷﺮﻭﻉ ﺑﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ ﺑﺴﻴﺎﺭ ﻓﻌﺎﻝ ﺍﺳﺖ ﻭ ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺮﮔﺮﺩﺍﻧﺪﻥ ﺍﺟﺴﺎﻡ ﻣﺨﺘﻠﻒ ﻭ ﻳﺎ ﺑﺮﺧﻮﺭﺩﺵ ﺑﺎ ﺍﺷﻴﺎﺀ ﻏﻴﺮﻗﺎﺑﻞ ﺍﺟﺘﻨﺎﺏ ﺍﺳﺖ.

ﺍﻳﻦ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﮐﻪ ﻣﻌﻠﻢ ﻣﺪﺭﺳﻪ-ﺧﺎﻧﻪ ﺍﻭ ﻫﺴﺘﻴﺪ ﺑﺮﺍﯼ ﻫﺮ ﺑﺎﺭ ﺯﻣﻴﻦ ﺧﻮﺭﺩﻧﺶ ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﻗﺎﺋﻞ ﻧﺸﻮﻳﺪ. ﺍﮔﺮ ﻫﺮ ﺑﺎﺭ ﮐﻪ ﺍﻭ ﺯﻣﻴﻦ ﻣﯽ ﺧﻮﺭﺩ، ﻧﺪﻭﻳﺪ ﻭ ﺑﻠﻨﺪﺵ ﻧﮑﻨﻴﺪ ﺧﻮﺩﺵ ﺑﻠﻨﺪ ﻣﯽ‌ﺷﻮﺩ. ﺍﮔﺮ ﻫﺮ ﺑﺎﺭ ﮐﻪ ﭼﻴﺰﯼ ﺭﺍ ﺑﺮﻣﯽ‌ﮔﺮﺩﺍﻧﺪ ﺁﻥ ﺭﺍ ﺑﺰﺭﮒ ﻧﮑﺮﺩﻩ ﻭ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺳﺨﻨﺮﺍﻧﯽ ﻧﮑﻨﻴﺪ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﻳﻦ ﺯﻣﻴﻦ ﺧﻮﺭﺩﻥ‌ﻫﺎ ﺑﺮﺍﯼ ﺍﻭ ﻓﻘﻂ ﺑﻪ‌ ﻣﺎﻧﻨﺪ ﻭﻗﻔﻪ‌ﻫﺎﯼ‌ﻫﺎﯼ ﻟﺤﻈﻪ‌ﺍﯼ ﻭ ﻣﺰﺍﺣﻢ ﻭ ﮔﺬﺭﺍ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺴﺶ ﺯﻳﺎﺩﺗﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺯﻣﻴﻦ ﺧﻮﺭﺩﻥ‌ﻫﺎ ﺭﺍ ﺟﺰﺋﯽ ﺍﺯ ﺍﮐﺘﺸﺎﻓﺎﺕ ﺭﻭﺯﻣﺮﻩ‌ﺍﺵ ﻣﯽ‌ﺷﻤﺮﺩ.‌

ﻣﺘﺎﺳﻔﺎﻧﻪ ﺑﻌﻀﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺷﺮﻭﻉ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺩﺭﻟﻨﺒﺮﯼ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻭﺳﻴﻠﻪ ﺁﻣﻮﺯﺷﯽ ﻣﯽ‌ﮐﻨﻨﺪ. ﺯﺩﻥ ﺩﺭﻟﻨﺒﺮﯼ ﺑﻪ ﮐﻮﺩﮎ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺩﻭ ﺳﺎﻟﮕﯽ، ﻣﮕﺮ ﺩﺭ ﻣﻮﺍﺭﺩ ﺑﺴﻴﺎﺭ ﻣﻌﺪﻭﺩ ﺑﯽ‌ﻣﻌﻨﯽ ﺍﺳﺖ. ﺍﮔﺮ ﮐﻮﺩﮎ ﻣﻮﻗﻊ ﺭﺩ ﺷﺪﻥ ﺍﺯ ﺧﻴﺎﺑﺎﻥ ﺑﺎ ﺳﻤﺎﺟﺖ ﺑﻪ ﻭﺳﻂ ﺧﻴﺎﺑﺎﻥ ﺑﺪﻭﺩ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺠﺒﻮﺭ ﺷﻮﻳﺪ ﮐﻪ ﺑﻪ ﻟﻨﺒﺮﺵ ﺑﺰﻧﻴﺪ.  ﻭﻟﯽ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﻣﺤﻴﻂ ﺧﺎﻧﻮﺍﺩﮔﯽ ﻳﮏ ﮐﻮﺩﮎ ﺧﻴﻠﯽ ﮐﻮﭼﮏ ﺭﺍ ﻃﻮﺭﯼ ﺗﻨﻈﻴﻢ ﮐﻨﻴﺪ ﮐﻪ ﺻﺪﻫﺎ ﻣﻤﻨﻮﻋﻴﺖ ﻭﺟﻮﺩ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺑﺎ ﺟﻠﺐ ﮐﺮﺩﻥ ﺗﻮﺟﻬﺶ ﺑﻪ ﭼﻴﺰﻫﺎﯼ ﺩﻳﮕﺮ ﺭﻓﺘﺎﺭ ﺍﻭ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ،  ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺍﺣﺘﻴﺎﺟﯽ ﺑﻪ ﺗﻨﺒﻴﻪ ﺑﺪﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺩﺭﻟﻨﺒﺮﯼ ﺑﺮﺍﯼ ﺑﻌﺪ ﺍﺯ ﺩﻭ ﺳﺎﻟﮕﯽ ﺍﺳﺖ ﻭ ﻗﺒﻞ ﺍﺯ ﺁﻥ ﻗﺪﻏﻦ ﺍﺳﺖ. ﺍﮔﺮ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﺧﻴﻠﯽ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﻳﺪ ﻭ ﺍﻭ ﺭﺍ ﺯﻳﺎﺩ «ﺗﻨﺒﻴﻪ» ﻣﯽ‌ﮐﻨﻴﺪ ﺑﻪ ﻳﮏ ﺭﻭﺍﻧﺸﻨﺎﺱ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻳﺪ ﮐﻪ ﺑﺮﺍﯼ ﺭﻓﻊ ﻣﺴﺎﺋﻠﺘﺎﻥ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﮐﻨﺪ.

ﻳﮑﯽ ﺍﺯ ﺩﻻﻳﻠﯽ ﮐﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺭﺍ ﺩﺭ ﺳﻨﻴﻦ ﭘﺎﻳﻴﻦ ﮐﺘﮏ ﻣﯽ‌ﺯﻧﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺁﻧﻬﺎ ﺍﺯ ﺭﻭﯼ ﻗﺼﺪ ﺧﺮﺍﺑﮑﺎﺭﯼ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﺜﻼ ﻣﺎﺩﺭ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﺩﺭ ﮐﺘﺎﺑﺨﺎﻧﻪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺩﺍﺭﺩ ﺯﻳﺒﺎﺗﺮﻳﻦ ﮐﺘﺎﺑﺶ ﺭﺍ ﭘﺎﺭﻩ ﻣﯽ‌ﮐﻨﺪ ﻏﺎﻓﻠﮕﻴﺮ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﮐﻮﺩﮎ ﺍﺯ ﺭﻭﯼ ﻣﺨﺎﻟﻔﺖ ﻭ ﺑﺪﺟﻨﺴﯽ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺍﻭ ﺭﺍ ﮐﺘﮏ ﻣﯽ‌ﺯﻧﺪ. ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﺻﻼ ﺍﻳﻦ ﻃﻮﺭ ﻧﻴﺴﺖ. ﻳﮏ ﺑﭽﻪ ﭼﻬﺎﺭﺳﺎﻟﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺍﺣﺴﺎﺱ ﻣﺨﺎﻟﻔﺖ ﻭ ﺧﺮﺍﺑﮑﺎﺭﯼ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﻭﻟﯽ ﻳﮏ ﮐﻮﺩﮎ ﮐﻤﺘﺮ ﺍﺯ ﺩﻭ ﺳﺎﻝ ﻧﻪ.

ﺑﺮﺍﯼ ﺑﭽﻪ ﮐﻤﺘﺮ ﺍﺯ ﺩﻭﺳﺎﻝ، ﭘﺎﺭﻩ ﮐﺮﺩﻥ ﺻﻔﺤﺎﺕ ﮐﺘﺎﺏ ﺑﺨﺸﯽ ﺍﺯ ﺍﮐﺘﺸﺎﻓﺎﺕ ﻋﻠﻤﯽ ﺟﻬﺎﻥ ﮐﺘﺎﺏ ﺍﺳﺖ ﻭ ﺑﯽ‌ﺷﮏ ﭼﻨﺪ ﺻﻔﺤﻪ‌ﺍﯼ ﺍﺯ ﮐﺘﺎﺏ ﺭﺍﻫﻢ ﺩﺭ ﺩﻫﺎﻧﺶ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﻳﮏ ﻣﺎﺩﺭ ﻧﺒﺎﻳﺪ ﺭﻓﺘﺎﺭ ﻃﺒﻴﻌﯽ ﻳﮏ ﮐﻮﭼﻮﻟﻮ ﺭﺍ ﮐﻪ ﻣﺸﻐﻮﻝ ﮐﺸﻒ ﺩﻧﻴﺎﺳﺖ ﺑﺎ ﺭﻓﺘﺎﺭ ﻟﺠﻮﺟﺎﻧﻪ ﻳﮏ ﺑﭽﻪ ﺑﺰﺭﮒ ﺍﺷﺘﺒﺎﻩ ﺑﮕﻴﺮﺩ. ﮐﻮﺩﮎ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮﺩﺍﺷﺘﻦ ﺧﺼﻮﺻﻴﺎﺕ ﺧﺎﺹ ﺳﻨﺶ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺗﻮﺑﻴﺦ ﮐﺮﺩ.

ﺩﺭ ﺩﺭﺱ ﺁﻳﻨﺪﻩ ﺩﺭ ﻣﻮﺭﺩ ﻏﺬﺍ ﺩﺍﺩﻥ ﺑﻪ ﻛﻮﺩﻙ ﻭ ﺭﻓﺘﺎﺭ ﺍﻭ ﻧﺴﺒﺖ ﺑﻪ ﺑﺪﻥ ﺧﻮﺩﺵ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﻛﺮﺩ.
