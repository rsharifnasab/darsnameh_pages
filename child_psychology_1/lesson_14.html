ﺩﺭﺱ ﻳﺎﺯﺩﻫﻢ- ﺩﻭ ﺗﺎ ﺳﻪ ﺳﺎﻟﮕﯽ، ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ- ﺑﺨﺶ ﺩﻭﻡ

ﻣﺸﮑﻞ ﺑﺘﻮﺍﻥ ﺗﻔﺎﻭﺗﯽ ﺭﺍ ﮐﻪ ﺑﻴﻦ ﺍﺣﺴﺎﺳﯽ ﮐﻪ ﻃﻔﻞ ﻧﺴﺒﺖ ﺑﻪ ﺷﺨﺼﻴﺖ ﺧﻮﺩ ﺩﺭ ﺯﻣﺎﻧﯽ ﮐﻪ ﺷﺮﻭﻉ ﺑﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻫﻤﺎﻥ ﮐﻮﺩﮎ، ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﺭﺍ ﻃﯽ ﻣﯽ‌ﮐﻨﺪ ﺑﺎ ﮐﻠﻤﺎﺕ ﺗﺸﺮﻳﺢ ﮐﺮﺩ.

ﮐﻮﺩﮎ ﻧﻮﭘﺎ ﺑﻪ ﺭﻏﻢ ﺍﻳﻦ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻳﻦ ﻃﺮﻑ ﻭ ﺁﻥ ﻃﺮﻑ ﺑﺮﻭﺩ ﻫﻨﻮﺯ ﻭﺍﻗﻌﺎ ﻳﮏ ﻧﻮﺯﺍﺩ ﺍﺳﺖ. ﺩﺭ ﺍﻭ ﻗﺎﺑﻠﻴﺖ ﺳﺮﻭﺭ، ﭘﺎﮐﯽ ﻭ ﺧﻮﺩ ﺍﻧﮕﻴﺨﺘﮕﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻟﻴﮑﻦ ﮐﻮﺩﮎ ﺩﺭ ﺩﻭﺭﺍﻥ «ﺑﻠﻮﻍ ﺍﻭﻟﻴﻪ» ﺍﻳﻦ ﺧﺼﻠﺖ ﺳﺎﺩﮔﯽ، ﺧﻮﺩﺍﻧﮕﻴﺨﺘﮕﯽ ﻭ ﭘﺎﮐﯽ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻩ ﺍﺳﺖ. ﺯﻧﺪﮔﯽ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﺍﻭ ﻣﺜﻞ ﺧﻴﺎﺑﺎﻧﯽ ﻳﮏ ﻃﺮﻓﻪ ﻧﻴﺴﺖ، ﺑﻠﮑﻪ ﺟﺎﺩﻩ‌ﺍﯼ ﺍﺳﺖ ﺩﻭ ﻃﺮﻓﻪ. ﺍﻭ ﺑﺮﺍﯼ ﺍﻭﻟﻴﻦ ﺑﺎﺭ ﺩﺭ ﺯﻧﺪﮔﯽ ﮐﻮﺗﺎﻫﺶ ﺑﻪ ﮔﺮﺍﻳﺶ‌ﻫﺎﯼ ﻗﻮﯼ ﻭ ﻣﺘﻨﺎﻗﻀﯽ ﺩﺭ ﺧﻮﺩ ﺑﺮﺧﻮﺭﺩ ﻣﯽ‌ﮐﻨﺪ. ﺭﻓﺘﺎﺭ ﻭﯼ ﺩﺍﺋﻤﺎ ﻧﺸﺎﻥ‌ﺩﻫﻨﺪﻩ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ:«ﺧﺐ، ﮐﺠﺎ ﻣﯽ‌ﺧﻮﺍﻡ ﺑﺮﻡ؟ ﻋﻘﺐ ﮔﺮﺩ ﮐﻨﻢ ﺗﺎ ﺑﺎﺯ ﻳﻪ ﻧﻮﺯﺍﺩ ﺑﺸﻢ ﻳﺎ ﺑﻪ ﻃﺮﻑ ﺟﻠﻮ ﺣﺮﮐﺖ ﮐﻨﻢ ﺗﺎ ﺗﺒﺪﻳﻞ ﺑﻪ ﻳﻪ ﮐﻮﺩﮎ ﺑﺸﻢ؟ ﭼﯽ ﻣﯽ‌ﺧﻮﺍﻡ؟ ﻣﯽ‌ﺧﻮﺍﻡ ﮐﺎﺭﯼ ﮐﻪ ﺩﻟﻢ ﻣﯽ‌ﺧﻮﺍﺩ ﺭﻭ ﺑﮑﻨﻢ ﻳﺎ ﮐﺎﺭﯼ ﮐﻪ ﻣﺎﻣﺎﻥ ﺍﻳﻨﺎ ﻣﯽ‌ﮔﻦ ﻭ ﺩﺭﺳﺖ ﺧﻼﻑ ﺍﻭﻥ ﭼﻴﺰﻳﻪ ﮐﻪ ﻣﻦ ﻣﯽ‌ﺧﻮﺍﻡ؟ ﭘﺲ ﺍﻳﻦ «ﻣﻦ»ﮐﻪ ﻫﻤﻪ ﺍﻳﻦ ﭼﻴﺰﺍ ﺭﻭ ﻣﯽ‌ﺧﻮﺍﺩ ﮐﻴﻪ؟ ﻣﻦ ﺩﺭ ﻭﺍﻗﻊ ﮐﯽ ﻫﺴﺘﻢ؟»

ﺍﻳﻦ ﺳﻦ ﺍﻏﻠﺐ ﺍﻭﻗﺎﺕ ﺍﻳﻦ ﺧﺼﻠﺖ ﺭﺍ ﺩﺍﺭﺩ «ﻣﺎﻣﺎﻥ ﺑﺬﺍﺭ، ﺧﻮﺩﻡ ﻣﯽ‌ﺧﻮﺍﻡ ﺍﻳﻦ ﮐﺎﺭﻭ ﺑﮑﻨﻢ». ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﮐﻮﺩﮎ ﻧﻤﯽ‌ﮔﺬﺍﺭﺩ ﻣﺎﺩﺭ ﺩﺭ ﻟﺒﺎﺱ ﭘﻮﺷﻴﺪﻥ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﮐﻨﺪ ﻭ ﻓﺮﻳﺎﺩ ﻣﯽ‌ﮐﺸﺪ «ﺑﺬﺍﺭ ﺧﻮﺩﻡ ﺑﭙﻮﺷﻢ!»
ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﻧﻴﺰ ﺍﺯ ﻣﺎﺩﺭﺵ ﺟﺪﺍ ﻧﻤﯽ‌ﺷﻮﺩ ﻭ ﻣﯽ‌ﮔﻮﻳﺪ ﮐﻪ ﻧﯽ‌ﻧﯽ ﻛﻮﭼﻮﻟﻮﺳﺖ ﻭ ﻣﺎﻣﺎﻥ ﺑﺎﻳﺪ ﺗﻤﺎﻡ ﮐﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺑﮑﻨﺪ. ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﻪ ﻣﺎﻧﻨﺪ ﻳﮏ ﻧﻮﺟﻮﺍﻥ ﺍﺯ ﺳﺎﻋﺘﯽ ﺑﻪ ﺳﺎﻋﺖ ﺩﻳﮕﺮ ﻭ ﺍﺯ ﺭﻭﺯﯼ ﺑﻪ ﺭﻭﺯ ﺩﻳﮕﺮ، ﺑﻴﻦ ﺧﻮﺍﺳﺘﻪ ﺍﺳﺘﻘﻼﻝ ﻭ ﻭﺍﺑﺴﺘﮕﯽ ﮐﻮﺩﮐﺎﻧﻪ‌ﺍﺵ ﺑﻪ ﻣﺎﺩﺭ ﮐﻠﻪ ﻣﻌﻠﻖ ﻣﯽ‌ﺯﻧﺪ. ﺍﻭ ﻳﺎ «ﺑﻠﻪ» ﺍﺳﺖ ﻭ ﻳﺎ «ﻧﻪ». ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﻣﺤﺪﻭﺩﻳﺖ ﻭ ﻗﻮﺍﻧﻴﻨﯽ ﮐﻪ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺩﻭﺳﺎﻟﻪ ﺩﺭ ﻧﻈﺮ ﻣﯽ‌ﮔﻴﺮﻳﺪ ﺑﺎﻳﺪ ﻗﺎﺑﻞ ﺍﻧﻌﻄﺎﻑ ﺑﺎﺷﺪ. ﺑﺮﺍﯼ ﮐﻮﺩﮐﯽ ﺑﻪ ﺍﻳﻦ ﺳﻦ ﺍﺻﻮﻝ ﺑﺴﻴﺎﺭ ﺩﻗﻴﻘﯽ ﺩﺭ ﻣﻮﺭﺩ ﻟﺒﺎﺱ ﭘﻮﺷﻴﺪﻥ، ﺳﺎﻋﺖ ﺣﻤﺎﻡ ﻭ ﻏﻴﺮﻩ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻥ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﻋﻠﺖ ﻧﻴﺰ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﺍﺳﺖ. ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪﺵ ﻣﻤﻠﻮ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻓﮑﺎﺭ ﭘﻴﭽﻴﺪﻩ ﺍﺳﺖ ﻭ ﻗﻮﺍﻧﻴﻦ ﻣﻄﻠﻖ ﻭ ﻏﻴﺮﻗﺎﺑﻞ ﺍﻧﻌﻄﺎﻑ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻗﺎﺑﻞ ﺍﺟﺮﺍ ﻧﻴﺴﺘﻨﺪ. ﻣﻌﻤﻮﻻ ﺧﻴﻠﯽ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﻭ ﻗﻮﺍﻧﻴﻦ ﻣﺸﺨﺺ ﺭﺍ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﻮﺩﮎ ﺗﻘﺮﻳﺒﺎ ﺳﻪ ﺳﺎﻟﻪ ﺷﺪ ﺑﻪ ﻣﻮﺭﺩ ﺍﺟﺮﺍ ﺑﮕﺬﺍﺭﻧﺪ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﻭﻇﻴﻔﻪ‌ﺍﯼ ﮐﻪ ﻃﻔﻞ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪﺵ ﺩﺍﺭﺩ ﻛﺴﺐ ﺩﺭﮎ ﻣﺤﮑﻤﯽ ﺍﺯ ﺷﺨﺼﻴﺖ ﺧﻮﺩ ﺍﺳﺖ ﻭ ﺩﺭﮎ ﻋﻤﻴﻖ ﺍﻳﻦ ﻛﻪ ﮐﻴﺴﺖ، ﺩﺭ ﻋﻴﻦ ﺍﻳﻦ ﻛﻪ ﺑﺎﻳﺪ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺁﻧﭽﻪ ﺟﺎﻣﻌﻪ (ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﺧﺼﻮﺻﺎ ﺷﺎﻣﻞ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻣﯽ‌ﺷﻮﺩ) ﺍﺯ ﺍﻭ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﺩ ﻭﻓﻖ ﺩﻫﺪ.

ﺩﺭ ﺍﻳﻦ ﺟﺎ ﻣﺴﺎﺋﻞ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺩﻭ ﺻﻮﺭﺕ ﭘﻴﭽﺪﻩ ﺷﻮﻧﺪ. ﺩﺭﺣﺎﻟﺖ ﺍﻭﻝ، ﻭﺍﻟﺪﻳﻦ ﻗﺎﺩﺭ ﺑﻪ ﮐﻨﺘﺮﻟﯽ ﺷﺪﻳﺪ ﺑﻮﺩﻩ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺭﺍ ﻭﺍﺩﺍﺭ ﺑﻪ ﺍﺟﺮﺍﯼ ﻗﻮﺍﻧﻴﻦ ﺑﺴﻴﺎﺭ ﺳﺨﺖ ﮐﻨﻨﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻮﺩﮎ ﺯﻳﺎﺩﻩ ﺍﺯ ﺣﺪ ﻣﻄﻴﻊ ﻭ ﺭﺍﻡ ﺷﻮﺩ.  ﭘﺴﺮﯼ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﻢ ﮐﻪ ﻫﻴﭻ ﮔﺎﻩ ﻣﺮﺣﻠﻪ ﻣﺨﺎﻟﻔﺖ ﻧﻤﻮﻧﻪ‌ﻭﺍﺭ ﺩﻭ ﺳﺎﻟﻪ‌ﻫﺎ ﺭﺍ ﭘﺸﺖ ﺳﺮﻧﮕﺬﺍﺷﺖ ﺯﻳﺮﺍ ﻣﺎﺩﺭﺵ ﻫﻴﭻ ﮔﺎﻩ ﺍﺟﺎﺯﻩ ﭼﻨﻴﻦ ﮐﺎﺭﯼ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺪﺍﺩ. ﺍﻭ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﻣﺠﺒﻮﺭ ﺑﻪ ﺍﻃﺎﻋﺘﯽ ﮐﻮﺭﮐﻮﺭﺍﻧﻪ ﮐﺮﺩﻩ ﺑﻮﺩ. ﻣﺎﺩﺭ ﺍﺟﺎﺯﻩ ﺑﺮﻭﺯ ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺣﺮﮐﺎﺕ ﻧﺎﺷﯽ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﻳﺎ ﺟﻬﺶ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﺳﺮﭼﺸﻤﻪ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﻭ ﻋﻼﻣﺖ ﻣﺸﺨﺼﻪ ﺍﻳﻦ ﺳﻦ ﻫﺴﺘﻨﺪ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﻧﻤﯽ‌ﺩﺍﺩ. ﺩﺭ ﻋﻮﺽ ﻫﺮ ﮔﺎﻩ ﮐﻪ ﻓﺮﺯﻧﺪﺵ ﺁﺭﺍﻡ، ﻣﻨﻔﻌﻞ ﻭ «ﻣﻬﺮﺑﺎﻥ» ﺑﻮﺩ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﺮﺩ. ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﮐﻪ ﺑﺮﺍﯼ ﻣﺎﺩﺭِ ﻃﻔﻞ ﺧﻴﻠﯽ ﺧﻮﺏ ﭘﻴﺶ ﻣﯽ‌ﺭﻓﺖ ﻭ ﻣﻮﺟﺐ ﺷﺪﻩ ﺑﻮﺩ ﮐﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺵ ﺑﺪﻭﻥ ﻣﺴﺎﻟﻪ ﺧﺎﺻﯽ ﺑﮕﺬﺭﺩ، ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﭼﻨﺪﺍﻥ ﻣﻔﻴﺪ ﻧﺒﻮﺩ. ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﻭ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺭﻓﺖ، ﻣﻌﻠﻤﺶ ﻣﺸﺎﻫﺪﻩ ﮐﺮﺩ ﮐﻮﺩﮎ ﺗﺮﺳﻮ ﺍﺳﺖ ﻭ ﺑﺎ ﺑﭽﻪ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺑﺎﺯﯼ ﻧﻤﯽ‌ﮐﻨﺪ ﻭ ﺩﻧﻴﺎﯼ ﺑﺴﺘﻪ‌ﺍﯼ ﺩﺍﺭﺩ. ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺁﻣﺪ ﮐﻪ ﻣﺎﺩﺭﺵ ﺍﻭ ﺭﺍ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﺁﺭﺍﻡ ﻭ ﻣﻄﻴﻊ ﺑﺎﺭ ﺁﻭﺭﺩﻩ ﺑﻮﺩ.
ﺍﻳﻦ ﮐﻮﺩﮎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺷﺨﺺ ﺑﺎﻟﻐﯽ ﺷﻮﺩ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﻋﻠﺖ ﭼﻨﻴﻦ ﺗﺮﺑﻴﺘﯽ، ﺷﺨﺼﯽ ﺧﺠﺎﻟﺘﯽ ﺷﻮﺩ ﮐﻪ ﺟﺮﺃﺕ ﻣﺨﺎﻟﻔﺖ ﺑﺎ ﮐﺴﯽ ﻭ ﭼﻴﺰﯼ ﺭﺍ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺍﺯ ﻭﺍﺭﺩ ﺷﺪﻥ ﺑﻪ ﺯﻧﺪﮔﯽ ﺍﺟﺘﻤﺎﻋﯽ ﻭ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺍﻣﺘﺤﺎﻥ ﮐﺮﺩﻥ ﻫﺮ ﭼﻴﺰ ﺟﺪﻳﺪﯼ ﻫﺮﺍﺱ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺗﻤﺎﻡ ﮐﻮﺩﮐﺎﻥ ﺑﻪ ﺍﻳﻦ ﺭﺍﺣﺘﯽ ﺍﺳﺘﺒﺪﺍﺩ ﺷﺪﻳﺪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺭﺍ ﻗﺒﻮﻝ ﻧﻤﯽ‌ﮐﻨﻨﺪ. ﺗﺮﺱ‌ﻫﺎﯼ ﮐﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﺍﺯ ﺑﻘﻴﻪ ﺗﺮﺱ ﻫﺎ ﺩﻳﺮﭘﺎﺗﺮﻧﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ، ﺍﺳﺘﺒﺪﺍﺩ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺑﺎ ﻣﻘﺎﻭﻣﺖ ﺷﺪﻳﺪ ﮐﻮﺩﮎ ﺑﺮﺧﻮﺭﺩ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﻣﯽ‌ﮐﻮﺷﺪ ﺗﺎ ﺁﻧﺠﺎ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻘﺎﻭﻣﺖ ﮐﻨﺪ ﻭ ﺳﺮ ﻓﺮﻭﺩ ﻧﻴﺎﻭﺭﺩ. ﻭ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺗﻤﺎﻡ ﺩﻭﺭﻩ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﺗﺒﺪﻳﻞ ﺑﻪ ﻣﺒﺎﺭﺯﻩ‌ﺍﯼ ﺑﻴﻦ ﮐﻮﺩﮎ ﻭ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺩﺭ ﻣﻘﺎﺑﻞ ﺧﻮﺍﺳﺘﻪ ﻫﺎﯼ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ. ﺍﻳﻦ ﮐﻪ ﭼﻪ ﮐﺴﯽ ﺩﺭ ﺍﻳﻦ ﺟﻨﮓ ﭘﻴﺮﻭﺯ ﺑﺸﻮﺩ ﻣﻬﻢ ﻧﻴﺴﺖ، ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﻣﺴﻠﻤﺎ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺴﺖ ﺩﺭ ﺍﻳﻦ ﻣﺪﺕ ﺑﻪ ﺩﺳﺖ ﺑﻴﺎﻭﺭﺩ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﺪ.

ﻣﻤﮑﻦ ﻫﻢ ﻫﺴﺖ ﮐﻪ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺳﺘﺒﺪﺍﺩ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ، ﮐﻮﺩﮎ ﺧﻮﺩ ﺭﺍ ﻇﺎﻫﺮﺍ ﻣﻄﻴﻊ ﻧﺸﺎﻥ ﺩﻫﺪ، ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻣﺨﺎﻟﻔﺘﺶ ﺭﺍ ﺩﺭ ﻋﻤﻖ ﻭﺟﻮﺩﺵ ﺳﺮﮐﻮﺏ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﺑﺮﺧﻼﻑ ﻣﻴﻠﺶ ﺑﻪ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺗﻦ ﺩﺭ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺩﻭ ﺭﻭ ﻣﯽ‌ﺷﻮﺩ. ﻫﺮ ﮔﺎﻩ ﻓﺮﺻﺖ ﺩﺳﺖ ﺩﻫﺪ، ﭘﻨﻬﺎﻧﯽ ﮐﻤﯽ ﺍﺯ ﺍﻳﻦ ﺩﺷﻤﻨﻴﺶ ﺭﺍ ﺭﻭﯼ ﺍﻃﺮﺍﻓﻴﺎﻧﺶ ﺧﺎﻟﯽ ﻣﯽ‌ﮐﻨﺪ. ﭼﻴﺰﯼ ﺭﺍ ﻣﯽ‌ﺷﮑﻨﺪ، ﺧﻮﺍﻫﺮﺵ ﺭﺍ ﻧﻴﺸﮕﻮﻥ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﻳﺎ ﻋﻤﻞ ﻣﺨﺎﻟﻔﺖ ﻳﺎ ﺧﺮﺍﺑﮑﺎﺭﺍﻧﻪ ﺩﻳﮕﺮﯼ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﺪ. ﭼﻨﻴﻦ ﮐﻮﺩﮐﯽ ﻭﻗﺘﯽ ﺑﺰﺭﮒ ﺷﻮﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺗﺒﺪﻳﻞ ﺑﻪ ﺷﺨﺼﯽ ﺷﻮﺩ ﻛﻪ  ﻇﺎﻫﺮﺍ ﺑﻪ ﺍﺻﻮﻝ ﺍﺧﻼﻗﯽ ﺑﺴﻴﺎﺭ ﭘﺎﻳﺒﻨﺪ ﺍﺳﺖ ﻭﻟﯽ ﺩﺭ ﺍﺻﻞ ﻣﻤﻠﻮ ﺍﺯ ﺑﻐﺾ ﻭ ﻋﺪﺍﻭﺕ ﺍﺳﺖ.

ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻣﺴﺎﻟﻪ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﻄﺮﺡ ﺷﻮﺩ ﻭ ﺁﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺎﺩﺭ ﺍﺯ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﺍﻗﺘﺪﺍﺭﺵ ﺗﺮﺱ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﻭ ﺩﺭ ﻣﻘﺎﺑﻞ ﺗﻤﺎﻡ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﯼ ﻓﺮﺯﻧﺪﺵ ﺳﺮ ﻓﺮﻭ ﻣﯽ‌ﺁﻭﺭﺩ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﻮﺩﮎ ﻣﺤﺪﻭﺩﻩ‌ﺍﯼ ﮐﻪ ﻣﺎﺩﺭﺵ ﺑﺮﺍﻳﺶ ﺗﻌﻴﻴﻦ ﮐﺮﺩﻩ ﻗﺒﻮﻝ ﻧﻤﯽ‌ﮐﻨﺪ، ﻣﺎﺩﺭ ﺍﺳﺖ ﮐﻪ ﻋﻘﺐ ﻧﺸﻴﻨﯽ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺧﻼﺻﻪ ﻫﻤﻴﺸﻪ ﺣﺮﻑ ﺁﺧﺮ ﺑﺎ ﮐﻮﺩﮎ ﺍﺳﺖ. ﺑﻪ ﺯﻭﺩﯼ ﻧﻘﺶ‌ﻫﺎ ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ ﺗﻐﻴﻴﺮ ﻣﯽ‌ﮐﻨﻨﺪ. ﺩﻳﮕﺮ ﮐﻮﺩﮎ ﺍﺳﺖ ﮐﻪ ﺧﺎﻧﻪ ﺭﺍ ﺍﺩﺍﺭﻩ ﻣﯽ‌ﮐﻨﺪ ﻧﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ. ﺩﺭ ﭼﻨﻴﻦ ﺣﺎﻟﺘﯽ ﮐﻮﺩﮎ ﻏﻴﺮﻗﺎﺑﻞ ﺗﺤﻤﻞ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﮔﻮﻧﻪ ﺍﻃﻔﺎﻝ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﺩﺭ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﻧﻤﯽ‌ﺁﻣﻮﺯﻧﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻭﺍﻗﻌﻴﺖ ﺗﻄﺒﻴﻖ ﺩﻫﻨﺪ ﻭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﻪ ﮐﻮﺩﮐﺴﺘﺎﻥ ﺑﺮﻭﻧﺪ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﮐﻪ ﻣﻌﻠﻢ ﻭ ﺍﻃﻔﺎﻝ ﺩﻳﮕﺮ ﺣﺪﺍﻗﻞ ﺍﺣﺘﺮﺍﻡ ﺍﺑﺘﺪﺍﻳﯽ ﻧﺴﺒﺖ ﺑﻪ ﻗﻮﺍﻧﻴﻦ ﺟﺎﻣﻌﻪ ﺭﺍ ﻃﻠﺐ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺩﭼﺎﺭ ﻫﺰﺍﺭﺍﻥ ﻣﺸﮑﻞ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﻳﻦ ﮐﻮﺩﮐﺎﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺭﺍ ﺑﺮﺍﯼ ﺧﻮﺩﺷﺎﻥ ﺣﻞ ﮐﻨﻨﺪ، ﭼﺮﺍ ﮐﻪ ﻣﺎﺩﺭﺷﺎﻥ ﺑﻪ ﺁﻧﻬﺎ ﺩﺭ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﮐﻤﮑﯽ ﺟﻬﺖ ﻳﺎﺩﮔﻴﺮﯼ ﺍﺣﺘﺮﺍﻡ ﺑﻪ ﻗﻮﺍﻧﻴﻦ ﺟﺎﻣﻌﻪ ﻧﮑﺮﺩﻩ ﺍﺳﺖ.

ﺑﻪ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻐﯽ ﮐﻪ ﺩﻭﺭﻭﺑﺮﺗﺎﻥ ﻫﺴﺘﻨﺪ ﻧﮕﺎﻫﯽ ﺑﮑﻨﻴﺪ. ﻣﻄﻤﺌﻨﺎ ﺩﺭ ﺑﻴﻦ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﻫﻢ ﺁﺩﻡ‌ﻫﺎﯼ ﺷﺪﻳﺪﺍ ﺧﺠﺎﻟﺘﯽ ﻭ ﻫﻢ ﺍﻧﺴﺎﻥ‌ﻫﺎﻳﯽ ﺑﺴﻴﺎﺭ ﻣﻄﻴﻊ ﮐﻪ ﺷﺪﻳﺪﺍ ﺗﺎﺑﻊ ﺭﺳﻢ ﻭ ﺭﺳﻮﻣﺎﺕ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ. ﮔﻮﻳﯽ ﺍﻳﻨﻬﺎ ﻫﻴﭽﻮﻗﺖ ﺭﺍﺣﺖ ﻧﻴﺴﺘﻨﺪ، ﺍﻳﻦ ﮔﻮﻧﻪ ﺍﻓﺮﺍﺩ ﺧﺸﮏ، ﭘﺮﺗﻮﻗﻊ ﻭ ﻣﻤﻠﻮ ﺍﺯ ﭘﻴﺶ ﺩﺍﻭﺭﯼ ﻫﺴﺘﻨﺪ. ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﻫﻢ ﻫﻤﻴﺸﻪ ﺑﺎ ﻫﻤﻪ ﭼﻴﺰ ﻣﺨﺎﻟﻔﻨﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﻫﺮ ﻣﻮﺭﺩﯼ ﻣﻮﺿﻊ ﻣﺨﺎﻟﻒ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺩﺭ ﺑﺎﺷﮕﺎﻩ‌ﻫﺎ ﻭ ﺍﻧﺠﻤﻦ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻫﻤﻴﺸﻪ ﺍﻳﻦ‌ﻫﺎ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﺴﺎﻟﻪ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻭﺭﻧﺪ. ﺍﻳﻨﻬﺎ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﺑﺎ ﮐﺴﯽ ﻳﺎ ﭼﻴﺰﯼ ﻣﺨﺎﻟﻔﺖ ﮐﻨﻨﺪ. ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﺍﮐﺜﺮ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻐﯽ ﮐﻪ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﻣﻄﻴﻊ‌ﺍﻧﺪ ﻳﺎ ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﻫﻤﻴﺸﻪ ﺑﺎ ﻫﻤﻪ ﭼﻴﺰ ﻣﺨﺎﻟﻔﺖ ﻣﯽ‌ﮐﻨﻨﺪ، ﻛﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﻭﺭﺍﻥ ﻃﻔﻮﻟﻴﺖ ﺭﺍ ﺑﺪ ﮔﺬﺭﺍﻧﺪﻩ‌ﺍﻧﺪ، ﻳﻌﻨﯽ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﮐﻪ ﻣﺎ ﺁﻥ ﺭﺍ«ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﻧﺎﻣﻴﺪﻩ‌ﺍﻳﻢ.‌
