ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﭼﺸﻢ ﺍﻧﺪﺍﺯ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺍﻫﻤﻴﺖ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﻧﻴﺎﺯﻫﺎﯼ ﺍﺳﺎﺳﯽ ﻧﻮﺯﺍﺩ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﻭ ﺍﺯ ﺗﺎﺛﻴﺮﯼ ﮐﻪ ﺩﺭﺳﺖ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﺍﻳﻦ ﻧﻴﺎﺯﻫﺎ ﺑﻪ ﺩﻳﺪ ﺍﻭ ﺑﻪ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻭ ﺩﺭﺱ ﺑﻌﺪ ﻧﻈﺮﯼ ﺍﺟﻤﺎﻟﯽ ﺑﻪ ﺭﺷﺪ ﻓﺮﺯﻧﺪ ﻧﻮﺯﺍﺩﺗﺎﻥ ﺩﺭ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ‌ﺍﺵ ﻣﯽ‌ﺍﻧﺪﺍﺯﻳﻢ.

ﺑﺮﺍﯼ ﺑﺮﺭﺳﯽ ﺍﻳﻦ ﺳﺎﻝ ﻭ ﺍﺯ ﺟﻬﺖ ﺭﺍﺣﺘﯽ ﮐﺎﺭ، ﺍﻳﻦ ﺩﻭﺭﻩ ﺭﺍ ﺑﻪ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﺳﻪ ﻣﺎﻫﻪ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﮐﻨﻴﻢ. ﻭﻟﯽ ﺑﻼﻓﺎﺻﻠﻪ ﺑﺎﻳﺪ ﮔﻮﺷﺰﺩ ﮐﻨﻴﻢ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﺳﻪ ﻣﺎﻩ ﮐﺎﻣﻼً ﺍﺧﺘﻴﺎﺭﯼ ﺍﺳﺖ. ﻗﺼﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻧﻈﺮﯼ ﮐﻠﯽ ﺩﺭ ﻣﻮﺭﺩ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺪﻫﻴﻢ ﻭ ﺗﻮﺻﻴﻪ‌ﻫﺎﻳﯽ ﺩﺭ ﺯﻣﻴﻨﻪ ﺑﺎﺯﯼ‌ﻫﺎ ﻭ ﺍﻳﺠﺎﺩ ﻣﺤﻴﻄﯽ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺍﻭ ﺟﻬﺖ ﺗﺴﻬﻴﻞ ﺭﺷﺪ ﺫﻫﻨﯽ‌ﺍﺵ ﺑﻪ ﺷﻤﺎ ﺑﮑﻨﻴﻢ.

ﻣﻄﺎﻟﻌﺎﺕ ﻣﺘﻌﺪﺩ ﻧﺸﺎﻥ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﮐﻪ ﻧﻮﺯﺍﺩﺍﻥ ﺗﻔﺎﻭﺕ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺟﻬﺎﺕ ﻣﺨﺘﻠﻒ ﺑﺎﻫﻢ ﺩﺍﺭﻧﺪ: ﺁﺭﺍﻡ ﻳﺎ ﻧﺎﺁﺭﺍﻡ ﺑﻮﺩﻥ، ﺣﺴﺎﺳﻴﺖ ﺑﻪ ﻧﻮﺭ، ﺻﺪﺍ، ﺗﻤﺎﺱ، ﭼﮕﻮﻧﮕﯽ ﺧﻮﺭﺩﻥ ﺳﻴﻨﻪ ﻳﺎ ﺷﻴﺸﻪ ﺷﻴﺮ، ﺍﺯ ﻧﻈﺮ ﺳﺮﺷﺖ ﻭ ﺧﺼﻮﺻﻴﺎﺕ ﺍﺧﻼﻗﯽ، ﺗﺮﮐﻴﺐ ﻋﻀﻼﻧﯽ، ﮔﺮﻭﻩ ﺧﻮﻧﯽ ﻭ ﺗﻌﺎﺩﻝ ﻫﻮﺭﻣﻮﻧﯽ.

ﻓﺮﺯﻧﺪﺍﻥ ﺷﻤﺎ ﻧﻴﺰ ﻫﻤﮕﯽ ﺑﺎ ﻫﻢ ﺗﻔﺎﻭﺕ ﺧﻮﺍﻫﻨﺪ ﺩﺍﺷﺖ. ﺑﻪ ﺷﺨﺼﻴﺖ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﻳﺪ. ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺁﻧﻬﺎ ﮐﺎﻣﻼً ﺑﺎ ﺵِﻣﺎﯼ ﮐﻠﯽ ﺍﺯ ﻗﺒﻞ ﺩﺭﺳﺖ ﺷﺪﻩ‌ﺍﯼ ﻣﻄﺎﺑﻘﺖ ﻧﺪﺍﺭﻧﺪ. ﺑﺎ ﺑﻪ ﺧﺎﻃﺮ ﺳﭙﺮﺩﻥ ﺍﻳﻨﮑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻓﺮﺩﯼ ﮐﺎﻣﻼً ﻳﮑﺘﺎﺳﺖ، ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﺩﺭ ﺍﻭﻟﻴﻦ ﻣﺎﻩ‌ﻫﺎﯼ ﺯﻧﺪﮔﯽ، ﻧﻮﺯﺍﺩ ﭼﻪ ﺧﺼﻮﺻﻴﺎﺕ ﮐﻠﯽ‌ﺍﯼ ﺩﺍﺭﺩ.

 

ﺳﻪ ﻣﺎﻩ ﺍﻭﻝ
ﻧﻮﺯﺍﺩ ﺍﮐﺜﺮ ﺍﻭﻗﺎﺕﺵ ﺩﺭ ﺧﻮﺍﺏ ﻣﯽ‌ﮔﺬﺭﺩ، ﮔﺎﻫﯽ ﺣﺘﯽ ﺗﺎ ﺑﻴﺴﺖ ﺳﺎﻋﺖ ﺩﺭ ﺷﺒﺎﻧﻪ ﺭﻭﺯ (ﻭﻟﯽ ﺑﺎﺯ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻣﺘﻔﺎﻭﺕ ﺑﺎﺷﺪ). ﻧﻮﺯﺍﺩ، ﻣﻌﻤﻮﻻً ﺩﺭ ﺳﻪ ﻣﺎﻩ ﺍﻭﻝ ﺯﻧﺪﮔﯽ‌ﺍﺵ ﺁﺭﺍﻡ ﻭ ﺳﺎﮐﺖ ﺍﺳﺖ. ﺍﻭ ﻫﻨﻮﺯ ﻗﺎﺩﺭ ﺑﻪ ﺑﻠﻨﺪ ﮐﺮﺩﻥ ﺳﺮﺵ ﻧﻴﺴﺖ، ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻏﻠﺖ ﺑﺰﻧﺪ (ﻣﮕﺮ ﺑﻪ ﻃﻮﺭ ﺍﺗﻔﺎﻗﯽ) ﻭ ﻗﺎﺩﺭ ﺑﻪ ﺗﮑﺎﻥ ﺩﺍﺩﻥ ﺟﺪﺍﮔﺎﻧﻪ ﺍﻧﮕﺸﺘﺎﻥ ﻭ ﺷﺼﺘﺶ ﻧﻴﺴﺖ. ﺍﺯ ﻧﻈﺮ ﺍﺩﺭﺍﮐﯽ، ﺩﻧﻴﺎ ﺩﺭ ﻧﻈﺮ ﺍﻭ ﭼﻴﺰﯼ ﻋﻈﻴﻢ، ﻣﺘﺤﺮﮎ ﻭ ﭘﺮﻃﻨﻴﻦ ﺍﺳﺖ. ﻫﺮﭼﻨﺪ ﮐﻪ ﺍﺯ ﻫﻤﺎﻥ ﻟﺤﻈﻪ ﺍﻭﻝ ﺗﻮﻟﺪ ﺑﻪ ﺻﻮﺭﺕ ﺍﻓﺮﺍﺩ ﺩﻗﺖ ﻣﯽ‌ﻛﻨﺪ ﻭﻟﯽ ﻫﻨﻮﺯ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﺍﺯ ﻳﮑﺪﻳﮕﺮ ﺗﺸﺨﻴﺺ ﺑﺪﻫﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻧﻮﺯﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﭼﻴﺰﻫﺎﯼ ﻗﺎﺑﻞ ﺗﻮﺟﻬﯽ ﺭﺍ ﺩﺭ ﻣﻐﺰ ﮐﻮﭼﮑﺶ ﺛﺒﺖ ﮐﻨﺪ.

ﻧﺘﺎﻳﺞ ﺗﺤﻘﻴﻘﺎﺕ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﻧﻮﺯﺍﺩ ﺷﻤﺎ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﻣﯽ‌ﺷﻨﻮﺩ، ﻣﯽ‌ﺑﻮﻳﺪ ﻭ ﻣﯽ‌ﺑﻴﻨﺪ ﺩﺭ ﻣﻐﺰﺵ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﺍﺯ ﻟﺤﻈﻪ ﺗﻮﻟﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺑﻌﻀﯽ ﺗﺎﺛﺮﺍﺕ ﺧﺎﺭﺟﯽ ﺑﻪ ﻭﺳﻴﻠﻪ ﺣﻮﺍﺱ ﻟﺬﺕ ﺑﺒﺮﺩ ﻭ ﺑﻪ ﻣﺤﺾ ﺍﻳﻨﮑﻪ ﻗﺎﺩﺭ ﺑﺎﺷﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺍﻭ ﺑﺎﺯﯼ ﮐﻨﻴﺪ. ﺍﻟﺒﺘﻪ ﻣﻨﻈﻮﺭ ﺍﺯ ﺑﺎﺯﯼ ﮐﺮﺩﻥ، ﻗﺎﻳﻢ ﻣﻮﺷﮏ ﻭ ﭘﺮﺕ ﮐﺮﺩﻥ ﺍﻭ ﺑﻪ ﻫﻮﺍ ﻧﻴﺴﺖ. ﻣﻨﻈﻮﺭ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺩﺭ ﺳﻄﺢ ﺑﺴﻴﺎﺭ ﺍﺑﺘﺪﺍﻳﯽ ﺍﺳﺖ.

ﻣﺜﻼً ﺑﺎ ﺑﻪ ﮐﺎﺭ ﺍﻧﺪﺍﺧﺘﻦ ﺣﺲ ﺷﻨﻮﺍﻳﯽ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ. ﻧﻮﺯﺍﺩﻫﺎﯼ ﻛﻮﭼﻚ ﮔﻮﺵ ﺑﺴﻴﺎﺭ ﺣﺴﺎﺳﯽ ﺩﺍﺭﻧﺪ ﻭ ﺑﺎ ﺻﺪﺍﻫﺎﯼ ﺷﺪﻳﺪ ﻭ ﻧﺎﮔﻬﺎﻧﯽ ﺍﺯ ﺟﺎ ﻣﯽ‌ﭘﺮﻧﺪ. ﻭﻟﯽ ﺍﮔﺮ ﺑﺎ ﺁﻧﻬﺎ ﺑﻪ ﺁﺭﺍﻣﯽ ﺻﺤﺒﺖ ﮐﻨﻨﺪ ﻭ ﻳﺎ ﺁﻭﺍﺯ ﺑﺨﻮﺍﻧﻨﺪ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﻧﺪ. ﺁﻧﻬﺎ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﮐﻪ ﻭﻗﺘﯽ ﺻﺪﺍﯼ ﭘﺎﻳﯽ ﺑﺸﻨﻮﻧﺪ ﮔﺮﻳﻪ‌ﺷﺎﻥ ﺭﺍ ﻗﻄﻊ ﮐﻨﻨﺪ.

ﺑﺎ ﺍﻭ ﺻﺤﺒﺖ ﮐﻨﻴﺪ ﻭ ﺑﺮﺍﻳﺶ ﺁﻭﺍﺯ ﺑﺨﻮﺍﻧﻴﺪ. ﺻﺪﺍﻫﺎﯼ ﺟﺎﻟﺐ ﺑﺮﺍﻳﺶ ﺩﺭ ﺑﻴﺎﻭﺭﻳﺪ. ﻭﺳﺎﻳﻞ ﻣﻮﺳﻴﻘﯽ ﻣﺨﺘﻠﻒ ﺑﺮﺍﻳﺶ ﺑﺰﻧﻴﺪ. ﺻﺪﺍﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﺜﻞ ﺗﻴﮏ‌ﺗﻴﮏ ﺳﺎﻋﺖ، ﺻﺪﺍﯼ ﺑﺮﺧﻮﺭﺩ ﻗﺎﺷﻖ ﻭ ﺷﻴﺸﻪ ﺭﺍ ﺑﮕﺬﺍﺭﻳﺪ ﮔﻮﺵ ﮐﻨﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﺑﻴﺎﻳﺪ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺩﺭ ﺫﻫﻦ ﺍﻭ ﻫﻴﭻ ﺍﺛﺮﯼ ﻧﻤﯽ‌ﮔﺬﺍﺭﻧﺪ. ﺍﻣﺎ ﺍﻳﻦ ﺻﺪﺍﻫﺎ ﺩﺭ ﻣﻐﺰ ﺍﻭ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭﺗﺎﻥ ﺍﻧﮕﻴﺰﻩ ﺣﺴﯽ ﻣﺜﺒﺘﯽ ﺑﺮﺍﯼ ﺍﻭ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﮐﻨﻴﺪ.

ﭘﻮﺳﺖ ﺣﻴﻄﻪ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﺑﺮﺍﻧﮕﻴﺨﺘﻦ ﻧﻮﺯﺍﺩ ﺍﺳﺖ. ﺍﻭ ﺑﺎﺯﯼ ﻧﻮﺍﺯﺵ ﻭ ﺗﻤﺎﺱ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﺩ. ﻣﺜﻼً ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﻨﭻ ﺩﻗﻴﻘﻪ ﻗﺒﻞ ﻳﺎ ﺑﻌﺪ ﺍﺯ ﺣﻤﺎﻡ ﺍﻭ ﺭﺍ ﻣﺎﺳﺎﮊ ﺑﺪﻫﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻓﻘﻂ ﺍﮔﺮ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺑﮑﻨﻴﺪ ﻭ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﮐﺎﺭ ﮐﺴﻞ ﮐﻨﻨﺪﻩ‌ﺍﯼ ﺍﺳﺖ ﺍﺯ ﺁﻥ ﺻﺮﻓﻨﻈﺮ ﮐﻨﻴﺪ.

ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺳﺖ ﻭ ﭘﺎﻳﺶ ﺭﺍ ﺣﺮﮐﺖ ﺩﻫﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺍﺛﺮ ﺧﻮﺑﯽ ﺭﻭﯼ ﻋﻀﻼﺗﺶ ﺩﺍﺭﺩ. ﻭﻗﺘﯽ ﻃﺎﻕ‌ﺑﺎﺯ ﺧﻮﺍﺑﻴﺪﻩ، ﺩﺳﺖ‌ﻫﺎ ﻭ ﭘﺎﻫﺎﻳﺶ ﺭﺍ ﺑﻪ ﺩﻭ ﻃﺮﻑ ﺑﺎﺯ ﮐﺮﺩﻩ ﻭ ﺑﻌﺪ ﺭﻭﯼ ﺳﻴﻨﻪ‌ﺍﺵ ﺧﻢ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺣﺮﮐﺖ ﺭﺍ ﭼﻨﺪ ﺑﺎﺭ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﺎﻫﺎﻳﺶ ﺭﺍ ﻫﻢ ﺑﻠﻨﺪ ﮐﻨﻴﺪ ﻭ ﭘﺎﯼ ﺩﻭ ﭼﺮﺧﻪ ﺑﺰﻧﻴﺪ. ﺑﻪ ﻣﺤﺾ ﺍﻳﻨﮑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭﮎ ﮐﻨﺪ، ﺍﻳﻦ ﮐﺎﺭﻫﺎﯼ ﺷﻤﺎ ﺑﺎ ﺧﻨﺪﻩ ﻭ ﻟﺒﺨﻨﺪ ﺍﻭ ﻫﻤﺮﺍﻩ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﺧﺼﻮﺻﺎً ﺍﮔﺮ ﺍﻳﻦ ﺣﺮﮐﺎﺕ ﺑﺎ ﺁﻭﺍﺯ ﻳﺎ ﺻﺪﺍﯼ ﺟﺎﻟﺐ ﺩﻳﮕﺮﯼ ﻫﻤﺮﺍﻩ ﺑﺎﺷﻨﺪ.

ﭼﺸﻢ‌ﻫﺎ ﻫﻢ ﻭﺳﻴﻠﻪ ﺍﻧﮕﻴﺰﺷﯽ ﺩﻳﮕﺮ ﺍﺳﺖ. ﮔﻬﻮﺍﺭﻩ ﻧﻮﺯﺍﺩ ﺩﻧﻴﺎﯼ ﺍﻭﺳﺖ . ﺍﻳﻦ ﺩﻧﻴﺎ ﺑﺮﺍﯼ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻧﻮﺯﺍﺩﺍﻥ ﻏﻢ ﺍﻧﮕﻴﺰ ﺍﺳﺖ! ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺁﭘﺎﺭﺗﻤﺎﻧﯽ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﻏﻴﺮ ﺍﺯ ﺩﺭ ﻭ ﺩﻳﻮﺍﺭ ﭼﻴﺰ ﺩﻳﮕﺮﯼ ﻧﺒﺎﺷﺪ. ﺩﻧﻴﺎﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﺎ ﭼﻴﺰﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻣﻔﺮﺡ ﺳﺎﺯﻳﺪ: ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﮑﻪ‌ﻫﺎﯼ ﭘﺎﺭﭼﻪ ﻳﺎ ﮐﺎﻏﺬ ﻳﺎ ﻧﺎﻳﻠﻮﻥ ﺑﻪ ﺭﻧﮓ‌ﻫﺎﯼ ﺗﻨﺪ ﻭ ﻧﻘﺶ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺭﺍ ﮐﻨﺎﺭ ﺁﻥ ﺁﻭﻳﺰﺍﻥ ﮐﻨﻴﺪ. ﺑﻴﺸﺘﺮ ﻣﺎﺩﺭﺍﻥ ﺍﻳﻦ ﻭﺳﻴﻠﻪ ﺭﺍ ﺑﺎﻻﯼ ﺳﺮ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺁﻭﻳﺰﺍﻥ ﻣﯽ‌ﮐﻨﻨﺪ. ﺩﺭ ﻃﻮﻝ ﺷﺶ ﻫﻔﺘﻪ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﮐﺎﺭ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﺍﻳﻦ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ ﺑﺎﻳﺪ ﻃﺮﻑ ﺭﺍﺳﺖ ﻳﺎ ﭼﭗ ﮔﻬﻮﺍﺭﻩ ﺁﻭﻳﺰﺍﻥ ﺷﻮﺩ. ﺍﺯ ﻧﻈﺮ ﻓﻴﺰﻳﻮﻟﻮﮊﻳﮏ، ﻧﻮﺯﺍﺩ ﺑﻌﺪ ﺍﺯ ﺷﺶ ﻫﻔﺘﮕﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺗﺪﺭﻳﺞ ﻗﺎﺩﺭ ﺑﻪ ﭼﺮﺧﺎﻧﺪﻥ ﺳﺮﺵ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﮕﺎﻫﺶ ﺭﺍ ﻣﺘﻮﺟﻪ ﺳﻘﻒ ﮐﻨﺪ.

ﺩﺭ ﺿﻤﻦ ﺑﺎﻳﺪ ﺗﺎﮐﻴﺪ ﮐﻨﻴﻢ ﮐﻪ ﻧﺒﺎﻳﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﻫﻤﻴﺸﻪ ﺩﺭ ﮔﻬﻮﺍﺭﻩ‌ﺍﺵ ﺑﮕﺬﺍﺭﻳﺪ. ﮔﻬﮕﺎﻩ ﺍﻭ ﺭﺍ ﺭﻭﯼ ﺯﺍﻧﻮﻳﺘﺎﻥ ﺑﮕﻴﺮﻳﺪ ﺗﺎ ﺑﺘﻮﺍﻧﺪ ﺍﺷﻴﺎ ﺭﺍ ﺍﺯ ﺯﺍﻭﻳﻪ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﺑﺒﻴﻨﺪ. ﻳﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺭﻭﯼ ﻳﮏ ﺻﻨﺪﻟﯽ ﮐﻮﺗﺎﻩ ﻣﺨﺼﻮﺹ ﺑﭽﻪ ﺩﺭ ﺍﺗﺎﻕ ﺩﻳﮕﺮﯼ ﻏﻴﺮ ﺍﺯ ﺍﺗﺎﻕ ﺧﻮﺩﺵ ﺑﮕﺬﺍﺭﻳﺪ ﮐﻪ ﺑﺒﻴﻨﺪ ﭼﻪ ﻣﯽ‌ﮔﺬﺭﺩ.

ﻭﻗﺘﯽ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﺑﺮﺍﻳﺶ ﺁﻭﺍﺯ ﻣﯽ‌ﺧﻮﺍﻧﻴﺪ، ﺍﻭ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﺫﻫﻨﺶ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺟﻮﺍﺑﯽ ﺑﻪ ﺷﻤﺎ ﻧﻤﯽ‌ﺩﻫﺪ. ﻭﻗﺘﯽ ﺩﻭ ﻣﺎﻫﻪ ﺷﺪ، ﺷﺎﻳﺪ ﺷﺎﻫﺪ ﺗﻐﻴﻴﺮﯼ ﺑﺎﺷﻴﺪ.

ﻭﻗﺘﯽ ﺑﺎ ﺍﻭ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺑﺮﺍﯼ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺑﺎ ﺍﻭ  ﺻﺪﺍﻫﺎﯼ ﻋﺠﻴﺐ ﻭ ﻏﺮﻳﺐ ﺩﺭ ﻣﯽ‌ﺁﻭﺭﻳﺪ، ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﺍﻭ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﺑﻪ ﺷﻤﺎ «ﺟﻮﺍﺏ» ﺑﺪﻫﺪ. ﺩﻫﺎﻧﺶ ﺭﺍ ﺑﺎﺯ ﻣﯽ‌ﮐﻨﺪ، ﮔﻮﻳﯽ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﺎ ﺷﻤﺎ ﺻﺤﺒﺖ ﮐﻨﺪ. ﺑﻌﺪﻫﺎ ﺍﻳﻦ ﺻﺪﺍﻫﺎﯼ ﺑﯽ‌ﻣﻌﻨﯽ ﺗﺒﺪﻳﻞ ﺑﻪ ﮔﻔﺘﮕﻮﯼ «ﻧﻮﻋﯽ» ﺑﻴﻦ ﻣﺎﺩﺭ ﻭ ﻓﺮﺯﻧﺪ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﮐﻮﺷﺶ‌ﻫﺎﯼ ﻧﻮﺯﺍﺩ ﺑﺮﺍﯼ ﺟﻮﺍﺏ ﺩﺍﺩﻥ ﺑﻪ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺗﻘﺮﻳﺒﺎً ﻣﻘﺎﺭﻥ ﺑﺎ ﻣﺮﺣﻠﻪ ﻣﻬﻤﯽ ﺍﺯ ﺭﺷﺪ ﺍﻭﺳﺖ: ﻫﺪﺍﻳﺖ ﺩﻳﺪ، ﮐﻪ ﻳﮑﯽ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻳﻦ ﮐﺸﻒ‌ﻫﺎﯼ ﺍﻭ ﺩﺭ ﻳﺎﺩﮔﻴﺮﯼ ﺑﺮﺍﯼ ﺗﺴﻠﻂ ﺑﺮ ﺟﻬﺎﻥ‌ﺍﺵ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻧﻮﺯﺍﺩ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻗﺎﺩﺭ ﺑﻪ ﺩﻳﺪﻥ ﻳﮏ ﺷﯽﺀ، ﮔﺮﻓﺘﻦ ﻭ ﻟﻤﺲ ﺁﻥ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺩﺭِ ﺟﻬﺎﻥ ﺩﻳﮕﺮﯼ ﺑﻪ ﺭﻭﯼ ﺍﻭ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ.

ﺍﺣﺘﻴﺎﺟﺎﺕ ﻧﻮﺯﺍﺩ ﺷﻤﺎ ﺑﻪ ﻣﺎﻳﺤﺘﺎﺝ ﺭﻭﺯﻣﺮﻩ ﺧﻼﺻﻪ ﻧﻤﯽ‌ﺷﻮﺩ، ﺍﻭ ﻣﺤﺘﺎﺝ ﺑﻪ ﺷﺮﮐﺖ ﺩﺭ ﺩﻧﻴﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺍﻭ ﺭﺍ ﺩﺭ ﺑﺮﮔﺮﻓﺘﻪ. ﺍﻭ ﺑﺎﻳﺪ ﺑﻴﺎﻣﻮﺯﺩ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻌﻀﯽ ﮐﺎﺭﻫﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﻭ ﺭﻭﯼ ﺍﻳﻦ ﺩﻧﻴﺎ ﺗﺎﺛﻴﺮ ﺑﮕﺬﺍﺭﺩ. ﻗﺒﻼً ﺭﻭﯼ ﺿﺮﻭﺭﺕ ﺁﻭﺍﺯ ﺧﻮﺍﻧﺪﻥ ﻭ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺑﺎ ﺍﻭ ﺗﺎﮐﻴﺪ ﮐﺮﺩﻩ‌ﺍﻳﻢ. ﺟﻮﺍﺏ ﺩﺍﺩﻥ ﺑﻪ ﺻﺪﺍﻫﺎﻳﯽ ﮐﻪ ﺍﻭ ﺍﺯ ﺧﻮﺩ ﺩﺭ ﻣﯽ‌ﺁﻭﺭﺩ ﻧﻴﺰ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭ ﺩﺍﺭﺩ. ﻓﺮﺯﻧﺪﺗﺎﻥ ﺧﻴﻠﯽ ﺯﻭﺩ ﺑﺎ ﺻﺪﺍﻫﺎ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ. ﺟﻮﺍﺏ ﺩﺍﺩﻥ ﺑﻪ ﺧﻤﻴﺎﺯﻩ ﻭ ﻳﺎ ﺻﺪﺍﻫﺎﯼ ﺑﯽ ﻣﻌﻨﯽ ﺍﻭ، ﺑﺎ ﻫﻤﺎﻥ ﻟﺤﻦ ﺻﺪﺍﯼ ﺧﻮﺩﺵ، ﺑﺮﺍﯼ ﺍﻭ ﺧﻴﻠﯽ ﺟﺎﻟﺐ ﺍﺳﺖ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻭ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﺍﻃﺮﺍﻓﻴﺎﻧﺶ ﺭﺍ ﺩﺭ ﻣﻘﺎﺑﻞ ﻋﻤﻠﺶ ﺩﺭﻳﺎﻓﺖ ﻣﯽ‌ﮐﻨﺪ. ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﺍﻭ ﻣﺜﻞ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﻳﮏ ﮐﻮﺩﮎ ﺑﺎ ﻳﮏ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ ﺟﺪﻳﺪ ﺍﺳﺖ. ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﮐﻪ ﺍﻳﻦ ﺻﺪﺍﻫﺎ ﺭﺍ ﺗﮑﺮﺍﺭ ﮐﻨﺪ ﺗﺎ ﺟﻮﺍﺑﺶ ﺭﺍ ﺑﺪﻫﻨﺪ.‌

ﻧﻮﺯﺍﺩﺗﺎﻥ ﺍﺯ ﺍﻳﻦ ﻣﺒﺎﺩﻟﻪ ﺳﻮﺩ ﻓﺮﺍﻭﺍﻧﯽ ﻣﯽ‌ﺑﺮﺩ. ﺍﻭ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﯽ‌ﺁﻣﻮﺯﺩ ﻭ ﻣﯽ‌ﺑﻴﻨﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻌﻀﯽ ﮐﺎﺭﻫﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﻭ ﺩﺭ ﺩﻧﻴﺎﻳﺶ ﺗﺎﺛﻴﺮ ﺑﮕﺬﺍﺭﺩ. ﺍﻭ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺟﻬﺎﻧﯽ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺁﻥ ﺷﺮﮐﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﮐﻪ ﺍﺯ ﻫﻤﺎﻥ ﺳﻨﻴﻦ ﺑﺴﻴﺎﺭ ﭘﺎﻳﻴﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﭘﻴﺪﺍ ﮐﻨﺪ.‌

ﺩﺭﺳﺖ ﮐﺮﺩﻥ ﺩﻓﺘﺮﭼﻪ ﮐﻮﭼﮑﯽ ﺍﺯ ﺗﮑﻪ‌ﻫﺎﯼ ﭘﺎﺭﭼﻪ ﺍﺯ ﺟﻨﺲ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺭﺍﻩ ﺩﻳﮕﺮﯼ ﺟﻬﺖ ﺗﺸﻮﻳﻖ ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ ﺍﺳﺖ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﭘﺎﺭﭼﻪ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺯ ﺟﻨﺲ ﻭ ﺭﻧﮓ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﻓﺘﺮﭼﻪ‌ﻫﺎﯼ ﺑﻪ ﻃﻮﻝ ﻭ ﻋﺮﺽ ﺗﻘﺮﻳﺒﺎً ﺷﺼﺖ ﺳﺎﻧﺘﻴﻤﺘﺮ ﺑﺮﺍﯼ ﺍﻭ ﺑﺪﻭﺯﻳﺪ. ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎ ﻟﻤﺲ ﮐﺮﺩﻥ ﺍﻳﻦ ﭘﺎﺭﭼﻪ‌ﻫﺎ ﺗﻔﺎﻭﺕ ﺑﻴﻦ ﺁﻧﻬﺎ ﺭﺍ ﺍﺣﺴﺎﺱ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺣﺮﮐﺘﯽ ﮐﻮﭼﮏ ﺍﺯ ﭘﺎﺭﭼﻪ‌ﺍﯼ ﺑﻪ ﭘﺎﺭﭼﻪ ﺩﻳﮕﺮ، ﺗﺠﺮﺑﻪ ﺟﺪﻳﺪﯼ ﺍﺯ ﺩﻧﻴﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺍﻭ ﺭﺍ ﺩﺭ ﺑﺮﮔﺮﻓﺘﻪ ﻭ ﺍﻳﻦ ﺗﺠﺮﺑﻪ ﺍﻭ ﺭﺍ ﺑﺮﺍﯼ ﮐﺸﻒ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﻭ ﻳﺎﻓﺘﻦ ﺟﻮﺍﺏ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﺪ.

ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﺑﺮﺍﯼ ﺍﻭ ﺩﺭﺳﺖ ﮐﻨﻴﺪ ﮐﻪ ﺍﻭ ﺭﺍ ﺩﺭ ﺟﻬﺖ ﺩﺭﮎ ﻣﻮﺛﺮ ﺑﻮﺩﻥ ﺩﺭ ﺩﻧﻴﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ. ﻗﻄﻌﻪ‌ﺍﯼ ﻧﺦ ﺿﺨﻴﻢ ﻳﺎ ﻃﻨﺎﺏ ﻳﺎ ﮐﺶ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﻭ ﺑﻪ ﺁﻥ ﻣﺜﻼً ﻳﮏ ﺳﺮﯼ ﻗﺎﺷﻖ ﻳﺎ ﻳﮏ ﺩﺳﺘﺒﻨﺪ ﭘﻼﺳﺘﻴﮑﯽ ﺭﻧﮕﯽ ﮐﻪ ﺧﻄﺮﯼ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺁﻭﻳﺰﺍﻥ ﮐﻨﻴﺪ. ﺍﺷﻴﺎﻳﯽ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﺪ ﻧﺒﺎﻳﺪ ﺧﻴﻠﯽ ﮐﻮﭼﮏ ﺑﺎﺷﻨﺪ، ﮐﻪ ﺍﮔﺮ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺩﻫﺎﻧﺶ ﮔﺬﺍﺷﺖ ﺧﻄﺮ ﺧﻔﮕﯽ ﺑﺮﺍﻳﺶ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻧﺦ ﻳﺎ ﮐﺶ ﻣﺰﺑﻮﺭ ﺭﺍ ﺩﺭﺳﺖ ﺟﻠﻮ ﺩﺳﺘﺶ ﺁﻭﻳﺰﺍﻥ ﮐﻨﻴﺪ ﮐﻪ ﺑﺘﻮﺍﻧﺪ ﺁﻥ ﺭﺍ ﺑﺎ ﺩﺳﺘﺶ ﺑﮕﻴﺮﺩ. ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﮐﻪ ﮐﺶ ﻳﺎ ﻧﺨﯽ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﻣﺤﮑﻢ ﻭ ﺿﺨﻴﻢ ﺑﺎﺷﺪ ﮐﻪ ﻫﻨﮕﺎﻡ ﺑﺎﺯﯼ ﺑﺎ ﺁﻥ ﺑﻪ ﻃﺮﺯ ﺧﻄﺮﻧﺎﮐﯽ ﺑﻪ ﺩﺳﺖ ﻭ ﭘﺎﻳﺶ ﮔﻴﺮ ﻧﮑﻨﺪ ﻭ ﺧﻮﻧﺶ ﺭﺍ ﺑﻨﺪ ﻧﻴﺎﻭﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﺯ ﻧﺦ ﻗﺮﻗﺮﻩ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﺪ.


ﺳﻪ ﺗﺎ ﺷﺶ ﻣﺎﻫﮕﯽ  
ﺳﻪ ﻣﺎﻫﮕﯽ ﻧﺸﺎﻥ ﺩﻫﻨﺪﻩ ﻳﮏ ﮔﺬﺍﺭ ﺍﺳﺖ. ﮔﺬﺍﺭ ﺍﺯ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺍﻧﻔﻌﺎﻟﯽ ﻭ ﻏﻴﺮﺍﺭﺍﺩﯼ ﺑﻪ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﻓﻌﺎﻝ ﮐﻪ ﺩﺭ ﺁﻥ ﻧﻮﺯﺍﺩ ﺍﺯ ﻃﺮﻳﻖ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﭼﻴﺰﻫﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺭﺍ ﮐﺸﻒ ﮐﻨﺪ ﻭ ﺑﺸﻨﺎﺳﺪ.

ﺑﺮﺭﺳﯽ ﻧﻮﺯﺍﺩ ﮐﻮﭼﮏ ﺍﺯ ﺩﻧﻴﺎ ﺩﺭ ﻃﯽ ﺳﻪ ﻣﺎﻩ ﺍﻭﻟﻴﻪ ﺯﻧﺪﮔﻴﺶ، ﺍﺯ ﻃﺮﻳﻖ ﭼﺸﻢ‌ ﻭ ﮔﻮﺵ‌ ﻭ ﺩﻫﺎﻧﺶ (ﺍﺯ ﻃﺮﻳﻖ ﻣﮑﻴﺪﻥ) ﺍﺳﺖ. ﭘﺲ ﺍﺯ ﺁﻥ ﺍﻭ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﺎ ﺩﺳﺘﺶ ﻣﯽ‌ﮐﻨﺪ. ﭼﻴﺰﯼ ﮐﻪ ﺩﮐﺘﺮ ﮔﺰﻝ  «ﻋﻄﺶ ﻟﻤﺲ ﮐﺮﺩﻥ» ﻧﺎﻣﻴﺪﻩ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ. ﺍﻭ ﺷﻴﻔﺘﻪ ﮔﺮﻓﺘﻦ ﺍﺷﻴﺎﺀ ﺍﺳﺖ ﻭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺑﺎ ﻟﻤﺲ ﮐﺮﺩﻥ ﺁﻧﻬﺎ ﮐﺸﻒ ﮐﻨﺪ ﭼﻪ ﻫﺴﺘﻨﺪ. ﺗﻘﺮﻳﺒﺎً ﺩﺭ ﭼﻬﺎﺭ ﻣﺎﻫﮕﯽ ﺍﻫﻤﻴﺖ ﺩﺳﺖ‌ﻫﺎﻳﺶ ﺑﻴﺸﺘﺮ ﻣﯽ‌ﺷﻮﺩ. ﺗﺎ ﺑﻪ ﺣﺎﻝ ﺍﺯ ﻃﺮﻳﻖ ﺩﻳﺪ ﮐﺸﻒ ﮐﺮﺩﻩ ﺑﻮﺩ ﮐﻪ ﺍﺷﻴﺎﺀ ﺷﮑﻞ ﻭ ﺭﻧﮕﯽ ﺩﺍﺭﻧﺪ. ﻭﻟﯽ ﺣﺎﻻ ﺍﺯ ﻃﺮﻳﻖ ﺩﺳﺖ‌ﻫﺎﻳﺶ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﺟﺴﺎﻡ ﺧﺼﺎﻳﻞ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﺩﺍﺭﻧﺪ: ﻧﺮﻣﯽ، ﺯﺑﺮﯼ، ﺍﻧﺴﺠﺎﻡ. ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻧﻮﺯﺍﺩ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺑﻪ ﺩﻫﺎﻧﺶ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺩﺭ ﻭﺍﻗﻊ ﺩﻫﺎﻥ، ﻳﮑﯽ ﺍﺯ ﺣﻮﺍﺱ ﺍﺻﻠﯽ ﺍﻭ ﺑﺮﺍﯼ ﮐﺸﻒ ﺩﻧﻴﺎﺳﺖ. ﺍﻭ ﺗﺎ ﺳﺎﻝ‌ﻫﺎ ﺍﻳﻦ ﺧﺼﻠﺖ ﺭﺍ ﺣﻔﻆ ﻣﯽ‌ﮐﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺧﻴﻠﯽ ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﮐﻪ ﭼﻴﺰﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ ﺩﺭ ﺩﺳﺘﺮﺱ ﺍﻭ ﻧﺒﺎﺷﺪ. ﺣﺎﻻﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺍﺷﻴﺎﯼ ﻇﺮﻳﻒ ﻭ ﺷﮑﻨﻨﺪﻩ ﻭ ﭼﻴﺰﻫﺎﯼ ﺯﻳﺒﺎ ﻭ ﻧﻮﮎ ﺗﻴﺰ ﺭﺍ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺒﻠﻌﺪ ﺟﻤﻊ ﮐﻨﻴﺪ (ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺍﺷﻴﺎﺀ ﺗﺎ ﺳﻪ ﻣﺎﻫﮕﯽ ﺑﺮﺍﯼ ﮔﺴﺘﺮﺵ ﺩﻳﺪ ﺍﻭ ﺑﺴﻴﺎﺭ ﻣﻔﻴﺪ ﺑﻮﺩﻩ ﺑﺎﺷﻨﺪ). ﺍﮐﻨﻮﻥ ﺑﺎﻳﺪ ﺍﺷﻴﺎﻳﯽ ﺩﺭ ﺩﺳﺘﺮﺱ ﺍﻭ ﺑﺎﺷﺪ ﮐﻪ ﻣﺤﮑﻢ ﺑﺎﺷﻨﺪ ﻭ ﺍﻭ ﻧﺘﻮﺍﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﺑﺒﻠﻌﺪ.

ﺟﻐﺠﻐﻪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺟﺎﯼ ﺧﺎﺻﯽ ﺩﺍﺭﺩ. ﺍﮔﺮ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﻮﺍﻉ ﻭ ﺍﻗﺴﺎﻡ ﺟﻐﺠﻐﻪ ﺑﺮﺍﻳﺶ ﺑﺨﺮﻳﺪ. ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺩﻳﮕﺮﯼ ﻫﻢ ﮐﻪ ﺑﺮﺍﯼ ﺳﻨﻴﻦ ﺑﺎﻻﺗﺮ ﻫﺴﺖ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﺍﺳﺖ، ﻋﺮﻭﺳﮏ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﮐﺎﺋﻮﭼﻮﻳﯽ ﮐﻪ ﺑﻪ ﻫﻤﻪ ﻃﺮﻑ ﺧﻢ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺑﭽﻪ‌ﻫﺎﯼ ﺍﻳﻦ ﺳﻦ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﺩﻫﺎﻥ ﺑﺮﺩﻩ ﻭ ﮔﺎﺯ ﺑﮕﻴﺮﻧﺪ. ﺍﺳﺒﺎ‌ﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﮐﺎﺋﻮﭼﻮﻳﯽ ﮐﻪ ﺑﺘﻮﺍﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻓﺸﺎﺭ ﺩﺍﺩ ﻧﻴﺰ ﻣﻨﺎﺳﺐ ﺍﻳﻦ ﺳﻦ ﻫﺴﺘﻨﺪ. ﻭﻟﯽ ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﺯﻳﺮﺍ ﺑﻌﻀﯽ ﺍﺯ ﺁﻧﻬﺎ ﺳﻮﺳﻮﺗﮑﯽ ﻓﻠﺰﯼ ﺩﺍﺭﻧﺪ ﮐﻪ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﺁﻥ ﺟﺪﺍ ﺷﻮﺩ ﻭ ﺩﺭ ﺍﻳﻦ  ﺻﻮﺭﺕ ﺧﻄﺮ ﺧﻔﮕﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﺳﭙﺲ ﺯﻣﺎﻧﯽ ﻣﯽ‌ﺭﺳﺪ ﮐﻪ ﮐﻮﺩﮎ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺑﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﻳﺎ ﻋﺮﻭﺳﮏ‌ﻫﺎﯼ ﻧﺮﻡ (ﭘﺮﺯﺩﺍﺭ ﻭ ﻧﺮﻡ) ﺑﺎﺯﯼ ﮐﻨﺪ. ﻣﻮﺍﻇﺐ ﭼﺸﻢ‌ﻫﺎﯼ ﺷﻴﺸﻪ‌ﺍﯼ ﻋﺮﻭﺳﮏ ﺑﺎﺷﻴﺪ ﮐﻪ ﻃﻔﻞ ﻣﻤﮑﻦ ﺍﺳﺖ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﺁﻭﺭﺩﻩ ﻭ ﺑﻪ ﺩﻫﺎﻧﺶ ﺑﮕﺬﺍﺭﺩ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺎﺯ ﻫﻢ ﺧﻄﺮ ﺧﻔﮕﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﺨﺼﻮﺹ ﻧﻮﺍﺯﺵ ﮐﺮﺩﻥ ﻫﺴﺘﻨﺪ ﺍﺯ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ ﻓﺮﻭﺷﯽ ﺧﺮﻳﺪ.

ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺧﻮﺩﺗﺎﻥ ﻫﻢ ﺍﺯ ﺍﻳﻦ ﻧﻮﻉ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﺩﺭﺳﺖ ﮐﻨﻴﺪ. ﻗﺴﻤﺖ ﺑﻴﺮﻭﻧﯽ ﺑﺎﻳﺪ ﻗﺎﺑﻞ ﺷﺘﺸﻮ ﺑﺎﺷﻨﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺜﻼً ﺍﺯ ﻳﮏ ﺣﻮﻟﻪ ﮐﻬﻨﻪ ﺑﺮﺍﯼ ﺍﻳﻦ ﻣﻨﻈﻮﺭ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﺯ ﺍﺑﺮ، ﮐﺎﺋﻮﭼﻮ ﻳﺎ ﺗﻜﻪ‌ﻫﺎﯼ ﭘﺎﺭﭼﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﭘﺮﮐﺮﺩﻥ ﺗﻮﯼ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

ﮔﺬﺍﺷﺘﻦ ﻧﻮﺯﺍﺩ ﺩﺭ ﭘﺎﺭﮎ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺷﺮﻭﻉ ﺷﻮﺩ. ﻳﻌﻨﯽ ﺣﺪﻭﺩ ﭼﻬﺎﺭﻣﺎﻫﮕﯽ، ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﺑﺘﻮﺍﻧﺪ ﺑﻨﺸﻴﻨﺪ ﻭ ﺳﻴﻨﻪ ﺧﻴﺰ ﺑﺮﻭﺩ ﻭ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﺑﺪﺍﻧﺪ ﮐﻪ ﺭﻭﯼ ﺯﻣﻴﻦ، ﻣﯽ‌ﺗﻮﺍﻧﺪ «ﻓﺮﺍﺭ» ﮐﻨﺪ. ﺍﮔﺮ ﺑﻌﺪﺍً ﺍﻭ ﺭﺍ ﺩﺭ ﭘﺎﺭﮎ ﺑﮕﺬﺍﺭﻳﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﻧﻈﺮﺵ ﺯﻧﺪﺍﻥ ﺑﻴﺎﻳﺪ. ﭘﺎﺭﮎ ﺍﻭ ﺭﺍ ﺟﺎﻳﯽ ﺑﮕﺬﺍﺭﻳﺪ ﮐﻪ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻴﺪ، ﺍﺗﺎﻕ ﻧﺸﻴﻤﻦ ﻳﺎ ﺁﺷﭙﺰﺧﺎﻧﻪ، ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻭ ﺍﺯ ﺑﺎ ﺷﻤﺎ ﺑﻮﺩﻥ ﻟﺬﺕ ﺑﻴﺸﺘﺮﯼ ﻣﯽ‌ﺑﺮﺩ. ﻭﻟﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺩﺭ ﭘﺎﺭﮐﺶ ﺑﺎ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﮔﺬﺍﺷﺘﻪ‌ﺍﻳﺪ ﻫﻢ ﺑﮕﺬﺍﺭﻳﺪ، ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻭ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ‌ﻫﺎ ﺑﺎﺯﯼ ﮐﻨﺪ.
