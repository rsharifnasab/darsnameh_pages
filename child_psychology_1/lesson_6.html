ﺩﺭﺱ ﭘﻨﺠﻢ- ﭼﺸﻢ ﺍﻧﺪﺍﺯ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ - ﺑﺨﺶ ﺩﻭﻡ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺑﺮﺭﺳﯽ ﺭﺷﺪ ﻧﻮﺯﺍﺩﺗﺎﻥ ﺩﺭ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﺭﺍ ﺷﺮﻭﻉ ﮐﺮﺩﻳﻢ. ﺑﺮﺍﯼ ﺭﺍﺣﺘﯽ ﮐﺎﺭ ﺍﻳﻦ ﺳﺎﻝ ﺭﺍ ﺑﻪ ﭼﻬﺎﺭ ﺩﻭﺭﻩ‌ﯼ ﺳﻪ ﻣﺎﻫﻪ ﺗﻘﺴﻴﻢ ﮐﺮﺩﻳﻢ ﻭ ﻧﮕﺎﻫﯽ ﺍﺟﻤﺎﻟﯽ ﺑﻪ ﺷﺶ ﻣﺎﻩ ﺍﻭﻝ ﺍﻧﺪﺍﺧﺘﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺷﺶ ﻣﺎﻩ ﺩﻭﻡ ﺍﺯ ﺳﺎﻝ ﻧﺨﺴﺖ ﺯﻧﺪﮔﯽ ﮐﻮﺩﮎ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﻢ.

ﺷﺶ ﺗﺎ ﻧﻪ ﻣﺎﻫﮕﯽ                                                                                                   
ﺣﺪﻭﺩ ﺷﺶ ﻣﺎﻫﮕﯽ ﻣﻘﺎﺭﻥ ﺑﺎ ﻇﻬﻮﺭ ﺍﺣﺴﺎﺱ ﺗﺎﺯﻩ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺁﻥ ﺭﺍ ﺗﺮﺱ ﺍﺯ ﺗﺎﺯﮔﯽ ﻧﺎﻣﻴﺪﻩ‌ﺍﻧﺪ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺭ ﺷﺶ ﻣﺎﻩ ﺍﻭﻟﻴﻪ ﺯﻧﺪﮔﻴﺶ ﻣﻔﻬﻮﻡ ﻣﺸﺨﺼﯽ ﺍﺯ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﺍﻭ ﺁﺷﻨﺎ ﻫﺴﺘﻨﺪ، ﺻﻮﺭﺕ ﻭ ﺍﻓﺮﺍﺩ، ﮐﺴﺐ ﻣﯽ‌ﮐﻨﺪ. ﺍﮐﻨﻮﻥ ﺫﻫﻦ ﺍﻭ ﺑﻪ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﺭﺷﺪ ﮐﺮﺩﻩ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﭼﻴﺰﻫﺎ ﻭ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﺮﺍﯼ ﺍﻭ ﺁﺷﻨﺎ ﻧﻴﺴﺘﻨﺪ ﺗﺸﺨﻴﺺ ﺩﻫﺪ. ﮔﺬﺍﺷﺘﻦ ﻧﻮﺯﺍﺩ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻧﺰﺩ ﺍﻓﺮﺍﺩ ﻧﺎﺷﻨﺎﺱ ﺑﺎﻳﺪ ﺑﺴﻴﺎﺭ ﺑﻪ ﺗﺪﺭﻳﺞ ﺍﻧﺠﺎﻡ ﮔﻴﺮﺩ. ﺍﻭ ﺭﺍ ﻧﺎﮔﻬﺎﻥ ﺩﺭ ﺷﺮﺍﻳﻂ ﺟﺪﻳﺪﯼ ﻗﺮﺍﺭ ﻧﺪﻫﻴﺪ. ﺍﮔﺮ ﺁﺷﻨﺎﻳﯽ ﻧﻮﺯﺍﺩ ﺑﺎ ﻓﺮﺩ ﻳﺎ ﺍﻓﺮﺍﺩﯼ ﺍﻭ ﺭﺍ ﺑﻪ ﮔﺮﻳﻪ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ، ﺩﺭ ﻭﺍﻗﻊ ﺍﻭ ﺑﺎ ﮔﺮﻳﻪ ﮐﺮﺩﻧﺶ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﮔﻮﻳﺪ ﮐﻪ ﻣﯽ‌ﺗﺮﺳﺪ. ﺑﻪ ﺍﻭ ﻓﺮﺻﺖ ﺑﺪﻫﻴﺪ، ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺑﻬﺘﺮ.

ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﻭ ﻋﻼﻗﻪ ﺯﻳﺎﺩﯼ ﺑﻪ ﺻﺪﺍ ﺩﺭ ﺁﻭﺭﺩﻥ ﺍﺯ ﺧﻮﺩﺵ ﺩﺍﺭﺩ، ﺍﮐﻨﻮﻥ ﺁﻥ ﺻﺪﺍﻫﺎﯼ ﺑﯽ‌ﻣﻌﻨﯽ ﮐﻪ ﺣﺪﻭﺩ ﺩﻭ ﺳﻪ ﻣﺎﻫﮕﯽ ﺻﺤﺒﺖ ﺑﻴﻦ ﺷﻤﺎ ﻭ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﺍﺩ ﺗﺒﺪﻳﻞ ﺑﻪ ﻳﮏ ﺳﻴﺴﺘﻢ ﺛﺎﺑﺖ ﻭ ﻣﻨﺴﺠﻢ ﺑﺎﺯﯼ ﮐﻠﻤﺎﺕ ﻣﯽ‌ﺷﻮﺩ. ﻣﻌﻤﻮﻻ ﺍﻭﻟﻴﻦ ﺩﻧﺪﺍﻥ ﺣﺪﻭﺩ ﺳﻦ ﻫﻔﺖ ﻣﺎﻫﮕﯽ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ. ﺑﻪ ﻫﻨﮕﺎﻡ ﺩﻧﺪﺍﻥ ﺩﺭ ﺁﻭﺭﺩﻥ، ﮐﻮﺩﮎ ﺍﺣﺘﻴﺎﺝ ﻓﺮﺍﻭﺍﻧﯽ ﺑﻪ ﮔﺎﺯ ﮔﺮﻓﺘﻦ ﺍﺷﻴﺎﺀ ﺩﺍﺭﺩ، ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﺍﻭ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺘﻮﺍﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﻣﻴﺎﻥ ﺁﺭﻭﺍﺭﻩ‌ﻫﺎﻳﺶ ﻓﺸﺎﺭ ﺑﺪﻫﺪ ﺗﻬﻴﻪ ﮐﻨﻴﺪ.

ﺩﺭ ﻓﺎﺻﻠﻪ ﺑﻴﻦ ﺷﺶ ﺗﺎ ﻧﻪ ﻣﺎﻫﮕﯽ ﮐﻮﺩﮎ ﺷﻴﻔﺘﻪ ﺗﮑﺮﺍﺭ ﺍﺳﺖ. ﺍﻭ ﻳﮏ ﮐﺎﺭ ﺭﺍ ﺁﻧﻘﺪﺭ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺍﺣﺴﺎﺱ ﮐﻨﺪ ﺑﺮ ﺁﻥ ﺗﺴﻠﻂ ﮐﺎﻣﻞ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﺍﺳﺖ. ﻣﺜﻼ ﭼﻴﺰﯼ ﮐﻪ ﺭﻭﯼ ﻣﻴﺰ ﺻﻨﺪﻟﯽ‌ﺍﺵ ﺍﺳﺖ ﺑﺮ ﻣﯽ‌ﺩﺍﺭﺩ ﻭ ﺑﺎﺯ ﺑﺮ ﻣﯽ‌ﺩﺍﺭﺩ. ﺷﺨﺺ ﺑﺎﻟﻎ ﺍﺯ ﺗﮑﺮﺍﺭ ﻣﮑﺮﺭﺍﺕ ﺧﻴﻠﯽ ﺯﻭﺩ ﺧﺴﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻟﺬﺗﯽ ﺭﺍ ﮐﻪ ﻧﻮﺯﺍﺩ ﺍﺯ ﺗﮑﺮﺍﺭ ﮐﺎﺭﻫﺎ ﻣﯽ‌ﺑﺮﺩ ﺩﺭﮎ ﮐﻨﺪ.

ﺑﺎﺯﯼ ﻭ ﻟﺬﺕ ﺗﻘﻠﻴﺪ ﮐﺮﺩﻥ ﺍﺯ ﺳﻦ ﺷﺶ ﻣﺎﻫﮕﯽ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﻓﻌﺎﻟﻴﺖ ﺩﺭ ﺗﻤﺎﻡ ﺩﻭﺭﺍﻥ ﻃﻔﻮﻟﻴﺖ ﻳﮑﯽ ﺍﺯ ﻗﻮﯼ‌ﺗﺮﻳﻦ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺎﻗﯽ ﺧﻮﺍﻫﺪ ﻣﺎﻧﺪ. ﮐﻮﺩﮎ ﺷﺶ ﻣﺎﻫﻪ ﺣﺮﮐﺎﺕ ﻭ ﺻﺪﺍﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﺭﺍ ﺗﻘﻠﻴﺪ ﻣﯽ‌ﮐﻨﺪ (ﻣﺜﻼ ﺗﻤﻴﺰ ﮐﺮﺩﻥ ﻣﻴﺰ ﺑﺎ ﻳﮏ ﺩﺳﺘﻤﺎﻝ). ﺍﻭ ﺧﻴﻠﯽ ﭘﻴﺸﺘﺮ ﺍﺯ ﺁﻧﮑﻪ ﺑﺘﻮﺍﻧﺪ ﺻﺤﺒﺖ ﮐﻨﺪ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﻧﺘﻘﺎﻝ ﭘﻴﻐﺎﻡ‌ﻫﺎﻳﺶ ﭘﻴﺪﺍ ﮐﻨﺪ. ﺍﻭ ﻣﺎﻧﻨﺪ ﻣﺴﺎﻓﺮﯼ ﺩﺭ ﮐﺸﻮﺭﯼ ﺧﺎﺭﺟﯽ ﺍﺳﺖ ﮐﻪ ﮔﺮﭼﻪ ﺯﺑﺎﻥ ﺁﻥ ﻣﻤﻠﮑﺖ ﺭﺍ ﻧﻤﯽ‌ﺩﺍﻧﺪ ﻭﻟﯽ ﺑﺎ ﺣﺮﮐﺎﺕ ﺩﺳﺖ ﻭ ﺻﻮﺭﺕ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻔﻬﻤﺎﻧﺪ.

ﺑﯽ‌ﺷﮏ ﺣﺪﻭﺩ ﺳﻦ ﻫﺸﺖ ﻣﺎﻫﮕﯽ، ﺍﻭ ﻗﺎﺩﺭ ﺑﻪ ﺳﻴﻨﻪ ﺧﻴﺰ ﺭﻓﺘﻦ ﺍﺳﺖ. ﺑﺎ ﺍﻳﻦ ﻓﻌﺎﻟﻴﺖ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﻧﻴﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺭﺍ ﺑﻬﺘﺮ ﺑﺮﺭﺳﯽ ﮐﻨﺪ. ﺩﺭ ﻫﺸﺖ ﻳﺎ ﻧﻪ ﻣﺎﻫﮕﯽ، ﻧﻮﺯﺍﺩ ﺑﺰﺭﮒ‌ﺗﺮ ﺍﺯ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺩﺳﺘﺸﻮﻳﯽ ﻳﺎ ﻭﺍﻥ ﮐﻮﭼﮏ ﺍﻭ ﺭﺍ ﺣﻤﺎﻡ ﮐﺮﺩ. ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻭﺍﻥ ﺑﺰﺭﮒ ﺣﻤﺎﻡ ﮐﻨﺪ. ﻭﻟﯽ ﺑﺎﻳﺪ ﺁﺏ ﺧﻴﻠﯽ ﮐﻤﯽ ﺩﺭ ﻭﺍﻥ ﺑﺮﻳﺰﻳﺪ. ﺯﻳﺮﺍ ﺍﮔﺮ ﺍﻭ ﺭﺍ ﺗﻨﻬﺎ ﺑﮕﺬﺍﺭﻳﺪ ﺍﺣﺘﻤﺎﻝ ﺩﺍﺭﺩ ﺧﻔﻪ ﺷﻮﺩ. ﻣﻮﻗﻊ ﺣﻤﺎﻡ ﮐﺮﺩﻥ ﺗﻌﺪﺍﺩﯼ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ ﻫﻢ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﮕﺬﺍﺭﻳﺪ، ﻓﻨﺠﺎﻥ ﭘﻼﺳﺘﻴﮑﯽ، ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﭘﻼﺳﺘﻴﮑﯽ ، ﻟﻴﻒ ﻭ ﻏﻴﺮﻩ. ﺁﺏ ﺑﺎﺯﯼ ﻳﮑﯽ ﺍﺯ ﻣﺤﺒﻮﺏ‌ﺗﺮﻳﻦ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺑﭽﻪ‌ﻫﺎﺳﺖ. ﺷﺎﻳﺪ ﺑﻪ ﺍﻳﻦ ﺟﻬﺖ ﮐﻪ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺯﻧﺪﮔﯽ‌ﺷﺎﻥ ﺩﺍﺧﻞ ﺭﺣﻢ ﻣﺎﺩﺭ، ﺩﺭ ﺩﺍﺧﻞ ﻣﺎﻳﻊ، ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ. ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﻋﻠﺖ ﺍﻳﻦ ﻋﻼﻗﻪ ﻫﺮ ﭼﻪ ﺑﺎﺷﺪ، ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺩﺭ ﺁﺏ ﻳﮑﯽ ﺍﺯ ﺁﺭﺍﻡ‌ﮐﻨﻨﺪﻩ‌ﺗﺮﻳﻦ ﺑﺎﺯﯼ‌ﻫﺎ ﺑﺮﺍﯼ ﺍﻃﻔﺎﻝ ﺍﺳﺖ.

ﺑﻌﺪ ﺍﺯ ﺳﻴﻨﻪ ﺧﻴﺰ ﺭﻓﺘﻦ، ﮐﻮﺩﮎ ﺷﺮﻭﻉ ﺑﻪ ﭼﻬﺎﺭ ﺩﺳﺖ ﻭ ﭘﺎ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﻣﺤﺾ ﺍﻳﻨﮑﻪ ﺷﺮﻭﻉ ﺑﻪ ﺳﻴﻨﻪ ﺧﻴﺰ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺧﺼﻮﺻﺎ ﺍﮔﺮ ﺑﺘﻮﺍﻧﺪ ﺭﻭﯼ ﭘﺎﻫﺎﻳﺶ ﺑﺎﻳﺴﺘﺪ، ﺑﺎﻳﺪ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺑﺮﺍﻳﺶ ﺧﻄﺮﻧﺎﮎ ﺍﺳﺖ ﺍﺯ ﺩﺳﺘﺮﺳﺶ ﺩﻭﺭ ﮐﻨﻴﺪ. ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﮐﻪ ﺳﻮﺯﻥ، ﻣﻴﺦ، ﭘﻮﻧﺰ ﻳﺎ ﻫﺮ ﺷﯽﺀ ﺩﻳﮕﺮﯼ ﮐﻪ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﮐﻮﺩﮎ ﺩﺭ ﺩﻫﺎﻧﺶ ﺑﮕﺬﺍﺭﺩ ﺭﻭﯼ ﺯﻣﻴﻦ ﻧﺒﺎﺷﺪ. ﻫﻤﻴﺸﻪ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﮐﻮﺩﮎ ﻫﺮ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﭘﻴﺪﺍ ﮐﻨﺪ ﺍﻭﻝ ﺑﻪ ﺩﻫﺎﻧﺶ ﻣﯽ‌ﺑﺮﺩ. ﻫﺮ ﭼﻴﺰﯼ ﮐﻪ ﺍﺯ ﺟﺎﻳﯽ ﺁﻭﻳﺰﺍﻥ ﺑﺎﺷﺪ، ﻣﺜﻼ ﻧﺦ ﻳﺎ ﻃﻨﺎﺏ، ﺍﻭ ﺭﺍ ﺑﺮ ﻣﯽ‌ﺍﻧﮕﻴﺰﺩ ﮐﻪ ﺁﻥ ﺭﺍ ﺑﮑﺸﺪ. ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻳﻦ ﮐﻮﭼﻮﻟﻮﯼ ﺧﺴﺘﮕﯽ‌ﻧﺎﭘﺬﻳﺮ ﺳﺮ ﺑﻪ ﻫﻮﺍﻳﺘﺎﻥ ﺷﯽﺀ ﺳﻨﮕﻴﻨﯽ ﺭﺍ ﺭﻭﯼ ﺧﻮﺩﺵ ﻧﻴﻨﺪﺍﺯﺩ.

ﮔﻬﮕﺎﻩ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺧﺎﺭﺝ ﺍﺯ ﭘﺎﺭﮐﺶ ﺑﮕﺬﺍﺭﻳﺪ. ﮔﻮﺷﻪ ﻳﮏ ﺍﺗﺎﻕ ﻳﺎ ﺣﺘﯽ ﺗﻤﺎﻡ ﻳﮏ ﺍﺗﺎﻕ ﺧﺎﻟﯽ ﺍﺯ ﭼﻴﺰﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ، ﺣﻴﻄﻪ ﻋﻤﻞ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﺗﻤﺎﻡ ﻳﮏ ﺭﻭﺯ ﺍﺳﺖ. ﻋﻼﻗﻪ ﻓﺮﺍﻭﺍﻥ ﮐﻮﺩﮎ ﺑﻴﻦ ﺷﺶ ﺗﺎ ﻧﻪ ﻣﺎﻫﮕﯽ ﺑﻪ ﮐﺸﻒ ﭼﻴﺰﻫﺎﯼ ﺟﺪﻳﺪ ﺍﻭ ﺭﺍ ﺑﺮ ﻣﯽ‌ﺍﻧﮕﻴﺰﺩ ﮐﻪ ﮔﺎﻩ ﻧﻴﻢ ﺳﺎﻋﺖ ﭘﺸﺖ ﺳﺮﻫﻢ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﺎﺯﯼ ﮐﻨﺪ. ﻭﺳﺎﻳﻞ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﺟﺎﺭﯼ ﺧﺎﻧﻪ، ﺑﻬﺘﺮﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻫﺎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻫﺴﺘﻨﺪ(ﺩﻳﮓ، ﮐﺎﺳﻪ ﻭ...).

ﺳﻦ ﻣﺸﺨﺼﯽ ﮐﻪ ﺩﺭ ﺁﻥ ﮐﻮﺩﮎ ﺑﺘﻮﺍﻧﺪ ﺷﺴﺘﺶ ﺭﺍ ﺑﺎﺯ ﮐﻨﺪ ﻭ ﺍﺷﻴﺎﺀ ﺭﺍ ﺭﻫﺎ ﮐﻨﺪ ﺍﺯ ﮐﻮﺩﮐﯽ ﺑﻪ ﮐﻮﺩﮎ ﺩﻳﮕﺮ ﺑﺴﻴﺎﺭ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﻫﺮﮔﺎﻩ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺍﺯ ﺭﺷﺪ ﻓﻴﺰﻳﮑﯽ ﮐﻮﺩﮎ ﻓﺮﺍ ﺭﺳﺪ، ﺩﻭﺭﻩ ﻣﺸﮑﻠﯽ ﺑﺮﺍﯼ ﻣﺎﺩﺭ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ. ﮐﻮﺩﮎ ﺑﺎﺯﯼ ﺟﺪﻳﺪﯼ ﺭﺍ ﻓﺮﺍ ﮔﺮﻓﺘﻪ ﮐﻪ «ﺑﺎﺯﯼ ﮐﻦ، ﺭﻫﺎﮐﻦ، ﻭ ﭘﺮﺕ ﮐﻦ» ﻧﺎﻡ ﺩﺍﺭﺩ. ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﺯ ﭘﺎﺭﮎ ﺑﻴﺮﻭﻥ ﻣﯽ‌ﭘﺮﺩ ﻭ ﺍﺯ ﺑﺎﻻﯼ ﺻﻨﺪﻟﻴﺶ ﺧﻮﺩ ﺭﺍ ﺭﻭﯼ ﺯﻣﻴﻦ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ. ﺍﻳﻦ ﮐﺎﺭ ﺍﻭ ﺑﺮﺍﯼ ﻧﺎﺭﺍﺣﺖ ﮐﺮﺩﻥ ﺷﻤﺎ ﻧﻴﺴﺖ. ﺍﻳﻦ ﺑﺎﺯﯼ ﺟﺰﻳﯽ ﺍﺳﺖ ﺍﺯ ﻣﺠﻤﻮﻉ ﺗﺤﻘﻴﻘﺎﺕ ﺍﻭ ﺩﺭﺑﺎﺭﻩ ﺟﻬﺎﻥ ﻭ ﺍﺟﺰﺍﯼ ﺗﺸﮑﻴﻞ‌ﺩﻫﻨﺪﻩ ﺁﻥ. ﺍﻳﻦ ﮐﺎﺭ ﺩﺭ ﻭﺍﻗﻊ ﮐﺸﻒ ﺭﺍﺑﻄﻪ ﺑﻴﻦ ﻗﺎﺑﻠﻴﺖ ﺟﺪﻳﺪ ﺍﺳﺖ ﻭ ﻗﺎﻧﻮﻥ ﺟﺎﺫﺑﻪ ﺯﻣﻴﻦ، ﻫﺮﭼﻨﺪ ﮐﻮﺩﮎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺁﻥ ﺭﺍ ﺑﺎ ﻓﺮﻣﻮﻝ‌ﻫﺎﯼ ﺭﻳﺎﺿﯽ ﺑﻴﺎﻥ ﮐﻨﺪ.

ﻣﻮﺿﻮﻉ ﺍﺻﻠﯽ ﺑﺎﺯﯼ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﮔﺬﺍﺷﺘﻦ ﻭ «ﺑﺮﺩﺍﺷﺘﻦ» ﺍﺳﺖ. ﺍﺷﻴﺎﯼ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﺩﺭ ﺧﺎﻧﻪ ﺟﺎﻟﺐ‌ﺗﺮﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﻫﺴﺘﻨﺪ. ﺑﺎﺯﯼ ﺑﺎ ﺑﻄﺮﯼ ﺑﺰﺭﮒ ﭘﻼﺳﺘﻴﮑﯽ ﺑﺎ ﺩﻫﺎﻧﻪ ﮔﺸﺎﺩ ﻳﺎ ﻭﺳﺎﻳﻞ ﻣﺘﺪﺍﻭﻝ ﻣﺨﺘﻠﻒ ﺑﺮﺍﯼ ﺍﻭ ﺑﺴﻴﺎﺭ ﻟﺬﺕ‌ﺑﺨﺶ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺳﺒﺪ ﮐﺎﻏﺬﻫﺎﯼ ﺑﺎﻃﻠﻪ، ﻣﻤﻠﻮ ﺍﺯ ﮐﺎﻏﺬ ﻭ ﭘﺎﮐﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﮐﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻣﺪﺗﻬﺎ ﻣﻔﺘﻮﻥ ﺧﻮﺩ ﻣﯽ‌ﮐﻨﺪ.

 

ﻧﻪ ﺗﺎ ﺩﻭﺍﺯﺩﻩ ﻣﺎﻫﮕﯽ
ﺑﺮﺧﯽ ﺍﺯ ﺑﭽﻪ‌ﻫﺎ ﺩﺭﺳﻦ ﻧﻪ ﻣﺎﻫﮕﯽ ﺭﺍﻩ ﺍﻓﺘﺎﺩﻩ‌ﺍﻧﺪ. ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﺩﺭ ﻳﮏ ﺳﺎﻟﮕﯽ ﻭ ﺑﺮﺧﯽ ﺩﻳﮕﺮ ﺩﺭ ﺳﻴﺰﺩﻩ، ﭼﻬﺎﺭﺩﻩ ﻣﺎﻫﮕﯽ ﺷﺮﻭﻉ ﺑﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﻨﺪ. ﻭﻟﯽ ﺍﻳﻦ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭ ﭼﻪ ﺳﻨﯽ ﺷﺮﻭﻉ ﺑﻪ ﺭﺍﻩ ﺭﻓﺘﻦ ﻣﯽ‌ﮐﻨﺪ ﺍﻫﻤﻴﺖ ﭼﻨﺪﺍﻧﯽ ﻧﺪﺍﺭﺩ. ﺳﻪ ﻣﺎﻩ ﺁﺧﺮ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ ﺍﻭ، ﺩﻭﺭﻩ ﮔﺬﺍﺭ ﺍﺯ ﺣﺎﻟﺖ ﺍﻓﻘﯽ ﺑﻪ ﺣﺎﻟﺖ ﻋﻤﻮﺩﯼ ﺍﺳﺖ. ﺍﻭ ﺩﻳﮕﺮ ﺑﻪ ﻫﻨﮕﺎﻡ ﻋﻮﺽ ﮐﺮﺩﻥ ﮐﻬﻨﻪ‌ﺍﺵ، ﺑﻪ ﺁﺭﺍﻣﯽ ﺩﺭﺍﺯ ﻧﻤﯽ‌ﮐﺸﺪ ﮐﻪ ﺷﻤﺎ ﮐﺎﺭﺗﺎﻥ ﺭﺍ ﺑﮑﻨﻴﺪ. ﺑﻠﮑﻪ ﺑﺮﻋﮑﺲ ﺩﺭﺳﺖ ﻣﺜﻞ ﻳﮏ «ﮐﺮﻡ ﺧﺎﮐﯽ» ﺷﺪﻩ ﺍﺳﺖ. ﻭ ﺍﺣﺘﻤﺎﻻ ﺑﻪ ﻫﻨﮕﺎﻡ ﻟﺒﺎﺱ ﭘﻮﺷﻴﺪﻥ ﺑﺎ ﺷﻤﺎ ﻫﻤﮑﺎﺭﯼ ﻣﯽ‌ﮐﻨﺪ. ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺳﻪ ﻣﺎﻫﻪ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﺷﺮﻭﻉ ﺑﻪ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﭘﻴﺸﺮﻓﺘﻪ‌ﺍﯼ ﭼﻮﻥ «ﺩﺳﺖ ﺯﺩﻥ» ﻭ ﺩﻳﮕﺮ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺗﻘﻠﻴﺪﯼ ﻣﺨﺼﻮﺹ ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﮐﻮﺩﮐﯽ ﻣﯽ‌ﮐﻨﺪ. ﺍﮔﺮ ﭼﻪ ﻧﻮﺯﺍﺩ ﻫﻨﻮﺯ ﻗﺎﺩﺭ ﺑﻪ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﻧﻴﺴﺖ، ﺍﻣﺎ ﻗﺴﻤﺖ ﺍﻋﻈﻢ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺑﻪ ﺍﻭ ﻣﯽ‌ﮔﻮﻳﻴﺪ ﻣﯽ‌ﻓﻬﻤﺪ. ﺩﺳﺘﻮﺭﻫﺎﯼ ﺳﺎﺩﻩ ﺭﺍ ﺩﺭﮎ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺗﻌﺪﺍﺩﯼ ﮐﻠﻤﺎﺕ ﺭﺍ ﮐﻪ ﻧﻘﺶ ﻭﺍﮊﻩ‌ﻫﺎﯼ ﮐﻠﻴﺪﯼ ﺩﺭ ﺑﺎﺯﯼ‌ﻫﺎ ﻭ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﺩﺍﺭﻧﺪ، ﻣﺜﻞ ﺣﻤﺎﻡ ﻳﺎ ﻏﺬﺍ ﻣﯽ‌ﻓﻬﻤﺪ.

ﺍﮐﻨﻮﻥ ﺑﺎﻳﺪ ﮐﻤﮏ ﮐﻨﻴﺪ ﻧﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺍﻭ ﺭﺍ ﺍﺣﺎﻃﻪ ﮐﺮﺩﻩ ﺍﺳﺖ ﻳﺎﺩ ﺑﮕﻴﺮﺩ. ﺍﻳﻦ ﮐﺎﺭ ﺳﺎﺩﻩ‌ﺍﯼ ﺍﺳﺖ. ﻫﺮ ﺩﻓﻌﻪ ﻓﻘﻂ ﻳﮏ ﮐﻠﻤﻪ ﺑﻪ ﮐﺎﺭ ﺑﺮﻳﺪ. ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺩﻧﻴﺎﯼ ﮐﻮﺩﮎ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ ﺷﻨﺎﺳﺎﻳﯽ ﮐﺮﺩﻩ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺩﻫﻴﺪ. ﻭﻗﺘﯽ ﺍﻭ ﺭﺍ ﺣﻤﺎﻡ ﻣﯽ‌ﮐﻨﻴﺪ، ﺩﺳﺘﺘﺎﻥ ﺭﺍ ﺩﺭ ﺁﺏ ﮐﺮﺩﻩ ﻭ ﺗﮑﺎﻥ ﺩﻫﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ «ﺁﺏ».  ﻭﻗﺘﯽ ﺑﻪ ﺍﻭ ﻳﮏ ﻗﺎﺷﻖ ﮐﻤﭙﻮﺕ ﺳﻴﺐ ﻣﯽ‌ﺩﻫﻴﺪ ﺑﮕﻮﻳﻴﺪ «ﮐﻤﭙﻮﺕ». ﺍﮔﺮ ﺩﺭ ﻣﺎﺷﻴﻦ ﻫﺴﺘﻴﺪ ﻭ ﺍﺯ ﮐﻨﺎﺭ ﻳﮏ ﮐﺎﻣﻴﻮﻥ ﺭﺩ  ﻣﯽ‌ﺷﻮﻳﺪ ﮐﺎﻣﻴﻮﻥ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺑﺪﻫﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ «ﮐﺎﻣﻴﻮﻥ».  ﺍﻳﻦ ﺑﺎﺯﯼ ﺍﺳﻢ‌ﮔﺬﺍﺭﯼ ﺭﺍ ﺩﺭ ﻫﻤﻪ ﺟﺎ ﻭ ﻫﻤﻴﺸﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮑﻨﻴﺪ.

ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻓﻘﻂ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺑﻪ ﺍﻭ ﻣﯽ‌ﮔﻮﻳﻴﺪ ﺩﺭ ﺫﻫﻨﺶ ﺛﺒﺖ ﮐﻨﺪ. ﺩﺭ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﮐﻪ ﺷﺮﻭﻉ ﺑﻪ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﮐﻨﺪ، ﺧﻴﻠﯽ ﺑﻌﺪﺗﺮ، ﺍﻭ ﺍﻳﻦ ﮐﻠﻤﺎﺕ ﺭﺍ ﺗﮑﺮﺍﺭ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺍﻳﻦ ﺑﺎﺯﯼ ﻧﺎﻡ‌ﮔﺬﺍﺭﯼ ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﮐﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺭﺷﺪ ﺯﺑﺎﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.

ﺩﺭ ﺍﻳﻦ ﺩﻭ ﻳﺎ ﺳﻪ ﻣﺎﻩ ﺁﺧﺮ ﻧﺨﺴﺘﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺑﺎ ﮐﺘﺎﺏ ﺁﺷﻨﺎ ﮐﻨﻴﺪ. ﺷﺎﻳﺪ ﺗﻌﺠﺐ ﮐﻨﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ «ﮐﺘﺎﺏ؟... ﻣﺴﺨﺮﻩ ﺍﺳﺖ. ﺍﻭﻥ ﮐﺘﺎﺑﻮ ﻣﯽ‌ﺫﺍﺭﻩ ﺗﻮ ﺩﻫﻨﺶ»،  ﻭﻟﯽ ﻳﺎﺩﺗﺎﻥ ﺑﺎﺷﺪ ﮐﻪ ﺍﻭ ﺍﺯ ﺍﻳﻦ ﺭﺍﻩ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻔﻬﻤﺪ ﺍﺟﺴﺎﻡ ﭼﻪ ﻫﺴﺘﻨﺪ، ﮐﺘﺎﺏ ﻫﻢ ﺍﺯ ﺁﻥ ﺟﻤﻠﻪ ﺍﺳﺖ. ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻭ ﻣﯽ‌ﮔﺬﺍﺭﻳﺪ ﺑﺎﻳﺪ ﻳﺎ ﭘﺎﺭﭼﻪ‌ﺍﯼ ﺑﺎﺷﻨﺪ ﻳﺎ ﺍﺯ ﺟﻨﺲ ﻣﻘﻮﺍﯼ ﺿﺨﻴﻢ. ﻻﺯﻡ ﻧﻴﺴﺖ ﮐﻪ ﮐﺘﺎﺏ‌ﻫﺎ ﺩﺍﺳﺘﺎﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ، ﺯﻳﺮﺍ ﺍﻭ ﺑﺮﺍﯼ ﻓﻬﻤﻴﺪﻥ ﺩﺍﺳﺘﺎﻥ ﻫﻨﻮﺯ ﭘﺨﺘﻪ ﻧﻴﺴﺖ، ﺑﻠﮑﻪ ﻓﻘﻂ ﺗﺼﺎﻭﻳﺮ ﺍﺷﻴﺎﯼ ﺁﺷﻨﺎ (ﺑﻪ ﻫﻤﺮﺍﻩ ﻓﻘﻂ ﻳﮏ ﮐﻠﻤﻪ) ﮐﺎﻓﯽ ﺍﺳﺖ. ﺍﻳﻦ ﺍﻭﻟﻴﻦ ﮐﺘﺎﺏ‌ﻫﺎ ﺩﺭ ﻭﺍﻗﻊ ﺟﺰﻳﯽ ﺍﺯ ﺑﺎﺯﯼ ﻧﺎﻡ‌ﮔﺬﺍﺭﯼ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.
ﺗﺼﺎﻭﻳﺮ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺩﻫﻴﺪ ﻭ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﮐﻠﻤﻪ‌ﺍﯼ ﺭﺍ ﮐﻪ ﺑﺎ ﺁﻥ ﻣﻨﺎﺳﺒﺖ ﺩﺍﺭﺩ ﺍﺩﺍ ﮐﻨﻴﺪ. ﺑﻌﺪ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺧﻮﺩﺵ ﮐﺘﺎﺏ ﺭﺍ ﺑﮕﻴﺮﺩ، ﮐﺘﺎﺏ ﺭﺍ ﻧﺎﺯ ﻣﯽ‌ﮐﻨﺪ، ﺍﻳﻦ ﻃﺮﻑ ﻭ ﺁﻥ ﻃﺮﻑ ﻣﯽ‌ﺯﻧﺪ ﻭ ﺑﻌﺪ ﻫﻢ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﺩﺭ ﺩﻫﺎﻧﺶ! ﺑﻌﺪﻫﺎ، ﺗﺼﺎﻭﻳﺮ ﺭﺍ ﻧﮕﺎﻩ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﻭ ﮐﻠﻤﺎﺕ ﻏﻴﺮ ﻗﺎﺑﻞ ﻓﻬﻤﯽ ﺑﻴﺎﻥ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺮﺍﯼ ﺍﻭ ﻳﻌﻨﯽ «ﺧﻮﺍﻧﺪﻥ». ﻭﻟﯽ ﺍﻳﻦ ﮐﺎﺭ ﺍﻭ ﺭﺍ ﺗﺤﻘﻴﺮ ﻧﮑﻨﻴﺪ. ﺑﺎ ﮔﺬﺍﺷﺘﻦ ﮐﺘﺎﺏ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺳﻦ، ﭘﺎﻳﻪ‌ﻫﺎﯼ ﻋﻼﻗﻪ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺭﺍ ﺩﺭ ﺍﻭ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻭﺭﻳﺪ.

ﺩﺭ ﺩﺭﺱ ﺁﻳﻨﺪﻩ ﭼﻬﺎﺭ ﺗﻮﺻﻴﻪ ﺑﺮﺍﯼ ﺍﻳﻦ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ ﺧﻮﺍﻫﻴﻢ ﺁﻭﺭﺩ ﻛﻪ ﺑﻪ ﺷﻤﺎ ﻛﻤﻚ ﻣﯽ‌ﮐﻨﺪ ﺍﺯ ﺑﺮﺧﯽ ﺍﺯ ﺍﺷﺘﺒﺎﻫﺎﺕ ﺑﺮ ﺣﺬﺭ ﺑﺎﺷﻴﺪ.
