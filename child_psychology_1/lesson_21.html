ﺁﺏ‌ﺑﺎﺯﯼ ﻫﻤﭽﻨﺎﻥ ﺍﺯ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺁﺏ‌ﺑﺎﺯﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﻭﺍﻥ ﺣﻤﺎﻡ ﻭ ﻳﺎ ﻇﺮﻑ‌ﺷﻮﻳﯽ ﻭ ﺩﺳﺘﺸﻮﻳﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ. ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻋﻼﻗﻪ ﻣﻔﺮﻃﯽ ﺑﻪ ﺧﻴﺲ ﮐﺮﺩﻥ ﺍﺑﺮ ﺩﺭ ﺁﺏ ﻭ ﭼﻼﻧﺪﻥ ﺁﻥ ﻭ ﭘﺮﻭ ﺧﺎﻟﯽ ﮐﺮﺩﻥ ﻟﻴﻮﺍﻥ ﻭ ﻳﺎ ﻫﺮ ﻇﺮﻑ ﭘﻼﺳﺘﻴﮑﯽ ﺩﻳﮕﺮﯼ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻭ ﻣﯽ­‌ﮔﺬﺍﺭﻳﺪ.

ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻇﺮﻑ‌ﻫﺎﯼ ﺁﺷﭙﺰﺧﺎﻧﻪ ﻳﺎ ﻟﺒﺎﺱ ﻋﺮﻭﺳﮏ ﻭ ﻳﺎ ﺩﺳﺘﻤﺎﻟﯽ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﺪ ﮐﻪ ﺁﻧﻬﺎ ﺭﺍ «ﺑﺸﻮﻳﺪ». ﻥِﯼِ ﭘﻼﺳﺘﻴﮑﯽ ﻫﻢ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﮕﺬﺍﺭﻳﺪ، ﺑﺎ ﻓﻮﺕ ﮐﺮﺩﻥ ﻫﻮﺍ ﺩﺭ ﺁﺏ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺳﻄﺢ ﺧﻮﺩﺵ، ﺑﻪ ﺍﻭﻟﻴﻦ ﺁﺯﻣﺎﻳﺶ ﻓﻴﺰﻳﮑﯽ ﺩﺳﺖ ﺑﺰﻧﺪ. ﻋﻼﻗﻪ ﺑﻪ ﺧﻤﻴﺮﺑﺎﺯﯼ ﻧﻴﺰ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺧﻴﻠﯽ ﺯﻳﺎﺩ ﺍﺳﺖ. ﺧﻤﻴﺮ ﺭﺍ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺑﻴﺮﻭﻥ ﺑﺨﺮﻳﺪ ﻭ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺧﻮﺩﺗﺎﻥ ﺩﺭﺳﺖ ﮐﻨﻴﺪ. ﻃﺮﺯ ﺩﺭﺳﺖ ﮐﺮﺩﻥ ﺁﻥ ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺍﺳﺖ: ﺑﻪ ﻧﺴﺒﺖ 2 ﺑﻪ 1 ﺁﺭﺩ ﻭ ﻧﻤﮏ ﺭﺍ ﻣﺨﻠﻮﻁ ﮐﻨﻴﺪ، ﺳﭙﺲ ﺁﺏ ﻻﺯﻡ ﺭﺍ ﺑﻪ ﺁﻥ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ ﺗﺎ ﺧﻤﻴﺮﯼ ﺷﺒﻴﻪ ﺑﻪ ﺧﻤﻴﺮ ﻧﺎﻥ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﻳﺪ. ﻳﮑﯽ ﺩﻭ ﻗﻄﺮﻩ ﺭﻭﻏﻦ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺑﺘﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﻣﺪﺗﯽ ﻃﻮﻻﻧﯽ ﻧﮕﻬﺪﺍﺭﻳﺪ. ﺍﮔﺮ ﺍﻳﻦ ﺧﻤﻴﺮ ﺭﺍ ﺩﺭ ﮐﻴﺴﻪ‌ﺍﯼ ﻧﺎﻳﻠﻮﻧﯽ ﺑﮕﺬﺍﺭﻳﺪ، ﺗﻘﺮﻳﺒﺎ ﻳﮏ ﻣﺎﻩ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ. ﮐﻮﺩﮐﺎﻥ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﻛﻪ ﺍﺯ ﺧﻤﻴﺮ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ، ﻫﻴﭻ ﭼﻴﺰ ﺧﺎﺻﯽ ﺩﺭﺳﺖ ﻧﻤﯽ­ﮐﻨﻨﺪ. ﺍﻳﻦ ﺧﻮﺩ ﺧﻤﻴﺮ ﺍﺳﺖ ﮐﻪ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﺁﻧﻬﺎﺳﺖ. ﮐﻮﺩﮎ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺁﻥ ﺭﺍ ﻟﻪ ﮐﻨﺪ، ﻗﺎﻟﺐ ﺑﮕﻴﺮﺩ، ﻓﺸﺎﺭ ﺑﺪﻫﺪ ﻭ ﺩﺳﺖ­ﻣﺎﻟﯽ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺁﻥ ﺷﮑﻞ ﺩﻫﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺧﻴﻠﯽ ﺑﻪ ﺁﻧﻬﺎ ﺁﺭﺍﻣﺶ ﻣﯽ‌ﺩﻫﺪ. ﺑﻬﺘﺮﻳﻦ ﻭﺳﻴﻠﻪ ﮐﺎﺭ ﺁﻧﻬﺎ ﺩﺳﺖ‌ﻫﺎﻳﺸﺎﻥ ﺍﺳﺖ. ﻭﻟﯽ ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﻣﺪﺗﯽ ﺍﺯ ﺩﺳﺖ‌ﻫﺎﻳﺸﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩﻧﺪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ‌ ﺁﻧﻬﺎ ﻗﺎﻟﺐ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﺑﺎﺯﯼ ﭘﻴﺸﻨﻬﺎﺩ ﮐﺮﺩ.

ﻭﺳﻴﻠﻪ ﺩﻳﮕﺮﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻭ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺩﺭ ﺗﻤﺎﻡ ﺩﻭﺭﺍﻥ ﻗﺒﻞ ﺍﺯ ﻣﺪﺭﺳﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ، ﻳﮏ ﻭﺍﻳﺖ ﺑﻮﺭﺩ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺑﺰﺭﮒ ﺑﺎ ﻣﺎﮊﻳﻚ ﻣﺨﺼﻮﺹ ﺁﻥ ﺍﺳﺖ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﺗﺎﺑﻠﻮ ﺭﺍ ﺑﻪ ﺩﻳﻮﺍﺭ ﺍﺗﺎﻕ ﻧﺼﺐ ﮐﻨﻴﺪ. ﺍﻳﻦ ﻳﮑﯽ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻱ ﭘﺮﻭﺭﺷﯽ ﺍﺳﺖ ﮐﻪ ﺍﻣﮑﺎﻧﺎﺕ ﻣﺘﻌﺪﺩﯼ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭﺑﺮ ﺩﺍﺭﺩ.

ﺧﻂ‌ﺧﻄﯽ ﮐﺮﺩﻥ ﭘﻴﺶ ﺩﺭ ﺁﻣﺪ ﻧﻘﺎﺷﯽ ﻭ ﻧﻮﺷﺘﻦ ﺍﺳﺖ. ﮐﻮﺩﮐﯽ ﮐﻪ ﺍﻣﮑﺎﻧﺎﺕ ﻻﺯﻡ ﺑﺮﺍﯼ ﺧﻂ‌ﺧﻄﯽ ﮐﺮﺩﻥ ﺑﺮﺍﻳﺶ ﻓﺮﺍﻫﻢ ﻧﺸﺪﻩ ﺍﺳﺖ، ﻧﺴﺒﺖ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺩﻳﺮﺗﺮ ﻧﻘﺎﺷﯽ‌ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺩﻳﺮﺗﺮ ﻣﯽ‌ﻧﻮﻳﺴﺪ. ﻫﻴﭻ ﮐﻮﺩﮐﯽ ﺑﺎ ﻏﺮﻳﺰﻩ ﻫﻢ‌ﺁﻫﻨﮕﯽ ﺍﻧﮕﺸﺖ ﺷﺼﺖ ﺑﺎ ﺩﻳﮕﺮ ﺍﻧﮕﺸﺘﺎﻥ ﺑﺮﺍﯼ ﺑﻪ ﺩﺳﺖ ﮔﺮﻓﺘﻦ ﻣﺪﺍﺩ ﻭ ﻗﻠﻢ ﺑﻪ ﺩﻧﻴﺎ ﻧﻤﯽ‌ﺁﻳﺪ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺗﻤﺮﻳﻦ ﮐﻨﺪ ﺗﺎ ﻋﻀﻼﺕ ﺍﻧﮕﺸﺘﺎﻧﺶ ﺁﻣﺎﺩﮔﯽ ﻻﺯﻡ ﺭﺍ ﺟﻬﺖ ﻧﻘﺎﺷﯽ ﮐﺮﺩﻥ ﻭ ﻧﻮﺷﺘﻦ ﭘﻴﺪﺍ ﮐﻨﻨﺪ. ﺍﺯ ﻃﺮﻳﻖ ﺧﻂ‌ﺧﻄﯽ ﮐﺮﺩﻥ ﺍﺳﺖ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻓﺮﺍ ﻣﯽ­‌ﮔﻴﺮﺩ.
ﺍﮐﻨﻮﻥ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺪﺍﺩ ﺷﻤﻌﯽ ﻧﻴﺰ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻭ ﺑﮕﺬﺍﺭﻳﺪ.

ﮐﻮﺩﮎ ﺩﺭ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﻫﻢ ﺑﻪ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺁﺭﺍﻡ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﺩ ﻭ ﻫﻢ ﺑﻪ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺑﺎﺗﺤﺮﮎ. ﺑﻪ ﺗﺪﺭﻳﺞ ﮐﻪ ﺯﺑﺎﻧﺶ ﭘﻴﺸﺮﻓﺖ‌ﻫﺎﯼ ﺑﺰﺭﮔﯽ ﻣﯽ‌ﮐﻨﺪ، ﺗﻮﺟﻪ ﺯﻳﺎﺩﯼ ﺑﻪ ﮐﻠﻤﺎﺕ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﺑﺎﺯﯼ ﺑﺎ ﮐﻠﻤﺎﺕ ﺭﺍ ﺧﻴﻠﯽ ﺩﻭﺳﺖ ﺩﺍﺭﺩ، ﺻﺪﺍﻫﺎ ﺭﺍ ﺗﻘﻠﻴﺪ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﮐﻮﺩﮐﺎﻧﻪ ﺭﺍ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﮐﻨﺪ. ﻭﻗﺘﯽ ﮐﻪ ﺑﺮﺍﻳﺶ ﮐﺘﺎﺏ ﻣﯽ‌ﺧﻮﺍﻧﻴﺪ ﻋﺎﺷﻖ ﭼﻴﺰﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﺷﻮﻧﺪ، ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﻳﯽ ﻣﺜﻞ «ﺷﻨﮕﻮﻝ ﻭ ﻣﻨﮕﻮﻝ ﻭ ﻫﺴﺘﻪ ﺍﻧﮕﻮﺭ» ﻭ «ﺳﻔﻴﺪ ﺑﺮﻓﯽ» ﺧﻮﺷﺶ ﻣﯽ‌ﺁﻳﺪ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﺮﺍﻳﺶ ﮐﺘﺎﺏ ﻣﯽ‌ﺧﻮﺍﻧﻴﺪ ﺍﻭ ﺑﺎ ﺧﻮﺷﺤﺎﻟﯽ ﺟﺮﻳﺎﻥ ﺑﻌﺪﯼ­‌ﺍﯼ ﺭﺍ ﮐﻪ ﭘﻴﺶ ﺧﻮﺍﻫﺪ ﺁﻣﺪ ﺑﻴﺎﻥ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﻭﺍﯼ ﺑﻪ‌ﺣﺎﻟﺘﺎﻥ ﺍﮔﺮ ﺩﺭ ﺗﻌﺮﻳﻒ ﮐﺮﺩﻥ ﻗﺼﻪ ﻳﺎ ﺩﺍﺳﺘﺎﻧﯽ ﮐﻪ ﺍﻭ ﻣﯽ‌ﺷﻨﺎﺳﺪ ﺗﻐﻴﻴﺮﯼ ﺑﺪﻫﻴﺪ ﻳﺎ ﻗﺴﻤﺘﯽ ﺍﺯ ﺁﻥ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﮐﻨﻴﺪ. ﮐﻮﺩﮎ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﺗﺠﺮﺑﻪ‌ﻫﺎﯼ ﺧﻮﺩ ﺍﻭﺳﺖ ﺑﺴﻴﺎﺭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ.

ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺁﺭﺍﻡ ﺩﺭ ﺍﻳﻦ ﺳﻦ «ﺑﺎﺯﯼ ﺳﮑﻮﺕ» ﻳﺎ «ﻫﻴﺲ ﺑﺎﺯﯼ» ﺍﺳﺖ. ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﮕﻮﻳﻴﺪ: ﻣﻲ‌‌ﺧﻮﺍﻡ ﻳﻪ ﺑﺎﺯﯼ ﺟﺪﻳﺪ ﺑﻪ ﺍﺳﻢ«ﻫﻴﺲ ﺑﺎﺯﯼ» ﺑﻬﺖ ﻳﺎﺩ ﺑﺪﻡ. ﻫﺮ ﺩﻭ ﺗﺎﯼ ﻣﺎ ﺑﺎﻳﺪ ﺳﺎﮐﺖ ﺑﺎﺷﻴﻢ ﻭ ﻫﻴﭻ ﺻﺪﺍﻳﯽ ﺩﺭﻧﻴﺎﺭﻳﻢ ﻭ ﻓﻘﻂ ﮔﻮﺵ ﮐﻨﻴﻢ. ﺣﺎﻻ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﻢ، ۱، ۲، ۳، ﻫﻴﺲ! ﺧﻮﺏ ﮔﻮﺵ ﮐﻦ ﻭ ﺑﮕﻮ ﮐﻪ ﭼﯽ ﻣﯽ‌ﺷﻨﻮﯼ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﻤﺎﻡ ﺻﺪﺍﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﻣﺤﻞ ﺑﺎﺯﯼ ﻣﯽ‌ﺷﻨﻮﺩ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﺎﺯﮔﻮ ﮐﻨﺪ: ﺻﺪﺍﯼ ﻣﺎﺷﻴﻦ‌ﻫﺎ ﺩﺭ ﺧﻴﺎﺑﺎﻥ، ﺁﻭﺍﺯ ﭘﺮﻧﺪﮔﺎﻥ ﻭ ﻳﺎﺻﺪﺍﯼ ﺭﺍﺩﻳﻮﯼ ﻫﻤﺴﺎﻳﻪ ﻭ ﻏﻴﺮﻩ.

ﻧﻮﻉ ﺩﻳﮕﺮﯼ ﺍﺯ «ﺑﺎﺯﯼ ﺳﮑﻮﺕ» ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﮐﻮﺩﮎ ﺑﺨﻮﺍﻫﻴﺪ ﭼﺸﻢ‌ﻫﺎﻳﺶ ﺭﺍ ﺑﺒﻨﺪﺩ ﻭ ﺣﺪﺱ ﺑﺰﻧﺪ ﮐﻪ ﺻﺪﺍﻳﯽ ﺷﻤﺎ ﺩﺭﻣﯽ‌ﺁﻭﺭﻳﺪ ﺻﺪﺍﯼ ﭼﻪ ﭼﻴﺰﯼ ﺍﺳﺖ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﻗﺎﺷﻖ ﺭﻭﯼ ﺷﻴﺸﻪ ﺑﺰﻧﻴﺪ، ﺑﺎ ﻧﺎﺧﻦ ﺭﻭﯼ ﻣﻴﺰ ﺑﮑﺸﻴﺪ ﻭ ﻳﺎ ﻫﺮ ﺻﺪﺍﯼ ﺟﺎﻟﺒﯽ ﮐﻪ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﻣﯽ‌ﺁﻳﺪ ﺩﺭﺑﻴﺎﻭﺭﻳﺪ. ﻳﮏ ﻧﻮﻉ ﺩﻳﮕﺮ ﺍﺯ ﺍﻳﻦ ﺑﺎﺯﯼ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺣﺎﻻ ﻣﻦ ﭼﻴﺰﯼ ﺭﺍ ﺧﻴﻠﯽ ﺧﻴﻠﯽ ﺁﻫﺴﺘﻪ ﺩﺭ ﮔﻮﺷﺖ ﺯﻣﺰﻣﻪ ﻣﯽ‌ﮐﻨﻢ، ﺑﺒﻴﻨﻢ ﺗﻮ ﻣﯽ‌ﻓﻬﻤﯽ ﭼﻪ ﻣﯽ‌ﮔﻢ؟

ﻳﮑﯽ ﺍﺯ ﺗﻐﻴﻴﺮﺍﺕ ﻣﻬﻢ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﺍﻥ ﺗﻐﻴﻴﺮﺍﺕ ﺩﺭ ﻋﺎﺩﺍﺕ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻃﻔﻞ ﺍﺳﺖ. ﺍﮔﺮ ﺷﻤﺎ ﺭﻭﯼ ﮔﺮﺳﻨﮕﯽ ﮐﻮﺩﮎ ﺑﻪ ﻋﻨﻮﺍﻥ ﻋﺎﻣﻠﯽ ﺑﺮﺍﯼ ﺗﺸﻮﻳﻖ ﺍﻭ ﺑﻪ ﺧﻮﺭﺩﻥ ﺣﺴﺎﺏ ﮐﻨﻴﺪ، ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻗﺎﻋﺪﺗﺎ ﻧﺒﺎﻳﺪ ﻫﻴﭻ‌ﮔﺎﻩ ﻣﺴﺌﻠﻪ‌ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﻭ ﺭﺍ ﻣﺠﺒﻮﺭ ﺑﻪ ﺧﻮﺭﺩﻥ ﻧﮑﻨﻴﺪ. ﺑﻪ ﺍﻭ ﻧﮕﻮﻳﻴﺪ ﮐﻪ ﺯﻭﺩﺗﺮ ﻏﺬﺍﻳﺶ ﺭﺍ ﺑﺨﻮﺭﺩ. ﻣﺴﻠﻤﺎ ﺍﻭ ﮐﻤﺘﺮ ﻭ ﮐﻤﺘﺮ ﺍﺯ ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺷﻤﺎ ﺩﻟﺘﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺧﻮﺍﻫﺪ ﺧﻮﺭﺩ، ﻭﻟﯽ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺟﻬﺖ ﺟﺒﺮﺍﻥ ﺍﻧﺮﮊﯼ ﻋﻈﻴﻤﯽ ﮐﻪ ﺭﻭﺯﺍﻧﻪ ﻣﺼﺮﻑ ﻣﯽ‌ﮐﻨﺪ ﻏﺬﺍ ﻣﯽ‌ﺧﻮﺭﺩ. ﻏﺬﺍﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺩﻫﻴﺪ ﺑﺎﻳﺪ ﺩﺍﺭﺍﯼ ﻣﻮﺍﺩ ﻻﺯﻡ ﺑﺎﺷﺪ ﻭ ﺍﻭ ﺭﺍ ﺑﺮﺍﯼ ﺧﻮﺭﺩﻧﺶ ﺭﺍﺣﺖ ﺑﮕﺬﺍﺭﻳﺪ. ﺍﺻﺮﺍﺭﯼ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺣﺘﻤﺎ ﺳﺒﺰﻳﺠﺎﺕ ﻭ ﮔﻮﺷﺘﺶ ﺭﺍ ﺍﻭﻝ ﺗﻤﺎﻡ ﮐﻨﺪ ﺗﺎ ﺑﺘﻮﺍﻧﺪ ﺩﺳﺮﺵ ﺭﺍ ﺑﺨﻮﺭﺩ.

ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ، ﻣﺴﺎﺋﻠﯽ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻳﺪ ﮐﻪ ﻧﺎﺷﯽ ﺍﺯ ﺍﻧﺘﻈﺎﺭ ﻭﺍﻟﺪﻳﻦ ﺍﺳﺖ ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﻪ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻣﺜﻞ ﻳﮏ ﺁﺩﻡ ﺑﺎﻟﻎ ﺳﺮ ﻏﺬﺍﻳﺶ ﺑﻨﺸﻴﻨﺪ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﮐﻪ ﺷﺎﻫﺪ ﺭﺷﺪ ﺳﺮﻳﻊ ﺯﺑﺎﻥ ﺩﺭ ﮐﻮﺩﮐﺸﺎﻥ ﻫﺴﺘﻨﺪ، ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻮﺩﮎ ﺍﻳﻦ ﺭﺷﺪ ﺭﺍ ﺩﺭ ﺗﻤﺎﻡ ﺯﻣﻴﻨﻪ‌ﻫﺎ ﮐﺮﺩﻩ ﺍﺳﺖ. ﻭﻟﯽ ﻫﻨﻮﺯ ﺧﻴﻠﯽ ﺯﻭﺩ ﺍﺳﺖ ﮐﻪ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻮﺩﮎ «ﺩﺭﺳﺖ ﺳﺮ ﺳﻔﺮﻩ ﺑﻨﺸﻴﻨﺪ».


ﻗﺒﻼ ﮔﻔﺘﻴﻢ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺗﻤﺎﻡ ﭼﻴﺰ‌ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﺷﻮﺩ ﺩﻭﺳﺖ ﺩﺍﺭﺩ. ﭘﺲ ﺍﺯ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺮﺍﯼ ﻋﺎﺩﺕ ﺩﺍﺩﻥ ﺍﻭ ﺑﻪ ﺧﻮﺍﺑﻴﺪﻥ ﺳﺮ ﺳﺎﻋﺖ ﻣﺸﺨﺺ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ. ﺍﻳﻦ ﮐﺎﺭ ﺍﻣﺘﻴﺎﺯﺍﺕ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺭﺩ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﮐﻮﺩﮎ «ﺍﺣﺘﻴﺎﺝ ﻃﺒﻴﻌﯽ ﺑﻪ ﺧﻮﺍﺑﻴﺪﻥ» ﺭﺍ ﺍﺣﺴﺎﺱ ﻧﻤﯽ‌ﮐﻨﺪ. ﻣﺎ ﺑﻪ ﻫﺮ ﭼﻴﺰﯼ ﮐﻪ ﺑﺮﺍﯼ ﺧﻮﺍﺑﺎﻧﺪﻥ ﺍﻭ ﻣﺎ ﺭﺍ ﻳﺎﺭﯼ ﻣﯽ‌ﺩﻫﺪ ﻣﺤﺘﺎﺝ ﻫﺴﺘﻴﻢ ﻭ ﺍﮔﺮ ﺍﻳﻦ ﺧﺼﻠﺖ ﻋﻼﻗﻪ ﺑﻪ ﺗﮑﺮﺍﺭ ﭼﻴﺰﯼ ﺑﺎﺷﺪ  ﺑﺎﻳﺪ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ. ﺑﺎﻳﺪ ﺑﻪ ﮐﻮﺩﮎ ﺑﻔﻬﻤﺎﻧﻴﻢ ﮐﻪ ﺧﻮﺍﺑﻴﺪﻥ ﺩﺭﺳﺖ ﻣﺜﻞ ﻏﺮﻭﺏ ﺧﻮﺭﺷﻴﺪ ﻏﻴﺮ ﻗﺎﺑﻞ ﺍﺟﺘﻨﺎﺏ ﺍﺳﺖ.

ﺩﺭ ﺍﻳﻦ ﺟﺎ ﻳﮏ ﭘﻴﺸﻨﻬﺎﺩ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﮐﻨﻢ: ﺍﻭﻻ ﺑﻌﺪ ﺍﺯ ﺷﺎﻡ ﻫﻴﭻ ﮔﻮﻧﻪ ﺑﺎﺯﯼ ﭘﺮ ﺗﺤﺮﻛﯽ ﺑﺎ ﮐﻮﺩﮎ ﻧﮑﻨﻴﺪ، ﭼﺮﺍ ﮐﻪ ﻣﻮﺟﺐ ﻫﻴﺠﺎﻥ ﺯﺩﮔﯽ ﺍﻭ ﻣﯽ­‌ﺷﻮﺩ ﻭ ﺑﺎﺯﮔﺸﺖ ﺑﻪ ﺁﺭﺍﻣﺶ ﺭﺍ ﺳﺨﺖ‌ﺗﺮ ﻣﯽ‌ﮐﻨﺪ. ﻋﺎﺩﺕ ﺧﻮﺍﺑﺎﻧﺪﻥ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺣﻤﺎﻡ ﮐﺮﺩﻥ ﻃﻔﻞ ﺷﺮﻭﻉ ﮐﻨﻴﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺣﻤﺎﻡ ﮐﺮﺩﻥ ﻓﻘﻂ ﺑﺮﺍﯼ ﺗﻤﻴﺰ ﺷﺪﻥ ﺍﺳﺖ ﻭ ﺁﻥ ﺭﻭﯼ ﺳﮑﻪ ﺭﺍ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭ ﺣﻤﺎﻡ ﺁﺏ‌ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ ﻧﻤﯽ‌ﺑﻴﻨﻨﺪ. ﺑﻪ ﺍﻭ ﺍﻣﮑﺎﻥ ﺁﺏ‌ﺑﺎﺯﯼ ﺑﺪﻫﻴﺪ، ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﮐﻪ ﺍﻭ ﺍﺯ ﻫﻤﻪ ﺗﻤﻴﺰ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺩﺭ ﻭﺍﻥ ﺣﻤﺎﻡ ﺑﺎﺯﯼ ﮐﻨﺪ ﺗﺎ ﺧﻮﺩﺵ ﺧﺴﺘﻪ ﺷﻮﺩ ﻭ ﺑﮕﻮﻳﺪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻴﺮﻭﻥ ﺑﻴﺎﻳﺪ (ﺍﮔﺮ ﺑﻪ ﻋﻠﺘﯽ، ﻳﮏ ﺷﺐ ﻋﺠﻠﻪ ﺩﺍﺷﺘﻴﺪ، ﺳﻌﯽ ﮐﻨﻴﺪ ﺯﻭﺩﺗﺮ ﮐﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺑﮑﻨﺪ.) ﻭﻗﺘﯽ ﺩﺭ ﻭﺍﻥ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﮕﺬﺍﺭﻳﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻋﺚ ﺁﺭﺍﻣﺶ ﺍﻭ ﻣﯽ‌ﺷﻮﺩ. ﺑﻌﺪ ﺍﺯ ﺣﻤﺎﻡ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﻭﺩ ﺩﺭ ﺗﺨﺘﺶ ﻭ ﻏﺬﺍﻳﺶ ﺭﺍ ﺩﺭ ﺭﺧﺘﺨﻮﺍﺑﺶ ﺑﺨﻮﺭﺩ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﺎ ﺍﺟﺎﺯﻩ ﺩﺍﺩﻥ ﺧﻮﺭﺩﻥ ﭼﻴﺰﯼ ﺩﺭ ﺭﺧﺘﺨﻮﺍﺑﺶ، ﺧﻮﺍﺑﻴﺪﻥ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﻴﺪ. ﺍﻳﻦ ﻏﺬﺍﯼ ﮐﻮﭼﮑﯽ ﮐﻪ ﺩﺭ ﺭﺧﺘﺨﻮﺍﺑﺶ ﻣﯽ‌ﺧﻮﺭﺩ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺠﺎﻝ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺻﺤﺒﺖ‌ﻫﺎﯼ ﻣﺤﺒﺖ ﺁﻣﻴﺰ ﻭ ﺧﻮﺩﺟﻮﺵ ﺑﺎﺷﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ، ﺑﺎ ﺣﻀﻮﺭ ﻣﺎﺩﺭﺍﻧﻪ ﺷﻤﺎ، ﺧﻮﺍﺑﻴﺪﻥ ﺭﺍﺣﺖ‌ﺗﺮ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﮔﻴﺮﺩ. ﺑﻌﺪ ﺍﺯ ﻏﺬﺍ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻳﮏ ﻳﺎ ﺩﻭ ﺩﺍﺳﺘﺎﻥ ﺑﺮﺍﻳﺶ ﺑﺨﻮﺍﻧﻴﺪ.

ﺍﮐﻨﻮﻥ ﻟﺤﻈﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﭘﺪﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺍﻳﻦ ﺗﺸﺮﻳﻔﺎﺕ ﺧﻮﺍﺑﻴﺪﻥ ﺩﺧﺎﻟﺖ ﮐﻨﺪ. ﺑﻌﻀﯽ ﺍﺯ ﭘﺪﺭﻫﺎ ﺩﻭﺳﺖ ﻧﺪﺍﺭﻧﺪ ﺩﺍﺳﺘﺎﻥ ﺑﺮﺍﯼ ﮐﻮﺩﮐﺎﻥ ﮐﻮﭼﮏ ﺑﺨﻮﺍﻧﻨﺪ. ﺟﺎﯼ ﺑﺴﯽ ﺗﺎﺳﻒ ﺍﺳﺖ.

ﻭﻗﺘﯽ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﮐﺘﺎﺏ ﻣﯽ‌ﺧﻮﺍﻧﻴﺪ، ﺁﻥ ﺭﺍ ﻃﻮﺭﯼ ﻧﮕﻪ‌ﺩﺍﺭﻳﺪ ﮐﻪ ﺍﻭ ﺑﺘﻮﺍﻧﺪ ﻋﮑﺲ‌ﻫﺎﯼ ﺁﻥ ﺭﺍ ﺑﺒﻴﻨﺪ. ﺍﻳﻦ ﺗﺼﺎﻭﻳﺮ ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﺩﺍﺭﻧﺪ. ﺑﺎ ﺳﻮﺍﻝ ﮐﺮﺩﻥ ﺩﺭ ﻣﻮﺭﺩ ﺟﺰﺋﻴﺎﺗﯽ ﮐﻪ ﺩﺭ ﺗﺼﺎﻭﻳﺮ ﮐﺘﺎﺏ ﻫﺴﺖ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ «ﺩﻗﺖ ﮐﺮﺩﻥ» ﻭ «ﺩﻳﺪﻥ» ﺗﻤﺎﻡ ﺟﺰﺋﻴﺎﺕ ﺭﺍ ﻧﻴﺰ ﺑﻴﺎﻣﻮﺯﻳﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﻮﺩﮐﺎﻥ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﺯ ﺗﺎﺭﻳﮑﯽ ﻣﯽ‌ﺗﺮﺳﻨﺪ. ﭼﺮﺍ ﺑﺎﻳﺪ ﮐﻮﺩﮎ ﺭﺍ ﻣﺠﺒﻮﺭ ﮐﻨﻴﻢ ﺩﺭ ﺍﺗﺎﻗﯽ ﺗﺎﺭﻳﮏ ﺑﺨﻮﺍﺑﺪ؟ ﻫﻴﭻ ﻋﻠﺘﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﭼﺮﺍﻍ ﺧﻮﺍﺏ ﮐﻮﭼﮑﯽ ﺩﺭ ﺍﺗﺎﻗﺶ ﺭﻭﺷﻦ ﻧﮑﻨﻴﺪ. ﺩﺍﺷﺘﻦ ﺍﻧﺪﮐﯽ ﺭﻭﺷﻨﺎﻳﯽ ﺩﺭ ﺍﺗﺎﻗﺶ ﺍﻳﻦ ﺗﺮﺱ ﺭﺍ ﺍﺯ ﺑﻴﻦ ﻣﯽ‌ﺑﺮﺩ.

ﻋﻠﺘﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻟﻪ ﻳﺎ ﺩﻭ ﺳﺎﻝ­‌ ﻭ­ ﻧﻴﻤﻪ ﺷﺐ ﻣﻮﻗﻊ ﺧﻮﺍﺏ ﻳﮏ ﺷﻴﺸﻪ ﺷﻴﺮ ﻧﺨﻮﺭﺩ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻴﻠﯽ ﺍﺯ ﻫﻤﺴﺎﻳﻪ‌ﻫﺎﯼ ﻣﺘﺨﺼﺺ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﻨﺪ: «ﭼﯽ، ﻫﻨﻮﺯ ﺗﻮﯼ ﺍﻳﻦ ﺳﻦ ﺑﻬﺶ ﺗﻮ ﺷﻴﺸﻪ ﺷﻴﺮ ﻣﻴﺪﯼ؟» ﮐﺠﺎﯼ ﺍﻳﻦ ﮐﺎﺭ ﻏﻴﺮ ﻃﺒﻴﻌﯽ ﺍﺳﺖ؟ ﺍﮔﺮ ﺑﺮﺍﯼ ﺍﺣﺴﺎﺱ ﺍﻣﻨﻴﺖ ﮐﺮﺩﻥ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻳﮏ ﺷﯽﺀ ﺧﺎﺹ ﺩﺍﺭﺩ، ﻫﻴﭻ ﻋﻠﺘﯽ ﻧﺪﺍﺭﺩ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﻭ ﺭﺍ ﻣﺠﺒﻮﺭ ﺑﻪ ﺗﺮﮎ ﺍﻳﻦ ﻋﺎﺩﺕ ﺑﮑﻨﻴﺪ.

ﺑﺴﻴﺎﺭ ﺧﻮﺏ ﺍﺳﺖ ﮐﻪ ﻋﺎﺩﺕ ﺧﻮﺍﺑﺎﻧﺪﻥ ﮐﻮﺩﮎ ﻫﻤﺮﺍﻩ ﺑﺎ ﻟﺬﺕ ﺑﺎﺷﺪ. ﺍﻭ ﺑﺎﻳﺪ ﺑﺎ ﺑﯽ‌ﺻﺒﺮﯼ ﺑﻪ ﺗﺨﺘﺶ ﺑﺮﻭﺩ. ﻧﺒﺎﻳﺪ ﻟﺤﻈﻪ ﺭﻓﺘﻦ ﺑﻪ ﺗﺨﺘﺶ ﺭﺍ ﺷﮑﻨﺠﻪ‌ﺍﯼ ﺑﺪﺍﻧﺪ ﻭ ﻓﮑﺮ ﮐﻨﺪ ﺩﺭ ﺣﺎﻟﯽ‌ﮐﻪ ﺑﻘﻴﻪ ﺍﻓﺮﺍﺩ ﻓﺎﻣﻴﻞ ﺑﻴﺪﺍﺭ ﻣﯽ‌ﻣﺎﻧﻨﺪ ﻭ ﺗﻔﺮﻳﺢ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﻭ ﺑﺎﻳﺪ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺗﺎﺭﻳﮑﯽ ﺣﺒﺲ ﮐﻨﺪ. ﻣﺮﺍﺳﻢ ﺧﻮﺍﺑﻴﺪﻥ ﮐﻮﺩﮎ ﺑﺎﻳﺪ ﻧﻪ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺑﻠﮑﻪ ﺑﺮﺍﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﻧﻴﺰ ﻟﺬﺕ ﺑﺨﺶ ﺑﺎﺷﺪ.
 


ﺩﺭ ﺍﻳﻦ‌ﺟﺎ ﺑﻪ ﭘﺎﻳﺎﻥ ﻣﺒﺤﺚ ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﻣﯽ‌ﺭﺳﻴﻢ، ﺍﻣﺎ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ‌ﮐﻪ ﺍﻳﻦ ﺑﺤﺚ ﺭﺍ ﺑﺒﻨﺪﻳﻢ ﻳﮏ ﻣﻄﻠﺐ ﺭﺍ ﺍﺿﺎﻓﻪ ﮐﻨﻴﻢ. ﺩﺭ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ، ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻭﺟﻪ ﺭﺷﺪ، ﮐﻴﻔﻴﺖ ﭘﻮﻳﺎﯼ ﺷﺨﺼﻴﺖ ﮐﻮﺩﮎ ﺍﺳﺖ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺍﻳﻦ ﮐﻴﻔﻴﺖ ﭘﻮﻳﺎﯼ ﺷﺨﺼﻴﺖ ﻃﻔﻞ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﻨﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻨﺪ ﺑﻪ ﺍﻭ ﺟﻬﺖ ﮐﺴﺐ ﺧﻮﺩ ﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻗﻮﯼ ﻳﺎﺭﯼ ﺭﺳﺎﻧﻨﺪ.

ﻋﺠﻴﺐ ﺍﺳﺖ ﮐﻪ ﺍﮐﺜﺮ ﻭﺍﻟﺪﻳﻦ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﻧﺪ ﻗﻮﯼ ﻭ ﭘﻮﻳﺎ ﺑﺎﺷﻨﺪ ﺍﻣﺎ ﻫﻤﻴﻦ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺍﻳﻦ ﭘﻮﻳﺎﻳﯽ ﺭﺍ ﺩﺭ ﺩﻭ ﺳﺎﻟﮕﯽ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﻗﺒﻮﻝ ﮐﻨﻨﺪ. ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻟﻪ ﻣﻈﻬﺮ ﭘﻮﻳﺎﻳﯽ ﺍﺳﺖ. ﻣﮕﺮ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﺨﺎﻟﻔﺖ ﺷﺪﻳﺪ ﺍﻭ ﺭﺍ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ، ﻟﺬﺕ ﺑﺮﺩﻥ ﺍﺯ ﺯﻧﺪﮔﯽ، ﺗﻮﻗﻊ ﺑﺮﺁﻭﺭﺩﻥ ﻓﻮﺭﯼ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﻳﺶ، ﺷﻮﺭ ﻭ ﺍﺷﺘﻴﺎﻕ ﺍﻭ ﺭﺍ ﺑﺮﺍﯼ ﺷﺮﮐﺖ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﮐﻪ ﺭﻭﺯ ﺑﻪ ﺭﻭﺯ ﭼﻴﺰﻫﺎﯼ ﺟﺪﻳﺪﯼ ﺍﺯ ﺁﻥ ﮐﺸﻒ ﻣﯽ‌ﮐﻨﺪ، ﻧﺪﻳﺪﻩ ﮔﺮﻓﺖ؟ ﺗﻤﺎﻣﯽ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﻣﻨﺤﺼﺮ ﺑﻪ ﻓﺮﺩ ﺑﻮﺩﻥ ﺍﻭ ﻣﯽ‌ﺍﻧﺠﺎﻣﺪ، ﻧﺎﺷﯽ ﺍﺯ ﺍﻳﻦ ﭘﻮﻳﺎﻳﯽ ﺍﺳﺖ. ﺑﺪﻭﻥ ﺍﻳﻦ ﭘﻮﻳﺎﻳﯽ ﻧﺸﺴﺘﻦ، ﺳﻴﻨﻪ­‌ﺧﻴﺰ ﺭﻓﺘﻦ، ﺭﺍﻩ ﺭﻓﺘﻦ ﻭ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺭﺍ ﻳﺎﺩ ﻧﻤﯽ‌ﮔﺮﻓﺖ. ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﺍﻳﻦ ﺧﺼﻠﺖ ﭘﻮﻳﺎﻳﯽ، ﻣﻨﺒﻊ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﺗﻌﺎﺩﻝ ﺭﻭﺣﯽ ﺍﻭﺳﺖ. ﮐﻮﺩﮎ ﻧﺒﺎﻳﺪ ﺍﻳﻦ ﺧﺼﻠﺖ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺑﺪﻫﺪ. ﺑﺮﻋﮑﺲ، ﺑﺎﻳﺪ ﺍﻳﻦ ﺧﺼﻠﺖ ﺭﺍ ﺑﻪ ﻣﺜﺎﺑﻪ ﻭﺭﻗﯽ ﺑﺮﻧﺪﻩ ﺣﻔﻆ ﮐﻨﻴﻢ، ﻧﻪ ﺁﻥ ﮐﻪ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻋﻴﺐ ﺳﻌﯽ ﮐﻨﻴﻢ ﺁﻥ ﺭﺍ ﺩﺭ ﺍﻭ ﺍﺯ ﺑﻴﻦ ﺑﺒﺮﻳﻢ.

ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ، ﺑﻪ ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ ﭼﻴﺰﯼ ﻋﺮﺿﻪ ﮐﻨﻴﻢ: ﺍﺣﺘﺮﺍﻡ ﺑﻪ ﭘﻮﻳﺎﻳﯽ ﺣﻴﺎﺗﯽ ﺍﻭ. ﻗﻮﺍﻧﻴﻨﯽ ﻣﻄﺎﺑﻖ ﺑﺎ ﺍﻳﻦ ﭘﻮﻳﺎﻳﯽ ﻭ ﻗﺎﺑﻠﻴﺖ ﺍﻧﻄﺒﺎﻗﯽ ﺍﻧﻌﻄﺎﻑ‌ﭘﺬﻳﺮ ﺍﺯ ﺟﺎﻧﺐ ﺧﻮﺩﻣﺎﻥ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻴﻢ ﺗﺎ ﻋﻴﻨﻜﯽ ﻛﻪ ﺍﺯ ﻭﺭﺍﯼ ﺁﻥ ﺟﻬﺎﻥ ﺭﺍ ﻣﯽ‌­ﻧﮕﺮﺩ ﺑﺎ ﺷﻴﺸﻪ­‌ﺍﯼ ﮐﻪ ﻣﺨﺼﻮﺹ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺳﺖ ﮐﺎﻣﻞ ﺳﺎﺯﺩ.
