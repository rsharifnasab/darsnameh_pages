ﺩﺭﺱ ﺍﻭﻝ- ﺳﻮﺀ ﺗﻔﺎﻫﻢ ﻫﺎﻳﯽ ﺩﺭﺑﺎﺭﻩ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﮐﻮﺩﮎ

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺑﺮﺩﺍﺷﺖ ﻏﻠﻄﯽ ﺍﺯ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺩﺍﺭﻧﺪ. ﭘﻴﺶ ﺍﺯ ﺻﺤﺒﺖ ﺩﺭﺑﺎﺭﻩ ﺁﻧﭽﻪ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺩﺭ ﻣﻮﺭﺩ ﺗﺮﺑﻴﺖ ﮐﻮﺩﮎ ﻣﻄﺮﺡ ﻣﯽ‌ﮐﻨﺪ، ﺑﺪ ﻧﻴﺴﺖ ﺍﺷﺎﺭﻩ‌ﺍﯼ ﮐﻨﻴﻢ ﺑﻪ ﺁﻥ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻗﺒﻮﻝ ﻧﺪﺍﺭﺩ.‌

ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﻗﺒﻮﻝ ﻧﺪﺍﺭﺩ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﺑﺎﻳﺪ ﺍﺟﺎﺯﻩ ﺩﻫﻨﺪ ﮐﻮﺩﮎ ﻫﺮ ﭼﻪ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﮑﻨﺪ.  ﻫﻴﭻ ﺭﻭﺍﻧﺸﻨﺎﺱ ﻋﺎﻗﻠﯽ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﺍﺯ ﭼﻨﻴﻦ ﻋﻘﻴﺪﻩ ﻋﺠﻴﺒﯽ ﺩﻓﺎﻉ ﻧﮑﺮﺩﻩ ﺍﺳﺖ. ﻭﻟﯽ ﺷﺎﻳﺪ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﻋﺠﻴﺐ ﺑﺎﺷﺪ ﺑﮕﻮﻳﻴﻢ ﺑﻌﻀﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺭﺍ ﺁﺯﺍﺩ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ ﺣﺘﯽ ﺭﻭﯼ ﺩﻳﻮﺍﺭﻫﺎ ﺑﺎ ﺧﻮﺩﮐﺎﺭ ﻭ ﻣﺎﮊﻳﻚ ﻭ ﮔﭻ‌ﻫﺎﯼ ﺭﻧﮕﯽ ﺧﻂ ﺑﮑﺸﻨﺪ، ﺯﻳﺮﺍ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﮔﺮ ﺟﻠﻮ ﺍﻳﻦ ﮐﺎﺭ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﮕﻴﺮﻧﺪ، ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﮐﺎﺭ ﺑﺪﯼ ﮐﺮﺩﻩ‌ﺍﻧﺪ!  ﺩﺭ ﻭﺍﻗﻊ ﻋﻠﺖ ﺍﻳﻦ ﺍﻣﺮ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻣﺎﺩﺭ ﺟﺮﺍﺕ «ﻧﻪ» ﮔﻔﺘﻦ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﻧﺪﺍﺭﺩ ﻭ ﺑﺮﺍﯼ ﺗﻮﺟﻴﻪ ﺍﻳﻦ ﮐﺎﺭﺵ ﻣﺘﻮﺳﻞ ﺑﻪ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻣﯽ‌ﺷﻮﺩ.

ﺍﺻﻞ ﺩﻭﻣﯽ ﮐﻪ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻣﺮﺩﻭﺩ ﺍﺳﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﻣﺮﺍﺣﻞ ﻣﺨﺘﻠﻒ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻧﺒﺎﻳﺪ ﺩﺧﺎﻟﺘﯽ ﺑﮑﻨﻨﺪ ﻭ ﺑﺎﻳﺪ ﺑﮕﺬﺍﺭﻧﺪ ﻫﺮ ﻣﺮﺣﻠﻪ ﺩﻭﺭﻩ ﺧﻮﺩ ﺭﺍ ﻃﯽ ﮐﻨﺪ.
ﻣﺴﻠﻢ ﺍﺳﺖ ﮐﻮﺩﮐﺎﻥ، ﺧﺼﻮﺻﺎ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺳﻦ ﺷﺶ ﺳﺎﻟﮕﯽ، ﻣﺮﺍﺣﻞ ﻣﺘﻔﺎﻭﺕ ﻭ ﻣﺸﺨﺺ ﺭﺷﺪ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ. ﻣﺎﺩﺭﯼ ﮐﻪ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻓﺮﺯﻧﺪ ﭼﻬﺎﺭ ﺳﺎﻟﻪ‌ﺍﺵ ﻣﺎﻧﻨﺪ ﮐﻮﺩﮐﯽ ﺳﻪ‌ ﺳﺎﻟﻪ ﺭﻓﺘﺎﺭ ﮐﻨﺪ ﺣﺘﻤﺎ ﺑﺎ ﻧﺎﮐﺎﻣﯽ ﺭﻭﺑﻪ‌ﺭﻭ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺑﺎ ﺍﻳﻦ ﻫﻤﻪ، ﺍﻳﻨﮑﻪ ﻭﺍﻟﺪﻳﻦ ﻗﺒﻮﻝ ﮐﻨﻨﺪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺑﺎﻳﺪ ﻣﺮﺍﺣﻞ ﻣﺨﺘﻠﻒ ﺭﺷﺪ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﺑﮕﺬﺍﺭﻧﺪ، ﺑﻪ  ﺍﻳﻦ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﺁﻧﻬﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﻫﻴﭻ ﭼﻴﺰﯼ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﺗﻐﻴﻴﺮ ﺩﻫﻨﺪ.  ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﻮﺛﺮ ﺑﺎﺷﻨﺪ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﮐﻪ ﺭﻓﺘﺎﺭﯼ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﻫﺮ ﺩﻭﺭﻩ ﺍﺯ ﺭﺷﺪ ﻃﻔﻞ ﺩﺍﺭﻧﺪ ﺍﺛﺮﯼ ﻣﺴﺘﻘﻴﻢ ﺑﺮ ﭼﮕﻮﻧﮕﯽ ﻃﯽ ﮐﺮﺩﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺍﺭﺩ.

ﻧﻈﺮ  ﺑﺴﻴﺎﺭ ﺷﺎﻳﻊ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﺧﻮﺏ ﻧﺒﺎﻳﺪ ﻫﻴﭻ‌ﮔﺎﻩ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻧﺪ، ﻳﺎ ﺍﺯ ﮐﺎﺭﯼ ﺟﻠﻮﮔﻴﺮﯼ ﮐﻨﻨﺪ. ﺑﺮﺍﯼ ﻫﺮ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﻣﻮﺍﻗﻊ ﺧﻮﺷﯽ ﻭ ﺍﻓﺴﺮﺩﮔﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺭﻓﺘﺎﺭ ﺑﺪ ﻳﮑﯽ ﺍﺯ ﻭﺭﻭﺟﮏ ﻫﺎﻳﻤﺎﻥ ﺭﺍ ﺑﺎ ﺭﺍﺣﺘﯽ ﻭ ﻣﻼﻳﻤﺖ ﺗﻤﺎﻡ ﺗﺤﻤﻞ ﮐﻨﻴﻢ. ﻭﻟﯽ ﻣﻮﺍﻗﻌﯽ ﻫﻢ ﻫﺴﺖ ﮐﻪ ﮐﻮﭼﮑﺘﺮﻳﻦ ﮐﺎﺭ ﺍﺣﻤﻘﺎﻧﻪ ﻓﺮﺯﻧﺪﻣﺎﻥ ﻓﺮﻳﺎﺩﻣﺎﻥ ﺭﺍ ﺑﻪ ﺁﺳﻤﺎﻥ ﻣﯽ‌ﺑﺮﺩ.

ﺻﺪﺍﻗﺖ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺗﺎﺛﺮﺍﺗﺸﺎﻥ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ. ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺑﻪ ﻣﺎ ﻧﻤﯽ‌ﮔﻮﻳﺪ ﻫﻤﻴﺸﻪ ﺁﺭﺍﻡ ﺑﺎﺷﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻭﺍﻟﺪﻳﻦ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﻓﺸﺎﺭ ﺭﻭﺣﯽ ﺷﺪﻳﺪﯼ ﺭﺍ ﺗﺤﻤﻞ ﮐﻨﻨﺪ.
ﻧﻤﻲ‌ﺗﻮﺍﻥ ﺑﻪ ﻭﺍﻟﺪﻳﻦ ﮔﻔﺖ ﭼﮕﻮﻧﻪ ﺍﺣﺴﺎﺳﺎﺗﯽ ﺑﺎﻳﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺗﻨﻬﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﮔﻔﺖ ﭼﻪ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﮑﻨﻨﺪ. ﺍﻋﻤﺎﻝ ﺷﻤﺎ ﺗﺤﺖ ﺍﺧﺘﻴﺎﺭ ﺷﻤﺎ ﺍﺳﺖ ﻭﻟﯽ ﺍﺣﺴﺎﺳﺎﺗﺘﺎﻥ ﻧﻪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﻣﻔﺎﻫﻴﻤﯽ ﺭﺍ ﻋﺮﺿﻪ ﮐﻨﻴﻢ ﮐﻪ ﺑﺮﺍﯼ ﺩﺭﮎ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﮐﻨﺪ ﻭ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﭼﻪ ﮐﺎﺭﻫﺎﻳﯽ ﺑﮑﻨﻴﺪ ﺗﺎ ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺑﺰﺭﮒ ﺷﺪﻧﺪ ﺍﻧﺴﺎﻥ ﻫﺎﯼ ﺧﻮﺷﺒﺨﺖ ﻭ ﺑﺎﻫﻮﺷﯽ ﺑﺎﺷﻨﺪ.

ﺣﺎﻝ ﮐﻪ ﺳﻮﺀ ﺗﻔﺎﻫﻤﺎﺕ ﺩﺭﺑﺎﺭﻩ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ، ﺑﺒﻴﻨﻴﻢ ﻭﺍﻟﺪﻳﻦ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﺍﺯ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻨﺪ.


 ﺍﻫﻤﻴﺖ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻟﯽ ﺯﻧﺪﮔﯽ

ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﮑﺘﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺳﺎﻝ‌ﻫﺎﯼ ﺷﮑﻞ‌ﮔﻴﺮﯼ ﺍﻭ ﻭ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺳﺎﻝ‌ﻫﺎﯼ ﺯﻧﺪﮔﯽ ﺍﻭ ﻫﺴﺘﻨﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﺳﺎﻝ‌ﻫﺎﯼ ﺑﻌﺪ ﺍﻫﻤﻴﺖ ﮐﻤﺘﺮﯼ ﺩﺍﺭﻧﺪ ﻭ ﺍﻳﻦ ﮐﻪ ﻫﻤﻪ ﭼﻴﺰ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺷﺸﻤﻴﻦ ﺳﺎﻟﮕﺮﺩ ﺗﻮﻟﺪ ﺍﺗﻔﺎﻕ ﻣﯽ‌ﺍﻓﺘﺪ. ﻭﻟﯽ ﻣﻄﻤﺌﻨﺎ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﺳﺎﻝ‌ﻫﺎﯼ ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ‌ﺍﯼ ﻫﺴﺘﻨﺪ.

ﺩﺭ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﺳﺎﺧﺘﺎﺭ ﺍﺳﺎﺳﯽ ﺷﺨﺼﻴﺖ ﮐﻮﺩﮎ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﺩ. ﻳﻌﻨﯽ ﺷﺨﺼﻴﺘﯽ ﮐﻪ ﺗﻤﺎﻡ ﻋﻤﺮ ﺍﻭ ﺭﺍ ﻫﻤﺮﺍﻫﯽ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺍﻳﻦ ﺩﻭﺭﺍﻥ ﺗﺎﺛﻴﺮ ﻋﻤﺪﻩ ﻭ ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ‌ﺍﯼ ﺩﺭ ﻣﻮﻓﻘﻴﺖ ﺍﻭ ﺩﺭ ﺗﺤﺼﻴﻼﺕ ﻭ ﻫﻤﻴﻦ ﻃﻮﺭ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﺍﻧﺴﺎﻥ ﺑﺎﻟﻎ ﺩﺍﺭﺩ: ﺭﻓﺘﺎﺭ ﺍﻭ ﺩﺭ ﺟﺎﻣﻌﻪ، ﭼﮕﻮﻧﮕﯽ ﺭﻓﺘﺎﺭﺵ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﺴﺎﻳﻞ ﺟﻨﺴﯽ، ﺩﻭﺭﺍﻥ ﺟﻮﺍﻧﯽ، ﻧﻮﻉ ﺍﻧﺴﺎﻧﯽ ﮐﻪ ﺑﺎ ﺍﻭ ﺍﺯﺩﻭﺍﺝ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﻭ ﻣﻮﻓﻘﻴﺖ ﻭ ﻋﺪﻡ ﻣﻮﻓﻘﻴﺖ ﺍﻭ ﺩﺭ ﺍﺯﺩﻭﺍﺝ.
ﺍﻟﺒﺘﻪ ﻧﺒﺎﻳﺪ ﻫﻮﺵ ﺭﺍ ﺑﺎ ﺩﺍﻧﺶ ﺍﺷﺘﺒﺎﻩ ﮐﺮﺩ. ﻣﺴﻠﻤﺎ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﭘﻨﺠﺎﻩ ﺩﺭﺻﺪ ﻣﻌﻠﻮﻣﺎﺗﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﺑﻠﻮﻍ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ ﺗﺎ ﺳﻦ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﮐﺴﺐ ﻧﻤﯽ‌ﮐﻨﺪ. ﻫﻮﺵ ﻗﺎﺑﻠﻴﺖ ﺫﻫﻨﯽ ﺍﻧﺴﺎﻥ ﺑﺮﺍﯼ ﺑﻪ ﮐﺎﺭﺑﺮﺩﻥ ﺑﻪ‌ﺟﺎﯼ ﻣﻌﻠﻮﻣﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﮐﺴﺐ ﻣﯽ‌ﮐﻨﺪ. ﻭ ﭘﻨﺠﺎﻩ ﺩﺭﺻﺪ ﺍﻳﻦ ﻇﺮﻓﻴﺖ ﺗﺎ ﺳﻦ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﮐﺴﺐ ﺷﺪﻩ ﺍﺳﺖ.

ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ‌ﯼ ﺩﺭﺳﻨﺎﻣﻪ ﺑﻪ ﮐﻮﺩﮐﯽ ﺗﺨﻴﻠﯽ ﺍﺷﺎﺭﻩ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﺩﻭﺭﺍﻥ ﻣﺨﺘﻠﻒ ﺭﺷﺪ ﺍﻭ ﺭﺍ ﺍﺯ ﺯﻣﺎﻥ ﺗﻮﻟﺪ ﺗﺎ ﺳﻦ ﺷﺶ ﺳﺎﻟﮕﯽ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﻢ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﻘﺎﻳﺴﻪ ﮐﺮﺩﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎ ﺍﻳﻦ ﮐﻮﺩﮎ ﮐﺎﺭ ﺍﺷﺘﺒﺎﻫﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺩﺭ ﺩﺭﺱ‌‌ ﺑﻌﺪ، ﺍﺯ ﻧﻴﺎﺯﻫﺎﯼ ﺍﺳﺎﺳﯽ ﻧﻮﺯﺍﺩ ﻭ ﺍﻫﻤﻴﺖ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﺁﻥ‌ﻫﺎ ﺧﻮﺍﻫﻴﻢ ﮔﻔﺖ.
