ﺩﺭﺱ ﺩﻭﺍﺯﺩﻫﻢ- ﺍﻧﻀﺒﺎﻁ ﺩﺭ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ

ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ، ﺳﺨﻦ ﺍﺯ ﮐﻨﺘﺮﻝ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺮ ﮐﻮﺩﮎ ﮔﻔﺘﻴﻢ. ﺍﻳﻦ ﺑﺤﺚ ﻣﺎ ﺭﺍ ﻧﺎﭼﺎﺭ ﺑﻪ ﺻﺤﺒﺖ ﺩﺭﺑﺎﺭﻩ ﺍﻧﻀﺒﺎﻁ ﻣﯽ‌ﮐﻨﺪ. ﺗﺎﮐﻨﻮﻥ، ﺑﻪ ﻣﻌﻨﯽ ﻭﺍﻗﻌﯽ ﮐﻠﻤﻪ، ﻣﺎ ﻣﻮﺭﺩﯼ ﺑﺮﺍﯼ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﻧﺪﺍﺷﺘﻴﻢ. ﺍﮔﺮ ﻣﺎﺩﺭ ﻣﺤﻴﻄﯽ ﻣﻨﺎﺳﺐِ ﺣﺎﻝ ﻳﮏ ﻧﻮﭘﺎﯼ ﮐﻮﭼﮏ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﺩﻩ ﺑﺎﺷﺪ، ﻋﻠﺘﯽ ﻧﺪﺍﺭﺩ ﮐﻪ ﭼﻴﺰﯼ ﺭﺍ ﺑﺮ ﺍﻭ ﺗﺤﻤﻴﻞ ﮐﻨﺪ. ﻭﻟﯽ ﺩﺭ ﻣﺮﺣﻠﻪ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﮐﻮﺩﮎ ﺑﺎﻳﺪ ﺍﻧﻀﺒﺎﻁ ﺭﺍ ﻓﺮﺍ ﮔﻴﺮﺩ ﻭ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺑﺪﺍﻧﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﻋﻤﻞ ﮐﻨﺪ.

ﺩﺭ ﺍﻳﻦ ﺑﺨﺶ ﻓﻘﻂ ﺍﺯ ﺍﻧﻀﺒﺎﻁ ﺩﺭ ﻣﻮﺭﺩ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ ﻭﻟﯽ ﺑﺪ ﻧﻴﺴﺖ ﮐﻪ ﻧﻈﺮﯼ ﮐﻠﯽ ﺩﺭﺑﺎﺭﻩ ﺍﻧﻀﺒﺎﻁ ﻭ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﮐﻮﺩﮎ ﺑﺎﻳﺪ ﺭﻋﺎﻳﺖ ﮐﻨﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

ﺍﻭﻟﻴﻦ ﮐﺎﺭﯼ ﮐﻪ ﻭﺍﻟﺪﻳﻦ ﺑﺎﻳﺪ ﺩﺭ ﻣﻮﺭﺩ ﮐﻮﺩﮐﺎﻥ ﺍﻳﻦ ﺳﻦ ﺍﻧﺠﺎﻡ ﺩﻫﻨﺪ ﺗﻔﺎﻭﺕ ﻗﺎﺋﻞ ﺷﺪﻥ ﺑﻴﻦ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻋﻤﺎﻝ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺍﺳﺖ. ﺍﻭ ﺍﺯ ﺧﻴﺎﺑﺎﻥ ﺭﺩ ﻣﯽ‌ﺷﻮﺩ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﺮﺍﯼ ﺍﻭ ﻗﺪﻏﻦ ﮐﺮﺩﻩ‌ﺍﻳﺪ، ﮐﻮﺩﮎ ﺩﻳﮕﺮﯼ ﺭﺍ ﮐﺘﮏ ﻣﯽ‌ﺯﻧﺪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺑﻪ ﺍﻭ ﮔﻔﺘﻪ ﺑﻮﺩﻳﺪ ﭼﻨﻴﻦ ﮐﺎﺭﯼ ﺭﺍ ﻧﮑﻨﺪ، ﺷﻦ ﭘﺮﺕ ﻣﯽ‌ﮐﻨﺪ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺑﻪ ﺍﻭ ﮔﻔﺘﻪ ﺑﻮﺩﻳﺪ «ﺁﺩﻡ ﺷﻦ ﭘﺮﺕ ﻧﻤﯽ‌ﮐﻨﻪ.» ﻣﻨﻈﻮﺭ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ، ﺗﺎﺛﺮﺍﺕ ﺩﺍﺧﻠﯽ ﮐﻮﺩﮎ ﺍﺳﺖ. ﮐﻮﺩﮎ ﻣﻤﮑﻦ ﺍﺳﺖ ﻋﺼﺒﺎﻧﯽ ﺑﺎﺷﺪ، ﺧﻮﺷﺤﺎﻝ ﺑﺎﺷﺪ، ﺑﺘﺮﺳﺪ، ﺧﺠﺎﻟﺖ ﺑﮑﺸﺪ ﻭ ﻳﺎ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺗﺒﻌﻴﺾ ﺑﻴﻦ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻋﻤﺎﻝ ﮐﻮﺩﮎ ﺣﺎﺋﺰ ﺍﻫﻤﻴﺖ ﻓﺮﺍﻭﺍﻥ ﺍﺳﺖ. ﺯﻳﺮﺍ ﮐﻮﺩﮎ ﻗﺎﺩﺭ ﺑﻪ ﮐﻨﺘﺮﻝ ﺍﻋﻤﺎﻟﺶ ﻫﺴﺖ ﻭﻟﯽ ﮐﻨﺘﺮﻝ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻧﻴﺴﺖ . ﺍﺣﺴﺎﺳﺎﺕ ﮐﻮﺩﮎ، ﺑﻪ ﻣﺎﻧﻨﺪ ﺍﻓﮑﺎﺭﺵ، ﻓﯽ‌ﺍﻟﺒﺪﺍﻫﻪ ﺑﻪ ﺫﻫﻨﺶ ﻣﯽ‌ﺭﺳﻨﺪ. ﺍﻭ ﻧﻪ ﺯﻣﺎﻥ ﺁﻧﻬﺎ ﺭﺍ ﻣﯽ‌ﺩﺍﻧﺪ ﻭ ﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﺤﻮﻩ ﺑﺮﻭﺯ ﺁﻧﻬﺎ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﺪ.

ﻣﺜﻼ ﻋﺼﺒﺎﻧﻴﺖ ﺭﺍ ﺩﺭﻧﻈﺮ ﺑﮕﻴﺮﻳﻢ: ﮐﻮﺩﮎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﻋﺼﺒﺎﻧﯽ ﻧﺸﻮﺩ ﻳﺎ ﻧﺴﺒﺖ ﺑﻪ ﭼﻴﺰﯼ ﺿﺪﻳﺖ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻭﻟﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﻌﻘﻮﻝ ﺍﺯ ﺍﻭ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺍﻋﻤﺎﻝ ﻧﺎﺷﯽ ﺍﺯ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﺪ: ﮐﺘﮏ ﺯﺩﻥ، ﺷﻦ ﭘﺮﺗﺎﺏ ﮐﺮﺩﻥ ﻭ ﻳﺎ ﮔﺎﺯ ﮔﺮﻓﺘﻦ.

ﺍﻭﻟﻴﻦ ﮐﺎﺭﯼ ﮐﻪ ﺑﺎ ﮐﻮﺩﮎ ﺩﺭ ﺳﻦ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﻣﯽ‌ﺗﻮﺍﻥ ﮐﺮﺩ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﮐﻨﻴﻢ ﮐﻪ ﺣﺪﻭﺩﯼ ﻣﻌﻘﻮﻝ ﺑﺮﺍﯼ ﺍﻋﻤﺎﻟﺶ ﻗﺎﺋﻞ ﺷﻮﺩ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺑﺎ ﺍﺳﺘﻨﺒﺎﻃﯽ ﻏﻠﻂ ﺍﺯ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺟﺪﻳﺪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﻣﺸﺨﺺ ﮐﺮﺩﻥ ﻣﺤﺪﻭﺩﻩ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﻋﻤﺎﻝ ﮐﻮﺩﮎ ﺑﻪ ﻣﻌﻨﯽ ﻣﺤﺪﻭﺩ ﮐﺮﺩﻥ ﺭﺷﺪ ﺷﺨﺼﻴﺖ ﺍﻭ ﺍﺳﺖ.

ﻭﻟﯽ ﻣﺤﺪﻭﺩﻩ ﻣﻌﻘﻮﻝ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﭼﻴﺴﺖ؟ ﻫﻴﭻ ﻗﺎﻧﻮﻥ ﺩﻗﻴﻘﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﻭﺿﻊ ﮐﺮﺩ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺍﻳﻦ ﺳﻮﺍﻝ ﮐﻪ «ﭼﻨﺪ ﻣﻤﻨﻮﻋﻴﺖ ﻣﻄﻠﻖ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺖ» ﺷﺮﻭﻉ ﮐﻨﻴﺪ. ﻭﻗﺘﯽ ﻓﻬﺮﺳﺘﯽ ﺍﺯ ﺍﻳﻦ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﺗﻬﻴﻪ ﮐﺮﺩﻳﺪ ﺷﺎﻳﺪ ﺗﻌﺠﺐ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﻓﻬﺮﺳﺖ ﺧﻴﻠﯽ ﮐﻮﺗﺎﻩ‌ﺗﺮ ﺍﺯ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺗﺼﻮﺭ ﻣﯽ‌ﮐﺮﺩﻳﺪ.

 ﻫﺮ ﻭﻗﺖ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻓﺮﺯﻧﺪ ﺩﻭﺳﺎﻟﻪ‌ﺗﺎﻥ ﺭﺍ ﻣﺠﺒﻮﺭ ﺑﻪ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﮐﻨﻴﺪ ﻫﻤﻴﺸﻪ ﭘﻴﺶ ﺧﻮﺩﺗﺎﻥ ﻓﻜﺮ ﻛﻨﻴﺪ : ﭼﻪ ﺍﻫﻤﻴﺘﯽ ﺩﺍﺭﺩ ﺍﻭ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﮑﻨﺪ ﻳﺎ ﻧﮑﻨﺪ؟ ﺁﻳﺎ ﺍﺭﺯﺵ ﺍﻳﻦ ﺭﺍ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩﺵ ﺻﺤﺒﺖ ﮐﻨﻴﺪ؟ ﺑﺮﺍﯼ ﺍﻭ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﭼﻮﻥ ﺑﺨﺎﺭﯼ ﻭ ﺍﺟﺎﻕ ﮔﺎﺯ ﺩﺍﻍ، ﺧﻄﺮ ﻋﺒﻮﺭ ﺍﺯ ﺧﻴﺎﺑﺎﻥ، ﭘﺮﺕ ﻧﮑﺮﺩﻥ ﺷﻦ ﺑﻪ ﺭﻭﯼ ﺑﭽﻪ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻭ ﻏﻴﺮﻩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺩﻳﮕﺮ ﺑﺎ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎ ﻭ ﻣﺤﺪﻭﺩ ﮐﺮﺩﻥ‌ﻫﺎﯼ ﺟﺪﻳﺪ، ﺑﻴﻬﻮﺩﻩ ﻭ ﺑﯽ‌ﺍﻫﻤﻴﺖ ﺯﻧﺪﮔﯽ ﺧﻮﺩﺗﺎﻥ ﻭ ﺍﻭ ﺭﺍ ﭘﻴﭽﻴﺪﻩ‌ﺗﺮ ﻧﮑﻨﻴﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﻓﻬﺮﺳﺘﯽ ﺟﺎﺩﻭﻳﯽ ﺍﺯ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﻣﻮﺭﺩ ﮐﻮﺩﮐﺎﻥ ﺑﻪ ﻣﺮﺣﻠﻪ ﺍﺟﺮﺍ ﮔﺬﺍﺷﺖ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﭼﻨﻴﻦ ﻓﻬﺮﺳﺘﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ، ﺯﻳﺮﺍ ﻫﺮ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﺷﺨﺼﯽ ﻭ ﺭﻭﺵ ﺯﻧﺪﮔﯽ ﺧﺎﺹ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺩﺍﺭﻧﺪ. ﺑﻌﻀﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﮐﻪ ﻧﺴﺒﺘﺎ ﺣﻮﺻﻠﻪ ﺑﻴﺸﺘﺮ ﻭ ﺍﻋﺼﺎﺏ ﺁﺭﺍﻡ‌ﺗﺮﯼ ﺩﺍﺭﻧﺪ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﮐﻤﺘﺮﯼ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻗﺎﺋﻞ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﺑﺮﻋﮑﺲ ﻣﻤﻨﻮﻋﻴﺖ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺩﺭ ﻧﻈﺮ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺍﻳﻦ ﺩﺳﺘﻪ ﻣﺴﻠﻤﺎ ﺍﮔﺮ ﻓﺮﺯﻧﺪﺷﺎﻥ ﮐﺎﺭﻫﺎﻳﯽ ﺭﺍ ﺑﮑﻨﺪ ﮐﻪ ﻓﺮﺯﻧﺪﺍﻥ ﺩﺳﺘﻪ ﺍﻭﻝ ﺍﺟﺎﺯﻩ ﺍﻧﺠﺎﻣﺶ ﺭﺍ ﺩﺍﺭﻧﺪ، ﺧﻴﻠﯽ ﻧﺎﺭﺍﺣﺖ ﺧﻮﺍﻫﻨﺪ ﺷﺪ.

ﺍﻏﻠﺐ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺩﺭ ﻣﻮﺭﺩ ﻣﺤﺪﻭﺩﻳﺘﯽ ﮐﻪ ﺑﺮﺍﯼ ﺍﻋﻤﺎﻝ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻗﺎﺋﻞ ﻣﯽ‌ﺷﻮﻧﺪ ﺁﮔﺎﻫﻨﺪ ﻭﻟﯽ ﺍﮐﺜﺮ ﻣﻮﺍﻗﻊ ﺩﺭ ﻣﻮﺭﺩ ﺍﺣﺴﺎﺳﺎﺕ ﺁﻧﻬﺎ ﻧﻤﯽ‌ﺩﺍﻧﻨﺪ ﭼﻪ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﮑﻨﻨﺪ.

ﺑﺎﻳﺪ ﺗﺎﺛﺮﺍﺕ ﺭﻭﺣﯽ ﮐﻮﺩﮎ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺖ
ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﯽ ﺳﺎﻟﻢ ﺑﺎﺷﻨﺪ ﻭ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻗﻮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ، ﺑﺎﻳﺪ ﺑﻪ ﺁﻧﻬﺎ ﺍﺟﺎﺯﻩ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺑﺪﻫﻴﻢ.  ﻣﺘﺎﺳﻔﺎﻧﻪ ﻭﺍﻗﻌﻴﺖ ﺑﺴﻴﺎﺭ ﻣﺘﻔﺎﻭﺕ ﺑﺎ ﺍﻳﻦ ﺍﺳﺖ ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺍﺟﺎﺯﻩ ﭼﻨﻴﻦ ﮐﺎﺭﯼ ﺭﺍ ﻧﻤﯽ‌ﺩﻫﻨﺪ.

ﻣﺜﺎﻟﯽ ﺍﺯ ﻣﻴﺎﻥ ﺻﺪﻫﺎ ﻣﺜﺎﻝ ﺑﻴﺎﻭﺭﻳﻢ. ﺭﻭﺯﯼ ﺩﺭ ﭘﺎﺭﮐﯽ ﺑﺎ ﺗﻌﺠﺐ ﻓﺮﺍﻭﺍﻥ ﻣﮑﺎﻟﻤﻪ ﺯﻳﺮ ﺭﺍ ﺷﻨﻴﺪﻡ. ﻣﺎﺩﺭﯼ ﺑﺎ ﺩﻭ ﻓﺮﺯﻧﺪﺵ، ﻳﮑﯽ ﭘﺴﺮﯼ ﺗﻘﺮﻳﺒﺎ ﺷﺶ ﺳﺎﻟﻪ ﻭ ﺩﻳﮕﺮﯼ ﺩﺧﺘﺮﯼ ﺣﺪﻭﺩﺍ ﭼﻬﺎﺭﺳﺎﻟﻪ، ﺩﺭ ﭘﺎﺭﮎ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﺮﺩﻧﺪ.  ﭘﺴﺮ، ﮐﻪ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﻴﺪ ﺧﻴﻠﯽ ﺍﺯ ﺩﺳﺖ ﺧﻮﺍﻫﺮﺵ ﻋﺼﺒﺎﻧﯽ ﺍﺳﺖ ﻣﯽ‌ﮔﻔﺖ «ﺍﻟﻴﺰ ﺍﺯ ﺗﻮ ﻣﺘﻨﻔﺮﻡ». ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﻣﺎﺩﺭﺵ ﮔﻔﺖ: «ﻭَﻧﺴﺎﻥ ﺑﮕﻮ ﺑﺒﻴﻨﻢ ﭼﻪ ﺍﺣﺴﺎﺳﯽ ﺩﺍﺭﯼ؟» ﻧﺨﻴﺮ. ﺍﻭ ﺑﻪ ﭘﺴﺮﺵ ﮔﻔﺖ: «ﺑﺒﻴﻦ ﻭَﻧﺴﺎﻥ ﺟﻮﻥ، ﺧﻮﺍﻫﺮ ﮐﻮﭼﻮﻟﻮﯼ ﺗﻮ ﺩﺧﺘﺮ ﺧﻴﻠﯽ ﺧﻮﺑﻴﻪ، ﺗﻮ ﺍﺯ ﺍﻭﻥ ﻣﺘﻨﻔﺮ ﻧﻴﺴﺘﯽ، ﺩﻭﺳﺘﺶ ﺩﺍﺭﯼ.» ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﻳﻦ ﺩﺭﻭﻍ ﺑﻮﺩ ﻭ ﻭَﻧﺴﺎﻥ ﻫﻢ ﺍﻳﻦ ﺭﺍ ﻣﯽ‌ﺩﺍﻧﺴﺖ. ﻣﺎﺩﺭ ﺩﺍﺷﺖ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺭﺍ ﺧﻼﻑ ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺑﻮﺩ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﺍﺩ. ﻃﺒﻴﻌﺘﺎ ﻣﺎﺩﺭ ﻭﺍﻗﻌﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺴﺖ ﮐﺎﺭﯼ ﺩﺭ ﻣﻮﺭﺩ ﻋﺼﺒﺎﻧﻴﺖ ﺍﻭ ﺍﺯ ﺧﻮﺍﻫﺮﺵ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. ﺗﻨﻬﺎ ﭼﻴﺰﯼ ﮐﻪ ﻣﺎﺩﺭ ﺑﺎ ﺍﻳﻦ ﻧﻮﻉ ﺑﺮﺧﻮﺭﺩ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﺁﻣﻮﺧﺖ ﺍﻳﻦ ﺑﻮﺩ ﮐﻪ ﺍﻭ ﺭﻭﺷﯽ ﻧﻴﺮﻧﮓ‌ﺁﻣﻴﺰ ﺩﺭ ﻣﻘﺎﺑﻞ ﺁﻥ‌ﭼﻪ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ ﺑﻪ ﮐﺎﺭ ﮔﻴﺮﺩ، ﺍﻭ ﺑﺎ ﺍﻳﻦ ﺭﻭﺵِ ﺑﺮﺧﻮﺭﺩ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﺁﻣﻮﺧﺖ ﮐﻪ ﮔﺮﺍﻳﺶ‌ﻫﺎﯼ ﺩﺭﻭﻧﯽ ﻭ ﻃﻐﻴﺎﻥ‌ﻫﺎﯼ ﺩﺭﻭﻧﯽ ﻧﺎ ﺁﮔﺎﻫﺎﻧﻪ‌ﺍﺵ ﺭﺍ ﮐﻪ ﺍﻋﻤﺎﻟﺶ ﻧﺎﺷﯽ ﺍﺯ ﺁﻥ‌ﻫﺎﺳﺖ، ﻣﻬﺎﺭ ﮐﻨﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻭ ﺩﺳﺖ ﺑﻪ ﮐﺎﺭﻫﺎﯼ ﻣﻮﺫﻳﺎﻧﻪ ﻣﯽ‌ﺯﻧﺪ: ﻭﻗﺘﯽ ﻣﺎﺩﺭﺵ ﻧﻴﺴﺖ، ﺧﻮﺍﻫﺮﺵ ﺭﺍ ﮐﺘﮏ ﻣﯽ‌ﺯﻧﺪ. ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺩﺭ ﻣﻮﺭﺩ ﺗﻤﺎﻡ ﺍﺣﺴﺎﺳﺎﺗﯽ ﮐﻪ ﻣﺎ ﺁﻧﻬﺎ ﺭﺍ ﻣﻨﻔﯽ ﺍﺭﺯﻳﺎﺑﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﺻﺤﺖ ﺩﺍﺭﺩ: ﻋﺼﺒﺎﻧﻴﺖ، ﺗﺮﺱ، ﺧﺠﺎﻟﺖ، ﺩﺭﺩ ﻭ ﻳﺎ ﻧﮕﺮﺍﻧﯽ.

ﺗﻨﻬﺎ ﺍﺣﺴﺎﺳﺎﺗﯽ ﮐﻪ ﻣﺎ ﺳﻌﯽ ﺩﺭ ﻣﻨﺤﺮﻑ ﮐﺮﺩﻧﺸﺎﻥ ﻧﻤﯽ‌ﮐﻨﻴﻢ ﺍﺣﺴﺎﺳﺎﺕ «ﻣﺜﺒﺖ» ﻋﺸﻖ ﻭ ﻣﺤﺒﺖ ﺍﺳﺖ. ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻫﻴﭻ ﻣﺎﺩﺭﯼ ﺭﺍ ﻧﺪﻳﺪﻩ‌ﺍﻡ ﮐﻪ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﺍﺯ ﮔﻔﺘﻦ ﺟﻤﻠﻪ ﺯﻳﺮ ﻣﻨﻊ ﮐﻨﺪ،«ﻣﺎﻣﺎﻥ، ﺩﻭﺱِﺕ ﺩﺍﺭﻡ!»

ﺑﺎﻳﺪ ﺍﺟﺎﺯﻩ ﺩﻫﻴﻢ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺗﻤﺎﻡ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺑﻴﺎﻥ ﮐﻨﻨﺪ، ﭼﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺏ ﻭ ﭼﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺑﺪ. ﻓﻘﻂ ﺑﺎ ﺑﻪ ﺯﺑﺎﻥ ﺁﻭﺭﺩﻥ ﻭ ﺑﻴﺮﻭﻥ ﺭﻳﺨﺘﻦ ﺗﻤﺎﻳﻼﺕ ﺑﺪ ﺍﺳﺖ ﮐﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺏ ﺟﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺎﺯ ﻣﯽ‌ﻳﺎﺑﻨﺪ، ﺍﮔﺮ ﺑﻪ ﮐﻮﺩﮐﺎﻥ ﺍﺟﺎﺯﻩ ﻧﺪﻫﻴﻢ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺿﺪﻳﺖ ﺧﻮﺩ ﺭﺍ ﺑﺮﻭﺯ ﺩﻫﻨﺪ، ﺟﺎﻳﯽ ﺑﺮﺍﯼ ﺍﺑﺮﺍﺯ ﻋﺸﻖ ﻭ ﻣﺤﺒﺖ ﻭﺟﻮﺩ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺳﺮﮐﻮﺏ ﺷﻮﻧﺪ، ﺳﻼﻣﺖ ﺭﻭﺡ ﺩﭼﺎﺭ ﺍﺷﮑﺎﻝ ﻣﯽ‌ﺷﻮﺩ. ﻣﻌﻤﻮﻻ، ﮐﻮﺩﮐﺎﻥ ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﺎ ﺍﺯ ﺁﻧﻬﺎ ﻧﺨﻮﺍﺳﺘﻪ‌ﺍﻳﻢ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻨﺪ ﭘﻨﻬﺎﻥ ﮐﻨﻨﺪ، ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺑﻴﺎﻥ ﻣﯽ‌ﮐﻨﻨﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺻﺒﺢ ﺩﻭ ﮐﻮﺩﮎ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﮐﺘﮏ‌ﮐﺎﺭﯼ ﮐﻨﻨﺪ ﻭ ﺑﻌﺪ ﺍﺯ ﻇﻬﺮ ﺑﻬﺘﺮﻳﻦ ﺩﻭﺳﺖ‌ﻫﺎﯼ ﺩﻧﻴﺎ ﺑﺎﺷﻨﺪ. ﺑﺮﻋﮑﺲ، ﻭﺍﻟﺪﻳﻦ ﻫﻤﻴﻦ ﺑﭽﻪ‌ﻫﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﻣﺎﻩ‌ﻫﺎ ﺭﻭﺍﺑﻄﺸﺎﻥ ﺑﻪ ﻋﻠﺖ ﺍﻳﻦ ﺩﻋﻮﺍ ﺗﻴﺮﻩ ﺑﺎﻗﯽ ﺑﻤﺎﻧﺪ.

ﺍﻓﺮﺍﺩ ﺑﺎﻟﻐﯽ ﮐﻪ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻣﺸﮑﻼﺗﯽ ﺩﺍﺭﻧﺪ، ﺑﻪ ﮐﻤﮏ ﺭﻭﺍﻧﺸﻨﺎﺱ، ﻗﺒﻞ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﻃﺒﻴﻌﯽ ﺑﻴﺎﻥ ﮐﻨﻨﺪ. ﻭﻟﯽ ﺍﮔﺮ ﻣﺎ ﺍﺯ ﻫﻤﺎﻥ ﺳﻨﻴﻦ ﭘﺎﻳﻴﻦ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺑﻴﺎﻣﻮﺯﻳﻢ، ﺁﻧﻬﺎ، ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻐﯽ ﺷﻮﻧﺪ ﺍﺯ ﺁﻥ ﺑﻬﺮﻩ ﺧﻮﺍﻫﻨﺪ ﺟﺴﺖ.

ﮐﻮﺩﮐﺎﻥ ﺧﺮﺩﺳﺎﻝ ﻗﺎﺩﺭ ﺑﻪ ﺳﺮﮐﻮﺏ ﮐﺮﺩﻥِ ﺗﻨﻬﺎ ﺑﺨﺸﯽ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﭼﻮﻥ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺗﺮﺱ ﻧﻴﺴﺘﻨﺪ، ﺩﺭ ﻭﺍﻗﻊ ﺑﺎ ﭼﻨﻴﻦ ﮐﺎﺭﯼ ﺁﻧﻬﺎ ﺗﻤﺎﻡ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺳﺮﮐﻮﺏ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﮔﺮ ﺯﻣﺎﻧﯽ ﮐﻪ ﻫﻨﻮﺯ ﮐﻮﭼﮏ ﻫﺴﺘﻨﺪ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺑﻴﺎﻣﻮﺯﻳﻢ ﮐﻪ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺗﺮﺳﺸﺎﻥ ﺭﺍ ﺳﺮﮐﻮﺏ ﮐﻨﻨﺪ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺑﻪ ﺁﻧﻬﺎ ﺁﻣﻮﺧﺘﻪ‌ﺍﻳﻢ ﮐﻪ ﺍﺣﺴﺎﺳﺎﺕ «ﺧﻮﺏ» ﻭ ﻣﺜﺒﺖ ﻋﺸﻖ ﻭ ﻣﺤﺒﺖ ﺭﺍ ﻧﻴﺰ ﺳﺮﮐﻮﺏ ﮐﻨﻨﺪ.

ﺍﻣﺎ ﺍﻳﻦ‌ﮐﻪ ﮐﻮﺩﮎ ﻓﻘﻂ ﻗﺎﺩﺭ ﺑﻪ ﺑﻴﺎﻥ ﺁﻥ ﭼﻴﺰﯼ ﺑﺎﺷﺪ ﮐﻪ ﺩﺭ ﺍﻭ ﻣﯽ‌ﮔﺬﺭﺩ ﮐﺎﻓﯽ ﻧﻴﺴﺖ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻧﻴﺰ ﺑﺎﻳﺪ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﺩﺭﮎ ﮐﻨﻨﺪ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﻮﺩﮐﺎﻥ ﻧﺎﺭﺍﺣﺖ، ﻋﺼﺒﺎﻧﯽ ﻭ ﺭﻧﺠﻴﺪﻩ ﺧﺎﻃﺮ ﻫﺴﺘﻨﺪ ﺩﻟﺸﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻭﺍﻟﺪﻳﻨﺸﺎﻥ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭﮎ ﮐﻨﻨﺪ. ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻧﺸﺎﻥ ﺩﻫﻴﺪ ﮐﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺭﺍ ﺩﺭﮎ ﻣﯽ‌ﮐﻨﻴﺪ؟ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻃﻮﺭ ﺳﻄﺤﯽ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﻣﯽ‌ﺩﻭﻧﻢ ﭼﻪ ﺍﺣﺴﺎﺳﯽ ﺩﺍﺭﯼ، ﻭﻟﯽ ﺍﻳﻦ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﻭ ﺳﺮﻳﻊ ﺍﺳﺖ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻭ ﺭﺍ ﻣﺘﻘﺎﻋﺪ ﻧﮑﻨﺪ. ﺭﻭﺵ ﺩﻳﮕﺮﯼ ﺑﺮﺍﯼ ﻓﻬﻤﺎﻧﺪﻥ ﺍﻳﻨﮑﻪ ﺍﻭ ﺭﺍ ﻋﻤﻴﻘﺎ ﺩﺭﮎ ﻣﯽ‌ﮐﻨﻴﺪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻳﻦ ﺭﻭﺵ ﺍﻧﻌﮑﺎﺱ ﺍﺣﺴﺎﺳﺎﺕ ﻧﺎﻡ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﺑﻪ ﺁﻥ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ.
