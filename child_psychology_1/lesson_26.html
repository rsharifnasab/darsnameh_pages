ﺩﺭﺱ ﺑﻴﺴﺖ ﻭ ﻳﮑﻢ- ﺩﺧﺘﺮ ﻳﺎ ﭘﺴﺮ

ﺩﺭ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﺍﺳﺖ ﮐﻪ ﮐﻮﺩﮎ ﻣﯽ­‌ﻓﻬﻤﺪ ﮐﻪ ﻣﺘﻌﻠﻖ ﺑﻪ ﭼﻪ ﺟﻨﺴﯽ ﺍﺳﺖ. ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﭘﻴﺶ ﻧﻴﺰ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ ﺗﻘﺮﻳﺒﺎ ﺗﺎ ﺳﻦ ﺳﻪ ﺳﺎﻟﮕﯽ، ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﻫﺮ ﺩﻭ، ﺍﮔﺮ ﺑﺸﻮﺩ ﮔﻔﺖ، ﺑﺪﻭﻥ ﺟﻨﺲ ﻫﺴﺘﻨﺪ. ﺁﻧﻬﺎ ﺑﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﻳﮑﺴﺎﻧﯽ ﺑﺎﺯﯼ ﻣﯽ­‌ﮐﻨﻨﺪ ﻭ ﺍﺯ ﭼﻴﺰﻫﺎﯼ ﻣﺸﺎﺑﻬﯽ ﺧﻮﺷﺸﺎﻥ ﻣﯽ‌ﺁﻳﺪ. ﻣﺜﻼ ﭘﺴﺮﻫﺎﯼ ﺧﺮﺩﺳﺎﻝ ﻣﺎﻧﻨﺪ ﺩﺧﺘﺮﻫﺎﯼ ﮐﻮﭼﮏ ﺩﺭ ﺍﻭﻟﻴﻦ ﺳﺎﻝ‌­ﻫﺎﯼ ﺯﻧﺪﮔﯽ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺑﺎ ﻋﺮﻭﺳﮏ ﻭ ﻳﺎ ﺣﻴﻮﺍﻧﺎﺕ ﭘﺮﺯﺩﺍﺭ ﻭ ﭘﺎﺭﭼﻪ‌ﺍﯼ ﺑﺎﺯﯼ ﮐﻨﻨﺪ.

ﻣﺴﻠﻤﺎ، ﺣﺘﯽ ﺩﺭ ﺍﻳﻦ ﺍﻭﻟﻴﻦ ﺳﺎﻝ‌ﻫﺎﯼ ﺯﻧﺪﮔﯽ ﻧﻴﺰ ﺗﻔﺎﻭﺕ‌ﻫﺎﻳﯽ ﺑﻴﻦ ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻣﻌﻤﻮﻻ ﮔﺮﺍﻳﺶ ﺑﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺧﺘﺮﻫﺎ ﭘﺨﺘﮕﯽ ﺯﻭﺩﺗﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﭘﺴﺮﻫﺎ ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺯﻣﻴﻨﻪ‌ﻫﺎ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﺩﻫﻨﺪ(ﻣﺜﻼ ﺻﺤﺒﺖ ﮐﺮﺩﻥ). ﭘﺴﺮﻫﺎ ﻣﻌﻤﻮﻻ ﺍﺯ ﻧﻈﺮ ﺟﺴﻤﯽ ﻓﻌﺎﻝ­‌ﺗﺮﻧﺪ ﻭ ﺍﺯ ﺩﺧﺘﺮﻫﺎ ﭘﺮﺧﺎﺷﮕﺮﺗﺮﻧﺪ. ﻭﻟﯽ ﺩﺭ ﮐﻞ ﻭ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﺸﺨﺺ، ﺩﺭ ﺳﻪ ﺳﺎﻟﮕﯽ ﺗﻔﺎﻭﺕ ﺟﻨﺲ ﺑﺎﺭﺯ ﻣﯽ‌ﺷﻮﺩ. ﺍﺯ ﺍﻳﻦ ﺳﻦ ﺑﻪ ﺑﻌﺪ، ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﻫﺮ ﮐﺪﺍﻡ ﺭﻓﺘﺎﺭﯼ ﻣﺘﻔﺎﻭﺕ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﻣﯽ­‌ﺩﻫﻨﺪ ﻭ ﺩﻳﺪﯼ ﻣﺨﺘﻠﻒ ﺍﺯ ﺧﻮﺩ ﻭ ﺩﻧﻴﺎ ﭘﻴﺪﺍ ﻣﯽ‌­ﮐﻨﻨﺪ.

ﭘﺴﺮﻫﺎ ﺑﻴﺸﺘﺮ ﺩﻋﻮﺍ ﺭﺍﻩ ﻣﯽ‌­ﺍﻧﺪﺍﺯﻧﺪ، ﺑﻴﺸﺘﺮ ﺳﺮ ﻭ ﺻﺪﺍ ﻣﯽ‌­ﮐﻨﻨﺪ، ﺭﻳﺴﮏ ﺑﻴﺸﺘﺮﯼ ﻣﯽ­‌ﮐﻨﻨﺪ، ﻓﮑﺮﺷﺎﻥ ﻣﺴﺘﻘﻞ­‌ﺗﺮ ﺍﺳﺖ ﻭ ﺑﺮﺍﯼ ﺗﺮﺑﻴﺖ ﺍﺯ ﺩﺧﺘﺮﻫﺎ ﻣﺸﮑﻞ‌ﺗﺮ ﻫﺴﺘﻨﺪ ﻭ ﺻﺪﻣﻪ ﭘﺬﻳﺮﺗﺮ ﺍﺯ ﺩﺧﺘﺮﻫﺎ ﻫﺴﺘﻨﺪ. ﺩﺭ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ، ﻫﻤﻴﻨﻄﻮﺭ ﺳﺎﻝ‌ﻫﺎﯼ ﺑﻌﺪ، ﻣﻴﺰﺍﻥ ﻣﺮﮒ ﻭ ﻣﻴﺮ ﭘﺴﺮﻫﺎ ﺑﻴﺸﺘﺮ ﺍﺯ ﺩﺧﺘﺮ‌ﻫﺎ ﺍﺳﺖ. ﺍﻣﮑﺎﻥ ﮔﺮﻓﺘﻦ ﻟﮑﻨﺖ ﺯﺑﺎﻥ، ﺩﺍﺷﺘﻦ ﻣﺸﮑﻼﺗﯽ ﺑﺮﺍﯼ ﺧﻮﺍﻧﺪﻥ ﻭ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﻋﺎﻃﻔﯽ ﺩﺭ ﭘﺴﺮﻫﺎ ﺑﻴﺶ ﺍﺯ ﺩﺧﺘﺮﺍﻥ ﺍﺳﺖ. ﺍﺯ ﻧﻈﺮ ﺭﺷﺪ ﺟﺴﻤﺎﻧﯽ، ﭘﺴﺮﻫﺎ ﺗﻘﺮﻳﺒﺎ ﻳﮑﺴﺎﻝ ﺍﺯ ﺩﺧﺘﺮﻫﺎ ﻋﻘﺐ‌ﺗﺮﻧﺪ. ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﺪﺭﺳﻪ ﺭﻓﺘﻦ ﺭﺍ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﻨﺪ، ﺣﺘﯽ ﻋﻀﻼﺕ ﺩﺳﺘﺸﺎﻥ ﺭﺷﺪ ﮐﻤﺘﺮﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﺍﺳﺖ.  ‌


ﺩﺧﺘﺮﻫﺎ ﺑﺮﻋﮑﺲ ﻣﺤﮑﻢ‌­ﺗﺮ ﻭ ﭘﺨﺘﻪ­‌ﺗﺮ ﻫﺴﺘﻨﺪ ﻭﻟﯽ ﺧﻴﻠﯽ ﻣﻄﻴﻊ­‌ﺗﺮ ﻭ ﺻﻠﺢ‌ﻃﻠﺐ‌­ﺗﺮﻧﺪ، ﺧﻮﺩ ﺭﺍ ﺑﻬﺘﺮ ﺑﺎ ﻗﻮﺍﻧﻴﻦ ﻭ ﺳﻨﻦ ﺗﻄﺒﻴﻖ ﻣﯽ‌­ﺩﻫﻨﺪ، ﺑﻴﺸﺘﺮ ﻓﺮﻣﺎﻧﺒﺮﺩﺍﺭﻧﺪ ﻭ ﮔﻮﺷﻪ‌ﻧﺸﻴﻦ­‌ﺗﺮﻧﺪ. ﺁﻧﻬﺎ ﺑﻪ ﺍﻓﺮﺍﺩ ﺑﻴﺸﺘﺮ ﺍﺯ ﺍﺷﻴﺎﺀ ﺗﻮﺟﻪ ﺩﺍﺭﻧﺪ، ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻋﻼﻗﻪ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﻧﺪ ﻭ ﻧﺴﺒﺖ ﺑﻪ ﻋﮑﺲ‌­ﺍﻟﻌﻤﻞ ﺁﻧﻬﺎ ﺣﺴﺎﺱ‌­ﺗﺮﻧﺪ. ﻗﺎﺑﻠﻴﺖ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﻳﺎﺩﮔﻴﺮﯼ ﻧﺎﻡ ﺍﺷﺨﺎﺹ ﻭ ﻣﺤﻞ­‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﺍﺭﻧﺪ.

ﺍﺯ ﻧﻈﺮ ﻋﻠﻤﯽ ﻫﻴﭻ ﺗﻔﺎﻭﺗﯽ ﺩﺭ ﻣﻮﺭﺩ ﺿﺮﻳﺐ ﻫﻮﺷﯽ ﺑﻴﻦ ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﺩﺭ ﺩﻭﺭﺍﻥ ﮐﻮﺩﮐﯽ ﻧﻤﯽ­‌ﺗﻮﺍﻥ ﻣﺸﺎﻫﺪﻩ ﮐﺮﺩ، ﺑﺎ ﺍﻳﻦ‌ﻫﻤﻪ ﻃﺮﺯ ﻓﮑﺮ ﻭ ﭼﮕﻮﻧﮕﯽ ﻓﮑﺮ ﻭ ﺩﺭﮎ ﺫﻫﻨﯽ ﺁﻧﻬﺎ ﺑﺴﻴﺎﺭ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺩﺧﺘﺮﻫﺎ ﺍﺯ ﻧﻈﺮ ﺷﻔﺎﻫﯽ ﺧﻴﻠﯽ ﺍﺯ ﭘﺴﺮﻫﺎ ﺟﻠﻮﺗﺮﻧﺪ. ﺁﻧﻬﺎ ﺯﻭﺩﺗﺮ ﺍﺯ ﭘﺴﺮﻫﺎ ﺻﺤﺒﺖ ﻣﯽ‌­ﮐﻨﻨﺪ ﻭ ﺑﻌﺪﻫﺎ ﻏﻠﻂ ﺩﻳﮑﺘﻪ ﮐﻤﺘﺮﯼ ﺩﺍﺭﻧﺪ ﻭ ﺑﻴﺸﺘﺮ ﻣﯽ­‌ﻧﻮﻳﺴﻨﺪ. ﭘﺴﺮﻫﺎ ﺩﺭ ﺯﻣﻴﻨﻪ‌ﻫﺎﯼ ﺍﻧﺘﺰﺍﻋﯽ، ﺍﺯ ﺟﻤﻠﻪ ﺭﻳﺎﺿﻴﺎﺕ ﻭ ﻋﻠﻮﻡ ﺍﺯ ﺁﻧﻬﺎ ﺟﻠﻮ ﻣﯽ­‌ﺯﻧﻨﺪ. ﮔﺮﺍﻳﺶ ﭘﺴﺮﻫﺎ ﺑﺮﺍﯼ ﺍﺑﺘﮑﺎﺭ ﺑﻴﺸﺘﺮ ﺍﺳﺖ.

ﭼﮕﻮﻧﻪ ﻣﯽ­‌ﺗﻮﺍﻥ ﺍﻳﻦ ﺍﺧﺘﻼﻓﺎﺕ ﻋﻤﻴﻖ ﺭﺍ ﮐﻪ ﺍﻳﻦ ﻗﺪﺭ ﺯﻭﺩ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﺗﻮﺿﻴﺢ ﺩﺍﺩ؟ ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺧﻴﻠﯽ ﺯﻳﺎﺩ، ﺍﻳﻦ ﺍﺧﺘﻼﻓﺎﺕ ﺍﺯ ﻃﺮﻓﯽ ﻧﺎﺷﯽ ﺍﺯ ﻋﻮﺍﻣﻞ ﻫﻮﺭﻣﻮﻧﯽ ﻭ ﮊﻧﺘﻴﮏ ﻭ ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ ﻧﺎﺷﯽ ﺍﺯ ﺭﻭﺵ ﺗﺮﺑﻴﺘﯽ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻩ ﺩﺭ ﻣﻮﺭﺩ ﮐﻮﺩﮐﺎﻥ ﺍﺳﺖ. ﺑﻪ ﻧﻈﺮ ﻣﺘﺨﺼﺼﻴﻦ، ﻋﺎﻣﻞ ﺗﺮﺑﻴﺘﯽ ﻭ ﻓﺮﻫﻨﮕﯽ ﺍﻫﻤﻴﺖ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﺩ ﺗﺎ ﻋﻮﺍﻣﻞ ﺑﻴﻮﻟﻮﮊﻳﮑﯽ.

ﺑﺎ ﺍﻳﻨﻬﻤﻪ، ﻣﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ، ﺑﺎﻳﺪ ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﻢ ﮐﻪ ﺩﺧﺘﺮﺍﻥ ﻭ ﭘﺴﺮﺍﻧﻤﺎﻥ ﻣﻮﺍﻓﻖ ﺑﺎ ﺟﻨﺴﯽ ﺑﺎﺷﻨﺪ ﮐﻪ ﺑﻪ ﺁﻥ ﺗﻌﻠﻖ ﺩﺍﺭﻧﺪ. ﺍﻳﻦ ﮐﻪ ﮐﻮﺩﮎ، ﺧﻮﺩ ﺭﺍ ﻧﻤﺎﻳﻨﺪﻩ ﮐﺎﻣﻠﯽ ﺍﺯ ﺟﻨﺴﯽ ﮐﻪ ﺑﻪ ﺁﻥ ﺗﻌﻠﻖ ﺩﺍﺭﺩ ﺑﺪﺍﻧﺪ ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻭ ﺳﻼﻣﺖ ﺫﻫﻨﯽ ﺍﻭ ﺩﺍﺭﺩ.

ﺑﺎﻳﺪ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﮐﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ، ﭼﻪ ﺩﺧﺘﺮ ﻭ ﭼﻪ ﭘﺴﺮ، ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺑﻴﺸﺘﺮ ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻣﺎﺩﺭﺵ ﻣﯽ­‌ﮔﺬﺭﺍﻧﺪ، ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺍﻭ ﻫﻢ‌­ﺫﺍﺕ ﻣﻲ­ﭘﻨﺪﺍﺭﺩ. ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﭘﺴﺮ ﺳﻪ ﺳﺎﻟﻪ‌ﺍﯼ ﺑﮕﻮﻳﺪ ﮐﻪ«ﻭﻗﺘﯽ ﺑﺰﺭﮒ ﺷﺪﻡ، ﻣﺜﻞ ﺗﻮ، ﻣﺎﻣﺎﻥ ﻣﯽ‌­ﺷﻢ!» ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺴﺮﺍﻥ ﻛﻔﺶ ﭘﺎﺷﻨﻪ ﺑﻠﻨﺪ ﻣﺎﺩﺭﺷﺎﻥ ﺭﺍ ﻣﯽ‌ﭘﻮﺷﻨﺪ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺟﻮﺭﺍﺏ ﭘﺪﺭﺷﺎﻥ ﺭﺍ ﻣﯽ‌ﭘﻮﺷﻨﺪ، ﻳﺎ ﻣﯽ­‌ﺧﻮﺍﻫﻨﺪ ﺍﺯ ﻣﺎﺗﻴﮏ ﻳﺎ ﻻﻙ ﻣﺎﺩﺭﺷﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪ.

ﺑﺎ ﺍﻳﻨﻬﻤﻪ، ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ، ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﻫﺮ ﮐﺪﺍﻡ، ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺭﺍﻩ ﺭﺷﺪ ﺟﺪﺍﮔﺎﻧﻪ‌ﺍﯼ ﺭﺍ ﺑﭙﻴﻤﺎﻳﻨﺪ. ﻣﺴﻠﻤﺎ، ﮐﻮﺩﮎ ﺩﺭ ﺳﻪ ﺳﺎﻟﮕﯽ ﺍﺯ ﻧﻈﺮ ﺫﻫﻨﯽ ﭘﺨﺘﮕﯽ ﻻﺯﻡ ﺑﺮﺍﯼ ﺩﺭﮎ ﺗﻌﻠﻘﺶ ﺑﻪ ﺍﻳﻦ ﻳﺎ ﺁﻥ ﺟﻨﺲ ﺭﺍ ﺩﺍﺭﺩ. ﻭﻟﯽ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻮﺩﮐﯽ ﮐﻪ ﺩﺭ ﺳﻨﻴﻦ ﻗﺒﻞ ﻣﺪﺭﺳﻪ ﺍﺳﺖ ﻫﻨﻮﺯ ﭼﻴﺰﯼ ﺍﺯ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺩﺭﮎ ﻧﮑﻨﺪ. ﭘﺴﺮﯼ ﭘﻨﺞ ﺳﺎﻟﻪ ﮐﻪ ﻧﻮﺯﺍﺩﯼ ﻟﺨﺖ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺩﺍﺩﻧﺪ ﻭ ﺍﺯ ﺍﻭ ﭘﺮﺳﻴﺪﻧﺪ ﮐﻪ ﺩﺧﺘﺮ ﺍﺳﺖ ﻳﺎ ﭘﺴﺮ ﺟﻮﺍﺏ ﺩﺍﺩ: «ﭼﻪ ﻣﯽ‌­ﺩﻭﻧﻢ، ﺑﺪﻭﻥ ﻟﺒﺎﺱ ﮐﻪ ﻣﻌﻠﻮﻡ ﻧﻤﯽ­‌ﺷﻪ.»

ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ، ﮐﻮﺩﮐﺎﻥ ﺑﻴﻦ ﺳﻪ ﺗﺎ ﺷﺶ ﺳﺎﻝ، ﺑﻪ ﻃﺮﻳﻘﯽ ﮐﺸﻒ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﭘﺴﺮﻫﺎ ﺁﻟﺖ ﻣﺮﺩﯼ ﺩﺍﺭﻧﺪ ﻭ ﺩﺧﺘﺮﻫﺎ ﻧﺪﺍﺭﻧﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺍﺯ ﺗﺄﺛﻴﺮ ﻣﻬﻢ ﺍﻳﻦ ﮐﺸﻒ ﺩﺭ ﺯﻧﺪﮔﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﻮﺩﮐﺎﻥ ﺑﯽ­‌ﺧﺒﺮﻧﺪ ﻭ ﻣﺘﻮﺟﻪ ﺍﻫﻤﻴﺘﯽ ﮐﻪ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺑﺮﺍﯼ ﮐﻮﺩﮐﺎﻥ ﺩﺍﺭﺩ ﻧﻤﯽ­‌ﺷﻮﻧﺪ. ﻣﺴﻠﻤﺎ ﺍﮔﺮ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺨﻮﺍﻫﻨﺪ ﭼﻴﺰﯼ ﺑﻪ ﺭﻭﯼ ﺧﻮﺩ ﻧﻴﺎﻭﺭﻧﺪ ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺳﺖ ﮐﻪ ﺁﻥ ﺭﺍ ﻣﺮﺗﺒﻂ ﺑﺎ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﻣﯽ­‌ﺩﺍﻧﻨﺪ ، ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺗﺤﺖ ﺗﺄﺛﻴﺮ ﺍﺣﺴﺎﺳﺎﺕ ﺳﺮﮐﻮﺏ ﺷﺪﻩ‌­ﺍﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﻮﺟﺐ ﻣﯽ‌­ﺷﻮﺩ ﺁﻧﻬﺎ ﻧﺒﻴﻨﻨﺪ ﮐﻪ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﭼﻪ ﺍﻫﻤﻴﺘﯽ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺩﺍﺭﺩ.

ﻫﺮ ﭼﻘﺪﺭ ﻣﺤﻴﻂ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺩﺭ ﻣﻮﺭﺩ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﺳﺎﻟﻢ‌ﺗﺮ ﺑﺎﺷﺪ، ﻫﻤﺎﻧﻘﺪﺭ ﺷﻮﮎ ﺭﻭﺣﯽ‌­ﺍﯼ ﮐﻪ ﺍﺯ ﺍﻳﻦ«ﮐﺸﻒ» ﺑﻪ ﺩﺧﺘﺮ ﻳﺎ ﭘﺴﺮ ﮐﻮﭼﮏ ﻭﺍﺭﺩ ﻣﯽ­‌ﺷﻮﺩ ﮐﻤﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻫﺮ ﭼﻘﺪﺭ ﻣﺤﻴﻂ ﺩﺭ ﻣﻮﺭﺩ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﺳﺮﮐﻮﺏ ﺷﺪﻩ ﺑﺎﺷﺪ ﻭ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﺑﻪ ﻣﺎﻧﻨﺪ ﮔﻨﺎﻫﯽ ﺗﻠﻘﯽ ﺷﻮﺩ، ﻫﻤﺎﻥ ﻗﺪﺭ ﮐﻮﺩﮎ ﺑﺮﺍﯼ ﻫﻀﻢ ﺍﻳﻦ «ﺷﻮﮎ» ﺑﺎ ﺍﺷﮑﺎﻻﺕ ﻣﻮﺍﺟﻪ ﻣﯽ‌­ﺷﻮﺩ. ﻫﺮ ﮐﻮﺩﮐﯽ، ﻣﺴﺘﻘﻞ ﺍﺯ ﻣﺤﻴﻂ ﺧﺎﻧﻮﺍﺩﮔﻴﺶ، ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﻣﺘﻔﺎﻭﺕ، ﺷﺨﺼﯽ ﻭ ﺧﺎﺹ ﺧﻮﺩﺵ ﻋﮑﺲ‌­ﺍﻟﻌﻤﻞ ﻧﺸﺎﻥ ﻣﯽ‌­ﺩﻫﺪ. ﺑﺎ ﺍﻳﻨﻬﻤﻪ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ­‌ﻫﺎﯼ ﻧﻮﻋﯽ ﮐﻪ ﺩﺧﺘﺮﻫﺎ ﻭ ﭘﺴﺮﻫﺎﯼ ﮐﻮﭼﮏ ﭘﺲ ﺍﺯ ﮐﺸﻒ ﺍﻳﻨﮑﻪ ﭘﺴﺮﻫﺎ ﺩﺍﺭﺍﯼ ﺁﻟﺖ ﻣﺮﺩﯼ ﻫﺴﺘﻨﺪ ﺩﺭ ﺻﻮﺭﺗﻴﮑﻪ ﺩﺧﺘﺮﻫﺎ ﻓﺎﻗﺪ ﺁﻧﻨﺪ، ﺗﻮﺿﻴﺢ ﺩﻫﻴﻢ.

ﻣﺴﺌﻠﻪ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺩﺧﺘﺮﻫﺎ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ: ﺩﺧﺘﺮﻫﺎﯼ ﮐﻮﭼﮏ ﮐﻤﯽ ﺁﺯﺭﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺑﻪ ﭘﺴﺮﻫﺎ ﺣﺴﻮﺩﯼ ﻣﯽ‌­ﮐﻨﻨﺪ. ﻭﻟﯽ ﺍﻳﻦ ﺑﺪﻳﻦ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﺣﺘﻤﺎ ﺷﻮﮐﯽ ﺑﻪ ﺁﻧﻬﺎ ﻭﺍﺭﺩ ﻣﯽ‌ﺷﻮﺩ. ﻣﻌﻤﻮﻻ، ﺍﮔﺮ ﺩﺧﺘﺮ ﮐﻮﭼﮏ ﺧﻮﺩ ﺍﻧﮕﺎﺭﻩ ﻣﺘﻌﺎﺩﻟﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻗﺒﻮﻝ ﻣﯽ‌­ﮐﻨﺪ ﮐﻪ ﺩﺧﺘﺮﻫﺎ ﺑﺎ ﭘﺴﺮﻫﺎ ﻓﺮﻕ ﺩﺍﺭﻧﺪ. ﺩﺭﮎ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﮐﻪ ﺩﺧﺘﺮﻫﺎ ﻫﻢ ﺧﻴﻠﯽ ﭼﻴﺰﻫﺎﯼ ﺩﻳﮕﺮﯼ ﺩﺍﺭﻧﺪ ﮐﻪ ﮐﻤﺒﻮﺩ ﺍﻳﻦ ﺯﺍﺋﺪﻩ ﺭﺍ ﺟﺒﺮﺍﻥ ﻣﯽ­‌ﮐﻨﺪ.

ﻣﺎﺩﺭ ﺑﺎ ﺗﻮﺿﻴﺢ ﺍﻳﻨﮑﻪ ﺍﮔﺮ ﭘﺴﺮﻫﺎ ﺩﺍﺭﺍﯼ ﺁﻟﺖ ﻣﺮﺩﯼ ﻫﺴﺘﻨﺪ، ﺩﺧﺘﺮﻫﺎ ﻫﻢ ﺩﺭ ﻋﻮﺽ ﭼﻴﺰﯼ ﺩﺍﺭﻧﺪ ﮐﻪ ﭘﺴﺮﻫﺎ ﻧﺪﺍﺭﻧﺪ، ﻳﻌﻨﯽ ﺭﺣﻢ، ﻣﯽ­‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﮐﺴﺐ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﺑﮑﻨﺪ. ﻣﺎﺩﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺩﺧﺘﺮﺵ ﺗﻮﺿﻴﺢ ﺩﻫﺪ ﮐﻪ ﺭﺣﻢ ﭼﻴﺰﯼ ﻣﺜﻞ ﻳﮏ ﮐﻴﺴﻪ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺩﺍﺧﻞ ﺷﮑﻢ ﺩﺧﺘﺮﻫﺎ ﻭ ﺯﻥ‌ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﻭ ﺩﺭ ﺍﻳﻦ ﮐﻴﺴﻪ ﺍﺳﺖ ﮐﻪ ﺑﭽﻪ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﺩ، ﻫﻤﻴﻨﻄﻮﺭ ﻣﯽ­‌ﺗﻮﺍﻧﺪ ﺗﻮﺿﻴﺢ ﺩﻫﺪ ﮐﻪ ﭘﺴﺮﻫﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﭽﻪ ﺩﺍﺭ ﺷﻮﻧﺪ ﺩﺭ ﺻﻮﺭﺗﻴﮑﻪ ﺯﻥ­ﻫﺎ ﻣﯽ­‌ﺗﻮﺍﻧﻨﺪ.

ﭘﺴﺮﻫﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﺸﻒ ﺍﻳﻨﮑﻪ ﺯﻥ­‌ﻫﺎ ﻭ ﺩﺧﺘﺮﻫﺎ ﺁﻟﺖ ﻣﺮﺩﯼ ﻧﺪﺍﺭﻧﺪ، ﺩﭼﺎﺭ ﺗﺮﺱ ﻋﻤﻴﻘﯽ ﺑﺸﻮﻧﺪ. ﻳﮏ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻝ، ﺑﺎ ﺗﺠﺰﻳﻪ ﻭ ﺗﺤﻠﻴﻞ ﺍﺑﺘﺪﺍﻳﯽ ﻭ ﺩﺭ ﺳﻄﺢ ﺧﻮﺩﺵ، ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﮑﺮ ﮐﻨﺪ ﮐﻪ ﺁﻟﺖ ﻣﺮﺩﯼ ﺯﻥ‌ﻫﺎ ﻭ ﺩﺧﺘﺮﻫﺎ ﺭﺍ ﺑﺮﻳﺪﻩ‌­ﺍﻧﺪ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺑﻼ ﺭﻭﺯﯼ ﺑﻪ ﺳﺮ ﺍﻭ ﻧﻴﺰ ﺑﻴﺎﻳﺪ.

ﺩﺭ ﺧﺎﻧﻪ‌ﺍﯼ ﮐﻪ ﺟﻮ ﺩﺭ ﻣﻮﺭﺩ ﺳﻮﺍﻝ‌ﻫﺎﯼ ﺟﻨﺴﯽ ﺳﺎﻟﻢ ﺑﺎﺷﺪ ﻭ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺩﺭ ﻣﻮﺭﺩ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ ﺧﻴﻠﯽ ﺭﻭﺷﻦ، ﭘﺴﺮ ﮐﻮﭼﮑﯽ ﮐﻪ ﻧﮕﺮﺍﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺒﺎﺩﺍ ﺑﻼﻳﯽ ﺑﺮ ﺳﺮ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ­‌ﺍﺵ ﺑﻴﺎﻳﺪ، ﺧﻴﻠﯽ ﺯﻭﺩ ﻧﮕﺮﺍﻧﯽ‌­ﺍﺵ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺍﺯ ﻣﻴﺎﻥ ﻣﯽ­‌ﺭﻭﺩ. ﻣﻌﻤﻮﻻ ﭘﺴﺮﻫﺎ ﺧﻴﻠﯽ ﻣﻐﺮﻭﺭ ﻫﺴﺘﻨﺪ ﻭ ﺩﻟﺸﺎﻥ ﻣﯽ­‌ﺧﻮﺍﻫﺪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺩﻭﺳﺘﺎﻥ ﻭ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺷﺎﻥ ﻧﺸﺎﻥ ﺩﻫﻨﺪ. ﺍﮔﺮ ﻣﺎﺩﺭ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺑﺮﺧﻮﺭﺩ ﺭﺍﺣﺘﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺍﺯ ﺳﺎﺩﻩ ﻟﻮﺣﯽ ﻓﺮﺯﻧﺪﺵ ﺩﺭ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻣﺮﺩﺍﻧﮕﯽ‌­ﺍﺵ، ﺍﺣﺘﻤﺎﻻ ﺗﻔﺮﻳﺢ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﻣﺎﺩﺭ ﭼﻪ ﮐﺎﺭﯼ ﺑﺮﺍﯼ ﮐﻤﮏ ﺑﻪ ﻃﻔﻞ ﺟﻬﺖ ﺗﺄﮐﻴﺪ ﺗﻌﻠﻖ ﺍﻭ ﺑﻪ ﺟﻨﺲ ﻣﻮﻧﺚ ﻳﺎ ﻣﺬﮐﺮ ﺑﻮﺩﻥ ﻣﯽ­‌ﺗﻮﺍﻧﺪ ﺑﮑﻨﺪ؟ ﻋﺎﻣﻞ ﺍﺳﺎﺳﯽ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺩﺧﺘﺮ ﻳﺎ ﭘﺴﺮ ﺑﻮﺩﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺍﺯ ﺍﻳﻦ ﮐﻪ ﺍﻭ ﭘﺴﺮ ﺍﺳﺖ ﺧﻮﺷﺤﺎﻝ ﺑﺎﺷﻴﺪ، ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺧﻴﻠﯽ ﺯﻳﺎﺩ ﺍﻭ ﺍﺯ ﭘﺴﺮ ﺑﻮﺩﻧﺶ ﺧﻮﺷﺤﺎﻝ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﮔﺮ ﺍﺯ ﺍﻳﻦ ﮐﻪ ﺍﻭ ﺩﺧﺘﺮ ﺍﺳﺖ ﺭﺍﺿﯽ ﺑﺎﺷﻴﺪ، ﺍﻭ ﺍﺯ ﺩﺧﺘﺮ ﺑﻮﺩﻧﺶ ﺧﻮﺷﺤﺎﻝ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻣﺪﻟﯽ ﺩﺍﺭﻧﺪ ﺗﺎ ﺍﺯ ﺍﻭ ﺗﻘﻠﻴﺪ ﮐﻨﻨﺪ. ﻓﮑﺮ ﻣﯽ­‌ﮐﻨﻢ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ، ﺩﺧﺘﺮﻫﺎ ﺍﺯ ﭘﺴﺮﻫﺎ ﺭﺍﺣﺖ‌­ﺗﺮ ﺑﺎﺷﻨﺪ ﺗﺎ ﺗﻌﻠﻘﺸﺎﻥ ﺑﻪ ﺟﻨﺲ ﻣﻮﻧﺚ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﻨﺪ. ﻋﻠﻞ ﻣﺨﺘﻠﻔﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﺍﻣﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﺯ ﺟﻤﻠﻪ، ﺍﻳﻨﻜﻪ ﺍﻣﺮﻭﺯﻩ، ﭘﺪﺭﻫﺎ ﺻﺒﺢ ﺯﻭﺩ ﻣﯽ‌­ﺭﻭﻧﺪ ﺳﺮﮐﺎﺭ ﻭ ﺷﺐ ﺩﻳﺮﻭﻗﺖ ﺑﻪ ﺧﺎﻧﻪ ﺑﺮ ﻣﯽ­‌ﮔﺮﺩﻧﺪ. ﭘﺴﺮ ﮐﻮﭼﮏ، ﺑﻪ ﻏﻴﺮ ﺍﺯ ﺗﻌﻄﻴﻼﺕ ﺁﺧﺮ ﻫﻔﺘﻪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻴﻠﯽ ﮐﻢ ﭘﺪﺭﺵ ﺭﺍ ﺑﺒﻴﻨﺪ.

ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﻋﻠﻞ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﻃﻼﻕ ﻭ ﻋﺪﻡ ﻭﺟﻮﺩ ﻣﺪﻝ ﻣﺮﺩﺍﻧﻪ ﺑﺮﺍﯼ ﭘﺴﺮﻫﺎ ﺑﻪ ﺣﺴﺎﺏ ﺁﻭﺭﺩ. ﻣﻌﻤﻮﻻ ﻣﺎﺩﺭ ﮐﻮﺩﮎ ﺭﺍ ﻧﮕﺎﻩ ﻣﯽ‌­ﺩﺍﺭﺩ. ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ، ﻭﺿﻌﻴﺖ ﺑﺮﺍﯼ ﺩﺧﺘﺮﻫﺎﯼ ﮐﻮﭼﮏ ﺭﺍﺣﺖ‌ﺗﺮ­ ﺍﺳﺖ، ﺯﻳﺮﺍ ﺁﻧﻬﺎ ﺑﻪ ﻣﺪﻟﯽ ﺯﻧﺎﻧﻪ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻧﺪ. ﻭﻟﯽ ﻭﻗﺘﯽ ﭘﺴﺮﯼ ﭘﺪﺭﺵ ﺭﺍ ﺧﻴﻠﯽ ﮐﻢ ﻣﯽ‌ﺑﻴﻨﺪ، ﭼﻪ ﭼﻴﺰﯼ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ؟ ﺍﻭ ﻣﺪﻝ ﻣﺮﺩﺍﻧﻪ‌­ﺍﺵ ﺭﺍ ﺍﺯ ﮐﺠﺎ ﭘﻴﺪﺍ ﺧﻮﺍﻫﺪ ﮐﺮﺩ؟

ﻇﺎﻫﺮﺍ ﻋﻼﺝ ﮐﺎﺭ ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﺳﺎﺩﻩ ﺍﺳﺖ، ﻭﻟﯽ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﺁﻥ ﻣﺸﮑﻞ ﺍﺳﺖ: ﭘﺪﺭ ﺑﺎﻳﺪ ﻭﻗﺖ ﺯﻳﺎﺩﯼ ﺭﺍ ﺑﺎ ﻓﺮﺯﻧﺪﺵ، ﭼﻪ ﺩﺧﺘﺮ ﻭ ﭼﻪ ﭘﺴﺮ، ﺑﮕﺬﺍﺭﻧﺪ. ﻛﺎﺵ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺟﺎﺩﻭﺋﯽ ﺑﺮﺍﯼ ﻣﺘﻘﺎﻋﺪ ﮐﺮﺩﻥ ﭘﺪﺭﻫﺎ ﺑﻪ ﺍﻫﻤﻴﺖ ﻧﻘﺸﺸﺎﻥ ﺩﺭ ﺯﻧﺪﮔﯽ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺷﺖ. ﻣﺘﺄﺳﻔﺎﻧﻪ، ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺮﺩﻫﺎ، ﺁﻧﭽﻨﺎﻥ ﺑﺮﺩﻩ ﺟﺎﻩ‌ﻃﻠﺒﯽ ﻭ ﻣﻮﻓﻘﻴﺖ ﺩﺭ ﮐﺎﺭﺷﺎﻥ ﻫﺴﺘﻨﺪ ﮐﻪ ﻭﻗﺘﯽ ﺑﺮﺍﯼ ﮔﺬﺭﺍﻧﺪﻥ ﺑﺎ ﻓﺮﺯﻧﺪﺍﻥ ﺧﺮﺩﺳﺎﻟﺸﺎﻥ ﭘﻴﺪﺍ ﻧﻤﯽ‌­ﮐﻨﻨﺪ.
ﺁﻧﻬﺎ ﺑﺎ ﮔﻔﺘﻦ ﺍﻳﻨﮑﻪ ﺑﺎ ﮐﺎﺭ ﮐﺮﺩﻥ ﺁﻳﻨﺪﻩ ﻣﺎﻟﯽ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺭﺍ ﺗﺄﻣﻴﻦ ﻣﯽ­‌ﮐﻨﻨﺪ، ﺑﻪ ﺧﻴﺎﻝ ﺧﻮﺩ ﺗﻮﺿﻴﺤﯽ ﻣﻨﻄﻘﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭﺷﺎﻥ ﺍﺭﺍﺋﻪ ﻣﯽ‌­ﺩﻫﻨﺪ.

ﭘﺪﺭﻫﺎ ﻓﮑﺮ ﻣﯽ­‌ﮐﻨﻨﺪ ﮐﻪ ﺑﻌﺪﻫﺎ، ﻭﻗﺘﯽ ﮐﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺑﺰﺭﮔﺘﺮ ﺷﺪﻧﺪ، ﻭﻗﺖ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﮔﺬﺭﺍﻧﺪﻥ ﺑﺎ ﺁﻧﻬﺎ ﺧﻮﺍﻫﻨﺪ ﺩﺍﺷﺖ. ﻣﺘﺄﺳﻔﺎﻧﻪ، ﻭﺿﻊ ﮐﻤﺘﺮ ﺑﺪﻳﻦ ﻣﻨﻮﺍﻝ ﺍﺳﺖ. ﻭﻗﺘﯽ ﭘﺪﺭﯼ ﻭﻗﺖ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﺶ ﺩﺭ ﺍﻳﻦ ﺳﻨﻴﻦ ﺣﻴﺎﺗﯽ ﻧﮕﺬﺍﺭﺩ، ﺑﻌﺪﻫﺎ، ﻭﻗﺘﯽ ﮐﻪ ﺁﻧﻬﺎ ﺑﺰﺭﮒ­ﺗﺮ ﺷﻮﻧﺪ، ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﺑﻪ ﭘﺪﺭﺷﺎﻥ ﻧﻤﯽ‌­ﺩﻫﻨﺪ. ﻭ ﺍﻳﻦ ﺑﺴﻴﺎﺭ ﻏﻢ ﺍﻧﮕﻴﺰ ﺍﺳﺖ.

ﺣﺘﯽ ﭘﺪﺭﯼ ﮐﻪ ﮐﺎﺭ ﺑﺴﻴﺎﺭﯼ ﺩﺍﺭﺩ، ﻣﯽ ﺗﻮﺍﻧﺪ ﮐﺎﺭﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﭘﺴﺮ ﺧﺮﺩﺳﺎﻟﺶ(ﻭ ﻳﺎ ﺩﺧﺘﺮ ﮐﻮﭼﮑﺶ) ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﮐﻪ ﺑﻪ ﻓﮑﺮﺵ ﻫﻢ ﺧﻄﻮﺭ ﻧﮑﺮﺩﻩ ﺍﺳﺖ. ﻣﺜﻼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺍﺩﺍﺭﻩ‌ﺍﺵ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺵ ﮐﺎﺭﺕ ﺑﻔﺮﺳﺘﺪ. ﮐﻮﺩﮐﺎﻥ ﺧﻴﻠﯽ ﮐﻢ ﻧﺎﻣﻪ ﻭ ﮐﺎﺭﺕ ﺩﺭﻳﺎﻓﺖ ﻣﯽ­‌ﮐﻨﻨﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﮐﻪ ﺍﺯ ﭘﺪﺭﺷﺎﻥ ﻧﺎﻣﻪ‌ﺍﯼ ﺩﺭﻳﺎﻓﺖ ﮐﻨﻨﺪ ﺑﺴﻴﺎﺭ ﺧﻮﺷﺤﺎﻝ ﺧﻮﺍﻫﻨﺪ ﺷﺪ. ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮔﻬﮕﺎﻩ ﺑﻪ ﺍﻭ ﺗﻠﻔﻦ ﺑﺰﻧﺪ. ﻣﮑﺎﻟﻤﻪ­‌ﺍﯼ ﭘﻨﺞ ﺩﻗﻴﻘﻪ­­‌ﺍﯼ ﻣﯽ‌­ﺗﻮﺍﻧﺪ ﺍﺛﺮ ﺧﻴﻠﯽ ﺧﻮﺑﯽ ﺑﺮ ﺭﻭﯼ ﻓﺮﺯﻧﺪﺵ ﺑﮕﺬﺍﺭﺩ.

ﻫﺮ ﭘﺪﺭﯼ ﺑﺎﻳﺪ ﻃﻮﺭﯼ ﺗﺮﺗﻴﺐ ﮐﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺑﺪﻫﺪ ﮐﻪ ﻓﺮﺯﻧﺪﺵ ﻣﺤﻞ ﮐﺎﺭﺵ ﺭﺍ ﺑﺒﻴﻨﺪ، ﻭ ﺑﺒﻴﻨﺪ ﮐﻪ ﺍﻭ ﭼﻪ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﺪ. ﭘﺪﺭﻫﺎ ﺑﺎﻳﺪ ﺭﺍﺟﻊ ﺑﻪ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﻓﮑﺮ ﮐﻨﻨﺪ. ﺑﺎﻳﺪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺳﻄﺢ ﺍﻭ ﻧﺰﻭﻝ ﺩﻫﻴﺪ ﻭ ﺑﻪ ﺍﻭ ﺗﻮﺿﻴﺢ ﺩﻫﻴﺪ ﮐﻪ ﭼﻪ ﮐﺎﺭﻫﺎﻳﯽ ﻣﯽ­‌ﮐﻨﻴﺪ. ﺑﻌﻀﯽ ﺍﺯ ﮐﺎﺭﻫﺎ ﺭﺍ ﺑﻬﺘﺮ ﺍﺯ ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﻣﯽ‌­ﺗﻮﺍﻥ ﺗﻮﺿﻴﺢ ﺩﺍﺩ. ﻣﺎﺩﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﺑﺎ ﺑﺎﺯﯼ ﮐﺮﺩﻥ «ﺑﺎﺑﺎ ﮐﺎﺭ ﻣﯽ­‌ﮐﻨﻪ» ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﮐﻤﮏ ﮐﻨﺪ.

ﻃﺮﺯ ﺭﻓﺘﺎﺭ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎ ﺩﺧﺘﺮ ﻳﺎ ﭘﺴﺮﺧﺮﺩﺳﺎﻟﺸﺎﻥ ﺍﺯ ﺍﻳﻨﻬﺎ ﻧﻴﺰ ﻣﻬﻢ‌ﺗﺮ ﺍﺳﺖ. ﺍﮔﺮ ﻣﺎﺩﺭﯼ ﺑﻪ ﺩﺧﺘﺮ ﺑﻮﺩﻥ ﻭ ﻣﺮﺩﺍﻧﮕﯽ ﭘﺴﺮﺵ ﺍﻓﺘﺨﺎﺭ ﮐﻨﺪ، ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﺧﻮﺑﯽ ﭘﻴﺶ ﻣﯽ‌ﺭﻭﺩ. ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﻗﺒﻮﻝ ﮐﻨﺪ ﮐﻪ ﺩﺭ ﻃﺒﻴﻌﺖ ﻳﮏ ﭘﺴﺮ ﺍﺳﺖ ﮐﻪ ﻣﻘﺎﻭﻣﺖ ﮐﻨﺪ، ﺷﻴﻄﺎﻥ ﺑﺎﺷﺪ، ﻗﻴﻞ ﻭ ﻗﺎﻝ ﺭﺍﻩ ﺑﻴﻨﺪﺍﺯﺩ ﻭ  ﻭﻗﺎﺣﺖ ﺑﻪ ﺧﺮﺝ ﺩﻫﺪ. ﺍﻭ ﻧﺒﺎﻳﺪ ﺳﻌﯽ ﮐﻨﺪ ﮐﻪ ﭘﺴﺮﺵ ﺗﺒﺪﻳﻞ ﺑﻪ ﻣﻮﺟﻮﺩﯼ ﺁﺭﺍﻡ ﻭ ﻣﻄﻴﻊ ﻣﺜﻞ ﺩﺧﺘﺮﻫﺎ ﺑﺸﻮﺩ.

ﭘﺪﺭ ﻧﻴﺰ ﻧﻘﺶ ﺑﺴﻴﺎﺭ ﻣﻬﻤﯽ ﺩﺍﺭﺩ. ﺍﻭ ﺑﺎﻳﺪ ﺑﺎ ﭘﺴﺮﺵ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﻣﺮﺩﺍﻧﻪ ﻭ ﺧﺸﻦ ﺑﮑﻨﺪ. ﺍﻳﻦ ﻧﻮﻉ ﺑﺎﺯﯼ‌ﻫﺎ ﺑﺮﺍﯼ ﺍﻭ ﻻﺯﻣﻨﺪ. ﺩﺧﺘﺮﺵ ﺭﺍ ﻧﻴﺰ ﻣﯽ­‌ﺗﻮﺍﻧﺪ ﻧﺎﺯ ﻭ ﻧﻮﺍﺯﺵ ﮐﻨﺪ ﺗﺎ ﺯﻧﺎﻧﮕﯽ ﻭ ﻃﻨﺎﺯﯼ ﺭﺍ ﺩﺭ ﺍﻭ ﺗﺤﺮﻳﮏ ﮐﻨﺪ.

ﺩﺭ ﺩﺭﺱ ﺑﺤﺚ ﺩﺭﺑﺎﺭﻩ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺳﻮﺍﻝ‌ﻫﺎﯼ ﺟﻨﺴﯽ ﮐﻮﺩﮐﺎﻥ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
