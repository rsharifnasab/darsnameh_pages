ﺩﺭﺱ ﺳﻮﻡ- ﻧﻴﺎﺯﻫﺎﯼ ﻧﻮﺯﺍﺩ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺍﺯ ﺍﻫﻤﻴﺖ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﻧﻴﺎﺯﻫﺎﯼ ﺍﺳﺎﺳﯽ ﻧﻮﺯﺍﺩ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﻭ ﮔﻔﺘﻴﻢ ﺍﮔﺮ ﺷﻤﺎ، ﻭﺍﻟﺪﻳﻦ، ﺑﺘﻮﺍﻧﻴﺪ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺍﺳﺎﺳﯽ ﻧﻮﺯﺍﺩ ﺧﻮﺩ ﺭﺍ ﺑﺮﺁﻭﺭﺩﻩ ﮐﻨﻴﺪ ﻭ ﺍﻭ ﺭﺍﺿﯽ ﺑﺎﺷﺪ، ﺑﻪ ﺣﺪﺍﮐﺜﺮ ﺷﮑﻮﻓﺎﻳﯽ ﺧﻮﺩ ﺩﺳﺖ ﻣﯽ‌ﻳﺎﺑﺪ. ﺳﭙﺲ ﺑﺤﺚ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺍﺳﺎﺳﯽ ﮐﻮﺩﮎ ﺭﺍ ﺑﺎ ﺗﻐﺬﻳﻪ ﻭ ﺣﺮﺍﺭﺕ ﺷﺮﻭﻉ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺳﺎﻳﺮ ﺍﻳﻦ ﻧﻴﺎﺯﻫﺎ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ.

ﺧﻮﺍﺏ
ﺳﻮﻣﻴﻦ ﺍﺣﺘﻴﺎﺝ ﺍﺳﺎﺳﯽ ﻧﻮﺯﺍﺩ ﺧﻮﺍﺏ ﺍﺳﺖ. ﮐﻮﺩﮎ ﺑﻪ ﻃﻮﺭ ﻃﺒﻴﻌﯽ ﺍﻳﻦ ﺍﺣﺘﻴﺎﺟﺶ ﺭﺍ ﺑﺮﺁﻭﺭﺩﻩ ﻣﯽ‌­ﮐﻨﺪ. ﺍﻭ ﺁﻥ ﻗﺪﺭ ﮐﻪ ﻻﺯﻡ ﺍﺳﺖ ﻣﯽ‌­ﺧﻮﺍﺑﺪ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺧﻮﺍﺑﻴﺪ ﺑﻴﺪﺍﺭ ﻣﯽ­‌ﺷﻮﺩ. ﺑﺎ ﺗﻤﺎﻡ ﺍﻳﻨﻬﺎ ﺫﮐﺮ ﭼﻨﺪ ﻧﮑﺘﻪ ﺩﺭ ﻣﻮﺭﺩ ﺑﺮﻧﺎﻣﻪ ﻭ ﻋﺎﺩﺍﺕ ﺧﻮﺍﺏ ﮐﻮﺩﮎ ﺑﯽ­‌ﻓﺎﻳﺪﻩ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ.

ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﻪ ﻧﻮﺯﺍﺩ ﺑﺨﻮﺍﺑﺪ ﻻﺯﻡ ﻧﻴﺴﺖ ﮐﻪ ﺧﺎﻧﻪ ﺁﺭﺍﻡ ﻭ ﺳﺎﮐﺖ ﺑﺎﺷﺪ. ﻫﻤﻪ ﻣﺎ ﻣﺎﺩﺭﺍﻧﯽ ﺭﺍ ﺩﻳﺪﻩ‌ﺍﻳﻢ ﮐﻪ ﺩﺭ ﺧﺎﻧﻪ ﺗﻚ ﭘﺎ ﺭﺍﻩ ﻣﯽ‌­ﺭﻭﻧﺪ ﻭ ﺗﺎ ﮐﺴﯽ ﻭﺍﺭﺩ ﺧﺎﻧﻪ‌ﺷﺎﻥ ﻣﯽ‌ﺷﻮﺩ ﻣﯽ‌ﮔﻮﻳﻨﺪ «ﻫﻴﺲ، ﺑﭽﻪ ﺧﻮﺍﺑﻴﺪﻩ!» ﺍﻳﻦ ﮐﺎﺭ ﻭﺍﻗﻌﺎ ﻻﺯﻡ ﻧﻴﺴﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺍﮔﺮ ﺑﺎ ﮐﻮﺷﺶ ﻓﺮﺍﻭﺍﻥ ﺧﺎﻧﻪ ﺭﺍ ﺳﺎﮐﺖ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ ﺍﻳﻦ ﺳﮑﻮﺕ ﻣﻄﻠﻖ ﺷﺮﻁ ﻻﺯﻡ ﺑﺮﺍﯼ ﺧﻮﺏ ﺧﻮﺍﺑﻴﺪﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺧﻮﺍﻫﺪ ﺷﺪ.
ﺑﻨﺎﺑﺮﺍﻳﻦ ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺧﻮﺍﺏ ﺍﺳﺖ ﻫﻤﺎﻥ ﻃﻮﺭﯼ ﺭﻓﺘﺎﺭ ﮐﻨﻴﺪ ﮐﻪ ﻣﻌﻤﻮﻻ ﻣﯽ‌­ﮐﻨﻴﺪ ﻭ ﺧﺼﻮﺻﺎ ﺍﮔﺮ ﺩﻟﺘﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺩﺭ ﺍﺗﺎﻕ ﺩﻳﮕﺮﯼ ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺭﺍﺩﻳﻮ ﮔﻮﺵ ﮐﻨﻴﺪ ﻳﺎ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻧﮕﺎﻩ ﮐﻨﻴﺪ.

ﻧﻜﺘﻪ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﺳﺎﻋﺖ‌ﻫﺎﯼ ﺧﻮﺍﺏ ﮐﻮﺩﮎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮔﻮﺷﺰﺩ ﮐﻨﻴﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺳﺎﻋﺖ‌ﻫﺎ ﮐﺎﻣﻼ ﺑﺎ ﺳﺎﻋﺖ ﺧﻮﺍﺏ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﻧﻮﺯﺍﺩ ﺧﻴﻠﯽ ﮐﻮﭼﮏ ﺑﻪ ﻣﺤﺾ ﺍﻳﻦ ﮐﻪ ﺷﻴﺮﺵ ﺭﺍ ﺧﻮﺭﺩ ﻣﯽ‌ﺧﻮﺍﺑﺪ. ﺍﻣﺎ ﺑﻪ ﺗﺪﺭﻳﺞ ﺑﻴﺪﺍﺭﯼ ﺑﻌﺪ ﺍﺯ ﺷﻴﺮ ﺧﻮﺭﺩﻥ ﻃﻮﻻﻧﯽ ﻣﯽ‌­ﺷﻮﺩ. ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺩﺭ ﺭﻭﺯ ﺍﺻﻼ ﻧﺎﺭﺍﺣﺖ‌ﮐﻨﻨﺪﻩ ﻧﻴﺴﺖ ﻭﻟﯽ ﺑﻪ ﻫﻨﮕﺎﻡ ﺷﺐ ﺍﺫﻳﺖ‌ﮐﻨﻨﺪﻩ ﺍﺳﺖ. ﻧﻴﻤﻪ‌ ﺷﺐ ﻭﻗﺘﯽ ﺷﻴﺮ ﻧﻮﺯﺍﺩ ﺭﺍ ﻣﯽ‌ﺩﻫﻴﺪ ﺩﻟﺘﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻼﻓﺎﺻﻠﻪ ﺑﺨﻮﺍﺑﻴﺪ، ﺯﻳﺮﺍ ﻓﺮﺩﺍ ﺻﺒﺢ ﺯﻭﺩ ﺑﺎﻳﺪ ﺑﻴﺪﺍﺭ ﺷﻮﻳﺪ. ﻭﻟﯽ ﻧﻮﺯﺍﺩ ﭼﻴﺰﯼ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻧﻤﯽ‌­ﻓﻬﻤﺪ. ﺍﻭ ﺭﺍﺿﯽ ﻭ ﺧﻮﺷﺤﺎﻝ ﺍﺳﺖ ﻭ ﻧﻤﯽ­‌ﺧﻮﺍﻫﺪ ﺑﻼﻓﺎﺻﻠﻪ ﺑﺨﻮﺍﺑﺪ. ﻣﯽ­‌ﺧﻮﺍﻫﺪ ﺑﺎﺯﯼ ﮐﻨﺪ ﻭ ﮐﻤﯽ ﺑﻴﺪﺍﺭ ﺑﻤﺎﻧﺪ.
ﻣﻤﮑﻦ ﺍﺳﺖ ﻧﻮﺯﺍﺩ ﺑﻪ ﻋﻠﺘﯽ ﻧﻴﻤﻪ ﺷﺐ ﺍﺯ ﺧﻮﺍﺏ ﺑﻴﺪﺍﺭ ﺷﻮﺩ ﻭ ﮔﺮﻳﻪ ﮐﻨﺪ. ﺷﺎﻳﺪ ﺑﻪ ﻋﻠﺖ ﺩﺭﺩ ﺭﻭﺩﻩ ﻳﺎ ﻣﻌﺪﻩ، ﮐﻪ ﮐﺎﺭﯼ ﺍﺯ ﺩﺳﺖ ﺷﻤﺎ ﺑﺮﻧﻤﯽ‌­ﺁﻳﺪ. ﻣﯽ­‌ﺑﻴﻨﻴﺪ ﮐﻪ ﺣﺘﯽ ﻭﻗﺘﯽ ﺍﻭ ﺭﺍ ﺑﻐﻞ ﻣﯽ­‌ﮐﻨﻴﺪ ﻭ ﻧﻮﺍﺯﺵ‌ﺍﺵ ﻣﯽ­‌ﮐﻨﻴﺪ ﺁﺭﺍﻡ ﻧﻤﯽ­‌ﺷﻮﺩ ﻭ ﮔﺮﻳﻪ­‌ﺍﺵ ﺗﻤﺎﻣﯽ ﻧﺪﺍﺭﺩ. ﮐﻮﺩﮎ ﺍﺯ ﺯﻭﺭ ﮔﺮﻳﻪ ﻧﻔﺲ‌ﺍﺵ ﺑﻨﺪ ﺁﻣﺪﻩ ﻭ ﺷﻤﺎ ﮐﻪ ﺍﺯ ﺑﯽ­‌ﺧﻮﺍﺑﯽ ﺳﺮﭘﺎ ﺑﻨﺪ ﻧﻴﺴﺘﻴﺪ ﺳﻌﯽ ﻣﯽ­‌ﮐﻨﻴﺪ ﺍﻭ ﺭﺍ ﺁﺭﺍﻡ ﮐﻨﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺩﻭﺑﺎﺭﻩ ﺑﺨﻮﺍﺑﻴﺪ. ﺷﺎﻳﺪ ﺍﺯ ﺩﺳﺖ ﺍﻭ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻳﺪ. ﺷﺎﻳﺪ ﻫﻢ ﺩﻟﺘﺎﻥ ﺑﺨﻮﺍﻫﺪ ﺍﻭ ﺭﺍ ﻫﻞ ﺑﺪﻫﻴﺪ، ﮐﺘﮑﺶ ﺑﺰﻧﻴﺪ ﻭ ﺳﺮﺵ ﻓﺮﻳﺎﺩ ﺑﺰﻧﻴﺪ ﺗﺎ ﺳﺎﮐﺖ ﺷﻮﺩ: «ﺧﻔﻪ ﺷﻮ! ﻧﻤﯽ‌­ﻓﻬﻤﯽ ﺑﺎﻳﺪ ﺑﺨﻮﺍﺑﻴﻢ!»

ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﮐﺴﯽ ﺍﺯ ﭼﻨﻴﻦ ﻟﺤﻈﺎﺗﯽ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﺻﺤﺒﺖ ﻧﮑﺮﺩﻩ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺣﺴﺎﺱ ﮔﻨﺎﻩ ﺷﺪﻳﺪﯼ ﮐﻨﻨﺪ. ﻭﻟﯽ ﺍﮔﺮ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺳﯽ ﺑﻪ ﺷﻤﺎ ﺩﺳﺖ ﺩﻫﺪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﮐﺎﻣﻼ ﻃﺒﻴﻌﯽ ﻫﺴﺘﻴﺪ. ﮔﻔﺘﻢ «ﺍﺣﺴﺎﺱ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ». ﻭﻟﯽ ﺍﮔﺮ ﻭﺍﻗﻌﺎ ﮐﻨﺘﺮﻝ‌ﺗﺎﻥ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻩ ﻭ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﮐﺘﮏ ﺯﺩﻩ ﺑﺎﺷﻴﺪ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺎﻳﺪ ﺑﺎ ﻳﮏ ﭘﺰﺷﮏ ﻣﺸﻮﺭﺕ ﮐﻨﻴﺪ. ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﺍﻧﺴﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﻴﭽﺎﺭﻩ ﻭ ﻋﺼﺒﺎﻧﯽ ﺍﺣﺴﺎﺱ ﮐﻨﺪ ﻭﻟﯽ ﺑﺎﻳﺪ ﺍﻋﻤﺎﻟﺶ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﺪ.
       

ﻋﻮﺽ ﮐﺮﺩﻥ ﮐﻬﻨﻪ
ﺍﺣﺘﻴﺎﺝ ﺍﺳﺎﺳﯽ ﺩﻳﮕﺮ ﻧﻮﺯﺍﺩ ﺩﻓﻊ ﻣﻮﺍﺩ ﺯﺍﺋﺪ ﺑﺪﻧﺶ ﺍﺳﺖ (ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ). ﺍﻳﻦ ﺍﺣﺘﻴﺎﺝ ﺭﺍ ﻫﻢ ﺍﻭ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﺮﺁﻭﺭﺩﻩ ﻣﯽ­‌ﮐﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻋﻤﻼ ﻧﺎﺭﺍﺣﺘﯽ ﺧﺎﺻﯽ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﻣﮕﺮ ﺁﻥ ﮐﻪ ﺍﻳﻦ ﺍﺷﺘﺒﺎﻩ ﻋﻈﻴﻢ ﺭﺍ ﻣﺮﺗﮑﺐ ﺷﻮﻳﺪ ﮐﻪ ﺑﺨﻮﺍﻫﻴﺪ ﺩﺭ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ‌ﺍﺵ، ﮐﻨﺘﺮﻝ ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ ﺭﺍ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ.

ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺧﻮﺩﺗﺎﻥ ﻣﻘﻴﺪ ﺑﻪ ﻋﺎﺩﺕ­‌ﻫﺎﯼ ﺧﺎﺻﯽ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﻪ ﻫﻨﮕﺎﻡ ﮐﻮﺩﮐﯽ ﺑﻪ ﺷﻤﺎ ﺗﺤﻤﻴﻞ ﮐﺮﺩﻩ‌­ﺍﻧﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺯ ﮐﻬﻨﻪ ﮐﺜﻴﻒ ﺑﻴﺰﺍﺭ ﺑﺎﺷﻴﺪ.  ﺍﮔﺮ ﺩﺭ ﭼﻨﻴﻦ ﻭﺿﻌﻴﺘﯽ ﻗﺮﺍﺭ ﺩﺍﺭﻳﺪ ﺳﻌﯽ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﻨﺘﻘﻞ ﻧﮑﻨﻴﺪ. ﺍﻭ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺳﯽ ﻧﺴﺒﺖ ﺑﻪ ﺿﺎﻳﻌﺎﺕ ﺑﺪﻧﺶ ﻧﺪﺍﺭﺩ. ﺗﻨﻬﺎ ﻧﺘﻴﺠﻪ ﺍﻧﺘﻘﺎﻝ ﺍﻳﻦ ﮐﺮﺍﻫﺖ ﻣﺸﮑﻞ‌ﺗﺮ ﺷﺪﻥ ﮐﺎﺭﺗﺎﻥ ﺩﺭ ﺯﻣﺎﻥ ﺁﻣﻮﺯﺵ ﮐﻨﺘﺮﻝ ﺍﺩﺭﺍﺭ ﻭ ﻣﺪﻓﻮﻉ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.


ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺗﻤﺎﺱ ﺑﺪﻧﯽ
ﺍﺣﺘﻴﺎﺝ ﺍﺳﺎﺳﯽ ﺩﻳﮕﺮ ﻧﻮﺯﺍﺩ ﻧﻮﺍﺯﺵ ﻭ ﻳﺎ ﺑﻪ ﻗﻮﻟﯽ «ﻗﻮﺕ ﻗﻠﺐ ﺗﻤﺎﺱ ﻓﻴﺰﻳﮑﯽ» ﺍﺳﺖ. ﮐﻮﺩﮎ ﻓﻘﻂ ﺍﺯ ﺭﺍﻩ ﺗﻤﺎﺱ ﺑﺪﻧﯽ ﻣﺘﻮﺟﻪ ﻣﯽ‌­ﺷﻮﺩ ﺩﻭﺳﺘﺶ ﺩﺍﺭﻧﺪ: ﻳﻌﻨﯽ ﺑﺎ ﺑﻐﻞ ﮐﺮﺩﻥ، ﻧﻮﺍﺯﺵ ﮐﺮﺩﻥ، ﺗﮑﺎﻥ ﺩﺍﺩﻥ ﻭ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺑﺎ ﺍﻭ، ﺁﻭﺍﺯ ﺧﻮﺍﻧﺪﻥ ﺑﺮﺍﯼ ﺍﻭ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﺮﮔﺎﻩ ﻏﺮﻳﺰﻩ‌­ﺗﺎﻥ ﺣﮑﻢ ﮐﺮﺩ، ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺩﺭ ﺁﻏﻮﺵ ﺑﮕﻴﺮﻳﺪ، ﻧﻮﺍﺯﺷﺶ ﮐﻨﻴﺪ، ﺗﮑﺎﻧﺶ ﺩﻫﻴﺪ ﻭ ﺑﺮﺍﻳﺶ ﺁﻭﺍﺯ ﺑﺨﻮﺍﻧﻴﺪ. ﺑﻐﻠﺶ ﮐﻨﻴﺪ! ﺍﮔﺮ ﺩﻟﺘﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻫﺰﺍﺭ ﺑﺎﺭ ﻓﺸﺎﺭﺵ ﺑﺪﻫﻴﺪ! ﻧﻪ، ﻧﺘﺮﺳﻴﺪ، ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺯﻳﺎﺩﯼ ﻟﻮﺳﺶ ﻧﻤﯽ‌ﮐﻨﻴﺪ. ﺍﻭ ﺗﻨﻬﺎ ﺍﺯ ﺍﻳﻦ ﺭﺍﻩ ﻣﯽ­‌ﻓﻬﻤﺪ ﺩﻭﺳﺘﺶ ﺩﺍﺭﻧﺪ.


ﺍﺣﺘﻴﺎﺟﺎﺕ ﺫﻫﻨﯽ
ﺍﺣﺘﻴﺎﺟﺎﺕ ﻧﻮﺯﺍﺩ ﺷﻤﺎ ﻣﺤﺪﻭﺩ ﺑﻪ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﻧﺎﻡ ﺑﺮﺩﻳﻢ ﻧﻤﯽ‌­ﺷﻮﺩ. ﺍﻭ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺍﺳﺎﺳﯽ ﺫﻫﻨﯽ ﻧﻴﺰ ﺩﺍﺭﺩ. ﺑﺮﺍﯼ ﺩﺭﮎ ﺍﻳﻦ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺑﺎﻳﺪ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﺩﻧﻴﺎ ﺍﺯ ﻧﻈﺮ ﻧﻮﺯﺍﺩ ﭼﮕﻮﻧﻪ ﺍﺳﺖ.

ﻫﻤﻴﺸﻪ ﺑﺎﻳﺪ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﻧﻈﺮ ﮐﻮﺩﮎ ﺟﻬﺎﻥ ﻣﺎﻧﻨﺪ ﻓﻴﻠﻤﯽ ﺍﺳﺖ ﮐﻪ ﺗﺼﻮﻳﺮﻫﺎﻳﺶ ﺑﻪ ﺗﺪﺭﻳﺞ ﺷﮑﻞ ﻣﯽ‌­ﮔﻴﺮﻧﺪ. ﻣﺜﻼ  ﺷﻤﺎ ﻭﻗﺘﯽ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺭﻭﯼ ﻳﮏ ﺻﻨﺪﻟﯽ ﻣﯽ­‌ﻧﺸﻴﻨﻴﺪ ، ﮐﺎﻣﻼ ﻣﯽ­‌ﺩﺍﻧﻴﺪ ﮐﻪ ﺻﻨﺪﻟﯽ‌­ﺍﯼ ﮐﻪ ﺭﻭﻳﺶ ﻧﺸﺴﺘﻪ­‌ﺍﻳﺪ ﻳﺎ ﮐﺘﺎﺑﯽ ﮐﻪ ﻣﯽ‌­ﺧﻮﺍﻧﻴﺪ ﻳﺎ ﭼﺮﺍﻏﯽ ﮐﻪ ﮐﻨﺎﺭ ﺩﺳﺖ‌ﺗﺎﻥ ﺍﺳﺖ «ﺷﻤﺎ» ﻧﻴﺴﺘﻴﺪ. ﺍﻣﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﻳﻦ ﺭﺍ ﻧﻤﯽ­‌ﺩﺍﻧﺪ. ﺑﺎ ﻗﺮﺍﺭ ﺩﺍﺩﻥ ﺍﻭ ﺩﺭ ﻣﻌﺮﺽ ﺍﻧﮕﻴﺰﻩ­‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻋﺎﻃﻔﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﮐﻤﮑﺶ ﮐﻨﻴﺪ ﺗﺎ ﺩﻧﻴﺎﻳﺶ ﺛﺎﺑﺖ ﮔﺮﺩﺩ. ﻫﺮ ﭼﻪ ﻗﺪﺭ ﮐﻮﺩﮎ ﺑﻴﺸﺘﺮ ﺑﺒﻴﻨﺪ ﻭ ﺑﺸﻨﻮﺩ ﻫﻤﺎﻥ ﻗﺪﺭ ﺑﻴﺸﺘﺮ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﺒﻴﻨﺪ ﻭ ﺑﺸﻨﻮﺩ.

ﺍﻋﺘﻘﺎﺩ ﻗﺪﻳﻤﯽ‌­ﺍﯼ ﮐﻪ ﺑﺮ ﺍﺳﺎﺱ ﺁﻥ ﻫﺮﮐﺴﯽ ﺑﻪ ﻫﻨﮕﺎﻡ ﺗﻮﻟﺪ ﺑﺎ ﻣﻘﺪﺍﺭ ﻣﺸﺨﺼﯽ ﻫﻮﺵ ﻭ ﺍﺳﺘﻌﺪﺍﺩ ﺑﻪ ﺩﻧﻴﺎ ﻣﯽ‌­ﺁﻳﺪ ﺩﻳﮕﺮ ﻗﺪﺭﺕ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﻧﺘﺎﻳﺞ ﺗﺤﻘﻴﻘﺎﺕ ﺭﺍ ﻧﺪﺍﺭﺩ: ﻧﺘﺎﻳﺞ ﻋﻠﻤﯽ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﻫﺮ ﮐﻮﺩﮐﯽ ﺍﺯ ﻧﻈﺮ ﺍﺭﺛﯽ ﺣﺪﺍﮐﺜﺮ ﻇﺮﻓﻴﺖ ﻫﻮﺷﯽ ﺑﺎﻟﻘﻮﻩ‌­ﺍﯼ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﺑﺰﺭﮒ ﺷﺪﻥ ﻣﯽ­‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻳﻦ ﺣﺪﺍﮐﺜﺮ ﺑﺮﺳﺪ. ﺑﺮﺍﯼ ﻳﮑﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺣﺪﺍﮐﺜﺮ ﻫﻮﺵ ﺩﺭ ﺳﻄﺢ ﻳﮏ ﻧﺎﺑﻐﻪ ﺑﺎﺷﺪ ﻭ ﺑﺮﺍﯼ ﺩﻳﮕﺮﯼ ﺳﻄﺢ ﻫﻮﺵ ﻣﺘﻮﺳﻂ ﻭ ﺑﺮﺍﯼ ﻳﮏ ﻧﻔﺮ ﺩﻳﮕﺮ ﺣﺘﯽ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺍﺯ ﺳﻄﺢ ﻣﺘﻮﺳﻂ. ﻭﻟﯽ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻳﻦ ﺣﺪﺍﮐﺜﺮ ﻇﺮﻓﻴﺖ ﻫﻮﺷﯽ، ﺑﺴﺘﮕﯽ ﺑﻪ ﻣﻴﺰﺍﻥ ﺍﻧﮕﻴﺰﻩ­‌ﻫﺎﯼ ﺍﺣﺴﺎﺳﯽ ﻭ ﺫﻫﻨﯽ‌­ﺍﯼ ﮐﻪ ﺍﻭ ﺩﺭ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﺩﺭ ﻣﻌﺮﺽ ﺁﻥ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ ﺩﺍﺭﺩ.

ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺮﺍﯼ ﺑﺎﺯﯼ ﺍﺷﻴﺎ ﻣﺨﺘﻠﻒ ﺑﺪﻫﻴﺪ، ﺍﺷﻴﺎﻳﯽ ﮐﻪ ﺑﺘﻮﺍﻧﻨﺪ ﺑﻴﻨﺎﻳﯽ ، ﺷﻨﻮﺍﻳﯽ، ﺑﻮﻳﺎﻳﯽ ﻭ ﺩﻳﮕﺮ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺭﺍ ﺑﺮﺍﻧﮕﻴﺰﻧﺪ. ﺍﺷﻴﺎﻳﯽ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﮕﺬﺍﺭﻳﺪ ﮐﻪ ﺑﺘﻮﺍﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﺪ، ﺑﻤﮑﺪ، ﺑﺠﻮﺩ ﻭ ﺍﺯ ﻫﻢ ﺟﺪﺍ ﮐﻨﺪ.

ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺗﻘﺮﻳﺒﺎ ﺩﺭ ﺗﻤﺎﻡ ﺧﺎﻧﻪ­‌ﻫﺎ ﭘﻴﺪﺍ ﻣﯽ‌­ﺷﻮﻧﺪ ﺍﺯ ﻋﻬﺪﻩ‌ﯼ‌ ﺍﻳﻦ ﻛﺎﺭ ﺑﺮ ﻣﯽ‌­ﺁﻳﻨﺪ : ﺗﮑﻪ­‌ﻫﺎﯼ ﭘﺎﺭﭼﻪ‌­ﻫﺎﯼ ﺗﻤﻴﺰ ﮐﻪ ﻣﯽ‌­ﺗﻮﺍﻧﺪ ﻟﻤﺲ ﮐﻨﺪ ﻭ ﺑﺠﻮﺩ، ﺑﻄﺮﯼ ﻭ ﻇﺮﻑ­‌ﻫﺎﯼ ﭘﻼﺳﺘﻴﮑﯽ ﺑﺮﺍﯼ ﻓﺸﺎﺭ ﺩﺍﺩﻥ ﻭ ﺍﻳﻦ ﻃﺮﻑ ﻭ ﺁﻥ ﻃﺮﻑ ﺯﺩﻥ، ﮔﻴﺮﻩ ﺭﺧﺖ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﺯ ﺁﻧﻬﺎ ﺑﺮﺍﯼ ﺧﻮﺩﺵ ﺩﻧﺪﺍﻥ ﺑﮕﺬﺍﺭﺩ ﻭ.... ﺍﻳﻦ ﻓﻬﺮﺳﺖ ﺗﻤﺎﻣﯽ ﻧﺪﺍﺭﺩ. ﺍﻟﺒﺘﻪ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﻫﺮ ﭼﻴﺰﯼ ﮐﻪ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﺪ ﻣﻘﺼﺪ ﺍﻭﻟﺶ ﺩﻫﺎﻥ ﺍﻭﺳﺖ، ﺑﺎﻳﺪ ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﮐﻪ ﭼﻴﺰﻫﺎﯼ ﺧﻴﻠﯽ ﮐﻮﭼﮏ ﺑﻪ ﺍﻭ ﻧﺪﻫﻴﺪ ﺯﻳﺮﺍ ﺧﻄﺮ ﺧﻔﮕﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻓﺮﻭﺷﯽ­‌ﻫﺎ ﻫﻢ ﭼﻴﺰﻫﺎﯼ ﺟﺎﻟﺒﯽ ﺍﺭﺍﺋﻪ ﻣﯽ­‌ﺩﻫﻨﺪ! ﺟﻐﺠﻐﻪ‌­ﺍﯼ ﮐﻪ ﺑﺘﻮﺍﻥ ﺁﻥ ﺭﺍ ﺁﻭﻳﺰﺍﻥ ﮐﺮﺩ ﻳﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ­‌ﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌­ﺗﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﻋﻤﻮﺩﯼ ﻭ ﻳﺎ ﺍﻓﻘﯽ ﺭﻭﯼ ﺗﺨﺖ ﻳﺎ ﮔﻬﻮﺍﺭﻩ ﮐﻮﺩﮎ ﺁﻭﻳﺰﺍﻥ ﮐﺮﺩ ﺗﺎ ﺍﻭ ﺑﺘﻮﺍﻧﺪ ﺁﻥ­ﻫﺎ ﺭﺍ ﻟﻤﺲ ﮐﻨﺪ. ﻳﺎ ﻣﺜﻼ ﻳﮏ ﺁﮐﻮﺍﺭﻳﻮﻡ ﮐﻮﭼﮏ ﭘﻼﺳﺘﻴﮑﯽ ﺑﺎ ﻣﺎﻫﯽ­‌ﻫﺎﯼ ﺭﻧﮕﺎﺭﻧﮓ ﮐﻪ ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﮐﻨﺎﺭ ﺗﺨﺘﺶ ﺑﮕﺬﺍﺭﻳﺪ. ﺧﻮﺩﺗﺎﻥ ﻫﻢ ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ­‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺍﻧﮕﻴﺰﺵ ﺣﺴﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭﺳﺖ ﮐﻨﻴﺪ. ﻫﺮ ﺷﯽﺀ ﺭﻧﮕﯽ ﮐﻪ ﺑﺘﻮﺍﻥ ﺁﻥ ﺭﺍ ﺩﺭ ﺩﺳﺖ ﮔﺮﻓﺖ ﻭ ﺑﺎ ﺁﻥ ﺑﺎﺯﯼ ﮐﺮﺩ، ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺍﺟﺴﺎﻡ ﺑﺎﻳﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺑﺰﺭﮒ ﻭ ﺑﯽ‌­ﺧﻄﺮ ﺑﺎﺷﻨﺪ ﮐﻪ ﺑﺸﻮﺩ ﺍﺯ ﺁﻧﻬﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ. ﮔﻮﺷﻪ ﻭ ﮐﻨﺎﺭ ﺧﺎﻧﻪ­‌ﺗﺎﻥ ﺭﺍ ﺑﮕﺮﺩﻳﺪ ﻳﺎ ﺳﺮﯼ ﺑﻪ ﺑﺎﺯﺍﺭ ﺑﺰﻧﻴﺪ ﺑﺒﻴﻨﻴﺪ ﺧﻮﺩﺗﺎﻥ ﭼﻪ ﭼﻴﺰﯼ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﺍﻭ ﺩﺭﺳﺖ ﮐﻨﻴﺪ.

 ﻭﻟﯽ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﮐﻮﺩﮎ ﺍﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺭﺍ ﺑﻴﺸﺘﺮ ﺍﺯ ﻫﺮﭼﻴﺰ ﺩﻳﮕﺮﯼ ﺩﻭﺳﺖ ﺩﺍﺭﺩ «ﻣﺎﺩﺭ - ﭘﺪﺭ».  ﻫﺮ ﻃﻮﺭ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺻﺤﺒﺖ ﻭ ﺑﺎﺯﯼ ﮐﻨﻴﺪ. ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺷﺮﻭﻉ ﮐﻨﻴﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﺷﻴﺮ ﺑﭽﻪ ﺭﺍ ﻣﯽ‌ﺩﻫﻨﺪ، ﮐﻬﻨﻪ‌­ﺍﺵ ﺭﺍ ﻋﻮﺽ ﻣﯽ­‌ﮐﻨﻨﺪ، ﺣﻤﺎﻣﺶ ﻣﯽ­‌ﮐﻨﻨﺪ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﻳﮏ ﮐﻠﻤﻪ ﺻﺤﺒﺖ ﮐﻨﻨﺪ. ﭼﺮﺍ ﺍﺯ ﺍﻳﻦ ﻣﻮﻗﻌﻴﺖ‌­ﻫﺎ ﺑﺮﺍﯼ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﻢ.  ﻣﺴﻠﻤﺎ ﺍﻭ ﭼﻴﺰﯼ ﺭﺍ ﻛﻪ ﻣﯽ‌ﮔﻮﻳﻴﺪ ﻧﻤﯽ­‌ﻓﻬﻤﺪ، ﻭﻟﯽ ﻟﺤﻦ ﺻﺪﺍﯼ ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﻭ ﺍﻧﮕﻴﺰﻩ­ﺍﯼ ﺣﺴﯽ ﻭ ﺫﻫﻨﯽ ﺍﺳﺖ.

ﺑﻪ ﺍﻭ ﭼﻪ ﺑﮕﻮﻳﻴﺪ؟ ﻫﺮ ﭼﻪ ﺩﻟﺘﺎﻥ ﻣﯽ­‌ﺧﻮﺍﻫﺪ. ﻣﺜﻼ: «ﺍﻻﻥ ﻭﻗﺖ ﺣﻤﻮﻣﻪ، ﻧﯽ­‌ﻧﯽ ﮐﻮﭼﻮﻟﻮ ﺧﻮﺷﺤﺎﻟﻪ. ﺣﻤﻮﻡ ﺍﻣﺮﻭﺯ ﻣﯽ­‌ﭼﺴﺒﻪ؟ ﻳﺎ «ﺣﺎﻻ ﺑﺎﻳﺪ ﻋﻮﺿﺖ ﮐﻨﻢ ، ﭘﻮﺷﻚ ﻗﺪﻳﻤﯽ ﺭﻭ ﺍﻳﻦ‌ﻃﻮﺭﯼ ﺑﺮ ﻣﯽ‌­ﺩﺍﺭﻡ ﻭ ﺑﻌﺪ ﭘﻮﺷﻚ ﺗﻤﻴﺰﻭ ﺗﺎ ﻣﯽ‌­ﮐﻨﻢ، ﻣﯽ­‌ﺫﺍﺭﻡ ﺯﻳﺮﭘﺎﺕ، ﺍﻳﻨﻄﻮﺭﯼ ، ﺁﻫﺎﻥ ﺣﺎﻻ ﻣﯽ‌­ﺑﻨﺪﻣﺖ. ﺗﻤﻮﻡ ﺷﺪ».  ﻫﺮ ﻣﺎﺩﺭﯼ‌ ﺧﻮﺩﺵ ﻣﯽ‌­ﺗﻮﺍﻧﺪ ﭘﻴﺪﺍ ﮐﻨﺪ ﭼﻪ ﭼﻴﺰﯼ ﺑﻪ ﺑﭽﻪ‌­ﺍﺵ ﺑﮕﻮﻳﺪ ﻭ ﭼﻪ ﺁﻭﺍﺯﯼ ﺑﺮﺍﻳﺶ ﺑﺨﻮﺍﻧﺪ.
