ﺩﺭﺱ ﺑﻴﺴﺖ ﻭ ﺩﻭﻡ- ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺳﻮﺍﻝ ﻫﺎﯼ ﺟﻨﺴﯽ ﮐﻮﺩﮎ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺗﻔﺎﻭﺕ‌ﻫﺎﯼ ﺑﻴﻦ ﺩﺧﺘﺮ ﺑﭽﻪ‌ﻫﺎ ﻭ ﭘﺴﺮ ﺑﭽﻪ‌ﻫﺎ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺑﺤﺚ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺳﻮﺍﻝ‌ﻫﺎﯼ ﺟﻨﺴﯽ ﮐﻮﺩﮎ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ. 

ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﻴﭻ ﮐﻮﺩﮐﯽ ﺻﺪ ﺩﺭ ﺻﺪ ﻣﺬﮐﺮ ﻳﺎ ﻣﻮﻧﺚ ﻧﻴﺴﺖ. ﺍﮔﺮ ﻏﻴﺮ ﺍﺯ ﺍﻳﻦ ﺑﻮﺩ ﭼﻄﻮﺭ ﻣﻤﮑﻦ ﺑﻮﺩ ﺑﻴﻦ ﺩﻭ ﺟﻨﺲ ﻣﺨﺎﻟﻒ ﺗﻔﺎﻫﻢ ﺩﻭ ﺟﺎﻧﺒﻪ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻳﺪ. ﺑﺎﻳﺪ ﺍﺯ ﺗﺼﻮﻳﺮﻫﺎﻳﯽ ﭼﻮﻥ «ﺑﺮﺍﯼ ﻣﺮﺩ ﺑﻮﺩﻥ ﺑﺎﻳﺪ ﺧﺸﻦ ﺑﻮﺩ»،«ﭘﺴﺮﻫﺎ ﮔﺮﻳﻪ ﻧﻤﯽ‌­ﮐﻨﻨﺪ»ﻳﺎ«ﺩﺧﺘﺮﻫﺎ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻓﮑﺮ ﮐﺮﺩﻥ ﻧﺪﺍﺭﻧﺪ» ﻭ ﻏﻴﺮﻩ ﺍﺟﺘﻨﺎﺏ ﮐﺮﺩ. ﻣﺎ ﻣﯽ­‌ﺧﻮﺍﻫﻴﻢ ﭘﺴﺮﺍﻥ‌ﻣﺎﻥ ﺗﺒﺪﻳﻞ ﺑﻪ ﻣﺮﺩﻫﺎﻳﯽ ﺷﻮﻧﺪ ﮐﻪ ﺩﻟﺴﻮﺯﯼ ﻭ ﺩﺭﮎ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻧﻴﺰ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﻭ ﺩﺧﺘﺮﻫﺎﻳﻤﺎﻥ ﺗﺒﺪﻳﻞ ﺑﻪ ﺯﻥ­‌ﻫﺎﻳﯽ ﻣﺒﺘﮑﺮ ﻭ ﻣﺎﺟﺮﺍﺟﻮ ﺷﻮﻧﺪ ﮐﻪ ﻗﺎﺩﺭ ﺑﻪ ﺗﻔﮑﺮ ﺑﺎﺷﻨﺪ.

ﻭﻗﺘﯽ ﮐﻮﺩﮐﯽ ﺳﺌﻮﺍﻟﯽ ﺩﺭ ﻣﻮﺭﺩﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﻣﯽ­‌ﮐﻨﺪ، ﺑﺮﺍﯼ ﺍﻭ ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﭙﺮﺳﺪ ﭼﺮﺍ ﺑﺎﺭﺍﻥ ﻣﯽ‌ﺑﺎﺭﺩ ﻭ ﻳﺎ ﭼﺮﺍ ﺧﻮﺭﺷﻴﺪ ﻋﺼﺮﻫﺎ ﻏﺮﻭﺏ ﻣﯽ­‌ﮐﻨﺪ ﻭ ﻳﺎ ﭼﻪ ﭼﻴﺰﯼ ﺑﺎﻋﺚ ﺭﺷﺪ ﮔﻞ­‌ﻫﺎ ﻣﯽ­‌ﺷﻮﺩ. ﻭﻟﯽ ﻭﻗﺘﯽ ﺑﺒﻴﻨﺪ ﮐﻪ ﻣﺎ ﺩﺭ ﻣﻘﺎﺑﻞ ﺳﺌﻮﺍﻻﺕ ﻣﻌﺼﻮﻣﺎﻧﻪ ﺁﻧﻬﺎ ﺩﺭ ﻣﻮﺭﺩ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﻋﮑﺲ‌­ﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﻣﺘﻔﺎﻭﺗﯽ ﻧﺸﺎﻥ ﻣﯽ‌­ﺩﻫﻴﻢ، ﺑﻼﻓﺎﺻﻠﻪ ﺍﺣﺴﺎﺱ ﻣﯽ­‌ﮐﻨﺪ ﮐﻪ ﺍﻳﻦ ﭼﻴﺰﻫﺎ ﻣﺘﻔﺎﻭﺗﻨﺪ ﻭ ﺩﺭﻋﻴﻦ ﺍﻳﻦ ﮐﻪ ﻧﺒﺎﻳﺪ ﻃﺮﻑ ﺁﻧﻬﺎ ﺭﻓﺖ ﺑﺮﺍﻳﻤﺎﻥ ﺟﺬﺑﻪ ﺩﺍﺭﻧﺪ.

ﻗﺒﻞ ﺍﺯ ﻫﺮﭼﻴﺰ ﺑﺎﻳﺪ ﺳﻌﯽ ﮐﻨﻴﻢ ﺑﻪ ﺳﺌﻮﺍﻻﺕ ﺁﻧﻬﺎ ﺗﺎ ﺁﻧﺠﺎ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺻﻮﺭﺗﯽ ﺭﻭﺷﻦ، ﺻﺮﻳﺢ ﻭ ﺩﺭﺳﺖ ﭘﺎﺳﺦ ﺩﻫﻴﻢ. ﺍﻳﻦ ﺳﻮﺍﻻﺕ ﻣﻌﻤﻮﻻ ﺯﻣﺎﻧﯽ ﭘﺮﺳﻴﺪﻩ ﻣﯽ‌ﺷﻮﻧﺪ ﮐﻪ ﺍﺻﻼ ﺍﻧﺘﻈﺎﺭﺵ ﺭﺍ ﻧﺪﺍﺭﻳﻢ. ﺍﻭﻟﻴﻦ ﺳﺌﻮﺍﻟﯽ ﮐﻪ ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ ﻣﯽ‌ﮐﻨﺪ ﺳﺌﻮﺍﻝ ﻏﻴﺮﻗﺎﺑﻞ ﺍﺟﺘﻨﺎﺏ ﺯﻳﺮ ﺍﺳﺖ: «ﺑﭽﻪ ﭼﻪ ﺟﻮﺭﯼ ﺑﻪ ﺩﻧﻴﺎ ﻣﯽ‌‌ﺁﺩ؟» ﺑﻪ ﺍﻳﻦ ﺳﺌﻮﺍﻝ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﻣﯽ‌­ﺗﻮﺍﻥ ﺟﻮﺍﺏ ﺩﺍﺩ ﮐﻪ:«ﺑﭽﻪ‌­ﻫﺎ ﺍﺯ ﺗﻮﯼ ﺷﮑﻢ ﻣﺎﺩﺭ ﻣﯽ­‌ﺁﻥ. ﺍﻭﻧﺎ ﺩﺭ ﻗﺴﻤﺘﯽ ﺍﺯ ﺑﺪﻥ ﻣﺎﻣﺎﻥ ﮐﻪ ﺍﺳﻤﺶ ﺭﺣﻤﻪ ﺩﺭﺳﺖ ﻣﯽ‌­ﺷﻦ ﻭ ﺑﺰﺭﮒ ﻣﯽ­‌ﺷﻦ.» ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﺟﻮﺍﺏ ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺳﻪ ﺳﺎﻟﻪ ﮐﺎﻓﯽ ﺍﺳﺖ، ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺗﻮﺿﻴﺤﺎﺕ ﺑﻴﺸﺘﺮﯼ ﺑﺨﻮﺍﻫﺪ. ﺑﻪ ﻣﺎﻧﻨﺪ ﺟﻮﺍﺏ­‌ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺑﻘﻴﻪ ﺳﺌﻮﺍﻝ‌ﻫﺎﯼ ﻃﻔﻞ ﻣﯽ‌­ﺩﻫﻴﺪ، ﺟﻮﺍﺏ ﺳﻮﺍﻝ­‌ﻫﺎﯼ ﺍﻭ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﻧﻴﺰ ﮐﻮﺗﺎﻩ ﻭ ﺳﺎﺩﻩ ﺑﺪﻫﻴﺪ.

ﭼﮕﻮﻧﮕﯽ ﺧﺮﻭﺝ ﮐﻮﺩﮎ ﺍﺯ ﺭﺣﻢ ﻣﺎﺩﺭ ﻣﺴﺌﻠﻪ ﻇﺮﻳﻔﯽ ﺍﺳﺖ. ﺍﮔﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﻳﻦ ﺳﺌﻮﺍﻝ ﺭﺍ ﮐﺮﺩ، ﻗﺒﻞ ﺍﺯ ﺟﻮﺍﺏ ﺩﺍﺩﻥ ﺑﮕﺬﺍﺭﻳﺪ ﺍﻭ ﺣﺪﺱ ﺑﺰﻧﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺷﻤﺎ ﻣﯽ‌­ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺗﺼﻮﺭﯼ ﮐﻪ ﺍﻭ ﺍﺯ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺩﺍﺭﺩ ﭘﯽ ﺑﺒﺮﻳﺪ ﻭ ﺍﺣﺘﻤﺎﻻ ﺁﻥ ﺭﺍ ﺗﺼﺤﻴﺢ ﮐﻨﻴﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﮑﺮ ﮐﻨﺪ ﻧﻮﺯﺍﺩ ﺍﺯ ﻧﺎﻑ ﻭ ﻳﺎ ﺍﺯ ﻣﻘﻌﺪ ﺑﻴﺮﻭﻥ ﻣﯽ‌­ﺁﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮕﻮﺋﻴﺪ ﮐﻪ: «ﻧﻪ، ﺑﭽﻪ ﺍﺯ ﺍﻭﻥ ﺟﺎ ﻧﻤﯽ­‌ﺁﺩ. ﻣﺎﺩﺭ ﻳﮏ ﺳﻮﺭﺍﺥ ﻣﺨﺼﻮﺻﯽ ﺑﺮﺍﯼ ﺑﭽﻪ‌ﻫﺎ ﺩﺍﺭﻩ ﮐﻪ ﺷﺒﻴﻪ ﻳﻪ ﺗﻮﻧﻠﻪ. ﺍﻳﻦ ﺗﻮﻧﻞ ﻣﯽ­‌ﺗﻮﻧﻪ ﺑﺎﺯ ﺷﻪ. ﻭﻗﺘﯽ ﺑﭽﻪ ﻣﯽ­‌ﺧﻮﺍﺩ ﺩﻧﻴﺎ ﺑﻴﺎﺩ، ﺗﻮﻧﻞ ﭘﻬﻦ ﻣﯽ­‌ﺷﻪ ﺗﺎ ﺑﭽﻪ ﺑﺘﻮﻧﻪ ﺍﺯ ﺍﻭﻥ ﺭﺩ ﺑﺸﻪ. ﻭﻗﺘﯽ ﺑﭽﻪ ﺑﻪ ﺩﻧﻴﺎ ﺍﻭﻣﺪ، ﺗﻮﻧﻞ ﺩﻭﺑﺎﺭﻩ ﺗﻨﮓ ﻣﯽ­‌ﺷﻪ ﻭ ﻣﺜﻞ ﻗﺒﻞ ﻣﯽ‌­ﺷﻪ.»

ﺣﺎﻻ ﻣﯽ­‌ﺭﺳﻴﻢ ﺑﻪ ﻭﺟﻬﯽ ﺍﺯ ﺁﻣﻮﺯﺵ ﺟﻨﺴﯽ ﺍﻃﻔﺎﻝ ﺳﻪ ﺗﺎ ﭘﻨﺞ ﺳﺎﻟﻪ ﮐﻪ ﺷﺎﻳﺪ ﺑﺮﺍﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﮐﻤﯽ ﻣﺸﮑﻞ­‌ﺗﺮ ﺑﺎﺷﺪ. ﻭﻗﺘﯽ ﮐﻮﺩﮎ ﺑﺎ ﺁﻟﺖ ﺗﻨﺎﺳﻞ‌ﯼ­ﺍﺵ ﺑﺎﺯﯼ ﻣﯽ­‌ﮐﻨﺪ ﭼﻪ ﮐﺎﺭ ﮐﻨﻴﻢ؟ ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻮﺩﮐﯽ ﺳﻪ ﺳﺎﻟﻪ ﺣﻤﺎﻡ ﻣﯽ‌­ﮐﻨﺪ ﻭ ﺑﺎ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌­ﺍﺵ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ، ﻏﻠﻂ ﺍﺳﺖ ﮐﻪ ﺑﮕﻮﻳﻴﻢ ﺩﺭ ﺣﺎﻝ ﺍﺳﺘﻤﻨﺎ ﺍﺳﺖ.
ﻭﻗﺘﯽ ﮐﻪ ﺍﻭ ﮐﻮﭼﮏ‌ﺗﺮ ﺑﻮﺩ، ﻣﻤﮑﻦ ﺑﻮﺩ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻓﻘﻂ ﺍﺯ ﺭﻭﯼ ﮐﻨﺠﮑﺎﻭﯼ ﺍﻧﺠﺎﻡ ﺩﻫﺪ، ﺩﺭ ﺁﻥ ﺯﻣﺎﻥ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌­ﺍﺵ ﺑﺮﺍﯼ ﺍﻭ ﭼﻴﺰﯼ ﺑﻮﺩ ﻣﺜﻞ ﮔﻮﺵ ﻳﺎ ﺑﻴﻨﯽ­‌ﺍﺵ. ﺍﻣﺎ ﺍﮐﻨﻮﻥ ﮐﺸﻒ ﻣﯽ­‌ﮐﻨﺪ ﮐﻪ ﻧﻮﺍﺯﺵ ﺁﻟﺖ ﺗﻨﺎﺳﻠﯽ‌ﺍﺵ ﻟﺬﺕ­‌ﺑﺨﺶ ﺍﺳﺖ. ﻗﺪﻳﻢ ﻓﮑﺮ ﻣﯽ­‌ﮐﺮﺩﻧﺪ ﮐﻪ ﮐﻮﺩﮎ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺩﻭﺭﻩ ﺑﻠﻮﻍ ﻧﻤﯽ‌­ﺗﻮﺍﻧﺪ ﻫﻴﭻ ﻟﺬﺗﯽ ﺍﺯ ﺍﻳﻦ ﮐﺎﺭ ﺑﺒﺮﺩ، ﻭﻟﯽ ﺩﻳﮕﺮ ﺁﻥ ﺩﻭﺭﻩ ﺳﭙﺮﯼ ﺷﺪﻩ ﻭ ﺍﮐﻨﻮﻥ ﻣﯽ­‌ﺩﺍﻧﻴﻢ ﮐﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﻟﺬﺕ ﺑﺮﺩﻥ ﺟﺰﺋﯽ ﺍﺯ ﺭﺷﺪ ﻃﺒﻴﻌﯽ ﮐﻮﺩﮎ ﺩﺭ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﺍﺳﺖ.

ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﺩﺍﺭﺩ، ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟ ﺑﻬﺘﺮﻳﻦ ﮐﺎﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﮕﺬﺍﺭﻳﺪ ﮐﺎﺭﺵ ﺭﺍ ﺑﮑﻨﺪ. ﺑﻌﺪ ﺍﺯ ﻣﺪﺗﯽ، ﮐﻮﺩﮎ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺭﻫﺎ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﮐﺎﺭﻫﺎﯼ ﺩﻳﮕﺮ ﻣﯽ­‌ﭘﺮﺩﺍﺯﺩ. ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺑﺎ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺑﺮﺧﻮﺭﺩ ﮐﻨﻴﺪ ﮐﻪ ﭼﻪ ﺑﻬﺘﺮ. ﻭﻟﯽ ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ؟ ﺍﮔﺮ ﺍﻳﻦ ﮐﺎﺭ ﺷﻤﺎ ﺭﺍ ﺧﻴﻠﯽ ﻧﺎﺭﺍﺣﺖ ﻣﯽ‌ﮐﻨﺪ ﭼﻪ؟ ﻓﮑﺮ ﻣﯽ‌­ﮐﻨﻢ ﺑﻬﺘﺮﻳﻦ ﮐﺎﺭ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﮐﺎﺭ ﺩﻳﮕﺮﯼ ﺑﻪ ﺍﻭ ﭘﻴﺸﻨﻬﺎﺩ ﮐﻨﻴﺪ ﻭ ﺣﻮﺍﺳﺶ ﺭﺍ ﭘﺮﺕ ﮐﻨﻴﺪ. ﻭﻟﯽ ﻃﻮﺭﯼ ﺭﻓﺘﺎﺭ ﻧﮑﻨﻴﺪ ﮐﻪ ﮔﻮﻳﯽ ﺧﺎﻧﻪ ﺁﺗﺶ ﮔﺮﻓﺘﻪ ﻭ ﻓﺮﻳﺎﺩ ﺑﺰﻧﻴﺪ ﮐﻪ:« ﺑﺪﻭ ﺑﻴﺎ ﺍﺳﺐ ﺑﺎﺯﯼ ﮐﻨﻴﻢ.»

ﺗﺎ ﺍﻳﻨﺠﺎ، ﺻﺤﺒﺖ ﺍﺯ ﺯﻣﺎﻧﯽ ﺑﻮﺩ ﮐﻪ ﮐﻮﺩﮎ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﺑﺎﺯﯼ‌ﻫﺎ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ ﻭﻟﯽ ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﻳﻦ ﺑﺎﺯﯼ‌ﻫﺎ ﺩﺳﺘﻪ ﺟﻤﻌﯽ ﻣﯽ‌­ﺷﻮﻧﺪ ﭼﮕﻮﻧﻪ ﺭﻓﺘﺎﺭﯼ ﺑﺎﻳﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ؟ ﻏﺎﻟﺒﺎ ﺑﺎﺯﯼ‌ﻫﺎ ﺟﻨﺴﯽ ﺩﺳﺘﻪ ﺟﻤﻌﯽ ﻧﺎﺷﯽ ﺍﺯ ﮐﻨﺠﮑﺎﻭﯼ ﺍﺳﺖ ﻭ ﻣﻌﻤﻮﻻ ﺑﻪ ﺻﻮﺭﺕ«ﺩﮐﺘﺮﺑﺎﺯﯼ» ﻧﻤﺎﻳﺎﻥ ﻣﯽ­‌ﺷﻮﺩ. ﻳﮑﯽ ﺍﺯ ﺑﭽﻪ‌ﻫﺎ«ﺩﮐﺘﺮ» ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺧﺘﺮ، ﻳﺎ ﭘﺴﺮ ﮐﻮﭼﮑﯽ ﺭﺍ ﮐﻪ «ﻣﺮﻳﺾ» ﺷﺪﻩ ﺍﺳﺖ ﻣﻌﺎﻳﻨﻪ ﻣﯽ­‌ﮐﻨﺪ. ﻭ ﺗﻤﺎﻡ ﺑﭽﻪ‌ﻫﺎ ﺑﻪ ﻧﻮﺑﺖ ﻣﺮﻳﺾ ﻭ ﺩﮐﺘﺮ ﻣﯽ­‌ﺷﻮﻧﺪ. ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﻧﻮﻉ ﺑﺎﺯﯼ­‌ﻫﺎ، ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﮐﻨﺠﮑﺎﻭﯼ ﮐﻮﺩﮐﺎﻥ ﺭﺍ ﺑﺮﻃﺮﻑ ﮐﺮﺩ، ﺟﺬﺍﺑﻴﺘﺶ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺗﺎﺛﻴﺮ ﺑﺪﯼ ﺭﻭﯼ ﺷﺮﮐﺖ‌ﮐﻨﻨﺪﮔﺎﻥ ﺍﻳﻦ ﺑﺎﺯﯼ ﻧﻤﯽ­‌ﮔﺬﺍﺭﺩ.

ﺍﮔﺮ ﺯﻣﺎﻧﯽ ﺩﺭ ﺣﻴﻦ ﭼﻨﻴﻦ ﺑﺎﺯﯼ­‌ﺍﯼ ﺑﭽﻪ ﻫﺎﻳﺘﺎﻥ ﺭﺍ ﻏﺎﻓﻠﮕﻴﺮ ﮐﺮﺩﻳﺪ، ﻧﮕﺮﺍﻥ ﻧﺸﻮﻳﺪ ﻭ ﻓﮑﺮ ﻧﮑﻨﻴﺪ ﮐﻪ ﺑﺎ ﺗﻌﺪﺍﺩﯼ ﺍﻧﺴﺎﻥ ﻻﺕ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻳﺪ. ﺗﻨﺒﻴﻪ ﻧﮑﻨﻴﺪ، ﺩﻋﻮﺍ ﻧﮑﻨﻴﺪ. ﺑﺎ ﻟﺤﻨﯽ ﻃﺒﻴﻌﯽ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﻣﯽ­‌ﺩﺍﻧﻴﺪ ﮐﻪ ﺁﻧﻬﺎ ﮐﻨﺠﮑﺎﻭ ﻫﺴﺘﻨﺪ ﺑﺪﺍﻧﻨﺪ ﮐﻪ ﺩﻳﮕﺮﺍﻥ ﭼﮕﻮﻧﻪ ﺩﺭﺳﺖ ﺷﺪﻩ‌­ﺍﻧﺪ. ﻭﻟﯽ ﺣﺎﻻ ﮐﻪ ﮐﻨﺠﮑﺎﻭﯼ‌ﺗﺎﻥ ﺑﺮﻃﺮﻑ ﺷﺪﻩ «ﻳﮏ ﺑﺎﺯﯼ ﺩﻳﮕﺮ» ﺑﮑﻨﻴﺪ. ﻭ ﺁﻥ ﻭﻗﺖ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺑﺎﺯﯼ ﺩﻳﮕﺮﯼ ﺗﺸﻮﻳﻖ ﮐﻨﻴﺪ.

ﺑﺎ ﭼﻨﻴﻦ ﻋﮑﺲ‌­ﺍﻟﻌﻤﻠﯽ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻭ ﺩﻳﮕﺮﺍﻥ ﻧﺸﺎﻥ ﻣﯽ‌­ﺩﻫﻴﺪ ﮐﻪ، ﺭﻭﺵ ﮐﻮﺩﮐﺎﻧﻪ ﻭ ﻃﺒﻴﻌﯽ ﺁﻧﻬﺎ ﺑﺮﺍﯼ ﺑﺮﻃﺮﻑ ﮐﺮﺩﻥ ﮐﻨﺠﮑﺎﻭﯼ ﺟﻨﺴﯽ­ﺷﺎﻥ ﺭﺍ ﻗﺒﻮﻝ ﮐﺮﺩﻩ‌­ﺍﻳﺪ، ﺍﻣﺎ ﺍﺯ ﺍﻳﻦ ﺑﻪ ﺑﻌﺪ، ﺑﺎﻳﺪ ﺍﻳﻦ ﮔﻮﻧﻪ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺩﺳﺘﻪ ﺟﻤﻌﯽ ﺭﺍ ﻗﻄﻊ ﮐﻨﻨﺪ. ﺩﻧﺒﺎﻝ ﺍﻳﻦ ﻧﺒﺎﺷﻴﺪ ﮐﻪ ﺑﺒﻴﻨﻴﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﮐﺘﺮ ﺑﺎﺯﯼ ﻣﯽ­‌ﮐﻨﺪ ﻳﺎ ﻧﻪ. ﺑﻪ ﺯﻧﺪﮔﯽ ﺧﺼﻮﺻﯽ ﺍﻭ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﻳﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﻫﻴﭻ ﮔﺎﻩ ﺍﺯ ﭼﻨﻴﻦ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﮐﻮﺩﮎ ﺑﺎ ﮐﻮﺩﮐﺎﻥ ﻫﻢ ﺳﻦ ﻭ ﺳﺎﻟﺶ ﺧﺒﺮﺩﺍﺭ ﻧﺸﻮﻳﺪ. ﺍﻳﻨﻄﻮﺭﯼ ﺧﻴﻠﯽ ﺧﻮﺏ ﺍﺳﺖ. ﺍﮔﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﻣﺤﻴﻄﯽ ﺳﺎﻟﻢ ﺍﺯ ﻧﻈﺮ ﺟﻨﺴﯽ ﺑﺰﺭﮒ ﺷﻮﺩ، ﺍﻳﻦ ﮔﻮﻧﻪ ﺑﺎﺯ‌ﯼ‌ﻫﺎ ﻫﻴﭻ ﻭﻗﺖ ﺗﻤﺎﻡ ﻓﮑﺮ ﻭ ﺫﮐﺮ ﺍﻭ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﻣﺸﻐﻮﻝ ﻧﺨﻮﺍﻫﺪ ﮐﺮﺩ.

ﺣﺎﻻ ﺑﺎﻳﺪ ﻭﺿﻌﻴﺘﯽ ﺭﺍ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﮐﻨﻴﻢ ﮐﻪ ﻣﺎﺩﺭ ﺩﻳﮕﺮﯼ ﻏﻴﺮ ﺍﺯ ﺷﻤﺎ، ﭘﯽ ﺑﻪ ﻭﺟﻮﺩ ﺍﻳﻦ ﺑﺎﺯﯼ‌ﻫﺎ ﺑﺒﺮﺩ ﻭ ﺑﺎ ﺻﺪﺍﻳﯽ ﮔﺮﻓﺘﻪ ﺍﺯ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﻧﺎﺭﺍﺣﺘﯽ ﺑﻪ ﺷﻤﺎ ﺗﻠﻔﻦ ﺑﺰﻧﺪ ﻭ ﺑﮕﻮﻳﺪ ﮐﻪ ﭘﺴﺮ ﺷﻤﺎ  ﺩﺧﺘﺮ ﭼﻬﺎﺭﺳﺎﻟﻪ­‌ﺍﺵ ﺭﺍ ﺑﻪ ﮔﻮﺷﻪ ﮔﺎﺭﺍﮊ ﮐﺸﻴﺪﻩ ﺍﺳﺖ! ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻫﻤﻪ ﭼﻴﺰ ﺑﺴﺘﮕﯽ ﺑﻪ ﭼﮕﻮﻧﮕﯽ ﺭﻭﺍﺑﻂ ﺷﻤﺎ ﺑﺎ ﺍﻳﻦ ﻣﺎﺩﺭ ﺩﺍﺭﺩ. ﺑﺪﺍﻧﻴﺪ ﮐﻪ، ﺍﮔﺮ ﺑﺮﺣﺴﺐ ﺗﺼﺎﺩﻑ ﻣﺠﺒﻮﺭ ﺷﻮﻳﺪ ﮐﻪ ﭼﻨﻴﻦ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﺑﺸﻨﻮﻳﺪ، ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﻧﺎﺷﯽ ﺍﺯ ﺁﻣﻮﺯﺵ ﺟﻨﺴﯽ ﻏﻠﻄﯽ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﻣﺎﺩﺭ ﻋﺼﺒﺎﻧﯽ ﺩﺍﺷﺘﻪ ﺍﺳﺖ. ﺍﮔﺮ ﻣﺎﺩﺭﯼ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺍﻳﻦ ﮔﻮﻧﻪ ﺑﺎﺯﯼ­‌ﻫﺎ ﺳﺮﺯﻧﺶ ﮐﻨﺪ، ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﻪ ﮐﻨﺎﺭﯼ ﺑﮑﺸﻴﺪ ﻭ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺧﺎﻧﻢ ﻓﻼﻧﯽ ﺧﻴﻠﯽ ﻋﺼﺒﺎﻧﯽ ﺍﺳﺖ، ﻭﻟﯽ ﺷﻤﺎ ﻧﻪ. ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺍﻭ ﻧﺒﺎﻳﺪ ﺩﺭ ﮔﻮﺷﻪ ﮔﺎﺭﺍﮊ ﺑﺎ ﺩﺧﺘﺮ ﺧﺎﻧﻢ ﻓﻼﻧﯽ ﺍﻳﻦ ﺑﺎﺯﯼ ﺭﺍ ﻣﯽ­‌ﮐﺮﺩ، ﻭﻟﯽ ﺧﺎﻧﻢ ﻓﻼﻧﯽ ﻫﻢ ﻋﻠﺘﯽ ﻧﺪﺍﺷﺖ ﮐﻪ ﺍﺯ ﮐﺎﻩ ﮐﻮﻫﯽ ﺑﺴﺎﺯﺩ.
