ﺩﺭﺱ ﭼﻬﺎﺭﺩﻫﻢ- ﻓﻦ ﺍﻧﻌﻜﺎﺱ ﺍﺣﺴﺎﺳﺎﺕ- ﺑﺨﺶ ﺩﻭﻡ

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ، ﺩﺭ ﻣﻮﺭﺩ ﻓﻦ ﺍﻧﻌﮑﺎﺱ ﺍﺣﺴﺎﺳﺎﺕ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﻭ ﺑﺤﺚ ﺭﺍ ﺑﺎ ﻳﮏ ﻣﺜﺎﻝ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺍﻧﺘﻬﺎ ﻧﻴﺰ ﻳﺎﺩﺁﻭﺭ ﺷﺪﻳﻢ ﮐﻪ ﻫﻤﻴﺸﻪ ﺍﺟﺮﺍﯼ ﺍﻳﻦ ﻓﻦ ﮐﺎﺭ ﺁﺳﺎﻧﯽ ﻧﻴﺴﺖ ﻭ ﮔﺎﻫﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﻭﺍﻟﺪﻳﻦ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻧﺪ ﻭ ﻗﺎﺩﺭ ﺑﻪ ﺣﻔﻆ ﺧﻮﻧﺴﺮﺩﯼ ﺧﻮﺩ ﻧﺒﺎﺷﻨﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﻓﺮﻳﺎﺩ ﻣﯽ‌ﮐﺸﻨﺪ، ﺑﻌﺪﺵ ﻧﺎﺭﺍﺣﺖ ﻣﯽ‌ﺷﻮﻧﺪ. ﺩﺭ ﻭﺍﻗﻊ ﭼﻴﺰ ﺧﺎﺭﻕ‌ﺍﻟﻌﺎﺩﻩ‌ﺍﯼ ﺍﺗﻔﺎﻕ ﻧﻴﻔﺘﺎﺩﻩ ﺍﺳﺖ، ﻭﻟﯽ ﺁﻧﻬﺎ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﻫﻤﻴﺸﻪ ﺑﺎﻳﺪ ﺁﺭﺍﻡ ﺑﺎﺷﻨﺪ. ﺍﻳﻦ ﺭﻓﺘﺎﺭﯼ ﺍﻳﺪﻩ‌ﺁﻝ ﺍﺳﺖ ﻭﻟﯽ ﻣﻦ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻫﻴﭻ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﺭﺍ ﻧﺪﻳﺪﻩ‌ﺍﻡ ﮐﻪ ﻗﺎﺩﺭ ﺑﺎﺷﻨﺪ ﺍﻳﻦ ﺍﺻﻮﻝ ﺭﺍ ﺩﺭ ﺗﻤﺎﻡ ﺷﺮﺍﻳﻂ ﺑﻪ ﻣﻮﺭﺩ ﺍﺟﺮﺍ ﺑﮕﺬﺍﺭﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﮔﺮ ﭘﻴﺶ ﺁﻣﺪ ﮐﻪ ﺩﻟﺘﺎﻥ ﺑﺨﻮﺍﻫﺪ ﻓﺮﻳﺎﺩ ﺑﮑﺸﻴﺪ، ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﮑﻨﻴﺪ. ﻭ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻭﻗﺘﯽ ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻢ ﺁﻟﻮﺩﺗﺎﻥ ﺭﺍ ﺧﺎﻟﯽ ﮐﺮﺩﻳﺪ، ﻣﺴﻠﻤﺎ ﺭﻓﺘﺎﺭﺗﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺴﻴﺎﺭ ﻓﺮﻕ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ: «ﺧﻴﻠﯽ ﻧﺎﺭﺍﺣﺘﻢ ﮐﻪ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻡ، ﻭﻟﯽ ﺍﻻﻥ ﺑﻬﺘﺮﻡ.» ﻭ ﺍﻭ ﺍﻳﻦ ﺭﺍ ﺩﺭﮎ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺣﺎﻻ ﺍﮔﺮ ﺑﮕﻮﻳﻢ ﮐﻪ ﺗﻤﺎﻡ ﺍﻳﻦ«ﺩﻭ ﺳﺎﻟﻪ‌ﻫﺎﯼ ﻭﺣﺸﺘﻨﺎﮎ» ﻭﺍﻟﺪﻳﻨﺸﺎﻥ ﺭﺍ ﺍﻳﻦ ﻫﻤﻪ ﻧﺎﺭﺍﺣﺖ ﻧﻤﯽ‌ﮐﻨﻨﺪ ﺷﺎﻳﺪ ﮐﻤﯽ ﺳﺒﮏ‌ﺗﺮ ﺷﻮﻳﺪ. ﺑﻴﻦ ﺁﻧﻬﺎ ﺗﻔﺎﻭﺕ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻧﻈﺮ ﺧﻠﻖ ﻭ ﺧﻮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻫﺮﮐﺪﺍﻡ ﺍﺯ ﻧﻈﺮ ﺑﻴﻮﻟﻮﮊﻳﮑﯽ ﺗﻔﺎﻭﺕ ﺑﺴﻴﺎﺭﯼ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺩﺍﺭﺩ. ﺑﻌﻀﯽ ﺍﺯ ﺁﻧﻬﺎ ﺗﺤﺮﻳﮏ‌ﭘﺬﻳﺮﻧﺪ. ﻭ ﺑﻌﻀﯽ ﺩﻳﮕﺮ «ﺭﺍﺣﺖ»ﺗﺮ ﻫﺴﺘﻨﺪ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺪﺍﻳﺖ ﮐﺮﺩﻥ ﺑﺮﺧﯽ ﺍﺯ ﮐﻮﺩﮐﺎﻥ ﺩﺭ ﻳﮏ ﺳﺮﺍﺷﻴﺒﯽ ﭘﺮﺯﺣﻤﺖ ﺭﺍﺣﺖ‌ﺗﺮ ﺍﺯ ﺑﺮﺧﯽ ﺩﻳﮕﺮ ﺍﺳﺖ. ﺍﻣﺎ ﻫﺮ ﮐﻮﺩﮎ ﻃﺒﻴﻌﯽ ﻭ ﺳﺎﻟﻤﯽ ﺑﺎﻳﺪ ﻣﻘﺪﺍﺭﯼ ﻣﻨﻔﯽ‌ﮔﺮﺍﻳﯽ ﻭ ﻋﻨﺎﺩ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﺩﻫﺪ. ﻭ ﺍﮔﺮ ﺷﻬﺎﻣﺘﺘﺎﻥ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﻴﺪ ﻫﻤﻴﺸﻪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﯽ‌ﮔﺬﺭﺩ! ﺯﻣﺎﻧﯽ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺳﻪ ﺳﺎﻟﻪ ﺷﺪ، ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﮐﻪ ﺍﻭ ﭼﮕﻮﻧﻪ ﻧﺎﮔﻬﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﺩﻭﺳﺎﻝ ﻭ ﻧﻴﻤﮕﯽ ﻫﻤﺮﺍﻩ ﻭ «ﺭﺍﺣﺖ» ﻣﯽ‌ﺷﻮﺩ.

ﻣﺎﺩﺭﻫﺎ ﺧﻴﻠﯽ ﺧﻮﺏ ﻣﯽ‌ﺩﺍﻧﻨﺪ ﮐﻪ ﮐﻠﻤﻪ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺍﻳﻦ ﺳﻦ «ﻧﻪ» ﺍﺳﺖ ﻭ ﻧﺒﺎﻳﺪ ﺗﻌﺠﺐ ﮐﻨﻴﺪ ﮐﻪ ﮐﻮﺩﮎ ﺧﻴﻠﯽ ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﮐﻠﻤﻪ «ﺑﻠﻪ» ﺭﺍ ﻳﺎﺩ ﺑﮕﻴﺮﺩ «ﻧﻪ» ﺭﺍ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ. ﻫﺮﭼﻪ ﺑﺎﺷﺪ ﺍﻭ ﮐﻠﻤﻪ «ﻧﻪ» ﺭﺍ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﺍﺯ ﺩﻫﺎﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﺷﻨﻴﺪﻩ ﺍﺳﺖ ﺗﺎ ﮐﻠﻤﻪ «ﺑﻠﻪ» ﺭﺍ. ﻭ ﺷﺎﻳﺪ ﺍﮔﺮ ﺩﺭ ﺩﻭﺭﺍﻥ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ «ﻧﻪ»ﻫﺎ ﺭﺍ ﺑﻪ ﺣﺪﺍﻗﻞ ﻣﻤﮑﻦ ﮐﺎﻫﺶ ﺩﻫﻴﻢ، «ﻧﻪ»ﻫﺎﯼ ﮐﻤﺘﺮﯼ ﺑﺸﻨﻮﻳﻢ. ﻭﻟﯽ ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﻣﻨﻔﯽ ﺩﻳﮕﺮﯼ ﺍﺯ ﺍﻭ ﺧﻮﺍﻫﻴﻢ ﺩﻳﺪ، ﻭﻗﺘﯽ ﺻﺪﺍﻳﺶ ﻣﯽ‌ﮐﻨﻴﺪ ﻓﺮﺍﺭ ﻣﯽ‌ﮐﻨﺪ، ﻟﮕﺪ ﻣﯽ‌ﺯﻧﺪ، ﻫﻨﮕﺎﻣﯽ ﺍﺯ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﮐﺎﺭﯼ ﺭﺍ ﺑﮑﻨﺪ ﺑﻪ ﺧﻮﺩﺵ ﭘﻴﭻ ﻭ ﺗﺎﺏ ﻣﯽ‌ﺩﻫﺪ ﻭ ﻳﺎ ﻭﺍﻗﻌﺎ ﺩﺍﺩ ﻭ ﻓﺮﻳﺎﺩ ﺭﺍﻩ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ ﻭ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﺩ. ﻣﻌﻤﻮﻻ ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﺩﺭ ﺩﻭﺭﺍﻥ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﻇﺎﻫﺮ ﻧﻤﯽ‌ﺷﻮﺩ. ﺍﮐﻨﻮﻥ ﺑﺒﻴﻨﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﻳﻦ ﺭﻓﺘﺎﺭ ﺑﺮﺧﻮﺭﺩﮐﻨﻴﻢ.

ﺭﻓﺘﺎﺭﯼ ﮐﻪ «ﺗﺸﻮﻳﻖ» ﺷﻮﺩ ﮔﺮﺍﻳﺶ ﺑﻪ ﺗﮑﺮﺍﺭ ﺩﺍﺭﺩ. ﺍﮔﺮ ﻭﺍﻟﺪﻳﻦ ﺟﻮﺍﺏ ﺍﻳﻦ ﻣﺨﺎﻟﻔﺖ‌ﻫﺎ ﺭﺍ ﺑﺎ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻥ ﺑﺪﻫﻨﺪ، ﻧﺎﺧﻮﺩﺁﮔﺎﻩ  ﮐﻮﺩﮎ ﺭﺍ ﺑﻪ ﻣﺨﺎﻟﻔﺖ ﮐﺮﺩﻥ ﺗﺮﻏﻴﺐ ﻭ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺍﻣﺎ ﺭﻭﺵ‌ﻫﺎﯼ ﺑﻬﺘﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﺭﻭﺵ ﺗﺸﻮﻳﻖ ﻣﺨﺎﻟﻔﺖ‌ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻗﺒﻞ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﺑﺎﻳﺪ ﻣﺨﺎﻟﻔﺖ ﺷﻔﺎﻫﯽ ﺭﺍ ﺍﺯ ﻳﮏ ﺭﻓﺘﺎﺭ ﻣﺨﺎﻟﻒ ﺗﺸﺨﻴﺺ ﺩﺍﺩ. ﻣﺜﻼ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﮔﻮﻳﻴﺪ:«ﺧﺐ، ﺑﺎﻳﺪ ﻛﺎﭘﺸﻦ ﺑﭙﻮﺷﯽ ﮐﻪ ﺑﺮﻳﻢ ﺑﻴﺮﻭﻥ». ﻭ ﮐﻤﮑﺶ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻛﺎﭘﺸﻨﺶ ﺭﺍ ﺑﭙﻮﺷﺪ. ﺍﻭ ﻣﯽ‌ﮔﻮﻳﺪ «ﻧﻪ، ﻧﻤﯽ‌ﺧﻮﺍﻡ!» ﻭ ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺩﺭ ﻋﻤﻞ ﺣﺮﻑ‌ﻫﺎﻳﺶ ﺑﺎ ﺍﻋﻤﺎﻟﺶ ﺗﻨﺎﻗﺾ ﺩﺍﺭﺩ، ﭼﺮﺍ ﮐﻪ ﺩﺳﺖ‌ﻫﺎﻳﺶ ﺭﺍ ﺩﺭ ﺁﺳﺘﻴﻨﺶ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﻛﺎﭘﺸﻨﺶ ﺭﺍ ﺑﭙﻮﺷﺪ. ﺍﻳﻦ ﻣﺜﺎﻝ ﺧﻮﺑﯽ ﺍﺯ«ﻣﺨﺎﻟﻔﺖ ﺷﻔﺎﻫﯽ» ﺍﺳﺖ. ﮔﻮﻳﯽ ﮐﻮﺩﮎ ﺑﻪ ﻣﺎﺩﺭﺵ ﻣﯽ‌ﮔﻮﻳﺪ:«ﻣﯽ‌ﺩﻭﻧﻢ ﮐﻪ ﺑﻴﺮﻭﻥ ﻫﻮﺍ ﺳﺮﺩﻩ ﻭ ﻭﺍﻗﻌﺎ ﺑﻪ ﻛﺎﭘﺸﻦ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻡ. ﻣﯽ‌ﺩﻭﻧﻢ ﮐﻪ ﺗﻮ ﺍﺯ ﻣﻦ ﺑﺰﺭﮒ‌ﺗﺮﯼ ﻭ ﻣﯽ‌ﺗﻮﻧﯽ ﻣﺠﺒﻮﺭﻡ ﮐﻨﯽ ﮐﺎﭘﺸﻨﻤﻮ ﺑﭙﻮﺷﻢ. ﻭﻟﯽ ﺑﺒﻴﻦ ﻣﺎﻣﺎﻥ، ﻗﺒﻮﻝ ﮐﻦ ﮐﻪ ﻣﻨﻢ ﺁﺩﻣﻢ ﻭ ﺣﺪﺍﻗﻞ ﺍﻳﻦ ﺣﻘﻮ ﺩﺍﺭﻡ ﮐﻪ ﻳﻪ ﮐﻤﯽ ﺍﻋﺘﺮﺍﺽ ﮐﻨﻢ!» ﺍﻳﻦ ﻣﺨﺎﻟﻔﺖ ﻣﺜﻞ ﻳﮏ ﺑﺎﺯﯼ ﺍﺳﺖ: ﮐﻮﺩﮎ ﺑﻪ ﻧﻮﻋﯽ ﺑﺎ ﻣﺎﺩﺭﺵ«ﺑﺎﺯﯼ ﻋﺪﻡ ﺍﻃﺎﻋﺖ» ﻣﯽ‌ﮐﻨﺪ. ﻭ ﺍﮔﺮ ﻣﺎﺩﺭ ﻣﺘﻮﺟﻪ ﺣﺎﻟﺖ‌ ﺑﺎﺯﯼ‌ﻭﺍﺭ ﺍﻳﻦ ﺣﺮﮐﺖ ﻓﺮﺯﻧﺪﺵ ﻧﺸﻮﺩ ﻭ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﺷﺪﻳﺪﯼ ﻧﺸﺎﻥ ﺩﻫﺪ، ﺑﺎﻋﺚ ﺑﺤﺮﺍﻧﯽ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻋﻠﺘﯽ ﺑﺮﺍﯼ ﺁﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ.

ﺭﻓﺘﺎﺭ ﻣﺨﺎﻟﻒ ﻭﺍﻗﻌﯽ، ﮐﻪ ﺑﺎ ﻣﺨﺎﻟﻔﺖ ﺷﻔﺎﻫﯽ ﺗﻔﺎﻭﺕ ﺩﺍﺭﺩ، ﺩﺭ ﺍﻳﻦ ﺷﺮﺍﻳﻂ ﺑﺎ ﻓﺮﺍﺭﮐﺮﺩﻥ ﻳﺎ ﻣﺨﺎﻟﻔﺖ ﻓﻴﺰﻳﮑﯽ ﻭ ﺷﺪﻳﺪ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﺎﺩﺭ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻛﺎﭘﺸﻦ ﮐﻮﺩﮎ ﺭﺍ ﺗﻨﺶ ﮐﻨﺪ ﻇﺎﻫﺮ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺍﮔﺮ ﭼﻨﻴﻦ ﺷﻮﺩ ﻣﺎﺩﺭ ﺑﻪ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﯽ‌ﺗﻮﺍﻧﺪ«ﺟﻮﺍﺏ ﺑﺪﻫﺪ». ﺍﮔﺮ ﻗﺼﺪ ﺍﺯ ﺑﻴﺮﻭﻥ ﺭﻓﺘﻦ، ﺭﻓﺘﻦ ﺑﻪ ﺣﻴﺎﻁ ﻭ ﺑﺎﺯﯼ ﮐﺮﺩﻥ ﺑﺎﺷﺪ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﮕﻮﻳﺪ «ﺑﺮﺍﯼ ﺑﻴﺮﻭﻥ ﺭﻓﺘﻦ ﺑﺎﻳﺪ ﻛﺎﭘﺸﻨﺖ ﺭﻭ ﺑﭙﻮﺷﯽ، ﻭﻟﯽ ﺍﮔﺮ ﺩﻟﺖ ﺑﺨﻮﺍﺩ ﻣﯽ‌ﺗﻮﻧﻴﻢ ﺑﻤﻮﻧﻴﻢ ﺧﻮﻧﻪ.» ﺍﮔﺮ ﻭﺍﻗﻌﺎ ﺩﻟﺶ ﺑﺨﻮﺍﻫﺪ ﺑﻴﺮﻭﻥ ﺑﺮﻭﺩ ﻭ ﺑﺎﺯﯼ ﮐﻨﺪ، ﺑﺎ ﻧﺎﺭﺍﺣﺘﯽ ﺣﺎﺿﺮ ﺑﻪ ﭘﻮﺷﻴﺪﻥ ﻛﺎﭘﺸﻦ ﻣﯽ‌ﺷﻮﺩ. ﻭﻟﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺷﺮﺍﻳﻂ ﻃﻮﺭ ﺩﻳﮕﺮﯼ ﺑﺎﺷﺪ: ﻣﺜﻼ ﺍﮔﺮ  ﻣﺎﺩﺭ ﻗﺼﺪ ﺧﺮﻳﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﻋﺠﻠﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺩﻳﮕﺮ ﻓﺮﺻﺖ ﺍﻳﻨﮑﻪ ﻣﺴﺌﻠﻪ ﺭﺍ ﺑﺎ ﺁﺭﺍﻣﺶ ﺣﻞ ﮐﻨﺪ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺑﻨﺎﺑﺮﺍﻳﻦ، ﮐﺎﺭﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﮑﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺑﻪ ﺯﻭﺭ ﻛﺎﭘﺸﻦ ﺭﺍ ﺗﻦ ﻓﺮﺯﻧﺪﺵ ﻣﯽ‌ﮐﻨﺪ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺭﺍ ﺑﻪ ﺍﻭ ﻣﻨﻌﮑﺲ ﮐﻨﺪ:«ﻣﻴﺪﻭﻧﻢ ﮐﻪ ﻧﻤﯽ‌ﺧﻮﺍﯼ ﻛﺎﭘﺸﻨﺘﻮ ﺑﭙﻮﺷﯽ، ﻣﯽ‌ﺩﻭﻧﻢ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﻋﺼﺒﺎﻧﻴﺖ ﻣﯽ ﮐﻨﻪ!»
ﻓﻘﻂ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺑﺎﻟﻎ ﻧﻴﺴﺘﻨﺪ ﮐﻪ ﺗﻮﻫﻴﻦ ﺷﺨﺼﻴﺖ ﺁﻧﻬﺎ ﺭﺍ ﺧﺪﺷﻪ‌ﺩﺍﺭ ﻣﯽ‌ﮐﻨﺪ. ﮐﻮﺩﮐﺎﻥ «ﻧﺨﺴﺘﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﻧﻴﺰ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺩﻭﺳﺖ ﻧﺪﺍﺭﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺳﻌﯽ ﮐﻨﻴﻢ ﮐﻪ ﺗﺎ ﺁﻧﺠﺎ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﮐﻮﺩﮐﯽ ﮐﻪ ﺍﺯ ﺧﻮﺩ ﻣﺨﺎﻟﻔﺖ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﺍﺟﺘﻨﺎﺏ ﮐﻨﻴﻢ ﻭ ﺍﻣﺘﻴﺎﺯﺍﺗﯽ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﻢ ﺗﺎ ﺷﺨﺼﻴﺘﺶ ﺧﺪﺷﻪ‌ﺩﺍﺭ ﻧﺸﻮﺩ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺜﻼ ﻳﮏ ﻓﻌﺎﻟﻴﺖ ﺩﻳﮕﺮ ﺑﻪ ﺍﻭ ﭘﻴﺸﻨﻬﺎﺩ ﮐﻨﻴﺪ ﻭ ﺫﻫﻦ ﺍﻭ ﺭﺍ ﺍﺯ ﻣﺴﺌﻠﻪ ﻗﺒﻠﯽ ﺑﺮﮔﺮﺩﺍﻧﻴﺪ ﻭ ﻳﺎ ﺑﺎ ﻧﺎﺯﮐﺮﺩﻥ ﺍﻭ ﻣﺤﺒﺘﺶ ﺭﺍ ﺟﻠﺐ ﮐﻨﻴﺪ. ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﺍﻋﺼﺎﺑﺘﺎﻥ ﺍﺟﺎﺯﻩ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﺪﻫﺪ! ﺧﺼﻮﺻﺎ ﺑﺎ ﻣﻨﻌﮑﺲ ﮐﺮﺩﻥ ﺍﺣﺴﺎﺳﺎﺗﺶ، ﻣﺎﻧﻊ ﺍﺯ ﺧﺪﺷﻪ‌ﺩﺍﺭ ﺷﺪﻥ ﺷﺨﺼﻴﺘﺶ ﺷﻮﻳﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﻓﻘﻂ ﺑﻪ ﺍﻭ ﻣﯽ‌ﮔﻮﻳﻴﺪ:«ﻣﯽ‌ﺩﻭﻧﻢ ﮐﻪ ﺍﺻﻼ ﻣﻮﺍﻓﻖ ﻧﻴﺴﺘﯽ ﻭ ﺍﻳﻦ ﻫﻢ ﮐﺎﻣﻼ ﻃﺒﻴﻌﻴﻪ. ﻭﺍﻗﻌﺎ ﺧﻴﻠﯽ ﻣﺘﺄﺳﻔﻢ ﻭ ﻣﯽ‌ﺩﻭﻧﻢ ﭼﻪ ﺍﺣﺴﺎﺳﯽ ﺩﺍﺭﯼ، ﻭﻟﯽ ﻣﺘﺄﺳﻔﻢ ﺑﻬﺖ ﺑﮕﻢ ﮐﻪ ﺑﺎﻳﺪ ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﺍﺯ ﻣﻦ ﺍﻃﺎﻋﺖ ﮐﻨﯽ.»


ﭼﮕﻮﻧﻪ ﺑﺮﺧﻮﺭﺩﯼ ﺑﺎ ﺧﺸﻢ‌ﻫﺎﯼ ﺷﺪﻳﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ؟
ﺧﺸﻢ ﺷﺪﻳﺪ ﺗﻈﺎﻫﺮ ﻧﻬﺎﻳﯽ ﻣﻨﻔﯽ‌ﮔﺮﺍﻳﯽ ﺍﺳﺖ. ﻋﺼﺒﺎﻧﻴﺖ ﮐﻮﺩﮎ ﺁﻧﭽﻨﺎﻥ ﺍﺳﺖ ﮐﻪ ﺗﻨﻬﺎ ﮐﺎﺭﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﮑﻨﺪ ﮔﺮﻳﻪ ﻛﺮﺩﻥ ﺍﺳﺖ، ﻓﺮﻳﺎﺩ ﻭ ﺯﻭﺯﻩ ﺑﮑﺸﺪ ﻭ ﻳﺎ ﺧﻮﺩﺵ ﺭﺍ ﺭﻭﯼ ﺯﻣﻴﻦ ﺍﻧﺪﺍﺧﺘﻪ ﻭ ﺑﻪ ﻫﻤﻪ ﻃﺮﻑ ﻟﮕﺪ ﺑﻴﻨﺪﺍﺯﺩ. ﺍﮔﺮ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﻳﻦ ﻋﺼﺒﺎﻧﻴﺖ ﺗﺴﻠﻴﻢ ﺷﻮﻳﺪ ﻭ ﮐﺎﺭﯼ ﺭﺍ ﮐﻪ ﮐﻮﺩﮎ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ، ﺍﻳﻦ ﮐﺎﺭﺵ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﻴﺪ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﻳﺪ ﮐﻪ ﺑﺮﺍﯼ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻥ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺯﻭﺯﻩ ﻭ ﻓﺮﻳﺎﺩ ﺑﮑﺸﺪ.

ﺍﮔﺮ ﺷﻤﺎ ﻫﻢ ﺧﻮﻧﺴﺮﺩﻳﺘﺎﻥ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺑﺪﻫﻴﺪ ﻭ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻳﺪ، ﻏﻀﺐ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﻴﺪ. ﺍﺯ ﺁﻥ ﺑﻪ ﺑﻌﺪ ﮐﻮﺩﮎ ﻣﯽ‌ﺩﺍﻧﺪ ﮐﻪ ﻫﺮﻭﻗﺖ ﻫﺮﭼﻪ ﺑﺨﻮﺍﻫﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺷﻤﺎ ﺑﮕﻴﺮﺩ. ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭ ﭼﻨﻴﻦ ﺣﺎﻟﺖ ﻋﺼﺒﺎﻧﻴﺘﯽ ﺍﺳﺖ ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﺑﺮﺍﯼ ﺍﻭ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﺪ ﻭ ﺑﺎ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺣﻮﺍﺱ ﺍﻭ ﺭﺍ ﭘﺮﺕ ﮐﻨﻴﺪ. ﺍﻭ ﺩﺭ ﺣﺎﻟﺘﯽ ﻧﻴﺴﺖ ﮐﻪ ﺑﻪ ﺣﺮﻑ‌ﻫﺎﯼ ﻣﻨﻄﻘﯽ ﻭ ﻋﺎﻗﻼﻧﻪ ﮔﻮﺵ ﮐﻨﺪ. ﺧﺼﻮﺻﺎ ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﮐﻪ ﺁﺗﺶ ﻏﻀﺐ ﺍﻭ ﺭﺍ ﺑﺎ ﺗﻬﺪﻳﺪ ﮐﺮﺩﻥ ﻭ ﺯﺩﻥ ﻳﮏ ﺩﺭﻟﻨﺒﺮﯼ ﺧﺎﻣﻮﺵ ﮐﻨﻴﺪ. ﻭﺍﻟﺪﻳﻨﯽ ﮐﻪ ﺑﻪ ﻓﺮﺯﻧﺪ ﻋﺼﺒﺎﻧﻴﺸﺎﻥ ﻣﯽ‌ﮔﻮﻳﻨﺪ:«ﺳﺎﮐﺖ، ﺍﮔﺮﻧﻪ ﺑﻪ ﺯﻭﺩﯼ ﻣﯽ‌ﻓﻬﻤﯽ ﮐﻪ ﭼﺮﺍ ﮔﺮﻳﻪ ﻣﯽ‌ﮐﻨﯽ.» ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺨﻮﺍﻫﻨﺪ ﺁﺗﺶ ﺭﺍ ﺑﺎ ﺑﻨﺰﻳﻦ ﺧﺎﻣﻮﺵ ﮐﻨﻨﺪ.
