ﺩﺭﺱ ﺩﻫﻢ- ﺩﻭ ﺗﺎ ﺳﻪ ﺳﺎﻟﮕﯽ، ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ

ﺳﻦ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺍﺯ ﺯﻣﺎﻥ ﺭﺍﻩ ﺍﻓﺘﺎﺩﻥ ﮐﻮﺩﮎ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ، ﻳﻌﻨﯽ ﺣﺪﻭﺩ ﻳﮏ ﺳﺎﻟﮕﯽ ﻭ ﺗﻘﺮﻳﺒﺎ ﺗﺎ ﺩﻭ ﺳﺎﻟﮕﯽ ﺍﺩﺍﻣﻪ ﺩﺍﺭﺩ. ﻣﺪﺕ ﻫﺮ ﺩﻭﺭﻩ‌ﺍﯼ ﺍﺯ ﺭﺷﺪ ﮐﻮﺩﮎ، ﻫﺮ ﭼﻨﺪ ﺍﺯ ﻧﻈﺮ ﭼﮕﻮﻧﮕﯽ ﺗﺴﻠﺴﻞ ﺑﻪ ﺗﺮﺗﻴﺒﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺟﺎ ﺁﻣﺪﻩ، ﺍﻣﺎ ﺍﺯ ﻧﻈﺮ ﻣﺪﺕ ﺯﻣﺎﻥ ﺍﺯ ﮐﻮﺩﮐﯽ ﺑﻪ ﮐﻮﺩﮎ ﺩﻳﮕﺮ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺮ ﮐﻮﺩﮐﯽ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﻭﻟﯽ ﻫﺮ ﮐﺪﺍﻡ ﺑﺎ ﺁﻫﻨﮓ ﺧﺎﺹ ﺧﻮﺩ، ﺑﻌﻀﯽ ﺳﺮﻳﻊ‌ﺗﺮ ﻭ ﺑﺮﺧﯽ ﺩﻳﮕﺮ ﺁﻫﺴﺘﻪ‌ﺗﺮ. ﭘﻨﺠﺎﻩ ﺩﺭﺻﺪ ﮐﻮﺩﮐﺎﻥ ﺩﻭ ﺳﺎﻟﻪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻭﺍﺭﺩ ﻣﺮﺣﻠﻪ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﻣﯽ‌ﺷﻮﻧﺪ، ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺑﻴﺴﺖ ﻭ ﭘﻨﺞ ﺩﺭﺻﺪ ﻗﺒﻞ ﺍﺯ ﺍﻳﻦ ﺳﻦ ﻭ ﺑﻴﺴﺖ ﻭ ﭘﻨﺞ ﺩﺭﺻﺪ ﺑﻘﻴﻪ ﻫﻨﻮﺯ ﻭﺍﺭﺩ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻧﺸﺪﻩ‌ﺍﻧﺪ. 

ﺑﻨﺎﺑﺮﺍﻳﻦ، ﺗﻮﺿﻴﺤﺎﺕ ﮐﻠﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﺭﻓﺘﺎﺭ ﻧﻤﻮﻧﻪ‌ﻭﺍﺭ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺩﺍﺩﻩ‌ﺍﻳﻢ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺮﺍﺣﻠﯽ ﮐﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺣﺘﻤﺎ ﻭ ﺩﻗﻴﻘﺎ ﺑﮕﺬﺭﺍﻧﺪ ﻓﺮﺽ ﻧﮑﻨﻴﺪ. ﺍﻳﻦ ﻣﺜﺎﻝ‌ﻫﺎ ﻓﻘﻂ ﺑﺮﺍﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻣﺴﻴﺮ ﺍﺣﺘﻤﺎﻟﯽ ﺗﻐﻴﻴﺮﺍﺗﯽ ﺍﺳﺖ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺧﻮﺍﻫﺪ ﮔﺬﺭﺍﻧﺪ. ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺗﺮﺗﻴﺒﯽ ﺭﺍ ﮐﻪ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﭘﺸﺖ ﺳﺮﻫﻢ ﻣﯽ‌ﺁﻳﻨﺪ ﺑﺪﺍﻧﻴﺪ. ﻭﻟﯽ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﻪ ﺻﻮﺭﺗﯽ ﺍﺳﺎﺳﯽ ﻣﻬﺮ ﺧﻮﺩﺵ ﺭﺍ ﺑﺎ ﺷﺨﺼﻴﺖ ﺧﻮﺩﺵ ﺩﺭ ﻫﺮ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﻣﺮﺍﺣﻠﯽ ﮐﻪ ﻣﯽ‌ﮔﺬﺭﺍﻧﺪ ﻣﯽ‌ﺯﻧﺪ.

ﺭﺷﺪ ﮐﻮﺩﮎ ﺭﺍﻩ ﻣﺴﺘﻘﻴﻤﯽ ﺭﺍ ﻃﯽ ﻧﻤﯽ‌ﮐﻨﺪ، ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺮ ﭼﻘﺪﺭ ﮐﻮﺩﮎ ﺑﺰﺭﮔﺘﺮ ﺷﻮﺩ ﺭﻓﺘﺎﺭ ﺍﻭ ﭘﺨﺘﻪ‌ﺗﺮ ﻧﻤﯽ‌ﺷﻮﺩ. ﺑﺮﻋﮑﺲ ﺭﺷﺪ ﺍﻭ ﻣﺮﺍﺣﻞ ﻣﺨﺘﻠﻒ ﺗﻌﺎﺩﻝ ﻭ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺩﺍﺭﺩ ﮐﻪ ﻳﮑﯽ ﭘﺲ ﺍﺯ ﺩﻳﮕﺮﯼ ﻓﺮﺍ ﻣﯽ‌ﺭﺳﻨﺪ. ﻣﺜﻼ ﻣﺮﺣﻠﻪ ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﻳﮏ ﺩﻭﺭﻩ ﺗﻌﺎﺩﻝ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺁﻥ ﺩﻭﺭﻩ ﻋﺪﻡ ﺗﻌﺎﺩﻝ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» (ﺗﻘﺮﻳﺒﺎ ﺩﻭ ﺗﺎ ﺳﻪ ﺳﺎﻟﮕﯽ) ﻓﺮﺍ ﻣﯽ‌ﺭﺳﺪ. ﺳﭙﺲ ﻣﺮﺣﻠﻪ ﺳﻪ‌ ﺳﺎﻟﮕﯽ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻣﺮﺣﻠﻪ ﺗﻌﺎﺩﻝ ﺍﺳﺖ ﻭ ﭘﺲ ﺍﺯ ﺁﻥ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﮐﻪ ﻳﮏ ﺩﻭﺭﻩ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺍﺳﺖ. ﺑﻌﺪ، ﺗﻘﺮﻳﺒﺎ ﺣﺪﻭﺩ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺩﻭﺑﺎﺭﻩ ﻳﮏ ﺩﻭﺭﻩ ﺗﻌﺎﺩﻝ ﻓﺮﺍ ﻣﯽ‌ﺭﺳﺪ.

ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﻣﺮﺣﻠﻪ ﮔﺬﺍﺭ ﺍﺳﺖ. ﺍﻭﻟﻴﻦ ﮔﺬﺍﺭﯼ ﮐﻪ ﺩﺭ ﺭﺷﺪ ﮐﻮﺩﮎ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﺯ ﺍﻳﻦ ﺟﻬﺖ ﺍﻳﻦ ﻧﺎﻡ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﮐﻪ ﺍﻳﻦ ﺩﻭﺭﺍﻥ ﺷﺒﺎﻫﺖ ﻓﺮﺍﻭﺍﻧﯽ ﺑﻪ ﺩﻭﺭﺍﻥ ﻭﺍﻗﻌﯽ ﺑﻠﻮﻍ (ﺑﻴﻦ۱۳-۱۹ﺳﺎﻟﮕﯽ) ﺩﺍﺭﺩ- ﮐﻪ ﻣﺎ ﺁﻥ ﺭﺍ «ﺩﻭﻣﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» ﻧﺎﻣﻴﺪﻩ‌ﺍﻳﻢ.

ﺳﻴﺰﺩﻩ ﺗﺎ ﻧﻮﺯﺩﻩ ﺳﺎﻟﮕﯽ ﻣﺮﺣﻠﻪ ﮔﺬﺍﺭ ﺍﺯ ﮐﻮﺩﮐﯽ ﺑﻪ ﺑﻠﻮﻍ ﺍﺳﺖ. ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺳﻴﺰﺩﻩ ﺳﺎﻟﮕﯽ ﮐﻮﺩﮎ ﺩﺭ ﻣﺤﺪﻭﺩﻩ‌ﺍﯼ ﮐﻪ ﻭﺍﻟﺪﻳﻨﺶ ﺑﺮﺍﯼ ﺍﻭ ﺗﻌﻴﻴﻦ ﮐﺮﺩﻩ‌ﺍﻧﺪ ﻋﻤﻞ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﻫﻨﻮﺯ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﭘﺨﺘﻪ ﻧﺸﺪﻩ ﺗﺎ ﺑﺘﻮﺍﻧﺪ ﺩﺭ ﻣﺤﺪﻭﺩﻩ‌ﺍﯼ ﮐﻪ ﺧﻮﺩﺵ ﺗﻌﻴﻴﻦ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺑﺮ ﺍﺳﺎﺱ ﻗﻮﺍﻧﻴﻨﯽ ﮐﻪ ﺧﻮﺩﺵ ﺑﻪ ﺁﻧﻬﺎ ﻣﻌﺘﻘﺪ ﺍﺳﺖ ﻋﻤﻞ ﮐﻨﺪ. ﻭﻟﯽ ﺩﺭ ﻣﺤﺪﻭﺩﻩ‌ﺍﯼ ﮐﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺗﻌﻴﻴﻦ ﮐﺮﺩﻩ‌ﺍﻧﺪ، ﺟﻮﺍﻥ ﻗﺒﻞ ﺍﺯ ﺑﻠﻮﻍ ﺑﻪ ﺳﻄﺢ ﻣﺸﺨﺼﯽ ﺍﺯ ﺗﻌﺎﺩﻝ ﺭﻭﺍﻧﯽ ﺭﺳﻴﺪﻩ ﺍﺳﺖ. ﺭﺳﻴﺪﻥ ﺑﻪ ﺗﻌﺎﺩﻝ ﺷﺨﺺ ﺑﺎﻟﻎ ﺑﺎ ﺷﮑﺴﺘﻦ ﺗﻌﺎﺩﻝ ﮐﻮﺩﮐﯽ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ. ﻫﺮﮔﺎﻩ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺟﻮﺍﻧﯽ ﮐﻪ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﺭﺍ ﭘﺸﺖ ﺳﺮﻣﯽ‌ﮔﺬﺍﺭﺩ ﺍﻳﻦ ﺭﺍ ﻓﻬﻤﻴﺪﻧﺪ، ﭘﻴﭽﻴﺪﮔﯽ ﻣﻨﺤﺮﻑ‌ﮐﻨﻨﺪﻩ ﺭﻓﺘﺎﺭ ﻳﮏ ﺗﺎﺯﻩ ﺑﺎﻟﻎ ﺑﯽ‌ﺩﺭﻧﮓ ﺑﺮﺍﻳﺸﺎﻥ ﻗﺎﺑﻞ ﺩﺭﮎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﺯ ﺍﻳﻦ ﻟﺤﻈﻪ ﺑﻪ ﺑﻌﺪ ﻭﺍﻟﺪﻳﻦ ﻣﯽ‌ﻓﻬﻤﻨﺪ ﮐﻪ ﭼﺮﺍ ﻳﮏ ﺗﺎﺯﻩ ﺑﺎﻟﻎ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﯽ ﻣﺤﺘﺎﺝ ﻧﻮﻉ ﺩﻳﮕﺮﯼ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﺍﺳﺖ ﻳﺎ ﭼﺮﺍ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻣﻮﻫﺎﻳﺶ ﺭﺍ ﺑﻪ ﻣﺪﻝ ﺩﻳﮕﺮﯼ ﺁﺭﺍﻳﺶ ﮐﻨﺪ ﻭ ﻳﺎ ﻃﻮﺭ ﺩﻳﮕﺮﯼ ﻟﺒﺎﺱ ﺑﭙﻮﺷﺪ ﻭ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺭﻓﺘﺎﺭﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﻣﻌﻤﻮﻻ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺭﻭﯼ ﺁﻥ ﺻﺤﻪ ﻧﻤﯽ‌ﮔﺬﺍﺭﻧﺪ.

ﻻﺯﻣﻪ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﮐﻪ ﻣﻤﻠﻮ ﺍﺯ ﮐﺸﻤﮑﺶ ﻭ ﺁﺷﻔﺘﮕﯽ ﺍﺳﺖ، ﻧﻔﯽ ﻭ ﻋﺼﻴﺎﻥ ﺍﺳﺖ. ﺩﺭﮎ ﺣﺎﻻﺕ ﻣﺜﺒﺖ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﮐﻪ ﻣﺎ ﺁﻥ ﺭﺍ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﻧﺎﻣﻴﺪﻩ‌ﺍﻳﻢ، ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺩﺍﺭﺩ. ﻣﺴﺎﻟﻪ ﺭﺍ ﺍﺯ ﺩﻳﺪ ﺗﺎﺯﻩ‌ﺍﯼ ﺑﻨﮕﺮﻳﻢ: ﻣﺜﻼ ﮐﻮﺩﮐﯽ ﺭﺍ ﮐﻪ ﺑﻴﻦ ﺷﺎﻧﺰﺩﻩ ﺗﺎ ﻫﺠﺪﻩ ﻣﺎﻩ ﺩﺍﺭﺩ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﺍﻭ ﻫﻨﻮﺯ ﻳﮏ ﻧﻮﺯﺍﺩ ﺍﺳﺖ. ﭼﮕﻮﻧﻪ ﻃﺒﻴﻌﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻭ ﺭﺍ ﺗﺒﺪﻳﻞ ﺑﻪ «ﺳﻪ ﺳﺎﻟﻪ‌ﺍﯼ»ﮐﻨﺪ ﮐﻪ ﺷﺨﺼﻴﺘﺶ ﻧﺴﺒﺖ ﺑﻪ ﻳﮏ ﻧﻮﺯﺍﺩ ﭘﺨﺘﮕﯽ ﺧﺎﺹ ﻳﮏ ﮐﻮﺩﮎ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ؟ ﻣﺎﺩﺭ ﻃﺒﻴﻌﺖ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻓﻘﻂ ﺑﺎ ﺷﮑﺴﺘﻦ ﺗﻌﺎﺩﻟﯽ ﮐﻪ ﺍﻭ ﺑﻪ ﺁﻥ ﻧﺎﺋﻞ ﺁﻣﺪﻩ ﺑﻮﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻧﺠﺎﻡ ﺑﺮﺳﺎﻧﺪ. ﻫﻤﻴﺸﻪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺮﭼﻨﺪ ﺍﻳﻦ ﺭﻓﺘﺎﺭ ﻣﻨﻔﯽ ﻭ ﻣﺨﺎﻟﻔﺖ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﺸﮑﻼﺕ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﻪ ﺑﺎﺭ ﻣﯽ‌ﺁﻭﺭﺩ، ﻭﻟﯽ ﺍﻳﻦ ﻣﺮﺣﻠﻪ، ﻣﺮﺣﻠﻪ ﻣﺜﺒﺘﯽ ﺍﺯ ﺭﺷﺪ ﺍﻭﺳﺖ.  ﺑﺪﻭﻥ ﮔﺬﺭﺍﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﻭ ﺩﺭ ﺗﻌﺎﺩﻝ ﻧﻮﺯﺍﺩﯼ ﺑﺎﻗﯽ ﻣﯽ‌ﻣﺎﻧﺪ.

ﺩﺭ ﺍﻳﻦ ﺳﻦ ﮐﻮﺩﮎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﺧﻮﺑﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﻭﺍﻗﻌﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺭﺍﺑﻄﻪ ﮔﺮﻭﻫﯽ ﺑﺎ ﻣﺸﺎﺑﻬﺎﻥ ﺧﻮﺩ ﺑﺮﻗﺮﺍﺭ ﮐﻨﺪ. ﺍﻭ ﻫﻨﻮﺯ ﻣﺤﺘﺎﺝ ﻣﺎﺩﺭﺵ ﺍﺳﺖ. ﻣﺎﺩﺭ ﺧﻮﺭﺷﻴﺪﯼ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺳﻴﺎﺭﻩ ﺟﻮﺍﻥ ﺑﻪ ﺩﻭﺭ ﺁﻥ ﻣﯽ‌ﮔﺮﺩﺩ.

ﮐﻮﺩﮎ ﻣﻌﻤﻮﻻ ﺳﻤﺞ ﻭ ﻏﻴﺮﻗﺎﺑﻞ ﺍﻧﻌﻄﺎﻑ ﺍﺳﺖ. ﺗﻮﻗﻊ ﺩﺍﺭﺩ ﺗﻤﺎﻡ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﻳﺶ ﺑﻼﻓﺎﺻﻠﻪ ﺑﺮﺁﻭﺭﺩﻩ ﺷﻮﻧﺪ. ﺑﺎ ﻣﺼﺎﻟﺤﻪ ﻣﻴﺎﻧﻪ‌ﺍﯼ ﻧﺪﺍﺭﺩ ﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻭﺍﻗﻌﻴﺎﺕ ﺗﻄﺒﻴﻖ ﺩﻫﺪ. ﻫﻴﭻ ﻳﮏ ﺍﺯ ﭼﻴﺰﻫﺎﯼ ﻣﻮﺟﻮﺩ ﻧﺒﺎﻳﺪ ﺗﻐﻴﻴﺮ ﮐﻨﻨﺪ. ﺍﮔﺮ ﺑﺎﺑﺎ ﻳﺎ ﻣﺎﻣﺎﻥ ﮐﻠﻤﻪ‌ﺍﯼ ﺩﺭ ﺩﺍﺳﺘﺎﻥ ﻭ ﻳﺎ ﺁﻭﺍﺯﯼ ﮐﻪ ﺍﻭ ﻣﯽ‌ﺷﻨﺎﺳﺪ ﺍﺷﺘﺒﺎﻫﯽ ﺑﮕﻮﻳﻨﺪ ﻳﺎ ﻓﺮﺍﻣﻮﺵ ﮐﻨﻨﺪ ﺑﺎ ﺧﺸﻮﻧﺖ ﺍﻋﺘﺮﺍﺽ ﻣﯽ‌ﮐﻨﺪ. ﺗﻤﺎﻡ ﺍﻣﻮﺭ ﻋﺎﺩﯼ ﺧﺎﻧﻪ ﺑﺮﺍﯼ ﺍﻭ ﺣﮑﻢ ﻭﻗﺎﻳﻊ ﺩﻧﺒﺎﻝ ﻫﻤﯽ ﺭﺍ ﺩﺍﺭﻧﺪ ﮐﻪ ﻗﻮﺍﻧﻴﻦ ﺁﻥ ﻏﻴﺮﻗﺎﺑﻞ ﺗﻐﻴﻴﺮﻧﺪ ﻭ ﺑﻪ ﻧﻈﺮ ﺍﻭ ﺍﻳﻦ ﻗﻮﺍﻧﻴﻦ ﺑﺎﻳﺪ ﻣﻮ ﺑﻪ ﻣﻮ ﺍﺟﺮﺍ ﺷﻮﻧﺪ. ﻭﻟﯽ ﮐﻮﺩﮎ ﺧﻮﺩﺵ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﻳﮏ ﻣﺮﺗﺒﻪ ﺗﻐﻴﻴﺮ ﺩﻫﺪ. ﺁﺏ ﻣﻴﻮﻩ‌ﺍﺵ ﺭﺍ ﺩﺭ ﻟﻴﻮﺍﻥ ﺑﺪﻫﻴﺪ، ﺩﺭ ﻓﻨﺠﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ. ﺩﻓﻌﻪ ﺩﻳﮕﺮ ﺩﺭ ﻓﻨﺠﺎﻥ ﺑﺪﻫﻴﺪ، ﺍﻳﻦ ﺩﻓﻌﻪ ﺩﺭ ﻟﻴﻮﺍﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ.

ﮐﻮﺩﮎ ﺳﻠﻄﻪ‌ﻃﻠﺐ ﺍﺳﺖ. ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺩﺳﺘﻮﺭ ﺑﺪﻫﺪ. ﺧﻮﺩﺵ ﺭﺍ ﻓﺮﻣﺎﻧﺮﻭﺍﯼ ﺧﺎﻧﻪ ﻣﯽ‌ﺩﺍﻧﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﺎﺭﻫﺎﻳﯽ ﺑﮑﻨﺪ ﮐﻪ ﮐﺎﻣﻼ ﻣﺸﻬﻮﺩ ﺍﺳﺖ ﮐﻪ ﻗﺎﺩﺭ ﺑﻪ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻧﺶ ﻧﻴﺴﺖ (ﻣﺜﻞ ﺑﺴﺘﻦ ﺑﻨﺪ ﮐﻔﺸﺶ) ﻭ ﺑﺎ ﺧﺸﻨﻮﻧﺖ ﮐﻤﮏ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺭﺩ ﻣﯽ‌ﮐﻨﺪ. ﺳﭙﺲ ﺑﻌﺪ ﺍﺯ ﺍﻳﻦ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﺪ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ، ﮔﺮﻳﻪ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﻭ ﺣﺘﯽ ﺷﻤﺎ ﺭﺍ ﻣﻼﻣﺖ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﮐﻤﮑﺶ ﻧﻤﯽ‌ﮐﻨﻴﺪ.

ﺍﻳﻦ ﺳﻦ ﻫﻤﭽﻨﻴﻦ ﺳﻦ ﺗﺎﺛﺮﺍﺕ ﺷﺪﻳﺪ، ﺁﺷﻔﺘﮕﯽ ﻭ ﮐﺸﻤﮑﺶ ﻭ ﺧﺸﻢ‌ﻫﺎﯼ ﺷﺪﻳﺪ ﺍﺳﺖ. ﺍﻳﻦ ﻣﺮﺣﻠﻪ، ﻣﺮﺣﻠﻪ ﺍﻓﺮﺍﻁ ﻭ ﺗﻔﺮﻳﻂ ﺍﺳﺖ. ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﻏﺎﻟﺒﺎ ﺍﻧﺘﺨﺎﺑﯽ ﺳﺎﺩﻩ ﻭ ﻣﺸﺨﺺ ﻭ ﺍﻳﺴﺘﺎﺩﻥ ﺑﺮ ﺳﺮ ﺁﻥ ﻣﺸﮑﻞ ﺍﺳﺖ. ﺍﻭ ﻣﺮﺩﺩ ﺍﺳﺖ ﻭ ﺑﻴﻦ ﺍﺣﺴﺎﺳﺎﺕ ﮐﺎﻣﻼ ﻣﺘﻀﺎﺩ ﮐﻠﻪ ﻣﻌﻠﻖ ﻣﯽ‌ﺯﻧﺪ، ﺍﺣﺴﺎﺱ « ﺍﻳﻨﻮ ﻣﯽ‌ﺧﻮﺍﻡ» ﻭ « ﺍﻳﻨﻮ ﻧﻤﯽ‌ﺧﻮﺍﻡ»، ﻭ «ﻣﯽ‌ﺧﻮﺍﻡ ﺍﻳﻦ ﮐﺎﺭﻭ ﺑﮑﻨﻢ» ﻭ «ﻧﻤﯽ‌ﺧﻮﺍﻡ ﺍﻳﻦ ﮐﺎﺭﻭ ﺑﮑﻨﻢ». ﮐﻮﭼﮑﺘﺮﻳﻦ ﭼﻴﺰ ﻣﺜﻞ ﻣﺜﻼ ﺧﺮﻳﺪﻥ ﺑﺴﺘﻨﯽ ﺑﺮﺍﯼ ﺍﻭ ﻭ ﺍﻳﻦ ﮐﻪ ﻭﺍﺭﺩ ﻣﻐﺎﺯﻩ ﺷﻮﻳﺪ ﺗﺎ ﺁﻥ ﺭﺍ ﺑﺮﺍﻳﺶ  ﺑﺨﺮﻳﺪ ﺍﻭ ﺭﺍ ﺩﭼﺎﺭ ﺗﺮﺩﻳﺪ ﺩﺭ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﻣﯽ‌ﮐﻨﺪ.

ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺗﻤﺎﻡ ﺯﻣﻴﻨﻪ‌ﻫﺎ ﺭﮐﻮﺭﺩ ﺭﺍ ﺑﺸﮑﻨﺪ. ﻣﺜﻼ ﺑﻪ ﻃﻮﺭ ﺗﻤﺎﻡ ﻧﺸﺪﻧﯽ ﻳﮏ ﮐﺎﺭ ﺭﺍ ﺗﮑﺮﺍﺭ ﮐﻨﺪ ﻳﺎ ﺟﻤﻠﻪ ﻳﺎ ﮐﻠﻤﻪ‌ﺍﯼ ﺭﺍ ﺁﻥ‌ﻗﺪﺭ ﺗﮑﺮﺍﺭ ﮐﻨﺪ ﮐﻪ ﻣﺎﺩﺭ ﺑﻪ ﺧﺎﻃﺮ ﺍﻳﻦ ﮐﺎﺭﺵ ﺳﺮﺵ ﻓﺮﻳﺎﺩ ﺑﮑﺸﺪ. ﭼﻴﺰﻫﺎﯼ ﺟﺪﻳﺪ ﺭﺍ ﺧﻴﻠﯽ ﺳﺨﺖ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺍﻭ ﻗﺒﻮﻻﻧﺪ. ﻣﺜﻞ ﻟﺒﺎﺱ ﻧﻮ ﻳﺎ ﻏﺬﺍﻳﯽ ﮐﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻧﺨﻮﺭﺩﻩ ﺍﺳﺖ. ﺍﻭ ﻣﺤﺘﺎﺝ ﺑﻪ ﺍﻣﻨﻴﺘﯽ ﺍﺳﺖ ﮐﻪ ﻗﺪﻳﻤﯽ ﻭ ﺷﻨﺎﺧﺘﻪ ﺷﺪﻩ ﺑﺮﺍﻳﺶ ﻣﯽ‌ﺁﻭﺭﺩ. ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺍﺳﺖ ﮐﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﺷﻮﻧﺪ ﺩﻭﺳﺖ ﺩﺍﺭﺩ: ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺑﺎﻳﺪ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﺸﺨﺺ ﻭ ﻫﻤﻴﺸﻪ ﻳﮑﺴﺎﻥ ﺍﻧﺠﺎﻡ ﺩﺍﺩ. ﮐﻮﺩﮎ ﺩﻭ ﺳﺎﻝ ﻭ ﻧﻴﻤﻪ ﺑﻪ ﺧﺎﻃﺮ ﺑﺪﺍﺧﻼﻗﻴﺶ ﻣﺸﻬﻮﺭ ﺍﺳﺖ. ﻭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺍﻳﻦ ﺑﺪﺍﺧﻼﻗﯽ ﺑﺎ ﺑﺪﺍﺧﻼﻗﯽ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺗﻼﻗﯽ ﮐﻨﺪ، ﻭﺍﻭﻳﻼ! ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻓﺮﺯﻧﺪﺷﺎﻥ ﮔﺬﺷﺖ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺍﻭ ﺧﻴﻠﯽ ﮐﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺷﺮﺍﻛﺘﯽ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺍﺯ ﭼﻴﺰﯼ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﺪ، ﺻﺒﺮ ﮐﻨﺪ ﻭ ﻳﺎ ﮔﺬﺷﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻭ ﺍﻣﺎ ﺩﺭﺑﺎﺭﻩ ﻭﺟﻪ ﻣﺜﺒﺖ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ  ﺳﺮﺳﺨﺖ، ﻣﻤﻠﻮ ﺍﺯ ﺷﻮﺭ ﻭ ﺷﻮﻕ ﻭ ﺍﻧﺮﮊﯼ ﻓﺮﺍﻭﺍﻥ ﺍﺳﺖ. ﭼﻨﺎﻧﭽﻪ ﻣﺎﺩﺭ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﮔﻴﺮﺩ، ﺍﺯ ﮐﺸﻒ ﺣﺎﻻﺕ ﻣﺨﺘﻠﻒ ﺷﺨﺼﻴﺖ ﺟﺎﻟﺐ ﻓﺮﺯﻧﺪﺵ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ. ﺯﻳﺮﺍ ﮐﻮﺩﮎ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﻭ ﺳﺎﻝ ﺑﻪ ﺭﻏﻢ ﺗﻤﺎﻡ ﺍﻓﺮﺍﻁ‌ﻫﺎﻳﺶ، ﺑﺎ ﺳﺎﺩﮔﯽ‌ﻫﺎ ﻭ ﺗﻌﺠﺒﺶ ﺩﺭ ﻣﻘﺎﺑﻞ ﺟﻬﺎﻧﯽ ﮐﻪ ﺁﻥ ﺭﺍ ﺗﺎﺯﻩ ﻭ ﺩﺳﺖ ﻧﺨﻮﺭﺩﻩ ﻣﯽ‌ﻳﺎﺑﺪ، ﺑﺎ ﺗﺼﻮﺭﺍﺗﺶ ﻭ ﺑﺎ ﺷﻮﺭ ﻭ ﺍﺷﺘﻴﺎﻗﺶ ﺑﺮﺍﯼ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻄﺒﻮﻉ ﻭ ﺩﻟﻔﺮﻳﺐ ﺑﺎﺷﺪ.
    
ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺍﺯ ﺭﺷﺪﺵ ﭼﻪ ﭼﻴﺰﯼ ﻣﯽ‌ﺁﻣﻮﺯﺩ؟ ﺑﯽ‌ﺷﮏ ﻣﺎﺩﺭ ﻳﮏ ﺗﻴﻦ‌ﺍﻳﺠﺮ ﻭ ﮐﻮﺩﮐﯽ ﮐﻪ ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ ﺭﺍ ﻣﯽ‌ﮔﺬﺭﺍﻧﻨﺪ ﺟﻮﺍﺏ ﻣﺸﺎﺑﻬﯽ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﻣﯽ ﺩﻫﻨﺪ: «ﻫﻴﭽﯽ! ﻏﻴﺮ ﺍﺯ ﺍﺫﻳﺖ ﮐﺮﺩﻥ ﻭ ﻣﺨﺎﻟﻔﺖ ﮐﺮﺩﻥ ﺑﺎ ﻣﻦ ﻫﻴﭽﯽ ﻳﺎﺩ ﻧﻤﯽ‌ﮔﻴﺮﻩ»، ﺧﻮﺩﻣﺎﻧﻴﻢ ﺍﻳﻦ ﺵِﮐﻮﻩ ﻣﺎﺩﺭﺍﻧﻪ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺻﺤﻴﺢ ﺍﺳﺖ.
ﺍﻣﺎ ﺍﮔﺮ ﻓﺮﺍﺗﺮ ﺍﺯ ﻧﺎﺭﺍﺣﺘﯽ‌ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﺭﺍ ﺑﻨﮕﺮﻳﺪ، ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎ ﻣﺨﺎﻟﻔﺖ ﮐﺮﺩﻥ ﺑﺎ ﺁﻧﭽﻪ ﺩﺭ ﺟﺎﻣﻌﻪ ﻣﻮﺭﺩ ﻗﺒﻮﻝ ﺍﺳﺖ ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺣﺎﻝ ﮐﺸﻒ ﺷﺨﺼﻴﺖ ﺧﻮﻳﺶ ﺍﺳﺖ. (ﺍﻳﻦ ﺩﻗﻴﻘﺎ ﻫﻤﺎﻥ ﺗﺠﺮﺑﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺑﻴﻦ ﺳﻴﺰﺩﻩ ﺗﺎ ﻧﻮﺯﺩﻩ ﺳﺎﻟﮕﯽ- ﺑﺎ ﺷﺪﺗﯽ ﮐﻤﺘﺮ- ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ...)

ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪﮐﻪ ﻧﻮﺯﺍﺩ ﺑﻪ «ﻣﻦ» ﺁﮔﺎﻫﯽ ﻧﺪﺍﺭﺩ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﻭ «ﻣﻦ» ﺭﺍ ﺍﺯ «ﻏﻴﺮﻣﻦ» ﺗﺸﺨﻴﺺ ﺩﻫﺪ ﻣﺪﺗﯽ ﻭﻗﺖ ﻻﺯﻡ ﺍﺳﺖ. ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺍﺯ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﮐﻪ ﺍﮐﻨﻮﻥ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﻭﻟﻴﻦ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺍﻭ ﻭﺍﻗﻌﺎ ﺷﺨﺼﻴﺖ ﻳﮑﺘﺎﯼ ﺧﻮﺩ ﺭﺍ ﮐﺴﺐ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻳﮑﯽ ﺍﺯ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﺍﻟﺰﺍﻣﺎ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻳﻦ ﻫﺪﻑ ﺑﺎﻳﺪ ﺍﻧﺠﺎﻡ ﺩﻫﺪ، ﺍﻳﻦ ﻣﺨﺎﻟﻔﺖ‌ﻫﺎ ﻭ ﮐﻨﺶ‌ﻫﺎﯼ ﻣﻨﻔﯽ ﺩﺭ ﻣﻘﺎﺑﻞ ﻭﺍﻟﺪﻳﻦ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﻪ ﮐﻮﺩﮎ ﺑﺘﻮﺍﻧﺪ ﻣﺸﺨﺺ ﮐﻨﺪ ﭼﻪ ﮐﺴﯽ ﻫﺴﺖ ﻭ ﭼﻪ ﭼﻴﺰﯼ ﻣﯽ‌ﺧﻮﺍﻫﺪ، ﺑﺎﻳﺪ ﺍﺯ ﻣﺮﺣﻠﻪ ﻣﺨﺎﻟﻔﺖ ﻭ ﻣﺒﺎﺭﺯﻩ ﻃﻠﺒﯽ ﺑﮕﺬﺭﺩ.
