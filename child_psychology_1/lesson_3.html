ﺩﺭﺱ ﺩﻭﻡ- ﺍﻫﻤﻴﺖ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﻧﻴﺎﺯﻫﺎﯼ ﻧﻮﺯﺍﺩ

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺳﻮ‌ﺗﻔﺎﻫﻢ‌ﻫﺎﻳﯽ ﺩﺭ ﻣﻮﺭﺩ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﮐﻮﺩﮎ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﻭ ﺍﺯ ﺍﻫﻤﻴﺖ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﮐﻮﺩﮎ ﮔﻔﺘﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻭ ﺩﺭﺱ‌ ﺑﻌﺪ ﺍﺯ ﺍﻫﻤﻴﺖ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﻧﻴﺎﺯﻫﺎﯼ ﻧﻮﺯﺍﺩ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﻭ ﺍﻳﻦ ﻧﻴﺎﺯﻫﺎ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﻢ.

ﺷﻨﺎﺧﺖ ﮐﻮﺩﮎ ﺍﺯ ﺧﻮﻳﺶ ﺍﺯ ﻟﺤﻈﻪ ﺗﻮﻟﺪ ﺁﻏﺎﺯ ﻣﯽ‌ﺷﻮﺩ. ﺑﻴﺎﻳﻴﺪ ﺍﻳﻦ ﺷﻨﺎﺧﺖ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻧﺎﺷﯽ ﺍﺯ ﺁﻥ ﺭﺍ ﺑﺎ ﻳﮏ ﻋﻴﻨﮏ ﻣﻘﺎﻳﺴﻪ ﮐﻨﻴﻢ. ﮐﻮﺩﮎ ﺑﺮﺍﯼ ﻫﺮ ﻳﮏ ﺍﺯ ﻣﺮﺍﺣﻞ ﺭﺷﺪﺵ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ، ﺷﻴﺸﻪ‌ﺍﯼ ﻣﻨﺎﺳﺐ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﺪ. ﺷﻴﺸﻪ ﻫﺮ ﺩﻭﺭﻩ ﺑﻪ ﺷﻴﺸﻪ ﻣﺮﺣﻠﻪ ﻗﺒﻠﯽ ﺍﺿﺎﻓﻪ ﻣﯽ‌ﺷﻮﺩ.

ﺣﺎﻝ ﺑﻴﺎﻳﻴﺪ ﺷﻴﺸﻪ‌ﻫﺎﯼ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ. ﺍﻳﻦ ﺩﻭﺭﻩ ﺍﺯ ﻟﺤﻈﻪ ﺗﻮﻟﺪ ﺁﻏﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺗﺎ ﺯﻣﺎﻥ ﺭﺍﻩ ﺍﻓﺘﺎﺩﻥ ﻧﻮﺯﺍﺩ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﻳﺎﺑﺪ. ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮ ﮐﻮﺩﮐﺎﻥ، ﺍﻳﻦ ﺩﻭﺭﻩ ﻳﮏ ﺳﺎﻝ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺑﺮﺧﯽ ﮐﻪ ﺯﻭﺩﺗﺮ ﺭﺍﻩ ﻣﯽ‌ﺍﻓﺘﻨﺪ ﻧﻪ ﻣﺎﻩ ﻭ ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﺷﺎﻧﺰﺩﻩ ﻣﺎﻩ ﻭ ﺑﺮﺍﯼ ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﺑﻴﺸﺘﺮ ﻃﻮﻝ ﻣﯽ‌ﮐﺸﺪ.

ﺑﻌﻀﯽ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﻓﻜﺮ ﻣﯽ‌ﮐﻨﻨﺪ: «ﺍﻭﻥ ﮐﻪ ﭼﻴﺰﯼ ﻧﻤﯽ‌ﻓﻬﻤﻪ. ﻧﻮﺯﺍﺩﻩ ﻭ ﺑﻴﺸﺘﺮ ﻭﻗﺘﺎ ﺧﻮﺍﺑﻪ. ﻭﻗﺘﯽ ﺑﻴﺪﺍﺭ ﻣﯽ‌ﺷﻪ ﺷﻴﺮﺷﻮ ﻣﯽ‌ﺧﻮﺭﻩ، ﮐﻬﻨﻪ‌ﺍﺵ ﺭﻭ ﻋﻮﺽ ﻣﯽ‌ﮐﻨﻢ ﻭ ﺣﻤﺎﻣﺶ ﻣﯽ‌ﻛﻨﻢ. ﺯﻧﺪﮔﯽ ﻧﻮﺯﺍﺩ ﺗﻮ ﻫﻤﻴﻦ ﺧﻼﺻﻪ ﻣﯽ‌ﺷﻪ. ﻫﻨﻮﺯ ﮐﻮﭼﮏ‌ﺗﺮ ﺍﺯ ﺍﻭﻧﻪ ﮐﻪ ﺑﺘﻮﻧﻪ ﭼﻴﺰ ﺯﻳﺎﺩﯼ ﻳﺎﺩ ﺑﮕﻴﺮﻩ. ﺑﻌﺪﺍ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﻩ ﺑﻨﺸﻴﻨﻪ، ﺗﻮ ﺗﺨﺘﺶ ﺑﺎﺯﯼ ﮐﻨﻪ ﻭ ﭼﺎﺭ ﺩﺳﺖ ﻭ ﭘﺎ ﺭﺍﻩ ﺑﺮﻩ، ﻭﻟﯽ ﻫﻨﻮﺯ ﺑﺮﺍﯼ ﻳﺎﺩ ﮔﺮﻓﺘﻦ ﭼﻴﺰﯼ ﺧﻴﻠﯽ ﮐﻮﭼﻴﮑﻪ.»

ﺍﺷﺘﺒﺎﻫﯽ ﺑﺰﺭﮒ‌ﺗﺮ ﺍﺯ ﺍﻳﻦ ﻣﻤﮑﻦ ﻧﻴﺴﺖ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺍﺯ ﻫﻤﺎﻥ ﻟﺤﻈﻪ ﺍﻭﻝ ﺗﻮﻟﺪ ﺷﺮﻭﻉ ﺑﻪ ﻳﺎﺩﮔﻴﺮﯼ ﻣﯽ‌ﮐﻨﺪ. ﻃﯽ ﺍﻭﻟﻴﻦ ﺳﺎﻝ، ﻣﻬﻢ‌ﺗﺮﻳﻦ ﭼﻴﺰﯼ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﮐﺴﺐ ﻣﯽ‌ﮐﻨﺪ، ﺩﻳﺪ ﺍﺳﺎﺳﯽ ﻧﺴﺒﺖ ﺑﻪ ﺯﻧﺪﮔﯽ ﺍﺳﺖ. ﺍﻭ ﺍﺯ ﺩﻳﺪﮔﺎﻩ ﻧﻮﺯﺍﺩﺍﻧﻪ ﺧﻮﺩ ﻓﻠﺴﻔﻪ ﺣﻴﺎﺕ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺑﻨﻴﺎﺩﻳﻨﯽ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﺣﻘﻴﻘﺖ ﺯﻧﺪﮔﯽ ﺗﺪﻭﻳﻦ ﻣﯽ‌ﮐﻨﺪ. ﺍﻭ ﺩﺭ ﺣﺎﻝ ﭘﺎﻳﻪ‌ﺭﻳﺰﯼ ﺍﺣﺴﺎﺳﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺁﻳﻨﺪﻩ ﻧﺴﺒﺖ ﺑﻪ ﺯﻧﺪﮔﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ، ﺣﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻧﻈﺮ ﺍﻭ ﺯﻧﺪﮔﯽ ﺳﺮﺷﺎﺭ ﺍﺯ ﺷﺎﺩﯼ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﺎﺷﺪ ﻭ ﻳﺎ ﺑﺮﻋﮑﺲ ﻣﻤﻠﻮ ﺍﺯ ﻋﺪﻡ ﺍﻋﺘﻤﺎﺩ ﻭ ﺑﺪﺑﺨﺘﯽ.
ﻣﺤﻴﻄﯽ ﮐﻪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﮐﻨﻴﺪ، ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ ﺍﻋﺘﻤﺎﺩ ﻳﺎ ﻋﺪﻡ ﺍﻋﺘﻤﺎﺩﯼ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻭ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﺩ. ﺍﻳﻦ ﻣﺤﻴﻂ ﻋﻨﺼﺮﯼ ﺍﺯ ﻧﻈﺎﻡ ﺑﻴﻨﺸﯽ ﺍﻭ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﻭﺭﺍﯼ ﺁﻥ ﺑﻪ ﺟﻬﺎﻥ ﻣﯽ‌ﻧﮕﺮﺩ. ﺍﮔﺮ ﺩﺭ ﻃﻮﻝ ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﻋﻴﻨﮑﯽ ﺧﻮﺵ‌ﺑﻴﻨﺎﻧﻪ ﺑﻪ ﭼﺸﻢ ﺑﺰﻧﺪ، ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﺰﺭﮒ ﺷﻮﺩ ﺷﺨﺼﯽ ﺧﻮﺵ‌ﺑﻴﻦ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻭﻟﯽ ﺍﮔﺮ ﺑﺮﻋﮑﺲ ﻋﻴﻨﮑﯽ ﺑﺪﺑﻴﻨﺎﻧﻪ ﺑﻪ ﭼﺸﻢ ﺑﺰﻧﺪ، ﺩﺭ ﺯﻧﺪﮔﯽ ﺗﺒﺪﻳﻞ ﺑﻪ ﺍﻧﺴﺎﻧﯽ ﻋﻤﻴﻘﺎ ﺑﺪﺑﻴﻦ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺍﻭﻟﻴﻦ ﺳﺎﻝ ﺯﻧﺪﮔﯽ ﻧﺴﺒﺖ ﺑﻪ ﻣﺮﺍﺣﻞ ﺑﻌﺪﯼ، ﺍﺯ ﻧﻈﺮ ﺭﺷﺪ ﺭﻭﺍﻧﯽ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺩﺍﺭﺩ. ﺯﻳﺮﺍ ﻧﻮﺯﺍﺩ ﺑﺮﺍﯼ ﺯﻧﺪﮔﯽ ﮐﺎﻣﻼ ﻭﺍﺑﺴﺘﻪ ﺑﻪ ﺷﻤﺎﺳﺖ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﻪ ﻓﺮﺯﻧﺪ ﻧﻮﺯﺍﺩﺗﺎﻥ ﺣﺪﺍﮐﺜﺮ ﺍﻣﮑﺎﻧﺎﺗﺶ ﺭﺍ ﺭﺷﺪ ﺩﻫﺪ ﭼﻪ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﮑﻨﻴﺪ؟ ﺍﮔﺮ ﺷﻤﺎ، ﻭﺍﻟﺪﻳﻦ، ﺑﺘﻮﺍﻧﻴﺪ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺍﺳﺎﺳﯽ ﺍﻭ ﺭﺍ ﺑﺮﺁﻭﺭﺩﻩ ﮐﻨﻴﺪ ﻭ ﺍﻭ ﺭﺍﺿﯽ ﺑﺎﺷﺪ، ﺍﻭ ﺑﻪ ﺣﺪﺍﮐﺜﺮ ﺷﮑﻮﻓﺎﻳﯽ ﺧﻮﺩ ﺩﺳﺖ ﻣﯽ‌ﻳﺎﺑﺪ. ﭘﺲ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﺍﻳﻦ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺍﺳﺎﺳﯽ ﭼﻪ ﻫﺴﺘﻨﺪ.


‌ﺗﻐﺬﻳﻪ
ﻣﺒﺮﻡ‌ﺗﺮﻳﻦ ﺍﺣﺘﻴﺎﺟﺎﺕ ﻧﻮﺯﺍﺩ ﺗﻐﺬﻳﻪ ﺍﺳﺖ. ﻧﻮﺯﺍﺩ ﮔﺮﺳﻨﮕﯽ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻭﺍﻗﻌﻴﺘﯽ ﻋﻤﻴﻖ ﻭ ﺁﻧﯽ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ. ﺯﻣﺎﻧﯽ ﮐﻪ ﻫﻨﻮﺯ ﺧﻴﻠﯽ ﮐﻮﭼﮏ ﺍﺳﺖ، ﻣﺜﻼ ﺣﺪﻭﺩ ﻳﮏ ﻣﺎﻫﮕﯽ ﺍﺯ ﺩﺭﺩ ﮔﺮﺳﻨﮕﯽ ﺍﺯ ﺧﻮﺍﺏ ﺑﻴﺪﺍﺭ ﻣﯽ‌ﺷﻮﺩ. ﻭﻗﺘﯽ ﮐﻪ ﺷﻴﺮﺵ ﺭﺍ ﺧﻮﺭﺩ ﺩﻭﺑﺎﺭﻩ ﻣﯽ‌ﺧﻮﺍﺑﺪ ﺗﺎ ﺍﻳﻦ ﮐﻪ ﺩﻭﺑﺎﺭﻩ ﺍﺯ ﮔﺮﺳﻨﮕﯽ ﺍﺯ ﺧﻮﺍﺏ ﺑﻴﺪﺍﺭ ﺷﻮﺩ.  ﺑﻌﺪ، ﺩﻭﺭﻩ‌ﻫﺎﯼ ﺑﻴﺪﺍﺭﯼ ﻃﻮﻻﻧﯽ‌ﺗﺮ ﻣﯽ‌ﺷﻮﺩ. ﺩﻳﮕﺮ ﺑﻼﻓﺎﺻﻠﻪ ﺑﻌﺪ ﺍﺯ ﺷﻴﺮ ﺧﻮﺭﺩﻥ ﻧﻤﯽ‌ﺧﻮﺍﺑﺪ. ﭼﮕﻮﻧﻪ ﺍﻳﻦ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺗﻐﺬﻳﻪ ﮐﻮﺩﮎ ﺭﺍ ﺑﺮ ﻃﺮﻑ ﺳﺎﺯﻳﻢ؟ ﺧﻴﻠﯽ ﺳﺎﺩﻩ: ﺭﺿﺎﻳﺖ ﺧﺎﻃﺮ ﺍﻭ ﺭﺍ ﻓﺮﺍﻫﻢ ﺁﻭﺭﻳﺪ. ﺍﻳﻦ ﺑﻪ ﻧﻈﺮ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺩﺭ ﻭﺍﻗﻊ ﺳﺎﺩﻩ ﻫﻢ ﻫﺴﺖ، ﺍﻣﺎ ﺟﺎﻣﻌﻪ ﺑﻴﻬﻮﺩﻩ ﺁﻥ ﺭﺍ ﺗﺒﺪﻳﻞ ﺑﻪ ﻣﺴﺎﻟﻪ ﭘﻴﭽﻴﺪﻩ‌ﺍﯼ ﮐﺮﺩﻩ ﺍﺳﺖ.


ﺷﻴﺮ ﻣﺎﺩﺭ ﻳﺎ ﺷﻴﺮ ﺧﺸﮏ: ﻣﺴﺎﻟﻪ‌ﺍﯼ ﮐﻪ ﺑﺪ ﻣﻄﺮﺡ ﺷﺪﻩ
ﺑﺎ ﻣﺒﺎﺣﺜﻪ ﻣﻴﺎﻥ ﻃﺮﻓﺪﺍﺭﺍﻥ ﻭ ﻣﺨﺎﻟﻔﺎﻥ ﺷﻴﺮ ﻣﺎﺩﺭ ﻳﺎ ﺷﻴﺮ ﺧﺸﮏ ﺩﺍﺩﻥ ﺑﻪ ﻧﻮﺯﺍﺩ، ﺷﺮﺍﻳﻂ ﺭﺍ ﻣﻐﺸﻮﺵ ﮐﺮﺩﻩ‌ﺍﻳﻢ. ﺑﺤﺜﯽ ﺩﺍﺋﻤﯽ ﻣﻴﺎﻥ ﺩﮐﺘﺮﻫﺎ، ﭘﺮﺳﺘﺎﺭﻫﺎ ﻭ ﺍﻃﺮﺍﻓﻴﺎﻥ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻣﺜﻼ ﮔﺎﻫﯽ ﺍﻭﻗﺎﺕ ﺣﺘﯽ ﺍﻳﻦ ﮔﻮﻧﻪ ﺑﺤﺚ‌ﻫﺎ ﻣﻮﺟﺐ ﻣﯽ‌ﺷﻮﺩ ﻣﺎﺩﺭ ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﮐﻪ ﺷﻴﺮ ﺧﻮﺩﺵ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﻧﻤﯽ‌ﺩﻫﺪ ﺍﺣﺴﺎﺱ ﮔﻨﺎﻩ ﮐﻨﺪ.
ﺍﻳﻦ ﻣﺠﺎﺩﻟﻪ ﻓﻘﻂ ﻳﮏ ﺟﻮﺍﺏ ﺩﺍﺭﺩ. ﺗﺼﻤﻴﻢ ﺩﺭ ﻣﻮﺭﺩ ﺍﻧﺘﺨﺎﺏ ﺭﻭﺵ ﺷﻴﺮ ﺩﺍﺩﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻓﻘﻂ ﺑﺎ ﺧﻮﺩ ﺷﻤﺎﺳﺖ. ﺑﺎ ﺍﻳﻦ‌ﻫﻤﻪ، ﺍﮔﺮ ﺑﻪ ﻧﻮﺯﺍﺩﺗﺎﻥ ﺷﻴﺮ ﺧﺸﮏ ﻣﯽ‌ﺩﻫﻴﺪ، ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﻭ ﺭﺍ ﻃﻮﺭﯼ ﺑﻐﻞ ﮐﻨﻴﺪ ﮐﻪ ﺍﻧﮕﺎﺭ ﺷﻴﺮ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺩﻫﻴﺪ ﻭ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﮐﻮﺩﮎ ﺭﺍ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﻧﻮﺍﺯﺵ ﮐﻨﻴﺪ.‌

ﺁﻳﺎ ﺷﻴﺮ ﺩﺍﺩﻥ ﺑﻪ ﻧﻮﺯﺍﺩ ﺑﺎﻳﺪ ﺗﺎﺑﻊ ﺳﺎﻋﺖ‌ﻫﺎﯼ ﺧﺎﺹ ﻭ ﻣﻨﻈﻢ ﺑﺎﺷﺪ ﻳﺎ ﻣﻄﺎﺑﻖ ﻣﻴﻞ ﻧﻮﺯﺍﺩ؟
ﺣﺎﻝ ﺑﺒﻴﻨﻴﻢ ﮐﻪ ﭼﻪ ﻣﻮﻗﻊ ﺑﺎﻳﺪ ﺑﻪ ﻧﻮﺯﺍﺩ ﺷﻴﺮ ﺩﺍﺩ؟ ﺟﻮﺍﺏ ﺍﻳﻦ ﺳﻮﺍﻝ ﻃﺒﻴﻌﺘﺎ ﺍﻳﻦ ﺍﺳﺖ: ﻫﺮ ﻭﻗﺖ ﮐﻪ ﺍﻭ ﮔﺮﺳﻨﻪ ﺍﺳﺖ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﺟﻮﺍﺏ ﺍﻳﻦ ﺳﻮﺍﻝ ﺳﺎﺩﻩ ﺩﺭ ﺯﻣﺎﻥ ﻣﺎ ﺑﻴﻬﻮﺩﻩ ﭘﻴﭽﻴﺪﻩ ﺷﺪﻩ ﺍﺳﺖ.

ﺯﻣﺎﻧﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺮﺩﻡ ﻓﮑﺮ ﻣﯽ‌ﮐﺮﺩﻧﺪ ﮐﻪ ﻫﺮ ﭼﻪ ﺯﻭﺩﺗﺮ ﻳﮏ ﺳﺮﯼ ﻋﺎﺩﺍﺕ ﺭﺍ ﺑﻪ ﻧﻮﺯﺍﺩ ﺑﻴﺎﻣﻮﺯﻧﺪ ﺑﻬﺘﺮ ﺍﺳﺖ. ﺑﻪ ﻧﻈﺮ ﺍﻳﻦ ﺩﺳﺘﻪ ﻣﻬﻢ‌ﺗﺮﻳﻦِ ﺍﻳﻦ ﻋﺎﺩﺍﺕ ﺩﺍﺩﻥِ ﺷﻴﺮ ﺑﭽﻪ ﺭﺍﺱ ﺳﺎﻋﺖ‌ﻫﺎﯼ ﻣﺸﺨﺺ ﺑﻮﺩ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﻮﺩ ﮐﻪ ﺩﮐﺘﺮﻫﺎ ﻣﻘﺮﺭ ﮐﺮﺩﻧﺪ ﮐﻪ ﺑﺎﻳﺪ ﻫﺮ ﺳﻪ ﻳﺎ ﭼﻬﺎﺭ ﺳﺎﻋﺖ ﻳﮑﺒﺎﺭ ﺑﻪ ﻧﻮﺯﺍﺩ ﺷﻴﺮ ﺩﺍﺩ. ﻣﺎﺩﺭﻫﺎ ﻣﯽ‌ﺑﺎﻳﺴﺖ ﻃﺒﻖ ﺍﻳﻦ ﺑﺮﻧﺎﻣﻪ ﺑﻪ ﻧﻮﺯﺍﺩﺷﺎﻥ ﺷﻴﺮ ﻣﯽ‌ﺩﺍﺩﻧﺪ. ﺍﻳﻦ ﺭﮊﻳﻢ ﻏﺬﺍﻳﯽ ﻭﺣﺸﺘﻨﺎﮎ ﻳﮑﯽ ﺍﺯ ﺍﺳﺎﺳﯽ‌ﺗﺮﻳﻦ ﺍﺻﻮﻝ ﭘﺮﻭﺭﺵ ﮐﻮﺩﮎ ﺭﺍ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺯﻳﺮ ﭘﺎ ﻣﯽ‌ﮔﺬﺍﺷﺖ: ﺍﺣﺘﺮﺍﻡ ﺑﻪ ﺷﺨﺼﻴﺖ ﮐﻮﺩﮎ.

ﻫﺮ ﻧﻮﺯﺍﺩ ﻓﺮﺩﯼ ﻣﻨﺤﺼﺮ ﺑﻪ ﻓﺮﺩ ﺍﺳﺖ ﻭ ﺍﺣﺘﻴﺎﺟﺎﺗﺶ ﺑﺮﺍﯼ ﺗﻐﺬﻳﻪ ﺑﺎ ﺩﻳﮕﺮﯼ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺣﺎﻝ ﭼﻪ ﻃﻮﺭ ﺗﻮﺍﻧﺴﺘﻪ‌ﺍﻳﻢ ﻓﮑﺮ ﮐﻨﻴﻢ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺮﻧﺎﻣﻪ‌ﺍﯼ ﻣﺸﺨﺺ ﺭﺍ ﺑﻪ ﺗﻤﺎﻡ ﻧﻮﺯﺍﺩﺍﻥ ﺗﺤﻤﻴﻞ ﮐﺮﺩ ﮐﻪ ﺑﺮﺍﯼ ﺗﻤﺎﻡ ﺁﻧﻬﺎ ﻣﻨﺎﺳﺐ ﺑﺎﺷﺪ؟ ﻣﺴﻠﻤﺎ ﺷﻴﺮﺩﺍﺩﻥ ﻧﻮﺯﺍﺩ ﻫﺮ ﺳﻪ ﻳﺎ ﭼﻬﺎﺭ ﺳﺎﻋﺖ ﻏﻠﻂ ﺍﺳﺖ. ﻣﻨﻄﻘﺎ ﻏﻴﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺑﺮﻧﺎﻣﻪ ﻣﻨﺎﺳﺐ ﺣﺎﻝ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﺎﺷﺪ. ﺯﻳﺮﺍ ﻧﻪ ﺗﻨﻬﺎ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻣﺘﻤﺎﻳﺰ ﺍﺯ ﺗﻤﺎﻡ ﮐﻮﺩﮐﺎﻥ ﺩﻳﮕﺮ ﺍﺳﺖ، ﺑﻠﮑﻪ ﺣﺘﯽ ﺍﺣﺘﻴﺎﺝ ﺍﻭ ﺑﻪ ﺗﻐﺬﻳﻪ ﺍﺯ ﺭﻭﺯﯼ ﺑﻪ ﺭﻭﺯ ﺩﻳﮕﺮ ﻓﺮﻕ ﻣﯽ‌ﮐﻨﺪ.

ﺑﺒﻴﻨﻴﻢ ﺍﮔﺮ ﺑﻪ ﻃﻔﻠﯽ ﺳﺮ ﺳﺎﻋﺎﺕ ﻣﺸﺨﺺ ﻏﺬﺍ ﺑﺪﻫﻨﺪ ﭼﻪ ﺑﻪ ﺳﺮﺵ ﻣﯽ‌ﺁﻳﺪ؟ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻃﻔﻞ ﮔﺮﺳﻨﻪ ﺍﺳﺖ ﺧﻮﺩ ﺭﺍ ﻣﺤﺮﻭﻡ ﺣﺲ ﻣﯽ‌ﮐﻨﺪ. ﻧﻮﺯﺍﺩ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮔﺮﺳﻨﻪ ﺍﺳﺖ، ﮔﺮﺳﻨﮕﯽ ﺑﺮﺍﯼ ﺍﻭ ﻧﻴﺮﻭﻳﯽ ﻣﺴﻠﻂ ﺑﺮ ﻫﻤﻪ ﭼﻴﺰ ﺍﺳﺖ. ﺍﺣﺴﺎﺱ ﮔﺮﺳﻨﮕﯽ ﺑﺮﺍﯼ ﺍﻭ ﻳﮏ ﮐﻞ ﺍﺳﺖ. ﺍﻭ ﻫﻴﭻ ﺗﺎﺧﻴﺮﯼ ﻧﻤﯽ‌ﭘﺬﻳﺮﺩ: ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺧﻮﺭﺩﻥ ﺩﺍﺭﺩ.

 ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮔﺮﺳﻨﻪ ﺍﺳﺖ ﻭ ﺑﻪ ﺍﻭ ﻏﺬﺍ ﻧﻤﯽ‌ﺩﻫﻴﺪ، ﺑﺮﺍﯼ ﺟﻠﺐ ﺗﻮﺟﻪ ﺷﻤﺎ ﺑﻪ ﮔﺮﺳﻨﮕﯽ‌ﺍﺵ ﺍﺯ ﺗﻨﻬﺎ ﺭﺍﻫﯽ ﮐﻪ ﺑﻠﺪ ﺍﺳﺖ، ﻳﻌﻨﯽ ﮔﺮﻳﻪ ﮐﺮﺩﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﺪ. ﻫﺮ ﻗﺪﺭ ﺯﻣﺎﻥ ﺑﮕﺬﺭﺩ ﻭ ﺑﻪ ﺍﻭ ﻏﺬﺍ ﻧﺪﻫﻴﺪ، ﻫﻤﺎﻥ ﻗﺪﺭ ﻓﺮﻳﺎﺩﻫﺎ ﺑﻠﻨﺪﺗﺮ ﻭ ﺧﺸﻦ‌ﺗﺮ ﻣﯽ ﺷﻮﺩ. ﮔﻮﻳﯽ ﻧﻮﺯﺍﺩ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﮔﻮﻳﺪ: «ﺑﻪ ﻋﺠﺐ ﺩﻧﻴﺎﻳﯽ ﺍﻭﻣﺪﻡ؟ ﻣﻦ ﻧﺸﺴﺘﻢ ﺍﻳﻨﺠﺎ، ﮔﺮﻳﻪ ﻭ ﺩﺍﺩ ﻭ ﺑﻴﺪﺍﺩ ﻣﯽ‌ﮐﻨﻢ ﮐﻪ ﺑﺎﺑﺎ ﻣﻦ ﮔﺮﺳﻨﻪ‌ﺍﻣﻪ ﻭﻟﯽ ﻛﺴﯽ ﺑﻪ ﻣﻦ ﻏﺬﺍ ﻧﻤﯽ‌ﺩﻩ!» ﻫﻤﻴﻦ ﻃﻮﺭ ﮐﻪ ﺩﻗﺎﻳﻖ ﻣﯽ‌ﮔﺬﺭﺩ ﻧﻮﻉ ﻓﺮﻳﺎﺩﻫﺎ ﻫﻢ ﻋﻮﺽ ﻣﯽ‌ﺷﻮﺩ: ﺑﻪ ﻋﻮﺽ ﺑﻴﺎﻥ ﺭﻭﺷﻦ ﻭ ﺭﺳﺎﯼ ﻳﮏ ﺍﺣﺘﻴﺎﺝ ﺳﺎﺩﻩ ﺑﻪ ﺧﻮﺭﺩﻥ، ﮔﺮﻳﻪ‌ﻫﺎ ﺗﺒﺪﻳﻞ ﺑﻪ ﻓﺮﻳﺎﺩﻫﺎﯼ ﺧﺸﻢ ﻭ ﻧﺎﺍﻣﻴﺪﯼ ﻣﯽ‌ﺷﻮﺩ.

ﺍﮔﺮ ﺷﻴﺮ ﺩﺍﺩﻥ ﺑﻪ ﻧﻮﺯﺍﺩ ﻃﺒﻖ ﺳﺎﻋﺖ‌ﻫﺎﯼ ﻣﺸﺨﺺ ﺑﺎﺷﺪ ﺍﻭ ﺑﻪ ﺗﺪﺭﻳﺞ ﮐﺸﻒ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﺗﻤﺎﻡ ﻧﻴﺮﻭ ﮔﺮﻳﻪ ﮐﻨﺪ ﺑﯽ ‌ﺍﻳﻦ ﮐﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﻗﻊ ﭼﻴﺰﯼ ﺑﻪ ﺍﻭ ﻧﺨﻮﺍﻫﻨﺪ ﺩﺍﺩ. ﻧﻮﺯﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﻭﻋﮑﺲ ﺍﻟﻌﻤﻞ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ: ﺧﺸﻢ ﻭ ﻳﺎ ﺑﺮﻋﮑﺲ ﺑﯽ‌ﻗﻴﺪﯼ، ﮔﻮﻳﯽ ﺩﻳﮕﺮ ﻫﻴﭻ ﺍﻣﻴﺪﯼ ﺑﻪ ﺍﺭﺿﺎﯼ ﺍﺣﺘﻴﺎﺟﺎﺗﺶ ﻧﺪﺍﺭﺩ. ﺍﻭ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ ﺧﺸﻤﺶ ﺭﺍ ﺳﺮﻛﻮﺏ ﮐﻨﺪ ﻭ ﺗﺴﻠﻴﻢ ﺧﻄﺮﻧﺎﮐﯽ ﺟﺎﯼ ﺍﻳﻦ ﺧﺸﻢ ﺭﺍ ﻣﯽ‌ﮔﻴﺮﺩ. ﻭﻟﯽ ﮐﻮﺩﮎ،  ﭼﻪ ﻫﻤﻴﺸﻪ ﺧﺸﻤﮕﻴﻦ ﺑﺎﺷﺪ ﻭ ﭼﻪ ﺷﻴﻮﻩ ﺗﺴﻠﻴﻢ ﻭ ﺑﯽ‌ﻗﻴﺪﯼ ﺭﺍ ﺍﺗﺨﺎﺫ ﮐﻨﺪ، ﺩﺭ ﻫﺮ ﺩﻭ ﺣﺎﻟﺖ ﻧﺘﻴﺠﻪ ﻳﮑﯽ ﺍﺳﺖ: ﺑﯽ‌ﺍﻃﻤﻴﻨﺎﻧﯽ ﺍﺳﺎﺳﯽ ﻧﺴﺒﺖ ﺑﻪ ﺯﻧﺪﮔﯽ.

ﺭﻭﺯ ﺑﻪ ﺭﻭﺯ ﺗﻌﺪﺍﺩ ﺑﻴﺸﺘﺮﯼ ﺍﺯ ﭘﺰﺷﮑﺎﻥ ﺑﻪ ﻣﺎﺩﺭﺍﻥ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺑﻪ ﻧﻮﺯﺍﺩﺷﺎﻥ ﻣﻨﺎﺳﺐ ﺗﻘﺎﺿﺎﯼ ﺍﻭ ﻏﺬﺍ ﺑﺪﻫﻨﺪ. ﺍﻳﻦ ﺭﻭﺵ ﺑﺪﻳﻬﯽ ﻭ ﺭﻭﺷﻦ ﺍﺳﺖ. ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﻧﻮﺯﺍﺩ ﺧﻮﺩ ﺑﺎ ﺑﻴﺪﺍﺭ ﺷﺪﻥ ﻭ ﮔﺮﻳﻪ ﮐﺮﺩﻧﺶ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﺪ ﭼﻪ ﻣﻮﻗﻊ ﮔﺮﺳﻨﻪ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻪ ﺧﻮﺩ ﻭ ﺩﻧﻴﺎﻳﯽ ﮐﻪ ﺩﺭ ﺁﻥ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﺪ ﺍﻋﺘﻤﺎﺩﯼ ﻋﻤﻴﻖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ – ﺷﺮﻁ ﺍﺳﺎﺳﯽ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻗﻮﯼ ﻭ ﺳﺎﻟﻢ – ﻣﻬﻢ‌ﺗﺮﻳﻦ ﮐﺎﺭﯼ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮑﻨﻴﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻫﺮ ﻭﻗﺖ ﺍﻭ ﮔﺮﺳﻨﻪ ﺍﺳﺖ ﺑﻪ ﺍﻭ ﻏﺬﺍ ﺑﺪﻫﻴﺪ. ﻧﻮﺯﺍﺩﯼ ﮐﻪ ﻫﺮ ﻭﻗﺖ ﮔﺮﺳﻨﻪ‌ﺍﺵ ﺑﻮﺩﻩ ﺑﻪ ﺍﻭ ﺷﻴﺮ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ: «ﺯﻧﺪﮔﯽ ﭼﻪ ﺧﻮﺑﻪ. ﭼﻘﺪﺭ ﺧﻮﺑﻪ ﮐﻪ ﻣﯽ‌ﺗﻮﻧﻢ ﺷﻴﺮ ﺑﺨﻮﺭﻡ. ﻭﻗﺘﯽ ﻣﺎﻣﺎﻥ ﻣﻨﻮ ﺑﻐﻞ ﻣﯽ‌ﮐﻨﻪ ﻭ ﺷﻴﺮﻣﻮ ﻣﯽ‌ﺩﻩ ﮔﺮﻣﺎﯼ ﺑﺪﻧﺸﻮ ﺧﻴﻠﯽ ﺩﻭﺳﺖ ﺩﺍﺭﻡ. ﺍﻳﻦ ﺩﻧﻴﺎ ﻭﺍﻗﻌﺎ ﺟﺎﯼ ﺧﻮﺑﻴﻪ، ﭼﻮﻥ ﻫﺮ ﻭﻗﺖ ﺑﻪ ﺍﻭﻥ ﻣﯽ‌ﻓﻬﻤﻮﻧﻢ ﮐﻪ ﮔﺮﺳﻨﻪ‌ﺍﻣﻪ ﺑﻪ ﻣﻦ ﺷﻴﺮ ﻣﯽ‌ﺩﻩ. ﻣﻄﻤﺌﻨﻢ ﮐﻪ ﻫﻤﻪ ﭼﻴﺰ ﺑﺮﺍﯼ ﻣﻦ ﻭ ﺍﻃﺮﺍﻓﻴﺎﻧﻢ ﺧﻴﻠﯽ ﺧﻮﺏ ﭘﻴﺶ ﻣﯽ‌ﺭﻩ.»
