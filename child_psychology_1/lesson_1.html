ﺑﭽﻪ ﺩﺍﺭ ﺷﺪﻥ ﺑﻪ ﺧﻮﺩﯼ ﺧﻮﺩ ﺩﺍﻧﺎﻳﯽ ﻭ ﮐﺎﺭﺁﻳﯽ ﻻﺯﻡ ﺑﺮﺍﯼ ﺍﻳﻦ ﻫﻨﺮ ﺭﺍ ﺩﺭ ﺍﻧﺴﺎﻥ ﺑﻪ ﻭﺟﻮﺩ ﻧﻤﯽ‌ﺁﻭﺭﺩ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﻪ ﺑﺘﻮﺍﻧﻴﻢ ﺍﻳﻦ ﻫﻨﺮ ﺭﺍ ﺑﻪ ﻧﺤﻮ ﺍﺣﺴﻦ ﮐﺴﺐ ﮐﻨﻴﻢ، ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﮐﻮﺩﮐﺎﻥ ﭼﮕﻮﻧﻪ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩ ﺑﺎ ﺍﺭﺗﮑﺎﺏ ﺍﺷﺘﺒﺎﻫﺎﺕ ﺑﺴﻴﺎﺭ ﻭ ﺑﺎ ﺗﺠﺮﺑﻪ ﺑﻪ ﺍﻳﻦ ﺁﮔﺎﻫﯽ ﻭ ﻣﻌﺮﻓﺖ ﻣﯽ‌ﺭﺳﻨﺪ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺍﮔﺮ ﺍﺯ ﻗﺒﻞ ﺑﺪﺍﻧﻴﻢ ﺗﻤﺎﻡ ﮐﻮﺩﮐﺎﻥ ﺗﺎ ﺭﺳﻴﺪﻥ ﺑﻪ ﺭﺷﺪ ﮐﺎﻣﻞ، ﻣﺴﻴﺮ ﻣﺸﺨﺼﯽ ﺭﺍ ﻃﯽ ﻣﯽ‌ﮐﻨﻨﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺑﺮﺧﯽ ﺍﺯ ﺍﻳﻦ ﺍﺷﺘﺒﺎﻫﺎﺕ ﺑﺮﺣﺬﺭ ﺑﺎﺷﻴﻢ.
 
ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ ﺗﺮﺟﻤﻪ ﻭ ﺍﻗﺘﺒﺎﺳﯽ ﺍﺳﺖ ﺍﺯ ﮐﺘﺎﺏ Tout se joue avant six ans  ﮐﻪ ﺧﻮﺩ ﺗﺮﺟﻤﻪ ﻓﺮﺍﻧﺴﻪ ﮐﺘﺎﺏ How to parent ﻧﻮﺷﺘﻪ ﻓﻴﺘﺴﻬﻮﮎ ﺩﺍﺩﺳﻮﻥ ﺍﺳﺖ. ﮐﺘﺎﺏ ﺧﻄﺎﺏ ﺑﻪ ﻣﺎﺩﺭﺍﻥ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﻭ ﺳﻌﯽ ﺷﺪﻩ ﻣﺎﻧﻨﺪ ﮔﻔﺖ ﻭ ﮔﻮﻳﯽ ﺑﻴﻦ ﻧﻮﻳﺴﻨﺪﻩ ﻭ ﺧﻮﺍﻧﻨﺪﻩ ﺑﺎﺷﺪ. ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ ﺳﻌﯽ ﺷﺪﻩ ﺍﻳﻦ ﻟﺤﻦ ﺣﻔﻆ ﺷﻮﺩ . ﻫﺮ ﭼﻨﺪ ﮐﺘﺎﺏ - ﻭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ- ﺧﻄﺎﺏ ﺑﻪ ﻣﺎﺩﺭﺍﻥ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﻣﺎ ﭘﺪﺭﺍﻥ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﻣﺎﺩﺭﺍﻥ ﺍﺯ ﺍﻳﻦ ﺩﻭﺭﻩ ﻫﻤﺎﻥ‌ﻗﺪﺭ ﺑﻬﺮﻩ ﺧﻮﺍﻫﻨﺪ ﺑﺮﺩ. 
 
ﺭﺷﺪ ﮐﻮﺩﮎ ﺭﺍﻩ ﻣﺴﺘﻘﻴﻤﯽ ﺭﺍ ﻃﯽ ﻧﻤﯽ‌ﮐﻨﺪ، ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺮ ﭼﻘﺪﺭ ﮐﻮﺩﮎ ﺑﺰﺭﮒ‌ﺗﺮ ﺷﻮﺩ ﺭﻓﺘﺎﺭ ﺍﻭ ﭘﺨﺘﻪ‌ﺗﺮ ﻧﻤﯽ‌ﺷﻮﺩ. ﺑﺮﻋﮑﺲ ﺭﺷﺪ ﻭ ﻧﻤﻮ ﺍﻭ ﺩﺍﺭﺍﯼ ﻣﺮﺍﺣﻞ ﻣﺨﺘﻠﻒ ﺗﻌﺎﺩﻝ ﻭ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺍﺳﺖ ﮐﻪ ﻳﮑﯽ ﭘﺲ ﺍﺯ ﺩﻳﮕﺮﯼ ﻓﺮﺍ ﻣﯽ‌ﺭﺳﻨﺪ. ﻣﺜﻼ ﻣﺮﺣﻠﻪ «ﻗﺪﻡ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ» (ﺯﻣﺎﻧﯽ ﻛﻪ  ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍﻩ ﻣﯽ‌ﺍﻓﺘﺪ) ﻳﮏ ﺩﻭﺭﻩ ﺗﻌﺎﺩﻝ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺁﻥ ﺩﻭﺭﻩ ﻋﺪﻡ ﺗﻌﺎﺩﻝ «ﺍﻭﻟﻴﻦ ﺩﻭﺭﺍﻥ ﺑﻠﻮﻍ» (ﺗﻘﺮﻳﺒﺎ ﺩﻭ ﺗﺎ ﺳﻪ ﺳﺎﻟﮕﯽ) ﻓﺮﺍ ﻣﯽ‌ﺭﺳﺪ. ﺳﭙﺲ ﻣﺮﺣﻠﻪ ﺳﻪ‌ﺳﺎﻟﮕﯽ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻣﺮﺣﻠﻪ ﺗﻌﺎﺩﻝ ﺍﺳﺖ ﻭ ﭘﺲ ﺍﺯ ﺁﻥ ﭼﻬﺎﺭﺳﺎﻟﮕﯽ ﮐﻪ ﻳﮏ ﺩﻭﺭﻩ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺍﺳﺖ. ﺑﻌﺪ، ﺗﻘﺮﻳﺒﺎ ﺣﺪﻭﺩ ﭘﻨﺞ ﺳﺎﻟﮕﯽ ﺩﻭﺑﺎﺭﻩ ﻳﮏ ﺩﻭﺭﻩ ﺗﻌﺎﺩﻝ ﻓﺮﺍ ﻣﯽ‌ﺭﺳﺪ.
 
ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺗﻌﺮﻳﻔﯽ ﺳﺎﺩﻩ ﺍﺯ ﺗﻌﺎﺩﻝ ﻭ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺑﺪﻫﻴﻢ ﮐﺎﻓﯽ ﺍﺳﺖ ﺑﮕﻮﻳﻴﻢ ﻫﺮﮔﺎﻩ ﺑﻮﺩﻥِ ﮐﻮﺩﮎ ﺩﺭ ﮐﻨﺎﺭ ﻣﺎﺩﺭ ﻟﺬﺕ‌ﺑﺨﺶ ﺍﺳﺖ (ﻣﺜﻼ ﺳﻪ ﺳﺎﻟﮕﯽ) ﺍﻳﻦ ﺩﻭﺭﻩ، ﺩﻭﺭﻩ ﺗﻌﺎﺩﻝ ﺍﺳﺖ. ﻭﻟﯽ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﻮﺩﮎ ﻣﺎﻧﻨﺪ ﻏﻮﻝ ﮐﻮﭼﻮﻟﻮﯼ ﺑﺪﺟﻨﺴﯽ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﻨﺪ (ﻣﺜﻼ ﺩﻭ ﺳﺎﻟﮕﯽ)، ﺩﻭﺭﻩ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺭﺍ ﻃﯽ ﻣﯽ‌ﮐﻨﺪ.
 
ﺷﻨﺎﺧﺖ ﻣﺮﺍﺣﻞ ﻣﺨﺘﻠﻒ ﻭ ﻣﺘﻮﺍﻟﯽ ﺭﺷﺪ ﮐﻮﺩﮎ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﮐﺎﻓﯽ ﻧﻴﺴﺖ ﻭ ﻣﺎ ﻧﺒﺎﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﻣﻨﻔﻌﻞ ﻓﻘﻂ ﺷﺎﻫﺪ ﮔﺬﺭﺍﻥ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﺑﺎﺷﻴﻢ. ﺑﺎ ﺷﻨﺎﺧﺖ ﺍﻳﻦ ﻣﺮﺍﺣﻞ، ﻫﺮ ﭘﺪﺭ ﻭ ﻳﺎ ﻣﺎﺩﺭﯼ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﺪ ﺑﺎ ﻣﺸﮑﻼﺕ ﺭﻓﺘﺎﺭﯼ ﻓﺮﺯﻧﺪﺍﻧﺶ ﺑﺮﺧﻮﺭﺩ ﮐﻨﺪ ﻭ ﺳﻌﯽ ﮐﻨﺪ ﺍﻳﻦ ﻣﺸﮑﻼﺕ ﺭﺍ ﺩﺭ ﻫﺮ ﻣﺮﺣﻠﻪ ﺍﺯ ﺭﺷﺪ ﺁﻥ‌ﻫﺎ ﺣﻞ ﮐﻨﺪ.
 
ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﻣﻄﺎﻟﺒﯽ ﮐﻠﯽ ﻭ ﻋﻤﻮﻣﯽ ﺩﺭﺑﺎﺭﻩ ﺍﻃﻔﺎﻝ ﻣﯽ‌ﺩﺍﻧﻨﺪ، ﺍﻣﺎ ﺩﺭ ﭘﯽ ﺗﻮﺻﻴﻪ‌ﻫﺎﻳﯽ ﻣﺸﺨﺺ ﻫﺴﺘﻨﺪ، ﺍﻳﻦ‌ﮐﻪ ﭼﻪ ﮐﺎﺭ ﺑﺎﻳﺪ ﮐﺮﺩ ﻳﺎ ﭼﻪ ﮐﺎﺭ ﻧﺒﺎﻳﺪ ﮐﺮﺩ. ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ ﻣﻤﻠﻮ ﺍﺯ ﺗﻮﺻﻴﻪ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺳﺖ ﻛﻪ ﺑﺮ ﺍﺳﺎﺱ ﻧﻈﺮﻳﺎﺕ ﺭﻭﺍﻧﺸﻨﺎﺳﺎﻥ ﺑﺰﺭﮒ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﻣﺎ ﺍﺯ ﺫﮐﺮ ﻧﺎﻡ ﺁﻧﻬﺎ ﺧﻮﺩﺩﺍﺭﯼ ﺷﺪﻩ ﺍﺳﺖ. 
 
ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ  ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﻫﺮ ﻣﻄﻠﺐ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺭ ﺯﻣﻴﻨﻪ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺖ ﺍﻃﻔﺎﻝ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﺳﺖ، ﺑﻪ ﻃﺮﺡ ﻣﺴﺎﺋﻠﯽ ﮐﻠﯽ ﺩﺭ ﻣﻮﺭﺩ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ. ﻫﺮ ﻣﺎﺩﺭﯼ ﺑﻬﺘﺮ ﺍﺯ ﻫﺮ ﮐﺲ ﺩﻳﮕﺮﯼ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﮔﺮ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﮐﻪ ﺑﻌﻀﯽ ﺍﺯ ﺑﺨﺶ‌ﻫﺎﯼ ﺩﻭﺭﻩ ﺑﺎ ﺍﺣﺴﺎﺱ ﻣﺎﺩﺭﯼ ﺷﻤﺎ ﺩﻣﺴﺎﺯ ﻧﻴﺴﺖ، ﺁﻥ ﺭﺍ ﺑﻪ ﮐﻨﺎﺭﯼ ﮔﺬﺍﺷﺘﻪ ﻭ ﻏﺮﻳﺰﻩ ﻣﺎﺩﺭﻳﺘﺎﻥ ﺭﺍ ﺩﻧﺒﺎﻝ ﮐﻨﻴﺪ.
 
ﻣﻮﺛﺮﺗﺮﻳﻦ ﻣﻌﻠﻢ ﮐﻮﺩﮎ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﻫﺴﺘﻨﺪ. ﺣﺮﻓﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﭘﻴﭽﻴﺪﻩ‌ﺗﺮﻳﻦ ﺣﺮﻓﻪ‌ﻫﺎ ﺩﺭ ﺟﺎﻣﻌﻪ ﺍﺳﺖ، ﺯﻳﺮﺍ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﻮﺩﻥ ﺑﻪ ﻣﻌﻨﺎﯼ ﺑﻴﺴﺖ ﻭ ﭼﻬﺎﺭ ﺳﺎﻋﺘﻪ ﺭﻭﺍﻧﺸﻨﺎﺱ ﮐﻮﺩﮎ ﻭ ﻣﻌﻠﻢ ﮐﻮﺩﮎ ﺑﻮﺩﻥ ﻧﻴﺰ ﻫﺴﺖ. 
ﺷﻤﺎ ﺭﻭﺍﻧﺸﻨﺎﺱ ﮐﻮﺩﮎ ﻫﺴﺘﻴﺪ، ﭼﺮﺍ ﮐﻪ ﺑﺎﻳﺪ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﺪﺍﻧﻴﺪ. ﺣﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺧﻮﺏ، ﻣﺘﻮﺳﻂ ﻭ ﻳﺎ ﺣﺘﯽ ﺑﺪ ﺑﺎﺷﻴﺪ. ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ، ﭼﻪ ﺧﻮﺷﺘﺎﻥ ﺑﻴﺎﻳﺪ ﭼﻪ ﻧﻴﺎﻳﺪ، ﺷﻤﺎ ﺭﻭﺍﻧﺸﻨﺎﺱ ﮐﻮﺩﮎ ﻫﺴﺘﻴﺪ. ﺍﻳﻦ ﺭﺍ ﻫﻢ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺷﻤﺎ «ﻣﻮﺛﺮﺗﺮﻳﻦ» ﻣﻌﻠﻢ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺗﻤﺎﻡ ﻣﺪﺕ ﻋﻤﺮﺵ ﺧﻮﺍﻫﻴﺪ ﺑﻮﺩ.
 
ﺷﻤﺎ ﺍﻭﻟﻴﻦ ﻣﻌﻠﻢ ﻣﺪﺭﺳﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻫﺴﺘﻴﺪ. ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﻳﺪ (ﻭ ﺣﺘﯽ ﺁﻧﭽﻪ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﻳﺪ) ﻣﻬﻢ ﺗﺮ ﺍﺯ ﺗﻤﺎﻡ «ﺩﺭﺱ»ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻣﺪﺭﺳﻪ ﺑﻴﺎﻣﻮﺯﺩ. ﺑﺪ ﻧﻴﺴﺖ ﭘﻴﺶ ﺍﺯ ﺑﺮﺭﺳﯽ ﻣﺮﺍﺣﻞ ﺭﺷﺪ ﮐﻮﺩﮎ ﻭ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻣﻌﻠﻢ ﺧﻮﺑﯽ ﺑﺎﺷﻴﻢ، ﻧﻈﺮﯼ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺕ ﻣﺎﺩﺭ ﺑﻴﻨﺪﺍﺯﻳﻢ.

ﺗﺠﺮﺑﻪ ﻣﺎﺩﺭﻥ ﺷﺪﻥ، ﺑﻪ ﺭﻏﻢ ﮐﻼﺱ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﮐﻪ ﺍﺣﺘﻤﺎﻻ ﺑﺮﺍﯼ ﺁﻣﺎﺩﮔﯽ ﺟﻬﺖ ﮔﺬﺭﺍﻧﺪﻥ ﺗﺠﺮﺑﻪ ﻣﺎﺩﺭﯼ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﻭ ﺑﻪ ﺭﻏﻢ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﮐﻪ ﺍﺣﻴﺎﻧﺎ ﻣﻄﺎﻟﻌﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ، ﺑﺮﺍﯼ ﺷﻤﺎ ﺗﺎﺭﻩ ﻭ ﺟﺪﻳﺪ ﺍﺳﺖ.  ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﺎ ﻧﻮﺯﺍﺩ ﺯﻧﺪﮔﯽ ﻧﮑﺮﺩﻩ ﺑﺎﺷﻴﺪ ﻭﺍﻗﻌﻴﺖ ﺍﻳﻦ ﺗﺠﺮﺑﻪ ﺭﺍ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﻟﻤﺲ ﮐﻨﻴﺪ. ﺗﻨﻬﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﻧﻮﺯﺍﺩ ﺑﻴﺴﺖ ﻭ ﭼﻬﺎﺭ ﺳﺎﻋﺘﻪ ﻧﺰﺩ ﺷﻤﺎﺳﺖ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﻭﺍﻗﻌﻴﺖ ﺭﺍ ﺩﺭﮎ ﮐﻨﻴﺪ. ﺗﻄﺎﺑﻖ ﺑﺎ ﺷﺮﺍﻳﻂ ﺟﺪﻳﺪ، ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﮐﻮﺷﺶ ﺑﺴﻴﺎﺭ ﺩﺍﺭﺩ. ﻗﺒﻼ ﻫﻴﭻ‌ﮔﺎﻩ ﺑﺎ ﭼﻨﻴﻦ ﻣﺴﻮﻭﻟﻴﺘﯽ ﺭﻭ ﺑﻪ ﺭﻭ ﻧﺒﻮﺩﻩ‌ﺍﻳﺪ.

ﮔﺎﻫﯽ ﻣﺎﺩﺭ ﺟﻮﺍﻥ ﺁﻥ‌ﭼﻨﺎﻥ ﺩﭼﺎﺭ ﺷﮏ ﻭ ﺗﺮﺩﻳﺪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻫﺮ ﭼﻴﺰﯼ ﺑﺮﺍﯼ ﺍﻭ ﺑﻪ ﺻﻮﺭﺕ ﻣﺴﺎﻟﻪ‌ﺍﯼ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ. ﺍﻭ ﮐﻪ ﺗﺠﺮﺑﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺑﺰﺭﮒ ﮐﺮﺩﻥ ﻧﻮﺯﺍﺩ ﻧﺪﺍﺭﺩ، ﻧﻤﯽ‌ﺩﺍﻧﺪ ﻭﻗﺎﻳﻊ ﻣﺨﺘﻠﻒ ﺭﺍ ﭼﮕﻮﻧﻪ ﺗﻔﺴﻴﺮ ﮐﻨﺪ. ﺍﮔﺮ ﻧﻮﺯﺍﺩ ﺩﺭ ﺧﻮﺍﺏ ﻋﻤﻴﻘﯽ ﺑﺎﺷﺪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﺩﻳﮕﺮ ﻧﻔﺲ ﻫﻢ ﻧﻤﯽ‌ﮐﺸﺪ. ﺑﺎ ﺷﺘﺎﺏ ﺑﺎﻻﯼ ﺳﺮ ﮐﻮﺩﮎ ﻣﯽ‌ﺭﻭﺩ ﺑﺒﻴﻨﺪ ﻓﺮﺯﻧﺪﺵ ﻫﻨﻮﺯ ﺯﻧﺪﻩ ﺍﺳﺖ ﻳﺎ ﻧﻪ! ﺍﮔﺮ ﮐﻮﺩﮎ ﺩﺭﺳﺖ ﻧﺘﻮﺍﻧﺪ ﺷﻴﺮ ﺭﺍ ﻫﻀﻢ ﮐﻨﺪ ﻭ ﻳﺎ ﺷﻴﺮ ﺩﺭ ﮔﻠﻮﻳﺶ ﺑﭙﺮﺩ، ﻣﺎﺩﺭ ﺟﻮﺍﻥ ﺯﻭﺩ ﻧﮕﺮﺍﻥ ﻣﯽ‌ﺷﻮﺩ. ﻫﺮ ﺣﺎﺩﺛﻪ‌ﺍﯼ ﺍﺑﻌﺎﺩﯼ ﻏﻴﺮ ﻃﺒﻴﻌﯽ ﺑﻪ ﺧﻮﺩ ﻣﯽ‌ﮔﻴﺮﺩ.

ﻣﺎﺩﺭﺍﻥ ﺟﻮﺍﻥ ﻣﺘﺨﺼﺺ ﺩﺭﺳﺖ ﮐﺮﺩﻥ ﮐﻮﻩ ﺍﺯ ﮐﺎﻩ ﻫﺴﺘﻨﺪ. ﺁﻧﻬﺎ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﻫﺮ ﮔﻮﻧﻪ ﺗﻔﺎﻭﺕ ﺑﺎ ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺁﻧﻬﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﻗﻮﺍﻧﻴﻦ ﺗﻐﺬﻳﻪ ﻭ ﺧﻮﺍﺏ ﮐﻮﺩﮎ ﻣﯽ‌ﺷﻨﺎﺳﻨﺪ، ﻋﻼﻣﺖ ﺑﻴﻤﺎﺭﯼ ﺧﻄﺮﻧﺎﮐﯽ ﺍﺳﺖ. ﻋﻠﺖ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻣﮑﺮﺭ ﻭ ﺳﺎﻋﺖ ﺑﻪ ﺳﺎﻋﺖ ﻣﺎﺩﺭﻫﺎﯼ ﺟﻮﺍﻥ ﺑﻪ ﺩﮐﺘﺮ ﻧﻴﺰ ﻫﻤﻴﻦ ﺍﺳﺖ.

ﺑﺪﺗﺮ ﺍﺯ ﻫﻤﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺎﺩﺭ ﺟﻮﺍﻥ ﺟﺮﺍﺕ ﺍﺑﺮﺍﺯ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺑﯽ‌ﻟﻴﺎﻗﺘﯽ ﺭﺍ ﺑﻪ ﮐﺴﯽ ﻧﺪﺍﺭﺩ. ﻫﺮ ﭼﻪ ﺑﺎﺷﺪ ﺍﻭ ﺍﻧﺴﺎﻧﯽ ﺍﺳﺖ ﻗﺎﺩﺭ ﺑﻪ ﺍﺳﺘﺪﻻﻝ ﻭ ﺣﺎﻻ ﺩﺭ ﺍﻭﺝ ﺍﻓﺘﺨﺎﺭ ﺯﻧﺪﮔﯽ ﺯﻧﺎﻧﻪ‌ﺍﺵ – ﻣﺎﺩﺭ ﺷﺪﻥ – ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻋﮑﺲ‌ﺍﻟﻌﻤﻠﺶ ﺩﺭ ﻣﻘﺎﺑﻞ ﭼﻨﻴﻦ ﺑﺮﺩﺍﺷﺘﯽ ﭼﻴﺴﺖ؟ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﻣﺎﺩﺭﻫﺎﯼ ﺩﻳﮕﺮ ﺑﻪ ﺧﻮﺑﯽ ﺑﻪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻣﯽ‌ﺭﺳﻨﺪ ﻭ ﺍﻭ  ﺗﻨﻬﺎ ﮐﺴﯽ ﺍﺳﺖ ﮐﻪ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺱ ﺑﺪﯼ ﺩﺍﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﺪ ﺑﻤﻴﺮﺩ ﻭﻟﯽ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﮐﺴﯽ ﺍﻋﺘﺮﺍﻑ ﻧﮑﻨﺪ. ﺍﻣﺎ ﺍﮔﺮ ﺍﻭ ﻣﯽ‌ﺩﺍﻧﺴﺖ ﮐﻪ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺭﺍ ﺗﻤﺎﻡ ﻣﺎﺩﺭﻫﺎﯼ ﺟﻮﺍﻥ ﺩﺍﺭﻧﺪ، ﻣﺴﻠﻤﺎ ﺧﻴﺎﻟﺶ ﺧﻴﻠﯽ ﺭﺍﺣﺖ‌ﺗﺮ ﻣﯽ‌ﺷﺪ.  ﺑﺎﻭﺭ ﮐﻨﻴﺪ ﮐﻪ ﺍﻭﺿﺎﻉ ﺑﺮﺍﯼ ﻫﻤﻪ ﻣﺎﺩﺭﻫﺎﯼ ﺟﻮﺍﻥ ﺑﻪ ﻫﻤﻴﻦ ﻣﻨﻮﺍﻝ ﺍﺳﺖ.

ﺗﻤﺎﻣﯽ ﺯﻧﺪﮔﯽ ﻣﺎﺩﺭ ﺟﻮﺍﻥ ﺩﺭ ﺍﻳﻦ ﮐﻮﭼﻮﻟﻮﻳﯽ ﮐﻪ ﺩﺭ ﮔﻬﻮﺍﺭﻩ‌ﺍﺵ ﺁﺭﻣﻴﺪﻩ ﺧﻼﺻﻪ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﻣﺎﺩﺭ ﻫﻢ ﺑﺪﻭﻥ ﺭﻭﺩﺭﺑﺎﻳﺴﺘﯽ، ﺧﻮﺩ ﻣﺎﻧﻴﻢ، ﺍﺯ ﺍﻭ ﺩﻟﺨﻮﺭ ﺍﺳﺖ. ﻭ ﺍﻳﻦ ﮐﺎﻣﻼ ﻃﺒﻴﻌﯽ ﺍﺳﺖ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﻫﻴﭻ ﮐﺲ ﺍﻭ ﺭﺍ ﺑﺮﺍﯼ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﮐﻴﻨﻪ ﺁﻣﺎﺩﻩ ﻧﮑﺮﺩﻩ ﺍﺳﺖ. ﺍﮐﺜﺮ ﺍﻭﻗﺎﺕ ﺑﻪ ﺧﺎﻃﺮ ﮐﻴﻨﻪ‌ﺍﯼ ﮐﻪ ﺍﺯ ﻓﺮﺯﻧﺪﺵ ﺩﺍﺭﺩ ﺍﺣﺴﺎﺱ ﮔﻨﺎﻩ ﻣﯽ‌ﮐﻨﺪ، ﺯﻳﺮﺍ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﺩ.

ﻣﺎﺩﺭ ﺟﻮﺍﻥ ﺑﺎﻳﺪ ﺑﺪﺍﻧﺪ ﮐﻪ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺍﻭﻟﻴﻪ ﺑﻪ ﻧﻮﺯﺍﺩﺵ ﮐﺎﻣﻼ ﻃﺒﻴﻌﯽ ﺍﺳﺖ. ﺑﻌﺪ ﺍﺯ ﻳﮏ ﺩﻭﺭﻩ ﺗﻄﺎﺑﻖ ﺑﺎ ﺍﻳﻦ ﻣﺴﻮﻭﻟﻴﺖ ﻏﻴﺮ ﻣﺎﻟﻮﻑ ﺍﻳﻦ ﮐﻴﻨﻪ ﻣﺤﻮ ﺷﺪﻩ ﻭ ﺟﺎﻳﺶ ﺭﺍ ﻋﺸﻘﯽ ﻋﻤﻴﻖ ﻭ ﻏﻴﺮﻗﺎﺑﻞ ﻣﻘﺎﻭﻣﺖ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺵ ﻣﯽ‌ﮔﻴﺮﺩ.

ﻋﻠﺖ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﺑﺮﺍﯼ ﺩﻟﺨﻮﺭ ﺑﻮﺩﻥ ﻣﺎﺩﺭ ﺍﺯ ﻓﺮﺯﻧﺪﺵ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻣﺎﺩﺭ ﻣﻌﻤﻮﻻ ﺗﺼﻮﺭ ﻣﯽ‌ﮐﻨﺪ ﺁﻣﺪﻥ ﻓﺮﺯﻧﺪﺵ ﺑﺎﻋﺚ ﺻﻤﻴﻤﯽ‌ﺗﺮ ﺷﺪﻥ ﺭﺍﺑﻄﻪ‌ﺍﺵ ﺑﺎ ﺷﻮﻫﺮﺵ ﻣﯽ‌ﺷﻮﺩ. ﻫﺮ ﭼﻪ ﺑﺎﺷﺪ، ﺍﻳﻦ ﻧﻮﺯﺍﺩ ﺭﺍ ﺁﻧﻬﺎ ﺩﻭ ﻧﻔﺮﯼ ﺩﺭﺳﺖ ﮐﺮﺩﻩ‌ﺍﻧﺪ ﻭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﻭﻗﺘﯽ ﮐﻮﺩﮎ ﺑﻪ ﺩﻧﻴﺎ ﺑﻴﺎﻳﺪ، ﺁﻧﻬﺎ ﺧﺎﻧﻮﺍﺩﻩ ﺳﻪ ﻧﻔﺮﯼ ﻭﺍﺣﺪﯼ ﺭﺍ ﺗﺸﮑﻴﻞ ﺧﻮﺍﻫﻨﺪ ﺩﺍﺩ.

ﻣﺘﺎﺳﻔﺎﻧﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﻭﺿﻊ ﺩﺭﺳﺖ ﺑﺮﻋﮑﺲ ﺍﺳﺖ. ﮐﻮﺩﮎ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﯽ، ﻣﺎﻧﻨﺪ ﭘﺮﺩﻩ‌ﺍﯼ ﻣﻴﺎﻥ ﺍﻳﻦ ﺩﻭ ﺣﺎﻳﻞ ﻣﯽ‌ﺷﻮﺩ. ﻣﺎﺩﺭ ﻣﯽ‌ﺑﻴﻨﺪ ﮐﻪ ﺷﻮﻫﺮﺵ ﻧﺴﺒﺖ ﺑﻪ ﮐﻮﺩﮎ ﻭ ﺗﻮﺟﻬﯽ ﮐﻪ ﻫﻤﺴﺮﺵ ﺑﻪ ﺍﻭ ﻣﯽ‌ﮐﻨﺪ، ﺣﺴﺎﺩﺕ ﻣﯽ‌ﻭﺭﺯﺩ. ﺭﻓﺘﺎﺭ ﺍﻭ ﺑﻴﺸﺘﺮ ﺷﺒﻴﻪ ﺑﻪ ﻳﮏ ﺭﻗﻴﺐ ﺍﺳﺖ ﺗﺎ ﭘﺪﺭ.
ﻣﺎﺩﺭ ﺟﻮﺍﻥ ﺑﺎﻳﺪ ﻳﺎﺩ ﺑﮕﻴﺮﺩ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﻳﻦ ﮐﻴﻨﻪ ﺑﺮﺧﻮﺭﺩ ﮐﻨﺪ ﻭ ﺑﺘﻮﺍﻧﺪ ﺍﻳﻦ ﺍﻣﺮ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺑﺨﺸﯽ ﺍﺯ ﺗﻐﻴﻴﺮﺍﺕ ﻏﻴﺮ ﻗﺎﺑﻞ ﺍﺟﺘﻨﺎﺑﯽ ﮐﻪ  ﺑﻌﺪ ﺍﺯ ﺗﻮﻟﺪ ﺍﻭﻟﻴﻦ ﻓﺮﺯﻧﺪ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ ﺑﭙﺬﻳﺮﺩ. ﻫﺮ ﻣﺎﺩﺭﯼ ﺭﻭﺵ ﺧﺎﺹ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺗﺒﺪﻳﻞ ﺧﺎﻧﻮﺍﺩﻩ‌ﺍﺵ ﺑﻪ ﻳﮏ ﻭﺍﺣﺪ ﺟﺪﻳﺪ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ.

ﺑﭽﻪ ﺗﺎﺯﻩ ﻣﺘﻮﻟﺪ ﺷﺪﻩ ﻭ ﺑﻪ ﻧﻈﺮ ﺑﺴﻴﺎﺭ ﺷﮑﻨﻨﺪﻩ ﮐﻮﭼﮏ ﻣﯽ‌ﺁﻳﺪ. ﮐﻮﭼﮑﺘﺮﻳﻦ ﺑﯽ‌ﻗﺎﻋﺪﮔﯽ ﺍﻫﻤﻴﺖ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﻣﯽ‌ﺩﻫﻴﺪ. ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﻣﺎﺩﺭ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﺮﺍﻣﻮﺵ ﮐﻨﺪ  ﻋﻘﻞ ﺳﻠﻴﻢ ﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭ ﺩﺭ ﭘﯽ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻧﻔﺮ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﺪ ﭼﻪ ﮐﺎﺭ ﮐﻨﺪ. ﺍﮔﺮ ﻧﺘﻮﺍﻧﺪ ﺑﻼﻓﺎﺻﻠﻪ ﺑﺎ ﺩﮐﺘﺮ ﺍﻃﻔﺎﻝ ﺗﻤﺎﺱ ﺑﮕﻴﺮﺩ ﻓﻮﺭﯼ ﺑﻪ ﻫﻤﺴﺎﻳﻪ ﺗﻠﻔﻦ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺍﺯ ﺍﻭ ﺭﺍﻩ ﺣﻞ ﺑﺨﻮﺍﻫﺪ.

ﺍﻣﺎ ﻫﺮ ﻣﺎﺩﺭ ﺟﻮﺍﻧﯽ ﺑﺎﻳﺪ ﺧﻮﺏ ﺑﺪﺍﻧﺪ ﮐﻪ ﺗﻤﺎﻡ ﺍﻳﻦ ﺍﺿﻄﺮﺍﺏ‌ﻫﺎ ﮐﺎﻣﻼ ﻃﺒﻴﻌﯽ ﺍﺳﺖ. ﺍﺿﻄﺮﺍﺏ، ﮐﻴﻨﻪ ﻭ ﺍﺣﺴﺎﺱ ﺩﺳﺘﭙﺎﭼﮕﯽ ﺭﺍ ﻫﻤﻪ ﻣﺎﺩﺭﺍﻥ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻩ‌ﺍﻧﺪ ﻭ ﻣﯽ‌ﮐﻨﻨﺪ. ﮐﻢ‌ﮐﻢ ﺑﺮ ﺷﻤﺎ ﺛﺎﺑﺖ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻧﻮﺯﺍﺩ ﺁﻥ ﻃﻮﺭ ﮐﻪ ﺷﻤﺎ ﻓﮑﺮ ﻣﯽ‌ﮐﺮﺩﻳﺪ ﺷﮑﻨﻨﺪﻩ ﻭ ﺿﻌﻴﻒ ﻧﻴﺴﺖ. ﻧﻮﺯﺍﺩﻫﺎ ﺍﺯ ﻣﻴﻠﻴﻮﻥ‌ﻫﺎ ﺳﺎﻝ ﭘﻴﺶ ﻣﯽ‌ﺩﺍﻧﻨﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﺑﺮ ﺩﺳﺖ ﻭ ﭘﺎ ﭼﻠﻔﺘﯽ ﺑﻮﺩﻥ ﻭ ﺍﺣﺴﺎﺱ ﺩﺳﺘﭙﺎﭼﮕﯽ ﻣﺎﺩﺭﺍﻧﺸﺎﻥ ﭼﻴﺮﻩ ﺷﻮﻧﺪ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻫﻢ ﺍﺯ ﻧﮕﺮﺍﻧﯽ‌ﻫﺎﯼ ﺷﻤﺎ ﺟﺎﻥ ﺳﺎﻟﻢ ﺑﻪ‌ﺩﺭ ﺧﻮﺍﻫﺪ ﺑﺮﺩ. ﺑﺎ ﺗﺠﺮﺑﻪ، ﺍﻋﺘﻤﺎﺩ ﺷﻤﺎ ﻫﻢ ﺍﻓﺰﺍﻳﺶ ﻣﯽ‌ﻳﺎﺑﺪ. ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺩﺭ ﺁﻏﻮﺷﺶ ﻣﯽ‌ﮔﻴﺮﻳﺪ، ﺑﺎ ﺭﺍﺣﺘﯽ ﺑﻴﺸﺘﺮ ﻏﺬﺍﻳﺶ ﺭﺍ ﻣﯽ‌ﺩﻫﻴﺪ ﻭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻧﻮﺯﺍﺩ ﺩﻭ ﻣﺎﻫﻪ ﺷﺪ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺧﻴﻠﯽ ﺯﻳﺎﺩ ﺯﻧﺪﻩ ﺧﻮﺍﻫﺪ ﻣﺎﻧﺪ!

ﺻﺤﺒﺖ ﺑﺎ ﻣﺎﺩﺭﺍﻥ ﺩﻳﮕﺮ ﻭ ﻣﺸﺎﻫﺪﻩ ﻭﺟﻮﺩ ﺍﺣﺴﺎﺳﺎﺕ ﻣﺸﺎﺑﻪ ﺩﺭ ﻧﺰﺩ ﺁﻧﻬﺎ ﺑﺴﻴﺎﺭ ﺁﺭﺍﻣﺶ‌ﺑﺨﺶ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﮐﻪ ﺗﻨﻬﺎ ﺷﻤﺎ ﻧﻴﺴﺘﻴﺪ ﮐﻪ ﺩﺭ ﭼﻨﻴﻦ ﻭﺿﻌﻴﺘﯽ ﻗﺮﺍﺭ ﺩﺍﺭﻳﺪ. ﺻﺤﺒﺖ ﺩﺭ ﻣﻮﺭﺩ ﺍﺣﺴﺎﺳﺎﺗﺘﺎﻥ ﺑﺎ ﻭﺍﻟﺪﻳﻨﺘﺎﻥ ﻧﻴﺰ ﮐﻤﮏ ﺯﻳﺎﺩﯼ ﺑﻪ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﭘﺲ ﺑﺎ ﻣﺎﺩﺭﺍﻥ ﺩﻳﮕﺮ، ﺑﻴﺸﺘﺮ ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﺗﺠﺮﺑﻪ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﻧﺪ، ﺻﺤﺒﺖ ﮐﻨﻴﺪ.

ﺍﻣﺎ ﺍﻳﻦ ﺍﺷﺘﺒﺎﻩ ﺭﺍ ﻧﮑﻨﻴﺪ ﮐﻪ ﻫﺮ ﭼﻪ ﻫﻤﺴﺎﻳﻪ ﻳﺎ ﻓﻼﻥ ﻣﺎﺩﺭ ﮔﻔﺖ ﻭﺣﯽ ﻡُﻧﺰﻝ ﺑﭙﻨﺪﺍﺭﻳﺪ. ﺍﻳﻦ ﺩﻗﻴﻘﺎ ﮐﺎﺭﯼ ﺍﺳﺖ ﮐﻪ ﺍﮐﺜﺮ ﻣﺎﺩﺭﺍﻥ ﺟﻮﺍﻥ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺩﺳﺖ ﭘﺎﭼﻪ ﻣﯽ‌ﺷﻮﻧﺪ، ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﺯ ﻧﻈﺮﺍﺕ ﻣﺎﺩﺭﺍﻥ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﮐﻤﮏ ﺑﻪ ﺑﺮﺭﺳﯽ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩﺗﺎﻥ ﻭ ﺍﺭﺯﻳﺎﺑﯽ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﻭﻟﯽ ﻫﻴﭻ‌ﮔﺎﻩ ﺁﻥ ﺭﺍ ﻭﺣﯽ ﻡُﻧﺰﻝ ﺗﻠﻘﯽ ﻧﮑﻨﻴﺪ. ﻭ ﻫﻤﻴﺸﻪ ﺑﻪ ﻳﺎﺩﺗﺎﻥ ﺑﺎﺷﺪ ﮐﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺭ ﻧﻮﻉ ﺧﻮﺩﺵ ﺩﺭ ﺩﻧﻴﺎ ﺑﯽ‌ﻧﻈﻴﺮ ﺍﺳﺖ. ﭼﻴﺰﯼ ﮐﻪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪ ﻫﻤﺴﺎﻳﻪ ﺷﻤﺎ ﺑﺴﻴﺎﺭ ﺧﻮﺏ ﺑﻮﺩﻩ ﻣﻌﻠﻮﻡ ﻧﻴﺴﺖ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻫﻢ ﺩﻗﻴﻘﺎ ﻫﻤﺎﻥ ﻃﻮﺭ ﺑﺎﺷﺪ.
ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺭ ﻧﻮﻉ ﺧﻮﺩ ﺩﺭ ﺩﻧﻴﺎ ﻳﮕﺎﻧﻪ ﺍﺳﺖ، ﺩﺭﺳﺖ ﻣﺜﻞ ﺍﺛﺮ ﺍﻧﮕﺸﺘﺎﻧﺶ. ﻣﺎ ﺭﻭﯼ ﺧﺼﻠﺖ ﻳﮑﺘﺎ ﺑﻮﺩﻥ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺗﮑﻴﻪ ﺑﺴﻴﺎﺭ ﺩﺍﺭﻳﻢ ﻭ ﻣﺪﺍﻡ ﺑﻪ ﺁﻥ ﺍﺷﺎﺭﻩ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.


ﻳﻜﻲ ﺍﺯ ﺍﻫﺪﺍﻑ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻧﺸﺎﻥ ﺩﻫﺪ ﺯﻳﺒﺎﺗﺮﻳﻦ ﻫﺪﻳﻪ‌ﺍﯼ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻋﺮﺿﻪ ﮐﻨﻴﺪ ﻓﺮﺍﻫﻢ ﻛﺮﺩﻥ ﺍﻣﻜﺎﻧﺎﺗﯽ ﺍﺳﺖ ﻛﻪ ﺑﻪ ﺍﻭ ﺍﺟﺎﺯﻩ ﺩﻫﻨﺪ ﺁﺯﺍﺩﺍﻧﻪ ﺑﻪ ﺷﮑﻮﻓﺎﻳﯽ ﺑﺮﺳﺪ. ﺍﻭ ﺩﺭ ﻧﻮﻉ ﺧﻮﺩ ﻳﮑﺘﺎﺳﺖ. ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﻫﻤﻴﻦ ﻃﻮﺭ ﺑﻤﺎﻧﺪ.


ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻪ ﺗﺪﺭﻳﺞ ﮐﻪ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﺩ ﺗﺎﺭﻳﺦ ﺭﺷﺪ ﺧﻮﺩ ﺭﺍ «ﺭﻗﻢ» ﻣﯽ‌ﺯﻧﺪ. ﺍﻭ ﺭﺍ ﺁﺯﺍﺩ ﺑﮕﺬﺍﺭﻳﺪ ﺍﻳﻦ ﮐﺎﺭﺵ ﺭﺍ ﺑﮑﻨﺪ. ﻫﺮ ﺑﭽﻪ‌ﺍﯼ ﺭﻭﺵ ﺧﺎﺹ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺯﻧﺪﮔﯽ ﺩﺍﺭﺩ. ﻭ ﺍﻳﻦ ﺭﻭﺵ ﺍﺯ ﻟﺤﻈﻪ ﺗﻮﻟﺪ ﺁﻏﺎﺯ ﻣﯽ‌ﺷﻮﺩ.
