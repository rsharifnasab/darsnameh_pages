ﻣﻘﺪﻣﺎﺗﯽ ﺩﺭﺑﺎﺭﻩ ﺻﺪﺍ

 ﺍﮔﺮ ﺷﻤﺎ ﺭﺍ ﺑﺎ ﭼﺸﻢ ﺑﺴﺘﻪ ﺑﻪ ﻳﮏ ﺍﺗﺎﻕ ﺑﺒﺮﻧﺪ ﻭ ﻛﺴﻰ ﺑﺎ ﺷﻤﺎ ﺻﺤﺒﺖ ﻛﻨﺪ، ﺷﻤﺎ ﺑﻼﻓﺎﺻﻠﻪ ﺑﻪ ﺩﻭ ﻧﻜﺘﻪ ﺍﺳﺎﺳﻰ ﭘﻰ ﻣﻰ ﺑﺮﻳﺪ:

۱- ﻓﺎﺻﻠﻪ ﺗﺎﻥ ﺑﺎ ﮔﻮﻳﻨﺪﻩ ﭼﻘﺪﺭ ﺍﺳﺖ.

۲- ﺩﺭ ﭼﻪ ﻧﻮﻉ ﻓﻀﺎﺋﻰ ﻫﺴﺘﻴﺪ، ﺍﺗﺎﻕ ﺧﻮﺍﺏ ﻳﺎ ﺣﻤﺎﻡ ﻳﺎ ﺍﻧﺒﺎﺭ.

ﺩﻟﻴﻞِ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻋﻜﺲ ﺍﻟﻌﻤﻞ ﻣﺘﻔﺎﻭﺕ ﮔﻮﺵ ﺑﻪ ﺻﺪﺍﻯ ﻣﺴﺘﻘﻴﻢ ﻭ ﺻﺪﺍﻯ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﺍﺳﺖ.

ﻭﻗﺘﻰ ﻛﺴﻰ ﺣﺮﻑ ﻣﻰ ﺯﻧﺪ ﻳﺎ ﻫﺮ ﺻﺪﺍﻯ ﺩﻳﮕﺮﻯ ﻛﻪ ﺗﻮﻟﻴﺪ ﻣﻰ ﺷﻮﺩ ﺩﺭ ﻭﺍﻗﻊ ﻣﺠﻤﻮﻋﻪ ﺍﻯ ﺍﺯ ﺍﻣﻮﺍﺝ ﺷﺒﻴﻪ ﺍﻣﻮﺍﺟﻰ ﻛﻪ ﺍﺯ ﺍﻧﺪﺍﺧﺘﻦ ﻳﮏ ﺳﻨﮓ ﺩﺭ ﺁﺏ ﺍﻳﺠﺎﺩ ﻣﻰ ﺷﻮﺩ ﺗﻮﻟﻴﺪ ﻣﻰ ﻛﻨﺪ. ﺩﺭ ﻳﮏ ﺍﺗﺎﻕ ﻣﻌﻤﻮﻟﯽ ﺍﻣﻮﺍﺝ ﺻﻮﺗﯽ ﺑﻌﺪ ﺍﺯ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﺩﻳﻮﺍﺭﻫﺎ، ﺳﻘﻒ ﻭ ﮐﻒ ﺍﺗﺎﻕ ﺩﺭ ﺟﻬﺎﺕ ﻭ ﺯﺍﻭﻳﻪ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺑﺎﺯﺗﺎﺏ ﭘﻴﺪﺍ ﻣﯽ ﮐﻨﻨﺪ. ﭼﺮﺧﺶ ﺻﺪﺍ ﺩﺭ ﺍﺗﺎﻕ ﺑﻪ ﻫﻤﻪ ﺟﻬﺖ ﻫﺎﺳﺖ ﻭ ﻣﻮﺝ ﻫﺎﻯ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﻮﺍﻧﻌﻰ ﻛﻪ ﺳﺮ ﺭﺍﻫﺸﺎﻥ ﭘﻴﺪﺍ ﻣﻰ ﺷﻮﺩ ﺑﻪ ﺍﺷﻜﺎﻝ ﻣﺘﻔﺎﻭﺗﻰ ﺑﺎﺯﺗﺎﺏ ﭘﻴﺪﺍ ﻣﻰ ﻛﻨﻨﺪ ﻭ ﺧﻴﻠﯽ ﺯﻭﺩ ﺗﻤﺎﻡ ﻓﻀﺎﯼ ﺍﺗﺎﻕ ﺭﺍ ﭘﺮ ﻣﯽ ﮐﻨﻨﺪ.

ﺍﮔﺮ ﺷﻨﻮﻧﺪﻩ ﺑﻪ ﮔﻮﻳﻨﺪﻩ ﻧﺰﺩﻳﮏ ﺑﺎﺷﺪ ﺑﺨﺶ ﺍﺻﻠﻰ ﺻﺪﺍﻳﻰ ﻛﻪ ﻣﻰ ﺷﻨﻮﺩ ﺻﺪﺍﻯ ﻣﺴﺘﻘﻴﻢ ﺍﺳﺖ ﻭ ﻓﻘﻂ ﺍﻧﺪﻛﻰ ﺻﺪﺍﻯ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ، ﻳﻌﻨﻰ ﺻﺪﺍﻯ‌ ﺑﺮﮔﺸﺘﻪ ﺍﺯ ﺩﻳﻮﺍﺭﻫﺎ ﺭﺍ ﻣﻰ ﺷﻨﻮﺩ.

ﺍﮔﺮ ﺷﻨﻮﻧﺪﻩ ﺍﺯ ﮔﻮﻳﻨﺪﻩ ﺩﻭﺭ ﺑﺎﺷﺪ ﺑﻴﺸﺘﺮ ﺻﺪﺍﺋﻰ ﻛﻪ ﻣﻰ ﺷﻨﻮﺩ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﺍﺳﺖ.

ﺗﻨﺎﺳﺐ ﺑﻴﻦ ﺻﺪﺍﻫﺎﻯ ﻣﺴﺘﻘﻴﻢ ﻭ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻣﻐﺰ ﻣﻰ ﮔﻮﻳﺪ ﮔﻮﻳﻨﺪﻩ ﻳﺎ ﻣﻮﻟﺪ ﺻﺪﺍ ﺩﺭ ﭼﻪ ﻓﺎﺻﻠﻪ ﺍﻯ ﺍﺯ ﻓﺮﺩ ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﻛﻴﻔﻴﺖ ﺻﺪﺍﻯ ﺍﻧﻌﻜﺎﺳﻰ ﻧﻴﺰ ﺍﻃﻼﻋﺎﺕ ﺩﻳﮕﺮﻯ ﺩﺭ ﻣﻮﺭﺩ ﻣﺤﻞ ﺑﻪ ﺷﻨﻮﻧﺪﻩ ﻣﻰ ﺩﻫﺪ. ﻣﺜﻼ ﺩﺭ ﻳﮏ ﻣﺴﺠﺪ ﻳﺎ ﺳﺎﻟﻦ ﺑﺰﺭﮒ ﻛﻨﺴﺮﺕ ﺻﺪﺍﻯ ﺧﻮﺍﻧﺪﻥ ﺗﺎ ﺑﻪ ﺩﻳﻮﺍﺭﻫﺎ ﺑﺮﺧﻮﺭﺩ ﮐﻨﺪ ﻭ ﺑﺮﮔﺮﺩﺍﻧﺶ ﺑﻪ ﮔﻮﺵ ﺷﻨﻮﻧﺪﻩ ﺑﺮﺳﺪ ﺯﻣﺎﻥ ﺑﻴﺸﺘﺮﻯ ﻃﻮﻝ ﻣﻰ ﮐﺸﺪ ﻭ ﺍﮔﺮ ﻓﻀﺎ ﺧﻴﻠﻰ ﺑﺰﺭﮒ‌ ﺑﺎﺷﺪ، ﻓﺎﺻﻠﻪ ﺯﻣﺎﻧﯽ ﻣﻴﺎﻥ ﺷﻨﻴﺪﻥ ﺻﺪﺍﯼ ﺍﻭﻟﻴﻪ ﻭ ﺑﺎﺯﺗﺎﺏ ﺁﻥ ﺍﺯ ﻳﮏ ﺩﻳﻮﺍﺭ ﻳﺎ ﻣﺎﻧﻊ ﺩﻭﺭﺗﺮ ﻗﺎﺑﻞ ﺗﻔﮑﻴﮏ ﻣﯽ ﺷﻮﺩ ﻭ ﺷﻨﻮﻧﺪﻩ ﻳﺎ ﺣﺘﯽ ﮔﻮﻳﻨﺪﻩ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻧﻌﮑﺎﺱ (ﺍﮐﻮﯼ) ﺻﺪﺍ ﺭﺍ ﺑﻪ ﻭﺿﻮﺡ ﺑﺸﻨﻮﺩ. ﮔﺎﻫﯽ ﺗﻌﺪﺩ ﻣﺎﻧﻊ ﻫﺎ ﻭ ﻓﺎﺻﻠﻪ ﺁﻧﻬﺎ ﺑﺎ ﻫﻤﺪﻳﮕﺮ ﺑﻪ ﭼﻨﺪﺯﻣﺎﻧﻪ ﺷﺪﻥ ﺩﺭﻳﺎﻓﺖ ﺑﺎﺯﺗﺎﺏ ﺻﺪﺍ ﻣﻨﺠﺮ ﻣﯽ ﺷﻮﺩ.

ﺟﻨﺲ ﻣﻮﺍﻧﻊ ﻧﻴﺰ ﺑﺮ ﮐﻴﻔﻴﺖ ﺑﺎﺯﺗﺎﺏ ﺻﺪﺍ ﺗﺄﺛﻴﺮ ﺩﺍﺭﺩ. ﺩﻳﻮﺍﺭ ﺳﻨﮕﻰ ﻳﺎ ﺷﻴﺸﻪ ﺍﻯ ﺻﺪﺍ ﺭﺍ ﺧﻴﻠﻰ ﺑﻬﺘﺮ ﺍﺯ ﺩﻳﻮﺍﺭ ﭘﺎﺭﭼﻪ ﺍﻯ ﻭ ﻓﺮﺵ ﺑﺮ ﻣﻰ ﮔﺮﺩﺍﻧﺪ ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺍﻧﻌﻜﺎﺱ ﺻﺪﺍ ﺩﺭ ﻳﮏ ﺣﻤﺎﻡ ﻛﺎﺷﻰ ﻛﺎﺭﻯ ﺷﺪﻩ ﺑﻪ ﺩﻓﻌﺎﺕ ﺑﻴﺸﺘﺮ ﺍﺯ ﺍﺗﺎﻕ ﺧﻮﺍﺑﻰ ﺑﻪ ﻫﻤﺎﻥ ﺍﻧﺪﺍﺯﻩ ﺍﺳﺖ.

ﺩﺭ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﻯ ﺻﺪﺍ ﺍﻧﻮﺍﻉ ﻓﻴﻠﺘﺮﻫﺎ ﺑﺮﺍﻯ ﺗﻮﻟﻴﺪ ﻳﺎ ﺍﺻﻼﺡ ﺍﻳﻦ ﻧﻮﻉ ﺟﻠﻮﻩ ﻫﺎﻯ ﺻﻮﺗﻰ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺻﺪﺍﭘﺮﺩﺍﺯﺍﻥ ﺑﺮﺍﯼ ﺍﻟﻘﺎﯼ ﻓﻀﺎﻫﺎﯼ ﺧﺎﺹ ﻧﻤﺎﻳﺸﯽ ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ. ﮔﺎﻫﯽ ﺩﺭ ﺻﺪﺍﺑﺮﺩﺍﺭﯼ ﻫﻢ ﻣﺜﻞ ﻋﮑﺎﺳﯽ ﺑﻪ ﺗﻨﺎﺳﺐ ﺍﺯ ﺳﻄﺢ ﻫﺎﻳﯽ ﮐﻪ ﻳﮏ ﻃﺮﻑ ﺁﻥ ﺑﺮﺍﯼ ﺟﺬﺏ ﺻﺪﺍ ﻭ ﻃﺮﻑ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﺑﺎﺯﺗﺎﺏ ﺻﺪﺍ ﺳﺎﺧﺘﻪ ﺷﺪﻩ ﺑﺮﺍﯼ ﺗﻘﻮﻳﺖ ﻳﺎ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﻋﺒﻮﺭ ﺻﺪﺍ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﺷﻮﺩ. ﺩﺭ ﺻﺪﺍﺑﺮﺩﺍﺭﯼ ﻫﻢ ﺑﻪ ﺍﻳﻦ ﻧﻮﻉ ﻭﺳﻴﻠﻪ Reflector ﻳﺎ ﺑﺎﺯﺗﺎﺏ ﺩﻫﻨﺪﻩ ﻣﯽ ﮔﻮﻳﻨﺪ.

ﺯﻣﺎﻥِ ﺍﻧﻌﮑﺎﺱ ﻳﺎ ﻋﻤﺮِ ﺻﺪﺍ ﻳﺎ Reverbration Time ﺑﻪ ﺯﻣﺎﻧﻰ ﻣﻰ ﮔﻮﻳﻨﺪ ﻛﻪ ﺻﺪﺍ ﺍﺯ ﺍﻭﺝ ﻗﺪﺭﺕ ﺍﻭﻟﻴﻪ ﺗﺎ ﺭﺳﻴﺪﻥ ﺑﻪ ﻳﮏ ﻣﻴﻠﻴﻮﻧﻴﻢ ﺁﻥ، ﻳﻌﻨﯽ ﻋﻤﻼ ﺻﻔﺮ، ﻃﻰ ﻣﻰ ﻛﻨﺪ. ﻫﺮ ﭼﻪ ﺍﻳﻦ ﺯﻣﺎﻥ ﻛﻮﺗﺎﻩ ﺗﺮ ﺑﺎﺷﺪ ﺻﺪﺍ ﺷﻔﺎﻑ ﺗﺮ ﺑﻪ ﮔﻮﺵ ﺷﻨﻮﻧﺪﻩ ﻣﻰ ﺭﺳﺪ.

ﻋﻤﺮ ﺻﺪﺍ ﺩﺭ ﺍﺗﺎﻕ ﻣﻌﻤﻮﻟﻰ ﻧﻴﻢ ﺛﺎﻧﻴﻪ ﺍﺳﺖ.  ﺩﺭ ﻳﮏ ﺳﺎﻟﻦ ﻛﻨﺴﺮﺕ ﻳﮏ ﻭ ﻧﻴﻢ ﺗﺎ ﺩﻭ ﻭ ﻧﻴﻢ ﺛﺎﻧﻴﻪ.  ﻋﻤﺮ ﺻﺪﺍ ﺩﺭ ﺻﺤﻦ ﻳﮏ ﻛﻠﻴﺴﺎ ﻣﻰ ﺗﻮﺍﻧﺪ ﺗﺎ ٨ ﺛﺎﻧﻴﻪ ﺑﺎﺷﺪ.  ﻭ ﺩﺭ ﺍﺳﺘﻮﺩﻳﻮﻯ ﺭﺍﺩﻳﻮﻳﻰ ﻣﻌﻤﻮﻻ ﺯﻳﺮ ٢٥ ﺻﺪﻡ ﺛﺎﻧﻴﻪ ﺍﺳﺖ.

ﻋﻠﺖ ﺍﻳﻦ ﮐﻪ ﺩﺭ ﺍﺳﺘﻮﺩﻳﻮﯼ ﺭﺍﺩﻳﻮﻳﯽ‌ ﻣﯽ ﮐﻮﺷﻨﺪ ﻋﻤﺮ ﺻﺪﺍﯼ ﺍﻭﻟﻴﻪ ﺗﺎ ﺣﺪ ﻣﻤﮑﻦ ﮐﻮﺗﺎﻩ ﺑﺎﺷﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻭﻗﺘﯽ ﺑﺎﻳﺪ ﺻﺪﺍ ﺍﺯ ﻃﺮﻳﻖ ﻳﮏ ﺩﺳﺘﮕﺎﻩ ﻭﺍﺳﻄﻪ ﻣﺜﻼ ﺭﺍﺩﻳﻮ ﻳﺎ ﻛﺎﻣﭙﻴﻮﺗﺮ ﺑﻪ ﮔﻮﺵ ﻛﺴﻰ ﺑﺮﺳﺪ ﭼﻮﻥ ﺗﺮﻛﻴﺒﻰ ﺍﺯ ﺑﺎﺯﺗﺎﺏ ﻫﺎﻯ ﺻﺪﺍ ﺩﺭ ﻣﺤﻞ ﺷﻨﻴﺪﻩ ﺷﺪﻥ ﺁﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺍﮔﺮ ﺩﺭ ﻣﺤﻞ ﺿﺒﻂ ﺻﺪﺍ ﻫﻢ ﺍﻧﻌﮑﺎﺱ ﻫﺎﯼ ﻧﺎﺧﻮﺍﺳﺘﻪ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺗﺮﮐﻴﺐ ﺍﻳﻦ ﺩﻭ ﮔﻮﻧﻪ ﺑﺎﺯﺗﺎﺏ ﺗﺎﺛﻴﺮ ﺑﺴﻴﺎﺭ ﺑﺪﻯ ﺑﺮ ﻛﻴﻔﻴﺖ ﺻﺪﺍ ﻭ ﻣﻔﻬﻮﻡ ﺑﻮﺩﻥ ﺁﻥ ﻣﯽ ﮔﺬﺍﺭﺩ.

ﺍﻳﺠﺎﺩ ﻭ ﺩﺭﮎ ﺻﺪﺍ ﺩﺭ ﺳﻪ ﻣﺮﺣﻠﻪ ﺍﻧﺠﺎﻡ ﻣﻰ ﺷﻮﺩ:
 ۱- ﻣﺮﺣﻠﻪ ﺗﻮﻟﻴﺪ: ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻳﮏ ﺣﺮﮐﺖ ﻳﺎ ﻟﺮﺯﻩ‌ ﺍﻣﻮﺍﺟﻰ ﺩﺭ ﻫﻮﺍ ﺍﻳﺠﺎﺩ ﻣﻰ ﮐﻨﺪ. ﺍﻳﻦ ﺣﺮﮐﺖ ﺍﻭﻟﻴﻪ ﻣﻰ ﺗﻮﺍﻧﺪ ﻧﺎﺷﻰ ﺍﺯ ﻟﺮﺯﻩ ﺗﺎﺭﻫﺎﻯ ﺻﻮﺗﻰ ﮐﺴﻰ ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﻫﻮﺍﻯ ﺭﻳﻪ ﺍﺵ ﺑﺎ ﺁﻧﻬﺎ ﺑﺎﺷﺪ ﻳﺎ ﺣﺎﺻﻞ ﺯﺩﻥ ﺯﺧﻤﻪ ﺍﻯ ﺭﻭﻯ ﺳﻴﻢ ﺗﺎﺭ. 

۲- ﻣﺮﺣﻠﻪ‌ ﭘﺨﺶ: ﺍﻣﻮﺍﺝ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﺩﺭ ﻫﻤﻪ ﺟﻬﺖ ﺷﺮﻭﻉ ﺑﻪ ﺣﺮﮐﺖ ﻣﻰ ﮐﻨﻨﺪ. ﮔﺎﻫﻰ ﺑﻪ ﺩﻳﻮﺍﺭ ﻣﻰ ﺧﻮﺭﻧﺪ ﻭ ﺗﻐﻴﻴﺮ ﻣﺴﻴﺮ ﻣﻰ ﺩﻫﻨﺪ. ﮔﺎﻫﻰ ﺑﻪ ﺳﻄﺢ ﻧﺮﻣﻰ ﺑﺮ ﻣﻰ ﺧﻮﺭﻧﺪ ﻭ ﺍﺯ ﻗﺪﺭﺕ ﺷﺎﻥ ﮐﺎﺳﺘﻪ ﻣﻰ ﺷﻮﺩ ﻭ ﺑﻌﻀﻰ ﺍﻭﻗﺎﺕ ﺑﻪ ﮔﻮﺵ ﻣﺎ ﻳﺎ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻣﻰ ﺭﺳﻨﺪ ﻭ ﺑﻪ ﻣﺮﺣﻠﻪ ﺳﻮﻡ، ﻳﻌﻨﻰ ﺩﺭﻳﺎﻓﺖ، ﻣﻰ ﺭﺳﻨﺪ.

 ۳- ﻣﺮﺣﻠﻪ ﺩﺭﻳﺎﻓﺖ: ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺻﺪﺍ ﺑﻪ ﮔﻴﺮﻧﺪﻩ ﺍﻯ ﻣﻰ ﺭﺳﺪ ﮐﻪ ﻧﺴﺒﺖ ﺑﻪ ﺗﻐﻴﻴﺮﺍﺕ ﺍﻣﻮﺍﺝ ﺩﺭ ﻫﻮﺍ ﺣﺴﺎﺱ ﺍﺳﺖ ﻭ ﻣﻰ ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﺍﻣﻮﺍﺝ ﺭﺍ ﺑﻪ ﭼﻴﺰﻫﺎﻯ ﺩﻳﮕﺮ ﺗﺒﺪﻳﻞ ﮐﻨﺪ. ﮔﻮﺵ ﺍﻧﺴﺎﻥ ﺍﻳﻦ ﺍﻣﻮﺍﺝ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺍﺭﺗﻌﺎﺵ ﺟﺬﺏ ﻭ ﭘﺲ ﺍﺯ ﺗﺒﺪﻳﻞ ﺑﻪ ﺟﺮﻳﺎﻥ ﺑﺮﻕ ﺑﺮﺍﯼ ﻓﻬﻤﻴﺪﻩ ﺷﺪﻥ ﺑﻪ ﻣﻐﺰ ﻣﻨﺘﻘﻞ ﻣﻰ ﮐﻨﺪ.

ﻣﻴﮑﺮﻭﻓﻮﻥ ﻧﻴﺰ ﺍﺑﺰﺍﺭﯼ ﺑﺮﺍﯼ ﺗﺒﺪﻳﻞ ﺍﻣﻮﺍﺝ ﺻﻮﺗﯽ ﺩﺭ ﻫﻮﺍ ﺑﻪ ﺍﻣﻮﺍﺝ ﺍﻟﮑﺘﺮﻳﮑﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ ﺗﻮﺍﻥ ﺁﻥ ﺭﺍ ﺗﻘﻮﻳﺖ، ﺿﺒﻂ ﻭ ﺑﺎﺯﭘﺨﺶ ﮐﺮﺩ. ﺗﻔﺎﻭﺕ ﺍﺻﻠﻰ ﺑﻴﻦ ﮔﻮﺵ ﺍﻧﺴﺎﻥ ﻭ ﻣﻴﻜﺮﻭﻓﻮﻥ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺍﻭﻟﻰ ﺑﻪ ﻳﮏ ﺩﺳﺘﮕﺎﻩ ﺗﺸﺨﻴﺺ ﺑﺴﻴﺎﺭ ﭘﻴﭽﻴﺪﻩ ﺑﻪ ﺍﺳﻢ ﻣﻐﺰ ﻣﺘﺼﻞ ﺍﺳﺖ ﻭ ﺩﻭﻣﻰ، ﺣﺪﺍﻗﻞ ﺗﺎ ﺍﻣﺮﻭﺯ، ﺑﻪ ﻛﻠﻰ ﺍﺣﻤﻖ ﺍﺳﺖ.

ﻣﻌﻨﻰ ﺍﻳﻦ ﺍﺧﺘﻼﻑ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﻣﺜﻼ ﻭﻗﺘﯽ ﺷﻤﺎ ﺩﺭ ﻳﮏ ﺟﺎﯼ ﺷﻠﻮﻍ ﺑﺎ ﺩﻭﺳﺖ ﺗﺎﻥ ﺻﺤﺒﺖ ﻣﯽ ﮐﻨﻴﺪ ﺑﺮﺍﯼ ﺷﻨﻴﺪﻥ ﺣﺮﻑ ﻫﺎﯼ ﺍﻭ ﻣﺸﮑﻞ ﺯﻳﺎﺩﯼ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﺍﻣﺎ ﺍﮔﺮ ﺩﺭ ﻫﻤﺎﻧﺠﺎﻳﯽ ﮐﻪ ﺍﻳﺴﺘﺎﺩﻩ ﺍﻳﺪ، ﺻﺪﺍﯼ ﺍﻭ ﺭﺍ ﺿﺒﻂ ﮐﻨﻴﺪ ﻣﺘﻮﺟﻪ ﻣﯽ ﺷﻮﻳﺪ ﮐﻪ ﺩﺭ ﻻﺑﻼﯼ ﺍﺯﺩﺣﺎﻡ ﻣﺤﻞ ﺣﺮﻑ ﻫﺎﯼ ﺍﻭ ﺑﻪ ﮐﻠﯽ ﻧﺎﻣﻔﻬﻮﻡ ﺍﺳﺖ. ﺩﻟﻴﻠﺶ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﻐﺰ ﺍﻧﺴﺎﻥ ﺩﺭ ﻳﮏ ﻓﻀﺎﻯ ﭘﺮﻫﻤﻬﻤﻪ ﻫﻢ ﻣﻰ ﺗﻮﺍﻧﺪ ﮔﺰﻳﻨﻪ ﺩﻟﺨﻮﺍﻩ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻣﻴﺎﻥ ﺍﻧﺒﻮﻩ ﺻﺪﺍﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﮔﻮﺵ ﻣﻰ ﺭﺳﺪ ﺟﺪﺍ ﻛﻨﺪ؛ ﺍﻣﺎ ﻣﻴﻜﺮﻭﻓﻮﻥ ﻭ ﺩﺳﺘﮕﺎﻩ ﻫﺎﻯ ﺿﺒﻄﻰ ﮐﻪ ﺑﻪ ﺁﻥ ﻭﺻﻞ ﺷﺪﻩ ﺍﻧﺪ ﻗﺎﺩﺭ ﺑﻪ ﺗﻔﮑﻴﮏ ﺍﻣﻮﺍﺝ ﺻﻮﺗﻰ ﮐﻪ ﺩﺭﻳﺎﻓﺖ ﻣﻰ ﮐﻨﻨﺪ ﻧﻴﺴﺘﻨﺪ ﻭ ﺧﻮﺍﺳﺘﻪ ﻭ ﻧﺎﺧﻮﺍﺳﺘﻪ ﺭﺍ ﺑﺎ ﻫﻢ ﺿﺒﻂ ﻣﻰ ﮐﻨﻨﺪ.

ﺍﮔﺮ ﺩﺭ ﺷﮑﻞ ﻫﺎﯼ ۲ ﻭ ۳ ﺩﺭ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ﺑﻪ ﺟﺎﯼ ﺷﻨﻮﻧﺪﻩ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻗﺮﺍﺭ ﺑﺪﻫﻴﺪ ﺻﺪﺍﯼ ﺿﺒﻂ ﺷﺪﻩ ﺩﺭ ﺷﮑﻞ ۳ ﺑﺎ ﺻﺪﺍﯼ ﺿﺒﻂ ﺷﺪﻩ ﺩﺭ ﺷﮑﻞ ۲ ﺗﻔﺎﻭﺕ ﺍﺳﺎﺳﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺩﺭ ﻣﻮﺭﺩ ﺷﮑﻞ ۳ ﺑﺎﺯﺗﺎﺏ ﻫﺎﯼ ﺻﺪﺍ ﺍﺯ ﺟﻬﺎﺕ ﻣﺨﺘﻠﻒ ﺑﺮ ﮐﻴﻔﻴﺖ ﺁﻧﭽﻪ ﺿﺒﻂ ﻣﯽ ﺷﻮﺩ ﺗﺎﺛﻴﺮ ﺯﻳﺎﺩ ﻭ ﻣﺨﺮﺑﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﺩﺭﺑﺎﺭﻩ ﺍﻧﻮﺍﻉ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎ ﻭ ﺧﻮﺍﺹ ﺁﻧﻬﺎ ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ ﺍﻳﻢ. ﺩﺭ ﺍﻳﻨﺠﺎ ﻓﻘﻂ ﺑﻪ ﺍﺷﺎﺭﻩ ﻣﯽ ﮔﻮﺋﻴﻢ ﮐﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎ ﻗﺎﺑﻠﻴﺖ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﺩﺍﺭﻧﺪ ﻭ ﮐﻴﻔﻴﺖ ﺻﺪﺍﻳﯽ ﮐﻪ ﺿﺒﻂ ﻣﯽ ﮐﻨﻨﺪ ﺑﺎ ﻫﻢ ﻓﺮﻕ ﻣﯽ ﮐﻨﺪ. ﺑﻌﻀﯽ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎ ﺟﻬﺖ ﺩﺍﺭ ﻫﺴﺘﻨﺪ ﻭ ﺻﺪﺍﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺍﺯ ﻳﮏ ﺯﺍﻭﻳﻪ ﺧﺎﺹ ﻣﯽ ﺁﻳﺪ ﺿﺒﻂ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺑﻌﻀﯽ ﭼﻨﺪ ﺟﻬﺘﻪ ﻭ ﺑﺮﺍﯼ ﺿﺒﻂ ﻫﻤﻪ ﺻﺪﺍﻫﺎﯼ ﺍﻃﺮﺍﻑ ﺷﺎﻥ ﺳﺎﺧﺘﻪ ﺷﺪﻩ ﺍﻧﺪ.

ﺑﻌﻀﯽ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎﯼ ﮔﺮﺍﻥ ﻗﻴﻤﺖ ﺗﺮ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺑﺎ ﺩﮐﻤﻪ ﻫﺎﻳﯽ ﺯﺍﻭﻳﻪ ﺷﻨﻮﺩﺷﺎﻥ ﺭﺍ ﺑﺎﺯ ﻳﺎ ﺑﺴﺘﻪ ﮐﻨﻨﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺑﻪ ﺩﻟﺨﻮﺍﻩ ﺻﺪﺍﺑﺮﺩﺍﺭ ﻓﻀﺎﻫﺎﯼ ﺻﻮﺗﯽ ﻣﺘﻨﻮﻋﯽ ﺭﺍ ﭘﻮﺷﺶ ﺩﻫﻨﺪ.

ﺑﻌﻀﯽ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎ ﺍﺯ ﺑﺎﻃﺮﯼ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﻨﺒﻊ ﻧﻴﺮﻭ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ ﺍﻣﺎ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎﯼ ﺑﻬﺘﺮ ﻭ ﮔﺮﺍﻥ ﺗﺮ ﻣﻌﻤﻮﻻ ﺍﺯ ﻃﺮﻳﻖ ﺳﻴﻢ ﺭﺍﺑﻂ‌ﺷﺎﻥ ﺑﺮﻕ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺭﺍ ﺍﺯ ﺩﺳﺘﮕﺎﻩ ﺿﺒﻂ ﺻﻮﺗﯽ ﮐﻪ ﺑﻪ ﺁﻥ ﻣﺘﺼﻞ ﺍﻧﺪ، ﻣﯽ ﮔﻴﺮﻧﺪ. ﺍﻣﺎ ﻧﮑﺘﻪ ﺍﺻﻠﯽ ﺩﺭ ﻣﻮﺭﺩ ﻫﻤﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﮑﻞ ﻭ ﻧﺤﻮﻩ ﮐﺎﺭﺷﺎﻥ ﻧﻴﺴﺖ ﮐﻪ ﺍﻫﻤﻴﺖ ﺩﺍﺭﺩ؛ ﮐﻴﻔﻴﺖ ﺻﺪﺍﯼ ﺁﻧﻬﺎﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺷﻨﻮﻧﺪﻩ ﻣﻬﻢ ﺍﺳﺖ.

ﻫﻤﭽﻨﺎﻥ ﮐﻪ ﭘﻴﺶ ﺗﺮ ﮔﻔﺘﻪ ﺷﺪ ﺿﺒﻂ ﺻﺪﺍ ﺩﺭ ﺍﺳﺘﻮﺩﻳﻮ ﺑﺮﺍﯼ ﺟﺒﺮﺍﻥ ﻣﺸﮑﻼﺕ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻭ ﺻﺪﺍﯼ ﻣﺤﻴﻂ ﺭﺍﻩ ﺣﻞ ﻣﻨﺎﺳﺒﯽ ﺍﺳﺖ. ﻣﻬﻢ ﺗﺮﻳﻦ ﺷﺎﺧﺼﻪ ﻫﺎﯼ ﺍﺳﺘﻮﺩﻳﻮﯼ ﺻﺪﺍ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺍﺳﺖ:

- ﺩﺭ ﺑﺮﺍﺑﺮ ﺻﺪﺍﻫﺎﯼ ﺑﻴﺮﻭﻥ ﮐﺎﻣﻼ ﻋﺎﻳﻖ ﺑﻨﺪﯼ ﺷﺪﻩ ﺍﺳﺖ. 

- ﺩﻳﻮﺍﺭﻫﺎﻳﺶ ﻗﺪﺭﺕ ﺟﺬﺏ ﺻﺪﺍﯼ ﺯﻳﺎﺩ ﻭ ﺑﺎﺯﺗﺎﺏ ﺍﻧﺪﮐﯽ ﺩﺍﺭﻧﺪ. ﻳﻌﻨﯽ ﺧﻴﻠﯽ ﺯﻭﺩ ﺍﻣﻮﺍﺝ ﺻﻮﺗﯽ ﺭﺍ ﺑﻪ ﺻﻔﺮ ﻧﺰﺩﻳﮏ ﻣﯽ ﮐﻨﻨﺪ. (ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﮐﻴﻔﻴﺖ ﺑﺮﺍﯼ ﺿﺒﻂ ﮔﻔﺘﺎﺭ ﻣﻨﺎﺳﺐ ﺍﺳﺖ ﻭ ﺩﺭ ﺍﺳﺘﻮﺩﻳﻮﻫﺎﯼ ﺿﺒﻂ ﻣﻮﺳﻴﻘﯽ ﺑﻪ ﺗﻨﺎﺳﺐ ﻧﻮﻉ ﺁﻥ ﺍﺳﺘﻮﺩﻳﻮ ﺗﻐﻴﻴﺮ ﺷﮑﻞ  ﭘﻴﺪﺍ ﻣﯽ ﮐﻨﺪ.)

- ﺗﻬﻮﻳﻪ ﺧﻮﺏ ﺍﻣﺎ ﺑﺪﻭﻥ ﺻﺪﺍﺋﯽ ﺩﺍﺭﻧﺪ. 

- ﭘﻨﺠﺮﻩ ﻭ ﻧﻘﻄﻪ ﺩﻳﺪ ﻣﻨﺎﺳﺒﯽ ﺑﻪ ﺍﺗﺎﻕ ﻓﺮﻣﺎﻥ ﻭ ﺩﺳﺘﮕﺎﻩ ﻫﺎﯼ ﺿﺒﻂ ﺻﺪﺍ ﺩﺍﺭﻧﺪ.
