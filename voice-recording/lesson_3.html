ﺑﻌﻀﯽ ﺗﻌﺮﻳﻒ ﻫﺎ ﻭ ﺧﻮﺍﺹ

‎ﺍﮐﺜﺮ ﮔﺰﺍﺭﺷﮕﺮﻫﺎ ﺍﺯ ﺿﺒﻂ ﺻﺪﺍ ﺩﺭ ﻣﺼﺎﺣﺒﻪ ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ. ﺿﺒﻂ ﺻﺪﺍ ﺑﺮﺍﯼ ﻳﮏ ﮔﺰﺍﺭﺵ ﺍﻓﺰﻭﻧﻪ ﺍﯼ ﺁﺳﺎﻥ ﺍﻣﺎ ﺑﺴﻴﺎﺭ ﻣﻔﻴﺪ ﺩﺭ ﮔﺰﺍﺭﺷﮕﺮﯼ ﺑﺮﺍﯼ ﺭﺳﺎﻧﻪ ﻫﺎﯼ ﻧﻮ ﺍﺳﺖ. ﺍﮔﺮ ﺷﻤﺎ ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺻﺪﺍﯼ ﻣﺼﺎﺣﺒﻪ ﺷﻮﻧﺪﻩ ﺭﺍ ﺿﺒﻂ ﻣﯽ ﮐﻨﻴﺪ ﭼﺮﺍ ﻧﺒﺎﻳﺪ ﺑﺎ ﮐﻤﯽ ﺑﻬﺘﺮ ﮐﺮﺩﻥ ﮐﻴﻔﻴﺖ ﺁﻥ ﺩﺭ ﻣﺘﻦ ﮔﺰﺍﺭﺵ ﻫﻢ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﻭ ﭼﻴﺰﯼ ﻓﺮﺍﺗﺮ ﺍﺯ ﻳﮏ ﻳﺎ ﺩﻭ ﻧﻘﻞ ﻗﻮﻝ ﮐﻮﺗﺎﻩ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﮐﺎﺭﺑﺮ ﻭ ﻣﺨﺎﻃﺐ ﮔﺰﺍﺭﺵ ﺧﻮﺩﺗﺎﻥ ﻗﺮﺍﺭ ﺩﻫﻴﺪ.

ﺩﺭ ﻳﮏ ﮔﻔﺘﮕﻮﯼ ﺑﻠﻨﺪ ﻳﮏ ﺳﺎﻋﺘﻪ ﻧﮑﺘﻪ ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺍﮔﺮ ﺁﻥ ﺭﺍ ﺑﻪ ﻣﺘﻨﯽ ۱۰۰۰ ﺗﺎ ۱۵۰۰ ﮐﻠﻤﻪ ﺍﯼ ﺑﺮﺍﯼ ﭼﺎﭖ ﺗﺒﺪﻳﻞ ﮐﻨﻴﺪ ﺣﺪﺍﮐﺜﺮ ﻣﻌﺎﺩﻝ ﺩﻩ ﺩﻗﻴﻘﻪ ﺍﺯ ﺣﺮﻑ ﻫﺎﯼ ﻣﺼﺎﺣﺒﻪ ﺷﻮﻧﺪﻩ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻧﻮﺷﺘﺎﺭ ﺍﺭﺍﺋﻪ ﺧﻮﺍﻫﻴﺪ ﮐﺮﺩ. ﺑﺮﺍﯼ ﻣﺨﺎﻃﺒﺎﻧﯽ ﮐﻪ ﻋﻼﻗﻤﻨﺪ ﺑﻪ ﺍﻃﻼﻋﺎﺕ ﺑﻴﺸﺘﺮ ﻭ ﮐﺎﻣﻞ‌ ﺗﺮ ﺍﺯ ﻣﻮﺿﻮﻉ ﻫﺴﺘﻨﺪ ﺍﻣﮑﺎﻥ ﺷﻨﻴﺪﻥ ﺗﻤﺎﻣﯽ ﻳﮏ ﮔﻔﺘﮕﻮ ﻓﺮﺻﺖ ﺑﺴﻴﺎﺭ ﻣﻐﺘﻨﻤﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺍﻣﺮﻭﺯ ﺑﺎ ﺩﺍﺷﺘﻦ ﺳﺎﻳﺖ ﻫﺎﯼ ﺍﻳﻨﺘﺮﻧﺘﯽ ﺣﺘﯽ ﺭﺍﺩﻳﻮ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻫﺎﯼ ﻣﻌﺘﺒﺮ ﺟﻬﺎﻥ ﻫﻢ ﮔﻔﺘﮕﻮﻫﺎ ﺭﺍ ﺩﺭ ﺩﻭ ﻧﺴﺨﻪ، ﻳﮑﯽ ﮐﻮﺗﺎﻩ ﺷﺪﻩ ﺑﺮﺍﯼ ﭘﺨﺶ ﺯﻧﺪﻩ ﻭ ﺩﻳﮕﺮﯼ ﮐﺎﻣﻞ ﺑﺮﺍﯼ ﺳﺎﻳﺖ ﺍﻳﻨﺘﺮﻧﺘﯽ ﺷﺎﻥ، ﻣﻨﺘﺸﺮ ﻣﯽ ﮐﻨﻨﺪ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﺳﺎﺩﻩ ﺗﺮﻳﻦ ﺷﮑﻞ، ﺑﺎﻧﺪ ﺻﻮﺗﯽ ﺩﺭ ﻳﮏ ﮔﺰﺍﺭﺵ ﭼﻨﺪﺭﺳﺎﻧﻪ ﺍﯼ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻓﺰﻭﻧﻪ ﺍﯼ ﺑﺮﺍﯼ ﺩﺍﺩﻥ ﺍﻃﻼﻋﺎﺕ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ. ﺍﻣﺎ ﺻﺪﺍ ﮔﺎﻩ ﺑﻴﺶ ﺍﺯ ﺍﻳﻦ ﻧﻘﺶ ﺩﺍﺭﺩ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺑﺎﻧﺪ ﺻﺪﺍﻳﯽ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺘﻮﺍﻧﺪ ﺑﻌﺪ ﺗﺎﺯﻩ ﺍﯼ ﺍﺯ ﺩﺍﺳﺘﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﮐﺎﺭﺑﺮﺍﻥ ﺑﻴﺎﻥ ﮐﻨﺪ. ﺍﮔﺮ ﺩﺭ ﺗﻈﺎﻫﺮﺍﺗﯽ ﮐﻪ ﺑﺎ ﻣﻤﻨﻮﻋﻴﺖ ﻫﺎﯼ ﺣﮑﻮﻣﺘﯽ ﻣﻮﺍﺟﻪ ﺍﺳﺖ ﻳﮏ ﺗﮑﻪ ﺻﺪﺍﯼ ﺻﺤﻨﻪ ﻭ ﺣﺮﻑ ﻫﺎﯼ ﻣﺮﺩﻡ، ﻭﻟﻮ ﺑﺎ ﮐﻴﻔﻴﺖ ﺿﺒﻂ ﭘﺎﺋﻴﻦ، ﺑﻪ ﺭﻭﺍﻳﺖ ﻣﮑﺘﻮﺏ ﺷﻤﺎ ﺍﺿﺎﻓﻪ ﺷﻮﺩ ﮐﺎﺭﺑﺮ ﺗﺼﻮﺭﯼ ﺍﺯ ﺍﺑﻌﺎﺩ ﺟﻤﻌﻴﺖ ﭘﻴﺪﺍ ‌ﻣﯽ ﮐﻨﺪ. ﻃﺒﻌﺎ ﻳﮏ ﺑﺮﻳﺪﻩ ﻭﻳﺪﺋﻮﻳﯽ ﺍﺯ ﺻﺤﻨﻪ ﺑﺴﻴﺎﺭ ﻓﺮﺍﺗﺮ ﺍﺯ ﻣﺘﻦ ﻳﺎ ﺻﺪﺍﯼ ﺗﻨﻬﺎ، ﮐﺎﺭﺁﺋﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﺷﻤﺎ ﺑﺎ ﮔﺰﺍﺭﺵ ﻫﺎﯼ ﺭﺍﺩﻳﻮﺋﯽ ﺁﺷﻨﺎﻳﯽ ﺩﺍﺭﻳﺪ. ﺩﺭ ﻫﺮ ﺑﺮﻧﺎﻣﻪ ﺧﺒﺮﯼ ﭼﻨﺪﻳﻦ ﮔﺰﺍﺭﺵ ﮐﻮﺗﺎﻩ ﺭﺍﺩﻳﻮﺋﯽ ﺑﻪ ﻣﻨﻈﻮﺭ ﺗﻮﺿﻴﺢ ﻣﺸﺮﻭﺡ ﺗﺮ ﻳﮏ ﺧﺒﺮ ﻭ ﺭﻧﮓ ﺁﻣﻴﺰﯼ ﻣﺘﻨﻮﻉ ﺗﺮ ﺑﺮﻧﺎﻣﻪ ﺍﺭﺍﺋﻪ ﻣﯽ ﺷﻮﺩ. ﺍﻳﻦ ﮔﺰﺍﺭﺵ ﻫﺎ ﻣﻌﻤﻮﻻ ﺩﺭ ﻣﻮﺍﻗﻌﯽ ﮐﻪ‌ ﮔﺰﺍﺭﺷﮕﺮ ﻓﺮﺻﺖ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﺟﻤﻊ ﺁﻭﺭﯼ ﺍﻃﻼﻋﺎﺕ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺟﺬﺍﺏ ﺗﺮ ﻭ ﭘﺮ ﻣﻄﻠﺐ ﺗﺮﻧﺪ.

ﮔﺰﺍﺭﺵ ﻫﺎﯼ ﺻﻮﺗﯽ ﻣﻌﻤﻮﻻ ﺍﺯ ﻋﻮﺍﻣﻠﯽ ﻣﺜﻞ ﻣﺼﺎﺣﺒﻪ ﻫﺎ، ﺻﺪﺍﯼ ﮔﻮﻳﻨﺪﻩ، ﺻﺪﺍﻫﺎﯼ ﻣﺤﻴﻂ ﻭ ﮔﻬﮕﺎﻩ ﻣﻮﺳﻴﻘﯽ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺯﻣﺎﻥ ﺁﻧﻬﺎ ﺑﺴﺘﻪ ﺑﻪ ﻃﻮﻝ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺧﺒﺮﯼ ﺑﻴﻦ ﻳﮏ ﺗﺎ ﭼﻬﺎﺭ ﺩﻗﻴﻘﻪ ﺍﺳﺖ. ﭼﻨﻴﻦ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﻋﻼﻭﻩ ﺑﺮ ﺍﻧﺘﻘﺎﻝ ﺍﻃﻼﻋﺎﺗﯽ ﺍﺯ ﺯﺑﺎﻥ ﻣﺼﺎﺣﺒﻪ ﺷﻮﻧﺪﻩ ﻫﺎ ﻭ ﮔﺰﺍﺭﺷﮕﺮ ﺑﻪ ﻣﺨﺎﻃﺒﺎﻥ ﺑﺎ ﺍﻓﺰﻭﺩﻥ ﺻﺪﺍﻫﺎﯼ ﺯﻣﻴﻨﻪ، ﺗﺼﻮﺭﯼ ﺍﺯ ﻣﺤﻞ ﮔﺰﺍﺭﺵ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﺪ. ﺻﺪﺍﯼ ﻣﺮﻍ ﻭ ﺧﺮﻭﺳﯽ ﺩﺭ ﭘﺲ ﺯﻣﻴﻨﻪ ﻳﺎ ﺻﺪﺍﯼ ﻋﺒﻮﺭ ﺧﻮﺩﺭﻭﻫﺎ ﻭ ﺑﻮﻕ ﻭ ﺳﺮﻭﺻﺪﺍﯼ ﺗﺮﺍﻓﻴﮏ ﺍﻧﻮﺍﻉ ﻣﺘﻔﺎﻭﺗﯽ ﺍﺯ ﻓﻀﺎ ﺭﺍ ﺑﻪ ﺫﻫﻦ ﺷﻨﻮﻧﺪﻩ ﺗﺪﺍﻋﯽ ﺧﻮﺍﻫﻨﺪ ﮐﺮﺩ.

ﺑﺮﺍﯼ ﺳﺎﺧﺘﻦ ﻳﮏ ﮔﺰﺍﺭﺵ ﺻﻮﺗﯽ ﺧﻮﺏ ﮔﺰﺍﺭﺷﮕﺮ ﺑﺎﻳﺪ ﻣﻘﺪﻣﺎﺗﯽ‌ ﺭﺍ ﺩﺭﺑﺎﺭﻩ ﺻﺪﺍ ﺑﺪﺍﻧﺪ. ﺩﺭ ﮐﺎﺭ ﺑﺎ ﺻﺪﺍ ﻫﺪﻑ ﻣﺎ ﺗﻮﻟﻴﺪ ﺻﺪﺍﻯ ﺧﻮﺏ ﺍﺳﺖ. ﺻﺪﺍﻯ ﺧﻮﺏ ﺻﺪﺍﻳﻰ ﺍﺳﺖ ﻛﻪ ﺷﻨﻴﺪﻧﺶ ﺁﺯﺍﺭﻧﺪﻩ ﻧﺒﺎﺷﺪ، ﻣﮕﺮ ﺑﻪ ﻋﻤﺪ.

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ ﺻﺪﺍ ﺑﻪ ﺻﻮﺭﺕ ﻣﮑﺎﻧﻴﮑﻰ ﻭ ﺩﺭ ﺍﺛﺮ ﺍﻧﻘﺒﺎﺽ ﻭ ﺍﻧﺒﺴﺎﻁ ﺍﻣﻮﺍﺝ ﺳﺎﺧﺘﻪ، ﻣﻨﺘﻘﻞ ﻭ ﺩﺭﻳﺎﻓﺖ ﻣﻰ ﺷﻮﺩ. ﻣﻴﮑﺮﻭﻓﻮﻥ ﻫﺎ ﺍﺯ ﻃﺮﻳﻖ ﭘﺮﺩﻩ ﺍﻯ ﮐﻪ ﺣﺮﮐﺖ ﺭﺍ ﺑﻪ ﺟﺮﻳﺎﻥ ﺑﺮﻕ ﺗﺒﺪﻳﻞ ﻣﻰ ﮐﻨﺪ ﺻﺪﺍ ﺭﺍ ﻗﺎﺑﻞ ﺿﺒﻂ ﻣﯽ ﮐﻨﻨﺪ. ﺷﺪﺕ ﻭ ﺿﻌﻒ ﺍﻳﻦ ﺟﺮﻳﺎﻥ ﺑﺮﻕ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﻧﺤﻮﻩ‌ ﻭ ﻣﻴﺰﺍﻥ ﻓﺸﺎﺭ‌ ﻫﻮﺍﺳﺖ. ﺑﻠﻨﺪﮔﻮﻫﺎ ﺑﺎ ﻟﺮﺯﻩ ﻫﺎﻯ ﭘﺮﺩﻩ ﺷﺎﻥ ﺑﻪ ﻋﻘﺐ ﻭ ﺟﻠﻮ، ﻫﻮﺍﻯ ﺍﻃﺮﺍﻑ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺑﻪ ﺣﺮﮐﺖ ﺩﺭ ﻣﻰ ﺁﻭﺭﻧﺪ ﻭ ﻧﻮﺳﺎﻥ ﻫﺎﯼ ﺑﺮﻗﯽ ﺭﺍ ﮐﻪ ﺍﺯ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺩﺭﻳﺎﻓﺖ ﺷﺪﻩ ﺩﻭﺑﺎﺭﻩ ﺑﻪ ﺻﻮﺭﺕ ﻣﻮﺝ ﺑﺎﺯﺗﻮﻟﻴﺪ ﻣﯽ ﮐﻨﻨﺪ.

ﮔﻮﺵ ﺍﻧﺴﺎﻥ ﻫﻢ ﺑﻪ ﻧﺤﻮ ﻣﺸﺎﺑﻬﻰ ﻋﻤﻞ ﻣﻰ ﮐﻨﺪ. ﻳﻌﻨﻰ ﻧﻮﺳﺎﻧﺎﺕ ﭘﺮﺩﻩ ﮔﻮﺵ ﺑﻪ ﺟﺮﻳﺎﻥ ﺑﺮﻕ ﺗﺒﺪﻳﻞ ﻣﻰ ﺷﻮﺩ ﻭ ﺑﻪ ﻣﻐﺰ ﻣﻨﺘﻘﻞ ﻣﻰ ﮔﺮﺩﺩ. ﺻﺪﺍﻫﺎﻳﯽ ﮐﻪ ﻣﺎ ﻣﻰ ﺷﻨﻮﻳﻢ، ﺣﺘﯽ ﺻﺪﺍﻫﺎﻯ ﻣﻤﺘﺪ، ﺩﺭ ﻭﺍﻗﻊ ﺟﺎ‌ﺑﻪ ﺟﺎﻳﯽ ﺳﺮﻳﻊ ﻭ ﻣﺪﺍﻭﻡ ﻫﻮﺍ ﻫﺴﺘﻨﺪ. ﺍﮔﺮ ﻓﺸﺎﺭ ﻫﻮﺍ ﻧﻮﺳﺎﻥ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺻﺪﺍﻳﯽ ﻭﺟﻮﺩ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﻣﺎ ﺻﺪﺍ ﺭﺍ ﺑﻪ ﺷﮑﻞ ﻣﻮﺝ ﺗﺼﻮﻳﺮ ﻣﻰ ﮐﻨﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺷﮑﻞِ ﻧﻤﺎﻳﺶ، ﻓﺮﮐﺎﻧﺲ (ﺍﺭﺗﻌﺎﺵ‌ﻫﺎﻯ) ﺻﺪﺍ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﺍﻧﻘﺒﺎﺽ ﻭ ﺍﻧﺒﺴﺎﻁ ﻫﻮﺍ ﻭ ﺩﺍﻣﻨﻪ ﺁﻥ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻣﻰ ﺷﻮﺩ.

ﺷﮑﻞ ﻭ ﻧﻤﻮﻧﻪ ﻳﮏ ﺗﮏ ﺻﺪﺍﻯ ﻣﻼﻳﻢ ﺑﺎ ﺗﻨﺎﻭﺏ ۴۴۰ ﻫﺮﺗﺰ

ﻧﻤﻮﻧﻪ ﻧﻮﺍﻯ ﺍﺭﮐﺴﺘﺮ (ﻓﺎﻳﻞ Sample3 ﺭﺍ ﮐﻪ ﺿﻤﻴﻤﻪ ﺷﺪﻩ ﺍﺳﺖ ﺩﺍﻧﻠﻮﺩ ﮐﻨﻴﺪ.)

 

ﺑﻌﻀﻰ ﺗﻌﺮﻳﻒ‌ﻫﺎ:
ﺩﺳﻴﺒﻞ (dB)
ﺩﺳﻴﺒﻞ ﻭﺍﺣﺪ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﻯ ﻗﺪﺭﺕ ﺻﺪﺍﺳﺖ. ﺩﺳﻴﺒﻞ ﻭﺍﺣﺪ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﻯ ﻋﺠﻴﺒﻰ ﺍﺳﺖ. ﮔﻮﺵ ﺁﺩﻡ ﻗﺎﺩﺭ ﺑﻪ ﺷﻨﻴﺪﻥ ﻃﻴﻒ ﻭﺳﻴﻌﻰ ﺍﺯ ﺻﺪﺍﻫﺎﺳﺖ؛ ﺍﺯ ﻣﺎﻟﻴﺪﻥ ﻧﻮﮎ ﺍﻧﮕﺸﺖ ﺭﻭﻯ ﭘﺎﺭﭼﻪ ﺗﺎ ﺻﺪﺍﻯ ﻣﻮﺗﻮﺭ ﻳﮏ ﺟﺖ ﮐﻪ ﺗﻘﺮﻳﺒﺎ ﺩﻩ ﻣﻴﻠﻴﺎﺭﺩ ﻣﺮﺗﺒﻪ ﺍﺯ ﺁﻥ ﻗﻮﯼ ﺗﺮ ﺍﺳﺖ. ﺩﺭ ﻣﻘﻴﺎﺱ ﺩﺳﻴﺒﻞ ﺻﺪﺍﻯ ﻧﺰﺩﻳﮏ ﺑﻪ ﺳﮑﻮﺕ ﻣﻄﻠﻖ ﺭﺍ ﺻﻔﺮ ﺩﺳﻴﺒﻞ ﻓﺮﺽ ﻣﻰ ﮐﻨﻨﺪ.

ﺻﺪﺍﻳﯽ ﮐﻪ ﺩﻩ ﻣﺮﺗﺒﻪ ﺍﺯ ﺁﻥ ﻗﻮﻯ ﺗﺮ ﺑﺎﺷﺪ ۱۰ ﺩﺳﻴﺒﻞ ﺍﺳﺖ ﻭ ﺻﺪﺍﻳﯽ ﮐﻪ ﺻﺪ ﻣﺮﺗﺒﻪ ﻗﻮﻯ ﺗﺮ ﺑﺎﺷﺪ ۲۰ ﺩﺳﻴﺒﻞ ﺍﺳﺖ. ﺑﺮﺍﻯ ﺍﻳﻦ ﻛﻪ ﺗﺼﻮﺭﻯ ﺍﺯ ﻣﻌﻨﻰ ﺁﻥ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﺻﺪﺍﻯ ﻧﻔﺲ ﻛﺸﻴﺪﻥ ﺍﻧﺴﺎﻥ ﺣﺪﻭﺩ ﺩﻩ ﺩﺳﻴﺒﻞ ﺍﺳﺖ. ﻳﮏ ﮔﻔﺘﮕﻮﻯ ﻣﻌﻤﻮﻟﻰ ﺑﻴﻦ ٥٠ ﺗﺎ ٦٥ ﺩﺳﻴﺒﻞ ﺍﺳﺖ. ﺻﺪﺍﻯ ﻣﺎﺷﻴﻦ ﭼﻤﻦ ﺯﻧﻰ ﺣﺪﻭﺩ ۹۰ ﺩﺳﻴﺒﻞ ﺍﺳﺖ. ﺑﻮﻕ ﻣﻌﻤﻮﻟﻰ ﻣﺎﺷﻴﻦ ﺣﺪﻭﺩ ۱۱۰ ﺩﺳﻴﺒﻞ ﺍﺳﺖ. ﻣﻮﺗﻮﺭ ﺟﺖ ۱۲۰ ﺩﺳﻴﺒﻞ ﺍﺳﺖ ﻭ ﺻﺪﺍﻯ ﮔﻠﻮﻟﻪ ۱۴۰ ﺩﺳﻴﺒﻞ.

ﺍﻟﺒﺘﻪ ﻫﻤﻪ ﺍﻳﻦ ﺍﻧﺪﺍﺯﻩ ﻫﺎ ﺩﺭ ﻓﺎﺻﻠﻪ ﻧﺰﺩﻳﮏ ﺑﺎ ﻣﻨﺒﻊ ﺻﺪﺍ ﺩﺭ ﺍﻳﻦ ﺣﺪﻭﺩ‌ ﺍﺳﺖ ﻭ ﺑﺎ ﺯﻳﺎﺩ ﺷﺪﻥ ﻓﺎﺻﻠﻪ ﻃﺒﻌﺎ ﺍﺯ ﻗﺪﺭﺕ ﺻﺪﺍ ﮐﻢ ﻣﻰ ﺷﻮﺩ. ﻫﺮ ﺻﺪﺍﻳﻰ ﮐﻪ ﺑﻴﺶ ﺍﺯ ۸۵ ﺩﺳﻴﺒﻞ ﻗﺪﺭﺕ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻣﻰ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﮔﻮﺵ ﺁﺳﻴﺐ ﺑﺰﻧﺪ.

ﺑﻴﺖ ﻭ ﺑﺎﻳﺖ ﻭ ﺗﻌﺎﺭﻳﻒ ﺩﻳﮕﺮ  ﺩﺭ ﺍﻭﺍﺋﻞ ﺩﻫﻪ‌ ۱۹۶۰ ﻣﻴﻼﺩﻯ ﺭﻭﺵ ﻣﻨﺎﺳﺒﻰ ﺑﺮﺍﻯ ﺗﺒﺪﻳﻞ ﺻﺪﺍ ﺑﻪ ﺑﻴﺖ ﻫﺎ ﻭ ﺑﺎﻳﺖ ﻫﺎ ﭘﻴﺪﺍ ﺷﺪ. ﻣﺒﺪﻝ ﻫﺎﻳﯽ ﺳﺎﺧﺘﻪ ﺷﺪ ﮐﻪ ﻫﺮ ﺛﺎﻧﻴﻪ ﺻﺪﺍﻯ ﺁﻧﺎﻟﻮﮒ ﺭﺍ ﺑﻪ ﻧﺰﺩﻳﮏ ﺳﻴﺼﺪ ﻫﺰﺍﺭ ﺟﺮﻳﺎﻥ ﺑﺎﺯ ﻭ ﺑﺴﺘﻪ ﺑﺮﻕ ﻳﺎ ﺑﻪ ﺗﻌﺒﻴﺮﻯ ﺻﻔﺮﻫﺎ (۰) ﻭ ﻳﮏ ﻫﺎ (۱) ﺗﺒﺪﻳﻞ ﻣﻰ ﮐﺮﺩ. ﺑﺎ ﺍﻳﻦ ﺗﺪﺑﻴﺮ ﺑﺎﺯﺳﺎﺯﻯ ﺻﺪﺍ ﺑﻪ ﺻﻮﺭﺕ ﺩﻳﺠﻴﺘﺎﻝ ﻣﻴﺴﺮ ﺷﺪ. ﺑﺎ ﻫﻤﻴﻦ ﺍﻣﮑﺎﻥ ﺍﺳﺖ ﮐﻪ ﺍﻣﺮﻭﺯ ﺻﺪﺍﻯ ﮐﺸﺘﻰ ﻭ ﻗﻄﺎﺭ ﻭ ﺩﻭﺳﺖ ﻫﻤﺮﺍﻩ ﺷﻤﺎ ﺑﻪ ﺻﻮﺭﺕ ﻣﺠﻤﻮﻋﻪ ﺍﻯ ﺍﺯ ﺻﻔﺮ ﻭ ﻳﮏ ﻗﺎﺑﻞ ﺍﻧﺘﻘﺎﻝ ﺑﻪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺷﺪﻩ ﺍﺳﺖ.

ﺍﻟﺒﺘﻪ ﺑﺎﺯﺳﺎﺯﻯ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺷﺒﻴﻪ ﺻﺪﺍﻯ ﺍﻭﻟﻴﻪ ﭼﻨﺪﺍﻥ ﺗﻮﻓﻴﻖ ﻣﻬﻤﻰ ﻧﺒﻮﺩ ﭼﻮﻥ ﺳﺎﺑﻘﻪ ﺿﺒﻂ ﻭ ﺗﮑﺮﺍﺭ ﺻﺪﺍﻫﺎ ﺑﻪ ﺻﻮﺭﺕ ﺁﻧﺎﻟﻮﮒ ﺑﺮ ﺭﻭﻯ ﺻﻔﺤﻪ ﻭ ﻧﻮﺍﺭ ﺍﺯ ﭘﻴﺶ ﻭﺟﻮﺩ‌ ﺩﺍﺷﺖ. ﺁﻧﭽﻪ ﺍﻳﻦ ﻓﻨﺎﻭﺭﻯ ﺗﺎﺯﻩ ﺭﺍ ﻣﻤﺘﺎﺯ ﻣﻰ‌ﮐﺮﺩ ﻗﺎﺑﻠﻴﺖ ﺩﺳﺘﮑﺎﺭﻯ ﻭ ﻭﻳﺮﺍﻳﺶ ﺍﻳﻦ ﺻﺪﺍﻫﺎ ﺑﻪ ﺷﮑﻠﻰ ﻣﺸﺎﺑﻪ ﺩﺳﺘﮑﺎﺭﻯ ﺩﺭ ﻫﺮ ﺫﺧﻴﺮﻩ ﺍﻃﻼﻋﺎﺗﻰ ﺩﻳﮕﺮ ﺩﺭ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺑﻮﺩ. ﺍﻳﻦ ﺗﺤﻮﻝ ﺑﻮﺩ ﮐﻪ ﻣﻌﻨﻰ ﺻﺪﺍﺑﺮﺩﺍﺭﻯ، ﺻﺪﺍﭘﺮﺩﺍﺯﻯ ﻭ ﺣﺘﯽ ﺩﺭﻳﺎﻓﺖ ﻣﺎ ﺍﺯ ﻣﻮﺳﻴﻘﻰ ﺭﺍ ﻣﺘﺤﻮﻝ ﮐﺮﺩ.

ﺩﻳﺠﻴﺘﺎﻝ ﺷﺪﻥ ﺻﺪﺍ ﺍﻣﺎ ﺩﻭ ﻭﺍﺣﺪ ﺗﺎﺯﻩ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﻯ ﺑﺎ ﺧﻮﺩﺵ ﺁﻭﺭﺩ: ﺑﻴﺖ ﺭﻳﺖ ﺭﺯﻭﻟﻮﺷﻦ (Bit Rate Resolution) ﻭ ﺳﻤﭙﻞ ﺭﻳﺖ (Sample Rate).

Sample Rate ﺗﻌﺪﺍﺩ ﻧﻤﻮﻧﻪ ﻫﺎﻯ ﺍﻃﻼﻋﺎﺗﻰ ﺑﺮﺩﺍﺷﺘﻪ ﺷﺪﻩ ﺍﺯ ﻳﮏ ﺻﺪﺍ ﺩﺭ ﻳﮏ ﺛﺎﻧﻴﻪ ﺍﺳﺖ. ﻭﺍﺣﺪ ﺁﻥ ﻫﺮﺗﺰ ﺍﺳﺖ ﮐﻪ ﺑﺎ Hz ﻧﻤﺎﻳﺶ ﺩﺍﺩﻩ ﻣﻰ ﺷﻮﺩ. ﻳﮏ ﻫﺮﺗﺰ ﻳﻌﻨﻰ ﻳﮏ ﻧﻤﻮﻧﻪ ﺩﺭ ﺛﺎﻧﻴﻪ ﻭ ۴۴/۱ ﮐﻴﻠﻮﻫﺮﺗﺰ ﻳﻌﻨﻰ ۴۴۱۰۰ ﻧﻤﻮﻧﻪ ﺩﺭ ﺛﺎﻧﻴﻪ.

Bit rate Resolution ﺗﻌﺪﺍﺩ ﺻﻔﺮ ﻭ ﻳﮏ ﻫﺎﻯ ﺍﻃﻼﻋﺎﺗﻰ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺛﺒﺖ ﻫﺮ ﻧﻤﻮﻧﻪ ﺻﻮﺗﻰ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ. ﺑﻴﺖ ﺭﻳﺖ ۸، ﻳﻌﻨﻰ ﻫﺮ ﻧﻤﻮﻧﻪ ﺍﺯ ۸ ﺑﻴﺖ ﺗﺸﮑﻴﻞ ﺷﺪﻩ ﻭ ﺑﻴﺖ ﺭﻳﺖ ۶۴، ﻳﻌﻨﻰ ﺍﺯ ﺷﺼﺖ ﻭ ﭼﻬﺎﺭ ﺑﻴﺖ. ﻫﺮ ﭼﻪ ﺗﻌﺪﺍﺩ ﺑﻴﺖ ﻫﺎ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ ﺩﻗﺖ ﺻﺪﺍ ﺑﻴﺸﺘﺮ ﺍﺳﺖ. ﻳﮏ ﻣﺜﺎﻝ ﺑﺮﺍﻯ ﺩﺭﮎ ﺑﻬﺘﺮ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﻰ ﺗﻮﺍﻧﻴﺪ ﻟﺒﺎﺱ ﺗﺎﺯﻩ ﺩﻭﺳﺖ ﺗﺎﻥ ﺭﺍ ﺩﺭ ۸ ﮐﻠﻤﻪ ﺗﻌﺮﻳﻒ ﮐﻨﻴﺪ ﻳﺎ ﺩﺭ ۱۲۸ ﮐﻠﻤﻪ. ﻭﺍﺿﺢ ﺍﺳﺖ ﮐﻪ ﺩﺭ ۱۲۸ ﮐﻠﻤﻪ ﺍﻃﻼﻋﺎﺕ ﺑﻴﺸﺘﺮ ﻭ ﺩﻗﻴﻖ ﺗﺮﻯ ﺑﻪ ﺷﻨﻮﻧﺪﻩ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺩ.
