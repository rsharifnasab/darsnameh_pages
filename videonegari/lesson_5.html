ﺏ- ﺍﺯ Zoom ﮐﺮﺩﻥ ﺑﺎ ﺩﻭﺭﺑﻴﻦ، ﺗﺎ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺑﻬﺘﺮ ﺍﺯ Zoom ﮐﺮﺩﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﭘﺲ ﺍﺯ ﺍﻳﻨﮑﻪ ﺷﺎﺕ ﻋﺮﻳﺾ ﺗﺮ ﺭﺍ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﮐﺮﺩﻳﺪ، ﺑﻪ ﻣﺤﻞ ﻭﺍﻗﻌﻪ (ﺳﻮﮊﻩ ﺍﯼ ﮐﻪ ﻣﯽ ﺧﻮﺍﺳﺘﻴﺪ ﺭﻭﯼ ﺁﻥ ﺯﻭﻡ ﮐﻨﻴﺪ) ﻧﺰﺩﻳﮏ ﺷﻮﻳﺪ ﻭ ﺍﺯ ﻧﺰﺩﻳﮏ ﻭ ﺑﻪ ﺣﺎﻟﺖ ﻋﺎﺩﯼ (ﻭ ﻧﻪ Zoom ﺷﺪﻩ) ﺍﺯ ﺁﻥ ﻓﻴﻠﻢ ﺑﮕﻴﺮﻳﺪ. ﺍﮔﺮ ﺍﺣﺘﻤﺎﻝ ﺧﻄﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻳﺎ ﺍﻣﮑﺎﻥ ﺗﺼﻮﻳﺮ ﺑﺮﺩﺍﺭﯼ ﺍﺯ ﻧﺰﺩﻳﮏ ﻧﻴﺴﺖ، ﺍﺑﺘﺪﺍ ﻳﮏ ﺷﺎﺕ ﻋﺮﻳﺾ ﺭﺍ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺳﭙﺲ ﺭﻭﯼ ﺳﻮﮊﻩ ﻣﻮﺭﺩ ﻧﻈﺮ Zoom ﮐﻨﻴﺪ ﻭ ﺑﺪﻭﻥ ﺣﺮﮐﺖ ﺩﺍﺩﻥ ﺩﺳﺖ ﺧﻮﺩ، ﺑﺮﺍﯼ ﻣﺪﺕ ﺣﺪﺍﻗﻞ ﺩﻩ ﺛﺎﻧﻴﻪ، ﺍﺯ ﺳﻮﮊﻩ ﻣﻮﺭﺩ ﻧﻈﺮ ﻓﻴﻠﻢ ﺑﮕﻴﺮﻳﺪ. ﺍﮔﺮ ﻣﺘﻮﺟﻪ ﺷﺪﻳﺪ ﺷﺎﺕ Zoom ﺷﺪﻩ، ﮐﻴﻔﻴﺖ ﻣﻨﺎﺳﺐ ﻧﺪﺍﺭﺩ، ﺷﺎﻳﺪ ﻻﺯﻡ ﺑﺎﺷﺪ ﺭﻭﯼ ﺁﻥ ﺻﺤﺒﺖ ﮐﻨﻴﺪ ﻭ ﺗﻮﺿﻴﺢ ﺑﺪﻫﻴﺪ ﮐﻪ ﻣﻨﻈﻮﺭﺗﺎﻥ ﺍﺯ ﺍﻳﻦ ﺗﺼﻮﻳﺮﺑﺮﺩﺍﺭﯼ ﭼﻴﺴﺖ. ﺑﺎﺯ ﻫﻢ ﻳﺎﺩﺁﻭﺭﯼ ﻣﯽ ﺷﻮﺩ ﺩﺭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺍﺯ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﮐﻮﭼﮏ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻴﺪ ﻭ ﺭﻭﯼ ﺳﻮﮊﻩ ﺍﯼ Zoom ﻣﯽ ﮐﻨﻴﺪ، ﻟﺮﺯﺵ ﺩﺳﺘﺎﻥ، ﺭﺍﺣﺖ ﺗﺮ  ﺑﻪ ﺗﺼﻮﻳﺮ ﻣﻨﺘﻘﻞ ﻣﯽ ﺷﻮﺩ. ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﺩﻭ ﺩﺳﺘﯽ ﺑﮕﻴﺮﻳﺪ. ﺍﮔﺮ ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺍﺯ ﺑﺎﻻﯼ ﺳﺮ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﮐﻨﻴﺪ، ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﺑﺎﻻ ﺑﺮﺩﻩ ﻭ ﺩﺭ ﺻﻔﺤﻪ ﻧﻤﺎﻳﺶ ﺁﻥ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ ﺍﺯ ﭼﻪ ﻓﻴﻠﻢ ﻣﯽ ﮔﻴﺮﻳﺪ. ﻟﺮﺯﺵ ﺩﺳﺖ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺯﻳﺎﺩ ﺍﺳﺖ.

ﺩﻭﻡ- ﺩﺭ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺁﻣﺎﺗﻮﺭﯼ، ﻣﻌﻤﻮﻻ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺩﺭ ﺩﺭﻭﻥ ﺩﺳﺘﮕﺎﻩ ﺗﻌﺒﻴﻪ ﺷﺪﻩ ﺍﺳﺖ. ﺑﺎ ﺷﻨﺎﺳﺎﻳﯽ ﻣﺤﻞ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﻣﻴﮑﺮﻭﻓﻮﻥ، ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﺁﻥ ﺭﺍ ﺑﺎ ﺩﺳﺖ ﺧﻮﺩ ﻧﭙﻮﺷﺎﻧﻴﺪ. ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﭼﻮﻥ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺑﺮ ﺭﻭﯼ ﺩﻭﺭﺑﻴﻦ ﺳﻮﺍﺭ ﺍﺳﺖ (ﻭ ﺑﻪ ﺳﻮﮊﻩ ﻧﺰﺩﻳﮏ ﻧﻴﺴﺖ)، ﻣﻤﮑﻦ ﺍﺳﺖ ﺻﺪﺍﯼ ﺷﻤﺎ ﺭﺍ ﺑﻬﺘﺮ ﺍﺯ ﺻﺪﺍﯼ ﻣﺤﻴﻂ ﺿﺒﻂ ﮐﻨﺪ، ﺩﺭ ﻧﺘﻴﺠﻪ ﺳﺮ ﻭ ﺻﺪﺍ ﻧﮑﻨﻴﺪ.

ﺳﻮﻡ- ﻭﻗﺘﯽ ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺳﻮﮊﻩ ﺭﻭﺷﻦ ﮐﺮﺩﻳﺪ، ﭘﻴﺶ ﺍﺯ ﺿﺒﻂ ﮐﺮﺩﻥ، ﺗﺼﻮﻳﺮ ﺭﺍ ﺩﺭﺳﺖ ﻧﮕﺎﻩ ﮐﻨﻴﺪ ﺗﺎ ﺷﻔﺎﻓﻴﺖ ﮐﺎﻓﯽ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﺗﻨﻈﻴﻢ ﺗﻤﺮﮐﺰ ﻭ ﺷﻔﺎﻓﻴﺖ ﺍﮐﺜﺮ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺁﻣﺎﺗﻮﺭﯼ ﺩﺭ ﺣﺎﻟﺖ ﺧﻮﺩﮐﺎﺭ ﺍﺳﺖ (Auto Focus) ﺷﺎﻳﺪ ﻻﺯﻡ ﺑﺎﺷﺪ، ﺩﺭ ﺻﻮﺭﺕ ﻣﺎﺕ ﺑﻮﺩﻥ ﺗﺼﻮﻳﺮ، ﻳﮏ ﺑﺎﺭ ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﺭﻭﺷﻦ ﻭ ﺧﺎﻣﻮﺵ ﮐﻨﻴﺪ. ﺭﻭﺵ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﻌﻴﺎﺭﺗﺎﻥ ﺭﺍ، ﺷﻔﺎﻓﻴﺖ ﺗﺼﻮﻳﺮ ﺩﺭ ﺩﻭﺭﺗﺮﻳﻦ ﻧﻘﻄﻪ، ﻗﺮﺍﺭ ﺩﻫﻴﺪ.

ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﻪ ﺳﻴﺴﺘﻢ ﺧﻮﺩﮐﺎﺭ ﺩﻭﺭﺑﻴﻦ ﮐﻤﮏ ﮐﻨﻴﺪ ﺗﺎ ﺳﺮﻳﻌﺘﺮ ﻓﻮﮐﻮﺱ ﮐﻨﺪ، ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﺑﺪﻭﻥ ﺣﺮﮐﺖ ﺑﻪ ﺳﻤﺖ ﮐﻔﺶ ﻫﺎﯼ ﺧﻮﺩ (ﮐﻪ ﻧﻘﻄﻪ ﺑﯽ ﻧﻬﺎﻳﺖ ﺗﺼﻮﻳﺮ ﺣﺴﺎﺏ ﻣﯽ ﺷﻮﺩ) ﻧﮕﻪ ﺩﺍﺭﻳﺪ ﺗﺎ ﺷﻔﺎﻑ ﺗﺮﻳﻦ ﺗﺼﻮﻳﺮ ﺭﺍ ﺑﺒﻴﻨﻴﺪ. ﺣﺎﻻ ﺩﻭﺭﺑﻴﻦ ﺷﻤﺎ ﻗﺎﺩﺭ ﺍﺳﺖ ﺍﺯ ﺩﻭﺭﺗﺮﻳﻦ ﻧﻘﺎﻁ ﻧﻴﺰ، ﺗﺼﻮﻳﺮﯼ ﺷﻔﺎﻑ ﺿﺒﻂ ﮐﻨﺪ.

ﭼﻬﺎﺭﻡ- ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺑﻪ ﺳﻮﮊﻩ ﻧﺰﺩﻳﮏ ﺷﻮﻳﺪ. ﮔﻮﺷﯽ ﻳﺎ ﺩﻭﺭﺑﻴﻦ ﮐﻮﭼﮏ ﺧﻮﺩ ﺭﺍ ﺛﺎﺑﺖ ﻧﮕﻪ ﺩﺍﺭﻳﺪ، ﺣﺘﯽ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﻧﺮﺩﻩ ﻫﺎﯼ ﺧﻴﺎﺑﺎﻥ، ﻟﺒﻪ ﯼ ﭘﻨﺠﺮﻩ ﻳﺎ ﺣﺘﯽ ﺣﺎﺷﻴﻪ ﯼ ﺩﻳﻮﺍﺭ ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﮑﻴﻪ ﮔﺎﻩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﻭ ﺍﺯ ﻟﺮﺯﺵ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺩﻭﺭﺑﻴﻦ ﺑﮑﺎﻫﻴﺪ.

ﭘﻨﺠﻢ- ﻳﮏ ﻓﻴﻠﻢ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻣﻌﻤﻮﻻ ﺍﺯ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﺗﺼﻮﻳﺮ ﺩﻭﺭ ﻭ ﻧﺰﺩﻳﮏ ﻭ ﺧﻴﻠﯽ ﻧﺰﺩﻳﮏ ﺳﺎﺧﺘﻪ ﻣﯽ ﺷﻮﺩ. ﭘﺲ ﺍﺯ ﺍﻳﻨﮑﻪ ﺍﺯ ﻳﮏ ﻧﻤﺎﯼ ﺛﺎﺑﺖ ﺳﻮﮊﻩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﮐﺮﺩﻳﺪ، ﻧﻤﺎﻫﺎﯼ ﺑﺴﺘﻪ ﻳﺎ ﻧﺰﺩﻳﮑﺘﺮ ﺍﺯ ﺣﺎﺷﻴﻪ ﯼ ﺳﻮﮊﻩ ﺭﺍ ﻧﻴﺰ ﺗﻬﻴﻪ ﮐﻨﻴﺪ.

ﺷﺸﻢ- ﺗﺼﻮﻳﺮ ﮐﻢ ﻧﻮﺭ، ﺑﯽ ﺍﺭﺯﺵ ﺍﺳﺖ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺩﺭ ﺑﻬﺘﺮﻳﻦ ﻧﻮﺭ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺩﺭ ﻧﻮﺭ ﺭﻭﺯ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﻣﯽ ﮐﻨﻴﺪ، ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﺑﺎﻳﺴﺘﻴﺪ ﮐﻪ ﻧﻮﺭ ﺧﻮﺭﺷﻴﺪ ﭘﺸﺖ ﺳﺮ ﺷﻤﺎ ﻭ ﺩﻭﺭﺑﻴﻦ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺩﺭ ﻣﺤﻴﻂ ﺳﺮﺑﺴﺘﻪ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﻣﯽ ﮐﻨﻴﺪ، ﻣﻨﺒﻊ ﻧﻮﺭ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﺧﻮﺩ ﻭ ﺩﻭﺭﺑﻴﻦ ﺑﮕﺬﺍﺭﻳﺪ. ﻣﺮﺍﻗﺐ ﺑﺎﺷﻴﺪ ﺳﺎﻳﻪ ﺍﻳﺠﺎﺩ ﻧﮑﻨﻴﺪ. ﺍﮔﺮ ﺩﺭ ﺷﺐ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﻣﯽ ﮐﻨﻴﺪ، ﺳﻮﮊﻩ ﺭﺍ ﺑﻪ ﻣﻨﺒﻊ ﻧﻮﺭ ﻧﺰﺩﻳﮏ ﮐﻨﻴﺪ. ﺍﮔﺮ ﻣﻨﺒﻊ ﻧﻮﺭﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻧﺪﺍﺭﻳﺪ، ﺍﺯ ﺍﻣﮑﺎﻥ ﻓﻴﻠﻤﺒﺮﺩﺍﺭﯼ ﺩﺭ ﺷﺐ ﺩﻭﺭﺑﻴﻦ (Night Mode) ﮐﻪ ﺩﺭ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺁﻣﺎﺗﻮﺭﯼ ﻧﻴﺰ ﻣﻮﺟﻮﺩ ﺍﺳﺖ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺗﺼﻮﻳﺮ ﺷﻤﺎ ﺳﻴﺎﻩ ﻭ ﺳﻔﻴﺪ ﻭﻟﯽ ﭘﺮﻧﻮﺭ ﻣﯽ ﺷﻮﺩ.

ﻫﻔﺘﻢ- ﺑﻬﺘﺮ ﺍﺳﺖ ﻫﺮ ﺷﺎﺕ ﺭﺍ ﺩﺭ ﻳﮏ ﻓﺎﻳﻞ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﻳﻌﻨﯽ ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﺭﻭﺷﻦ ﮐﻨﻴﺪ، ﻳﮏ ﺷﺎﺕ ﺗﺼﻮﻳﺮ ﺑﮕﻴﺮﻳﺪ ﻭ ﺗﺼﻮﻳﺮﺑﺮﺩﺍﺭﯼ ﺭﺍ ﻣﺘﻮﻗﻒ ﮐﻨﻴﺪ. ﺑﺪﻳﻦ ﺗﺮﺗﻴﺐ، ﺑﺮ ﺭﻭﯼ ﻳﮏ ﻓﺎﻳﻞ، ﻳﮏ ﺷﺎﺕ ﺿﺒﻂ ﺷﺪﻩ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺷﺎﺕ ﺑﻌﺪﯼ ﺩﻭﺑﺎﺭﻩ ﺩﮐﻤﻪ ﯼ ﺿﺒﻂ ﺭﺍ ﺑﺰﻧﻴﺪ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺗﺮﺗﻴﺐ ﺿﺒﻂ ﮐﻨﻴﺪ. ﺍﻳﻨﮕﻮﻧﻪ، ﮐﺎﺭ ﺍﻧﺘﻘﺎﻝ، ﺗﻔﮑﻴﮏ ﻭ ﻣﻮﻧﺘﺎﮊ ﺳﺎﺩﻩ ﺗﺮ ﻣﯽ ﺷﻮﺩ.

ﻫﺸﺘﻢ- ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻣﺰﻳﺖ ﺗﺼﻮﻳﺮﯼ ﮐﻪ ﺿﺒﻂ ﻣﯽ ﮐﻨﻴﺪ، ﻋﻼﻭﻩ ﺑﺮ ﺛﺒﺖ ﮐﺮﺩﻥ ﻭﺍﻗﻌﻪ، ﺛﺒﺖ ﮐﺮﺩﻥ ﻭﺍﮐﻨﺶ ﺗﻤﺎﺷﺎﭼﯽ ﻫﺎ ﻧﻴﺰ ﺍﺳﺖ. ﺍﮔﺮ ﻣﺜﻼ ﺍﺯ ﺩﺭﮔﻴﺮﯼ ﺧﻴﺎﺑﺎﻧﯽ ﭘﻠﻴﺲ ﻭ ﻣﺮﺩﻡ، ﻓﻴﻠﻢ ﻣﯽ ﮔﻴﺮﻳﺪ، ﺗﺼﻮﻳﺮ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ ﺗﻮﺳﻂ ﭘﻠﻴﺲ ﻣﻬﻢ ﺍﺳﺖ، ﺍﻣﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﭼﺮﺧﺎﻧﺪﻥ ﺩﻭﺭﺑﻴﻦ ﺑﻪ ﺳﻤﺖ ﻣﺮﺩﻡ، ﻭﺍﮐﻨﺶ ﺁﻧﻬﺎ ﺭﺍ ﻧﻴﺰ ﺛﺒﺖ ﻭ ﺿﺒﻂ ﮐﻨﻴﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﻗﺎﻳﻊ ﻭ ﺣﻮﺍﺩﺙ ﺗﮑﺮﺍﺭﯼ ﻫﺴﺘﻨﺪ ﻭ  ﺗﻨﻬﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﺎ ﻭﺍﮐﻨﺶ ﻣﺮﺩﻡ ﺩﺭ ﻫﻢ ﺁﻣﻴﺨﺘﻪ ﻣﯽ ﺷﻮﻧﺪ، ﻣﻔﻬﻮﻡ ﺟﺪﻳﺪﯼ ﻣﯽ ﮔﻴﺮﻧﺪ.

ﻧﻬﻢ- ﺩﺭ ﺿﺒﻂ ﻫﺎﯼ ﻣﺨﻔﻴﺎﻧﻪ، ﻗﺮﺍﺭ ﺩﺍﺩﻥ ﺩﻭﺭﺑﻴﻦ ﺩﺭ ﮐﻴﻒ ﻳﺎ ﺟﻴﺐ ﻭ ﺭﻭﺷﻦ ﻧﮕﻪ ﺩﺍﺷﺘﻨﺶ (ﺑﻪ ﺳﻤﺖ ﺳﻮﮊﻩ ﺍﯼ ﻧﺎﻣﻌﻠﻮﻡ ﻳﺎ ﺑﯽ ﺭﺑﻂ) ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﮐﻪ ﺻﺪﺍ ﺭﺍ ﺿﺒﻂ ﮐﻨﺪ، ﮐﺎﺭﺁﻣﺪ ﺍﺳﺖ. ﻣﻌﻤﻮﻻ ﺟﺬﺍﺑﺘﺮﻳﻦ ﺻﺤﻨﻪ ﻫﺎ ﺩﺭ ﻓﻴﻠﻢ ﻫﺎﯼ ﻣﺴﺘﻨﺪ، ﻣﺘﻌﻠﻖ ﺑﻪ ﺯﻣﺎﻧﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ ﺩﺍﻧﻴﺪ ﻓﻴﻠﻤﺒﺮﺩﺍﺭ ﺁﻥ ﺭﺍ ﻣﺨﻔﻴﺎﻧﻪ ﺿﺒﻂ ﮐﺮﺩﻩ ﺍﺳﺖ.

ﺩﻫﻢ- ﺑﻌﺪ ﺍﺯ ﺍﻧﺘﻘﺎﻝ ﺗﺼﻮﻳﺮﻫﺎﻳﺘﺎﻥ ﺑﻪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻭ ﭘﻴﺶ ﺍﺯ ﺷﺮﻭﻉ ﻣﻮﻧﺘﺎﮊ، ﺷﺎﻳﺪ ﺑﺪ ﻧﺒﺎﺷﺪ ﺗﺼﺎﻭﻳﺮ ﻟﺮﺯﺍﻥ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺁﻣﺎﺗﻮﺭﯼ ﺭﺍ ﮐﻤﯽ ﺛﺎﺑﺖ ﺗﺮ ﻭ ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ ﺗﺮ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﺎﺭ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻟﺮﺯﻩ ﮔﻴﺮ ﻣﺠﺎﻧﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻟﺮﺯﺵ ﺩﺳﺘﺎﻥ ﺭﺍ ﺑﻪ ﺣﺪﺍﻗﻞ ﻣﯽ ﺭﺳﺎﻧﻨﺪ ﻭ ﺗﺼﻮﻳﺮ ﺭﺍ ﻭﻳﺮﺍﻳﺶ ﻣﯽ ﮐﻨﻨﺪ. ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﻫﺎﯼ ﺁﻥ ﺭﺍ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺍﻳﻨﺠﺎ ﺩﺭﻳﺎﻓﺖ ﮐﻨﻴﺪ.

ﻳﺎﺯﺩﻫﻢ- ﺍﮔﺮ ﺩﺭ ﻣﻮﻗﻌﻴﺘﯽ ﻣﺠﺒﻮﺭ ﺑﻪ ﻓﺮﺍﺭ ﻳﺎ ﺗﺮﮎ ﻧﺎﮔﻬﺎﻧﯽ ﺻﺤﻨﻪ ﺷﺪﻳﺪ، ﺩﺭ ﺻﻮﺭﺕ ﺍﻣﮑﺎﻥ ﺩﻭﺭﺑﻴﻦ ﺧﻮﺩ ﺭﺍ ﺭﻭﺷﻦ ﻧﮕﻪ ﺩﺍﺭﻳﺪ، ﻭﻟﻮ ﺍﻳﻨﮑﻪ ﺩﻭﺭﺑﻴﻦ ﺩﺭ ﺟﻴﺐ ﺷﻤﺎ ﺑﺎﺷﺪ. ﻫﻴﭻ ﺻﺤﻨﻪ ﺍﯼ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻧﺪﻫﻴﺪ.

ﺩﻭﺍﺯﺩﻫﻢ- ﺧﻮﺩ ﻭ ﺍﻃﺮﺍﻓﻴﺎﻥ ﺗﺎﻥ ﺭﺍ ﺑﻪ ﺧﻄﺮ ﻧﻴﺎﻧﺪﺍﺯﻳﺪ. ﺍﮔﺮ ﺍﺯ ﺣﺎﺩﺛﻪ ﺍﯼ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﻣﯽ ﮐﻨﻴﺪ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻌﺪﻫﺎ ﺑﺎﻋﺚ ﺑﻪ ﺩﺭﺩﺳﺮ ﺍﻓﺘﺎﺩﻥ ﺷﻤﺎ ﺷﻮﺩ، ﺩﺭ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﺧﻮﺩ ﺩﻗﺖ ﮐﻨﻴﺪ ﺗﺎ ﻣﻴﺰﺍﻥ ﺧﻄﺮﻫﺎﯼ ﺍﺣﺘﻤﺎﻟﯽ ﺭﺍ ﺑﻴﺶ ﺍﺯ ﭘﻴﺶ ﮐﺎﻫﺶ ﺑﺪﻫﻴﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺩﺭ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﺍﺯ ﭼﻬﺮﻩ ﻫﺎ ﻭ ﻳﺎ ﻋﻼﻣﺎﺕ ﻣﺸﺨﺼﻪ ﺩﻗﺖ ﮐﻨﻴﺪ.

ﺳﻴﺰﺩﻫﻢ- ﺑﺎﺯ ﻫﻢ ﺍﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﮐﻪ ﻫﻴﭻ ﺳﻮﮊﻩ ﺍﯼ ﺭﺍ ﻓﺪﺍ ﻧﮑﻨﻴﺪ. ﺣﺘﯽ ﺍﮔﺮ ﺛﺒﺖ ﺗﺼﻮﻳﺮ ﺑﻪ ﻗﻴﻤﺖ ﺯﻳﺮ ﭘﺎ ﮔﺬﺍﺷﺘﻦ ﺗﻤﺎﻡ ﻗﻮﺍﻧﻴﻦ ﺑﺎﻻ ﺑﻮﺩ. ﺗﻨﻬﺎ ﻳﮏ ﻗﺎﻧﻮﻥ ﺭﺍ ﻫﻴﭻ ﻭﻗﺖ ﺯﻳﺮ ﭘﺎ ﻧﮕﺬﺍﺭﻳﺪ: ﺭﻳﺴﮏ ﻧﮑﻨﻴﺪ ﻭ ﺟﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺧﻄﺮ ﻧﻴﺎﻧﺪﺍﺯﻳﺪ.
