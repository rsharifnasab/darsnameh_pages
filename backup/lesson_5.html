ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﮐﻠﻮﺩ (ﺑﮏ ﺁﭖ ﺁﻧﻼﻳﻦ)

ﺩﺭ ﺩﺭﻭﺱ ﮔﺬﺷﺘﻪ ﺍﺷﺎﺭﻩ ﺍﯼ ﮔﺬﺭﺍ ﺑﻪ ﺗﻬﻴﻪ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﺁﻧﻼﻳﻦ ﻳﺎ ﮐﻠﻮﺩ ﺩﺍﺷﺘﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻗﺼﺪ ﺩﺍﺭﻳﻢ ﺑﻪ ﺗﻌﺮﻳﻒ ﻭ ﺷﻴﻮﻩ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻦ ﺭﻭﺵ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﻭ ﻣﺰﺍﻳﺎ ﻭ ﻣﻌﺎﻳﺐ ﺁﻥ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﺩﺭ ﺍﻧﺘﻬﺎ ﻫﻢ ﺑﻪ ﭼﻨﺪ ﺳﺮﻭﻳﺲ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﺁﻧﻼﻳﻦ ﻭ ﺷﮕﺮﺩﻫﺎﯼ ﮐﺎﺭ ﺑﺎ ﺁﻧﻬﺎ ﻣﯽ ﭘﺮﺩﺍﺯﻳﻢ.

 

ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﮐﻠﻮﺩ ﭼﻴﺴﺖ؟

ﻓﻀﺎﯼ ﮐﻠﻮﺩ، ﻳﮏ ﺷﻴﻮﻩ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﺑﺮ ﺭﻭﯼ ﺍﻳﻨﺘﺮﻧﺖ ﺍﺳﺖ. ﺷﺮﮐﺖ ﻫﺎﯼ ﺍﺭﺍﺋﻪ ﺩﻫﻨﺪﻩ ﺳﺮﻭﻳﺲ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﺍﺑﺮﯼ (Cloud) ﺩﺭ ﻭﺍﻗﻊ ﮔﺮﻭﻩ ﻗﺎﺑﻞ ﺗﻮﺟﻬﯽ ﺍﺯ ﺳﺮﻭﺭﻫﺎﯼ ﺑﺎ ﺣﺠﻢ ﺑﺎﻻ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻧﺪ ﮐﻪ ﺑﺎ ﻳﮏ ﺭﺍﺑﻂ ﮐﺎﺭﺑﺮﯼ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﺗﺤﺖ ﻭﺏ، ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﻣﻴﺰﺍﻥ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺍﻳﻨﺘﺮﻧﺖ ﺩﺳﺘﻪ ﺑﻨﺪﯼ، ﺫﺧﻴﺮﻩ ﻭ ﺑﺎﺯﻳﺎﺑﯽ ﮐﻨﻨﺪ.

ﺍﺯ ﺩﻳﺪﮔﺎﻩ ﮐﺎﺭﺑﺮﺍﻥ ﻫﻢ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﮐﻠﻮﺩ ﺩﺭ ﻭﺍﻗﻊ ﻣﻘﺪﺍﺭ ﻧﺎﻣﺤﺪﻭﺩﯼ ﻓﻀﺎﯼ ﻫﺎﺭﺩ ﺁﻧﻼﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﻗﻴﻤﺖ ﭘﺎﻳﻴﻦ، ﺍﻣﻨﻴﺖ ﺑﺎﻻ ﻭ ﺩﺳﺘﺮﺳﯽ ﻧﺴﺒﺘﺎ ﺩﺍﺋﻢ، ﺍﻃﻼﻋﺎﺕ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻭ ﻟﭗ ﺗﺎﭖ ﻫﺎ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺧﻮﺩ ﺫﺧﻴﺮﻩ، ﻣﺮﺗﺐ ﺳﺎﺯﯼ ﻭ ﻗﺎﺑﻞ ﺑﺎﺯﻳﺎﺑﯽ ﻣﯽ ﮔﺮﺩﺍﻧﺪ.

 

ﭼﻪ ﮐﺴﺎﻧﯽ ﺑﻪ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﮐﻠﻮﺩ ﻧﻴﺎﺯ ﺩﺍﺭﻧﺪ؟

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮔﻔﺘﻴﻢ، ﻓﻀﺎﯼ ﮐﻠﻮﺩ ﻫﻤﺎﻥ ﻓﻀﺎﯼ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﺁﻧﻼﻳﻦ ﺍﺳﺖ. ﺷﻤﺎ ﺑﺮﺍﯼ ﺩﺳﺖ ﻳﺎﺑﯽ ﺑﻪ ﻓﻀﺎﯼ ﮐﻠﻮﺩ ﺑﻪ ﻳﮏ ﺧﻂ ﺍﻳﻨﺘﺮﻧﺖ ﭘﺮ ﺳﺮﻋﺖ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ. ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﻣﺨﺼﻮﺻﺎ ﺑﺮﺍﯼ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﺍﺯ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﻣﺴﺎﻟﻪ ﺯﻣﺎﻥ ﺩﺭﺑﺎﺭﻩ ﺁﻧﻬﺎ ﺍﻣﺮ ﻣﻬﻤﯽ ﺍﺳﺖ، ﺑﻴﺸﺘﺮ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﺑﻮﺩﻩ ﻭ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ.

ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﻬﻴﻪ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﺑﻪ ﺻﻮﺭﺕ ﮐﻠﻮﺩ، ﺍﻏﻠﺐ ﻫﻨﮕﺎﻣﯽ ﺑﻪ ﮐﺎﺭ ﻣﯽ ﺭﻭﻧﺪ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﻣﺪﺕ ﺯﻣﺎﻥ ﮐﻮﺗﺎﻩ ﻧﻴﺎﺯﯼ ﺑﻪ ﺑﺎﺯﻳﺎﺑﯽ ﺣﺠﻢ ﺑﺎﻻﻳﯽ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺯﻳﺮﺍ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ، ﺳﺮﻋﺖ ﻭ ﻫﺰﻳﻨﻪ ﺍﻳﻨﺘﺮﻧﺖ ﺩﻭ ﻋﺎﻣﻞ ﻣﻬﻢ ﻣﺤﺪﻭﺩ ﮐﻨﻨﺪﻩ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻤﺎ ﺍﺯ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﻫﺴﺘﻨﺪ.

ﻣﺜﻼ ﺷﻤﺎ ﻳﮏ ﻋﮑﺎﺱ ﻫﺴﺘﻴﺪ ﻭ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻧﯽ ﺍﺯ ﺗﻤﺎﻡ ﻣﺠﻤﻮﻋﻪ ﻋﮑﺲ ﺗﺎﻥ ﺗﻬﻴﻪ ﮐﻨﻴﺪ. ﺑﻪ ﻧﻈﺮ ﻣﯽ ﺭﺳﺪ ﮐﻪ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﺭﻭﯼ ﮐﻠﻮﺩ ﺍﻧﺘﺨﺎﺏ ﮐﺎﻣﻼ ﻣﻨﺎﺳﺒﯽ ﺍﺳﺖ. ﺯﻳﺮﺍ ﺷﻤﺎ ﻳﮏ ﻧﺴﺨﻪ ﺍﺯ ﻓﺎﻳﻞ ﻫﺎﯼ ﺗﺎﻥ ﺭﺍ ﺭﻭﯼ ﻟﭗ ﺗﺎﭖ ﻳﺎ ﻫﺎﺭﺩ ﺍﮐﺴﺘﺮﻧﺎﻝ ﺗﺎﻥ ﺩﺍﺭﻳﺪ ﻭ ﺗﻨﻬﺎ ﺩﺭ ﺻﻮﺭﺕ ﺑﺮﻭﺯ ﻣﺸﮑﻞ ﺑﺮﺍﯼ ﺁﻧﻬﺎ، ﺑﻪ ﻧﺴﺨﻪ ﮐﻠﻮﺩ ﻧﻴﺎﺯ ﭘﻴﺪﺍ ﻣﯽ ﮐﻨﻴﺪ. ﺍﻣﺎ ﺑﺴﺘﻪ ﺑﻪ ﺗﻌﺪﺍﺩ ﻋﮑﺲ ﻫﺎ ﻭ ﺳﺮﻋﺖ ﺍﻳﻨﺘﺮﻧﺖ، ﺯﻣﺎﻥ ﺗﻬﻴﻪ ﺍﻳﻦ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﺗﺎ ﭼﻨﺪ ﺳﺎﻋﺖ  ﻃﻮﻝ ﺑﮑﺸﺪ. ﺍﻣﺎ ﺩﺭ ﺧﺼﻮﺹ ﺑﺎﺯﻳﺎﺑﯽ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﻋﮑﺲ ﻫﺎ، ﺯﻣﺎﻥ ﺩﺭ ﺍﻳﻦ ﻣﻴﺎﻥ ﻧﻘﺶ ﺣﻴﺎﺗﯽ ﻧﺪﺍﺭﺩ.

ﭘﺲ ﻫﺮ ﮐﺲ ﮐﻪ ﺑﻪ ﻳﮏ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﺍﻣﻦ ﺑﺮ ﺭﻭﯼ ﺳﺮﻭﺭﯼ ﺧﺎﺭﺝ ﺍﺯ ﻣﺤﻴﻂ ﮐﺎﺭ ﺑﺪﻭﻥ ﻧﻴﺎﺯ ﺑﻪ ﻧﮕﺮﺍﻧﯽ ﺩﺭﺑﺎﺭﻩ ﻧﮕﻬﺪﺍﺭﯼ ﻭ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺯﻳﺮﺳﺎﺧﺖ ﻫﺎ، ﺍﺣﺘﻴﺎﺝ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺯ ﺷﻴﻮﻩ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﮐﻠﻮﺩ ﺑﺮﺍﯼ ﺗﻬﻴﻪ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ. ﺍﻟﺒﺘﻪ ﺑﻪ ﺍﻳﻦ ﺷﺮﻁ ﮐﻪ ﺑﺎﺯﻳﺎﺑﯽ ﺍﻃﻼﻋﺎﺕ ﺩﺭ ﺑﺎﺯﻩ ﺯﻣﺎﻧﯽ ﮐﻮﺗﺎﻩ ﻭ ﺑﺎ ﺳﺮﻋﺖ ﺑﺎﻻ، ﭼﻨﺪﺍﻥ ﺍﻫﻤﻴﺘﯽ ﺑﺮﺍﯼ ﻭﯼ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻳﺎ ﺍﻳﻨﮑﻪ ﺑﺘﻮﺍﻧﺪ ﭘﻬﻨﺎﯼ ﺑﺎﻧﺪ ﺍﻳﻨﺘﺮﻧﺘﯽ ﻣﻨﺎﺳﺒﯽ ﺭﺍ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺗﻬﻴﻪ ﮐﻨﺪ.

 

ﺭﻣﺰﻧﮕﺎﺭﯼ ﻭ ﺑﺎﺯﻳﺎﺑﯽ ﺍﻃﻼﻋﺎﺕ ﺩﺭ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﮐﻠﻮﺩ

ﺩﺭ ﺷﻴﻮﻩ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﮐﻠﻮﺩ، ﺩﺍﺩﻩ ﻫﺎﯼ ﺷﻤﺎ ﺑﺮ ﺭﻭﯼ ﻳﮏ ﺳﺮﻭﺭ ﺑﻴﺮﻭﻧﯽ، ﺧﺎﺭﺝ ﺍﺯ ﺧﺎﻧﻪ ﻳﺎ ﺍﺩﺍﺭﻩ ﺗﺎﻥ ﺍﻗﺎﻣﺖ ﺩﺍﺭﻧﺪ. ﭘﺲ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﻧﻴﺎﺯ ﺑﻪ ﻣﺤﺎﻓﻈﺖ ﺧﺎﺻﯽ ﺩﺍﺭﻧﺪ. ﺧﻮﺷﺒﺨﺘﺎﻧﻪ ﺍﮐﺜﺮ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﮐﻨﻮﻧﯽ ﺗﻬﻴﻪ ﻧﺴﺨﻪ ﭘﺸﺘﻴﺒﺎﻥ ﮐﻠﻮﺩ، ﺍﺯ ﻫﻨﺮ ﺭﻣﺰﻧﮕﺎﺭﯼ ﺩﺍﺩﻩ ﻫﺎ ﺑﻬﺮﻩ ﻣﯽ ﺑﺮﻧﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﻃﻼﻋﺎﺕ ﺷﻤﺎ ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﻓﺸﺮﺩﻩ ﺳﺎﺯﯼ ﻭ ﺭﻣﺰﻧﮕﺎﺭﯼ ﻣﯽ ﺷﻮﻧﺪ، ﮐﻪ ﺩﻳﮕﺮﺍﻥ ﻗﺎﺩﺭ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻧﻬﺎ ﻧﺒﺎﺷﻨﺪ. ﺩﺭ ﻣﻮﺍﺭﺩ ﺯﻳﺎﺩﯼ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺩﺭ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺷﻤﺎ ﺭﻣﺰﻧﮕﺎﺭﯼ ﺷﺪﻩ ﻭ ﺑﻪ ﺳﺮﻭﺭ ﮐﻠﻮﺩ ﺍﺭﺳﺎﻝ ﻣﯽ ﺷﻮﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺗﻮﺳﻂ ﺧﻮﺩ ﺻﺎﺣﺒﺎﻥ ﺍﻳﻦ ﺳﺮﻭﻳﺲ ﻫﻢ ﻗﺎﺑﻞ ﺩﺳﺘﺮﺱ ﻧﻴﺴﺖ. ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﺍﻃﻼﻋﺎﺕ ﺑﺴﻴﺎﺭ ﻣﻬﻤﯽ ﺩﺍﺭﻳﺪ ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻗﺒﻞ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﺗﺤﻘﻴﻘﺎﺕ ﻣﻔﺼﻠﯽ ﺑﮑﻨﻴﺪ. 

ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺤﺚ ﻫﺎﯼ ﻣﻔﺼﻠﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺍﺭﺍﻳﻪ ﮐﻨﻨﺪﮔﺎﻥ ﺳﺮﻭﻳﺲ ﻫﺎﯼ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﺍﺑﺮﯼ ﺗﺎ ﭼﻪ ﺣﺪ ﺑﻪ ﺍﻃﻼﻋﺎﺕ ﮐﺎﺭﺑﺮﺍﻥ ﺩﺳﺘﺮﺳﯽ ﺩﺍﺭﻧﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺩﺭ ﻣﻮﺭﺩ ﺳﺮﻭﻳﺲ ﺩﺭﺍﭖ ﺑﺎﮐﺲ ﺍﻳﻨﻄﻮﺭ ﺑﻪ ﻧﻈﺮ ﻣﻴﺮﺳﺪ ﮐﻪ ﺳﺎﺯﻧﺪﮔﺎﻥ ﺍﻳﻦ ﺳﺮﻭﻳﺲ ﺍﻣﮑﺎﻥ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﻓﺎﻳﻞ ﻫﺎﯼ ﺷﻤﺎ ﺭﺍ ﺩﺍﺭﻧﺪ.

 

ﻓﻮﺍﻳﺪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﮐﻠﻮﺩ

- ﺷﻴﻮﻩ ﻣﻄﻤﺌﻦ ﻭ ﺍﻣﻨﯽ ﺑﺮﺍﯼ ﭘﺸﺘﻴﺒﺎﻥ ﮔﻴﺮﯼ ﺍﺯ ﺩﺍﺩﻩ ﻫﺎ ﺑﺮ ﺭﻭﯼ ﻓﻀﺎﯼ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﺩﻭﺭ ﺩﺳﺖ ﺍﺳﺖ.

- ﺍﺯ ﻫﺮ ﺟﺎﻳﯽ ﮐﻪ ﺍﻳﻨﺘﺮﻧﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻗﺎﺑﻞ ﺩﺳﺘﺮﺱ ﻭ ﺑﺎﺯﻳﺎﺑﯽ ﺍﺳﺖ.
