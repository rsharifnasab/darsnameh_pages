ﺩﺭﺱ ﺷﺸﻢ- ﻭﺭﺍﺛﺖ (Inheritance)

ﭘﻴﺶ‌ﮔﻔﺘﺎﺭ: ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﭘﻴﺸﻴﻦ ﺑﺎ ﻣﻔﺎﻫﻴﻢ ﺍﺑﺘﺪﺍﻳﯽ ﻭ ﭘﺎﻳﻪ‌ﺍﯼ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺷﺊ‌ﮔﺮﺍ ﺁﺷﻨﺎ ﺷﺪﻳﻢ. ﺍﺯ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺑﻌﺪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﻪ ﺑﻪ ﻣﻔﺎﻫﻴﻢ ﭘﻴﺸﺮﻓﺘﻪ ﺷﺊ‌ﮔﺮﺍﻳﯽ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﺍﻳﻦ ﺩﺭﺱ ﺭﺍ ﺑﻪ ﺩﺭﮎ ﻳﮑﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﻣﻔﺎﻫﻴﻢ ﺷﺊ‌ﮔﺮﺍﻳﯽ ﻳﻌﻨﯽ ﻭﺭﺍﺛﺖ ﺍﺧﺘﺼﺎﺹ ﻣﯽ‌ﺩﻫﻴﻢ.

 

 

ﻭﺭﺍﺛﺖ (Inheritance)

ﻣﻌﻨﯽ ﻭﺭﺍﺛﺖ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﺯ ﻧﺎﻣﺶ ﭘﻴﺪﺍﺳﺖ ﺑﻪ ﺍﺭﺙ ﺑﺮﺩﻥ ﺍﺳﺖ. ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻳﮏ ﮐﻮﺩﮎ ﻣﯽ‌ﺗﻮﺍﻧﺪ  ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﻭﺍﻟﺪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺍﺭﺙ ﺑﺒﺮﺩ، ﻳﮏ ﮐﻼﺱ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﮐﻼﺱ ﺑﺎﻻﺗﺮ ﺧﻮﺩ ﺭﺍ ﺍﺭﺙ‌ﺑﺮﯼ ﮐﻨﺪ. ﺩﺭ ﻭﺍﻗﻊ ﻳﮏ ﮐﻼﺱ ﺍﺯ ﮐﻼﺳﯽ ﺩﻳﮕﺮ ﻣﺸﺘﻖ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﺁﻥ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﺑﻪ ﺍﺭﺙ ﻣﯽ‌ﺑﺮﺩ. ﺍﻟﺒﺘﻪ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ ﻭﻳﮋﮔﯽ‌ﻫﺎ (ﻣﺘﻐﻴﺮ‌ﻫﺎ) ﻭ ﻋﻤﻞ‌ﻫﺎ (ﺗﻮﺍﺑﻊ).

 

ﻧﮑﺘﻪ: ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﭘﺲ ﺍﺯ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﻧﺎﻡ ﻭﻳﮋﮔﯽ ﻭ ﻋﻤﻞ ﻣﺘﻌﺎﻗﺒﺎ ﺍﺯ "ﻣﺘﻐﻴﺮ" ﻭ "ﺗﺎﺑﻊ"  ﻧﺎﻡ ﻣﯽ‌ﺑﺮﻳﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺁﺷﻨﺎﻳﯽ ﺷﻤﺎ ﻋﺰﻳﺰﺍﻥ ﺑﺎ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺭﻭﺍﻝ‌ﮔﻮﻧﻪ ﺍﻳﻦ‌ﻃﻮﺭ ﺩﺭ ﺫﻫﻦ ﺷﻤﺎ ﺟﺎ ﺑﻴﻔﺘﺪ ﮐﻪ ﻣﻌﺎﺩﻝ "ﻣﺘﻐﻴﺮ" ﺩﺭ ﺷﺊ‌ﮔﺮﺍﻳﯽ "ﻭﻳﮋﮔﯽ" ﻭ ﻣﻌﺎﺩﻝ "ﺗﺎﺑﻊ"  ﺩﺭ ﺷﺊ‌ﮔﺮﺍﻳﯽ "ﻋﻤﻞ" (ﻣﺘﺪ) ﺍﺳﺖ.

 

ﺍﻣﺎ ﮐﺎﺭﺑﺮﺩ ﻭﺭﺍﺛﺖ ﭼﻴﺴﺖ؟ ﻓﺮﺽ ﮐﻨﻴﺪ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺮﻧﺎﻣﻪ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﺗﻤﺎﻡ ﺍﻣﻮﺭ ﻣﺎﻟﯽ ﻳﮏ ﺑﻴﻤﺎﺭﺳﺘﺎﻥ - ﺍﺯ ﺣﻘﻮﻕ ﻭ ﺩﺭﻳﺎﻓﺘﯽ ﮔﺮﻓﺘﻪ ﺗﺎ ﻭﺍﻡ ﻭ...-  ﺭﺍ ﺷﺎﻣﻞ ﺷﻮﺩ. ﺧﺐ، ﺩﺭ ﻫﺮ ﺑﻴﻤﺎﺭﺳﺘﺎﻥ ﺍﻓﺮﺍﺩ ﺯﻳﺎﺩﯼ ﺑﺎ ﺳﻤﺖ‌ﻫﺎ ﻭ ﻣﺘﻌﺎﻗﺒﺎ ﺩﺭﻳﺎﻓﺖ‌ﻫﺎﯼ (ﺣﻘﻮﻕ) ﻣﺘﻔﺎﻭﺗﯽ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻨﺪ. ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺭﻭﺍﻝ‌ﮔﻮﻧﻪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺩﺭ ﺳﻤﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﺘﻐﻴﺮﻫﺎ ﻭ ﺗﻮﺍﺑﻊ ﻣﺘﻔﺎﻭﺕ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﻳﺎ ﺍﺯ ﺟﻤﻠﻪ‌ﻫﺎﯼ ﺷﺮﻃﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﻧﻤﻮﻧﻪ ﺗﺎﺑﻌﯽ ﮐﻪ ﺣﻘﻮﻕ ﻳﮏ "ﺩﮐﺘﺮ" ﺭﺍ ﺣﺴﺎﺏ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺗﺎﺑﻌﯽ ﮐﻪ ﺣﻘﻮﻕ ﻳﮏ "ﭘﺮﺳﺘﺎﺭ" ﺭﺍ، ﺷﺎﻳﺪ ﺍﺯ ﻧﻈﺮ ﻋﻤﻠﮑﺮﺩ ﻳﮑﺴﺎﻥ ﺑﺎﺷﻨﺪ (ﻣﺤﺎﺳﺒﻪ ﺣﻘﻮﻕ) ﺍﻣﺎ ﺍﺯ ﻧﻈﺮ ﺧﺮﻭﺟﯽ ﻣﺘﻔﺎﻭﺕ‌ﺍﻧﺪ. ﭼﺮﺍﮐﻪ ﺩﺭﻳﺎﻓﺘﯽ‌ﻫﺎﯼ ﺁﻥ‌ﻫﺎ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﭘﺲ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﻫﺮ ﮔﺮﻭﻩ ﮐﺎﺭﯼ «ﮐﺪ ﻧﻮﻳﺴﯽ» ﮐﻨﻴﺪ ﻭ ﺍﻳﻦ ﻣﯽ‌ﺷﻮﺩ ﺳﺎﻋﺖ‌ﻫﺎ ﺑﺮﻧﺎﻣﻪ‌ﻧﻮﻳﺴﯽ ﻭ ﺗﮑﺮﺍﺭ ﻣﺘﻐﻴﺮﻫﺎ ﻭ ﺗﻮﺍﺑﻌﯽ ﮐﻪ ﻋﻤﻠﮑﺮﺩ ﻳﮑﺴﺎﻥ ﺑﺎ ﺧﺮﻭﺟﯽ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺩﺍﺭﻧﺪ.

 

ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻭﺭﺍﺛﺖ ﺩﺭ ﺑﺮﻧﺎﻣﻪ‌ﻧﻮﻳﺴﯽ ﺷﺊ‌ﮔﺮﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺗﮑﺮﺍﺭ ﺟﻠﻮﮔﻴﺮﯼ ﮐﻨﻴﺪ. ﺍﻣﺎ ﭼﻄﻮﺭ؟ ﺧﺐ، ﺍﻳﻦ‌ﻃﻮﺭ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ ﮐﻪ ﺗﻤﺎﻡ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺩﺭ ﺑﻴﻤﺎﺭﺳﺘﺎﻥ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﻗﺮﺍﺭ ﺍﺳﺖ ﺣﻘﻮﻕ‌ﺷﺎﻥ ﻣﺤﺎﺳﺒﻪ ﺷﻮﺩ ﺩﺭ ﺩﺭﺟﻪ ﺍﻭﻝ «ﺍﻧﺴﺎﻥ» ﻫﺴﺘﻨﺪ ﻭ ﺩﺍﺭﺍﯼ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﻣﺸﺘﺮﮎ، ﺑﺮﺍﯼ ﻧﻤﻮﻧﻪ ﻫﻢ ﺩﺍﺭﺍﯼ ﻧﺎﻡ، ﻧﺎﻡ ﺧﺎﻧﻮﺍﺩﮔﯽ، ﺷﻤﺎﺭﻩ ﺷﻨﺎﺳﻨﺎﻣﻪ ﻭ ﻏﻴﺮﻩ ﭘﺲ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻳﮏ ﮐﻼﺱ ﺩﺭﺳﺖ ﮐﻨﻴﻢ ﻭ ﻫﻤﻪ ﺍﻳﻦ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﻣﺸﺘﺮﮎ ﺭﺍ ﺩﺍﺧﻞ ﺁﻥ ﻗﺮﺍﺭ ﺩﻫﻴﻢ. ﺳﭙﺲ ﺩﺭ ﺯﻣﺎﻥ ﺗﻌﺮﻳﻒ ﮐﺮﺩﻥ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﮔﺮﻭﻩ‌ﻫﺎﯼ ﮐﺎﺭﯼ ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﮐﻼﺱ ﺍﺭﺙ ﺑﺒﺮﻳﻢ ﻭ ﺩﻳﮕﺮ ﻻﺯﻡ ﻧﻴﺴﺖ ﺗﻤﺎﻡ ﺁﻥ ﻭﻳﮋﮔﯽ‌ﻫﺎ ﺭﺍ ﺗﮑﺮﺍﺭ ﮐﻨﻴﻢ.

 

ﺍﺯ ﺩﻳﮕﺮ ﻣﻮﺍﺭﺩ ﻣﻬﻤﯽ ﮐﻪ ﺍﺭﺙ‌ﺑﺮﯼ ﺑﺎﻋﺚ ﺳﺎﺩﮔﯽ ﻭ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﺗﮑﺮﺍﺭ ﮐﺪ ﻧﻮﻳﺴﯽ ﻣﯽ‌ﺷﻮﺩ، ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﭘﺎﻳﮕﺎﻩ ﺩﺍﺩﻩ ﺍﺳﺖ. ﺩﺭ ﺷﻴﻮﻩ ﺭﻭﺍﻝ‌ﮔﻮﻧﻪ ﻫﺮ ﺑﺎﺭ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﭘﺎﻳﮕﺎﻩ ﻭﺻﻞ ﺷﻮﻳﻢ ﺑﺎﻳﺪ ﺗﻤﺎﻡ ﮐﺪﻫﺎ ﺭﺍ ﺗﮑﺮﺍﺭ ﮐﻨﻴﻢ. ﺑﺮﺍﯼ ﻧﻤﻮﻧﻪ ﻧﺎﻡ ﺟﺪﻭﻝ ﻭ ﮐﻠﻴﺪ ﺍﺻﻠﯽ ﺭﺍ ﺑﺎﻳﺪ ﻫﺮ ﺑﺎﺭ ﺗﻨﻈﻴﻢ ﮐﻨﻴﻢ. ﺍﻣﺎ ﺍﺯ ﻃﺮﻳﻖ ﺷﻴﻮﻩ ﺷﺊ‌ﮔﺮﺍﯼ ﻣﺎ ﺑﺎ ﺳﺎﺧﺘﻦ ﮐﻼﺱ ﭘﺎﻳﮕﺎﻩ ﻭ ﺍﺭﺙ ﺑﺮﯼ ﺍﺯ ﺁﻥ ﺑﺎﻋﺚ ﮐﺎﻫﺶ ﺩﺭ ﮐﺪ ﻧﻮﻳﺴﯽ ﻣﯽ‌ﺷﻮﻳﻢ. ﺍﺟﺎﺯﻩ ﺑﺪﻫﻴﺪ ﮐﻪ ﺑﻪ ﻫﻤﻴﻦ ﻣﻘﺪﺍﺭ ﺷﺮﺡ ﺍﮐﺘﻔﺎ ﮐﻨﻴﻢ ﻭ ﺩﺭ ﺍﺩﺍﻣﻪ ﺑﺎ ﺍﻧﺠﺎﻡ ﻧﻤﻮﻧﻪ ﮐﺪﻧﻮﻳﺴﯽ ﺍﺭﺙ‌ﺑﺮﯼ ﺭﺍ ﺑﻴﺸﺘﺮ ﺗﻮﺿﻴﺢ ﺩﻫﻴﻢ.

 

ﺑﻪ ﮐﻼﺳﯽ ﮐﻪ ﺍﺯ ﺁﻥ ﺍﺭﺙ ﻣﯽ‌ﺑﺮﻳﻢ ﮐﻼﺱ ﻭﺍﻟﺪ parent class  ﻭ ﺑﻪ ﮐﻼﺱ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﮐﻼﺱ ﻭﺍﻟﺪ ﺍﺭﺙ ﻣﯽ‌ﺑﺮﻧﺪ ﮐﻼﺱ ﻓﺮﺯﻧﺪ  child class  ﻣﯽ‌ﮔﻮﻳﻴﻢ. ﻣﻮﺭﺩ ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﮐﻼﺱ ﻓﺮﺯﻧﺪ ﺩﺭ ﺩﻧﻴﺎﯼ ﺑﺮﻧﺎﻣﻪ‌ﻧﻮﻳﺴﯽ ﺷﺊ‌ﮔﺮﺍ ﻧﻪ ﺗﻨﻬﺎ ﺗﻤﺎﻡ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎ (ﻭﻳﮋﮔﯽ‌ﻫﺎ ﻭ ﻋﻤﻞ‌ﻫﺎﯼ) ﮐﻼﺱ ﻭﺍﻟﺪ ﺭﺍ ﺑﻪ ﺍﺭﺙ ﻣﯽ‌ﺑﺮﺩ، ﺑﻠﮑﻪ ﺧﻮﺩ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﻳﯽ ﺩﺭ ﺧﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﺁﻥ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎ ﺩﺭ ﮐﻼﺱ ﻭﺍﻟﺪ ﻧﻴﺴﺘﻨﺪ.

 

ﻣﺜﺎﻝ ﮐﺎﺭﻣﻨﺪﺍﻥ ﺑﻴﻤﺎﺭﺳﺘﺎﻥ ﻳﺎﺩﺗﺎﻥ ﻫﺴﺖ؟ ﺍﺯ ﺍﻳﻦ ﻧﻈﺮ ﮐﻼﺱ ﻭﺍﻟﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﻧﺎﻡ «ﮐﺎﺭﻣﻨﺪ» ﺩﺍﺭﺍﯼ ﺗﻤﺎﻡ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﻣﺸﺘﺮﮎ ﺑﻴﻦ ﮐﺎﺭﻣﻨﺪﺍﻥ ﺑﺎﺷﺪ ﻭ ﮐﻼﺱ‌ﻫﺎﯼ ﻓﺮﺯﻧﺪ ﮐﻪ ﺑﺮﺍﯼ ﮐﺎﺭﻣﻨﺪﺍﻥ ﺩﮐﺘﺮ ﻭ ﭘﺮﺳﺘﺎﺭ ﻣﯽ‌ﻧﻮﻳﺴﻴﻢ (ﮐﻪ ﺍﺯ ﮐﻼﺱ «ﮐﺎﺭﻣﻨﺪ» ﺍﺭﺙ ﺑﺮﯼ ﻣﯽ‌ﮐﻨﻨﺪ) ﺩﺍﺭﺍﯼ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﮐﻪ ﻣﺨﺼﻮﺹ ﮐﺎﺭﻣﻨﺪﺍﻥ ﺩﮐﺘﺮ ﻭ ﭘﺮﺳﺘﺎﺭ ﺍﺳﺖ ﺑﺎﺷﻨﺪ.

 

ﻧﮑﺘﻪ: ﻫﺮ ﮐﻼﺱ ﻓﺮﺯﻧﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﻤﺎﻡ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﻳﺎ ﺍﻋﻀﺎﯼ (ﻭﻳﮋﮔﯽ ﻭ ﻋﻤﻞ) ﮐﻼﺱ ﻭﺍﻟﺪ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
 

 

ﻧﮑﺘﻪ: ﻫﺮ ﮐﻼﺱ ﻓﺮﺯﻧﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻏﻴﺮ ﺍﺯ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﮐﻼﺱ ﻭﺍﻟﺪ، ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﺧﻮﺩﺵ ﺭﺍ ﻫﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
