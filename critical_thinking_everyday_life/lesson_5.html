ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﺍﻋﺘﺒﺎﺭ ﻭ ﺻﺤﺖ ﺍﺳﺘﺪﻻﻝ

ﺍﺳﺘﺪﻻﻝ ﺍﺛﺒﺎﺕ ﻳﮏ ﺍﺩﻋﺎﺳﺖ ﻭ ﻫﺮ ﺍﺩﻋﺎﻳﯽ ﺑﺮﺍﯼ ﻣﻮﺭﺩ ﻗﺒﻮﻝ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﻧﻴﺎﺯﻣﻨﺪ ﺍﺭﺯﻳﺎﺑﯽ ﺍﺳﺖ. ﺍﺯ ﮐﺠﺎ ﺑﺪﺍﻧﻴﻢ ﺍﺳﺘﺪﻻﻟﯽ ﮐﻪ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﻳﺎ ﺩﻳﮕﺮﺍﻥ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻩ‌ﺍﻧﺪ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﺍﺳﺖ؟ ﻣﻌﻴﺎﺭﻫﺎﯼ ﺍﺭﺯﻳﺎﺑﯽ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﭼﻪ ﻫﺴﺘﻨﺪ؟

ﺍﺳﺘﺪﻻﻝ ﻣﺘﺸﮑﻞ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﻧﺘﻴﺠﻪ ﺍﺳﺖ ﻭ ﻧﺘﺎﻳﺞ ﺑﺎﻳﺪ ﺑﻪ ﻧﺤﻮﯼ ﺑﻪ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻣﺮﺗﺒﻂ ﺑﺎﺷﻨﺪ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺧﻮﺏ ﭘﻴﺶ ﺑﺮﻭﺩ. ﺍﻣﺎ ﻫﻤﻴﺸﻪ ﺍﻳﻦ ﻃﻮﺭ ﻧﻴﺴﺖ ﻭ ﺩﻭ ﺭﺍﻩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻣﺸﮑﻞ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺩﻭ ﻣﺸﮑﻞ ﺍﺻﻠﯽ ﺩﺭ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ:

۱- ﻣﺸﮑﻠﯽ ﺩﺭ ﻳﮏ ﻳﺎ ﭼﻨﺪ ﺗﺎ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻳﻌﻨﯽ ﻣﻘﺪﻣﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ.

۲- ﺩﺭ ﺭﺍﺑﻄﻪ‌ﯼ ﺑﻴﻦ ﻣﻘﺪﻣﻪ ﻭ ﻧﺘﻴﺠﻪ ﻣﺸﮑﻠﯽ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻳﻌﻨﯽ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﻟﻴﻞ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﻧﺘﻴﺠﻪ ﻓﺮﺍﻫﻢ ﻧﮑﻨﻨﺪ.

ﺑﺎﻳﺪ ﺍﺯ ﻫﺮ ﺩﻭﯼ ﺍﻳﻦ ﻣﺸﮑﻼﺕ ﺍﺟﺘﻨﺎﺏ ﮐﻨﻴﻢ ﻭ ﺑﺎ ﭘﺮﻫﻴﺰ ﺍﺯ ﺁﻧﻬﺎ ﺑﻪ ﻳﮏ ﺍﺭﺯﺵ ﻣﺘﻨﺎﻇﺮ ﺑﺎ ﻫﺮ ﻣﺸﮑﻞ ﺩﺳﺖ ﺧﻮﺍﻫﻴﻢ ﻳﺎﻓﺖ، ﻳﻌﻨﯽ ﺍﻋﺘﺒﺎﺭ ﻭ ﺻﺤﺖ.

ﺍﮔﺮ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻣﺸﮑﻠﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﻭ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ، ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﻣﻌﺘﺒﺮ ﺍﺳﺖ ﻭ ﺍﮔﺮ ﺭﺍﺑﻄﻪ‌ﯼ ﺑﻴﻦ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﻧﺘﻴﺠﻪ ﺑﺪﻭﻥ ﺍﻳﺮﺍﺩ ﺑﺎﺷﺪ ﻭ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺑﻪ ﺧﻮﺑﯽ ﺍﺯ ﻧﺘﻴﺠﻪ ﺣﻤﺎﻳﺖ ﮐﻨﻨﺪ ﺁﻧﮕﺎﻩ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﺻﺤﻴﺢ ﻧﻴﺰ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﻫﺮ ﺩﻭﯼ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﺑﭙﺮﺩﺍﺯﻳﻢ ﻭ ﺑﺒﻴﻨﻴﻢ ﺍﻳﻨﮑﻪ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﻭ ﺻﺤﻴﺢ ﺍﺳﺖ ﭼﻪ ﻣﻌﻨﺎﻳﯽ ﺩﺍﺭﺩ.

 

ﺍﻋﺘﺒﺎﺭ ﺍﺳﺘﺪﻻﻝ
ﺑﮕﺬﺍﺭﻳﺪ ﺍﺯ ﺭﺍﺑﻄﻪ ﺑﻴﻦ ﻣﻘﺪﻣﻪ ﻭ ﻧﺘﻴﺠﻪ ﺷﺮﻭﻉ ﮐﻨﻴﻢ. ﺳﻮﺍﻝ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﭼﻪ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﺑﻴﻦ ﻣﻘﺪﻣﻪ ﻭ ﻧﺘﻴﺠﻪ ﺭﺍﺑﻄﻪ ﺧﻮﺑﯽ ﺍﺳﺖ ﻳﺎ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﺑﻪ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺧﻮﺏ ﺑﺪﻝ ﻣﯽ‌ﮐﻨﺪ؟ ﺍﻟﺒﺘﻪ ﺟﻮﺍﺏ ﺍﻳﻦ ﺳﻮﺍﻝ ﺑﻪ ﻧﻮﻉ ﺍﺳﺘﺪﻻﻝ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﺑﻌﺪﯼ ﺑﻪ ﺍﻧﻮﺍﻉ ﺁﻥ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ. ﺍﻣﺎ ﺑﮕﺬﺍﺭﻳﺪ ﻓﻌﻼ ﺭﻭﯼ ﻧﻮﻉ ﺳﺎﺩﻩ ﻭ ﺳﺮ ﺭﺍﺳﺖ ﺍﺳﺘﺪﻻﻝ ﻳﻌﻨﯽ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺗﻤﺮﮐﺰ ﮐﻨﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ ﻧﺘﻴﺠﻪ ﺑﺎﻳﺪ ﭘﻴﺮﻭ ﻭ ﺑﻪ ﺍﺻﻄﻼﺡ ﻣﻨﻄﻖ ﻗﺪﻳﻢ ﺗﺎﻟﯽ ﻣﻘﺪﻡ ﺑﺎﺷﺪ.

ﺧﺐ ﺍﻳﻦ ﭼﻪ ﻣﻌﻨﺎﻳﯽ ﺩﺍﺭﺩ؟ ﻳﻌﻨﯽ ﭼﻪ ﮐﻪ ﻧﺘﻴﺠﻪ ﺑﺎﻳﺪ ﺩﻧﺒﺎﻟﻪ‌ﺭﻭ ﻣﻘﺪﻣﻪ ﺑﺎﺷﺪ؟ ﮐﺎﺭﯼ ﮐﻪ ﻣﻌﻤﻮﻻ ﻣﻨﻄﻖ‌ﺩﺍﻥ‌ﻫﺎ ﻣﯽ‌ﮐﻨﻨﺪ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺭﻭﯼ ﻣﻔﻬﻮﻡ ﺍﻋﺘﺒﺎﺭ ﺗﻤﺮﮐﺰ ﮐﻨﻨﺪ. ﻳﻌﻨﯽ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺑﺎﻳﺪ ﺑﻪ ﻧﺤﻮﯼ ﺳﺎﺧﺘﺎﺭﺑﻨﺪﯼ ﺷﺪﻩ ﺑﺎﺷﺪ ﮐﻪ ﻣﻌﺘﺒﺮ ﺑﺎﺷﺪ. ﺩﺭﺑﺎﺭﻩ ﻣﻔﻬﻮﻡ ﺍﻋﺘﺒﺎﺭ ﺩﺭ ﻫﻤﻴﻦ ﺩﺭﺱ ﺑﺤﺚ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ. ﺍﻣﺎ ﻳﺎﺩﻣﺎﻥ ﺑﺎﺷﺪ ﮐﻪ ﺗﻨﻬﺎ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻗﻴﺎﺳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ «ﺑﺎﻳﺪ» ﻣﻌﺘﺒﺮ ﺑﺎﺷﻨﺪ. ﻧﻮﻉ ﺩﻳﮕﺮﯼ ﺍﺯ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﻧﺎﻡ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻗﺮﺍﺭ ﻧﻴﺴﺖ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮﯼ ﺑﺎﺷﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺑﺎ ﺗﻌﺮﻳﻔﯽ ﮐﻪ ﺍﻳﻨﺠﺎ ﺍﺯ ﺍﻋﺘﺒﺎﺭ ﺩﺍﺭﻳﻢ. ﻳﻌﻨﯽ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺣﺘﯽ ﻭﺍﻧﻤﻮﺩ ﻧﻤﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﻣﻌﺘﺒﺮ ﻫﺴﺘﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﭘﻴﺮﻭﯼ ﻧﻤﯽ‌ﮐﻨﺪ.

ﺑﺤﺚ ﺍﻧﻮﺍﻉ ﺍﺳﺘﺪﻻﻝ ﻭ ﺗﻔﺎﻭﺕ‌ﻫﺎﯼ ﺁﻥ ﺭﺍ ﺑﺮﺍﯼ ﺩﺭﺱ ﺑﻌﺪﯼ ﺑﮕﺬﺍﺭﻳﻢ ﻭ ﺍﻳﻦ ﺟﺎ ﺑﺮﺍﯼ ﺳﺎﺩﮔﯽ ﻓﻘﻂ ﺑﺮ ﺭﻭﯼ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻗﻴﺎﺳﯽ ﺗﻤﺮﮐﺰ ﮐﻨﻴﻢ.

ﺍﻳﺪﻩ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺑﺎﻳﺪ ﺑﻪ ﻧﻮﻋﯽ ﺳﺎﺧﺘﺎﺭ ﺑﻨﺪﯼ ﺷﺪﻩ ﺑﺎﺷﺪ ﮐﻪ ﻣﻌﺘﺒﺮ ﺑﺎﺷﺪ. ﺑﻴﺎﻳﻴﺪ ﺑﺎ ﻳﮏ ﻣﺜﺎﻝ ﺳﺎﺩﻩ ﺷﺮﻭﻉ ﮐﻨﻴﻢ: ﻓﺮﺽ ﮐﻨﻴﺪ ﮐﻪ ﺷﻤﺎ ﻣﺮﻳﻢ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﻴﺪ،‌ ﺍﻣﺎ ﺑﭽﻪ‌ﻫﺎﻳﺶ ﺭﺍ ﻧﻤﯽ‌ﺷﻨﺎﺳﻴﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﮐﻪ ﺍﻭ ﻓﺮﺯﻧﺪﯼ ﺩﺍﺭﺩ ﮐﻪ ﺣﺎﻣﻠﻪ ﺍﺳﺖ. ﻭ ﻫﻤﻴﻦ‌ﻃﻮﺭ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﮐﻪ ﺗﻨﻬﺎ ﺯﻧﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺣﺎﻣﻠﻪ ﺷﻮﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﻪ ﻃﻮﺭ ﺍﺟﺘﻨﺎﺏ‌ﻧﺎﭘﺬﻳﺮﯼ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﺭﺍ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﮐﻪ: ﻣﺮﻳﻢ ﺣﺪﺍﻗﻞ ﻳﮏ ﺩﺧﺘﺮ ﺩﺍﺭﺩ. ﻭﻗﺘﯽ ﺑﻪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﻓﮑﺮ ﮐﻨﻴﺪ، ﺑﻪ ﻫﻴﭻ ﻋﻨﻮﺍﻥ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﮐﻪ ﻫﺮ ﺩﻭﯼ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ، ﺍﻣﺎ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺍﻳﻦ ﻭﻳﮋﮔﯽ ﺭﺍ ﺍﻋﺘﺒﺎﺭ ﻣﯽ‌ﻧﺎﻣﻴﻢ.

ﺧﺐ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﻋﺘﺒﺎﺭ ﺭﺍ ﻫﻢ ﺩﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺗﻌﺮﻳﻒ ﮐﻨﻴﻢ:

ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ ﺍﮔﺮ ﻭ ﺗﻨﻬﺎ ﺍﮔﺮ، ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻧﺒﺎﺷﺪ ﮐﻪ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﻳﻌﻨﯽ ﻭﺿﻌﻴﺘﯽ ﻣﻤﮑﻦ ﻧﺒﺎﺷﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﻣﻌﻴﺎﺭ ﺑﺮﺍﯼ ﺗﻌﺮﻳﻒ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺳﺎﺩﻩ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ، ﺍﻣﺎ ﮐﻠﻤﻪ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺸﮑﻞ‌ﺳﺎﺯ ﺑﺎﺷﺪ. ﺍﺯ ﮐﺠﺎ ﻣﯽ‌ﻓﻬﻤﻴﻢ ﮐﻪ ﻭﺿﻌﻴﺘﯽ ﮐﻪ ﺍﺯ ﺁﻥ ﺣﺮﻑ ﺯﺩﻳﻢ ﻣﻤﮑﻦ ﺍﺳﺖ ﻳﺎ ﻏﻴﺮﻣﻤﮑﻦ؟ ﺧﺐ ﭘﺎﺳﺦ ﻣﻌﻤﻮﻝ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻫﻴﭻ ﺭﺍﻩ ﺣﻞ ﻣﮑﺎﻧﻴﮑﯽ ﻭ ﻓﺮﻣﻮﻝ‌ﺑﻨﺪﯼ ﺷﺪﻩ ﺑﺮﺍﯼ ﺁﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﻭ ﺩﺭ ﻣﻨﻄﻖ ﻧﻴﺰ ﻣﻨﻄﻖ‌ﺩﺍﻧﺎﻥ ﺑﺎ ﺁﻥ ﺩﺳﺖ ﺑﻪ ﮔﺮﻳﺒﺎﻧﻨﺪ.

ﻓﻌﻼ ﺑﻴﺎﻳﻴﺪ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﺸﮑﻞ ﺍﺛﺒﺎﺕ ﺍﻋﺘﺒﺎﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﺣﻞ ﮐﻨﻴﻢ: ﺍﮔﺮ ﺷﻤﺎ ﺑﺘﻮﺍﻧﻴﺪ ﻳﮏ ﺩﺍﺳﺘﺎﻥ ﺳﺎﺯﮔﺎﺭ ﻭ ﺑﺪﻭﻥ ﺗﻨﺎﻗﺾ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ، ﺁﻥ‌ﮔﺎﻩ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﻧﺎﻣﻌﺘﺒﺮ ﺍﺳﺖ. ﺍﻣﺎ ﺍﮔﺮ ﻫﻴﭻ ﺭﺍﻫﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﻧﺘﻮﺍﻧﻴﺪ ﻳﮏ ﺩﺍﺳﺘﺎﻥ ﺳﺎﺯﮔﺎﺭ ﺑﺴﺎﺯﻳﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ، ﺁﻧﮕﺎﻩ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ.

ﺣﺎﻝ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﮐﻪ ﺑﺎ ﺍﻳﻦ ﻣﻌﻴﺎﺭﻫﺎ ﻣﺜﺎﻝ ﺧﻮﺩ ﺭﺍ ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﻴﻢ.

۱- ﻣﺮﻳﻢ ﻓﺮﺯﻧﺪﯼ ﺩﺍﺭﺩ ﮐﻪ ﺣﺎﻣﻠﻪ ﺍﺳﺖ.

۲- ﺗﻨﻬﺎ ﺯﻧﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺣﺎﻣﻠﻪ ﺷﻮﻧﺪ.

‍۳- ﺑﻨﺎﺑﺮﺍﻳﻦ، ﻣﺮﻳﻢ ﺣﺪﺍﻗﻞ ﻳﮏ ﻓﺮﺯﻧﺪ ﺩﺧﺘﺮ ﺩﺍﺭﺩ.

ﺧﺐ ﺁﻳﺎ ﻫﻴﭻ ﺭﺍﻫﯽ ﺑﺮﺍﯼ ﮔﻔﺘﻦ ﻳﮏ ﺩﺍﺳﺘﺎﻥ ﺳﺎﺯﮔﺎﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ؟ ﻳﻌﻨﯽ ﻣﺮﻳﻢ ﻓﺮﺯﻧﺪﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﺣﺎﻣﻠﻪ‌ ﺍﺳﺖ ﻭ ﺗﻨﻬﺎ ﺯﻧﺎﻥ ﺑﺘﻮﺍﻧﻨﺪ ﺣﺎﻣﻠﻪ ﺷﻮﻧﺪ‌ ﺍﻣﺎ ﻣﺮﻳﻢ ﺩﺧﺘﺮ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ؟

ﺑﮕﺬﺍﺭﻳﺪ ﮐﻤﯽ ﺑﺮﺍﯼ ﺳﺎﺧﺘﻦ ﭼﻨﻴﻦ ﺩﺍﺳﺘﺎﻧﯽ ﺗﻼﺵ ﮐﻨﻴﻢ. ﻓﺮﺽ ﮐﻨﻴﺪ ﻣﺮﻳﻢ ﺗﻨﻬﺎ ﻳﮏ ﻓﺮﺯﻧﺪ ﺩﺍﺭﺩ ﮐﻪ ﭘﺴﺮ ﺍﺳﺖ. ﺧﺐ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺗﺎ ﺍﻳﻨﺠﺎ ﻧﺘﻴﺠﻪ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ ﮐﻪ ﺧﺐ ﺗﺎ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺭﺍ ﺩﻭﺑﺎﺭﻩ ﺑﺎ ﺍﻳﻦ ﻓﺮﺽ ﺟﺪﻳﺪ ﺑﺮﺭﺳﯽ ﻧﮑﻨﻴﻢ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﭼﻴﺰﯼ ﺩﺭﺑﺎﺭﻩ ﺍﻋﺘﺒﺎﺭ ﺍﺳﺘﺪﻻﻝ ﺑﮕﻮﻳﻴﻢ. ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﻣﺮﻳﻢ ﻓﺮﺯﻧﺪﯼ ﺩﺍﺭﺩ ﮐﻪ ﺣﺎﻣﻠﻪ ﺍﺳﺖ ‌ﻭ ﻣﻘﺪﻣﻪ ﺩﻭﻡ ﮐﻪ ﻓﻘﻂ ﺯﻧﺎﻥ ﺣﺎﻣﻠﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻫﻢ ﺩﺭﺳﺖ ﺍﺳﺖ، ﺑﻨﺎﺑﺮﺍﻳﻦ ﭘﺴﺮ ﻣﺮﻳﻢ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺣﺎﻣﻠﻪ ﺑﺎﺷﺪ. ﭘﺲ ﻣﻘﺪﻣﻪ ﺍﻭﻝ ﺑﺎﻳﺪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ ﮐﻪ ﻣﺮﻳﻢ ﻳﮏ ﻓﺮﺯﻧﺪ ﺣﺎﻣﻠﻪ ﺩﺍﺭﺩ. ﺧﺐ ﺗﺎ ﺍﻳﻨﺠﺎﯼ ﮐﺎﺭ ﻧﺘﻮﺍﻧﺴﺘﻴﻢ ﺩﺍﺳﺘﺎﻧﯽ ﺩﺭﺳﺖ ﮐﻨﻴﻢ ﮐﻪ ﻫﺮ ﺩﻭﯼ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ ﻭ ﺩﻳﺪﻳﻢ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﺍﺟﺘﻨﺎﺏ‌ﻧﺎﭘﺬﻳﺮﯼ ﺑﺎﻳﺪ ﻗﺒﻮﻝ ﮐﻨﻴﻢ ﮐﻪ ﻳﮑﯽ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻫﻢ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﻪ ﺍﻋﺘﺒﺎﺭ ﺍﺳﺘﺪﻻﻝ ﻟﻄﻤﻪ‌ﺍﯼ ﻭﺍﺭﺩ ﻧﺸﺪﻩ ﺍﺳﺖ.

ﻳﮏ ﺑﺎﺭ ﺩﻳﮕﺮ ﺗﻼﺵ ﮐﻨﻴﻢ ﻭ ﺍﻳﻦ ﺑﺎﺭ ﺑﻪ ﻣﻘﺪﻣﻪ ﺩﻭﻡ ﻳﻌﻨﯽ ﺍﻳﻨﮑﻪ ﺗﻨﻬﺎ ﺯﻧﺎﻥ ﺣﺎﻣﻠﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﺣﻤﻠﻪ ﮐﻨﻴﻢ. ﻓﺮﺽ ﮐﻨﻴﺪ ﺑﻪ ﻫﺮ ﺩﻟﻴﻠﯽ ﮐﻪ ﻧﻤﯽ‌ﺩﺍﻧﻴﻢ ﭼﻴﺴﺖ، ﻣﺮﻳﻢ ﭘﺴﺮﯼ ﺩﺍﺭﺩ ﮐﻪ ﺍﺗﻔﺎﻗﺎ ﺣﺎﻣﻠﻪ ﺍﺳﺖ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﻘﺪﻣﻪ ﺩﻭﻡ ﮐﻪ ﻣﯽ‌ﮔﻮﻳﺪ ﻓﻘﻂ ﺯﻧﺎﻥ ﺣﺎﻣﻠﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻧﺎﺩﺭﺳﺖ ﺍﺯ ﺁﺏ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ ﻭ ﻫﻤﭽﻨﺎﻥ ﻧﺘﻮﺍﻧﺴﺘﻪ‌ﺍﻳﻢ ﺳﻨﺎﺭﻳﻮﻳﯽ ﺳﺮﻫﻢ ﮐﻨﻴﻢ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﻫﻤﭽﻨﺎﻥ ﺍﻋﺘﺒﺎﺭ ﺍﺳﺘﺪﻻﻝ ﺑﺪﻭﻥ ﺧﺪﺷﻪ ﺑﺎﻗﯽ ﻣﺎﻧﺪﻩ ﺍﺳﺖ.

ﺑﮕﺬﺍﺭﻳﺪ ﮐﻤﯽ ﺳﻤﺞ‌ﺗﺮ ﺑﺎﺷﻴﻢ ﻭ ﺍﻣﮑﺎﻧﺎﺕ ﺩﻳﮕﺮ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ. ﻣﺜﻼ ﻓﺮﺽ ﮐﻨﻴﻢ ﻣﺮﻳﻢ ﺩﻭ ﻓﺮﺯﻧﺪ ﺩﺍﺭﺩ ﮐﻪ ﻳﮑﯽ ﺩﺧﺘﺮ ﻭ ﺩﻳﮕﺮﯼ ﭘﺴﺮ ﺍﺳﺖ. ﺧﺐ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﺯﻳﺮﺍ ﺩﺧﺘﺮ ﻣﺮﻳﻢ ﺣﺎﻣﻠﻪ ﺍﺳﺖ ﻭ ﻧﺘﻴﺠﻪ ﻫﻢ ﺩﺭﺳﺖ ﺍﺳﺖ ﺯﻳﺮﺍ ﻣﯽ‌ﮔﻮﻳﺪ ﺍﻭ ﺣﺪﺍﻗﻞ ﻳﮏ ﺩﺧﺘﺮ ﺩﺍﺭﺩ ﻭ ﺍﻳﻨﮑﻪ ﻳﮏ ﺩﺧﺘﺮ ﻭ ﻳﮏ ﭘﺴﺮ ﺩﺍﺭﺩ ﻧﺘﻴﺠﻪ ﺭﺍ ﻧﺎﺩﺭﺳﺖ ﻧﻤﯽ‌ﮐﻨﺪ. ﺍﻣﺎ ﭼﻄﻮﺭ ﺍﺳﺖ ﺑﻪ ﺍﻳﻦ ﺳﻨﺎﺭﻳﻮ ﻓﮑﺮ ﮐﻨﻴﻢ ﮐﻪ ﻣﺮﻳﻢ ﻳﮏ ﻓﺮﺯﻧﺪ ﺩﺍﺭﺩ ﮐﻪ ﺍﺯ ﻟﺤﺎﻅ ﺯﻳﺴﺖ‌ﺷﻨﺎﺳﯽ ﺩﺧﺘﺮ ﺍﺳﺖ ﺍﻣﺎ ﺧﻮﺩ ﺭﺍ ﭘﺴﺮ ﺟﺎ ﻣﯽ‌ﺯﻧﺪ ﻳﺎ ﭘﺴﺮ ﺩﻳﺪﻩ ﻣﯽ‌ﺷﻮﺩ. ﻳﻌﻨﯽ ﻣﺮﻳﻢ ﻭ ﺩﻳﮕﺮﺍﻥ ﺍﻭ ﺭﺍ ﻫﻤﺎﻧﻨﺪ ﭘﺴﺮﺍﻥ ﻣﯽ‌ﺑﻴﻨﻨﺪ ﺍﻣﺎ ﺁﻥ ﻓﺮﺯﻧﺪ ﺣﺎﻣﻠﻪ ﺍﺳﺖ ﭼﻮﻥ ﺍﺯ ﻟﺤﺎﻅ ﺯﻳﺴﺖ‌ﺷﻨﺎﺳﯽ ﻣﻮﻧﺚ ﺍﺳﺖ. ﺧﺐ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ.

ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ ﮐﻢ ﮐﻢ ﺩﺍﺭﻳﻢ ﺑﻪ ﺟﺎﯼ ﺧﻮﺑﯽ ﻣﯽ‌ﺭﺳﻴﻢ. ﺍﻣﺎ ﺁﻳﺎ ﺍﻳﻦ ﺩﺍﺳﺘﺎﻥ ﻣﻌﻘﻮﻝ ﻭ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻭ ﺳﺎﺯﮔﺎﺭ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ؟ ﺻﺒﺮ ﮐﻨﻴﺪ. ﺍﮔﺮ ﮐﻤﯽ ﺩﻗﻴﻖ ﺑﺎﺷﻴﻢ، ﻓﺮﺯﻧﺪ ﻣﺮﻳﻢ ﻳﺎ ﺩﺧﺘﺮ ﺍﺳﺖ ﻳﺎ ﭘﺴﺮ. ﺍﮔﺮ ﺩﺧﺘﺮ ﺑﺎﺷﺪ ﻭ ﺣﺎﻣﻠﻪ ﻫﻢ ﺑﺎﺷﺪ ﻣﺸﮑﻠﯽ ﻧﻴﺴﺖ ﻭ ﻧﺘﻴﺠﻪ ﻫﻢ ﺩﺭﺳﺖ ﺍﺳﺖ ﻭ ﺍﺳﺘﺪﻻﻝ ﺍﻋﺘﺒﺎﺭ ﺩﺍﺭﺩ. ﺍﻣﺎ ﺍﮔﺮ ﭘﺴﺮ ﺑﺎﺷﺪ ﭼﻮﻥ ﺧﻮﺩﺵ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﭘﺴﺮ ﻣﯽ‌ﺑﻴﻨﺪ، ﺍﻳﻦ ﻣﻘﺪﻣﻪ ﮐﻪ ﻣﺮﻳﻢ ﻳﮏ ﻓﺮﺯﻧﺪ ﺣﺎﻣﻠﻪ ﺩﺍﺭﺩ ﺩﺭﺳﺖ ﺍﺳﺖ. ﺍﻣﺎ ﻣﻘﺪﻣﻪ ﺩﻭﻡ ﮐﻪ ﻓﻘﻂ ﺩﺧﺘﺮﺍﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺣﺎﻣﻠﻪ ﺑﺎﺷﻨﺪ ﭼﻄﻮﺭ؟ ﺍﻳﻦ ﻣﻘﺪﻣﻪ ﻧﺎﺩﺭﺳﺖ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻭ ﺑﺎﺯ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ.

ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻫﺮﮐﺎﺭﯼ ﮐﻪ ﮐﻨﻴﺪ ﺍﻣﮑﺎﻥ ﻧﺪﺍﺭﺩ ﮐﻪ ﻫﺮ ﺩﻭﯼ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺍﻳﻦ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ. ﺷﺎﻳﺪ ﻣﺎ ﻧﺘﻮﺍﻧﻴﻢ ﺩﺍﺳﺘﺎﻧﯽ ﭘﻴﺪﺍ ﮐﻨﻴﻢ ﮐﻪ ﻧﺸﺎﻥ ﺩﻫﺪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﻧﺎﻣﻌﺘﺒﺮ ﺍﺳﺖ ﺍﻣﺎ ﺍﻳﻦ ﻭﺍﻗﻌﻴﺖ ﮐﻪ ﻣﺎ ﺗﻼﺵ ﮐﺮﺩﻳﻢ ﻭ ﻧﺘﻮﺍﻧﺴﺘﻴﻢ ﻫﻴﭻ ﺍﻣﮑﺎﻧﯽ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﻢ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ.

ﺭﻭﺵ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﺳﻨﺠﺶ ﺍﻋﺘﺒﺎﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻓﺮﺽ ﮐﻨﻴﻢ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ ﻭ ﺁﻧﮕﺎﻩ ﺑﺎﻳﺪ ﺣﺪﺍﻗﻞ ﻳﮑﯽ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻫﻢ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺍﻣﮑﺎﻥ ﻧﺪﺍﺭﺩ ﻫﺮ ﺩﻭ ﻣﻘﺪﻣﻪ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﺭﻭﺵ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﻣﺜﺎﻟﯽ ﮐﻪ ﺯﺩﻳﻢ ﺑﻪ ﮐﺎﺭ ﺑﺒﺮﻳﺪ ﻭ ﺑﺎﺯ ﺑﻪ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﺑﺮﺳﻴﺪ ﮐﻪ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ.

ﻧﮑﺘﻪ ﺩﻳﮕﺮ ﺩﺭﺑﺎﺭﻩ ﺍﻋﺘﺒﺎﺭ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﻋﺘﺒﺎﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻭﺍﺑﺴﺘﻪ ﺑﻪ ﺁﻥ ﻧﻴﺴﺖ ﮐﻪ ﺁﻳﺎ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﻧﺘﻴﺠﻪ ﻭﺍﻗﻌﺎ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﻳﺎ ﻧﻪ. ﺑﻠﮑﻪ ﻭﺍﺑﺴﺘﻪ ﺑﻪ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﭼﻪ ﭼﻴﺰﯼ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﺍﺳﺖ. ﺍﻳﻨﮑﻪ ﺁﻳﺎ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭﺍﻗﻌﺎ ﺩﺭ ﻳﮏ ﺟﻬﺎﻥ ﻭﺍﻗﻌﯽ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ، ﻣﺴﺌﻠﻪ ﻧﻴﺴﺖ. ﺍﻋﺘﺒﺎﺭ ﻣﺒﺘﻨﯽ ﺑﺮ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﺍﺳﺖ.

ﺍﮔﺮ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ، ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺩﺭﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺁﻧﮕﺎﻩ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻭﺍﻗﻌﺎ ﺧﻮﺏ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﺧﺎﺻﻴﺘﯽ ﺑﻪ ﻧﺎﻡ «ﺻﺤﺖ» ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺻﺤﺖ ﺑﻪ ﻣﻌﻨﺎﯼ ﺳﺎﻟﻢ ﺑﻮﺩﻥ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺖ ﻭ ﺑﺎ ﺍﺻﻄﻼﺡ ﺩﺭﺳﺘﯽ ﻳﺎ ﻧﺎﺩﺭﺳﺘﯽ ﮐﻪ ﺑﻪ ﻳﮏ ﺟﻤﻠﻪ ﻳﺎ ﮔﺰﺍﺭﻩ ﻧﺴﺒﺖ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺻﺤﺖ ﻭ ﺍﻋﺘﺒﺎﺭ ﺧﺼﻮﺻﻴﺎﺕ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻫﺴﺘﻨﺪ ﻭ ﺩﺭﺳﺘﯽ ﻳﺎ ﻧﺎﺩﺭﺳﺘﯽ ﺧﺼﻮﺻﻴﺎﺗﯽ ﺑﺮﺍﯼ ﺟﻤﻠﻪ‌ﻫﺎ ﻭ ﮔﺰﺍﺭﻩ‌ﻫﺎ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﻧﺪ. ﺑﺎﻳﺪ ﺩﻗﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﻓﻠﺴﻔﻪ ﻭ ﻣﻨﻄﻖ ﻣﻤﮑﻦ ﺍﺳﺖ ﻭﺍﮊﻩ‌‌ﺍﯼ ﺭﺍ ﺩﺭ ﺯﺑﺎﻥ ﺭﻭﺯﻣﺮﻩ ﺑﻪ ﺩﻓﻌﺎﺕ ﺑﻪ ﮐﺎﺭ ﺑﺒﺮﻳﺪ ﻭ ﻣﻨﻈﻮﺭ ﺩﻳﮕﺮﯼ ﺍﺯ ﺁﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺍﻣﺎ ﺩﺭ ﺍﺻﻄﻼﺡ ﻓﻠﺴﻔﯽ ﻭ ﻣﻨﻄﻘﯽ ﮐﻪ ﺗﻌﺮﻳﻒ ﻫﻢ ﻣﯽ‌ﺷﻮﺩ، ﻳﮏ ﺍﺻﻄﻼﺡ ﻣﻌﻨﺎﯼ ﺩﻗﻴﻖ ﻭ ﻣﺘﻔﺎﻭﺗﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺩﻳﺪﻳﻢ ﺍﮔﺮ ﺩﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺑﺎﺷﺪ، ﻧﺘﻴﺠﻪ ﻫﻢ ﺑﺎﻳﺪ ﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺻﺤﻴﺢ ﻧﺘﻴﺠﻪ ﺑﺎﻳﺪ ﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺍﻳﻦ ﺑﺴﻴﺎﺭ ﻣﻌﻴﺎﺭ ﺍﺭﺯﺷﻤﻨﺪﯼ ﺍﺳﺖ ﺯﻳﺮﺍ ﻭﻗﺘﯽ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺻﺤﻴﺢ ﺑﺎﺷﺪ ﺁﻥ ﻭﻗﺖ ﭼﻴﺰ ﻣﻬﻤﯽ ﺩﺭ ﺩﺳﺖ ﺩﺍﺭﻳﻢ.
