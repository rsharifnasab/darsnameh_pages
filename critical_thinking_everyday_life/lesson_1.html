ﺩﺭﺱ ﺍﻭﻝ- ﻣﻨﻄﻖ ﭼﻴﺴﺖ؟

ﺩﺭ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﺩﺭﺳﻨﺎﻣﻪ ﺑﺎ ﻣﻮﺿﻮﻉ ﻓﻠﺴﻔﻪ، ﺩﻳﺪﻳﻢ ﮐﻪ ﻓﻠﺴﻔﻴﺪﻥ ﻭ ﺗﻔﮑﺮ ﻓﻠﺴﻔﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻭ ﺑﺎﻳﺪ ﻓﻌﺎﻟﻴﺘﯽ ﺭﻭﺯﻣﺮﻩ ﺑﺮﺍﯼ‌ ﻫﻤﻪ ﺍﻓﺮﺍﺩ ﺩﺭ ﻫﺮ ﻣﻮﻗﻌﻴﺖ ﻧﮋﺍﺩﯼ، ﺟﻨﺴﻴﺘﯽ، ﻃﺒﻘﺎﺗﯽ، ﺳﻨﯽ ﻭ ﻓﺮﻫﻨﮕﯽ ﺑﺎﺷﺪ. ﻓﻼﺳﻔﻪ ﺭﺍﻩ ﺣﻞ ﻣﺴﺎﺋﻞ ﺩﺷﻮﺍﺭ ﺭﺍ ﺑﻪ ﻣﺎ ﻧﺸﺎﻥ ﻧﻤﯽ‌ﺩﻫﻨﺪ، ﺑﻠﮑﻪ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﺧﻮﺩﻣﺎﻥ ﺑﻪ ﻃﻮﺭ ﻣﺴﺘﻘﻞ ﺩﺭﺑﺎﺭﻩ ﻫﻤﻪ ﭼﻴﺰ ﺑﻴﻨﺪﻳﺸﻴﻢ ﻭ ﺑﻪ ﭘﺎﺳﺦ‌ﻫﺎﯼ ﺧﻮﺩﻣﺎﻥ ﺑﺮﺳﻴﻢ.

ﺍﻣﺎ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﭘﺎﺳﺦ‌ﻫﺎﯼ ﻣﺎ ﺑﻪ ﺳﻮﺍﻻﺕ ﺩﺷﻮﺍﺭ ﺯﻧﺪﮔﯽ ﻭ ﺣﺘﯽ ﺳﻮﺍﻻﺕ ﻧﺴﺒﺘﺎ ﺁﺳﺎﻥ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺑﯽ‌ﻣﻌﻨﺎ ﻭ ﻣﻬﻤﻞ ﻧﺒﺎﺷﻨﺪ، ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺗﻔﮑﺮﻣﺎﻥ ﺭﻭﺷﻤﻨﺪ ﺑﺎﺷﺪ. ﺍﮔﺮ ﻳﺎﺩ ﺑﮕﻴﺮﻳﻢ ﮐﻪ ﻫﺮ ﺍﺩﻋﺎﻳﯽ ﺭﺍ ﺍﺯ ﺳﻮﯼ ﻫﺮ ﮐﺴﯽ ﮐﻪ ﺑﺎﺷﺪ ﺑﺪﻳﻬﯽ ﻧﭙﻨﺪﺍﺭﻳﻢ ﻭ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﺑﻴﻨﺪﻳﺸﻴﻢ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺩﺭ ﺭﺍﻩ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﻗﺪﻡ ﺑﺮﺩﺍﺭﻳﻢ.

ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﺎﻳﺪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺻﺮﻓﺎ ﻣﺨﺎﻟﻔﺖ ﺑﺎ ﻫﻤﻪ ﭼﻴﺰ ﻭ ﻫﻤﻪ ﮐﺲ ﺭﺍ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺗﻔﮑﺮ ﻭ ﺍﻧﺪﻳﺸﻪ ﺍﻧﺘﻘﺎﺩﯼ ﺑﻪ ﺷﻤﺎﺭ ﺁﻭﺭﺩ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﻓﻠﺴﻔﻪ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﻫﺮ ﺣﻮﺯﻩ ﺩﻳﮕﺮﯼ ﺍﺯ ﻣﻌﺮﻓﺖ ﺑﺸﺮﯼ ﺭﻭﺵ ﻭ ﻗﺎﻋﺪﻩ‌ﻫﺎﯼ ﺧﻮﺩﺵ ﺭﺍ ﺩﺍﺭﺩ. ﺭﺍﻩ ﻭ ﺭﻭﺵ ﻓﻠﺴﻔﻪ‌ﻭﺭﺯﯼ ﻭ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﻣﻨﻄﻖ ﻧﺎﻡ ﺩﺍﺭﺩ.

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺷﮏ ﻓﻠﺴﻔﯽ ﺭﺍﻫﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﭘﯽ ﺑﺮﺩﻥ ﺑﻪ ﺍﻳﻨﮑﻪ ﮐﺪﺍﻡ ﻳﮏ ﺍﺯ ﺑﺎﻭﺭﻫﺎ ﻭ ﺍﻋﺘﻘﺎﺩﺍﺕ ﻣﺎ ﻳﻘﻴﻨﯽ ﻧﻴﺴﺘﻨﺪ، ﻣﻨﻄﻖ ﻧﻴﺰ ﺭﻭﺷﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﻋﺒﻮﺭ ﺍﺯ ﻳﮏ ﺍﻋﺘﻘﺎﺩ ﻣﻮﺭﺩ ﺗﺮﺩﻳﺪ ﺑﻪ ﺳﻤﺖ ﺑﺎﻭﺭ ﺩﻳﮕﺮﯼ ﮐﻪ ﻳﻘﻴﻨﯽ‌ﺗﺮ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﻣﻨﻄﻖ ﺭﻭﺵ ﺳﺎﺧﺘﻦ ﺳﻨﮓ ﺑﻨﺎﯼ ﻳﻘﻴﻦ ﺩﺭ ﻓﻠﺴﻔﻪ ﺍﺳﺖ ﻭ ﻓﻴﻠﺴﻮﻓﺎﻥ ﮐﻪ ﺍﻟﮕﻮﻫﺎﯼ ﺍﻧﺪﻳﺸﻴﺪﻥ ﺭﺍ ﺍﺭﺯﻳﺎﺑﯽ ﻣﯽ‌ﮐﻨﻨﺪ، ﺍﺯ ﻃﺮﻳﻖ ﻣﻨﻄﻖ ﺩﺭ ﭘﯽ ﻳﺎﻓﺘﻦ ﺍﻟﮕﻮﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺗﻔﮑﺮ ﺑﺮ ﺁﻣﺪﻩ‌ﺍﻧﺪ.

ﺍﮐﺜﺮ ﺍﻓﺮﺍﺩ ﻋﻼﻗﻪ ﺩﺍﺭﻧﺪ ﮐﻪ ﻓﮑﺮ ﮐﻨﻨﺪ ﺁﺩﻡ‌ﻫﺎﻳﯽ ﻣﻨﻄﻘﯽ ﻫﺴﺘﻨﺪ. ﺍﮔﺮ ﺑﻪ ﮐﺴﯽ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺍﻭ ﻳﮏ ﻓﺮﺩ ﻏﻴﺮﻣﻨﻄﻘﯽ ﺍﺳﺖ، ﺑﻪ ﻃﻮﺭ ﻃﺒﻴﻌﯽ ﺑﻪ ﺍﻭ ﺍﻧﺘﻘﺎﺩ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﻭ ﺣﺘﯽ ﺩﺭ ﺑﻌﻀﯽ ﺍﺯ ﻣﻮﺍﺭﺩ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺣﺮﻑ ﻧﻮﻋﯽ ﺗﻮﻫﻴﻦ ﻳﺎ ﺍﻋﺘﺮﺍﺽ ﻗﻠﻤﺪﺍﺩ ﺷﻮﺩ. ﻏﻴﺮﻣﻨﻄﻘﯽ ﺑﻮﺩﻥ ﺳﺮﺩﺭﮔﻤﯽ، ﺍﺑﻬﺎﻡ ﻭ ﻏﻴﺮﻋﻘﻼﻧﯽ ﺑﻮﺩﻥ ﺍﺳﺖ. ﺍﻣﺎ ﻣﻨﻄﻘﯽ ﺑﻮﺩﻥ ﺑﻪ ﭼﻪ ﻣﻌﻨﺎﺳﺖ ﻭ ﻣﻨﻄﻖ ﭼﻴﺴﺖ؟

ﻣﻨﻄﻖ ﻳﮑﯽ ﺍﺯ ﻗﺪﻳﻤﯽ‌ﺗﺮﻳﻦ ﺣﻮﺯﻩ‌ﻫﺎﯼ ﺍﻧﺪﻳﺸﻪ ﺑﺸﺮﯼ ﺍﺳﺖ ﻭ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺍﺯ ﻣﺪﺭﻥ‌ﺗﺮﻳﻦ ﺷﺎﺧﻪ‌ﻫﺎﯼ ﺗﻔﮑﺮ ﻧﻴﺰ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ. ﭘﻴﺪﺍﻳﺶ ﻣﻨﻄﻖ ﺑﻪ ﻗﺮﻥ ﭼﻬﺎﺭﻡ ﭘﻴﺶ ﺍﺯ ﻣﻴﻼﺩ ﻭ ﺯﻣﺎﻥ ﻓﻼﺳﻔﻪ ﻳﻮﻧﺎﻥ ﺑﺎﺯ ﻣﯽ‌ﮔﺮﺩﺩ. ﺗﻨﻬﺎ ﺣﻮﺯﻩ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﻣﻨﻄﻖ ﻗﺪﻳﻤﯽ‌ﺗﺮﻧﺪ، ﻓﻠﺴﻔﻪ ﻭ ﺭﻳﺎﺿﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻫﺮ ﺩﻭﯼ ﺁﻧﻬﺎ ﻋﻤﻴﻘﺎ ﺑﺎ ﻣﻨﻄﻖ ﺩﺭﻫﻢ ﺗﻨﻴﺪﻩ‌ﺍﻧﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺑﻪ ﺳﺨﺘﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﻨﻄﻖ ﺭﺍ ﺍﺯ ﻓﻠﺴﻔﻪ ﻭ ﺭﻳﺎﺿﯽ ﺟﺪﺍ ﮐﺮﺩ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺁﻣﻮﺧﺘﻦ ﻓﻠﺴﻔﻪ ﺑﻪ ﺩﺍﻧﺴﺘﻦ ﻣﻨﻄﻖ ﻧﻴﺎﺯ ﺩﺍﺭﻳﻢ.

ﺍﺯ ﺳﻮﯼ ﺩﻳﮕﺮ ﻣﻨﻄﻖ ﺩﺭ ﺁﻏﺎﺯ ﻗﺮﻥ ﺑﻴﺴﺘﻢ ﺑﺎ ﺍﻧﻘﻼﺑﯽ ﺑﻨﻴﺎﺩﯼ ﻣﻮﺍﺟﻪ ﺷﺪ ﻭ ﺭﻳﺎﺿﻴﺎﺕ ﺟﺪﻳﺪ ﺁﻥ ﺭﺍ ﮐﺎﻣﻼ ﻣﺘﺤﻮﻝ ﮐﺮﺩ. ﻫﻤﭽﻨﻴﻦ ﺩﺭ ﻧﻴﻤﻪ ﭘﺎﻳﺎﻧﯽ ﻗﺮﻥ ﺑﻴﺴﺘﻢ، ﻋﻠﻮﻡ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻭ ﺍﻃﻼﻋﺎﺕ ﺗﺎﺛﻴﺮ ﺍﻧﻘﻼﺑﯽ ﺩﻳﮕﺮﯼ ﺑﺮ ﻣﻨﻄﻖ ﮔﺬﺍﺷﺘﻨﺪ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﻓﻠﺴﻔﻪ ﺗﺎ ﺭﻳﺎﺿﻴﺎﺕ ﮔﺮﻓﺘﻪ ﺗﺎ ﻋﻠﻮﻡ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻭ ﺣﺘﯽ ﻋﻠﻮﻡ ﻣﺤﺾ ﻭ ﮐﺎﺭﺑﺮﺩﯼ ﻫﻤﮕﯽ ﺑﻪ ﻣﻨﻄﻖ ﻭﺍﺑﺴﺘﻪ ﻫﺴﺘﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﻨﻄﻖ ﻣﻮﺿﻮﻋﯽ ﻣﺮﮐﺰﯼ ﺩﺭ ﺍﻧﺪﻳﺸﻪ ﺑﺸﺮﯼ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﻣﻨﻄﻖ ﺳﺘﻮﻥ ﻓﻘﺮﺍﺕ ﺗﻤﺪﻥ ﺑﺸﺮ ﺍﺳﺖ ﮐﻪ ﺗﻤﺎﻡ ﻧﻈﺎﻡ ﻋﻠﻢ، ﻓﻠﺴﻔﻪ ﻭ ﻗﺎﻧﻮﻥ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﻣﺤﮑﻢ، ﺍﺳﺘﻮﺍﺭ ﻭ ﺳﺎﺯﮔﺎﺭ ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﻧﮕﻪ ﻣﯽ‌ﺩﺍﺭﺩ. ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻣﻨﻄﻖ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻧﻴﺰ ﺑﺎ ﻣﺎﺳﺖ ﻭ ﺩﺭ ﺷﮑﻞ ﺩﺍﺩﻥ ﺑﺎﻭﺭﻫﺎﻳﻤﺎﻥ ﺑﻪ ﮐﻤﮏ ﻣﯽ‌ﺁﻳﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﻧﺪﺍﻧﻴﻢ ﮐﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻭ ﺩﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ‌ﻣﺎﻥ ﻣﻨﻄﻖ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺑﺮﻳﻢ؛ ﺍﻣﺎ ﭼﻪ ﺑﻪ ﺁﻥ ﺁﮔﺎﻩ ﺑﺎﺷﻴﻢ ﻭ ﭼﻪ ﻧﺒﺎﺷﻴﻢ، ﻣﺪﺍﻡ ﺩﺭ ﺣﺎﻝ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻨﻄﻖ ﻫﺴﺘﻴﻢ. ﭘﺲ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﻪ ﺁﻥ ﺁﮔﺎﻩ ﺑﺎﺷﻴﻢ ﺗﺎ ﺑﺘﻮﺍﻧﻴﻢ ﺑﻪ ﺩﺭﺳﺘﯽ ﻭ ﺑﻬﻴﻨﻪ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ ﺗﺎ ﻫﻢ ﺑﺎﻭﺭﻫﺎﻳﻤﺎﻥ ﺭﺍ ﻣﺤﮑﻢ‌ﺗﺮ ﮐﻨﻴﻢ ﻭ ﻫﻢ ﺑﺘﻮﺍﻧﻴﻢ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻫﻢ ﺑﻘﺒﻮﻻﻧﻴﻢ. ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﻴﻢ ﺑﺎﻭﺭﻫﺎ ﻭ ﺍﺩﻋﺎﻫﺎﯼ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻫﻢ ﺑﻪ ﭼﺎﻟﺶ ﺑﮑﺸﻴﻢ. ﺗﻤﺎﻡ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺑﺎ ﻣﻨﻄﻖ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﺍﺳﺖ.

ﺑﺎ ﻫﻤﻪ ﺍﻳﻦ ﺗﻌﺎﺭﻳﻒ ﻣﻤﮑﻦ ﺍﺳﺖ ﻫﻤﭽﻨﺎﻥ ﺑﻪ ﺩﻧﺒﺎﻝ ﺗﻮﺻﻴﻔﯽ ﺩﻗﻴﻖ‌ﺗﺮ، ﮐﺎﺭﺑﺮﺩﯼ‌ﺗﺮ ﻭ ﺭﻭﺷﻦ‌ﺗﺮ ﺍﺯ ﻣﻨﻄﻖ ﺑﺎﺷﻴﻢ. ﻣﺎﻧﻨﺪ ﻫﺮ ﺣﻮﺯﻩ ﺩﻳﮕﺮﯼ ﺗﺎ ﺑﺎ ﮐﺎﺭﺑﺮﺩﻫﺎﯼ ﻳﮏ ﻣﻔﻬﻮﻡ ﺁﺷﻨﺎ ﻧﺸﻮﻳﻢ، ﺷﺎﻳﺪ ﻧﺘﻮﺍﻧﻴﻢ ﺑﻪ ﻃﻮﺭ ﺩﻗﻴﻖ ﻭ ﻭﺍﺿﺢ ﺁﻥ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﻨﻴﻢ. ﭘﺲ ﺑﺒﻴﻨﻴﻢ ﻣﻨﻄﻖ ﺩﻗﻴﻘﺎ ﺑﻪ ﭼﻪ ﮐﺎﺭ ﻣﯽ‌ﺁﻳﺪ؟

 

ﮐﺎﺭﮐﺮﺩ ﻣﻨﻄﻖ
ﻭﻗﺘﯽ ﺍﺯ ﻣﻨﻄﻖ ﺣﺮﻑ ﻣﯽ‌ﺯﻧﻴﻢ ﺑﻪ ﻃﻮﺭ ﻋﻤﻠﯽ ﻭ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻣﻨﻈﻮﺭﻣﺎﻥ ﺩﻟﻴﻞ ﻭ ﺑﺮﻫﺎﻥ ﺁﻭﺭﺩﻥ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﺍﺩﻋﺎﻫﺎ ﻭ ﺑﺎﻭﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺩﺍﺭﻳﻢ. ﺑﺪﻳﻬﯽ ﺍﺳﺖ ﮐﻪ ﺑﺪﻭﻥ ﺍﺭﺍﺋﻪ ﺩﻟﻴﻞ ﻫﻴﭻ ﮐﺲ ﺍﺩﻋﺎﯼ ﻣﺎ ﺭﺍ ﻗﺒﻮﻝ ﻧﻤﯽ‌ﮐﻨﺪ ﻭ ﺧﻮﺩﻣﺎﻥ ﻧﻴﺰ ﻫﻴﭻ ﮔﺎﻩ ﺍﺩﻋﺎﻳﯽ ﺭﺍ ﺑﺪﻭﻥ ﺷﻨﻴﺪﻥ ﺩﻻﻳﻠﯽ ﻣﻌﺘﺒﺮ ﻧﻤﯽ‌ﭘﺬﻳﺮﻳﻢ.

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﮔﺮ ﮐﺴﯽ ﺍﺩﻋﺎ ﮐﻨﺪ ﮐﻪ «ﻣﺎﺳﺖ ﺳﻴﺎﻩ ﺍﺳﺖ» ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﺍﺩﻋﺎﻳﺶ ﺩﻟﻴﻞ ﻭ ﻣﺪﺭﮎ ﻭ ﺑﺮﻫﺎﻥ ﺑﻴﺎﻭﺭﺩ. ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺩﻟﻴﻞ ﻭ ﻣﺪﺭﮎ ﺭﺍ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﺍﺩﻋﺎﻫﺎﯼ ﻧﺎﻣﺘﻌﺎﺭﻓﯽ ﻣﺎﻧﻨﺪ ﺳﻴﺎﻫﯽ ﻣﺎﺳﺖ ﻧﺒﺎﻳﺪ ﺩﺭﺧﻮﺍﺳﺖ ﮐﻨﻴﻢ ﻭ ﺣﺘﯽ ﺍﮔﺮ ﮐﺴﯽ ﺑﮕﻮﻳﺪ «ﻣﺎﺳﺖ ﺳﻔﻴﺪ ﺍﺳﺖ» ﻧﻴﺰ ﺑﺎﻳﺪ ﺩﻟﻴﻞ ﻭ ﻣﺪﺭﮎ ﻣﻮﺟﻪ ﻭ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﺍﺭﺍﺋﻪ ﺩﻫﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﻴﭻ ﺍﺩﻋﺎﻳﯽ ﺑﺪﻭﻥ ﺍﺭﺍﺋﻪ ﺩﻻﻳﻠﯽ ﺑﺮﺍﯼ ﭘﺸﺘﻴﺒﺎﻧﯽ ﺍﺯ ﺁﻥ ﺍﺩﻋﺎ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻧﻴﺴﺖ.

ﺍﺭﺍﺋﻪ ﺩﻟﻴﻞ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﻫﺮ ﺍﺩﻋﺎ ﺩﺭ ﻣﻨﻄﻖ «ﺍﺳﺘﺪﻻﻝ» ﻧﺎﻡ ﺩﺍﺭﺩ. ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺳﻨﮓ ﺑﻨﺎﯼ ﻧﻈﺮﻳﻪ‌ﻫﺎﯼ ﻓﻠﺴﻔﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﺩﻻﻳﻞ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﻭ ﻣﻨﻄﻖ ﻣﻼﺗﯽ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺳﺎﺧﺘﺎﺭﺩﻫﻨﺪﻩ ﺭﺍ ﺑﻪ ﻳﮑﺪﻳﮕﺮ ﻣﺘﺼﻞ ﻣﯽ‌ﮐﻨﺪ. ﺍﻳﺪﻩ‌ﻫﺎﯼ ﺧﻮﺏ ﺍﺭﺯﺵ ﭼﻨﺪﺍﻧﯽ ﻧﺪﺍﺭﻧﺪ ﻣﮕﺮ ﺍﻳﻨﮑﻪ ﺗﻮﺳﻂ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺧﻮﺏ ﺣﻤﺎﻳﺖ ﺷﻮﻧﺪ. ﺍﻳﺪﻩ‌ﻫﺎﯼ ﺧﻮﺏ ﻧﻴﺎﺯﻣﻨﺪ ﺁﻥ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﻋﻘﻼﻧﯽ ﺗﻮﺟﻴﻪ ﺷﻮﻧﺪ ﻭ ﺍﻳﻦ ﺑﺪﻭﻥ ﻳﮏ ﺯﻳﺮﺑﻨﺎﯼ ﻣﻨﻄﻘﯽ ﻣﺤﮑﻢ ﻭ ﺍﺳﺘﻮﺍﺭ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻧﻴﺴﺖ.

ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺭﻭﺷﻨﯽ ﻭ ﻭﺿﻮﺡ ﻭ ﺑﺎ ﺩﻗﺖ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺑﺎﺷﻨﺪ ﺩﺭ ﻣﻘﺎﺑﻞ ﻫﺮﮔﻮﻧﻪ ﺍﺭﺯﻳﺎﺑﯽ ﻭ ﺍﻧﺘﻘﺎﺩ ﺑﺎﺯ ﻫﺴﺘﻨﺪ ﻭ ﻫﻤﻴﻦ ﻓﺮﺁﻳﻨﺪ ﮐﻪ ﺷﺎﻣﻞ ﻭﺍﮐﻨﺶ، ﺍﺻﻼﺡ ﻭ ﻋﺪﻡ ﭘﺬﻳﺮﺵ ﺍﺣﺘﻤﺎﻟﯽ ﺍﺳﺖ ﺑﻪ ﭘﻴﺸﺮﻓﺖ ﺍﻧﺪﻳﺸﻪ ﻣﻨﺠﺮ ﻣﯽ‌ﺷﻮﺩ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﺳﺎﺯﻧﺪﻩ ﺍﺳﺖ، ﺣﺘﯽ ﺍﮔﺮ ﺗﻤﺎﻡ ﺑﻨﻴﺎﺩ ﺍﻧﺪﻳﺸﻪ‌ﻣﺎﻥ ﺭﺍ ﺯﻳﺮ ﻭ ﺭﻭ ﮐﻨﺪ.

ﺩﺭﺑﺎﺭﻩ ﺍﺳﺘﺪﻻﻝ
ﻫﻤﻪ ﻣﺎ ﺍﺳﺘﺪﻻﻝ ﻣﯽ‌ﮐﻨﻴﻢ. ﻣﺎ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺑﺮ ﺍﺳﺎﺱ ﺁﻧﭽﻪ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ. ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﺎ ﺩﻟﻴﻞ ﻗﺎﻧﻊ ﮐﻨﻴﻢ ﻭ ﻣﻨﻄﻖ ﺩﺍﻧﺶ ﻣﻄﺎﻟﻌﻪ‌ﯼ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﭼﻪ ﭼﻴﺰﯼ ﺩﻟﻴﻞ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﭼﻴﺴﺖ ﻭ ﭼﺮﺍ؟

ﻫﻴﭻ ﭼﻴﺰﯼ ﺑﺮﺍﯼ ﻳﮏ ﻣﮑﺎﻟﻤﻪ ﻃﺒﻴﻌﯽ‌ﺗﺮ ﺍﺯ ﺍﺳﺘﺪﻻﻝ ﻧﻴﺴﺖ. ﻣﺎ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﺑﺎ ﺍﻭ ﺣﺮﻑ ﻣﯽ‌ﺯﻧﻴﻢ ﻗﺎﻧﻊ ﮐﻨﻴﻢ ﮐﻪ ﺩﺭﺳﺖ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﻭ ﺣﻖ ﺑﺎ ﻣﺎﺳﺖ ﻭ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﻢ ﮐﻪ ﻧﺸﺎﻥ ﺩﻫﻴﻢ ﺁﻧﭽﻪ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﺑﻪ ﻭﺿﻮﺡ ﻧﺘﻴﺠﻪ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﻣﺎ ﺩﺭ ﺑﺤﺚ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﺁﻥ ﺭﺍ ﻗﺒﻮﻝ ﺩﺍﺭﺩ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﺩﺭ ﻋﻤﻞ ﺑﻪ ﻣﻌﻨﯽ ﺭﺳﻴﺪﻥ ﺍﺯ ﻳﮏ ﺑﺎﻭﺭ ﺑﺪﻳﻬﯽ ﻭ ﻣﻮﺭﺩ ﻗﺒﻮﻝ ﻫﻤﮕﺎﻥ، ﺑﻪ ﻳﮏ ﺑﺎﻭﺭ ﺩﻳﮕﺮ ﺍﺳﺖ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺍﺑﺘﺪﺍ ﮐﺴﯽ ﺁﻥ ﺭﺍ ﻗﺒﻮﻝ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻫﻨﺮ ﻣﻨﻄﻖ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﺭﻭﺷﻤﻨﺪ ﻭ ﺑﺮ ﺍﺳﺎﺱ ﻗﻮﺍﻋﺪ ﻗﺎﺑﻞ ﻗﺒﻮﻝ ﻧﺸﺎﻥ ﺩﻫﺪ ﭼﻨﻴﻦ ﮔﺬﺍﺭﯼ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﺍﺳﺖ.

ﺍﺳﺘﺪﻻﻝ ﺍﺯ ﻣﻨﺎﺑﻊ ﺍﺻﻠﯽ ﺩﺍﻧﺶ ﻭ ﻣﻌﺮﻓﺖ ﺑﺸﺮﯼ ﺍﺳﺖ ﻭ ﻣﻨﻄﻖ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺑﻪ ﺯﺑﺎﻥ ﺩﻳﮕﺮ، ﻣﻨﻄﻖ ﺷﺎﺧﻪ‌ﺍﯼ ﺍﺯ ﻓﻠﺴﻔﻪ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺩﺭﺳﺖ ﻭ ﻧﺎﺩﺭﺳﺖ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ. ﺍﺭﺳﻄﻮ ﻣﻨﻄﻖ ﺭﺍ ﺑﻨﻴﺎﺩ ﻭ ﺳﺎﺯﻧﺪﻩ ﻓﻠﺴﻔﻪ ﻣﯽ‌ﺩﺍﻧﺴﺖ ﻭ ﺍﺯ ﻫﻤﺎﻥ ﺯﻣﺎﻥ ﻫﻤﻪ ﻓﻼﺳﻔﻪ ﻧﻘﺶ ﻣﻬﻤﯽ ﺭﺍ ﺑﻪ ﻣﻨﻄﻖ ﺩﺭ ﮐﺎﺭﺷﺎﻥ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﻭ ﻫﻤﭽﻨﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ. ﻣﻨﻄﻖ ﺣﻮﺯﻩ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﻗﻀﺎﻭﺕ ﻣﯽ‌ﮐﻨﺪ ﺁﻳﺎ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻣﻌﺘﺒﺮ ﻫﺴﺘﻨﺪ ﻳﺎ ﻧﻪ.

ﭘﺲ ﺍﺯ ﺗﻌﺮﻳﻒ ﻣﻨﻄﻖ ﻭ ﻣﻌﺮﻓﯽ ﻣﻘﺪﻣﺎﺗﯽ ﺍﺳﺘﺪﻻﻝ، ﻭﻗﺖ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺳﺖ ﺑﻪ ﮐﺎﺭ ﺷﻮﻳﻢ ﺑﻪ ﺍﺳﺘﺪﻻﻝ ﻭ ﺗﻌﺎﺭﻳﻒ ﺁﻥ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﺩﺭﺱ ﺑﻌﺪﯼ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺭﺩ،‌ ﺍﻣﺎ ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﺑﻪ ﺳﺮﺍﻍ ﺩﺭﺱ‌ ﺑﻌﺪﯼ ﺑﺮﻭﻳﻢ ﻧﮕﺎﻫﯽ ﺑﻪ ﻣﺴﻴﺮﯼ ﮐﻪ ﺩﺭ ﻃﻮﻝ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺑﺎ ﻣﻮﺿﻮﻉ ﻣﻨﻄﻖ ﻃﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ ﺑﻴﻨﺪﺍﺯﻳﻢ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺑﻪ ﺗﺮﺗﻴﺐ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ: ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﺍﺳﺘﺪﻻﻝ ﻣﻨﻄﻘﯽ ﺭﺍ ﻣﻌﺮﻓﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ. ﺩﺭﺱ ﺳﻮﻡ ﺑﻪ ﻫﺪﻑ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﻭ ﮐﺎﺭﺑﺮﺩﻫﺎﯼ ﺁﻥ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ. ﻣﻮﺿﻮﻉ ﺩﺭﺱ ﭼﻬﺎﺭﻡ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺍﺳﺎﺳﯽ ﻫﺮ ﺍﺳﺘﺪﻻﻝ ﻳﻌﻨﯽ ﺍﻋﺘﺒﺎﺭ ﻭ ﺻﺤﺖ ﺍﺳﺖ. ﺩﺭﺱ ﭘﻨﺠﻢ ﺑﻪ ﺍﻧﻮﺍﻉ ﺍﺳﺘﺪﻻﻝ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺭﺩ ﻭ ﺩﺭﺱ ﺷﺸﻢ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ ﮐﻪ ﺁﻳﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﻫﻤﻪ ﻣﺒﺎﺣﺚ ﭘﻴﺮﻭﺯ ﺷﺪ؟ ﺩﺭﺱ ﻫﻔﺘﻢ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻣﻨﻄﻘﯽ ﻭ ﺩﺭﺱ ﻫﺸﺘﻢ ﭘﺎﺭﺍﺩﻭﮐﺲ‌ﻫﺎ ﺭﺍ ﻣﻌﺮﻓﯽ ﻣﯽ‌ﮐﻨﺪ. ﺩﺭ ﺩﺭﺱ ﻧﻬﻢ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﭘﺎﺳﺦ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ ﮐﻪ ﻣﻨﻄﻖ ﺑﻪ ﭼﻪ ﮐﺎﺭ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺩﺭ ﻧﻬﺎﻳﺖ ﺩﺭ ﺩﺭﺱ ﺩﻫﻢ ﻣﻨﺎﺑﻌﯽ ﺭﺍ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺑﻴﺸﺘﺮ ﻣﻌﺮﻓﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
