ﺗﻘﺮﻳﺒﺎ ﺑﻌﻴﺪ ﺍﺳﺖ ﮐﻪ ﮐﺴﯽ ﺍﺳﻢ ﺷﺮﻟﻮﮎ ﻫﻠﻤﺰ، ﮐﺎﺭﺁﮔﺎﻩ ﺧﻴﺎﻟﯽ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﺁﺭﺗﻮﺭ ﮐﺎﻧﻦ ﺩﻭﻳﻞ ﺭﺍ ﻧﺸﻨﻴﺪﻩ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺑﻪ ﻣﺎﺟﺮﺍﻫﺎﯼ ﺷﺮﻟﻮﮎ‌ ﻫﻠﻤﺰ ﻋﻼﻗﻤﻨﺪ ﺑﺎﺷﻴﺪ، ﮐﺘﺎﺑﺶ ﺭﺍ ﺧﻮﺍﻧﺪﻩ ﺑﺎﺷﻴﺪ ﻳﺎ ﺍﻗﺘﺒﺎﺱ‌ﻫﺎﯼ ﺳﻴﻨﻤﺎﻳﯽ ﻭ ﺗﻠﻮﻳﺰﻳﻮﻧﯽ ﺁﻥ ﺭﺍ ﺩﻳﺪﻩ ﺑﺎﺷﻴﺪ، ﺣﺘﻤﺎ ﺍﺯ ﺷﻴﻔﺘﮕﯽ ﻋﻤﻴﻖ ﻫﻠﻤﺰ ﺑﻪ ﻣﻨﻄﻖ ﺧﺒﺮ ﺩﺍﺭﻳﺪ. ﺍﺯ ﻧﻈﺮ ﻫﻠﻤﺰ ﻫﺮ ﻣﻌﻤﺎﯼ ﺟﻨﺎﻳﯽ ﻳﮏ ﭘﺎﺯﻝ ﻣﻨﻄﻘﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﻃﺮﻳﻖ ﺍﺳﺘﻨﺘﺎﺝ ﻣﻨﻄﻘﯽ ﻭ ﺑﺎ ﮐﻤﮏ ﺭﻭﺵ‌ﻫﺎ ﻭ ﻗﻮﺍﻋﺪ ﺍﻳﻦ ﺣﻮﺯﻩ ﺁﻥ ﺭﺍ ﺣﻞ ﮐﺮﺩ. ﺣﺎﻻ ﮐﻪ ﺗﺎ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﺑﺎ ﺍﺳﺘﻨﺘﺎﺝ ﻣﻨﻄﻘﯽ ﻭ ﺭﻭﺵ‌ﻫﺎﯼ ﺁﻥ ﺁﺷﻨﺎ ﺷﺪﻩ‌ﺍﻳﻢ ﺷﺎﻳﺪ ﺑﺘﻮﺍﻧﻴﻢ ﺩﻟﻴﻞ ﻭﺳﻮﺍﺱ ﻋﺠﻴﺐ ﻫﻠﻤﺰ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﺁﻥ ﺩﺭﮎ ﮐﻨﻴﻢ.

ﮐﺎﺭﺁﮔﺎﻩ ﺧﻴﺎﻟﯽ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﮐﻼﺳﻴﮏ ﮐﺎﻧﻦ ﺩﻭﻳﻞ ﺑﺎﻭﺭ ﺩﺍﺭﺩ ﮐﻪ ﺟﻮﺍﺏ ﻫﺮ ﻣﻌﻤﺎﻳﯽ ﮐﻪ ﺑﺎ ﺁﻥ ﻣﻮﺍﺟﻪ ﺍﺳﺖ ﻫﻤﻴﺸﻪ ﻭﺍﺿﺢ ﻭ ﺑﺪﻳﻬﯽ ﺟﻠﻮﯼ ﭼﺸﻢ ﻫﻤﻪ ﺍﺳﺖ ﻭ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻥ ﺁﻥ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩ. ﺍﻭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﻧﺸﺎﻧﻪ‌ﻫﺎ ﻭ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﺩﺭ ﻣﺤﻴﻂ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻧﺘﻴﺠﻪ ﮐﺎﻓﯽ ﺍﺳﺖ. ﺍﮔﺮ ﺗﻮﺟﻪ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ ﻫﻠﻤﺰ ﺍﺯ ﺭﻭﺵ ﺍﺳﺘﻨﺘﺎﺝ «ﻗﻴﺎﺳﯽ» ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﭘﺎﺳﺦ ﻣﻌﻤﺎﻫﺎﯼ ﺟﻨﺎﻳﯽ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺑﺎ ﺩﺍﻧﺴﺘﻦ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﻓﺮﺽ ﺑﺮ ﺩﺭﺳﺘﯽ ﺁﻧﻬﺎ، ﭘﺎﺳﺦ ﻳﺎ ﻧﺘﻴﺠﻪ ﺑﻪ ﻃﻮﺭ ﺍﺟﺘﻨﺎﺏ ﻧﺎﭘﺬﻳﺮﯼ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻳﻦ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻣﯽ‌ﺁﻳﺪ.

ﺍﻣﺎ ﺑﻌﻀﯽ ﺍﺯ ﻣﻨﻄﻖ‌ﺩﺍﻧﺎﻥ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﺍﻳﻦ ﺭﻭﺵ ﺑﺮﺍﯼ ﺣﻞ ﻣﻌﻤﺎﻫﺎﯼ ﮐﺎﺭﺁﮔﺎﻫﯽ ﻳﺎ ﻣﺸﺎﺑﻪ ﺁﻥ ﻫﻴﭻ ﺭﺑﻄﯽ ﺑﻪ ﺍﺳﺘﻨﺘﺎﺝ ﻗﻴﺎﺳﯽ ﻧﺪﺍﺭﺩ ﻭ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻦ ﮐﻪ ﻫﻠﻤﺰ ﺑﺮ ﺍﺛﺮ ﺗﺤﻠﻴﻞ ﺍﻃﻼﻋﺎﺕ ﺑﻪ ﻧﺘﺎﻳﺞ ﺗﺎﺯﻩ‌ﺍﯼ ﺩﺳﺖ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ، ﺑﺎﻳﺪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺯﻣﺮﻩ ﺍﺳﺘﻨﺘﺎﺝ‌ﻫﺎﯼ «ﺍﺳﺘﻘﺮﺍﻳﯽ» ﺑﻪ ﺷﻤﺎﺭ ﺁﻭﺭﺩ.

ﺧﺐ ﺑﺒﻴﻨﻴﻢ ﺍﺻﻼ ﺍﻳﻦ ﺩﻭ ﺍﺻﻄﻼﺡ ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻳﻢ ﭼﻪ ﻣﻌﻨﺎﻳﯽ ﺩﺍﺭﻧﺪ ﻭ ﺍﺻﻼ ﺩﺭﺳﻨﺎﻣﻪ ﻣﻨﻄﻖ ﻣﺎ ﭼﻪ ﺭﺑﻄﯽ ﺑﻪ ﺷﺮﻟﻮﮎ ﻫﻠﻤﺰ ﻭ ﻣﻌﻤﺎﻫﺎﯼ ﮐﺎﺭﺁﮔﺎﻫﯽ ﺩﺍﺭﺩ؟

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺍﺯ ﺍﺑﺘﺪﺍﯼ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺑﺎﺭﻫﺎ ﺗﮑﺮﺍﺭ ﮐﺮﺩﻳﻢ، ﺍﺳﺘﺪﻻﻝ ﻣﻨﻄﻘﯽ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﺍﺩﻋﺎﻫﺎ، ﺭﺳﻴﺪﻥ ﺍﺯ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﯼ ﺑﺪﻳﻬﯽ ﺑﻪ ﻧﺘﺎﻳﺞ ﻏﻴﺮﺑﺪﻳﻬﯽ ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﻫﺪﺍﻑ ﺩﻳﮕﺮ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺭﻭﺩ. ﺍﻣﺎ ﺍﺳﺘﺪﻻﻝ ﻫﻢ ﻣﺎﻧﻨﺪ ﻫﺮ ﻓﺮﺁﻳﻨﺪ ﺭﻭﺷﻤﻨﺪ ﺩﻳﮕﺮﯼ ﺩﺭ ﻋﻠﻢ ﻭ ﻓﻠﺴﻔﻪ ﺍﻧﻮﺍﻉ ﻣﺘﻔﺎﻭﺗﯽ ﺩﺍﺭﺩ ﻭ ﻗﻮﺍﻋﺪ ﻭ ﺗﻌﺎﺭﻳﻒ ﻣﺘﻔﺎﻭﺗﯽ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻥ ﻧﺴﺒﺖ ﺩﺍﺩ.

ﺩﻭ ﻧﻮﻉ ﺍﺻﻠﯽ ﺍﺳﺘﺪﻻﻝ،‌ ﺍﺳﺘﻨﺘﺎﺝ‌ﻫﺎﯼ ﻗﻴﺎﺳﯽ ﻭ ﺍﺳﺘﻨﺘﺎﺝ‌ﻫﺎﯼ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺩﺭﺑﺎﺭﻩ ﺁﻧﻬﺎ ﺑﺤﺚ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

۳- ﺑﻨﺎﺑﺮﺍﻳﻦ، ﭘﺮﻧﺪﻩ‌ ﺑﺎﻻﯼ ﺩﺭﺧﺖ ﺧﺎﻧﻪ ﻣﺎ ﺳﻴﺎﻩ ﺍﺳﺖ.

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻣﺘﻮﺟﻪ ﺷﺪﻳﻢ ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ، ﺍﻃﻼﻋﺎﺗﯽ ﺑﻴﺶ ﺍﺯ ﺁﻧﭽﻪ ﺩﺭ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﻪ ﻣﺎ ﺍﺿﺎﻓﻪ ﻧﻤﯽ‌ﺷﻮﺩ. ﺣﺎﻝ ﺍﮔﺮ ﺑﻪ ﺩﺍﺳﺘﺎﻥ ﻣﺎﺟﺮﺍﻫﺎﯼ ﺷﺮﻟﻮﮎ ﻫﻠﻤﺰ ﺑﺎﺯﮔﺮﺩﻳﻢ، ﻣﻄﺎﺑﻖ ﺍﺩﻋﺎﯼ ﺧﻮﺩ ﻫﻠﻤﺰ ﮐﺎﺭﯼ ﮐﻪ ﺍﻭ ﻣﯽ‌ﮐﻨﺪ ﺻﺮﻓﺎ ﺍﺳﺘﺨﺮﺍﺝ ﺍﻃﻼﻋﺎﺕ ﺑﺪﻳﻬﯽ ﻭ ﻣﻮﺟﻮﺩ ﺍﺯ ﺷﻮﺍﻫﺪ ﻣﻮﺟﻮﺩ ﺍﺳﺖ. ﺍﻣﺎ ﻫﻤﺰﻣﺎﻥ ﻣﯽ‌ﺗﻮﺍﻥ ﮔﻔﺖ ﮐﻪ ﻫﻠﻤﺰ ﭼﻴﺰﯼ ﻓﺮﺍﺗﺮ ﺍﺯ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺻﺮﻑ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺭﻭﺍﺑﻂ ﻋﻠﺖ ﻭ ﻣﻌﻠﻮﻟﯽ ﺑﻴﻦ ﺭﻭﻳﺪﺍﺩﻫﺎ ﺭﺍ ﮐﺸﻒ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺩﺭ ﺯﻣﺮﻩ ﻧﻮﻉ ﺩﻳﮕﺮﯼ ﺍﺯ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻳﻌﻨﯽ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺟﺎ ﻣﯽ‌ﮔﻴﺮﺩ.

ﺩﺭ ﺍﻳﻦ ﻣﺜﺎﻝ ﺍﺯ ﺭﻭﺵ ﺍﺳﺘﻨﺘﺎﺝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺷﺪﻩ ﺍﺳﺖ. ﺧﺐ ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺍﺯ ﺷﮑﻞ ﻇﺎﻫﺮﯼ ﺁﻥ ﻫﻢ ﭘﻴﺪﺍﺳﺖ ﺍﻳﻦ ﻣﺜﺎﻝ ﺑﺎ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﮐﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺁﻥ ﺭﺍ ﺑﺎ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﻣﻘﺎﻳﺴﻪ ﮐﻨﻴﻢ ﺑﻪ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﻣﯽ‌ﺭﺳﻴﻢ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺣﺮﮐﺖ ﺍﺯ ﻳﮏ ﺣﮑﻢ ﮐﻠﯽ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻳﮏ ﻧﺘﻴﺠﻪ ﺟﺰﺋﯽ ﺍﺳﺖ ﻭ ﻳﺎ ﺩﺭ ﺍﻧﻮﺍﻉ ﺩﻳﮕﺮ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻗﻴﺎﺳﯽ ﻳﮏ ﺣﮑﻢ ﮐﻠﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻳﮏ ﺣﮑﻢ ﮐﻠﯽ ﺩﻳﮕﺮ ﺭﺍ ﻧﺘﻴﺠﻪ ﺩﻫﺪ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﺜﺎﻝ ﻣﺎ ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩﯼ ﺣﮑﻢ ﺟﺰﺋﯽ ﺩﺍﺭﻳﻢ ﻭ ﻳﮏ ﺣﮑﻢ ﮐﻠﯽ ﺭﺍ ﻧﺘﻴﺠﻪ ﮔﺮﻓﺘﻪ‌ﺍﻳﻢ. ﺍﻳﻦ ﺭﻭﺵ ﺍﺳﺘﺪﻻﻟﯽ ﺍﺳﺘﻘﺮﺍ ﻧﺎﻡ ﺩﺍﺭﺩ.

ﺭﺍﻳﺞ‌ﺗﺮﻳﻦ ﺷﮑﻞ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺗﻌﻤﻴﻢ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻫﻤﻴﻦ ﻣﺜﺎﻝ ﻫﻢ ﺁﻥ ﺭﺍ ﻣﻼﺣﻈﻪ ﮐﺮﺩﻳﻢ. ﺍﮔﺮ ﺑﻪ ﻣﻌﻨﺎﯼ ﻭﺍﮊﻩ ﺗﻌﻤﻴﻢ ﺩﻗﺖ ﮐﻨﻴﻢ، ﺁﻥ ﺭﺍ ﺑﻪ ﻣﻌﻨﺎﯼ ﻋﻤﻮﻣﻴﺖ ﺩﺍﺩﻥ ﻳﺎ ﮐﻠﯽ ﺳﺎﺯﯼ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺗﻌﺮﻳﻒ ﮐﻨﻴﻢ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﺎ ﻣﺜﺎﻝ‌ﻫﺎ ﻭ ﻧﻤﻮﻧﻪ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﮐﻪ ﺑﻪ ﻫﻢ ﺷﺒﺎﻫﺖ ﺩﺍﺭﻧﺪ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻳﻢ ﻭ ﺑﺎ ﻧﮕﺎﻩ ﮐﻠﯽ ﺑﻪ ﻫﻤﻪ ﺁﻧﻬﺎ، ﺧﺎﺻﻴﺘﯽ ﺩﺭ ﺁﻧﻬﺎ ﺑﻪ ﮐﻞ ﺍﻋﻀﺎﯼ ﺁﻥ ﮔﺮﻭﻩ ﺗﻌﻤﻴﻢ ﻣﯽ‌ﺩﻫﻴﻢ.

ﺩﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ، ﻳﮏ ﻗﺎﻧﻮﻥ ﻋﻤﻮﻣﯽ ﺍﺯ ﻣﺸﺎﻫﺪﺍﺕ ﺧﺎﺻﯽ ﺩﺭﺑﺎﺭﻩ ﻭﻳﮋﮔﯽ ﮐﻠﯽ ﭼﻴﺰﻫﺎ ﺩﺭ ﻃﺒﻴﻌﺖ ﺍﺳﺘﻨﺘﺎﺝ ﻣﯽ‌ﺷﻮﺩ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ، ﺍﺯ ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩ ﻣﺸﺎﻫﺪﺍﺕ ﺩﺭﺑﺎﺭﻩ ﺍﻳﻨﮑﻪ ﭘﺴﺘﺎﻧﺪﺍﺭﻥ ﺑﭽﻪ ﻣﯽ​ﺯﺍﻳﻨﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺳﺘﻨﺘﺎﺝ ﺷﻮﺩ ﮐﻪ ﻫﻤﻪ ﭘﺴﺘﺎﻧﺪﺍﺭﺍﻥ ﺍﻳﻦ ﮔﻮﻧﻪ ﻫﺴﺘﻨﺪ.

ﭼﻨﻴﻦ ﺍﺳﺘﺪﻻﻟﯽ ﻫﻴﭻ ﻭﻗﺖ ﻣﻌﺘﺒﺮ ﻧﻴﺴﺖ، ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﮐﻪ ﺍﻋﺘﺒﺎﺭ ﺭﺍ ﺑﺮﺍﯼ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻳﻢ. ﺑﻪ ﺯﺑﺎﻥ ﺑﻬﺘﺮ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺑﺮ ﺧﻼﻑ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺍﺻﻼ ﺍﺩﻋﺎﯼ ﻣﻌﺘﺒﺮ ﺑﻮﺩﻥ ﻧﺪﺍﺭﺩ ﻭ ﺧﺎﺻﻴﺖ ﺍﻋﺘﺒﺎﺭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻮﺩ. ﺩﺭ ﻣﻘﺎﺑﻞ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﺳﺘﺪﻻﻝ ﺧﻮﺑﯽ ﺑﺎﺷﺪ، ﻧﻴﺎﺯ ﺑﻪ ﻗﻮﺕ ﻳﺎ ﺍﺳﺘﺤﮑﺎﻡ ﺩﺍﺭﺩ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺭﺍﺑﻄﻪ ﺑﻴﻦ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﻧﺘﻴﺠﻪ ﺁﻧﻘﺪﺭ ﻗﻮﯼ ﻭ ﻣﺤﮑﻢ ﺑﺎﺷﺪ ﮐﻪ ﺑﻪ ﺭﺍﺣﺘﯽ ﺑﺘﻮﺍﻥ ﺣﮑﻢ ﮐﻠﯽ ﺭﺍ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺟﺰﺋﯽ ﺍﺳﺘﺨﺮﺍﺝ ﮐﺮﺩ.

 ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻧﺘﻴﺠﻪ ﺿﺮﻭﺭﺗﺎ ﺍﺯ ﻣﻘﺪﻣﻪ​‌ﻫﺎ ﺍﺳﺘﻨﺘﺎﺝ ﻧﻤﯽ‌​ﺷﻮﺩ ﻭ ﺑﻪ ﺯﺑﺎﻥ ﺩﻳﮕﺮ، ﻣﻮﺍﺭﺩﯼ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﮐﻪ ﻣﻘﺪﻣﻪ​‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﺍﻣﺎ ﻧﺘﻴﺠﻪ ﺩﺭﺳﺖ ﻧﺒﺎﺷﺪ. ﻣﺎﻧﻨﺪ ﭘﻴﺪﺍ ﺷﺪﻥ ﻣﺜﺎﻝ ﻧﻘﺾ ﺩﺭ ﻫﻤﻴﻦ ﻣﺜﺎﻝ ﺩﺭ ﻣﻮﺭﺩ ﭘﻼﺗﻴﭙﻮﺱ، ﭘﺴﺘﺎﻧﺪﺍﺭﯼ ﮐﻪ ﺗﺨﻢ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺍﻳﻦ ﺑﻪ ﺁﻥ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻫﻤﻴﺸﻪ ﻓﺮﺍﺗﺮ ﺍﺯ ﻣﻘﺪﻣﻪ​‌ﻫﺎﻳﺶ ﻣﯽ‌​ﺭﻭﺩ، ﻭ ﻫﻴﭻ ﻭﻗﺖ ﻧﺘﻴﺠﻪ ﻣﺸﺨﺺ ﻭ ﺍﺯ ﭘﻴﺶ ﺗﻌﻴﻴﻦ ﺷﺪﻩ‌ﺍﯼ ﺭﺍ ﻧﺘﻴﺠﻪ ﻧﻤﯽ‌​ﺩﻫﺪ. ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻣﻘﺪﻣﻪ​‌ﻫﺎ ﺻﺮﻓﺎ ﻧﺘﻴﺠﻪ ﺭﺍ ﺣﻤﺎﻳﺖ ﻣﯽ​‌ﮐﻨﻨﺪ ﻳﺎ ﺁﻥ ﺭﺍ ﺗﺎ ﺩﺭﺟﻪ‌ﺍﯼ ﻣﺤﺘﻤﻞ ﻣﯽ‌ﺳﺎﺯﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﺮﭼﻪ ﺍﺣﺘﻤﺎﻝ ﺣﻤﺎﻳﺖ ﻣﻘﺪﻣﻪ‌‌ﻫﺎ ﺍﺯ ﻧﺘﻴﺠﻪ ﺩﺭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ، ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﻣﺤﮑﻢ‌ﺗﺮ ﺍﺳﺖ.

 

ﺗﻌﻤﻴﻢ
ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﮔﻔﺘﻴﻢ ﺍﺳﺘﺪﻻﻝ​‌ﻫﺎﯼ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺗﻌﻤﻴﻢ‌​ﺳﺎﺯﯼ​‌ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﺍﺯ ﺍﻧﻮﺍﻉ ﻣﺨﺘﻠﻒ، ﺍﺯ ﺧﺎﺹ ﺑﻪ ﻋﺎﻡ، ﺍﺯ ﻣﺸﺎﻫﺪﻩ ﺷﺪﻩ ﺑﻪ ﻣﺸﺎﻫﺪﻩ ﻧﺸﺪﻩ، ﺍﺯ ﮔﺬﺷﺘﻪ ﺑﻪ ﺁﻳﻨﺪﻩ ﻳﺎ ﺣﺎﻟﺖ ﻓﻌﻠﯽ ﺑﻪ ﺣﺎﻟﺖ ﺁﻳﻨﺪﻩ. ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻫﻤﻪ ﺟﺎ ﺣﺎﺿﺮ ﻭ ﺿﺮﻭﺭﯼ ﺍﺳﺖ. ﻏﻴﺮﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺑﺘﻮﺍﻧﻴﻢ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ‌​ﻣﺎﻥ ﺭﺍ ﺑﺪﻭﻥ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻟﮕﻮﻫﺎﯼ ﻣﺸﺎﻫﺪﻩ ﺷﺪﻩ ﻭ ﺍﻣﺘﺪﺍﺩ ﺁﻧﻬﺎ ﺩﺭ ﮔﺬﺷﺘﻪ ﻭ ﺁﻳﻨﺪﻩ ﺳﭙﺮﯼ ﮐﻨﻴﻢ ﻭ ﭘﻴﺶ‌​ﺑﻴﻨﯽ ﮐﻨﻴﻢ ﮐﻪ ﭼﻴﺰﻫﺎ ﺩﺭ ﺁﻳﻨﺪﻩ ﭼﮕﻮﻧﻪ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ.

ﺩﺭ ﺣﻘﻴﻘﺖ، ﻗﻮﺍﻧﻴﻦ ﻭ ﻧﻈﺮﻳﻪ‌ﻫﺎﯼ ﻋﻠﻤﯽ ﺍﻏﻠﺐ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ‌ﻫﺎﻳﯽ ﺩﺭﺧﺸﺎﻥ ﺍﺯ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ ﻣﯽ‌​ﺷﻮﻧﺪ، ﺍﻣﺎ ﺁﻳﺎ ﭼﻨﻴﻦ ﺍﺳﺘﻨﺘﺎﺝ​‌ﻫﺎﻳﯽ ﻣﻮﺟﻪ ﻫﺴﺘﻨﺪ؟ ﺩﻳﻮﻳﺪ ﻫﻴﻮﻡ ﻓﻴﻠﺴﻮﻑ ﺍﺳﮑﺎﺗﻠﻨﺪﯼ ﻓﮑﺮ ﻣﯽ​‌ﮐﺮﺩ ﮐﻪ ﻧﻪ، ﻳﻌﻨﯽ ﻫﻴﭻ ﻣﺒﻨﺎﯼ ﻋﻘﻼﻧﯽ ﺑﺮﺍﯼ ﺗﻮﺳﻞ ﺑﻪ ﺍﺳﺘﻘﺮﺍ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺍﺯ ﻧﻈﺮ ﻫﻴﻮﻡ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ، ﺍﻳﻦ ﺑﺎﻭﺭ ﺭﺍ ﻓﺮﺽ ﻣﯽ​‌ﮔﻴﺮﺩ ﮐﻪ ﻃﺒﻴﻌﺖ ﻫﻤﮕﻦ ﺍﺳﺖ، ﮐﻪ ﺑﺮ ﺍﺳﺎﺱ ﺁﻥ ﺁﻳﻨﺪﻩ ﺷﺒﻴﻪ ﮔﺬﺷﺘﻪ ﺍﺳﺖ. ﺍﻣﺎ ﺧﻮﺩ ﺍﻳﻦ ﻓﺮﺽ ﺑﻪ ﺟﺰ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﭼﻪ ﻣﺒﻨﺎﻳﯽ ﺩﺍﺭﺩ؟ ﻭ ﺍﮔﺮ ﻫﻤﮕﻦ ﺑﻮﺩﻥ ﻃﺒﻴﻌﺖ ﺗﻨﻬﺎ ﺑﺮ ﺍﻳﻦ ﺍﺳﺎﺱ ﺗﻮﺟﻴﻪ ﺷﻮﺩ، ﺩﻳﮕﺮ ﺑﺪﻭﻥ ﺁﻧﮑﻪ ﺑﻪ ﺩﺍﻡ ﺍﺳﺘﺪﻻﻝ ﺩﻭﺭﯼ ﺑﻴﻔﺘﻴﻢ ﻧﻤﯽ‌​ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺍﺳﺘﻘﺮﺍ ﺩﻓﺎﻉ ﮐﻨﻴﻢ. ﺑﻌﻀﯽ ﻫﻢ ﺗﻼﺵ ﮐﺮﺩﻩ​‌ﺍﻧﺪ ﮐﻪ ﺍﺳﺘﻘﺮﺍ ﺭﺍ ﺑﺮ ﺍﻳﻦ ﺍﺳﺎﺱ ﺗﻮﺟﻴﻪ ﮐﻨﻨﺪ ﮐﻪ ﺩﺭ ﮔﺬﺷﺘﻪ ﺟﻮﺍﺏ ﺩﺍﺩﻩ ﻭ ﮐﺎﺭ ﮐﺮﺩﻩ ﺍﺳﺖ ﮐﻪ ﻫﻤﭽﻨﺎﻥ ﺧﻮﺩ ﺍﻳﻦ ﺍﺩﻋﺎ ﺑﺮ ﻣﺒﻨﺎﯼ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﻧﻤﯽ‌​ﺗﻮﺍﻧﺪ ﺍﺯ ﺯﻣﻴﻦ ﺑﻠﻨﺪ ﺷﻮﺩ.

ﺍﺯ ﻧﻈﺮ ﺧﻮﺩ ﻫﻴﻮﻡ ﻣﺎ ﭼﺎﺭﻩ‌​ﺍﯼ ﺟﺰ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﻧﺪﺍﺭﻳﻢ ﻭ ﺍﺗﻔﺎﻗﺎ ﻫﻴﻮﻡ ﭘﻴﺸﻨﻬﺎﺩ ﻧﻤﯽ​‌ﮐﻨﺪ ﮐﻪ ﻧﺒﺎﻳﺪ ﺑﻪ ﺍﻳﻦ ﺭﻭﺵ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ. ﺍﻣﺎ ﺍﺻﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻪ ﺁﮔﺎﻩ ﺑﺎﺷﻴﻢ ﮐﻪ ﺍﻳﻦ ﻋﻤﻞ ﻣﺎ ﺑﺮ ﺍﺳﺎﺱ ﻋﺎﺩﺕ ﻭ ﺳﻨﺖ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﻃﻮﺭ ﻋﻘﻼﻧﯽ ﻭ ﻣﻨﻄﻘﯽ ﻣﻮﺟﻪ ﻧﻴﺴﺖ. ﻣﺴﺌﻠﻪ ﺍﺳﺘﻘﺮﺍ ﮐﻪ ﻫﻴﻮﻡ ﺍﺯ ﺧﻮﺩ ﺑﻪ ﺟﺎ ﮔﺬﺍﺷﺘﻪ ﺣﺘﯽ ﺍﻣﺮﻭﺯ ﻧﻴﺰ ﻓﻼﺳﻔﻪ ﺭﺍ ﻣﺸﻐﻮﻝ ﮐﺮﺩﻩ ﺍﺳﺖ.
