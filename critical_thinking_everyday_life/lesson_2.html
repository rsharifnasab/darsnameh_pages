ﺑﺮﺍﯼ ﻣﺘﻘﺎﻋﺪ ﮐﺮﺩﻥ ﮐﺴﯽ ﺑﻪ ﭼﻴﺰﯼ ﮐﻪ ﺑﺪﺍﻥ ﻣﻌﺘﻘﺪﻳﻢ ﺭﺍﻩ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻓﺮﺽ ﮐﻨﻴﻢ ﺑﺎ ﮐﺴﯽ ﺩﺭﺑﺎﺭﻩ ﺭﻧﮓ ﻣﺎﺳﺖ ﺍﺧﺘﻼﻑ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ. ﻣﺎ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﮐﻪ ﻣﺎﺳﺖ ﺳﻔﻴﺪ ﺍﺳﺖ ﻭ ﺩﻭﺳﺘﻤﺎﻥ ﺑﺎﻭﺭ ﺩﺍﺭﺩ ﮐﻪ ﻧﺨﻴﺮ ﻣﺎﺳﺖ ﺳﻴﺎﻩ ﺍﺳﺖ! ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﻣﺎ ﺑﻪ ﺩﻧﺒﺎﻝ ﻗﺎﻧﻊ ﮐﺮﺩﻥ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺑﺮﺍﯼ ﻗﺒﻮﻝ ﺍﺩﻋﺎ ﻭ ﺑﺎﻭﺭ ﺧﻮﺩﻣﺎﻥ ﻫﺴﺘﻴﻢ. ﻳﮑﯽ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﻗﺎﻧﻊ‌ ﮐﺮﺩﻥ ﺩﻭﺳﺘﻤﺎﻥ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺳﺖ ﺍﻭ ﺭﺍ ﺑﮕﻴﺮﻳﻢ ﻭ ﺑﻪ ﻧﺰﺩﻳﮏ‌ﺗﺮﻳﻦ ﻣﺎﺳﺖ‌ﺑﻨﺪﯼ ﻣﺤﻞ ﺑﺮﻭﻳﻢ ﻭ ﻳﮏ ﻇﺮﻑ ﻣﺎﺳﺖ ﺑﺨﺮﻳﻢ ﻭ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺩﻫﻴﻢ، ﺗﺎ ﻗﺒﻮﻝ ﮐﻨﺪ ﮐﻪ ﻣﺎﺳﺖ ﺳﻔﻴﺪ ﺍﺳﺖ. ﺭﻭﺵ ﺩﻳﮕﺮ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﺩﻋﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﭼﻨﺎﻥ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺴﯽ ﺑﻴﺎﻥ ﮐﻨﻴﻢ ﮐﻪ ﺍﻭ ﻣﺎ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺮﺟﻊ ﺻﺎﺣﺐ‌ﻧﻈﺮ ﺩﺭ ﺯﻣﻴﻨﻪ ﻣﻮﺭﺩ ﺑﺤﺚ،‌ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺭﻧﮓ ﻣﺎﺳﺖ ﺍﺳﺖ، ﺑﭙﺬﻳﺮﺩ ﻭ ﻋﻘﻴﺪﻩ‌ﺍﺵ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﺪ. ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺖ ﻫﻴﭻ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ ﺩﻭ ﺭﻭﺵ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﮐﻪ ﻣﺎ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﻳﻢ ﺑﺮﺳﺪ. ﻣﯽ‌ﺩﺍﻧﻴﻢ ﮐﻪ ﻫﻤﻪ ﺁﺩﻡ‌ﻫﺎ ﻣﺜﻞ ﺧﻮﺩﻣﺎﻥ ﺳﺮﺳﺨﺖ ﻫﺴﺘﻨﺪ ﻭ ﺑﻪ ﺭﺍﺣﺘﯽ ﺗﻐﻴﻴﺮ ﻋﻘﻴﺪﻩ ﻧﻤﯽ‌ﺩﻫﻨﺪ. ﺩﺭ ﻣﻮﺭﺩ ﺍﻳﻦ ﻣﺜﺎﻝ ﺧﺎﺹ ﺩﻭﺳﺘﻤﺎﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎ ﺗﻮﺳﻞ ﺑﻪ ﺑﻬﺎﻧﻪ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﺍﻳﻨﮑﻪ ﻣﺎ ﺧﻮﺩﻣﺎﻥ ﺭﻧﮓ ﺳﻔﻴﺪ ﺩﺍﺧﻞ ﺳﻄﻞ ﻣﺎﺳﺖ ﺭﻳﺨﺘﻪ‌ﺍﻳﻢ، ﺍﺯ ﻗﺒﻮﻝ ﺍﺩﻋﺎﯼ ﻣﺎ ﺳﺮ ﺑﺎﺯ ﺯﻧﺪ.

 

ﺗﻌﺮﻳﻒ ﺍﺳﺘﺪﻻﻝ
ﺍﻣﺎ ﺭﺍﻩ ﺩﻳﮕﺮ ﻗﺎﻧﻊ ﮐﺮﺩﻥ ﺍﻓﺮﺍﺩ، ﻣﺒﺎﺣﺜﻪ ﻭ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺖ. ﻳﻌﻨﯽ ﺑﺮﺍﯼ ﺩﻭﺳﺘﻤﺎﻥ ﺩﻟﻴﻞ ﺑﻴﺎﻭﺭﻳﻢ ﺑﻪ ﺍﻭ ﭼﻴﺰﻫﺎﻳﯽ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺍﻭ ﺭﺍ ﺩﺭ ﺑﺎﻭﺭﻫﺎﯼ ﺧﻮﺩﻣﺎﻥ ﺷﺮﻳﮏ ﮐﻨﻴﻢ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺨﻮﺍﻫﻴﻢ ﺍﻳﻦ ﮔﻮﻧﻪ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ ﮐﻪ ﻣﺎﺳﺖ ﻧﻮﻋﯽ ﻓﺮﺁﻭﺭﺩﻩ ﻟﺒﻨﯽ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺷﻴﺮ ﺗﻬﻴﻪ ﺷﺪﻩ ﺍﺳﺖ. ﺷﻴﺮ ﺳﻔﻴﺪ ﺍﺳﺖ  ﻭ ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﺎﺳﺖ ﻧﻴﺰ ﮐﻪ ﺍﺯ ﺁﻥ ﻣﺸﺘﻖ ﻣﯽ‌ﺷﻮﺩ ﺳﻔﻴﺪ ﺍﺳﺖ.

ﻫﻤﻪ ﺍﻳﻦ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﻣﺘﻘﺎﻋﺪ ﮐﺮﺩﻥ ﺩﻭﺳﺘﻤﺎﻥ ﺑﻪ ﺍﻭ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﺍﺳﺘﺪﻻﻝ ﻧﺎﻡ ﺩﺍﺭﻧﺪ.

ﺍﺳﺘﺪﻻﻝ ﺑﺎ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﻳﯽ ﺁﻏﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﻣﻴﺪﻭﺍﺭﻳﻢ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺩﺭ ﺑﺤﺚ ﺁﻧﻬﺎ ﺭﺍ ﻗﺒﻮﻝ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺑﻪ ﺩﻧﺒﺎﻝ ﺁﻥ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﻣﺎ ﻣﯽ‌ﺁﻳﺪ ﮐﻪ ﺩﺭ ﻭﺍﻗﻊ ﻫﻤﺎﻥ ﺍﺩﻋﺎﯼ ﺍﻭﻟﻴﻪ‌ﻣﺎﻥ ﺍﺳﺖ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﻪ ﻣﺨﺎﻃﺐ ﺁﻥ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﺩﻧﺒﺎﻝ ﺁﻥ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﭙﺬﻳﺮﺩ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺳﺎﺧﺘﺎﺭ ﺑﻨﻴﺎﺩﯼ ﻫﺮ ﺍﺳﺘﺪﻻﻝ ﻳﺎﻓﺘﻦ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺁﻧﻬﺎ ﺭﺍ ﻗﺒﻮﻝ ﺩﺍﺭﻧﺪ ﻭ ﺳﭙﺲ ﮔﺎﻡ ﺑﻪ ﮔﺎﻡ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﻣﯽ‌ﺍﻧﺠﺎﻣﺪ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺁﻥ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﺪ.

ﺍﺳﺘﺪﻻﻝ ﻳﮏ ﻓﺮﺁﻳﻨﺪ ﺭﻭﺷﻨﮕﺮﺍﻧﻪ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺁﻥ ﻧﻪ ﺗﻨﻬﺎ ﺑﺎﻳﺪ ﺩﻳﺪﮔﺎﻩ ﺧﻮﺩ ﺭﺍ ﺗﻮﺿﻴﺢ ﺩﻫﻴﻢ، ﺑﻠﮑﻪ ﺑﺎﻳﺪ ﺩﻻﻳﻞ ﻣﺸﺨﺼﯽ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﺁﻥ ﺍﺭﺍﺋﻪ ﮐﻨﻴﻢ. ﺍﺳﺘﺪﻻﻝ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺑﻪ ﻫﻢ ﭘﻴﻮﺳﺘﻪ ﺍﺯ ﮔﺰﺍﺭﻩ‌ﻫﺎ ﻭ ﺟﻤﻠﻪ‌ﻫﺎﯼ ﺑﺎ ﻣﻨﻈﻮﺭ ﻣﺸﺨﺺ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻳﮏ ﻧﺘﻴﺠﻪ ﻣﻌﻴﻦ ﺍﺳﺖ. ﭘﺲ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺍﺯ ﺟﻤﻼﺕ ﺯﺑﺎﻧﯽ ﮐﻪ ﺑﺎ ﺁﻥ ﺣﺮﻑ ﻣﯽ‌ﺯﻧﻴﻢ ﻭ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﻴﻢ ﺳﺎﺧﺘﻪ ﺷﺪﻩ‌ﺍﻧﺪ.

ﺧﺐ ﺍﻳﻦ ﺗﻌﺮﻳﻒ ﻳﮏ ﺗﻘﺎﺑﻞ ﺧﻮﺏ ﺭﺍ ﺑﻪ ﻣﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺯﻳﺮﺍ ﻣﺠﻤﻮﻋﻪ ﺟﻤﻼﺕ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﺷﻤﺎﺭ ﻧﻤﯽ‌ﺭﻭﻧﺪ ﺯﻳﺮﺍ ﺑﻪ ﻣﻨﻈﻮﺭ ﺳﺎﺧﺘﻦ ﻳﮏ ﻧﺘﻴﺠﻪ ﻣﺸﺨﺺ ﺟﻤﻊ ﻧﺸﺪﻩ‌ﺍﻧﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻳﮏ ﺭﻣﺎﻥ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ ﮐﻪ ﺣﺎﻭﯼ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﺟﻤﻼﺕ ﺍﺳﺖ ﮐﻪ ﺗﻮﺿﻴﺢ ﻣﯽ‌ﺩﻫﺪ ﺷﺨﺼﻴﺖ‌ﻫﺎﯼ ﺩﺍﺳﺘﺎﻥ ﭼﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﭼﻪ ﺍﺗﻔﺎﻗﺎﺗﯽ ﺩﺭ ﺟﺮﻳﺎﻥ ﺍﺳﺖ. ﺍﻣﺎ ﺍﻳﻦ ﻣﺠﻤﻮﻋﻪ ﺍﺯ ﺟﻤﻠﻪ‌ﻫﺎ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﺷﻤﺎﺭ ﻧﻤﯽ‌ﺭﻭﺩ،‌ ﺯﻳﺮﺍ ﻫﺪﻑ ﺁﻥ ﺭﺳﻴﺪﻥ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﻣﺸﺨﺺ ﻧﻴﺴﺖ. ﻳﺎ ﻣﺜﻼ ﻓﺮﻫﻨﮓ ﻟﻐﺎﺕ ﮐﻪ ﺣﺎﻭﯼ ﺗﻌﺎﺭﻳﻒ ﺯﻳﺎﺩﯼ ﺍﺳﺖ ﻭ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﺟﻤﻠﻪ‌ﻫﺎ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﺪ ﺍﺳﺘﺪﻻﻝ ﻧﻴﺴﺖ ﺯﻳﺮﺍ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺭﺍ ﺩﻧﺒﺎﻝ ﻧﻤﯽ‌ﮐﻨﺪ.

ﺩﺭ ﺣﻘﻴﻘﺖ ﺭﻣﺎﻥ‌ﻫﺎ ﻳﺎ ﻓﺮﻫﻨﮓ‌ﻧﺎﻣﻪ‌ﻫﺎ ﺟﻤﻼﺕ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﻧﻈﻢ ﺩﺍﺩﻩ‌ﺍﻧﺪ. ﻣﺜﻼ ﺑﻪ ﺗﺮﺗﻴﺐ ﺯﻣﺎﻧﯽ ﻳﺎ ﺍﻟﻔﺒﺎﻳﯽ، ﺩﺭ ﺣﺎﻟﯽ‌ﮐﻪ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺟﻤﻼﺕ ﺭﺍ ﺑﻪ ﺷﮑﻞ ﺳﺎﺧﺘﺎﺭﯼ ﻣﺸﺨﺺ ﻣﺮﺗﺐ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﻧﺸﺎﻥ‌ﺩﻫﻨﺪﻩ ﺗﺮﺗﻴﺐ ﻭ ﻧﻈﻢ ﺩﻟﻴﻞ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺳﺎﺧﺘﻦ ﻧﺘﻴﺠﻪ ﻧﻬﺎﻳﯽ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﺩﻧﺒﺎﻝ ﻫﻢ ﻣﯽ‌ﺁﻳﻨﺪ.

ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﻤﻮﻧﻪ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﻓﮑﺮ ﮐﻨﻴﻢ ﮐﻪ ﺁﻳﺎ ﺗﻘﻮﻳﻢ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺖ؟ ﻣﻮﺍﺭﺩ ﻧﻤﺎﻳﺶ ﺩﺍﺩﻩ ﺷﺪﻩ ﺭﻭﯼ ﺗﻘﻮﻳﻢ ﻳﺎ ﺳﺮﺭﺳﻴﺪ ﺷﺨﺼﯽ ﻣﺎ ﺑﻪ ﺻﻮﺭﺕ ﺯﻣﺎﻥ‌ﺑﻨﺪﯼ ﺷﺪﻩ ﻣﻨﻈﻢ ﺷﺪﻩ‌ﺍﻧﺪ ﺍﻣﺎ ﺗﺮﺗﻴﺐ ﻣﺸﺨﺼﯽ ﺍﺯ ﺩﻻﻳﻞ ﻧﻴﺴﺘﻨﺪ. ﺗﻘﻮﻳﻢ ﺑﻪ ﻣﻨﻈﻮﺭ ﺍﺭﺍﺋﻪ ﺩﻻﻳﻠﯽ ﺑﺮﺍﯼ ﻳﮏ ﻧﺘﻴﺠﻪ ﻣﺸﺨﺺ ﻣﺮﺗﺐ ﻧﺸﺪﻩ ﺍﺳﺖ. ﻣﺜﻼ ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﻢ ﮐﻪ ۲۹ ﺍﺳﻔﻨﺪ ﻧﺘﻴﺠﻪ‌ﯼ ﻳﮏ ﺳﺎﻝ ﻧﻴﺴﺖ. ﭘﺎﻳﺎﻥ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﺟﻤﻼﺕ ﻳﺎ ﻣﻮﺍﺭﺩ ﻳﺎ ﻭﻗﺎﻳﻊ ﺑﺎ ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﻣﻨﻄﻘﯽ ﻧﺘﻴﺠﻪ ﻧﺎﻡ ﺩﺍﺭﺩ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ.

ﭘﺲ ﺍﺳﺘﺪﻻﻝ ﻳﮏ ﺣﺮﮐﺖ ﻋﻘﻼﻧﯽ ﺍﺳﺖ ﺍﺯ ﻣﺒﺎﻧﯽ ﭘﺬﻳﺮﻓﺘﻪ ﺷﺪﻩ، ﺑﻪ ﻧﺎﻡ «ﻣﻘﺪﻣﻪ»، ﺑﻪ ﻧﻘﻄﻪ‌ﺍﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺍﺛﺒﺎﺕ ﮐﻨﻴﻢ ﻳﺎ ﺩﺭﺳﺘﯽ‌ﺍﺵ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﻴﻢ ﮐﻪ «ﻧﺘﻴﺠﻪ» ﻧﺎﻡ ﺩﺍﺭﺩ. ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺟﻤﻠﻪ‌ﻫﺎ ﻳﺎ ﮔﺰﺍﺭﻩ‌ﻫﺎﻳﯽ ﭘﺎﻳﻪ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺎﻳﺪ ﺁﻧﻬﺎ ﺭﺍ،‌ ﺣﺪﺍﻗﻞ ﺑﻪ ﻃﻮﺭ ﻣﻮﻗﺖ، ﻗﺒﻮﻝ ﮐﻨﻴﻢ ﺗﺎ ﺍﺳﺘﺪﻻﻝ ﺑﺘﻮﺍﻧﺪ ﺷﮑﻞ ﺑﮕﻴﺮﺩ. ﺧﻮﺩ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺑﻪ ﺩﺳﺖ ﺁﻣﺪﻩ ﺑﺎﺷﻨﺪ. ﻣﺜﻼ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺮ ﻣﺒﻨﺎﯼ ﻳﮏ ﺍﺻﻞ ﻣﻨﻄﻘﯽ ﻳﺎ ﻳﮏ ﺷﺎﻫﺪ ﺗﺠﺮﺑﯽ ﺑﺎﺷﻨﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﻧﺘﻴﺠﻪ‌ﯼ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﺑﺎﺷﻨﺪ.

ﺣﺮﮐﺖ ﺍﺯ ﻣﻘﺪﻣﻪ ﺑﻪ ﻧﺘﻴﺠﻪ ﺭﺍ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﻳﺎ «ﺍﺳﺘﻨﺘﺎﺝ» ﻣﯽ‌ﮔﻮﻳﻨﺪ ﮐﻪ ﻗﺪﺭﺕ ﺁﻥ ﺍﺳﺘﺤﮑﺎﻡ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﻣﻌﻴﻦ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺩﺭ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﺎ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﺳﻔﻴﺪ ﺑﻮﺩﻥ ﻣﺎﺳﺖ ﺑﺮﺍﯼ ﺩﻭﺳﺘﻤﺎﻥ ﻣﯽ‌ﺁﻭﺭﻳﻢ، ﻫﻤﮕﯽ ﺑﻪ ﻳﮏ ﺍﻧﺪﺍﺯﻩ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ ﻭ ﻗﻮﯼ ﻧﻴﺴﺘﻨﺪ. ﺗﺸﺨﻴﺺ ﺍﺳﺘﻨﺘﺎﺝ‌ﻫﺎﯼ ﺧﻮﺏ ﺍﺯ ﺑﺪ ﻣﻬﻤﺘﺮﻳﻦ ﻭﻇﻴﻔﻪ ﻣﻨﻄﻖ ﺍﺳﺖ. ﻣﻨﻄﻖ ﻋﻠﻢ ﺗﺤﻠﻴﻞ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻭ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺻﻮﻝ ﻭ ﻣﺒﺎﻧﯽ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺍﺳﺘﻨﺘﺎﺝ‌ﻫﺎﯼ ﺻﺤﻴﺢ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺮ ﺍﺳﺎﺱ ﺁﻧﻬﺎ ﺷﮑﻞ ﺑﮕﻴﺮﻧﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﻨﻄﻖ ﭼﻨﺪﺍﻥ ﻧﮕﺮﺍﻥ ﻣﺤﺘﻮﺍﯼ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻧﻴﺴﺖ، ﺑﻠﮑﻪ ﺑﺎ ﺳﺎﺧﺘﺎﺭ ﻭ ﺷﮑﻞ ﮐﻠﯽ ﺁﻧﻬﺎ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﺩ.

 

ﺑﺮﺍﯼ ﭼﻪ ﺍﺳﺘﺪﻻﻝ ﻣﯽ‌ﮐﻨﻴﻢ؟
ﻫﺪﻑ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻫﻤﻴﺸﻪ ﻫﻢ ﺭﺳﻴﺪﻥ ﺑﻪ ﻳﮏ ﻧﺘﻴﺠﻪ ﻧﻴﺴﺖ. ﮔﺎﻫﯽ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ ﮐﻪ ﺟﻮﺍﺏ ﺁﻥ ﺭﺍ ﺍﺯ ﻗﺒﻞ ﻣﯽ‌ﺩﺍﻧﻴﻢ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺩﺭ ﻣﻮﺭﺩ ﺍﺛﺒﺎﺕ ﻗﻀﺎﻳﺎﯼ ﻫﻨﺪﺳﯽ ﻣﺎﻧﻨﺪ ﻗﻀﻴﻪ ﻓﻴﺜﺎﻏﻮﺭﺙ، ﺍﺯ ﻗﺒﻞ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﮐﻪ ﻧﺘﻴﺠﻪ ﭼﻴﺴﺖ ﺍﻣﺎ ﺍﺳﺘﺪﻻﻟﯽ ﮐﻪ ﺩﺭ ﺭﺳﻴﺪﻥ ﺍﺯ ﻣﻘﺪﻣﺎﺕ ﺑﻪ ﻧﺘﻴﺠﻪ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﺷﻮﺩ، ﺑﻪ ﺷﮑﻞ ﻣﻨﻄﻘﯽ ﻭ ﮔﺎﻡ ﺑﻪ ﮔﺎﻡ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﭼﺮﺍ ﺍﻳﻦ ﻗﻀﻴﻪ ﺩﺭﺳﺖ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺗﻮﺿﻴﺤﯽ ﻣﻨﻄﻘﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻧﺘﻴﺠﻪ ﺭﺍ ﺑﻔﻬﻤﻴﻢ. ﭘﺲ ﮔﺎﻫﯽ ﺍﻭﻗﺎﺕ ﺍﺳﺘﺪﻻﻝ‌ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻳﮏ ﮔﺰﺍﺭﻩ ﺭﺍ ﺑﻔﻬﻤﻴﻢ ﻭ ﺗﻮﺿﻴﺢ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﭼﺮﺍ ﺁﻥ ﮔﺰﺍﺭﻩ ﻳﺎ ﺟﻤﻠﻪ ﺩﺭﺳﺖ ﺍﺳﺖ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺩﻳﺪﻳﻢ ﺍﺳﺘﺪﻻﻝ ﻟﺰﻭﻣﺎ ﻧﺒﺎﻳﺪ ﺩﺭ ﺑﺤﺚ‌ﻫﺎﯼ ﭘﻴﭽﻴﺪﻩ ﻭ ﻋﺠﻴﺐ ﻭ ﻏﺮﻳﺐ ﻓﻠﺴﻔﯽ ﺑﺎ ﺟﻤﻠﻪ‌ﺑﻨﺪﯼ‌ﻫﺎ ﻭ ﮐﻠﻤﻪ‌ﻫﺎﯼ ﻗﻠﻨﺒﻪ ﺳﻠﻨﺒﻪ ﻭ ﭘﺮ ﺍﺯ ﺍﺻﻄﻼﺣﺎﺕ ﻓﻨﯽ ﻇﺎﻫﺮ ﺷﻮﺩ. ﻫﻤﻪ ﻣﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻭ ﺑﺎ ﻫﺮ ﺍﺩﻋﺎﻳﻤﺎﻥ ﺍﺳﺘﺪﻻﻝ ﻣﯽ‌ﮐﻨﻴﻢ. ﺣﺎﻝ ﺍﻳﻨﮑﻪ ﺩﻻﻳﻞ‌ﻣﺎﻥ ﻃﻮﺭﯼ ﺑﺎﺷﻨﺪ ﮐﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻞ‌ﻣﺎﻥ ﺩﺭ ﺑﺤﺚ ﺭﺍﺿﯽ ﺷﻮﺩ ﻭ ﺍﺩﻋﺎﯼ ﻣﺎ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﺪ ﺑﻪ ﺍﻳﻦ ﺑﺮﻣﯽ‌ﮔﺮﺩﺩ ﮐﻪ ﭼﻘﺪﺭ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﻣﻌﺘﺒﺮ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﻫﻤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻨﺪ، ﺍﻣﺎ ﺍﻳﻨﮑﻪ ﺁﻳﺎ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﻣﺤﮑﻢ ﻭ ﻣﻌﺘﺒﺮ ﺍﺳﺖ ﻭ ﺁﻳﺎ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﮐﻪ ﺍﺯ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﺣﺎﺻﻞ ﻣﯽ‌ﺷﻮﺩ ﺩﺭﺳﺖ ﺍﺳﺖ ﻳﺎ ﻧﻪ، ﺑﺤﺚ ﺩﻳﮕﺮﯼ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺣﻮﺯﻩ‌ﻫﺎﯼ ﻣﻬﻢ ﻣﻨﻄﻖ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩ ﺗﺼﻮﺭ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺻﺮﻓﺎ ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﺑﺮﺍﺯ ﭘﻴﺶ‌ﺩﺍﻭﺭﯼ‌ﻫﺎ ﻭ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺩﺍﺭﻧﺪ. ﺷﺎﻳﺪ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺑﺴﻴﺎﺭﯼ ﻧﻴﺰ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻧﺎﺧﻮﺷﺎﻳﻨﺪ ﻭ ﺑﯽ‌ﻣﻮﺭﺩ ﻫﺴﺘﻨﺪ. ﺍﻣﺎ ﺍﮔﺮ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﻗﺒﻼ ﮔﻔﺘﻴﻢ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺍﺭﺍﺋﻪ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﺩﻻﻳﻞ ﻳﺎ ﺷﻮﺍﻫﺪ ﺩﺭ ﺣﻤﺎﻳﺖ ﺍﺯ ﻳﮏ ﻧﺘﻴﺠﻪ ﺗﻌﺮﻳﻒ ﮐﻨﻴﻢ، ﺁﻥ‌ﮔﺎﻩ ﺩﻳﮕﺮ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﻣﻌﻨﺎﯼ ﺑﻴﺎﻥ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎﯼ ﺷﺨﺼﯽ ﻭ ﻳﺎ ﻣﺸﺎﺟﺮﻩ ﻭ ﻣﺒﺎﺣﺜﻪ ﺑﺮ ﺳﺮ ﻫﻴﭻ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﻣﻌﻨﺎﯼ ﺗﻼﺵ ﺑﺮﺍﯼ ﺣﻤﺎﻳﺖ ﺍﺯ ﺩﻳﺪﮔﺎﻫﯽ ﻣﺸﺨﺺ ﻭ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺩﻟﻴﻞ ﺍﺳﺖ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﺍﺳﺘﺪﻻﻝ ﺑﯽ‌ﻣﻮﺭﺩ ﻭ ﻣﻬﻤﻞ ﻧﻴﺴﺖ، ﺑﻠﮑﻪ ﺧﻴﻠﯽ ﻫﻢ ﺿﺮﻭﺭﯼ ﻭ ﻻﺯﻡ ﺍﺳﺖ.

ﻻﺯﻡ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺭﺍﻫﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﺗﺸﺨﻴﺺ ﺍﻳﻨﮑﻪ ﮐﺪﺍﻡ ﺩﻳﺪﮔﺎﻩ ﺍﺯ ﺩﻳﮕﺮ ﻋﻘﺎﻳﺪ ﺑﻬﺘﺮ ﺍﺳﺖ. ﻣﺴﻠﻤﺎ ﻫﻤﻪ‌ﯼ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎ ﻳﮑﺴﺎﻥ ﻭ ﻣﻌﺎﺩﻝ ﻳﮑﺪﻳﮕﺮ ﻧﻴﺴﺘﻨﺪ. ﺑﻌﻀﯽ ﻧﺘﺎﻳﺞ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺗﻮﺳﻂ ﺩﻻﻳﻞ ﺧﻮﺏ ﺣﻤﺎﻳﺖ ﺷﻮﻧﺪ، ﺍﻣﺎ ﺍﻏﻠﺐ ﻣﺎ ﻧﻤﯽ‌ﺩﺍﻧﻴﻢ ﮐﻪ ﮐﺪﺍﻡ ﺑﻪ ﮐﺪﺍﻡ ﺍﺳﺖ. ﻣﺎ ﻧﻴﺎﺯ ﺩﺍﺭﻳﻢ ﮐﻪ ﺑﺮﺍﯼ ﻧﺘﺎﻳﺞ ﻣﺨﺘﻠﻒ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ ﻭ ﺁﻥ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺭﺍ ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﻴﻢ ﺗﺎ ﺑﻔﻬﻤﻴﻢ ﻭﺍﻗﻌﺎ ﭼﻘﺪﺭ ﻣﺤﮑﻢ ﻭ ﻗﺎﺑﻞ ﺍﻋﺘﻤﺎﺩ ﻫﺴﺘﻨﺪ.

 

ﺍﺳﺘﺪﻻﻝ ﻳﺎ ﭘﻴﺶ‌ﺩﺍﻭﺭﯼ؟
ﻳﮑﯽ ﺍﺯ ﻣﺸﮑﻼﺗﯽ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﻣﺒﺎﺣﺜﻪ‌ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻣﺎ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﺑﻪ ﺷﮑﻞ ﺗﻬﺎﺟﻤﯽ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ. ﻳﻌﻨﯽ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﻢ ﮐﻪ ﺑﺎ ﻧﺘﺎﻳﺞ ﺩﻟﺨﻮﺍﻩ ﺧﻮﺩ-ﺍﻣﻴﺎﻝ ﻳﺎ ﻋﻘﺎﻳﺪﻣﺎﻥ- ﺑﺤﺚ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﻢ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺑﺨﻮﺍﻫﻴﻢ ﺩﻻﻳﻠﯽ ﺩﺭ ﺣﻤﺎﻳﺖ ﺍﺯ ﺁﻧﻬﺎ ﺑﻴﺎﻭﺭﻳﻢ. ﺍﻳﻦ ﺭﻭﺵ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺑﻌﻀﯽ ﺍﺯ ﻣﻮﺍﺭﺩ ﺟﻮﺍﺏ ﺩﻫﺪ، ﺑﻪ ﺧﺼﻮﺹ ﻭﻗﺘﯽ ﮐﻪ ﺟﻮﺍﻥ‌ﺗﺮ ﻫﺴﺘﻴﻢ ﻭ ﺑﺎ ﺷﻮﺭ ﻭ ﻫﻴﺠﺎﻥ ﻭ ﺣﺮﺍﺭﺕ ﺩﺭ ﺑﺤﺚ‌ﻫﺎ ﺻﺮﻓﺎ ﺑﻪ ﺑﻴﺎﻥ ﻋﻘﺎﻳﺪ ﻭ ﻧﻈﺮﺍﺕ ﺷﺨﺼﯽ‌ﻣﺎﻥ ﺍﮐﺘﻔﺎ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺩﻳﮕﺮﺍﻥ ﻫﻢ ﻣﯽ‌ﭘﺬﻳﺮﻧﺪ. ﺧﺐ ﭼﻪ ﭼﻴﺰﯼ ﺍﺯ ﺍﻳﻦ ﺑﻬﺘﺮ؟

ﺍﻣﺎ ﺩﺭ ﻣﻘﺎﺑﻞ،‌ ﺍﺳﺘﺪﻻﻝ ﻭﺍﻗﻌﯽ ﻧﻴﺎﺯ ﺑﻪ ﺯﻣﺎﻥ ﻭ ﺗﻤﺮﻳﻦ ﺩﺍﺭﺩ. ﺁﻣﺎﺩﻩ ﻭ ﻣﺮﺗﺐ ﮐﺮﺩﻥ ﺩﻻﻳﻞ، ﻣﺘﻨﺎﺳﺐ ﮐﺮﺩﻥ ﻧﺘﺎﻳﺞ ﺑﺎ ﺷﻮﺍﻫﺪ ﻭﺍﻗﻌﯽ، ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺍﻧﺘﻘﺎﺩﻫﺎ ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﻮﺍﺭﺩ ﺩﻳﮕﺮ ﮐﻪ ﻧﻴﺎﺯﻣﻨﺪ ﻣﻬﺎﺭﺕ ﻭ ﺁﮔﺎﻫﯽ ﺍﺳﺖ. ﻣﺎ ﺑﺎﻳﺪ ﮐﻤﯽ ﺑﺰﺭﮒ ﺷﻮﻳﻢ! ﺑﺎﻳﺪ ﺍﻣﻴﺎﻝ ﻭ ﻋﻘﺎﻳﺪ ﺷﺨﺼﯽ‌ﻣﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﻣﺪﺗﯽ ﮐﻨﺎﺭ ﺑﮕﺬﺭﻳﻢ ﻭ ﻭﺍﻗﻌﺎ ﻓﮑﺮ ﮐﻨﻴﻢ.

ﻫﺮ ﭼﻘﺪﺭ ﺑﻴﺸﺘﺮ ﺗﻤﺮﻳﻦ ﮐﻨﻴﻢ ﻭ ﺑﺮﺍﯼ ﺑﺤﺚ ﮐﺮﺩﻥ ﻭﺍﻗﻌﺎ ﺑﻴﻨﺪﻳﺸﻴﻢ، ﺩﻟﻴﻞ ﺁﻭﺭﺩﻥ ﻭ ﻓﺮﺁﻳﻨﺪ ﺍﺳﺘﺪﻻﻝ ﺑﺮﺍﻳﻤﺎﻥ ﺟﺬﺍﺏ‌ﺗﺮ ﻣﯽ‌ﺷﻮﺩ. ﺫﻫﻦ‌ﻣﺎﻥ ﺍﻧﻌﻄﺎﻑ‌ﭘﺬﻳﺮﺗﺮ، ﺑﺎﺯﺗﺮ ﻭ ﮔﻮﺵ ﺑﻪ ﺯﻧﮓ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺧﻴﻠﯽ ﺯﻭﺩ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺗﺤﺴﻴﻦ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ ﮐﻪ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﭼﻘﺪﺭ ﻣﺎ ﺭﺍ ﻣﺘﺤﻮﻝ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﺍﺯ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺗﺎ ﺳﻴﺎﺳﺖ، ﻋﻠﻢ ﻭ ﻓﻠﺴﻔﻪ ﻭ ﺣﺘﯽ ﺩﻳﻦ ﻭ ﻣﺬﻫﺐ، ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺩﺍﺋﻤﺎ ﺑﻪ ﻣﺎ ﻋﺮﺿﻪ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﮔﺮ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﺭﺍ ﺗﻤﺮﻳﻦ ﮐﺮﺩﻩ ﺑﺎﺷﻴﻢ ﺍﻳﻦ ﺗﻮﺍﻧﺎﻳﯽ ﺭﺍ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ ﮐﻪ ﺍﺩﻋﺎﻫﺎﯼ ﺩﻳﮕﺮﺍﻥ، ﺭﺳﺎﻧﻪ‌ﻫﺎ، ﺩﻭﻟﺖ‌ﻫﺎ ﻭ ... ﺭﺍ ﺑﯽ ﭼﻮﻥ ﻭ ﭼﺮﺍ ﻧﭙﺬﻳﺮﻳﻢ ﻭ ﻣﺎ ﻧﻴﺰ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﺍﺭﺍﺋﻪ ﺩﻫﻴﻢ. ﺍﺳﺘﺪﻻﻝ ﺭﺍﻫﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺟﺎﯼ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﻴﺎﻥ ﺩﻳﺎﻟﻮﮒ ﻣﺴﺘﻤﺮ ﻭ ﺑﯽ‌ﭘﺎﻳﺎﻥ ﺯﻧﺪﮔﯽ ﭘﻴﺪﺍ ﮐﻨﻴﻢ.
