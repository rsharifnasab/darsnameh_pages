ﺩﺭﺱ ﺷﺸﻢ- ﺁﻳﺎ ﻣﯽ ﺗﻮﺍﻥ ﺩﺭ ﻫﺮ ﺑﺤﺜﯽ ﺑﺮﻧﺪﻩ ﺷﺪ؟

ﻣﺒﺎﺣﺜﻪ ﻭ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﻣﻌﻨﺎﯼ ﺗﻮ ﺩﻫﻨﯽ ﺯﺩﻥ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻧﻴﺴﺖ. ﺍﻳﻦ ﺑﻴﺸﺘﺮ ﮐﺎﺭ ﮐﺸﺘﯽ ﮐﭻ ﺍﺳﺖ ﺗﺎ ﻣﺒﺎﺣﺜﻪ ﻭ ﺍﺳﺘﺪﻻﻝ! ﺑﺎﻳﺪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺑﺤﺚ ﺑﻪ ﻣﻌﻨﺎﯼ ﺩﻋﻮﺍ ﻧﻴﺴﺖ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺻﺮﻓﺎ ﺑﺎ ﺩﺍﺩ ﺯﺩﻥ ﺑﺮ ﺳﺮ ﮐﺴﯽ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﺑﺤﺚ ﭘﻴﺮﻭﺯ ﺷﺪ. ﭘﺮﻳﺪﻥ ﻭﺳﻂ ﺣﺮﻑ ﺩﻳﮕﺮﺍﻥ ﻣﺎ ﺭﺍ ﺑﺮﻧﺪﻩ‌ﯼ ﺍﺳﺘﺪﻻﻝ ﻧﻤﯽ‌ﮐﻨﺪ. ﺍﻳﻨﮑﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺳﺎﮐﺖ ﺷﻮﺩ ﻳﺎ ﮐﺴﯽ ﺣﺮﻑ ﺍﻭ ﺭﺍ ﻧﺸﻨﻮﺩ ﺩﻟﻴﻞ ﻧﻤﯽ‌ﺷﻮﺩ ﮐﻪ ﺣﻖ ﺑﺎ ﻣﺎ ﺑﺎﺷﺪ. ﺍﺳﺘﺪﻻﻝ ﺍﻭ ﻫﻤﭽﻨﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺧﻮﺏ ﺑﺎﺷﺪ ﺣﺘﯽ ﺍﮔﺮ ﻣﺎ ﺍﻭ ﺭﺍ ﺳﺎﮐﺖ ﮐﺮﺩﻩ ﺑﺎﺷﻴﻢ. ﺍﮔﺮ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﻣﺎ ﺩﺭ ﺑﺤﺚ ﻧﺘﻮﺍﻧﺪ ﺣﺮﻑ ﺑﺰﻧﺪ ﻳﺎ ﺻﺪﺍﻳﺶ ﺷﻨﻴﺪﻩ ﻧﺸﻮﺩ، ﻣﺎ ﻭ ﺩﻳﮕﺮ ﻣﺨﺎﻃﺒﺎﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺍﺳﺘﺪﻻﻝ ﺍﻭ ﭼﻴﺰﯼ ﻳﺎﺩ ﺑﮕﻴﺮﻳﻢ ﻭ ﺍﻳﻦ ﺗﻨﻬﺎ ﺑﻪ ﺿﺮﺭ ﺧﻮﺩﻣﺎﻥ ﺍﺳﺖ.
 
ﺍﺫﻳﺖ ﻭ ﺁﺯﺍﺭ ﺩﻳﮕﺮﺍﻥ ﻫﻢ ﻣﺒﺎﺣﺜﻪ ﺑﻪ ﺷﻤﺎﺭ ﻧﻤﯽ‌ﺭﻭﺩ. ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﺩﺭ ﻣﺒﺎﺣﺜﻪ‌ﺍﯼ ﺑﺮﻧﺪﻩ ﺷﻮﻳﻢ، ﺻﺮﻓﺎ ﺑﺎ ﺍﻳﻦ ﺭﻭﺵ ﮐﻪ ﻃﺮﻑ ﺑﺤﺚ ﺧﻮﺩ ﺭﺍ ﺍﺣﻤﻖ ﺧﻄﺎﺏ ﮐﻨﻴﻢ. ﺑﺎ ﻧﺴﺒﺖ ﺩﺍﺩﻥ ﺻﻔﺎﺕ ﻧﺎﺧﻮﺷﺎﻳﻨﺪ ﺑﻪ ﺩﻭﺳﺘﺎﻧﻤﺎﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﺩﺭ ﺑﺤﺚ‌ﻫﺎ ﺑﺮﻧﺪﻩ ﺷﻮﻳﻢ ﻭ ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﻭﺳﺘﯽ‌ﻫﺎﻳﻤﺎﻥ ﻫﻢ ﺧﺪﺷﻪ‌ﺩﺍﺭ ﺷﻮﺩ.
 
ﺷﮑﺎﻳﺖ ﮐﺮﺩﻥ، ﺍﻳﺮﺍﺩ ﮔﺮﻓﺘﻦ ﻭ ﺑﻬﺎﻧﻪ‌ﮔﻴﺮﯼ ﻧﻴﺰ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻣﺎ ﺭﺍ ﺩﺭ ﺑﺤﺚ ﭘﻴﺮﻭﺯ ﮐﻨﺪ. ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﻋﺬﺭ ﻭ ﺑﻬﺎﻧﻪ ﺁﻭﺭﺩﻥ ﻫﻢ ﺍﺳﺘﺪﻻﻝ ﻧﻴﺴﺖ. 
 
ﺭﺩ ﮐﺮﺩﻥ ﺣﺮﻑ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺍﺳﺘﺪﻻﻝ ﻧﻴﺴﺖ. ﻣﺜﻼ ﺍﮔﺮ ﮐﺴﯽ ﺑﮕﻮﻳﺪ ﺑﺴﺘﻨﯽ ﻣﻴﻬﻦ ﺧﻴﻠﯽ ﺧﻮﺷﻤﺰﻩ ﺍﺳﺖ ﻭ ﻣﺎ ﺑﮕﻮﻳﻴﻢ ﻧﻪ ﺍﻳﻨﻄﻮﺭ ﻧﻴﺴﺖ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺩﻟﻴﻠﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﻣﺨﺎﻟﻔﺖ‌ﻣﺎﻥ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻩ ﺑﺎﺷﻴﻢ، ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﻧﺘﻈﺎﺭ ﭘﻴﺮﻭﺯﯼ ﺩﺭ ﺑﺤﺚ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ. 
 
 
ﭘﺲ ﭼﻪ ﭼﻴﺰﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺎ ﺭﺍ ﺩﺭ ﻳﮏ ﻣﺒﺎﺣﺜﻪ ﺑﺮﻧﺪﻩ ﮐﻨﺪ؟ 
ﺍﺳﺘﺪﻻﻝ ﻫﻤﻴﺸﻪ ﻫﻤﻪ ﺭﺍ ﻗﺎﻧﻊ ﻧﻤﯽ‌ﮐﻨﺪ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ ﺑﺎﺷﺪ، ﺍﺳﺘﺤﮑﺎﻡ، ﺻﺤﺖ ﻭ ﺍﻋﺘﺒﺎﺭ ﮐﺎﻓﯽ ﻧﻴﺴﺖ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺳﺘﺪﻻﻟﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﻤﻪ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﯼ ﺁﻥ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﺳﺎﺧﺘﺎﺭ ﺍﺳﺘﺪﻻﻝ ﻧﻴﺰ ﻫﻴﭻ ﻣﺸﮑﻠﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻫﻤﭽﻨﺎﻥ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ ﻧﺒﺎﺷﺪ. ﻫﻤﻮﺍﺭﻩ ﮐﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﺩﺭﺳﺖ ﻧﻔﻬﻤﻨﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﮐﻮﺭﮐﻮﺭﺍﻧﻪ ﻋﻘﺎﻳﺪﯼ ﺩﺭ ﺗﻀﺎﺩ ﺑﺎ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻭ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﮐﻪ ﺩﺭ ﻭﺍﻗﻌﻴﺖ ﺑﺪﻳﻬﯽ ﻭ ﺍﺛﺒﺎﺕ ﺷﺪﻩ ﻫﺴﺘﻨﺪ.
 
ﻣﺴﺌﻠﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﺩﻡ‌ﻫﺎ ﻫﻤﻴﺸﻪ ﻫﻢ ﻣﻨﻄﻘﯽ ﻧﻴﺴﺘﻨﺪ ﻭ ﻟﺰﻭﻣﺎ ﺩﺭ ﻫﻤﻪ ﺍﻭﻗﺎﺕ ﺑﻪ ﻋﻠﺖ‌ﻫﺎ ﻭ ﻣﻌﻠﻮﻝ‌ﻫﺎﯼ ﺍﺯ ﻟﺤﺎﻅ ﻋﻠﻤﯽ ﺍﺛﺒﺎﺕ ﺷﺪﻩ ﺑﺎﻭﺭ ﻧﺪﺍﻧﺪ. ﺭﺩ ﺑﺮﺧﯽ ﺍﺯ ﺧﺮﺍﻓﺎﺕ ﺑﺴﻴﺎﺭ ﺩﺷﻮﺍﺭ ﺍﺳﺖ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺑﻪ ﺳﺨﺘﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﺑﺎﻭﺭ ﺩﺍﺭﺩ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﺷﺒﺪﺭ ﭼﻬﺎﺭﭘﺮ ﺷﺎﻧﺲ ﻣﯽ‌ﺁﻭﺭﺩ، ﻗﺎﻧﻊ ﮐﺮﺩﻥ ﮐﻪ ﻫﻴﭻ ﺭﺑﻄﯽ ﺑﻴﻦ ﺍﻳﻦ ﻣﺸﺎﻫﺪﻩ ﻭ ﺍﺗﻔﺎﻗﺎﺕ ﺩﻳﮕﺮﯼ ﮐﻪ ﺑﺮﺍﯼ ﺍﻭ ﺩﺭ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﺍﻓﺘﺪ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺣﺘﯽ ﺑﻬﺘﺮﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻧﻴﺰ ﻫﻤﻮﺍﺭﻩ ﺍﻓﺮﺍﺩﯼ ﺭﺍ ﺩﺭ ﻣﻘﺎﺑﻞ ﺧﻮﺩ ﺩﺍﺭﻧﺪ. ﺩﺭ ﺿﻤﻦ ﺑﺎﻳﺪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺍﻓﺮﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺷﺘﺒﺎﻩ ﮐﻨﻨﺪ ﺣﺘﯽ ﺍﮔﺮ ﻣﺎ ﺩﺭ ﺗﻮﺟﻴﻪ ﺑﺎﻭﺭ ﺧﻮﺩ ﻣﻮﻓﻖ ﺑﻮﺩﻩ ﺑﺎﺷﻴﻢ ﻭ ﺑﻨﺎﺑﺮﺍﻳﻦ ﻗﺎﻧﻊ‌ﺳﺎﺯﯼ ﺑﻪ ﻧﺘﻴﺠﻪ ﻧﺨﻮﺍﻫﺪ ﺭﺳﻴﺪ. 
 
ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻣﻮﻓﻘﻴﺖ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﻧﺒﺎﻳﺪ ﺩﺭ ﻗﺎﺑﻠﻴﺖ ﻗﺎﻧﻊ‌ﺳﺎﺯﯼ ﺁﻥ ﺟﺴﺘﺠﻮ ﮐﺮﺩ. ﺑﻠﮑﻪ ﺑﺎﻳﺪ ﺩﻳﺪ ﭼﻘﺪﺭ ﻭﺍﺿﺢ ﻭ ﺭﻭﺷﻦ ﻭ ﺩﻗﻴﻖ ﻣﻘﺪﻣﻪ‌ﻫﺎ، ﺭﻭﺵ ﺍﺳﺘﺪﻻﻟﯽ ﻭ ﻧﺘﻴﺠﻪ ﺭﺍ ﺑﻴﺎﻥ ﮐﺮﺩﻩ ﺍﺳﺖ ﻭ ﻋﻠﯽ‌ﺍﻻﺻﻮﻝ ﭼﻘﺪﺭ ﺑﺮﺍﯼ ﻣﺨﺎﻃﺐ ﺑﻪ ﻃﻮﺭ ﻋﺎﻡ ﻗﺎﺑﻞ ﻓﻬﻢ ﺍﺳﺖ. ﺍﮔﺮ ﮐﺴﯽ ﮐﻮﺭﮐﻮﺭﺍﻧﻪ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺍﻧﮑﺎﺭ ﮐﻨﺪ، ﺭﺑﻄﯽ ﺑﻪ ﺍﻳﻦ ﻧﺪﺍﺭﺩ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﺧﻮﺏ ﻧﺒﻮﺩﻩ ﺍﺳﺖ.
 
ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﮐﻨﺎﺭ ﺗﻮﺟﻴﻪ ﻭ ﺍﻗﻨﺎﻉ، ﻫﺪﻑ ﺩﻳﮕﺮ ﺍﺳﺘﺪﻻﻝ ﻓﻬﻤﻴﺪﻥ ﺍﺳﺖ. ﮔﺎﻫﯽ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﻢ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻗﺎﻧﻊ ﮐﻨﻴﻢ ﮐﻪ ﺩﻳﺪﮔﺎﻩ ﻣﺎ ﺭﺍ ﺑﺎﻭﺭ ﮐﻨﻨﺪ، ﺑﻠﮑﻪ ﺗﻨﻬﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﻪ ﺁﻧﻬﺎ ﺑﻔﻬﻤﻨﺪ ﻭ ﺩﺭﮎ ﮐﻨﻨﺪ ﮐﻪ ﭼﺮﺍ ﭼﻨﻴﻦ ﺩﻳﺪﮔﺎﻩ ﻭ ﻋﻘﻴﺪﻩ‌ﺍﯼ ﺩﺍﺭﻳﻢ.
 
ﻫﻤﻪ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﮐﻪ ﭼﻨﻴﻦ ﻫﺪﻓﯽ ﻣﺜﻼ ﺩﺭ ﺳﻴﺎﺳﺖ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺳﻴﺎﺳﺘﻤﺪﺍﺭﺍﻥ ﺑﻪ ﻳﮑﺪﻳﮕﺮ ﻣﯽ‌ﺗﺎﺯﻧﺪ ﻭ ﺑﺮ ﺳﺮ ﻫﻢ ﻓﺮﻳﺎﺩ ﻣﯽ‌ﮐﺸﻨﺪ. ﺁﻧﻬﺎ ﺻﺮﻓﺎ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎﯼ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﻓﺮﻳﺎﺩ ﻣﯽ‌ﺯﻧﻨﺪ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺩﺭ ﺧﻴﻠﯽ ﺍﺯ ﻣﻮﺍﺭﺩ ﺑﺮﺍﯼ ﻧﻈﺮﻫﺎﺷﺎﻥ ﺩﻟﻴﻞ ﺑﻴﺎﻭﺭﻧﺪ. ﺍﮔﺮ ﺍﺯ ﺁﻧﻬﺎ ﺩﻟﻴﻞ ﺑﺨﻮﺍﻫﻴﻢ ﺗﺎ ﺑﻔﻬﻤﻴﻢ ﭼﺮﺍ ﭼﻨﻴﻦ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎﯼ ﺩﺍﺭﻧﺪ ﺑﺮﺍﯼ ﻣﺎ ﻭ ﺣﺘﯽ ﺧﻮﺩﺷﺎﻥ ﺑﻬﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.
 
ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭﺱ ﻣﻬﻢ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺗﻮﻗﻊ ﺧﻮﺩ ﺭﺍ ﺑﺎﻻ ﻧﺒﺮﻳﻢ. ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﻗﺎﻧﻊ ﮐﺮﺩﻥ ﻫﻤﻪ ﺑﺎﺷﻴﻢ ﺧﻴﻠﯽ ﺯﻭﺩ ﺧﺴﺘﻪ ﺧﻮﺍﻫﻴﻢ ﺷﺪ. ﻣﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﻫﻤﻪ ﺭﺍ ﺭﺍﺿﯽ ﮐﻨﻴﻢ. ﺍﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺩﻻﻳﻠﯽ ﺑﺮﺍﯼ ﻓﻬﻢ ﻋﻘﺎﻳﺪﻣﺎﻥ ﺍﺭﺍﺋﻪ ﺩﻫﻴﻢ. ﻫﻤﻴﻦ ﻧﮑﺘﻪ ﺩﺳﺘﺎﻭﺭﺩ ﺑﺰﺭﮔﯽ ﺍﺳﺖ.
 
 
ﺑﺮﻧﺪﻩ ﺷﺪﻥ ﺑﻪ ﻫﺮ ﻗﻴﻤﺖ؟
ﺳﻮﯼ ﺩﻳﮕﺮ ﺗﻤﺎﻳﻞ ﺑﻪ ﺍﻳﻨﮑﻪ ﻫﻤﻴﺸﻪ ﺩﺭ ﻫﻤﻪ ﺑﺤﺚ‌ﻫﺎ ﭘﻴﺮﻭﺯ ﺑﺎﺷﻴﻢ ﻭ ﻫﻤﻪ ﺭﺍ ﻗﺎﻧﻊ ﮐﻨﻴﻢ، ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻫﺮ ﺭﻭﺷﯽ ﺭﺍ ﺑﺮﺍﯼ ﺑﺮﻧﺪﻩ ﺷﺪﻥ ﺑﻪ ﮐﺎﺭ ﺑﮕﻴﺮﻳﻢ. ﺧﻄﺮ ﭼﻨﻴﻦ ﺭﻭﻳﮑﺮﺩﯼ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﻣﻴﺪﺍﻧﯽ ﺑﺮﺍﯼ ﺟﻨﮓ ﺑﺪﺍﻧﻴﻢ ﻭ ﺟﻨﺒﻪ‌‌ﯼ ﺗﻔﺎﻫﻢ ﻭ ﻳﺎﺩﮔﻴﺮﯼ ﺩﺭ ﺁﻥ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﮐﻨﻴﻢ. ﺍﮔﺮ ﭼﻨﻴﻦ ﺭﻭﺣﻴﻪ ﻭ ﺭﻭﻳﮑﺮﺩﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ، ﻟﺰﻭﻣﯽ ﻧﺪﺍﺭﺩ ﮐﻪ ﺣﺘﻤﺎ ﺑﻪ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﺑﺘﺪﺍﯼ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺁﻥ ﺍﺷﺎﺭﻩ ﺷﺪ ﻣﺘﻮﺳﻞ ﺷﻮﻳﻢ. ﻳﻌﻨﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﭼﻨﺪﺍﻥ ﻫﻢ ﺗﻬﺎﺟﻤﯽ ﻧﺒﺎﺷﻴﻢ، ﺍﻣﺎ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺭﺍ ﺑﺮﺍﯼ ﺑﺮﻧﺪﻩ ﺷﺪﻥ ﺩﺭ ﻫﺮ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺭﻭﺵ‌ﻫﺎ ﺭﺍ ﺁﮔﺎﻫﺎﻧﻪ ﻳﺎ ﻧﺎﺁﮔﺎﻫﺎﻧﻪ ﺑﻪ ﮐﺎﺭ ﺑﺒﺮﻳﻢ. 
 
ﺣﺘﻤﺎ ﺷﻨﻴﺪﻩ‌ﺍﻳﺪ ﻳﺎ ﺣﺘﯽ ﺧﻮﺩﺗﺎﻥ ﺩﺭ ﺑﺤﺚ‌ﻫﺎ ﮔﻔﺘﻪ‌ﺍﻳﺪ ﮐﻪ ﮐﺴﯽ ﺩﺭ ﺣﺎﻝ ﺳﻔﺴﻄﻪ ﮐﺮﺩﻥ ﺩﺭ ﺑﺤﺚ ﺍﺳﺖ. ﺳﻔﺴﻄﻪ ﺍﺳﺘﺪﻻﻟﯽ ﺍﺳﺖ ﮐﻪ ﺳﻌﯽ ﺩﺭ ﻭﺍﺭﻭﻧﻪ ﮐﺮﺩﻥ ﺣﻘﺎﻳﻖ ﺩﺍﺭﺩ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺭﺍ ﮐﻪ ﺩﺭﺳﺖ ﻳﺎ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ، ﺑﺎ ﻣﻘﺪﻣﻪ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﻌﻤﻮﻻ ﻧﺎﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﺍﺛﺒﺎﺕ ﮐﻨﺪ. ﺳﻔﺴﻄﻪ ﻫﻤﺎﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﻣﻌﻤﻮﻻ ﺑﻪ ﺁﻥ ﻣﻐﺎﻟﻄﻪ ﻳﺎ ﻣﻐﻠﻄﻪ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﻪ ﻧﻮﻋﯽ ﺳﻮﺀ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻨﻄﻖ ﻭ ﺭﻭﺵ‌ﻫﺎﯼ ﺁﻥ ﺍﺳﺖ. ﺩﺭﺱ ﺑﻌﺪﯼ ﺑﻪ ﻣﻐﺎﻟﻄﻪ، ﺭﻭﺵ‌ﻫﺎ ﻭ ﺍﻧﻮﺍﻉ ﺍﻥ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺭﺩ.
