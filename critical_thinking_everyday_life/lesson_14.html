ﺩﺭﺱ ﺩﻫﻢ- ﺗﻌﺎﺭﻳﻒ ﻭ ﻣﻨﺎﺑﻌﯽ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺑﻴﺸﺘﺮ

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﻗﺮﺍﺭ ﮔﺬﺍﺷﺘﻴﻢ ﺗﺎ ﺍﻳﻨﺠﺎﯼ ﺩﺭﺳﻨﺎﻣﻪ ﺩﺭﺑﺎﺭﻩ ﻣﻨﻄﻖ ﻭ ﻣﺴﺎﺋﻞ ﻓﻠﺴﻔﯽ ﭘﻴﺮﺍﻣﻮﻥ ﺁﻥ ﺣﺮﻑ ﺯﺩﻳﻢ ﺑﺪﻭﻥ ﺁﻧﮑﻪ ﭼﻨﺪﺍﻥ ﺍﺯ ﺍﺻﻄﻼﺣﺎﺕ ﻭ ﺍﺳﺎﻣﯽ ﻗﻠﻨﺒﻪ ﺳﻠﻨﺒﻪ ﻭ ﺑﺰﺭﮒ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ. ﺍﻣﺎ ﺍﻳﻦ ﺑﻪ ﺁﻥ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﻧﺒﺎﻳﺪ ﺑﺎ ﺣﻮﺯﻩ‌ﻫﺎﯼ ﻣﺘﻨﻮﻉ ﻣﻨﻄﻖ ﻭ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﻭ ﺍﺻﻄﻼﺡ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﺩﺭ ﺁﻥ ﺁﺷﻨﺎ ﺷﻮﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﻃﻮﺭ ﻓﻬﺮﺳﺖ‌ﻭﺍﺭ ﻫﺮ ﺁﻧﭽﻪ ﺭﺍ ﮐﻪ ﻻﺯﻡ ﺍﺳﺖ ﺑﺪﺍﻧﻴﺪ ﻣﺮﻭﺭ ﮐﻨﻴﻢ. ﺑﺮﺧﯽ ﺍﺯ ﺍﻳﻦ ﺍﺻﻄﻼﺣﺎﺕ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﻌﺮﻓﯽ ﮐﺮﺩﻳﻢ ﻭ ﺑﺮﺧﯽ ﻧﻴﺰ ﻓﺮﺍﺗﺮ ﺍﺯ ﺳﻄﺢ ﺍﻳﻦ ﺩﺭﺱ ﻫﺴﺘﻨﺪ ﻭ ﺻﺮﻓﺎ ﺑﺮﺍﯼ ﺁﺷﻨﺎﻳﯽ ﺩﺭ ﺍﻳﻦ ﻓﻬﺮﺳﺖ ﺁﻣﺪﻩ‌ﺍﻧﺪ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻳﻦ ﺗﻨﻬﺎ ﻳﮏ ﺳﺮﻧﺦ ﺍﺳﺖ ﺑﺮﺍﯼ ﺁﻧﮑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﺮﻭﻳﺪ ﻭ ﺑﻴﺸﺘﺮ ﻣﻄﺎﻟﻌﻪ ﮐﻨﻴﺪ. ﺑﻪ ﻫﻤﻴﻦ ﻣﻨﻈﻮﺭ ﺩﺭ ﭘﺎﻳﺎﻥ ﺍﻳﻦ ﺩﺭﺱ ﮐﺘﺎﺏ‌ﻫﺎ ﻭ ﻣﻨﺎﺑﻌﯽ ﺭﺍ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺑﻴﺸﺘﺮ ﺳﻮﺩﻣﻨﺪ ﺑﺎﺷﻨﺪ ﻧﻴﺰ ﻣﻌﺮﻓﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

ﻣﻨﻄﻖ: ﺣﻮﺯﻩ‌ﺍﯼ ﺍﺯ ﻣﻌﺮﻓﺖ ﺑﺸﺮﯼ ﺍﺳﺖ ﮐﻪ ﺭﻭﺵ ﺩﺭﺳﺖ ﺍﻧﺪﻳﺸﻴﺪﻥ ﻭ ﺍﺳﺘﺪﻻﻝ ﮐﺮﺩﻥ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺩﺭ ﺍﺑﺘﺪﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺷﺎﺧﻪ‌ﺍﯼ ﺍﺯ ﻓﻠﺴﻔﻪ ﻣﺘﻮﻟﺪ ﺷﺪﻩ ﺍﺳﺖ ﺍﻣﺎ ﺑﻌﺪﻫﺎ ﺑﻪ ﺣﻮﺯﻩ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻣﺎﻧﻨﺪ ﺭﻳﺎﺿﯽ ﻭ ﻋﻠﻮﻡ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻭ ﺑﺴﻴﺎﺭﯼ ﻋﻠﻮﻡ ﺩﻳﮕﺮ ﺭﺍﻩ ﻳﺎﻓﺘﻪ ﺍﺳﺖ.

ﺍﺳﺘﺪﻻﻝ: ﻳﮑﯽ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﻋﺮﺿﻪ ﮐﺮﺩﻥ ﺩﻟﻴﻞ ﺑﺮﺍﯼ ﭘﺬﻳﺮﻓﺘﻦ ﺍﻣﺮﯼ ﺑﺎ ﺷﺮﻭﻉ ﮐﺮﺩﻥ ﺍﺯ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎﻳﯽ ﻭ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﻳﻦ ﮐﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﮐﻪ ﻣﺎﻳﻞ ﺑﻪ ﭘﺬﻳﺮﻓﺘﻪ ﺷﺪﻥ ﺁﻥ ﻫﺴﺘﻴﺪ ﭼﮕﻮﻧﻪ ﻣﺮﺣﻠﻪ ﺑﻪ ﻣﺮﺣﻠﻪ ﺍﺯ ﭘﻴﺶ‌ﻓﺮﺽ‌ﻫﺎ ﺣﺎﺻﻞ ﻣﯽ‌ﺷﻮﺩ.

ﺗﻨﺎﻗﺾ ﻣﻨﻄﻘﯽ: ﺍﮔﺮ ﮔﺰﺍﺭﻩ‌ﺍﯼ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﻫﻢ ﺩﺭﺳﺖ ﻭ ﻫﻢ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ، ﺁﻧﮕﺎﻩ ﺑﺎ ﻳﮏ ﺗﻨﺎﻗﺾ ﻣﻨﻄﻘﯽ ﺭﻭ ﺑﻪ ﺭﻭ ﻫﺴﺘﻴﻢ ﮐﻪ ﺍﺯ ﻧﻈﺮ ﻣﻨﻄﻘﯽ ﻏﻴﺮﻣﻤﮑﻦ ﺍﺳﺖ. ﻣﺜﻼ ﺍﻳﻦ ﺟﻤﻠﻪ ﮐﻪ ﻣﺎﺳﺖ ﺳﻔﻴﺪ ﻭ ﻏﻴﺮ ﺳﻔﻴﺪ ﺍﺳﺖ، ﺣﺎﻭﯼ ﺗﻨﺎﻗﺾ ﺩﺭﻭﻧﯽ ﺍﺳﺖ ﺯﻳﺮﺍ ﻣﺎﺳﺖ ﺩﺭ ﻭﺍﻗﻌﻴﺖ ﻳﺎ ﺳﻔﻴﺪ ﺍﺳﺖ ﻭ ﻳﺎ ﻧﻴﺴﺖ؛ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻫﻤﺰﻣﺎﻥ ﻫﺮ ﺩﻭﯼ ﺍﻳﻦ ﺻﻔﺎﺕ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺍﻋﺘﺒﺎﺭ ﺍﺳﺘﺪﻻﻝ: ﺍﻋﺘﺒﺎﺭ ﺧﺎﺻﻴﺘﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻧﺴﺒﺖ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﻭ ﮐﻴﻔﻴﺖ ﺁﻥ ﺭﺍ ﺗﻌﻴﻴﻦ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺧﻮﺏ ﺑﺎﻳﺪ ﻣﻌﺘﺒﺮ ﺑﺎﺷﺪ. ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ ﺍﺳﺖ ﺍﮔﺮ ﻭ ﺗﻨﻬﺎ ﺍﮔﺮ، ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻧﺒﺎﺷﺪ ﮐﻪ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﻳﻌﻨﯽ ﻭﺿﻌﻴﺘﯽ ﻣﻤﮑﻦ ﻧﺒﺎﺷﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﻭ ﻧﺘﻴﺠﻪ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ.

ﺻﺤﺖ ﺍﺳﺘﺪﻻﻝ: ﺍﮔﺮ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻣﻌﺘﺒﺮ، ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺩﺭﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺁﻧﮕﺎﻩ ﺁﻥ ﺍﺳﺘﺪﻻﻝ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻭﺍﻗﻌﺎ ﺧﻮﺏ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﺧﺎﺻﻴﺘﯽ ﺑﻪ ﻧﺎﻡ ﺻﺤﺖ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ: ﻧﻮﻋﯽ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺩﻭ ﻳﺎ ﭼﻨﺪ ﻣﻘﺪﻣﻪ ﺁﻏﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﻧﺘﻴﺠﻪ ﺑﻪ ﻃﻮﺭ ﻣﻨﻄﻘﯽ ﺍﺯ ﺁﻧﻬﺎ ﺣﺎﺻﻞ ﻣﯽ‌ﺷﻮﺩ. ﺑﻪ ﺷﮑﻠﯽ ﮐﻪ ﺍﮔﺮ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ، ﻧﺘﻴﺠﻪ ﺑﻪ ﻃﻮﺭ ﺍﺟﺘﻨﺎﺏ‌ﻧﺎﭘﺬﻳﺮﯼ ﺑﺎﻳﺪ ﺩﺭﺳﺖ ﺑﺎﺷﺪ. ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎﻳﯽ ﮐﻠﯽ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺟﺰﺋﯽ ﻣﯽ‌ﺭﺳﺪ.

ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ: ﻧﻮﻋﯽ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﻧﻤﻮﻧﻪ‌ﻫﺎ ﻳﺎ ﻣﺜﺎﻝ‌ﻫﺎﻳﯽ ﻣﺸﺎﺑﻪ ﻫﻢ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﻗﻴﻘﺎ ﺑﻪ ﺩﻟﻴﻞ ﻫﻤﻴﻦ ﻣﺸﺎﺑﻬﺖ، ﺍﺳﺘﺪﻻﻝ‌ﮐﻨﻨﺪﻩ ﻳﮏ ﺍﻟﮕﻮﯼ ﮐﻠﯽ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺁﻧﻬﺎ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﻣﯽ‌ﮐﻨﺪ. ﺍﺳﺘﺪﻻﻝ ﺍﺳﺘﻘﺮﺍﻳﯽ ﺑﻪ ﻣﻌﻨﺎﯼ ﺭﺳﻴﺪﻥ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺟﺰﺋﯽ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﮐﻠﯽ ﺍﺳﺖ.

ﻣﻐﺎﻟﻄﻪ: ﺍﺳﺘﺪﻻﻟﯽ ﺍﺳﺖ ﮐﻪ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﻧﺎﺩﺭﺳﺖ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﮔﺰﺍﺭﻩ ﺩﺭﺳﺖ ﺍﺛﺒﺎﺕ ﮐﻨﺪ. ﻣﻐﺎﻟﻄﻪ، ﻣﻐﻠﻄﻪ ﻳﺎ ﺳﻔﺴﻄﻪ ﻧﻴﺰ ﻧﺎﻣﻴﺪﻩ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﻪ ﺩﻭ ﺭﻭﺵ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ. ﺍﻳﻨﮑﻪ ﺣﺪﺍﻗﻞ ﻳﮑﯽ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ، ﻳﺎ ﺍﻳﻨﮑﻪ ﻧﺘﻴﺠﻪ ﺑﻪ ﻃﻮﺭ ﻣﻨﻄﻘﯽ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺍﺳﺘﺨﺮﺍﺝ ﻧﺸﻮﺩ. ﻣﻐﺎﻟﻄﻪ ﻣﻌﻤﻮﻻ ﺑﺮﺍﯼ ﻭﺍﺭﻭﻧﻪ ﮐﺮﺩﻥ ﺣﻘﻴﻘﺖ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﭘﺎﺭﺍﺩﻭﮐﺲ: ﺑﻪ ﺣﮑﻢ ﻳﺎ ﺍﺣﮑﺎﻣﯽ ﻇﺎﻫﺮﺍ ﺻﺤﻴﺢ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﺗﻨﺎﻗﺾ ﻣﯽ‌ﺷﻮﻧﺪ ﻳﺎ ﺑﺎ ﺷﻬﻮﺩ ﻣﻄﺎﺑﻘﺖ ﻧﻤﯽ‌ﻳﺎﺑﻨﺪ. ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺑﻪ ﺟﻤﻼﺕ ﻣﺘﻨﺎﻗﺾ ﻭ ﺣﺘﯽ ﻣﺨﺎﻟﻒ ﮐﻪ ﻳﮏ ﺣﻘﻴﻘﺖ ﻭﺍﺣﺪ ﺭﺍ ﺑﻴﺎﻥ ﻣﯽ‌ﮐﻨﻨﺪ ﻧﻴﺰ ﭘﺎﺭﺍﺩﻭﮐﺲ ﻣﯽ‌ﮔﻮﻳﻨﺪ. ﺍﮔﺮ ﭘﺎﺭﺍﺩﻭﮐﺲ ﺑﻪ ﻣﻌﻨﺎﯼِ ﺗﻨﺎﻗﺾ ﺑﺎ ﻳﮑﯽ ﺍﺯ ﮔﺰﺍﺭﻩ‌ﻫﺎﯼِ ﻫﻤﺎﻥ ﻧﻈﺮﻳﻪ‌ﺍﯼ ﺑﺎﺷﺪ ﮐﻪ ﭘﺎﺭﺍﺩﻭﮐﺲ ﺩﺭ ﺁﻥ ﭘﺪﻳﺪ ﺁﻣﺪﻩ ﺍﻳﻦ ﺍﻣﺮ ﻳﮏ ﺿﻌﻒ ﺟﺪﯼ ﺑﺮﺍﯼِ ﺁﻥ ﻧﻈﺮﻳﻪ ﻣﺤﺴﻮﺏ ﺷﺪﻩ ﻭ ﺁﻥ ﺭﺍ ﺑﯽ‌ﺍﻋﺘﺒﺎﺭ ﻣﯽ‌ﮐﻨﺪ.

ﻣﻨﻄﻖ ﮔﺰﺍﺭﻩ‌ﻫﺎ: ﺷﺎﺧﻪ‌ﺍﯼ ﺍﺯ ﻣﻨﻄﻖ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺗﺮﮐﻴﺐ ﮔﺰﺍﺭﻩ‌ﻫﺎ ﻳﺎ ﺟﻤﻠﻪ‌ﻫﺎﯼ ﺩﺭﺳﺖ ﻭ ﻧﺎﺩﺭﺳﺖ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﺩ. ﺩﺭ ﺍﻳﻦ ﺣﻮﺯﻩ ﺍﺯ ﻣﻨﻄﻖ ﺑﺎ ﻫﺮ ﮔﺰﺍﺭﻩ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻭﺍﺣﺪ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺩﺭﺳﺖ ﻳﺎ ﻧﺎﺩﺭﺳﺖ ﺑﺎﺷﺪ ﻭ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺗﺮﮐﻴﺐ‌ﻫﺎﯼ ﻣﻤﮑﻦ ﺑﻴﻦ ﺁﻧﻬﺎ ﺍﺯ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﺷﻮﺩ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﻣﯽ‌ﮐﻨﻨﺪ. ﮔﺰﺍﺭﻩ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻣﻨﻄﻖ ﮔﺰﺍﺭﻩ‌ﻫﺎ ﻓﻘﻂ ﮔﺰﺍﺭﻩ‌ﻫﺎﯼ ﺧﺒﺮﯼ ﺍﺳﺖ. ﺩﺭ ﻣﻨﻄﻖ ﮐﻼﺳﻴﮏ ﻳﺎ ﻣﻨﻄﻖ ﺩﻭ ﺍﺭﺯﺷﯽ، ﮔﺰﺍﺭﻩ‌ﻫﺎ ﺩﺍﺭﺍﯼ ﺍﺭﺯﺵ ﺩﺭﺳﺖ ﻳﺎ ﻏﻠﻂ ﻫﺴﺘﻨﺪ.

ﻣﻨﻄﻖ ﻣﺤﻤﻮﻻﺕ: ﺍﻳﻦ ﺷﺎﺧﻪ ﮐﻪ ﻣﻨﻄﻖ ﻣﺮﺗﺒﻪ ﺍﻭﻝ ﻧﻴﺰ ﻧﺎﻡ ﺩﺍﺭﺩ ﺯﺑﺎﻧﯽ ﻓﺮﺍﮔﻴﺮ ﺩﺭ ﻋﻠﻮﻡ ﻧﻤﺎﺩﯼ ﺍﺳﺖ، ﮐﻪ ﺗﻮﺳﻂ ﺣﻮﺯﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺭﻳﺎﺿﯽ، ﻓﻠﺴﻔﻪ، ﺯﺑﺎﻥ‌ﺷﻨﺎﺳﯽ، ﻋﻠﻮﻡ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ. ﺩﺭ ﺍﻳﻦ ﻣﻨﻄﻖ ﺑﺎ ﻣﺤﻤﻮﻻﺕ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺑﻪ ﺯﺑﺎﻥ ﺩﻳﮕﺮ ﮔﺰﺍﺭﻩ‌ﻧﻤﺎ ﻧﺎﻣﻴﺪﻩ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺩﺭ ﺟﻤﻠﻪ «ﺍﻧﺴﺎﻥ ﺣﻴﻮﺍﻥ ﻧﺎﻃﻖ ﺍﺳﺖ»، ﺍﻧﺴﺎﻥ ﻣﻮﺿﻮﻉ ﻭ ﺣﻴﻮﺍﻥ ﻧﺎﻃﻖ ﻣﺤﻤﻮﻝ ﺁﻥ ﺍﺳﺖ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺩﺭ ﻣﻨﻄﻖ ﻣﺤﻤﻮﻻﺕ ﮔﺰﺍﺭﻩ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺍﺟﺰﺍﯼ ﺳﺎﺯﻧﺪﻩ ﺁﻥ ﺗﺠﺰﻳﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺑﺎ ﻧﻤﺎﺩﮔﺰﺍﺭﯼ ﻭ ﺗﺮﮐﻴﺐ ﺁﻧﻬﺎ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺭﺍ ﺗﺤﻠﻴﻞ ﻣﯽ‌ﮐﻨﻨﺪ.

ﻣﻨﻄﻖ ﻓﺎﺯﯼ: ﺩﺭ ﻣﻨﻄﻖ ﮐﻼﺳﻴﮏ ﻣﻌﻤﻮﻝ ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﺎ ﺁﻥ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻳﻢ، ﮔﺰﺍﺭﻩ‌ﻫﺎﯼ ﻳﺎ ﺩﺭﺳﺖ ﻫﺴﺘﻨﺪ ﻭ ﻳﺎ ﻧﺎﺩﺭﺳﺖ ﻭ ﺩﺭﺳﺖ ﺑﺮ ﻫﻤﻴﻦ ﺍﺳﺎﺱ ﺩﺭ ﺯﺑﺎﻥ ﺑﺮﻧﺎﻣﻪ‌ﻧﻮﻳﺴﯽ ﮐﺎﻣﭙﻴﻮﺗﺮﯼ ﻧﻴﺰ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﺯ ﻗﻮﺍﻋﺪ ﻣﻨﻄﻘﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻮﺩ، ﺩﺭﺳﺖ ﻳﺎ ﻧﺎﺩﺭﺳﺖ ﺑﻪ ﺻﻔﺮ ﻭ ﻳﮏ ﺗﺮﺟﻤﻪ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻣﺎ ﺍﻳﺪﻩ ﻣﻨﻄﻖ ﻓﺎﺯﯼ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺻﺮﻓﺎ ﺩﺭﺳﺖ ﻳﺎ ﻧﺎﺩﺭﺳﺖ ﺭﺍ ﺑﻪ ﺧﻴﻠﯽ ﺍﺯ ﮔﺰﺍﺭﻩ‌ﻫﺎ ﻧﺴﺒﺖ ﺩﺍﺩ ﻭ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺭﺯﺷﯽ ﺑﻴﻦ ﺍﻳﻦ ﺩﻭ ﺗﻌﺮﻳﻒ ﺷﻮﺩ.

 

ﺑﻴﺸﺘﺮ ﺑﺪﺍﻧﻴﺪ
ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺑﻴﺸﺘﺮ ﺩﺭ ﺯﻣﻴﻨﻪ ﻣﻨﻄﻖ ﻭ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﮐﺘﺎﺏ‌ﻫﺎ ﻭ ﻣﻨﺎﺑﻊ ﺯﻳﺮ ﻣﺮﺍﺟﻌﻪ ﮐﻨﻴﺪ.

  • ﻣﻨﻄﻖ ﮐﺎﺭﺑﺮﺩﯼ، ﻧﻮﻳﺴﻨﺪﻩ: ﻋﻠﯽ‌ ﺍﺻﻐﺮ ﺧﻨﺪﺍﻥ، ﺍﻧﺘﺸﺎﺭﺍﺕ ﺳﻤﺖ، ﭼﺎﭖ ﻫﻔﺘﻢ ۱۳۹۱.
  • ﻣﻨﻄﻖ ﺻﻮﺭﯼ، ﻧﻮﻳﺴﻨﺪﻩ: ﻣﺤﻤﺪ ﺧﻮﺍﻧﺴﺎﺭﯼ، ﺍﻧﺘﺸﺎﺭﺍﺕ ﺩﻳﺪﺍﺭ، ﭼﺎﭖ ۴۵ ﺍﻡ ۱۳۹۱.
  • ﻣﺒﺎﻧﯽ ﻣﻨﻄﻖ ﺟﺪﻳﺪ، ﻧﻮﻳﺴﻨﺪﻩ:‌ﻟﻄﻒ‌ﺍﻟﻠﻪ ﻧﺒﻮﯼ، ﺍﻧﺘﺸﺎﺭﺍﺕ ﺳﻤﺖ، ﭼﺎﭖ ﻫﺸﺘﻢ ۱۳۹۱.
  • ﺩﺭﺁﻣﺪﯼ ﺑﺮ ﻣﻨﻄﻖ ﺟﺪﻳﺪ، ﻧﻮﻳﺴﻨﺪﻩ: ﮔﺮﺍﻫﺎﻡ ﭘﺮﻳﺴﺖ، ﻣﺘﺮﺟﻢ: ﺑﻬﺮﺍﻡ ﺍﺳﺪﻳﺎﻥ، ﻧﺸﺮ ﻣﺎﻫﯽ ۱۳۸۷.
  • ﻣﻐﺎﻟﻄﺎﺕ، ﻧﻮﻳﺴﻨﺪﻩ: ﻋﻠﯽ ﺍﺻﻐﺮ ﺧﻨﺪﺍﻥ، ﺍﻧﺘﺸﺎﺭﺍﺕ ﺑﻮﺳﺘﺎﻥ ﻗﻢ ۱۳۸۸.
  • ﺍﺯ ﺍﺭﺳﻄﻮ ﺗﺎ ﮔﻮﺩﻝ: ﻣﺠﻤﻮﻋﻪ‌ ﻣﻘﺎﻟﻪ‌ﻫﺎﯼ ﻓﻠﺴﻔﯽ-ﻣﻨﻄﻘﯽ، ﻧﻮﻳﺴﻨﺪﻩ: ﺿﻴﺎﺀ ﻣﻮﺣﺪ، ﺍﻧﺘﺸﺎﺭﺍﺕ ﻫﺮﻣﺲ ۱۳۸۹.
  • ﺷﮕﺮﺩﻫﺎ، ﺍﻣﮑﺎﻥ‌ﻫﺎ ﻭ ﻣﺤﺪﻭﺩﻳﺘﻬﺎﯼ ﺑﺤﺚ ﺑﺎ ﺑﻨﻴﺎﺩﮔﺮﺍﻳﺎﻥ: ﺩﺭﺁﻣﺪﯼ ﺑﺮ ﺭﻭﺷﻨﮕﺮﯼ، ﻧﻮﻳﺴﻨﺪﻩ: ﻫﻮﺑﺮﺕ ﺷﻼﻳﺸﺮﺕ، ﻣﺘﺮﺟﻢ: ﻣﺤﻤﺪﺭﺿﺎ ﻧﻴﮑﻔﺮ، ﺍﻧﺘﺸﺎﺭﺍﺕ ﻃﺮﺡ ﻧﻮ ۱۳۸۱.
