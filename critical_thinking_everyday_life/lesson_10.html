ﺩﺭﺱ ﻫﻔﺘﻢ- ﻣﻐﺎﻟﻄﻪ ﻳﺎ ﺳﻮﺀ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻨﻄﻖ

ﺩﺭ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺩﺭﺑﺎﺭﻩ ﻣﻨﻄﻖ ﻭ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﺗﺎ ﺍﻳﻨﺠﺎ ﺑﻪ ﺗﻌﺮﻳﻒ ﺍﺳﺘﺪﻻﻝ، ﺍﻧﻮﺍﻉ ﺁﻥ ﻭ ﺍﻋﺘﺒﺎﺭ ﻭ ﺻﺤﺖ ﻭ ﺭﻭﺵ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﭘﺮﺩﺍﺧﺘﻪ‌ﺍﻳﻢ ﻭ ﺗﻼﺵ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺧﻮﺏ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﻴﻢ ﻭ ﺗﻤﺮﻳﻦ ﮐﻨﻴﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﺷﮑﻞ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ‌ﺍﯼ ﺑﺤﺚ ﻭ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻴﻢ.

ﺍﻣﺎ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻫﻤﻴﺸﻪ ﺧﻮﺏ ﭘﻴﺶ ﻧﻤﯽ‌ﺭﻭﻧﺪ ﻭ ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﻮﺍﺭﺩ ﺑﻪ ﺧﺼﻮﺹ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ، ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻣﺎ ﺧﻄﺎﻫﺎﻳﯽ ﺩﺍﺭﻧﺪ. ﺑﻪ ﻣﺜﺎﻝ‌ﻫﺎﯼ ﺯﻳﺮ ﺩﻗﺖ ﮐﻨﻴﺪ ﺗﺎ ﻣﺘﻮﺟﻪ ﺷﻮﻳﺪ ﭼﻪ ﻧﻮﻉ ﺧﻄﺎﻫﺎﻳﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻣﺎ ﺭﺥ ﺩﻫﺪ.

  • ﺳﺨﻨﺮﺍﻧﯽ ﺍﻳﻦ ﺳﻴﺎﺳﺘﻤﺪﺍﺭ ﺳﺮﺍﺳﺮ ﺑﯽ‌ﻣﻌﻨﺎ ﻭ ﻣﻬﻤﻞ ﺍﺳﺖ، ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﻭ ﺩﺭ ﺟﻮﺍﻧﯽ ﻋﻀﻮ ﺣﺰﺏ ﮐﻤﻮﻧﻴﺴﺖ ﺑﻮﺩﻩ ﺍﺳﺖ.
  • ﺣﺮﻓﯽ ﮐﻪ ﺗﻮ ﻣﯽ‌ﺯﻧﯽ ﺩﺭﺳﺖ ﻣﺜﻞ ﺣﺮﻑ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻫﻴﺘﻠﺮ ﻣﯽ‌ﺯﺩ ﻭ ﺑﺮﺍﯼ ﻫﻤﻴﻦ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ ﻧﻴﺴﺖ.
  • ﻓﺮﻭﺷﻨﺪﻩ‌ﺍﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﺷﻬﺮ ﺍﺯ ﺍﻭ ﺧﺮﻳﺪ ﮐﺮﺩﻡ ﺳﺮﻡ ﮐﻼﻩ ﮔﺬﺍﺷﺖ. ﻫﻤﻪ ﻣﺮﺩﻡ ﺁﻥ ﺷﻬﺮ ﮐﻼﻫﺒﺮﺩﺍﺭ ﻫﺴﺘﻨﺪ.

ﺣﺘﻤﺎ ﻣﺘﻮﺟﻪ ﺷﺪﻩ‌ﺍﻳﺪ ﮐﻪ ﺩﺭ ﻫﺮ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﺟﻤﻠﻪ‌ﻫﺎ ﻣﺸﮑﻠﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻭﻗﺘﯽ ﺍﺩﻋﺎﯼ ﻧﺎﺩﺭﺳﺘﯽ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﺘﻴﺠﻪ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺍﺩﻋﺎﯼ ﺩﺭﺳﺘﯽ ﺁﻥ ﺭﺍ ﺩﺍﺭﻳﻢ ﻳﺎ ﻭﻗﺘﯽ ﺑﺎ ﺭﻭﺷﯽ ﻧﺎﺩﺭﺳﺖ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﻧﺘﻴﺠﻪ ﺩﻟﺨﻮﺍﻩ‌ﻣﺎﻥ ﺑﺮﺳﻴﻢ، ﺩﻳﮕﺮ ﺍﺳﺘﺪﻻﻝ‌ﻣﺎﻥ ﻗﺎﺑﻞ ﺍﻋﺘﻤﺎﺩ ﻧﻴﺴﺖ. ﺑﻪ ﭼﻨﻴﻦ ﺧﻄﺎﻫﺎﻳﯽ ﺩﺭ ﻣﻨﻄﻖ ﻣﻐﺎﻟﻄﻪ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﮐﻪ ﻫﻤﺎﻥ ﺑﻪ ﺍﺷﺘﺒﺎﻩ ﺍﻧﺪﺍﺧﺘﻦ ﻣﺨﺎﻃﺐ ﺍﺳﺖ ﻭ ﻟﺰﻭﻣﺎ ﻫﻢ ﻋﻤﺪﯼ ﻧﻴﺴﺖ. ﻳﻌﻨﯽ ﺍﺳﺘﺪﻻﻝ ﮐﻨﻨﺪﻩ ﻣﻌﻤﻮﻻ ﺑﺎ ﻗﺼﺪ ﻓﺮﻳﺐ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﻣﻐﻠﻄﻪ ﻧﻤﯽ‌ﮐﻨﺪ، ﺍﻣﺎ ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻧﺘﻴﺠﻪ ﺩﻟﺨﻮﺍﻫﺶ ﺭﺍ ﻫﺮ ﻃﻮﺭ ﺷﺪﻩ ﺍﺛﺒﺎﺕ ﮐﻨﺪ ﻭ ﺑﺮﺍﻳﺶ ﺩﻟﻴﻞ ﺑﻴﺎﻭﺭﺩ ﻧﺪﺍﻧﺴﺘﻪ ﺩﭼﺎﺭ ﻣﻐﺎﻟﻄﻪ ﻣﯽ‌ﺷﻮﺩ.

ﺍﺳﺘﺪﻻﻝ ﺻﺤﻴﺢ ﻣﺒﻨﺎﯼ ﺍﺻﻠﯽ ﺑﺮﻧﺪﻩ ﺷﺪﻥ ﺩﺭ ﻫﺮ ﻣﺒﺎﺣﺜﻪ ﺍﺳﺖ ﻭ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻣﻨﻄﻘﯽ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺭﺍ ﺧﺪﺷﻪ‌ﺩﺍﺭ ﻣﯽ‌ﮐﻨﻨﺪ. ﻳﮑﯽ ﺍﺯ ﺑﺨﺶ‌ﻫﺎﯼ ﺟﺬﺍﺏ ﻣﻨﻄﻖ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﻧﻮﺍﻉ ﻣﻐﺎﻟﻄﻪ ﺍﺳﺖ. ﺑﺎ ﺩﺍﻧﺴﺘﻦ ﺍﻳﻨﮑﻪ ﭼﻄﻮﺭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭﺍﺳﺘﺪﻻﻝ‌ﻣﺎﻥ ﺍﺷﺘﺒﺎﻩ ﮐﻨﻴﻢ، ﺑﺎ ﭘﺮﻫﻴﺰ ﺍﺯ ﺁﻥ ﺍﺷﺘﺒﺎﻩ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺩﺭﺳﺘﯽ ﺍﺭﺍﺋﻪ ﺩﻫﻴﻢ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺩﺍﻧﺴﺘﻦ ﺍﻧﻮﺍﻉ ﻣﻐﺎﻟﻄﻪ ﺑﻪ ﺟﺰ ﻣﭻ‌ﮔﻴﺮﯼ ﺍﺯ ﺩﻳﮕﺮﺍﻥ، ﺑﻪ ﺩﺭﺳﺘﯽ ﺍﺳﺘﺪﻻﻝ ﻭ ﺗﻔﮑﺮ ﺧﻮﺩﻣﺎﻥ ﻫﻢ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﻗﺮﺍﺭ ﻧﻴﺴﺖ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﻳﯽ ﻗﺎﻧﻊ ﮐﻨﻨﺪﻩ ﺑﺎﺷﻨﺪ، ﺍﻣﺎ ﻣﻌﻤﻮﻻ ﺍﻳﻨﻄﻮﺭ ﻫﺴﺘﻨﺪ ﻭ ﻫﻤﻴﻦ ﻧﮑﺘﻪ ﺍﺳﺖ ﮐﻪ ﺑﺎﻋﺚ ﺍﻫﻤﻴﺖ ﺁﻧﻬﺎ ﻣﯽ‌ﺷﻮﺩ. ﺿﻤﻦ ﺍﻳﻨﮑﻪ ﻫﻤﻴﺸﻪ ﻫﻢ ﻣﺎﻧﻨﺪ ﻣﺜﺎﻝ‌ﻫﺎﯼ ﺫﮐﺮ ﺷﺪﻩ ﺍﺷﺘﺒﺎﻩ‌ﻫﺎﯼ ﺁﺷﮑﺎﺭﯼ ﻧﻴﺴﺘﻨﺪ ﻭ ﺑﺮﺍﯼ ﺗﺸﺨﻴﺺ ﺁﻧﻬﺎ ﺩﻗﺖ ﻭ ﻣﻄﺎﻟﻌﻪ ﻻﺯﻡ ﺍﺳﺖ.

 

ﺍﻧﻮﺍﻉ ﻣﻐﺎﻟﻄﻪ
ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﻣﻐﺎﻃﻪ‌ﻫﺎﯼ ﻣﻨﻄﻘﯽ ﺭﺍ ﺍﺯ ﻟﺤﺎﻅ ﻣﺤﺘﻮﺍﯼ ﺁﻧﻬﺎ ﻃﺒﻘﻪ‌ﺑﻨﺪﯼ ﮐﻨﻴﻢ ﺑﺎ ﺩﻭ ﺩﺳﺘﻪ ﻣﻐﻠﻄﻪ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻳﻢ. ﺩﺳﺘﻪ ﺍﻭﻝ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﻭﺿﻮﺡ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺯﺑﺎﻥ ﺑﻪ ﮐﺎﺭ ﺭﻓﺘﻪ ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﻣﺮﺑﻮﻁ ﻣﯽ‌ﺷﻮﺩ. ﺍﮔﺮ ﺍﺯ ﺯﺑﺎﻧﯽ ﻣﺒﻬﻢ ﻳﺎ ﺩﻭﭘﻬﻠﻮ ﺩﺭ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ ﻣﺮﺗﮑﺐ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﺍﺯ ﻧﻮﻉ ﻭﺿﻮﺡ ﺷﺪﻩ‌ﺍﻳﻢ. ﻣﺜﺎﻝ ﺯﻳﺮ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﻤﻮﻧﻪ‌ﺍﯼ ﺍﺯ ﺍﻳﻦ ﻣﻐﺎﻟﻄﻪ ﻣﻼﺣﻈﻪ ﮐﻨﻴﺪ.

۱- ﺩﺭ ﺑﺎﺯ ﺍﺳﺖ.

۲- ﺑﺎﺯ ﭘﺮﻧﺪﻩ ﺍﺳﺖ.

۳- ﭘﺲ ﺩﺭ ﭘﺮﻧﺪﻩ ﺍﺳﺖ.

ﺑﻪ ﻭﺿﻮﺡ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﺣﺎﺻﻞ ﺷﺪﻩ، ﻣﻬﻤﻞ ﻭ ﺑﯽ‌ﻣﻌﻨﯽ ﺍﺳﺖ. ﺍﻣﺎ ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﺍﺯ ﻟﺤﺎﻅ ﺳﺎﺧﺘﺎﺭﯼ ﻭ ﺑﺮ ﻣﺒﻨﺎﯼ ﺁﻧﭽﻪ ﺗﺎ ﮐﻨﻮﻥ ﺁﻣﻮﺧﺘﻴﻢ ﻫﻴﭻ ﻣﺸﮑﻠﯽ ﻧﺪﺍﺭﺩ ﻭ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﻴﺎﺳﯽ ﺳﺎﺩﻩ ﻭ ﺳﺮ ﺭﺍﺳﺖ ﺍﺳﺖ. ﭘﺲ ﭼﻪ ﭼﻴﺰﯼ ﺑﺎﻋﺚ ﺷﺪﻩ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﭼﻨﻴﻦ ﻧﺘﻴﺠﻪ‌ﯼ ﻧﺎﺩﺭﺳﺘﯽ ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎﻳﯽ ﺩﺭﺳﺖ ﺑﺮﺳﻴﻢ؟ ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺍﺑﺘﺪﺍ ﺩﺭ ﻣﻌﺮﻓﯽ ﺍﻳﻦ ﻧﻮﻉ ﺍﺳﺘﺪﻻﻝ ﻣﻐﺎﻟﻄﻪ‌ﺁﻣﻴﺰ ﮔﻔﺘﻴﻢ، ﺍﻳﻦ ﻣﻐﻠﻄﻪ ﺭﺍ ﺩﺭ ﺩﺳﺘﻪ ﻣﻐﺎﻟﻄﻪ ﻣﺮﺗﺒﻂ ﺑﺎ ﻭﺿﻮﺡ ﻃﺒﻘﻪ‌ﺑﻨﺪﯼ ﻣﯽ‌ﮐﻨﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺑﻪ ﺩﻟﻴﻞ ﺯﺑﺎﻥ ﺑﻪ ﮐﺎﺭ ﺭﻓﺘﻪ ﺩﺭ ﺁﻥ ﺧﺪﺷﻪ‌ﺩﺍﺭ ﺷﺪﻩ ﺍﺳﺖ. ﮐﻠﻤﻪ «ﺑﺎﺯ» ﺩﺭ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﺩﻭ ﻣﻌﻨﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﺩﻭ ﻣﻔﻬﻮﻡ ﻣﺠﺰﺍ ﺍﺯ ﻫﻢ ﺍﺭﺟﺎﻉ ﻣﯽ‌ﺩﻫﺪ. ﺑﻪ ﺩﻟﻴﻞ ﻫﻤﻴﻦ ﺩﻭﭘﻬﻠﻮ ﺑﻮﺩﻥ ﺩﺭ ﻭﺍﮊﻩ‌ﻫﺎﯼ ﺑﻪ ﮐﺎﺭ ﺭﻓﺘﻪ، ﺩﺭ ﻭﺍﻗﻊ ﺍﺻﻼ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺍﺯ ﺍﻳﻦ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩ. ﺑﻪ ﺍﻳﻦ ﻧﻮﻉ  ﻣﻐﺎﻟﻄﻪ ﮐﻪ ﺩﺭ ﻣﺜﺎﻝ ﺩﻳﺪﻳﻢ، ﺍﺷﺘﺮﺍﮎ ﻟﻔﻆ ﻧﻴﺰ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﺯﻳﺮﺍ ﻳﮏ ﻭﺍﮊﻩ ﺩﺭ ﺩﻭ ﻣﻌﻨﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺑﻪ ﮐﺎﺭ ﺭﻓﺘﻪ ﺍﺳﺖ. ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺷﮑﻞ ﮐﻠﯽ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﺑﺎﺯﺳﺎﺯﯼ ﮐﻨﻴﻢ ﻭ ﺑﺎﺯ ﺭﺍ ﺩﺭ ﺩﻭ ﻣﻌﻨﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺑﻪ ﮐﺎﺭ ﺑﺒﺮﻳﻢ، ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ‌ﺑﻨﺪﯼ ﺧﻮﺍﻫﻴﻢ ﺭﺳﻴﺪ.

۲- ﺝ، ﻧﻮﻋﯽ ﺩ ﺍﺳﺖ.

۳- ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﻟﻒ، ﺩ ﺍﺳﺖ.

ﺧﺐ ﺑﺎ ﺍﻳﻦ ﺻﻮﺭﺕ‌ﺑﻨﺪﯼ ﺑﻪ ﻭﺿﻮﺡ ﻣﺸﺨﺺ ﺍﺳﺖ ﮐﻪ ﺍﺳﺘﺪﻻﻝ ﺍﺻﻼ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺷﮑﻞ ﺑﮕﻴﺮﺩ ﻭ ﺩﺭ ﻭﺍﻗﻊ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺍﺯ ﺩﻭ ﻣﻘﺪﻣﻪ‌ﺍﯼ ﮐﻪ ﺩﺍﺭﻳﻢ ﺣﺎﺻﻞ ﻧﻤﯽ‌ﺷﻮﺩ. ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﺎﻧﻨﺪ ﺍﻳﻦ ﺑﻪ ﻭﺿﻮﺡ ﻭ ﺯﺑﺎﻥ ﺑﻪ ﮐﺎﺭ ﺭﻓﺘﻪ ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﻣﺮﺗﺒﻂ ﻫﺴﺘﻨﺪ ﺩﺭ ﺯﺑﺎﻥ ﺭﻭﺯﻣﺮﻩ ﺯﻳﺎﺩ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺭﻭﻧﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﺣﺎﺻﻞ ﺍﺯ ﺍﺑﻬﺎﻡ ﺯﺑﺎﻥ ﻧﺎﻣﻴﺪ.

ﺩﺳﺘﻪ ﺩﻳﮕﺮ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻣﻐﻠﻄﻪ‌ﺁﻣﻴﺰ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻧﺎﻣﺮﺑﻮﻁ ﻭ ﺩﻳﮕﺮﯼ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﺧﺎﻟﯽ ﺍﺯ ﻣﺤﺘﻮﺍ ﻫﺴﺘﻨﺪ.

ﺩﺭ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻧﺎﻣﺮﺑﻮﻁ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﺍﺭﺗﺒﺎﻃﯽ ﺑﺎ ﻧﺘﻴﺠﻪ ﻧﺪﺍﺭﻧﺪ ﻭ ﺁﻥ ﺭﺍ ﺣﻤﺎﻳﺖ ﻧﻤﯽ‌ﮐﻨﻨﺪ. ﺍﻳﻦ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺍﻧﻮﺍﻉ ﻣﺨﺘﻠﻒ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﺩﺭ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﻧﺪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﭼﻨﺪ ﺷﮑﻞ ﺍﺯ ﺍﻳﻦ ﻧﻮﻉ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﻴﻢ. ﺩﺭ ﻧﻮﻋﯽ ﺍﺯ ﺍﻳﻦ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﻣﺴﺘﻘﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﻣﻘﺪﻣﻪ‌ﻫﺎ ﺩﺭﺳﺖ ﺑﺎﺷﻨﺪ ﻳﺎ ﻧﻪ، ﺍﺭﺗﺒﺎﻃﯽ ﺑﻪ ﻧﺘﻴﺠﻪ ﻧﺪﺍﺭﻧﺪ ﻭ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺁﻧﻬﺎ ﻧﺘﻴﺠﻪ ﺍﺩﻋﺎ ﺷﺪﻩ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﺣﻤﻠﻪ ﺷﺨﺼﯽ ﻭ ﺗﻮﺳﻞ ﺑﻪ ﻗﺪﺭﺕ ﺍﺯ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻧﺎﻣﺮﺗﺒﻂ ﻫﺴﺘﻨﺪ. ﻣﺜﺎﻝ ﺯﻳﺮ ﺭﺍ ﻣﻼﺣﻈﻪ ﮐﻨﻴﺪ ﺗﺎ ﺑﻴﺸﺘﺮ ﺩﺭ ﻣﻮﺭﺩ ﺍﻳﻦ ﻧﻮﻉ ﻣﻐﺎﻟﻄﻪ ﺑﺪﺍﻧﻴﺪ.

ﺑﻴﻞ: ﻣﻦ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻢ ﮐﻪ ﺳﻘﻂ ﺟﻨﻴﻦ ﺍﺯ ﻧﻈﺮ ﺍﺧﻼﻗﯽ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ.

ﺩﻳﻮﻳﺪ: ﻣﻌﻠﻮﻡ ﺍﺳﺖ ﮐﻪ ﺗﻮ ﺍﻳﻨﻄﻮﺭ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﯽ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻳﮏ ﮐﺸﻴﺶ ﻫﺴﺘﯽ.

ﺑﻴﻞ: ﭘﺲ ﺍﺳﺘﺪﻻﻟﯽ ﮐﻪ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻢ ﭼﻪ ﻣﯽ‌ﺷﻮﺩ؟

ﺩﻳﻮﻳﺪ: ﺁﻥ ﺣﺴﺎﺏ ﻧﻴﺴﺖ. ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﮔﻔﺘﻢ ﺗﻮ ﻳﮏ ﮐﺸﻴﺸﯽ ﻭ ﺑﺎﻳﺪ ﻫﻢ ﺑﮕﻮﻳﯽ ﮐﻪ ﺳﻘﻂ ﺟﻨﻴﻦ ﮐﺎﺭ ﺑﺪﯼ ﺍﺳﺖ. ﺩﺭ ﺿﻤﻦ ﺗﻮ ﻧﻮﮐﺮ ﭘﺎﭖ ﻫﺴﺘﯽ ﻭ ﺑﻨﺎﺑﺮﺍﻳﻦ ﻧﻤﯽ‌ﺗﻮﺍﻧﻢ ﺁﻧﭽﻪ ﺭﺍ ﮐﻪ ﻣﯽ‌ﮔﻮﻳﯽ ﺑﺎﻭﺭ ﮐﻨﻢ.

ﺧﺐ ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻣﺘﻮﺟﻪ ﺷﺪﻳﺪ، ﺩﺭ ﺍﻳﻦ ﻣﺒﺎﺣﺜﻪ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﺩﻳﻮﻳﺪ ﮐﺎﻣﻼ ﻣﻐﺎﻟﻄﻪ‌ﺁﻣﻴﺰ ﻫﺴﺘﻨﺪ ﻭ ﺑﻴﺶ ﺍﺯ ﺁﻧﮑﻪ ﺍﺳﺘﺪﻻﻝ ﺑﺎﺷﻨﺪ ﺣﻤﻠﻪ ﺷﺨﺼﯽ ﺑﻪ ﺣﺴﺎﺏ ﻣﯽ‌ﺁﻳﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺍﺯ ﻣﻐﺎﻟﻄﻪ‌ﯼ ﺣﻤﻠﻪ ﺷﺨﺼﯽ، ﺩﻳﻮﻳﺪ ﻣﺜﺎﻝ ﻣﺎ ﺑﻪ ﺍﻳﻦ ﻣﺘﻮﺳﻞ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﭼﻮﻥ ﻃﺮﻑ ﻣﻘﺎﺑﻠﺶ ﺩﺭ ﺑﺤﺚ ﻋﻀﻮ ﮔﺮﻭﻩ ﺧﺎﺻﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﺜﺎﻝ ﺟﺎﻣﻌﻪ ﮐﺸﻴﺸﺎﻥ ﺍﺳﺖ، ﮐﻞ ﺍﺩﻋﺎﯼ ﺍﻭ ﺭﺍ ﺯﻳﺮ ﺳﻮﺍﻝ ﻣﯽ‌ﺑﺮﺩ. ﺍﻳﻦ ﻧﻮﻉ ﺍﺳﺘﺪﻻﻝ ﻫﻴﭻ ﺗﻔﺎﻭﺗﯽ ﺑﺎ ﺷﮑﻞ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﻧﮋﺍﺩﭘﺮﺳﺘﯽ، ﺳﮑﺴﻴﺴﻢ ﻳﺎ ﺩﻳﻦ‌ﻫﺮﺍﺳﯽ ﻧﺪﺍﺭﺩ ﮐﻪ ﺻﺮﻓﺎ ﺑﻪ ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﺷﺨﺺ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﻧﮓ ﭘﻮﺳﺖ ﺧﺎﺹ، ﺟﻨﺴﻴﺖ ﺧﺎﺹ ﻳﺎ ﻣﺬﻫﺐ ﺧﺎﺻﯽ ﺩﺍﺭﺩ ﮐﻞ ﺍﺩﻋﺎﯼ ﺍﻭ ﺭﺍ ﺯﻳﺮ ﺳﻮﺍﻝ ﻣﯽ‌ﺑﺮﻧﺪ.

ﺍﮔﺮ ﺑﻪ ﺍﻃﺮﺍﻓﻤﺎﻥ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺗﻮﺟﻪ ﮐﻨﻴﻢ، ﻣﺜﺎﻝ‌ﻫﺎﯼ ﻓﺮﺍﻭﺍﻧﯽ ﺍﺯ ﺍﻳﻦ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ ﻭ ﺣﺘﯽ ﺧﻮﺩﻣﺎﻥ ﻧﻴﺰ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺮﺗﮑﺐ ﺁﻥ ﺷﻮﻳﻢ.

ﺩﺭ ﻧﻮﻉ ﺩﻳﮕﺮﯼ ﺍﺯ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺯﻣﺮﻩ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎﯼ ﻣﻐﻠﻄﻪ‌ﺁﻣﻴﺰ ﺑﻪ ﺩﻟﻴﻞ ﺳﺎﺧﺘﺎﺭ ﺟﺎ ﺩﺍﺭﻧﺪ، ﺍﺯ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺧﺎﻟﯽ ﺍﺯ ﻣﺤﺘﻮﺍ ﻧﺎﻡ ﺑﺮﺩﻳﻢ. ﺍﻳﻦ ﻧﻮﻉ ﺍﺳﺘﺪﻻﻝ‌‌ ﺑﻪ ﺻﻮﺭﺗﯽ ﭘﻴﺶ ﻣﯽ‌ﺭﻭﺩ ﮐﻪ ﺩﺭ ﻣﻘﺪﻣﻪ‌ﻫﺎﯼ ﺁﻥ ﺍﺯ ﻧﺘﻴﺠﻪ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺩﺭﺳﺘﯽ ﺁﻥ ﻓﺮﺽ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﺩ. ﺑﺮﺍﯼ ﻧﻤﻮﻧﻪ ﻣﺼﺎﺩﺭﻩ ﺑﻪ ﻣﻄﻠﻮﺏ ﻧﻮﻋﯽ ﻣﻐﺎﻟﻄﻪ ﺧﺎﻟﯽ ﺍﺯ ﻣﺤﺘﻮﺍﺳﺖ، ﺩﻗﻴﻘﺎ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﻫﻴﭻ ﮔﻮﻧﻪ ﭘﻴﺸﺮﻓﺘﯽ ﻓﺮﺍﺗﺮ ﺍﺯ ﻓﺮﺽ‌ﻫﺎﯼ ﺧﻮﺩﺵ ﻧﺪﺍﺭﺩ. ﺩﺭ ﻭﺍﻗﻊ ﺍﻳﻦ ﻧﻮﻉ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺑﻪ ﺍﺻﻄﻼﺡ ﻣﻨﻄﻖ‌ﺩﺍﻥ‌ﻫﺎ ﺩﻭﺭ ﺑﺎﻃﻞ ﻫﺴﺘﻨﺪ. ﻣﺜﺎﻝ ﺯﻳﺮ ﺭﺍ ﺑﺒﻴﻨﻴﺪ ﺗﺎ ﺩﺭﮎ ﺑﻬﺘﺮﯼ ﺍﺯ ﺍﻳﻦ ﻧﻮﻉ ﺍﺳﺘﺪﻻﻝ ﻣﻐﺎﻟﻄﻪ‌ﺁﻣﻴﺰ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺣﺴﻦ: ﻫﻤﻪ ﺯﻧﺎﻥ ﻣﻮﺟﻮﺩﺍﺗﯽ ﺍﺣﺴﺎﺳﺎﺗﯽ ﻭ ﻏﻴﺮﻗﺎﺑﻞ ﺍﻋﺘﻤﺎﺩ ﻫﺴﺘﻨﺪ.

ﺳﺎﺭﺍ: ﺑﺮ ﻣﺒﻨﺎﯼ ﮐﺪﺍﻡ ﺩﻟﻴﻞ ﭼﻨﻴﻦ ﺍﺩﻋﺎﻳﯽ ﺩﺍﺭﯼ؟

ﺣﺴﻦ: ﻭﺍﺿﺢ ﺍﺳﺖ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺯﻥ ﻫﺴﺘﻨﺪ.

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺍﻳﻦ ﺍﺳﺘﺪﻻﻝ ﻧﺘﻴﺠﻪ ﺭﺍ ﺩﺭ ﻣﻘﺪﻣﻪ‌ﻫﺎﻳﺶ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩﻩ ﺍﺳﺖ ﻭ ﭼﻴﺰﯼ ﻓﺮﺍﺗﺮ ﺍﺯ ﺍﺩﻋﺎﻳﯽ ﮐﻪ ﺍﺯ ﻫﻤﺎﻥ ﺍﺑﺘﺪﺍ ﻣﻄﺮﺡ ﮐﺮﺩﻩ ﺑﺮﺍﯼ ﺍﺭﺍﺋﻪ ﻧﺪﺍﺭﺩ. ﻣﻌﻤﻮﻻ ﺍﺩﻋﺎﻫﺎﯼ ﺯﻥ‌ﺳﺘﻴﺰﺍﻧﻪ ﻳﺎ ﻫﺮ ﺷﮑﻞ ﺩﻳﮕﺮﯼ ﺍﺯ ﺗﻌﻤﻴﻢ‌ﻫﺎﯼ ﺷﺘﺎﺑﺰﺩﻩ، ﺑﻪ ﻧﻮﻋﯽ ﺧﺎﻟﯽ ﺍﺯ ﻣﺤﺘﻮﺍ ﻭ ﺣﺎﻭﯼ ﻣﻐﺎﻟﻄﻪ ﻣﺼﺎﺩﺭﻩ ﺑﻪ ﻣﻄﻠﻮﺏ ﻫﺴﺘﻨﺪ.

 

ﮐﺎﺭﮐﺮﺩﻫﺎﯼ ﻣﻐﺎﻟﻄﻪ
ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻣﻨﻄﻘﯽ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﻣﻌﻴﺎﺭﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﻃﺒﻘﻪ‌ﺑﻨﺪﯼ ﮐﺮﺩ. ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺑﺮ ﺭﻭﯼ ﮐﺎﺭﮐﺮﺩ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺗﻤﺮﮐﺰ ﮐﻨﻴﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎ ﺭﺍ ﺑﻪ ﭼﻨﺪ ﺩﺳﺘﻪ‌ﯼ ﻣﻐﺎﻟﻄﻪ ﺩﺭ ﻣﻘﺎﻡ ﺍﺩﻋﺎ، ﻣﻐﺎﻟﻄﻪ ﺩﺭ ﻣﻘﺎﻡ ﻧﻘﺪ، ﻣﻐﺎﻟﻄﻪ ﺩﺭ ﻣﻘﺎﻡ ﺩﻓﺎﻉ ﺗﻘﺴﻴﻢ ﮐﻨﻴﻢ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﮐﻪ ﺩﺭ ﺍﺑﺘﺪﺍﯼ ﺍﻣﺮ ﻳﮏ ﺍﺩﻋﺎﯼ ﻣﻐﻠﻄﻪ‌ﺁﻣﻴﺰ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺑﺎﺷﺪ ﻭ ﻣﺎ ﺑﺨﻮﺍﻫﻴﻢ ﺁﻥ ﺭﺍ ﮐﺸﻒ ﮐﻨﻴﻢ. ﻳﺎ ﺍﻳﻨﮑﻪ ﺩﺭ ﻳﮏ ﺑﺤﺚ ﻭ ﺩﺭ ﭘﺎﺳﺦ ﺑﻪ ﺍﺳﺘﺪﻻﻝ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺁﮔﺎﻫﺎﻧﻪ ﻳﺎ ﻏﻴﺮﺁﮔﺎﻫﺎﻧﻪ ﻣﺮﺗﮑﺐ ﻣﻐﺎﻟﻄﻪ ﺷﻮﻳﻢ ﻭ ﮐﺎﺭﮐﺮﺩ ﺩﻳﮕﺮ ﻫﻢ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻣﻘﺎﻡ ﻧﻘﺪ ﻳﺎ ﺣﻤﻠﻪ ﺑﻪ ﻣﻮﺿﻊ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﻣﻐﺎﻟﻄﻪ ﮐﻨﻴﻢ.

ﺑﺮﺍﯼ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ ﻣﻘﺎﺻﺪ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺟﺰﺋﻴﺎﺕ ﺍﻧﻮﺍﻉ ﻣﻐﺎﻟﻄﻪ ﺁﺷﻨﺎ ﺑﺎﺷﻴﻢ ﺗﺎ ﺑﺘﻮﺍﻧﻴﻢ ﻣﺸﮑﻼﺕ ﺍﺳﺘﺪﻻﻝ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﮐﻨﻴﻢ، ﭼﻪ ﻭﻗﺘﯽ ﺧﻮﺩﻣﺎﻥ ﺩﺭ ﺣﺎﻝ ﺍﺭﺍﺋﻪ ﺁﻥ ﻫﺴﺘﻴﻢ ﻭ ﭼﻪ ﻭﻗﺘﯽ ﮐﻪ ﺩﻳﮕﺮﺍﻥ ﻣﺎ ﺭﺍ ﻧﻘﺪ ﻣﯽ ﮐﻨﻨﺪ ﻳﺎ ﻭﻗﺘﯽ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻧﻘﺪ ﮐﻨﻴﻢ. ﻣﻐﺎﻟﻄﻪ ﺩﺭ ﺍﺳﺘﺪﻻﻝ ﺑﺴﻴﺎﺭ ﺧﻄﺮﻧﺎﮎ ﺍﺳﺖ، ﺍﻣﺎ ﺷﻨﺎﺧﺘﻦ ﻣﻐﺎﻟﻄﻪ‌ﻫﺎﯼ ﻣﻨﻄﻘﯽ ﺑﺴﻴﺎﺭ ﺳﻮﺩﻣﻨﺪ ﺍﺳﺖ.
