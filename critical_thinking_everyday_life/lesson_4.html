ﺩﺭﺱ ﺳﻮﻡ- ﻫﺪﻑ ﻫﺎ ﻭ ﮐﺎﺭﺑﺮﺩﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ

ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺩﻳﺪﻳﻢ، ﺍﺳﺘﺪﻻﻝ ﺍﺑﺰﺍﺭ ﺗﻔﮑﺮ ﺍﻧﺘﻘﺎﺩﯼ ﺍﺳﺖ. ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻧﺪﻳﺸﻪ ﻣﻨﻄﻘﯽ ﻫﺴﺘﻴﻢ، ﺑﺎﻳﺪ ﻣﺎﻧﻨﺪ ﻳﮏ ﻧﺠﺎﺭ ﻣﺎﻫﺮ ﺍﺳﺘﺪﻻﻝ‌ﻫﺎ ﺭﺍ ﻫﻤﭽﻮﻥ ﺟﻌﺒﻪ ‌ﺍﺑﺰﺍﺭ ﻫﻤﻴﺸﻪ ﺩﺭ ﺩﺳﺘﺮﺱ ﺧﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭﺳﺖ ﻭ ﺑﻪ ﻣﻮﻗﻊ ﺑﻪ ﮐﺎﺭ ﺑﮕﻴﺮﻳﻢ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺑﺮﺍﯼ ﻓﻬﻤﻴﺪﻥ ﭼﮕﻮﻧﮕﯽ ﺍﺳﺘﺪﻻﻝ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﺒﻴﻨﻴﻢ، ﺍﻳﻦ ﺍﺑﺰﺍﺭ ﺑﻪ ﭼﻪ ﮐﺎﺭ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺍﻓﺮﺍﺩ ﻭﻗﺘﯽ ﺍﺳﺘﺪﻻﻟﯽ ﺭﺍ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﺩﻫﻨﺪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺭﺳﻴﺪﻥ ﺑﻪ ﭼﻪ ﻫﺪﻓﯽ ﻫﺴﺘﻨﺪ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﻫﺪﻑ‌ﻫﺎﯼ ﺍﺭﺍﺋﻪ ﺍﺳﺘﺪﻻﻝ ﺑﭙﺮﺩﺍﺯﻳﻢ ﺯﻳﺮﺍ ﮐﺎﺭﺑﺮﺩﻫﺎ، ﺍﻫﺪﺍﻑ ﻭ ﺍﺳﺘﻔﺎﺩﻩ‌ﻫﺎ ﺑﺮﺍﯼ ﻓﻬﻤﻴﺪﻥ ﻳﮏ ﻣﻔﻬﻮﻡ ﻳﺎ ﺍﺑﺰﺍﺭ ﺑﺴﻴﺎﺭ ﺍﻫﻤﻴﺖ ﺩﺍﺭﻧﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﮔﺮ ﺷﻴﺌﯽ ﺭﺍ ﺩﺭ ﻳﮏ ﻣﻨﻄﻘﻪ ﺑﺎﺳﺘﺎﻧﯽ ﮐﺸﻒ ﮐﺮﺩﻩ ﺑﺎﺷﻴﻢ، ﺗﻨﻬﺎ ﺩﺭ ﺻﻮﺭﺗﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻔﻬﻤﻴﻢ ﭼﻴﺴﺖ ﮐﻪ ﺑﺪﺍﻧﻴﻢ ﺑﺮﺍﯼ ﭼﻪ ﻣﻨﻈﻮﺭﯼ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﺮﺩﻧﺪ.

ﺩﺭ ﻣﻮﺭﺩ ﺍﺳﺘﺪﻻﻝ ﻫﻢ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺑﺪﺍﻧﻴﻢ ﭼﺮﺍ ﻳﮏ ﻧﻔﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻮﺩﺵ ﺭﺍ ﺑﻪ ﺯﺣﻤﺖ ﺑﻴﻨﺪﺍﺯﺩ ﻭ ﺍﺳﺘﺪﻻﻟﯽ ﺭﺍ ﺍﺭﺍﺋﻪ ﺩﻫﺪ ﺩﺭ ﺣﺎﻟﯽ‌ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﺘﻴﺠﻪ ﺭﺍ ﺍﺯ ﻫﻤﺎﻥ ﺍﺑﺘﺪﺍ ﺑﻴﺎﻥ ﮐﻨﺪ؟

 

ﺍﻗﻨﺎﻉ ﻣﺨﺎﻃﺐ
ﻓﺮﺽ ﮐﻨﻴﻢ ﻓﺮﻭﺷﻨﺪﻩ‌ﺍﯼ ﺑﺨﻮﺍﻫﺪ ﺁﺧﺮﻳﻦ ﻣﺪﻝ ﻳﮏ ﺍﺗﻮﻣﻮﻣﺒﻴﻞ ﺭﺍ ﺑﻪ ﻣﺎ ﺑﻔﺮﻭﺷﺪ. ﺧﺐ ﻣﺎ ﺻﺮﻓﺎ ﺑﺎ ﺣﺮﻑ ﺍﻭ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻧﺨﻮﺍﻫﻴﻢ ﮐﺮﺩ. ﺍﻣﺎ ﺍﮔﺮ ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ ﺟﻤﻠﻪ ﺩﻻﻳﻠﯽ ﻫﻢ ﺍﺭﺍﺋﻪ ﺩﻫﺪ ﮐﻪ ﻣﺜﻼ ﺍﻳﻦ ﻣﺪﻝ ﭘﮋﻭ ﺩﻧﺪﻩ ﺍﺗﻮﻣﺎﺗﻴﮏ ﺩﺍﺭﺩ ﻳﺎ ﻣﺠﻬﺰ ﺑﻪ ﺩﺳﺘﮕﺎﻩ ﺟﯽ‌ﭘﯽ‌ﺍﺱ ﺍﺳﺖ ﻳﺎ ﺧﺪﻣﺎﺕ ﭘﺲ ﺍﺯ ﻓﺮﻭﺵ ﺁﻥ ﺳﻪ ﺳﺎﻝ ﺑﻴﺸﺘﺮ ﺍﺯ ﻣﺪﻝ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺍﺳﺖ ﻭ ﺩﺭ ﺿﻤﻦ ﺍﮔﺮ ﻫﻤﻴﻦ ﺣﺎﻻ ﻣﻌﺎﻣﻠﻪ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﺍﻳﻦ ﻓﺮﻭﺷﻨﺪﻩ ﺗﺨﻔﻴﻒ ﺧﻮﺑﯽ ﻣﯽ‌ﺩﻫﺪ، ﺍﺣﺘﻤﺎﻝ ﺍﻳﻨﮑﻪ ﻗﺎﻧﻊ ﺑﻪ ﺧﺮﻳﺪﻥ ﺍﻳﻦ ﺍﺗﻮﻣﺒﻴﻞ ﺷﻮﻳﻢ ﺯﻳﺎﺩ ﺍﺳﺖ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﻳﮑﯽ ﺍﺯ ﻫﺪﻑ‌ﻫﺎﯼ ﺍﺳﺘﺪﻻﻝ ﻗﺎﻧﻊ ﮐﺮﺩﻥ ﻳﺎ ﺍﻗﻨﺎﻉ ﻣﺨﺎﻃﺐ ﺍﺳﺖ. ﻗﺎﻧﻊ‌ﺳﺎﺯﯼ ﻳﻌﻨﯽ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﺑﺎﻭﺭ ﭼﻴﺰﯼ ﻳﺎ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﮐﻪ ﺩﺭ ﺣﺎﻟﺖ ﻋﺎﺩﯼ ﺁﻥ ﺭﺍ ﻗﺒﻮﻝ ﻧﻤﯽ‌ﮐﻨﻨﺪ.

ﭘﺲ ﻓﺮﻭﺷﻨﺪﻩ ﺍﺗﻮﻣﺒﻴﻞ ﻣﻮﺭﺩ ﻧﻈﺮ ﻣﺎ ﻫﻢ ﻫﻤﻴﻦ ﻫﺪﻑ ﺭﺍ ﺩﺍﺭﺩ ﻭ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻭﺿﻌﻴﺖ ﺫﻫﻨﯽ ﻣﺎ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﭼﻴﺰﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻔﺮﻭﺷﺪ ﻋﻮﺽ ﮐﻨﺪ. ﺍﻟﺒﺘﻪ ﺍﻓﺮﺍﺩ ﻣﺨﺘﻠﻒ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺨﻮﺍﻫﻨﺪ ﭼﻴﺰﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﺑﻪ ﻣﺎ ﺑﻔﺮﻭﺷﻨﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻫﺮ ﭼﻴﺰﯼ ﺑﺎﺷﺪ، ﺍﺯ ﻣﺎﺷﻴﻦ ﮔﺮﻓﺘﻪ ﺗﺎ ﺑﻴﻤﻪ ﻋﻤﺮ ﺗﺎ ﮐﺘﺎﺏ ﺩﺍﺳﺘﺎﻥ ﻳﺎ ﺣﺘﯽ ﻋﻀﻮﻳﺖ ﺩﺭ ﻳﮏ ﺣﺰﺏ ﺳﻴﺎﺳﯽ ﻳﺎ ﮔﺮﻭﻩ ﻣﺬﻫﺒﯽ.

ﭘﺲ ﻳﮑﯽ ﺍﺯ  ﻫﺪﻑ‌ﻫﺎﯼ ﻣﻬﻢ ﺍﺳﺘﺪﻻﻝ ﺗﻼﺵ ﺑﺮﺍﯼ ﺗﺎﺛﻴﺮ ﺑﺮ ﺟﻬﺎﻥ ﻭ ﺗﻐﻴﻴﺮ ﭼﻴﺰﯼ ﺩﺭ ﺁﻥ ﺍﺳﺖ، ﺧﻮﺍﻩ ﺑﻪ ﻃﻮﺭ ﻓﻴﺰﻳﮑﯽ ﻭ ﺧﻮﺍﻩ ﺩﺭ ﺫﻫﻦ ﺍﻓﺮﺍﺩ.

ﺳﻮﺍﻝ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﻳﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﮐﺴﯽ ﺭﺍ ﺑﺎ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﻗﺎﻧﻊ ﮐﺮﺩ، ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺁﻥ ﻓﺮﺩ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﺭﺍ ﺑﻔﻬﻤﺪ؟ ﺍﮔﺮ ﻣﺨﺎﻃﺐ ﻳﺎ ﻃﺮﻑ ﺑﺤﺚ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﺭﺍ ﻧﻔﻬﻤﺪ، ﻣﺜﻼ ﻭﻗﺘﯽ ﺑﻪ ﺯﺑﺎﻧﯽ ﺍﺭﺍﺋﻪ ﺷﻮﺩ ﮐﻪ ﺍﻭ ﻧﻤﯽ‌ﺩﺍﻧﺪ، ﺁﻧﮕﺎﻩ ﺍﺳﺘﺪﻻﻝ ﻣﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻣﻨﺠﺮ ﺑﻪ ﺗﻐﻴﻴﺮ ﻋﻘﻴﺪﻩ ﻳﺎ ﻧﻈﺮ ﺍﻭ ﺷﻮﺩ. ﭘﺲ ﺍﺳﺘﺪﻻﻝ ﻗﺎﻧﻊ‌ﮐﻨﻨﺪﻩ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ.

 

ﺗﻮﺿﻴﺢ، ﺗﻮﺟﻴﻪ ﻳﺎ ﺍﻗﻨﺎﻉ؟
ﺍﻣﺎ ﺍﺳﺘﺪﻻﻝ ﻫﻤﻴﺸﻪ ﻫﻢ ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﻧﻈﺮ ﺩﻳﮕﺮﺍﻥ ﻧﻴﺴﺖ. ﮔﺎﻫﯽ ﺩﻻﻳﻠﯽ ﺭﺍ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﺗﺎ ﻋﻘﻴﺪﻩ ﻭ ﺑﺎﻭﺭ ﺩﻳﮕﺮﺍﻥ ﻳﺎ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﺗﻮﺟﻴﻪ ﮐﻨﻴﻢ.

ﺗﻮﺟﻴﻪ ﺑﻪ ﻣﻌﻨﺎﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺩﻟﻴﻞ ﺑﺮﺍﯼ ﺑﺎﻭﺭﻫﺎﻳﻤﺎﻥ ﺍﺳﺖ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﮔﺮ ﺩﻭﺳﺘﻤﺎﻥ ﺑﺨﻮﺍﻫﺪ ﻣﺎﺷﻴﻦ ﺑﺨﺮﺩ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺍﻭ ﻣﺪﻝ ﺧﺎﺻﯽ ﺭﺍ ﭘﻴﺸﻨﻬﺎﺩ ﮐﻨﻴﻢ ﺍﻣﺎ ﻭﺍﻗﻌﺎ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﻫﺮ ﻗﻴﻤﺘﯽ ﺷﺪﻩ ﺍﻭ ﺭﺍ ﻗﺎﻧﻊ ﮐﻨﻴﻢ. ﺑﻠﮑﻪ ﺑﺎ ﺍﺭﺍﺋﻪ ﺩﻻﻳﻞ ﺻﺮﻓﺎ ﺗﻮﺟﻴﻪ ﮐﻨﻴﻢ ﮐﻪ ﺁﻥ ﺍﺗﻮﻣﺒﻴﻞ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﺍﻫﺪﺍﻑ ﺍﻭ ﺍﻧﺘﺨﺎﺏ ﺧﻮﺑﯽ ﺑﺎﺷﺪ ﻭ ﺩﺭ ﻧﻬﺎﻳﺖ ﺍﻭ ﺧﻮﺩﺵ ﺗﺼﻤﻴﻢ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﻣﺎ ﺍﺯ ﻫﺮ ﺗﺼﻤﻴﻤﯽ ﮐﻪ ﺩﻭﺳﺘﻤﺎﻥ ﺑﮕﻴﺮﺩ ﺍﺳﺘﻘﺒﺎﻝ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﻣﺎ ﻓﺮﻭﺷﻨﺪﻩ ﺍﺗﻮﻣﺒﻴﻞ ﺗﻨﻬﺎ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﮐﻪ ﺍﻭ ﺭﺍ ﻗﺎﻧﻊ ﮐﻨﺪ. ﭘﺲ ﻫﺪﻑ‌ﻫﺎﯼ ﻣﺎ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﻣﺎ ﺑﻪ ﺩﻧﺒﺎﻝ ﺗﻮﺟﻴﻪ ﻫﺴﺘﻴﻢ ﻭ ﻓﺮﻭﺷﻨﺪﻩ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻗﻨﺎﻉ.

ﺗﻮﺟﻴﻪ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺩﻻﻳﻞ ﺧﻮﺑﯽ ﺍﺭﺍﺋﻪ ﺩﻫﺪ. ﺍﻗﻨﺎﻉ ﺗﻼﺵ ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﻋﻘﺎﻳﺪ ﻳﺎ ﺭﻓﺘﺎﺭ ﺩﻳﮕﺮﺍﻥ ﺩﺍﺭﺩ ﻭ ﺩﻻﻳﻞ ﺑﺪ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻨﺪ. ﺍﻟﺒﺘﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻫﺮ ﺩﻭﯼ ﺍﻳﻦ ﺍﻫﺪﺍﻑ ﺭﺍ ﻫﻤﺰﻣﺎﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺍﻳﻦ ﻣﺴﺎﺋﻞ ﺭﺍ ﭘﻴﭽﻴﺪﻩ ﻣﯽ‌ﮐﻨﺪ.

ﻋﻼﻭﻩ ﺑﺮ ﺍﻫﺪﺍﻑ ﻗﺒﻠﯽ، ﺍﺳﺘﺪﻻﻝ ﺑﺮﺍﯼ ﺗﻮﺿﻴﺢ ﭼﻴﺰﻫﺎ ﻧﻴﺰ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺭﻭﻧﺪ. ﺍﻣﺎ ﺗﻮﺿﻴﺢ ﺑﻪ ﭼﻪ ﻣﻌﻨﺎﺳﺖ؟ ﻣﺎ ﻫﻤﻮﺍﺭﻩ ﺩﺭ ﺣﺎﻝ ﺗﻮﺿﻴﺢ ﻫﺴﺘﻴﻢ. ﻣﺜﻼ ﺗﻮﺿﻴﺢ ﺍﻳﻨﮑﻪ ﭼﺮﺍ ﺗﻴﻢ ﻓﻮﺗﺒﺎﻝ ﻣﻮﺭﺩ ﻋﻼﻗﻪ‌ﻣﺎﻥ ﺩﺭ ﺟﺎﻡ ﺣﺬﻓﯽ ﻣﻘﺎﻣﯽ ﻧﻴﺎﻭﺭﺩ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺩﻻﻳﻠﯽ ﻣﺎﻧﻨﺪ ﻗﻮﯼ ﺑﻮﺩﻥ ﺣﺮﻳﻒ، ﺿﻌﻒ‌ﻫﺎﯼ ﺳﺮﻣﺮﺑﯽ، ﺍﺷﺘﺒﺎﻫﺎﺕ ﺩﺍﻭﺭﯼ ﻭ ﻳﺎ ﺣﺘﯽ ﺑﺪﺷﺎﻧﺴﯽ ﺩﺭ ﺿﺮﺑﻪ‌ﻫﺎﯼ ﭘﻨﺎﻟﺘﯽ ﺍﺷﺎﺭﻩ ﮐﻨﻴﻢ.

ﺗﻮﺿﻴﺢ ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﺭﺍﺋﻪ ﺩﻟﻴﻞ ﻳﺎ ﺩﻻﻳﻠﯽ ﺍﺳﺖ ﮐﻪ ﭼﺮﺍ ﻭﺍﻗﻌﻪ‌ﺍﯼ ﺭﺥ ﺩﺍﺩﻩ ﺍﺳﺖ ﻳﺎ ﺍﻳﻨﮑﻪ ﭼﺮﺍ ﺩﻳﺪﮔﺎﻫﯽ ﺩﺭﺳﺖ ﺍﺳﺖ.

ﺗﻔﺎﻭﺕ ﺗﻮﺿﻴﺢ ﺑﺎ ﺗﻮﺟﻴﻪ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻭﻗﺘﯽ ﭼﻴﺰﯼ ﺭﺍ ﺗﻮﺟﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ، ﻣﺨﺎﻃﺐ ﻣﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺁﻥ ﺭﺍ ﺑﺎﻭﺭ ﻧﮑﻨﺪ. ﺍﻣﺎ ﻭﻗﺘﯽ ﭼﻴﺰﯼ ﺭﺍ ﺗﻮﺿﻴﺢ ﻣﯽ‌ﺩﻫﻴﻢ، ﻫﻢ ﻣﺎ ﻭ ﻫﻢ ﻣﺨﺎﻃﺐ‌ﻣﺎﻥ ﻓﺮﺽ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﮐﻪ ﻫﺮ ﺩﻭ ﺑﻪ ﺁﻧﭽﻪ ﮐﻪ ﺩﺭ ﺗﻼﺵ ﺑﺮﺍﯼ ﺗﻮﺿﻴﺤﺶ ﻫﺴﺘﻴﻢ، ﺑﺎﻭﺭ ﺩﺍﺭﻳﻢ. ﻳﻌﻨﯽ ﺁﻥ ﻭﺍﻗﻌﻪ ﺭﺥ ﺩﺍﺩﻩ ﺍﺳﺖ ﻳﺎ ﺩﻳﺪﮔﺎﻫﯽ ﺩﺭﺳﺖ ﺍﺳﺖ ﻭ ﻣﺎ ﺑﻪ ﺩﻧﺒﺎﻝ ﺗﻮﺿﻴﺢ ﺩﻻﻳﻠﯽ ﺑﺮﺍﯼ ﺁﻥ ﻫﺴﺘﻴﻢ.

ﺳﻮﺍﻝ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﻳﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﮐﺴﯽ ﺭﺍ ﻗﺎﻧﻊ ﮐﻨﻴﻢ ﺗﺎ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﺪ ﺑﺪﻭﻥ ﺁﻥ‌ﮐﻪ ﺗﻮﺿﻴﺢ ﺩﻫﻴﻢ ﭼﺮﺍ ﺁﻥ ﻧﺘﻴﺠﻪ ﺩﺭﺳﺖ ﺍﺳﺖ؟ ﺑﻠﻪ! ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﮐﺴﯽ ﺭﺍ ﺑﺎ ﻳﮏ ﺍﺳﺘﺪﻻﻝ ﺑﺪ ﻗﺎﻧﻊ ﮐﻨﻴﻢ، ﺍﻣﺎ ﺍﺳﺘﺪﻻﻝ ﺑﺪ ﺗﻮﺿﻴﺢ ﻧﻤﯽ‌ﺩﻫﺪ ﭼﺮﺍ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﮐﻪ ﮔﺮﻓﺘﻪ‌ﺍﻳﻢ ﺩﺭﺳﺖ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﺪﻑ ﺗﻮﺿﻴﺢ ﺍﻗﻨﺎﻉ ﻳﺎ ﺗﻮﺟﻴﻪ ﻧﻴﺴﺖ. ﻫﺪﻑ ﺍﺯ ﺗﻮﺿﻴﺢ ﺍﻓﺰﺍﻳﺶ ﺩﺭﮎ ﻭ ﻓﻬﻢ ﺍﺳﺖ، ﻳﻌﻨﯽ ﮐﻤﮏ ﺑﻪ ﺍﻳﻨﮑﻪ ﺍﻓﺮﺍﺩ ﺑﻔﻬﻤﻨﺪ ﭼﺮﺍ ﻳﮏ ﭼﻴﺰﯼ ﺩﺭﺳﺖ ﺍﺳﺖ.

 

ﺍﻧﻮﺍﻉ ﺗﻮﺿﻴﺢ ﺍﺯ ﺩﻳﺪﮔﺎﻩ ﺍﺭﺳﻄﻮ
ﺍﺭﺳﻄﻮ ﭼﻬﺎﺭ ﻧﻮﻉ ﺗﻮﺿﻴﺢ ﺑﺮﺍﯼ ﭼﻴﺰﻫﺎ ﻣﻌﺮﻓﯽ ﮐﺮﺩﻩ ﺍﺳﺖ:

  • ﺗﻮﺿﻴﺢ ﻋﻞّﯼ: ﺩﺭ ﺗﻮﺿﻴﺢ ﺍﻳﻨﮑﻪ ﭼﺮﺍ ﻳﮏ ﭘﻞ ﻓﺮﻭﺭﻳﺨﺖ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﮕﻮﻳﻴﻢ ﺯﻟﺰﻟﻪ ﻋﻠﺖ ﺧﺮﺍﺏ ﺷﺪﻥ ﭘﻞ ﺑﻮﺩﻩ ﺍﺳﺖ. ﺍﻳﻦ ﻧﻮﻉ ﺗﻮﺿﻴﺢ ﺑﻪ ﻋﻠﺖ‌ﻫﺎﯼ ﺭﺥ ﺩﺍﺩﻥ ﻳﮏ ﻭﺍﻗﻌﻪ ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﻟﻴﻞ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﺪ.
  • ﺗﻮﺿﻴﺢ ﻏﺎﻳﯽ ﻳﺎ ﻧﺘﻴﺠﻪ‌ﻧﮕﺮ: ﺩﺭ ﭘﺎﺳﺦ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﮐﻪ ﭼﺮﺍ ﻣﺎﺩﺭﺑﺰﺭﮒ ﺑﻪ ﻣﻴﺪﺍﻥ ﻣﻴﻮﻩ ﻭ ﺗﺮﻩ‌ﺑﺎﺭ ﺭﻓﺖ؟ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﮕﻮﻳﻴﻢ ﺑﺮﺍﯼ ﺍﻳﻦ ﺭﻓﺖ ﮐﻪ ﮔﻮﺟﻪ ﻓﺮﻧﮕﯽ ﺑﺨﺮﺩ. ﺍﻳﻦ ﻧﻮﻉ ﺗﻮﺿﻴﺢ ﻫﺪﻑ ﺍﺯ ﺍﻧﺠﺎﻡ ﻳﮏ ﺳﻠﺴﻠﻪ ﺍﺯ ﺍﻋﻤﺎﻝ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﻟﻴﻞ ﺁﻧﻬﺎ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﺩﻫﺪ.
  • ﺗﻮﺿﻴﺢ ﺻﻮﺭﯼ ﻳﺎ ﺷﮑﻠﯽ: ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺩﺭ ﭘﺎﺳﺦ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﮐﻪ ﭼﺮﺍ ﺍﻳﻦ ﭘﻴﭻ ﺑﺎ ﭘﻴﭻ‌ﮔﻮﺷﺘﯽ ﭼﻬﺎﺭﺳﻮ ﺑﺎﺯ ﻧﻤﯽ‌ﺷﻮﺩ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﮕﻮﻳﻴﻢ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺗﻨﻬﺎ ﻳﮏ ﺷﻴﺎﺭ ﺩﺍﺭﺩ ﺍﻳﻦ ﺍﺑﺰﺍﺭ ﻣﻨﺎﺳﺐ ﺁﻥ ﻧﻴﺴﺖ. ﺍﻳﻦ ﻧﻮﻉ ﺗﻮﺿﻴﺢ ﺑﻪ ﻓﺮﻡ ﻳﺎ ﺷﮑﻞ ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﻟﻴﻞ ﻳﮏ ﻭﺍﻗﻌﻪ ﺗﮑﻴﻪ ﻣﯽ‌ﮐﻨﺪ.
  • ﺗﻮﺿﻴﺢ ﻣﺎﺩﯼ: ﭼﺮﺍ ﺍﻳﻦ ﺩﻭﭼﺮﺧﻪ ﺍﻳﻨﻘﺪﺭ ﺳﺒﮏ ﺍﺳﺖ؟ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺪﻧﻪ ﺁﻥ ﺍﺯ ﺁﻟﻴﺎﮊ ﮐﺮﺑﻦ ﺳﺎﺧﺘﻪ ﺷﺪﻩ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺗﻮﺿﻴﺢ، ﺟﻨﺲ ﻭ ﻣﺎﺩﻩ ﺗﺸﮑﻴﻞ‌ﺩﻫﻨﺪﻩ ﺩﺭ ﻓﻬﻤﻴﺪﻥ ﭼﺮﺍﻳﯽ ﻳﮏ ﭘﺪﻳﺪﻩ ﺍﻫﻤﻴﺖ ﺩﺍﺭﺩ.

ﻫﻤﻪ‌ﯼ ﺍﻳﻦ ﭼﻬﺎﺭ ﻧﻮﻉ ﺗﻮﺿﻴﺢ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺑﻔﻬﻤﻴﻢ ﭼﺮﺍ ﭼﻴﺰﯼ ﺍﺗﻔﺎﻕ ﺍﻓﺘﺎﺩﻩ ﺍﺳﺖ. ﺍﻣﺎ ﺍﻳﻦ ﺗﻮﺿﻴﺢ‌ﻫﺎ ﻟﺰﻭﻣﺎ ﺟﺪﺍ ﺍﺯ ﻫﻢ ﻧﻴﺴﺘﻨﺪ ﻭ ﺑﺮﺍﯼ ﻓﻬﻢ ﻳﮏ ﭘﺪﻳﺪﻩ ﻳﮑﺴﺎﻥ ﻣﯽ‌ﺗﻮﺍﻥ ﻫﺮ ﭼﻬﺎﺭ ﻧﻮﻉ ﺗﻮﺿﻴﺢ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﻳﻨﮑﻪ ﭼﺮﺍ ﺑﻮﻕ ﻗﻄﺎﺭﯼ ﮐﻪ ﺍﺯ ﻧﺰﺩﻳﮑﯽ ﺧﺎﻧﻪ‌ ﻣﯽ‌ﮔﺬﺭﺩ ﺍﻳﻨﻘﺪﺭ ﺑﻠﻨﺪ ﺍﺳﺖ، ﺗﻮﺿﻴﺢ ﻋﻞّﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﺯﻳﺮﺍ ﮐﺴﯽ ﺟﻠﻮﯼ ﻗﻄﺎﺭ ﭘﺮﻳﺪﻩ ﻭ ﺭﺍﻧﻨﺪﻩ ﺑﻮﻕ ﺯﺩﻩ ﺍﺳﺖ. ﺗﻮﺿﻴﺢ ﻏﺎﻳﯽ ﻳﺎ ﻧﺘﻴﺠﻪ ﻧﮕﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﻗﻄﺎﺭ ﺍﺯ ﻣﺤﻞ ﺗﻘﺎﻃﻊ ﺟﺎﺩﻩ ﺑﺎ ﺭﻳﻞ ﻋﺒﻮﺭ ﮐﺮﺩﻩ ﺍﺳﺖ ﻭ ﺭﺍﻧﻨﺪﻩ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺍﺯ ﻧﺰﺩﻳﮏ ﺷﺪﻥ ﻗﻄﺎﺭ ﻣﻄﻠﻊ ﮐﻨﺪ. ﺗﻮﺿﻴﺢ ﺻﻮﺭﯼ ﻳﺎ ﺷﮑﻠﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﺑﻮﻕ ﻗﻄﺎﺭ ﺷﮑﻞ ﺧﺎﺻﯽ ﺩﺍﺭﺩ ﻭ ﺩﺭ ﻣﺤﻞ ﺧﺎﺻﯽ ﺑﺎﻻﯼ ﻗﻄﺎﺭ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﻭ ﭼﻨﻴﻦ ﺻﺪﺍﻳﯽ ﺩﺍﺭﺩ. ﺩﺭ ﻧﻬﺎﻳﺖ ﺗﻮﺿﻴﺢ ﻣﺎﺩﯼ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﻫﻮﺍ ﭼﮕﺎﻟﯽ ﻣﻌﻴﻨﯽ ﺩﺍﺭﺩ ﮐﻪ ﻭﻗﺘﯽ ﺑﺎ ﺳﺮﻋﺖ ﺧﺎﺻﯽ ﺍﺯ ﻟﻮﻟﻪ‌ﺍﯼ ﻋﺒﻮﺭ ﮐﻨﺪ، ﭼﻨﻴﻦ ﺻﺪﺍﻳﯽ ﺗﻮﻟﻴﺪ ﻣﯽ‌ﮐﻨﺪ.

ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻳﮏ ﺗﻮﺿﻴﺢ ﻧﻮﻋﯽ ﺍﺯ ﺍﺳﺘﺪﻻﻝ ﺍﺳﺖ ﻭ ﺑﻪ ﺍﻳﻦ ﺷﮑﻞ ﺻﻮﺭﺕ‌ﺑﻨﺪﯼ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﺻﻮﻝ ﻋﻤﻮﻣﯽ ﻳﺎ ﻗﻮﺍﻧﻴﻦ ﻃﺒﻴﻌﯽ ﺩﺭ ﺻﻮﺭﺕ ﻭﺟﻮﺩ ﺷﺮﺍﻳﻂ ﺍﻭﻟﻴﻪ ﺧﺎﺹ ﻣﻨﺠﺮ ﺑﻪ ﭘﺪﻳﺪﻩ‌ﺍﯼ ﻣﯽ‌ﺷﻮﻧﺪ ﮐﻪ ﺩﺭ ﺻﺪﺩ ﺗﻮﺿﻴﺢ ﺁﻥ ﻫﺴﺘﻴﻢ. ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺗﻮﺿﻴﺢ ﺩﻫﻴﻢ ﮐﻪ ﭼﺮﺍ ﻭﻗﺘﯽ ﮐﺘﺎﺑﯽ ﺭﺍ ﺭﻫﺎ ﮐﻨﻴﻢ ﺑﻪ ﺭﻭﯼ ﺯﻣﻴﻦ ﻣﯽ‌ﺍﻓﺘﺪ ﻧﻴﺎﺯ ﺑﻪ ﻳﮏ ﻗﺎﻧﻮﻥ ﻃﺒﻴﻌﯽ ﻳﺎ ﺍﺻﻞ ﻋﻤﻮﻣﯽ ﺩﺍﺭﻳﻢ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﺜﺎﻝ ﻗﺎﻧﻮﻥ ﺟﺎﺫﺑﻪ ﻋﻤﻮﻣﯽ ﺍﺳﺖ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﺪﻑ ﺍﺯ ﺗﻮﺿﻴﺢ ﺟﺎ ﺩﺍﺩﻥ ﻳﮏ ﭘﺪﻳﺪﻩ ﺑﻪ ﺧﺼﻮﺹ ﺩﺭ ﻳﮏ ﺍﻟﮕﻮﯼ ﻋﻤﻮﻣﯽ ﻭ ﮐﻠﯽ ﺍﺳﺖ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻋﺚ ﺍﻓﺰﺍﻳﺶ ﻓﻬﻢ ﻣﺎ ﺍﺯ ﺁﻥ ﺧﻮﺍﻫﺪ ﺷﺪ ﮐﻪ ﭘﺪﻳﺪﻩ‌ﻫﺎ ﭼﮕﻮﻧﻪ ﺷﮑﻞ ﮔﺮﻓﺘﻪ‌ﺍﻧﺪ ﻳﺎ ﺑﺎﻭﺭﻫﺎ ﻭ ﻋﻘﺎﻳﺪ ﺍﺯ ﮐﺠﺎ ﺁﻣﺪﻩ‌ﺍﻧﺪ. ﻣﺎ ﻣﻌﻤﻮﻻ ﺑﺮﺍﯼ ﭘﺪﻳﺪﻩ‌ﻫﺎﻳﯽ ﺩﻧﺒﺎﻝ ﺗﻮﺿﻴﺢ ﻣﯽ‌ﮔﺮﺩﻳﻢ ﮐﻪ ﻋﺠﻴﺐ ﻳﺎ ﻏﻴﺮﻋﺎﺩﯼ ﺑﺎﺷﻨﺪ ﻭ ﺟﺎ ﺩﺍﺩﻥ ﺁﻧﻬﺎ ﺩﺭ ﻳﮏ ﺍﻟﮕﻮﯼ ﮐﻠﯽ ﻭ ﻋﻤﻮﻣﯽ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﮐﻤﺘﺮ ﻋﺠﻴﺐ ﻳﺎ ﻏﻴﺮﻋﺎﺩﯼ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﻨﺪ ﻭ ﺩﻗﻴﻘﺎ ﻫﻤﻴﻦ ﻋﺎﺩﯼ ﺷﺪﻥ ﺑﻪ ﻣﻌﻨﺎﯼ ﺗﻮﺿﻴﺢ ﺍﺳﺖ ﮐﻪ ﺑﺎﻋﺚ ﻓﻬﻤﻴﺪﻥ ﭼﻴﺰﻫﺎ ﻣﯽ‌ﺷﻮﺩ.
