ﺩﺭﺱ ﻫﻔﺘﻢ- ﺁﺩﻡ ﻫﺎﯼ ﺧﻼﻕ

ﺗﺎ ﺍﻳﻨﺠﺎ ﻣﻄﺎﻟﺐ ﻣﺨﺘﻠﻔﯽ ﺩﺭ ﻣﻮﺭﺩ ﺗﻔﮑﺮ ﺧﻼﻕ ﺑﺎ ﻫﻢ ﺧﻮﺍﻧﺪﻳﻢ. ﺣﺎﻻ ﺷﺎﻳﺪ ﺑﺪ ﻧﺒﺎﺷﺪ ﮐﻤﯽ ﺩﺭ ﻣﻮﺭﺩ ﺧﺼﻮﺻﻴﺎﺕ ﺁﺩﻡ ﻫﺎﻳﯽ ﮐﻪ ﺧﻼﻗﺎﻧﻪ ﻓﮑﺮ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺩﻧﺒﺎﻝ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﻧﻮ ﻫﺴﺘﻨﺪ، ﺑﺪﺍﻧﻴﻢ.

ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﻣﻨﻈﻮﺭ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﻫﻤﻪ ﺍﻳﻦ ﺧﺼﻮﺻﻴﺎﺕ ﺩﺭ ﻫﻤﻪ ﺟﻤﻊ ﺍﺳﺖ ﻭ ﺍﻳﻨﮑﻪ ﺗﻤﺎﻡ ﺍﻳﻦ ﺧﺼﻮﺻﻴﺎﺕ ﻧﺴﺒﯽ ﻭ  ﺍﮐﺘﺴﺎﺑﯽ ﻫﺴﺘﻨﺪ ﻭ ﺍﺯ ﻃﺮﻳﻖ ﻳﺎﺩﮔﻴﺮﯼ ﻭ ﺗﮑﺮﺍﺭ ﻭ ﺗﻤﺮﻳﻦ ﻗﺎﺑﻞ ﺭﺷﺪ ﻫﺴﺘﻨﺪ. ﺧﻮﺍﻧﺪﻥ ﺍﻳﻦ ﺩﺭﺱ ﺷﺎﻳﺪ ﮐﻤﮑﯽ ﻫﻢ ﺑﻪ ﺧﻮﺩﻣﺎﻥ ﺑﺎﺷﺪ ﮐﻪ ﺗﻐﻴﻴﺮﺍﺕ ﺭﻓﺘﺎﺭﻳﻤﺎﻥ ﺭﺍ ﺑﻌﺪ ﺍﺯ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻦ ﺷﻴﻮﻩ ﻫﺎ ﻭ ﺗﮑﻨﻴﮏ ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﻳﺎﺩﮔﺮﻓﺘﻪ ﺍﻳﻢ، ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﻴﻢ.

 

ﮐﻨﺠﮑﺎﻭﯼ
ﺁﺩﻡ ﻫﺎﯼ ﺧﻼﻕ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺩﺭ ﻣﻮﺭﺩ ﻫﻤﻪ ﭼﻴﺰ ﺑﻴﺸﺘﺮ ﺑﺪﺍﻧﻨﺪ. ﺩﺍﻧﺴﺘﻦ ﺩﻟﻴﻞ ﻧﻤﯽ ﺧﻮﺍﻫﺪ. «ﺑﺮﺍﯼ ﭼﯽ ﻣﯽ ﺧﻮﺍﻫﯽ ﺍﻳﻦ ﺭﻭ ﺑﺪﻭﻧﯽ؟» ﺍﻳﻦ ﺳﻮﺍﻝ ﺑﺮﺍﯼ ﻳﮏ ﻓﺮﺩ ﺧﻼﻕ ﻋﺠﻴﺐ ﺍﺳﺖ، «ﻣﯽ ﺧﻮﺍﻫﻢ ﺑﺪﺍﻧﻢ ﭼﻮﻥ ﭼﻴﺰﯼ ﺩﺭ ﻣﻮﺭﺩﺵ ﻧﻤﯽ ﺩﻭﻧﻢ». ﺩﺍﻧﺶ، ﻟﺬﺕ ﺑﺨﺶ ﺍﺳﺖ ﻭ ﻋﻤﻮﻣﺎ ﺩﺭ ﻣﻮﺍﻗﻌﯽ ﺩﻭﺭ ﺍﺯ ﺍﻧﺘﻈﺎﺭ ﺑﻪ ﮐﻤﮏ ﻣﺎ ﻣﯽ ﺁﻳﺪ.

ﺿﻤﻦ ﺍﻳﻨﮑﻪ، ﺩﺍﻧﺶ ﺑﻪ ﺧﺼﻮﺹ ﺩﺭ ﺳﻄﻮﺡ ﻣﺨﺘﻠﻒ، ﺑﺮﺍﯼ ﺑﺮﻭﺯ ﺧﻼﻗﻴﺖ ﻭ ﺑﺎﺭﻭﺭ ﺷﺪﻥ ﺁﻥ ﻻﺯﻡ ﺍﺳﺖ. ﺑﻴﺸﺘﺮﻳﻦ ﺧﻼﻗﻴﺖ ﺍﺯ ﺗﻨﻮﻉ ﺩﺍﻧﺶ ﻭ ﺗﺮﮐﻴﺐ ﺁﻧﻬﺎ ﺣﺎﺻﻞ ﻣﯽ ﺷﻮﺩ. ﻃﺒﻴﻌﺘﺎ ﺍﺯ ﻫﻴﭻ، ﭼﻴﺰﯼ ﺩﺭ ﻧﻤﯽ ﺁﻳﺪ.

ﭘﺲ ﻫﻴﭻ ﻫﺮﺍﺱ ﻭ ﺍﺑﺎﻳﯽ ﺍﺯ ﭘﺮﺳﻴﺪﻥ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺍﺯ ﻫﺮ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﭙﺮﺳﻴﺪ، ﻳﮏ ﺳﻮﺍﻝ ﺭﺍ ﺍﺯ ﭼﻨﺪﻳﻦ ﻧﻔﺮ ﺑﭙﺮﺳﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﭘﺎﺳﺦ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﻘﺎﻳﺴﻪ ﮐﻨﻴﺪ. ﻫﻴﭻ ﻣﺤﺪﻭﺩﻳﺘﯽ ﺑﺮﺍﯼ ﺩﺍﻧﺴﺘﻦ ﺑﻴﺸﺘﺮ ﺑﺮﺍﯼ ﺧﻮﺩ ﻧﮕﺬﺍﺭﻳﺪ، ﺑﻪ ﺳﺮﺍﻍ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺑﺮﻭﻳﺪ ﮐﻪ ﻫﻴﭻ ﭼﻴﺰﯼ ﺭﺍﺟﻊ ﺑﻪ ﺁﻧﻬﺎ ﻧﻤﯽ ﺩﺍﻧﻴﺪ، ﻣﺜﻼ ﻫﻮﺍﺷﻨﺎﺳﯽ، ﺧﺸﮏ ﺷﻮﻳﯽ، ﺍﻓﺰﻭﺩﻧﯽ ﻫﺎﯼ ﺧﻮﺭﺍﮐﯽ، ﮐﺸﺘﯽ ﺳﺎﺯﯼ، ﻓﺮﺵ ﺑﺎﻓﯽ، ﺣﺴﺎﺑﺪﺍﺭﯼ، ﭼﺎﭖ ﻭ... .

 

ﭼﺎﻟﺶ
ﺁﺩﻡ ﻫﺎﯼ ﮐﻨﺠﮑﺎﻭ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺑﺪﺍﻧﻨﺪ ﭘﻴﺶ ﻓﺮﺽ ﻫﺎﯼ ﭘﺸﺖ ﻣﺴﺎﺋﻞ، ﺩﻻﻳﻞ، ﺍﻳﺪﻩ ﻫﺎ، ﻣﺸﮑﻼﺕ ﻭ ﺑﺎﻭﺭﻫﺎ ﭼﻴﺴﺖ. ﺁﻧﻬﺎ ﺭﺍ ﺑﺸﻨﺎﺳﻨﺪ ﻭ ﺑﻪ ﭼﺎﻟﺶ ﺑﮑﺸﻨﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﻦ ﭘﻴﺶ ﻓﺮﺽ ﻫﺎ ﻣﺤﮑﻢ ﻭ ﻻﺯﻡ ﻫﺴﺘﻨﺪ، ﻭﻟﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺁﻧﻬﺎ ﻏﻴﺮﺿﺮﻭﺭﯼ ﻭ ﻣﺎﻧﻌﯽ ﺑﺮﺍﯼ ﺧﻼﻕ ﻓﮑﺮ ﮐﺮﺩﻥ ﻭ ﺑﺮﻭﺯ ﺍﻳﺪﻩ ﻫﺎﯼ ﻧﻮ ﻫﺴﺘﻨﺪ.

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻫﺮ ﻭﻗﺖ ﺑﻪ ﺩﺍﻧﺸﮕﺎﻩ ﻳﺎ ﻣﻮﺳﺴﻪ ﺁﻣﻮﺯﺷﯽ ﻓﮑﺮ ﻣﯽ ﮐﻨﻴﻢ، ﺑﻪ ﻃﻮﺭ ﺳﻨﺘﯽ ﺍﻭﻟﻴﻦ ﭼﻴﺰﯼ ﮐﻪ ﺑﻪ ﺫﻫﻦ ﻣﺎﻥ ﻣﯽ ﺭﺳﺪ ﻳﮏ ﺳﺎﺧﺘﻤﺎﻥ ﭼﻨﺪ ﻃﺒﻘﻪ، ﮐﻼﺱ ﻫﺎ، ﺩﺍﻧﺸﺠﻮﻳﺎﻥ ﻭ ﮐﺘﺎﺑﺨﺎﻧﻪ ﺍﺳﺖ. ﺍﺳﺎﺳﺎ ﭼﺮﺍ ﺩﺍﻧﺸﮕﺎﻩ ﻳﺎ ﻣﻮﺳﺴﻪ ﺁﻣﻮﺯﺷﯽ ﺑﺎﻳﺪ ﭼﻨﻴﻦ ﺷﮑﻠﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ (ﻣﺤﻠﯽ ﻣﺸﺨﺺ ﺑﺮﺍﯼ ﺁﻣﻮﺯﺵ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ)؟ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﺩﺍﻧﺸﮕﺎﻩ ﻫﺎ ﻭ ﻣﻮﺳﺴﺎﺕ ﺁﻣﻮﺯﺷﯽ ﻣﻌﺘﺒﺮ، ﻣﮑﺎﺗﺒﻪ ﺍﯼ ﻭ ﺍﻳﻨﺘﺮﻧﺘﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ ﺯﻳﺎﺩﯼ ﺍﺯ ﺧﺎﻧﻪ ﻳﺎ ﻣﺤﻞ ﮐﺎﺭ ﺧﻮﺩ ﺩﺭ ﺣﺎﻝ ﻋﻠﻢ ﺁﻣﻮﺯﯼ ﻫﺴﺘﻨﺪ.

ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﻭ ﺗﻤﺮﻳﻦ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﺤﺚ ﻫﺎ ﻭ ﺩﻻﻳﻞ ﻣﺪﺍﻓﻊ ﻣﮑﺎﻥ ﺁﻣﻮﺯﺷﯽ ﺩﺭ ﻣﮑﺎﻧﯽ ﻋﻴﻨﯽ ﻭ ﻣﺸﺨﺺ ﺭﺍ ﺑﻪ ﭼﺎﻟﺶ ﺑﮑﺸﻴﺪ.

 

ﻧﺎﺭﺿﺎﻳﺘﯽ ﺳﺎﺯﻧﺪﻩ
ﻣﻨﻈﻮﺭ ﻏﺮ ﺯﺩﻥ، ﺑﻬﺎﻧﻪ ﮔﻴﺮﯼ، ﺍﺷﮑﺎﻝ ﺗﺮﺍﺷﯽ ﻭ ﻃﻠﺒﮑﺎﺭ ﺑﻮﺩﻥ ﻧﻴﺴﺖ. ﺑﻠﮑﻪ ﺗﻮﺍﻧﺎﻳﯽ ﺩﻳﺪﻥ ﻭ ﺣﺲ ﮐﺮﺩﻥ ﻧﻴﺎﺯ ﺑﻪ ﭘﻴﺸﺮﻓﺖ ﻭ ﺑﻬﺘﺮ ﺷﺪﻥ ﻭ ﭘﻴﺸﻨﻬﺎﺩ ﺭﻭﺷﯽ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺗﻐﻴﻴﺮ ﻭ ﭘﻴﺸﺮﻓﺖ ﺍﺳﺖ.

ﻧﺎﺭﺿﺎﻳﺘﯽ ﺳﺎﺯﻧﺪﻩ ﺑﺎﺯﺗﺎﺏ ﺍﻳﻦ ﻃﺮﺯ ﺗﻔﮑﺮ ﺍﺳﺖ: «ﻣﻦ ﺭﺍﻩ ﻫﺎﻳﯽ ﺑﻠﺪﻡ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﻬﺘﺮ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ.»

ﻳﮑﯽ ﺍﺯ ﺳﺎﺯﻧﺪﻩ ﺗﺮﻳﻦ ﺍﺛﺮﺍﺕ ﺍﻳﻦ ﻧﻮﻉ ﻧﺎﺭﺿﺎﻳﺘﯽ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻫﺪﻑ ﻭ ﺩﻭﺭﻧﻤﺎﯼ ﺁﻥ ﺩﻧﺒﺎﻝ ﻣﺸﮑﻼﺕ ﺑﻮﺩﻥ ﺍﺳﺖ. ﻭﻗﺘﯽ ﺷﻤﺎ ﻫﻴﭻ ﻭﻗﺖ ﺑﻪ ﮐﻴﻔﻴﺖ ﭼﻴﺰﯼ ﮐﻪ ﻣﻮﺟﻮﺩ ﺍﺳﺖ ﺭﺿﺎﻳﺖ ﻧﺪﻫﻴﺪ، ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺩﻧﺒﺎﻝ ﻣﺸﮑﻼﺕ ﻭ ﭼﺎﻟﺶ ﻫﺎﯼ ﮐﺎﺭ ﻫﺴﺘﻴﺪ ﻭ ﺍﻳﻦ ﻳﻌﻨﯽ ﺷﻤﺎ ﺑﻪ ﺩﻧﺒﺎﻝ ﺭﺍﻩ ﻫﺎ ﻭ ﻣﻮﻗﻌﻴﺖ ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺑﻬﺘﺮ ﮐﺮﺩﻥ ﻭ ﭘﻴﺸﺮﻓﺖ ﮐﺎﺭ ﻫﺴﺘﻴﺪ. ﺍﻳﻦ ﻧﻮﻉ ﺑﺮﺧﻮﺭﺩ ﺑﺴﻴﺎﺭ ﺳﺎﺯﻧﺪﻩ ﺍﺳﺖ.

ﺩﻳﮕﺮ ﻧﺸﺎﻧﻪ ﻧﺎﺭﺿﺎﻳﺘﯽ ﺳﺎﺯﻧﺪﻩ، ﻟﺬﺕ ﺍﺯ ﭼﺎﻟﺶ ﺍﺳﺖ. ﺁﺩﻡ ﻫﺎﯼ ﺧﻼﻕ ﻣﺪﺍﻡ ﺩﻧﺒﺎﻝ ﺍﻳﻦ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﺤﺪﻭﺩﻳﺖ ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﻣﺤﮏ ﺑﺰﻧﻨﺪ ﻭ ﺍﺯ ﺁﻧﻬﺎ ﻋﺒﻮﺭ ﮐﻨﻨﺪ، ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺯ ﭼﺎﻟﺶ ﺍﺳﺘﻘﺒﺎﻝ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺍﺯ ﺩﺳﺖ ﻭ ﭘﻨﺠﻪ ﻧﺮﻡ ﮐﺮﺩﻥ ﺑﺎ ﺁﻥ ﻟﺬﺕ ﻣﯽ ﺑﺮﻧﺪ.

ﺩﺭ ﺣﻘﻴﻘﺖ ﺷﺎﻳﺪ ﺑﺘﻮﺍﻥ ﮔﻔﺖ، ﺑﺨﺶ ﻋﻤﺪﻩ ﺍﯼ ﺍﺯ ﺍﻳﻦ ﻧﺎﺭﺿﺎﻳﺘﯽ (ﺑﺎ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﻣﻔﻬﻮﻡ ﻏﺎﻟﺐ ﻭ ﺭﺍﻳﺞ ﻧﺎﺭﺿﺎﻳﺘﯽ)، ﻣﺼﻨﻮﻋﯽ ﻫﻢ ﺑﺎﺷﺪ، ﭼﺮﺍ ﮐﻪ ﺩﺭ ﺣﻘﻴﻘﺖ ﻧﺎﺭﺿﺎﻳﺘﯽ ﻧﻴﺴﺖ، ﺍﺯ ﺷﺮﺍﻳﻂ ﻣﻮﺟﻮﺩ ﺭﺍﺿﯽ ﻧﻴﺴﺘﻨﺪ ﭼﻮﻥ ﻗﺼﺪ ﺩﺍﺭﻧﺪ ﺁﻥ ﺭﺍ ﺑﻬﺘﺮ ﻭ ﺑﻬﺘﺮ ﮐﻨﻨﺪ ﻭ ﺑﺎﻭﺭ ﺩﺍﺭﻧﺪ ﻫﻤﻴﺸﻪ ﺟﺎ ﺑﺮﺍﯼ ﺑﻬﺘﺮ ﺷﺪﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

 

ﺑﺎﻭﺭ ﺑﻪ ﺍﻳﻨﮑﻪ ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮ ﻣﺸﮑﻼﺕ ﺭﺍﻩ ﺣﻞ ﻭﺟﻮﺩ ﺩﺍﺭﺩ
ﺍﺑﺘﺪﺍ ﺑﺮﺍﺳﺎﺱ ﺑﺎﻭﺭ ﻗﻠﺒﯽ ﻭ ﺑﻪ ﺗﺪﺭﻳﺞ ﺑﺮﺍﺳﺎﺱ ﺗﺠﺮﺑﻪ، ﺁﺩﻡ ﻫﺎﯼ ﺧﻼﻕ ﺍﻃﻤﻴﻨﺎﻥ ﭘﻴﺪﺍ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮ ﻣﺸﮑﻼﺕ ﺭﺍﻩ ﺣﻞ ﻫﺎﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻣﺸﮑﻼﺕ ﺑﺎ ﺻﺮﻑ ﻣﺘﻌﻬﺪﺍﻧﻪ ﺯﻣﺎﻥ ﻭ ﺍﻧﺮﮊﯼ ﺍﺗﻔﺎﻕ ﻣﯽ ﺍﻓﺘﺪ.

ﺩﺭ ﻭﻫﻠﻪ ﺍﻭﻝ، ﺑﺎﻭﺭ ﺑﻪ ﻗﺎﺑﻞ ﺣﻞ ﺑﻮﺩﻥ ﻣﺸﮑﻼﺕ ﺑﺮﺍﯼ ﺣﻤﻠﻪ ﺑﻪ ﻣﺸﮑﻞ ﺑﺴﻴﺎﺭ ﺳﻮﺩﻣﻨﺪ ﺍﺳﺖ؛ ﭼﺮﺍ ﮐﻪ ﺑﻴﺸﺘﺮ ﻣﺸﮑﻼﺕ ﺩﺭ ﻧﮕﺎﻩ ﺍﻭﻝ ﻻﻳﻨﺤﻞ ﺑﻪ ﻧﻈﺮ ﻣﯽ ﺭﺳﻨﺪ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺭ ﺍﻳﻨﮑﻪ ﺑﻪ ﻃﺮﻑ ﺷﺎﻥ ﺑﺮﻭﻳﻢ ﺑﺘﺮﺳﻴﻢ. ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﺗﺮﺱ ﻋﺒﻮﺭ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺳﻌﯽ ﻣﯽ ﮐﻨﻨﺪ ﺑﺮ ﻣﺸﮑﻞ ﺳﻮﺍﺭ ﺷﻮﻧﺪ، ﺗﻘﺮﻳﺒﺎ ﺑﺎ ﻗﻄﻌﻴﺖ ﻣﯽ ﺗﻮﺍﻥ ﮔﻔﺖ ﮐﺴﺎﻧﯽ ﻧﻴﺴﺘﻨﺪ ﮐﻪ ﻓﮑﺮ ﻣﯽ ﮐﻨﻨﺪ ﻣﺸﮑﻞ ﻻﻳﻨﺤﻞ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

 

ﻗﺪﺭﺕ ﻣﻬﺎﺭ ﻗﻀﺎﻭﺕ ﻭ ﺍﻧﺘﻘﺎﺩ
ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﺪﻩ ﻫﺎﯼ ﻧﻮ ﻭ ﺗﺎﺯﻩ، ﺩﻗﻴﻘﺎ ﺑﻪ ﺩﻟﻴﻞ ﻧﻮ ﺑﻮﺩﻥ ﻭ ﻧﺎﺁﺷﻨﺎ ﺑﻮﺩﻥ ﺑﻪ ﻧﻈﺮ ﻋﺠﻴﺐ ﻭ ﻏﺮﻳﺐ ﻣﯽ ﺁﻳﻨﺪ. ﮐﻤﺎ ﺍﻳﻨﮑﻪ ﭼﻪ ﺑﺴﺎ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﺪﻩ ﻫﺎﯼ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻨﯽ ﻭ ﺗﺤﺴﻴﻦ ﺑﺮﺍﻧﮕﻴﺰ ﺍﻣﺮﻭﺯ، ﺍﻳﺪﻩ ﻫﺎﯼ ﻋﺠﻴﺐ ﻭ ﻣﻄﺮﻭﺩ ﺩﻫﻪ ﻫﺎﯼ ﻗﺒﻞ ﺑﺎﺷﻨﺪ.

ﺑﺮﺍﯼ ﮐﺴﯽ ﮐﻪ ﺩﻧﺒﺎﻝ ﻧﻮﺁﻭﺭﯼ ﻭ ﺍﻳﺪﻩ ﻫﺎﯼ ﺟﺪﻳﺪ ﺍﺳﺖ، ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﻣﻮﺍﺟﻬﻪ ﺑﺎ ﺍﻳﺪﻩ ﺟﺪﻳﺪ، ﺑﺘﻮﺍﻧﺪ ﺟﻠﻮﯼ ﻗﻀﺎﻭﺕ ﺧﻮﺩ ﺭﺍ ﺑﮕﻴﺮﺩ.

ﺍﻇﻬﺎﺭ ﻧﻈﺮﻫﺎﻳﯽ ﻣﺜﻞ، «ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﻧﺪﺍﺭﻩ ﺑﻔﺮﻭﺷﻪ!»، «ﻣﺤﺎﻟﻪ، ﺍﺻﻼ ﮐﺴﯽ ﺗﺎ ﺣﺎﻻ ﺩﻳﺪﻩ ﻳﮏ ﻫﻤﭽﻴﻦ ﭼﻴﺰﯼ» ﻭ... .

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺤﺼﻮﻻﺕ ﮐﻪ ﺯﻧﺪﮔﯽ ﻣﺎ ﺑﺪﻭﻥ ﺁﻧﻬﺎ ﺩﺭ ﺍﻳﻦ ﻋﺼﺮ ﻣﻴﺴﺮ ﻧﻴﺴﺖ، ﺩﺭ ﺑﺪﻭ ﺳﺎﺧﺖ ﻭ ﺗﻮﻟﻴﺪ ﺑﺎ ﻗﻀﺎﻭﺕ ﻫﺎ ﻭ ﺍﻧﺘﻘﺎﺩﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺭﻭﺑﻪ ﺭﻭ ﺑﻮﺩﻩ ﺍﻧﺪ، ﻣﺜﻞ ﻣﺤﺼﻮﻻﺕ ﺿﺪﻋﻔﻮﻧﯽ ﮐﻨﻨﺪﻩ، ﺭﺍﺩﻳﻮ، ﺍﺳﺘﻴﻞ، ﻇﺮﻑ ﻫﺎﯼ ﺗﻔﻠﻦ ﻭ... .

ﻗﻀﺎﻭﺕ ﻫﺎ ﻭ ﺍﻧﺘﻘﺎﺩﻫﺎﯼ ﺍﻳﻨﭽﻨﻴﻨﯽ ﺭﺍﻩ ﻣﺎ ﺭﺍ ﺑﻪ ﺩﻧﻴﺎﻫﺎﯼ ﺟﺪﻳﺪ ﺳﺪ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺑﺎﺭﻫﺎ ﻭ ﺑﺎﺭﻫﺎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺗﺎﮐﻴﺪ ﺷﺪﻩ ﺍﺳﺖ، ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺩﺭ ﺗﻔﮑﺮ ﺧﻼﻕ ﺑﻴﺮﻭﻥ ﺁﻣﺪﻥ ﺍﺯ ﻗﺎﻟﺐ ﻫﺮ ﻣﺤﺪﻭﺩﻳﺖ ﺫﻫﻨﯽ ﻭ ﻓﮑﺮﯼ ﺍﺳﺖ.

 

ﻣﺸﮑﻼﺕ ﺭﺍﻫﯽ ﺑﺮﺍﯼ ﭘﻴﺸﺮﻓﺖ
ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺩﺭ ﺑﺨﺶ «ﻧﺎﺭﺿﺎﻳﺘﯽ ﺳﺎﺯﻧﺪﻩ» ﻫﻢ ﺩﻳﺪﻳﻢ، ﺍﻓﺮﺍﺩﯼ ﺑﺎ ﺍﻳﻦ ﻧﻮﻉ ﻧﮕﺎﻩ، ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﺸﮑﻼﺕ ﻫﺴﺘﻨﺪ ﺑﺮﺍﯼ ﺍﺭﺗﻘﺎ. ﻭﻟﯽ ﺩﺭ ﺯﻧﺪﮔﯽ ﻋﺎﺩﯼ ﻫﻤﻴﺸﻪ ﺍﻳﻨﻄﻮﺭ ﻧﻴﺴﺖ. ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﻮﺍﺭﺩ ﻣﺸﮑﻼﺕ ﺑﺴﻴﺎﺭ ﺩﻭﺭ ﺍﺯ ﺍﻧﺘﻈﺎﺭ ﻣﺎ ﺍﺗﻔﺎﻕ ﻣﯽ ﺍﻓﺘﻨﺪ، ﺩﺭ ﻣﻮﺍﺟﻬﻪ ﺑﺎ ﺍﻳﻦ ﻧﻮﻉ ﻣﺸﮑﻼﺕ ﻏﻴﺮﻣﺘﺮﻗﺒﻪ ﻫﻢ ﻧﻮﻉ ﺑﺮﺧﻮﺭﺩ ﻭ ﻧﮕﺎﻩ ﺑﻪ ﻣﺸﮑﻼﺕ ﺩﺭ ﻳﺎﻓﺘﻦ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﻧﻮ ﻭ ﺧﻼﻗﺎﻧﻪ ﻗﻄﻌﺎ ﻣﻮﺛﺮ ﺍﺳﺖ. ﻧﮕﺎﻫﯽ ﮐﻪ ﻫﺮ ﻣﺸﮑﻞ ﺭﺍ ﺩﻟﻴﻠﯽ ﺑﺮﺍﯼ ﺑﺎﻻ ﺑﺮﺩﻥ ﮐﻴﻔﻴﺖ ﺁﻧﭽﻪ ﻫﺴﺖ ﻣﯽ ﺑﻴﻨﺪ.

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻣﺎﺭﮔﺎﺭﻳﻦ ﺩﺭ ﺍﺑﺘﺪﺍ ﺍﺯ ﭼﺮﺑﯽ ﮔﺎﻭ، ﺷﻴﺮ، ﺁﺏ ﻭ ﺗﮑﻪ ﻫﺎﯼ ﭘﺴﺘﺎﻥ ﮔﺎﻭ ﺗﻬﻴﻪ ﻣﯽ ﺷﺪ، ﺍﺯ ﻧﻈﺮ ﻃﻌﻢ ﻭ ﺳﻼﻣﺘﯽ ﻣﺤﺼﻮﻝ ﺧﻴﻠﯽ ﻓﻮﻕ ﺍﻟﻌﺎﺩﻫﺎﯼ ﻧﺒﻮﺩ. ﺑﻌﺪ ﺍﺯ ﺳﺎﻝ ﻫﺎ، ﻣﺸﮑﻞ ﮐﻤﺒﻮﺩ ﭼﺮﺑﯽ ﮔﺎﻭﯼ ﭘﻴﺶ ﺁﻣﺪ ﻭ ﺳﺎﺯﻧﺪﮔﺎﻥ ﻣﺎﺭﮔﺎﺭﻳﻦ ﺑﻪ ﻓﮑﺮ ﺗﻐﻴﻴﺮ ﻣﻮﺍﺩ ﺍﻭﻟﻴﻪ ﺍﻓﺘﺎﺩﻧﺪ ﻭ ﺑﺎﻻﺧﺮﻩ ﻣﺎﺭﮔﺎﺭﻳﻨﯽ ﺑﺴﻴﺎﺭ ﺷﺒﻴﻪ ﺗﺮﮐﻴﺐ ﺍﻣﺮﻭﺯﯼ ﺑﻪ ﺑﺎﺯﺍﺭ ﻋﺮﺿﻪ ﮐﺮﺩﻧﺪ. ﭼﺮﺑﯽ ﮔﻴﺎﻫﯽ، ﺩﺍﻧﻪ ﻫﺎﯼ ﺳﻮﻳﺎ، ﺫﺭﺕ ﻭ ﺭﻭﻏﻦ ﺁﻓﺘﺎﺑﮕﺮﺩﺍﻥ، ﻧﺘﻴﺠﻪ ﻣﺤﺼﻮﻟﯽ ﺑﻪ ﻣﺮﺍﺗﺐ ﺳﺎﻟﻢ ﺗﺮ ﻭ ﺧﻮﺵ ﻃﻌﻢ ﺗﺮ ﺷﺪ.

 

ﻣﺸﮑﻞ ﺑﻌﻨﻮﺍﻥ ﺍﻣﺮﯼ ﭘﺬﻳﺮﻓﺘﻪ ﻭ ﺟﺬﺍﺏ
ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺁﺩﻡ ﻫﺎ، ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﻨﺪ ﻣﺸﮑﻼﺕ ﺭﺍ ﻧﺎﺩﻳﺪﻩ ﺑﮕﻴﺮﻧﺪ ﻭ ﺍﻧﮑﺎﺭ ﮐﻨﻨﺪ. ﻣﺸﮑﻼﺗﯽ ﺩﺭ ﺧﺎﻧﻪ، ﻣﺤﻞ ﮐﺎﺭ، ﺑﺎ ﺩﻭﺳﺘﺎﻥ، ﺑﺎ ﻓﺮﺯﻧﺪ ﻭ ﻳﺎ ﻫﻤﺴﺮﺷﺎﻥ ﺩﺍﺭﻧﺪ ﻭ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﻣﺸﮑﻼﺕ ﻣﺎﺩﺍﻣﯽ ﮐﻪ ﺑﺮﺍﻳﺸﺎﻥ ﺭﺍﻩ ﺣﻞ ﭘﻴﺪﺍ ﻧﺸﻮﺩ، ﻣﺮﺗﻔﻊ ﻧﻤﯽ ﺷﻮﻧﺪ، ﻭﺟﻮﺩ ﻣﺸﮑﻼﺕ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﻣﺮﺯ ﺟﻨﻮﻥ ﻣﯽ ﺭﺳﺎﻧﺪ.

ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻳﺪﻩ ﻫﺎﯼ ﺟﺪﻳﺪ ﻭ ﺧﻼﻗﺎﻧﻪ ﻫﺴﺘﻨﺪ، ﻫﺮ ﻣﺸﮑﻞ ﭼﺎﻟﺶ ﺟﺬﺍﺑﯽ ﺍﺳﺖ ﮐﻪ ﺍﺭﺯﺵ ﻣﻮﺍﺟﻬﻪ ﺩﺍﺭﺩ. ﺭﻭﻧﺪ ﺣﻞ ﻣﺸﮑﻞ ﺑﺮﺍﯼ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺟﺬﺍﺏ، ﺁﻣﻮﺯﻧﺪﻩ ﻭ ﻣﻔﻴﺪ ﺑﺮﺍﯼ ﺧﻮﺩ ﻭ ﺟﺎﻣﻌﻪ ﺷﺎﻥ ﺍﺳﺖ.

 

ﺍﻧﻌﻄﺎﻑ ﺩﺭ ﺗﺨﻴﻞ
ﺁﺩﻡ ﻫﺎﯼ ﺧﻼﻕ ﺑﺎ ﺗﺨﻴﻞ ﻭ ﭘﺮﻭﺭﺍﻧﺪﻥ ﺍﻳﺪﻩ ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﻧﻈﺮ ﺩﻳﮕﺮﺍﻥ ﻋﺠﻴﺐ ﻭ ﻏﺮﻳﺐ ﻭ ﺑﯽ ﺣﺎﺻﻞ ﻣﯽ ﺁﻳﺪ ﻣﺸﮑﻠﯽ ﻧﺪﺍﺭﻧﺪ.

ﺩﺭ ﺭﻭﻧﺪ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﺭﺍﻩ ﺣﻞ ﻳﺎ ﻫﻤﻔﮑﺮﯼ ﺟﻤﻌﯽ، ﻫﺮ ﺍﻳﺪﻩ ﺍﯼ ﺻﺮﻑ ﻧﻈﺮ ﺍﺯ ﺍﻳﻨﮑﻪ ﺩﻳﮕﺮﺍﻥ ﭼﻪ ﻗﻀﺎﻭﺗﯽ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﺩﺍﺭﻧﺪ، ﻣﯽ ﺗﻮﺍﻧﺪ ﺭﺍﻫﮕﺸﺎ ﺑﺎﺷﺪ.

ﭘﺲ ﺑﺮﺍﯼ ﺑﺮﻭﺯ ﺧﻼﻗﻴﺖ ﺑﺎﻳﺪ ﺩﺭ ﻣﻘﺎﺑﻞ ﺗﺨﻴﻞ ﺍﻧﻌﻄﺎﻑ ﺩﺍﺷﺖ.
