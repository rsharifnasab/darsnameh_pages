ﺩﺭﺱ ﭘﻨﺠﻢ- ﻧﺤﻮﻩ ﺍﺟﺮﺍﯼ ﺗﮑﻨﻴﮏ ﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ

ﺑﻌﺪ ﺍﺯ ﭘﺸﺖ ﺳﺮ ﮔﺬﺍﺷﺘﻦ ﺩﻭ ﺩﺭﺱ ﺩﺭ ﻣﻮﺭﺩ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ ﻭ ﺧﻼﻗﺎﻧﻪ ﺍﻧﺪﻳﺸﻴﺪﻥ، ﺣﺘﻤﺎ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺑﺪﺍﻧﻴﺪ ﭼﮕﻮﻧﻪ ﺍﺯ ﺍﻳﻨﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻧﮑﺎﺗﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﻣﯽ ﮐﻨﻴﻢ.

 

ﻫﺮ ﭼﻪ ﺯﻭﺩﺗﺮ، ﺑﻬﺘﺮ
ﺍﻭﻟﻴﻦ ﻧﮑﺘﻪ، ﺷﺮﻭﻉ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺗﮑﻨﻴﮏ ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﻧﻮ ﻓﮑﺮ ﮐﺮﺩﻥ ﻳﺎﺩ ﮔﺮﻓﺘﻴﺪ؛ ﻫﺮ ﭼﻪ ﺯﻭﺩﺗﺮ ﺷﺮﻭﻉ ﮐﻨﻴﺪ ﺑﻬﺘﺮ ﺍﺳﺖ، ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺁﻧﻬﺎ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﺧﻮﺍﻫﻴﺪ ﮐﺮﺩ.

ﺣﺘﯽ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻗﺒﻞ ﺍﺯ ﺗﻤﺎﻡ ﮐﺮﺩﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﺷﺮﻭﻉ ﮐﻨﻴﺪ ﺑﻪ ﺗﻤﺮﻳﻦ ﺍﻳﻦ ﺭﻭﺵ ﻫﺎ ﻭ ﺍﺑﺰﺍﺭﻫﺎ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺑﺮﺍﯼ ﻫﺮ ﭼﺎﻟﺶ ﻳﺎ ﻣﺸﮑﻠﯽ ﮐﻪ ﺩﺍﺭﻳﺪ، ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻫﺮ ﻣﺸﮑﻠﯽ/ﻫﺮ ﭼﺎﻟﺸﯽ، ﻫﺮ ﮐﺠﺎ ﻭ ﻫﺮ ﺯﻣﺎﻥ. ﻫﺮ ﻣﻮﻗﻌﻴﺘﯽ، ﻣﯽ ﺗﻮﺍﻧﺪ ﻓﺮﺻﺖ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺑﺮﻭﺯ ﻭ ﺗﻤﺮﻳﻦ ﺧﻼﻗﻴﺖ ﺑﺎﺷﺪ.

ﺗﺎ ﺑﻪ ﺣﺎﻝ ﺑﺮﺍﯼ ﺷﻤﺎ ﭘﻴﺶ ﺁﻣﺪﻩ ﺩﺭ ﺩﻭﺭﻩ ﺁﻣﻮﺯﺵ ﮔﻴﺘﺎﺭ ﻳﺎ ﭘﻴﺎﻧﻮ ﺷﺮﮐﺖ ﮐﻨﻴﺪ؟ ﻳﺎ ﮐﻼﺱ ﻭﺭﺯﺵ؟ ﻣﯽ ﺩﺍﻧﻴﺪ ﺍﮔﺮ ﻣﻬﺎﺭﺗﯽ ﺭﺍ ﮐﻪ ﺗﺎﺯﻩ ﻳﺎﺩﮔﺮﻓﺘﻪ ﺍﻳﺪ، ﺗﻤﺮﻳﻦ ﻧﮑﻨﻴﺪ ﭼﻪ ﺍﺗﻔﺎﻗﯽ ﻣﯽ ﺍﻓﺘﺪ؟ ﺑﻠﻪ ﺷﻤﺎ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺁﻣﻮﺧﺘﻪ ﺍﻳﺪ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻣﯽ ﮐﻨﻴﺪ.

ﻭﻗﺘﯽ ﺑﻴﻦ ﻓﺮﺍ ﮔﺮﻓﺘﻦ ﻳﮏ ﻣﻬﺎﺭﺕ ﻭ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻦ ﻭ ﺗﻤﺮﻳﻦ ﮐﺮﺩﻥ ﺁﻥ ﻓﺎﺻﻠﻪ ﺑﻴﻔﺘﺪ، ﺷﻤﺎ ﺑﻪ ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﺑﺨﺶ ﻫﺎﻳﯽ ﺍﺯ ﺁﻧﭽﻪ ﻳﺎﺩﮔﺮﻓﺘﻪ ﺍﻳﺪ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﮐﺮﺩﻩ ﺍﻳﺪ، ﺩﭼﺎﺭ ﻋﺪﻡ ﺍﻃﻤﻴﻨﺎﻥ ﻭ ﺷﮏ ﻣﯽ ﺷﻮﻳﺪ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺷﺘﺒﺎﻩ ﮐﻨﻴﺪ ﻭ ﺗﺼﺤﻴﺢ ﺗﺎﺛﻴﺮﺍﺕ ﺁﻥ ﺍﺷﺘﺒﺎﻩ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻴﺸﺘﺮ ﺍﺯ ﻳﺎﺩ ﮔﺮﻓﺘﻦ ﺁﻥ ﻣﻬﺎﺭﺕ ﺯﻣﺎﻥ ﺑﺒﺮﺩ.

ﭘﺲ ﻫﺮ ﭼﻪ ﺯﻭﺩﺗﺮ ﺁﻧﭽﻪ ﺧﻮﺍﻧﺪﻩ ﻭ ﻳﺎﺩ ﮔﺮﻓﺘﻴﺪ ﺭﺍ ﺗﻤﺮﻳﻦ ﮐﻨﻴﺪ، ﻫﺮ ﭼﻪ ﺯﻭﺩﺗﺮ ﺑﻬﺘﺮ.

 

ﺗﮑﺮﺍﺭ، ﻣﻬﺎﺭﺕ ﺷﻤﺎ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﺩﻫﺪ
ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﻫﺮ ﻳﮏ ﺍﺯ ﺗﮑﻨﻴﮏ ﻫﺎ ﺭﺍ ﺗﻤﺮﻳﻦ ﮐﺮﺩﻳﺪ ﻭ ﺍﺯ ﻧﺘﻴﺠﻪ ﺭﺍﺿﯽ ﺑﻮﺩﻳﺪ، ﺩﺭ ﺍﻭﻟﻴﻦ ﻓﺮﺻﺖ ﻣﻤﮑﻦ ﻫﻤﺎﻥ ﻣﻬﺎﺭﺕ ﺭﺍ ﺩﻭﺑﺎﺭﻩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺑﺎ ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﺑﮑﺎﺭﮔﻴﺮﯼ ﺁﻥ ﻣﻬﺎﺭﺕ ﻳﺎ ﺷﻴﻮﻩ ﺗﻔﮑﺮ، ﺁﻥ ﺗﮑﻨﻴﮏ ﺩﺭ ﺷﻤﺎ ﺩﺭﻭﻧﯽ ﻣﯽ ﺷﻮﺩ. ﭘﺲ ﻓﻘﻂ ﺗﻤﺮﻳﻦ ﻧﮑﻨﻴﺪ، ﺁﻥ ﺭﺍ ﺑﻪ ﺑﻬﺘﺮﻳﻦ ﺷﮑﻞ ﺑﻪ ﮐﺎﺭ ﺑﺒﻨﺪﻳﺪ.

ﺍﮔﺮ ﺗﮑﻨﻴﮑﯽ ﮐﻪ ﺑﺮﺍﯼ ﺍﻭﻟﻴﻦ ﺑﺎﺭ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻳﺪ، ﭼﻨﺪﺍﻥ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﺧﻮﺏ ﻧﻴﺎﻣﺪ، ﻣﺪﺗﯽ ﺭﻭﯼ ﺁﻥ ﮐﺎﺭ ﮐﻨﻴﺪ. ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺷﻤﺎ ﺑﺎ ﺗﻐﻴﻴﺮ ﺷﻴﻮﻩ ﺗﻔﮑﺮ ﻭ ﻧﻮﻉ ﻧﮕﺎﻩ ﺗﺎﻥ ﺑﻪ ﻣﺸﮑﻼﺕ ﻭ ﭼﺎﻟﺶ ﻫﺎ ﻭ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﺎﺯﻩ، ﺑﻪ ﺟﻨﮓ ﻋﺎﺩﺕ ﻫﺎ ﻭ ﺍﻟﮕﻮﻫﺎﯼ ﻓﮑﺮﯼ ﭼﻨﺪﻳﻦ ﻭ ﭼﻨﺪ ﺳﺎﻟﻪ ﺧﻮﺩ ﺭﻓﺘﻪ ﺍﻳﺪ. ﻗﻄﻌﺎ ﺯﻣﺎﻥ ﻻﺯﻡ ﺍﺳﺖ ﺗﺎ ﺍﻳﻦ ﺷﻴﻮﻩ ﻫﺎ ﺩﺭ ﺷﻤﺎ ﻧﻬﺎﺩﻳﻨﻪ ﺷﻮﻧﺪ، ﻭﻟﯽ ﺍﻳﻦ ﺍﺗﻔﺎﻕ ﺑﺎ ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﺣﺘﻤﺎ ﻣﯽ ﺍﻓﺘﺪ.

 

ﺳﺮﺍﻍ ﺭﻭﺵ ﺑﻌﺪﯼ ﺑﺮﻭﻳﺪ
ﺍﮔﺮ ﺍﺣﺴﺎﺱ ﻣﯽ ﮐﻨﻴﺪ ﺗﮑﻨﻴﮏ ﻭ ﺭﻭﺷﯽ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ ﺍﻳﺪ ﺑﻪ ﻫﺮ ﺩﻟﻴﻞ ﻣﻨﺎﺳﺐ ﺷﻤﺎ ﻧﻴﺴﺖ ﻭ ﺍﻧﮕﺎﺭ ﺑﺮﺍﯼ ﺷﻤﺎ ﮐﺎﺭ ﻧﻤﯽ ﮐﻨﺪ. ﺍﺻﻼ ﻧﮕﺮﺍﻥ ﻧﺒﺎﺷﻴﺪ. ﻗﺮﺍﺭ ﻧﻴﺴﺖ ﺗﻤﺎﻡ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﺑﺮﺍﯼ ﻫﻤﻪ ﮐﺎﺭ ﮐﻨﺪ. ﺍﻳﻦ ﺍﺑﺰﺍﺭﻫﺎ ﻣﺜﻞ ﺁﭼﺎﺭ ﻧﻴﺴﺘﻨﺪ ﮐﻪ ﺣﺘﻤﺎ ﺑﺎﻳﺪ ﭘﻴﭻ ﺑﺎﺯ ﻭ ﺑﺴﺘﻪ ﮐﻨﻨﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺗﮑﻨﻴﮏ/ﺷﻴﻮﻩ ﻣﻨﺎﺳﺐ ﺷﻤﺎ ﻧﺒﺎﺷﺪ، ﭘﺲ ﺑﺮﻭﻳﺪ ﺳﺮﺍﻍ ﺑﻌﺪﯼ. ﻳﮏ ﺭﻭﺵ ﺩﻳﮕﺮ ﺭﺍ ﺍﻣﺘﺤﺎﻥ ﮐﻨﻴﺪ. ﺩﻗﻴﻘﺎ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺭﻭﺵ ﻫﺎ ﻭ ﺷﻴﻮﻩ ﻫﺎﯼ ﻣﺘﻌﺪﺩﯼ ﺩﺭ ﺩﻭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺍﺳﺖ.

 

ﺍﻳﺮﺍﺩ ﮐﺎﺭ ﮐﺠﺎﺳﺖ
ﺍﮔﺮ ﺗﺤﺖ ﻫﻴﭻ ﺷﺮﺍﻳﻄﯽ، ﺣﺘﯽ ﺑﺎ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻦ ﺗﻤﺎﻡ ﺷﻴﻮﻩ ﻫﺎ ﻭ ﺗﮑﻨﻴﮏ ﻫﺎﻳﯽ ﮐﻪ ﻳﺎﺩ ﮔﺮﻓﺘﻴﺪ، ﺍﻳﺪﻩ ﻧﻮ ﻭ ﺟﺪﻳﺪﯼ ﺑﻪ ﺫﻫﻦ ﺗﺎﻥ ﻧﻤﯽ ﺭﺳﺪ، ﺣﺘﻤﺎ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺑﺪﺍﻧﻴﺪ ﺍﻳﺮﺍﺩ ﮐﺎﺭ ﮐﺠﺎﺳﺖ ﻭ ﺷﺎﻳﺪ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﮕﻮﻳﻴﺪ ﺍﻳﻦ ﺭﻭﺵ ﻫﺎ ﻭ ﺗﮑﻨﻴﮏ ﻫﺎ ﺑﻪ ﮐﺎﺭ ﻧﻤﯽ ﺁﻳﻨﺪ ﻭ ﺍﻣﮑﺎﻥ ﻧﺪﺍﺭﺩ ﺑﺘﻮﺍﻥ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﻬﺘﺮ ﺍﺯ ﺍﻳﻦ ﺍﻧﺠﺎﻡ ﺩﺍﺩ! ﺍﮔﺮ ﺩﺭ ﭼﻨﻴﻦ ﻣﻮﻗﻌﻴﺘﯽ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻴﺪ ﺑﻪ ﻃﻮﺭ ﺟﺪﯼ ﺑﻪ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﺮﺍﺩ ﺍﺯ ﺷﻤﺎﺳﺖ ﻳﺎ ﺍﺯ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺷﻴﻮﻩ ﻫﺎ.

ﺑﻪ ﻃﻮﺭ ﻗﻄﻊ ﺁﺩﻡ ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﭼﻨﺎﻥ ﺑﻪ ﻋﻘﺎﻳﺪ ﻭ ﺷﻴﻮﻩ ﻫﺎﯼ ﻗﺪﻳﻤﯽ ﺧﻮﺩ ﻋﺎﺩﺕ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﺁﻧﻬﺎ ﻭﺍﺑﺴﺘﻪ ﺍﻧﺪ ﮐﻪ ﻭﺍﻗﻌﺎ ﺍﻫﺮﻡ ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﻗﻮﯼ ﻻﺯﻡ ﺍﺳﺖ ﺗﺎ ﺑﺘﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﺍﺯ ﺍﻳﻦ ﺗﻌﻠﻖ ﺟﺪﺍ ﮐﺮﺩ.

ﺍﻟﺒﺘﻪ ﺑﻌﻴﺪ ﻣﯽ ﺩﺍﻧﻢ ﺷﻤﺎ ﺍﺯ ﺍﻳﻦ ﺩﺳﺘﻪ ﺁﺩﻡ ﻫﺎ ﺑﺎﺷﻴﺪ، ﮐﻪ ﺍﮔﺮ ﺑﻮﺩﻳﺪ ﺩﺭ ﺩﻭﺭﻩ «ﺗﻔﮑﺮﺧﻼﻕ» ﺷﺮﮐﺖ ﻧﻤﯽ ﮐﺮﺩﻳﺪ!

 

ﺗﻀﻤﻴﻦ ﻣﺎ ﺑﻪ ﺷﻤﺎ
ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﺎ ﺿﻤﺎﻧﺖ ﻫﻢ ﻣﯽ ﺩﻫﻴﻢ. ﺍﮔﺮ ﺍﺯ ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺷﻴﻮﻩ ﻫﺎ ﻳﺎ ﺗﮑﻨﻴﮏ ﻫﺎﻳﯽ ﮐﻪ ﻳﺎﺩ ﮔﺮﻓﺘﻪ ﺍﻳﺪ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﺪ، ﺍﻳﻨﻬﺎ ﺑﻪ ﮐﺎﺭ ﻧﻤﯽ ﺁﻳﻨﺪ، ﺍﻳﻦ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺗﻀﻤﻴﻦ ﮐﺮﺩ.

ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ، ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺍﺯ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﺍﺳﺘﻔﺎﺩ ﮐﻨﻴﺪ، ﺁﻧﻬﺎ ﺑﻬﺘﺮ ﻭ ﮐﺎﺭﺁﻣﺪﺗﺮ ﺧﻮﺍﻫﻨﺪ ﺷﺪ. ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺍﺯ ﺍﻳﻦ ﺷﻴﻮﻩ ﻫﺎ ﻭ ﺗﮑﻨﻴﮏ ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﻣﺎﻫﺮﺗﺮ ﻣﯽ ﺷﻮﻳﺪ ﻭ ﺩﺭ ﺣﻘﻴﻘﺖ ﺷﻴﻮﻩ ﻫﺎﯼ ﻓﮑﺮ ﮐﺮﺩﻥ ﺧﻼﻗﺎﻧﻪ ﺩﺭ ﺷﻤﺎ ﻧﻬﺎﺩﻳﻨﻪ ﻣﯽ ﺷﻮﺩ.

 

ﭼﺮﺍ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﺭﺍ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﯽ ﮐﻨﻴﻢ؟
ﺩﻟﻴﻠﺶ ﺳﺎﺩﻩ ﺍﺳﺖ، ﻫﻨﻮﺯ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺷﻴﻮﻩ ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺭ ﺷﻤﺎ ﻧﻬﺎﺩﻳﻨﻪ ﻧﺸﺪﻩ ﻭ ﺩﻟﻴﻞ ﺁﻥ ﻫﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺍﺗﻔﺎﻕ ﺳﺮﻳﻊ ﻭ ﻟﺤﻈﻪ ﺍﯼ ﻧﻤﯽ ﺍﻓﺘﺪ. ﺷﺎﻳﺪ ﻧﻴﺎﺯ ﺑﻪ ﺳﺎﻝ ﻫﺎ ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺁﻧﮕﻮﻧﻪ ﻓﮑﺮ ﮐﻨﻴﺪ ﻭ ﺍﻳﻦ ﻧﻮﻉ ﻓﮑﺮ ﮐﺮﺩﻥ ﻭ ﻧﮕﺎﻩ ﺑﻪ ﻣﺸﮑﻼﺕ/ﭼﺎﻟﺶ ﻫﺎ ﻣﻠﮑﻪ ﺫﻫﻦ ﺷﻤﺎ ﺷﻮﺩ.

ﻧﮑﺘﻪ ﺍﯼ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﺯﻣﺎﻥ ﺑﺮ ﻫﺴﺘﻨﺪ ﺗﺎ ﺷﻤﺎ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﻣﺘﻌﺪﺩ ﻓﮑﺮ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﺣﻞ ﻳﮏ ﻣﺸﮑﻞ ﻭ ﻳﺎ ﺑﺮﺍﯼ ﻳﺎﻓﺘﻦ ﺭﺍﻩ ﺣﻞ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻫﻴﭻ ﻗﻴﺪﯼ ﻣﻘﻴﺪ ﻧﮑﻨﻴﺪ.

ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﺪ ﻭ ﺍﺭﺍﺩﻩ ﮐﻨﻴﺪ، ﺑﺎ ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﺍﻳﻦ ﺷﻴﻮﻩ ﻫﺎ، ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﺻﻮﺭﺕ ﺧﻼﻗﺎﻧﻪ ﺑﻪ ﺑﺨﺸﯽ ﺍﺯ ﺯﻧﺪﮔﯽ ﺷﻤﺎ ﺗﺒﺪﻳﻞ ﺧﻮﺍﻫﺪ ﺷﺪ ﻭ ﺩﻳﮕﺮ ﻧﻤﯽ ﺗﻮﺍﻧﻴﺪ ﺧﻼﻕ ﻧﺒﺎﺷﻴﺪ ﻭ ﺧﻼﻗﺎﻧﻪ ﻓﮑﺮ ﻧﮑﻨﻴﺪ.

 

ﺧﻮﺩﺁﮔﺎﻩ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﻣﯽ ﺷﻮﺩ
ﺣﺘﻤﺎ ﺩﺭ ﻣﻮﺭﺩ ﺑﻌﻀﯽ ﺑﺎﺯﻳﮑﻦ ﻫﺎﯼ ﻓﻮﺗﺒﺎﻝ ﺩﻳﺪﻩ ﻳﺎ ﺷﻨﻴﺪﻩ ﺍﻳﺪ، ﺿﺮﺑﺎﺗﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺁﻧﻬﺎ ﺣﺘﻤﺎ ﮔﻞ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺍﻣﮑﺎﻥ ﻧﺪﺍﺭﺩ ﺁﻥ ﺿﺮﺑﻪ ﻫﺎ ﺭﺍ ﮔﻞ ﻧﮑﻨﻨﺪ، ﺍﻳﻦ ﻧﻮﻉ ﺣﺮﮐﺖ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﻧﺎﺧﻮﺁﮔﺎﻩ ﺷﺪﻩ ﺍﺳﺖ.

ﺁﻥ ﺑﺎﺯﻳﮑﻦ ﺩﺭ ﺁﻥ ﻟﺤﻈﻪ ﻓﮑﺮ ﻧﻤﯽ ﮐﻨﺪ ﺑﻪ ﺁﻧﭽﻪ ﻳﺎﺩﮔﺮﻓﺘﻪ، ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﺩﺭﺳﺖ ﺑﻪ ﺗﻮﭘﯽ ﮐﻪ ﺷﻮﺕ ﺷﺪﻩ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ ﻭ ﮔﻞ ﻣﯽ ﺯﻧﺪ. ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﺯﻳﺎﺩ ﺍﻳﻦ ﻣﻬﺎﺭﺕ ﺭﺍ  ﺍﺯ ﺳﻄﺢ ﺧﻮﺩﺁﮔﺎﻩ ﺑﻪ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺭﺳﺎﻧﺪﻩ ﺍﺳﺖ.

ﺑﻌﺪ ﺍﺯ ﮔﺬﺭﺍﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﻭ ﺣﺘﯽ ﺑﻌﺪ ﺍﺯ ﻫﺮ ﺩﺭﺱ، ﺧﻮﺏ ﺑﻪ ﺁﻧﭽﻪ ﺧﻮﺍﻧﺪﻩ ﺍﻳﺪ ﻓﮑﺮ ﮐﻨﻴﺪ؛ ﺷﺎﻳﺪ ﻻﺯﻡ ﺑﺎﺷﺪ ﺩﻭﺑﺎﺭﻩ ﻭ ﭼﻨﺪ ﺑﺎﺭﻩ ﺑﺨﻮﺍﻧﻴﺪ. ﻭﻟﯽ ﻳﮏ ﺍﺻﻞ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺑﺴﭙﺎﺭﻳﺪ: ﺩﺭ ﺍﺳﺮﻉ ﻭﻗﺖ ﻳﮑﯽ ﺍﺯ ﺷﻴﻮﻩ ﻫﺎﻳﯽ ﮐﻪ ﺁﻣﻮﺧﺘﻪ ﺍﻳﺪ ﺭﺍ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﻭ ﺁﻥ ﺭﺍ ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ.

 

ﻳﺎﺩﻣﺎﻥ ﺑﺎﺷﺪ

  • ﺍﮔﺮ ﺑﻪ ﻣﻮﻗﻊ ﻭ ﻫﺮ ﭼﻪ ﺳﺮﻳﻊ ﺗﺮ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺭﻭﺵ ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺁﻣﻮﺧﺘﻪ ﺍﻳﻢ ﺑﻪ ﮐﺎﺭ ﻧﺒﻨﺪﻳﻢ، ﻓﺮﺍﻣﻮﺷﺸﺎﻥ ﻣﯽ ﮐﻨﻴﻢ. ﻣﺜﻞ ﻫﺮ ﻣﻬﺎﺭﺕ ﺩﻳﮕﺮﯼ.
  • ﺗﮑﺮﺍﺭ ﺑﺎﻋﺚ ﺗﻘﻮﻳﺖ ﻋﺎﺩﺍﺕ ﺟﺪﻳﺪ ﻣﯽ ﺷﻮﺩ. ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺗﮑﺮﺍﺭ ﻭ ﺗﻤﺮﻳﻦ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﺑﻬﺘﺮ ﺍﻳﻦ ﻣﻬﺎﺭﺕ ﻫﺎ ﺩﺭ ﺷﻤﺎ ﻧﻬﺎﺩﻳﻨﻪ ﻣﯽ ﺷﻮﺩ.
  • ﺍﮔﺮ ﻳﮏ ﺗﮑﻨﻴﮏ ﻭ ﺷﻴﻮﻩ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﺑﺮﺍﯼ ﺷﻤﺎ ﻣﻨﺎﺳﺐ ﻧﺒﻮﺩ، ﻧﮕﺮﺍﻥ ﻧﺸﻮﻳﺪ، ﺷﻴﻮﻩ ﺑﻌﺪﯼ ﺭﺍ ﺍﻣﺘﺤﺎﻥ ﮐﻨﻴﺪ.
  • ﺍﮔﺮ ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺷﻴﻮﻩ ﻫﺎ ﺭﺍ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﺪ، ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﻔﻴﺪ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ.
  • ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ ﺭﺍ ﺍﺳﺘﻔﺎﺩﻩ ﻭ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ، ﮐﺎﺭﺑﺮﺩﯼ ﺗﺮ ﻣﯽ ﺷﻮﻧﺪ ﻭ ﻣﻬﺎﺭﺕ ﺷﻤﺎ ﻫﻢ ﺩﺭ ﺑﮑﺎﺭﮔﻴﺮﯼ ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﻭ ﺑﻬﺘﺮ ﺁﻧﻬﺎ ﺑﻴﺸﺘﺮ ﻣﯽ ﺷﻮﺩ.
