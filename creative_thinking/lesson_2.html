ﺩﺭﺱ ﺍﻭﻝ- ﺩﺭ ﻣﻮﺭﺩ ﺗﻔﮑﺮ ﺧﻼﻕ ﺑﻴﺸﺘﺮ ﺑﺪﺍﻧﻴﻢ

ﻗﺒﻞ ﺍﺯ ﻭﺭﻭﺩ ﺑﻪ ﻣﺒﺤﺚ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺗﻤﺮﻳﻦ ﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺑﻴﺸﺘﺮ ﺑﺪﺍﻧﻴﻢ.

ﺑﻴﺸﺘﺮ ﺩﺍﻧﺴﺘﻦ ﻭ ﺧﻮﺍﻧﺪﻥ ﺩﺭ ﻣﻮﺭﺩ ﮐﺎﺭﺍﻳﯽ ﺗﻔﮑﺮ ﺑﺪﻳﻊ ﻭ ﺧﻼﻗﺎﻧﻪ، ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ ﮐﻨﺪ ﺗﺎ ﻧﻴﺎﺯ ﺑﻪ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﭼﻨﻴﻦ ﻣﻬﺎﺭﺗﯽ ﺭﺍ ﺑﻬﺘﺮ ﻭ ﺑﻴﺸﺘﺮ ﺣﺲ ﮐﻨﻴﻢ. ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ ﺍﻳﻦ ﻓﺮﺻﺖ ﺭﺍ ﺑﻪ ﻣﺎ ﻣﯽ ﺩﻫﺪ ﺗﺎ ﺑﻪ ﺭﻭﺵ ﻫﺎﯼ ﺧﻮﺩﻣﺎﻥ ﻫﻢ ﻓﮑﺮ ﮐﻨﻴﻢ ﺗﺎ ﺑﻬﺘﺮ ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﻢ ﭼﻪ ﺧﺼﻮﺻﻴﺖ ﻳﺎ ﻃﺮﺯ ﻓﮑﺮﯼ ﺩﺭ ﻣﺎ ﻣﺎﻧﻊ ﺧﻼﻕ ﻓﮑﺮ ﮐﺮﺩﻥ ﺍﺳﺖ.

 

ﻃﺮﺯ ﺗﻔﮑﺮﻣﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﺗﻐﻴﻴﺮ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﻢ
ﺧﻼﻗﻴﺖ ﭼﻪ ﻣﯽ ﮐﻨﺪ؟ ﺧﻼﻗﻴﺖ ﺗﻐﻴﻴﺮ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﺪ ﮐﻪ ﺍﻳﻦ ﺧﻮﺩ ﻣﯽ ﺗﻮﺍﻧﺪ ﻣﺴﺌﻠﻪ ﺳﺎﺯ ﺷﻮﺩ. ﺍﻳﻦ ﻣﺸﮑﻼﺕ ﺧﻮﺩ ﻣﻨﺠﺮ ﺑﻪ ﺍﺩﺍﻣﻪ ﭼﺮﺧﻪ ﺧﻼﻗﻴﺖ ﻣﯽ ﺷﻮﻧﺪ. ﻣﺴﺌﻠﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺎ ﮐﺠﺎﯼ ﺍﻳﻦ ﭼﺮﺧﻪ ﻫﺴﺘﻴﻢ؟

ﺗﻐﻴﻴﺮ ﺍﺟﺘﻨﺎﺏ ﻧﺎﭘﺬﻳﺮ ﺍﺳﺖ. ﺑﻪ ﻫﺮﺣﺎﻝ ﺯﻣﺎﻧﯽ ﺍﺗﻔﺎﻕ ﺧﻮﺍﻫﺪ ﺍﻓﺘﺎﺩ ﻳﺎ ﮐﺴﯽ ﺁﻥ ﺭﺍ ﺍﻳﺠﺎﺩ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺍﻳﻦ ﻣﺎ ﻫﺴﺘﻴﻢ ﮐﻪ ﻫﻤﮕﯽ ﺑﺎﻳﺪ ﺧﻼﻗﺎﻧﻪ ﻣﺸﮑﻼﺕ ﺭﺍ ﺣﻞ ﮐﻨﻴﻢ. ﺍﮔﺮ ﺷﻤﺎ ﻧﻘﺶ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﺻﺮﻓﺎ ﺑﻪ ﻣﺸﮑﻼﺕ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ ﺑﺎﺯﯼ ﮐﻨﻴﺪ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﮐﻪ ﺑﻪ ﺣﻞ ﻣﺸﮑﻠﯽ ﮐﻪ ﺑﻪ ﺳﺒﺐ ﺗﻐﻴﻴﺮ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﭘﺎﺳﺦ ﺩﻫﻴﺪ، ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﺭﻗﻴﺒﺎﻥ ﺧﻮﺩ ﺧﻮﺍﻫﻴﺪ ﺑﻮﺩ. ﺍﮔﺮ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺑﺮﻧﺪﻩ ﺑﺎﺯﯼ ﺑﺎﺷﻴﺪ ﻭ ﺍﺑﺘﮑﺎﺭ ﻋﻤﻞ ﺑﻪ ﺩﺳﺖ ﺧﻮﺩﺗﺎﻥ ﺑﺎﺷﺪ، ﻧﺎﭼﺎﺭﻳﺪ ﺧﻼﻕ ﺑﺎﺷﻴﺪ.

ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻋﺎﻣﻠﻴﺖ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺣﻞ ﻣﺴﺎﺋﻞ ﺯﻧﺪﮔﯽ ﭼﻪ ﺩﺭ ﺣﻮﺯﻩ ﺷﺨﺼﯽ ﻭ ﭼﻪ ﺩﺭ ﺣﻮﺯﻩ ﺷﻐﻠﯽ ﺣﻔﻆ ﮐﻨﻴﻢ، ﺑﺎﻳﺪ ﺗﻐﻴﻴﺮ ﺍﻳﺠﺎﺩ ﮐﻨﻴﻢ؛ ﻧﻤﯽ ﺗﻮﺍﻥ ﺍﺯ ﻧﻘﻄﻪ ﺍﯼ ﻭﺍﺭﺩ ﻋﻤﻞ ﺷﺪ ﮐﻪ ﺑﻪ ﺩﻟﻴﻞ ﺗﻐﻴﻴﺮ، ﻣﺸﮑﻠﯽ ﺑﺮﻭﺯ ﮐﺮﺩﻩ ﺍﺳﺖ، ﺑﻠﮑﻪ ﺑﺎﻳﺪ ﺍﺑﺘﮑﺎﺭﯼ ﺑﻪ ﮐﺎﺭ ﺑﺒﻨﺪﻳﻢ ﺗﺎ «ﻃﺮﺣﯽ ﻧﻮ ﺩﺭ ﺍﻧﺪﺍﺯﻳﻢ» ﻭ ﺗﻐﻴﻴﺮ ﺑﯽ ﺁﻓﺮﻳﻨﻴﻢ.

 

ﺑﺮﻧﺪﻩ ﮐﺴﯽ ﺍﺳﺖ ﮐﻪ ﺁﻓﺮﻳﻨﻨﺪﻩ ﻣﺸﮑﻞ ﻭ ﺧﺎﻟﻖ ﺗﻐﻴﻴﺮ ﺑﺎﺷﺪ.
ﺑﻠﻪ، ﻗﻄﻌﺎ ﻭﻗﺘﯽ ﺷﻤﺎ ﺗﻐﻴﻴﺮ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻭﺭﻳﺪ، ﻣﺸﮑﻞ ﻫﻢ ﺑﺮﺍﯼ ﺷﻤﺎ ﺍﻳﺠﺎﺩ ﻣﯽ ﺷﻮﺩ، ﻭﻟﯽ ﮐﻴﺴﺖ ﮐﻪ ﻫﺮ ﺭﻭﺯ ﻋﻤﺮﺵ ﺩﺭ ﺣﺎﻝ ﺩﺳﺖ ﻭ ﭘﻨﺠﻪ ﻧﺮﻡ ﮐﺮﺩﻥ ﺑﺎ ﻣﺸﮑﻼﺕ ﻧﺒﺎﺷﺪ!

ﻭﻗﺘﯽ ﺷﻤﺎ ﺑﺎﺷﻴﺪ ﮐﻪ ﺗﻐﻴﻴﺮ ﺭﺍ ﺑﻪ ﻭﺟﻮﺩ ﻣﻴﺎﻭﺭﻳﺪ، ﺷﻤﺎﻳﻴﺪ ﮐﻪ ﺑﻘﻴﻪ ﻣﺴﻴﺮ ﺭﺍ ﻫﺪﺍﻳﺖ ﻣﯽ ﮐﻨﻴﺪ.

ﺁﻧﭽﻪ ﺩﺭ ﻣﻮﺭﺩ ﺗﻐﻴﻴﺮ ﺟﺎﻟﺐ ﺍﺳﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ ﺑﺮﺍﯼ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﺑﻬﺘﺮﻳﻦ ﻫﺎﺳﺖ ﻭﻟﯽ ﺑﻴﺸﺘﺮ ﺁﺩﻡ ﻫﺎ ﺩﺭ ﺑﺮﺍﺑﺮﺵ ﻣﻘﺎﻭﻣﺖ ﻣﯽ ﮐﻨﻨﺪ. ﺑﻌﻀﯽ ﻫﺎ ﻣﺜﻞ ﻃﺎﻋﻮﻥ ﺍﺯ ﺗﻐﻴﻴﺮ ﻓﺮﺍﺭ ﻣﯽ ﮐﻨﻨﺪ.

ﺍﮔﺮ ﺷﻤﺎ ﺳﻌﯽ ﮐﻨﻴﺪ، ﻧﻮﻉ ﻧﮕﺎﻩ ﻭ ﻃﺮﺯ ﻓﮑﺮﺗﺎﻥ ﺑﻪ ﺗﻐﻴﻴﺮ ﺭﺍ ﺗﻐﻴﻴﺮ ﺭﺍ ﺩﻫﻴﺪ، ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﻣﺴﻴﺮ ﺗﻘﻮﻳﺖ ﺗﻔﮑﺮ ﺧﻼﻗﺎﻧﻪ ﻫﺮ ﭼﻘﺪﺭ ﮐﻪ ﺑﺨﻮﺍﻫﻴﺪ ﺟﻠﻮ ﺭﻭﻳﺪ.

 

ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ
ﺷﻴﻮﻩ ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺑﺮﺍﯼ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﻧﺘﺎﻳﺞ ﺧﻼﻗﺎﻧﻪ ﺷﻨﺎﺧﺘﻪ ﺷﺪﻩ ﺍﻧﺪ ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﺮﺍﯼ ﺁﺷﻨﺎﻳﯽ ﺑﻴﺸﺘﺮ ﺑﻪ ﭘﻨﺞ ﺭﻭﺵ ﮐﻼﺳﻴﮏ ﺍﺷﺎﺭﻩ ﻣﯽ ﮐﻨﻴﻢ.

ﺗﺪﺭﻳﺠﯽ
ﺍﻳﻦ ﺭﻭﺵ ﺑﻪ ﺻﻮﺭﺕ ﺗﺪﺭﻳﺠﯽ ﻋﻤﻞ ﻣﯽ ﮐﻨﺪ. ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﮐﻪ ﺍﻳﺪﻩ ﻫﺎﯼ ﺟﺪﻳﺪ ﺑﻪ ﺍﻳﺪﻩ ﻫﺎﯼ ﺩﻳﮕﺮ ﺷﺎﺥ ﻭ ﺑﺮﮒ ﻣﯽ ﺩﻫﻨﺪ، ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﺟﺪﻳﺪ ﺍﺯ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﻗﺒﻠﯽ ﺍﻟﻬﺎﻡ ﻣﯽ ﮔﻴﺮﻧﺪ ﻭ ﺟﺪﻳﺪﺗﺮ ﻗﺒﻠﯽ ﺭﺍ ﺗﮑﻤﻴﻞ ﻣﯽ ﮐﻨﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﺑﺰﺍﺭﻫﺎﯼ ﭘﻴﭽﻴﺪﻩ ﺍﯼ ﮐﻪ ﻣﺎ ﺍﻣﺮﻭﺯﻩ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻴﻢ، ﺣﺎﺻﻞ ﻳﮏ ﺩﻭﺭﻩ ﻃﻮﻻﻧﯽ ﻭ ﺗﺮﻳﺠﯽ ﭘﻴﺸﺮﻓﺖ ﻫﺴﺘﻨﺪ. ﺑﻪ ﺗﺪﺭﻳﺞ ﻭ ﺑﻪ ﺁﻫﺴﺘﮕﯽ ﺑﻬﺘﺮ ﻭ ﺑﻬﺘﺮ ﺷﺪﻩ ﺍﻧﺪ ﺗﺎ ﺑﻪ ﻣﺤﺼﻮﻟﯽ ﮔﺎﻫﯽ ﮐﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﻭ ﺑﻪ ﻣﺮﺍﺗﺐ ﺑﺎ ﮐﺎﺭﺁﻳﯽ ﭘﻴﭽﻴﺪﻩ ﺗﺮ ﺗﺒﺪﻳﻞ ﺷﺪﻩ ﺍﻧﺪ.

ﻳﮏ ﻣﺜﺎﻝ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﭘﻴﺸﺮﻓﺖ ﺧﻮﺩﺭﻭﻫﺎﺳﺖ ﭼﻄﻮﺭ ﻫﺮ ﻣﺪﻝ ﺑﺎ ﺑﮑﺎﺭﮔﻴﺮﯼ ﺭﻭﺵ ﻫﺎﯼ ﺧﻼﻕ ﻭ ﻧﻮﺁﻭﺭﯼ ﺑﻪ ﺗﺪﺭﻳﺞ ﻣﺪﻝ ﻗﺒﻠﯽ ﺭﺍ ﮐﺎﻣﻞ ﺗﺮ ﻭ ﺑﻬﺘﺮ ﻣﯽ ﮐﻨﺪ ﺑﻪ ﺷﮑﻠﯽ ﮐﻪ ﻣﺜﻼ ﺩﺭ ﻳﮏ ﺑﺎﺯﻩ ﺯﻣﺎﻧﯽ ﺩﻩ ﺳﺎﻟﻪ ﺗﻐﻴﻴﺮ ﺍﻳﻦ ﻓﻨﺎﻭﺭﯼ ﻭ ﻣﺤﺼﻮﻝ ﺑﺮﺍﯼ ﺷﻤﺎ ﮐﺎﻣﻼ ﻣﺤﺴﻮﺱ ﻭ ﺷﮕﻔﺖ ﺁﻭﺭ ﺍﺳﺖ.

ﺍﺻﻼ ﻣﻮﺑﺎﻳﻞ ﻫﺎﻳﻤﺎﻥ ﺑﻬﺘﺮﻳﻦ ﻣﺜﺎﻝ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺭ ﻃﻮﻝ ﺑﻴﺶ ﺍﺯ ﻳﮏ ﺩﻫﻪ ﭼﻘﺪﺭ ﺗﮑﺎﻣﻞ ﭘﻴﺪﺍ ﮐﺮﺩﻧﺪ ﻭ ﺑﻪ ﺗﺪﺭﻳﺞ ﺍﻣﮑﺎﻧﺎﺕ ﺑﻴﺸﺘﺮﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻣﺎ ﻗﺮﺍﺭ ﺩﺍﺩﻧﺪ. ﻭﺍﻗﻌﺎ ﺑﺎﻭﺭﻧﮑﺮﺩﻧﯽ ﺍﺳﺖ!

ﺍﻳﻦ ﺷﻴﻮﻩ  ﺍﻳﻦ ﻧﮑﺘﻪ ﻣﻬﻢ ﺭﺍ ﺑﻪ ﻣﺎ ﻳﺎﺩﺁﻭﺭﯼ ﻣﯽ ﮐﻨﺪ ﮐﻪ ﻫﺮ ﻣﺸﮑﻞ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﻭﺑﺎﺭﻩ ﻭ ﭼﻨﺪ ﺑﺎﺭﻩ ﺑﻪ ﺷﮑﻞ ﺑﻬﺘﺮ ﻭ ﻣﻮﺛﺮﺗﺮﯼ ﺣﻞ ﺷﻮﺩ.

ﻓﺮﺩﯼ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﺪ ﺧﻼﻕ ﺑﺎﺷﺪ، ﻫﻴﭽﻮﻗﺖ ﻓﮑﺮ ﻧﻤﯽ ﮐﻨﺪ ﻣﺸﮑﻠﯽ ﮐﻪ ﺣﻞ ﺷﺪ ﻓﺮﺍﻣﻮﺵ ﻣﯽ ﺷﻮﺩ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﮔﺮ ﮐﺎﺭ ﻣﯽ ﮐﻨﺪ ﺑﮕﺬﺍﺭ ﮐﺎﺭ ﮐﻨﺪ، «ﺍﮔﺮ ﻧﺸﮑﺴﺘﻪ ﭼﯽ ﺭﻭ ﻣﯽ ﺧﻮﺍﯼ ﺩﺭﺳﺖ ﮐﻨﯽ؟!!». ﻫﻤﻴﺸﻪ ﺭﺍﻩ ﺣﻞ/ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﺑﻬﺘﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

 

ﺗﺮﮐﻴﺐ ﺍﻳﺪﻩ ﻫﺎ
ﺩﺭ ﺍﻳﻦ ﺭﻭﺵ ﺩﻭ (ﻳﺎ ﺑﻴﺸﺘﺮ ﺍﺯ ﺩﻭ) ﺍﻳﺪﻩ ﻣﻮﺟﻮﺩ، ﺍﻳﺪﻩ ﻧﻮﯼ ﺳﻮﻣﯽ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ ﺩﻫﻨﺪ.

ﻣﺜﻼ ﮐﺘﺎﺏ ﻫﺎﯼ ﺻﻮﺗﯽ ﮐﻪ ﺍﺯ ﺗﺮﮐﻴﺐ ﺍﻳﺪﻩ ﮐﺘﺎﺏ ﻭ ﻓﺎﻳﻞ ﺻﻮﺗﯽ ﺗﺸﮑﻴﻞ ﺷﺪﻩ ﮐﻪ ﻫﻢ ﺑﺮﺍﯼ ﻧﺎﺑﻴﻨﺎﻳﺎﻥ ﻣﻔﻴﺪ ﺍﺳﺖ ﻭ ﻫﻢ ﺑﺮﺍﯼ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﺭﻭﺯ ﻣﺠﺒﻮﺭ ﻫﺴﺘﻨﺪ ﺳﺎﻋﺖ ﻫﺎ ﺭﺍﻧﻨﺪﮔﯽ ﮐﻨﻨﺪ.

 

ﺍﻳﺪﻫﺎﯼ ﺍﻧﻘﻼﺑﯽ
ﺩﺭ ﺑﻌﻀﯽ ﺍﺯ ﻣﻮﺍﺭﺩ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﺪﻩ ﺟﺪﻳﺪ ﮐﺎﻣﻼ ﺑﺎ ﺁﻧﭽﻪ ﻗﺒﻼ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﻓﺮﻕ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻣﺜﻼ ﺍﮔﺮ ﻳﮏ ﺍﺳﺘﺎﺩ ﺩﺍﻧﺸﮕﺎﻩ ﺑﭙﺮﺳﺪ ﻣﻦ ﭼﻄﻮﺭ ﺩﺭ ﮐﻼﺱ ﻫﺎﻳﻢ ﺑﻬﺘﺮ ﻭ ﺑﻬﺘﺮ ﺳﺨﻨﺮﺍﻧﯽ ﮐﻨﻢ، ﭘﺎﺳﺦ ﺍﻳﻦ ﺭﻭﺵ ﺍﻳﻦ ﺍﺳﺖ: ﭼﺮﺍ ﺍﺯ ﺍﺳﺎﺱ ﺷﻴﻮﻩ ﺗﺪﺭﻳﺲ ﺭﺍ ﻋﻮﺽ ﻧﻤﯽ ﮐﻨﻴﺪ؟ ﺷﻤﺎ ﺩﻳﮕﺮ ﺗﻨﻬﺎ ﺳﺨﻨﮕﻮ ﻭ ﻣﺪﺭﺱ ﻧﺒﺎﺷﻴﺪ، ﭼﺮﺍ ﺍﺯ ﺧﻮﺩ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ ﺑﺮﺍﯼ ﺗﺪﺭﻳﺲ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﯽ ﮐﻨﻴﺪ؟ ﭼﺮﺍ ﺩﺭ ﻗﺎﻟﺐ ﻭﺍﮔﺬﺍﺭﯼ ﻣﺒﺎﺣﺚ ﺑﻪ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ ﺩﺭ ﻣﺪﻝ ﻫﺎﯼ ﮐﺎﺭ ﮔﺮﻭﻫﯽ، ﻣﺒﺎﺣﺚ ﺁﻣﻮﺯﺷﯽ ﺭﺍ ﻣﻨﺘﻘﻞ ﻧﻤﯽ ﮐﻨﻴﺪ؟

 

ﺩﻭﺑﺎﺭﻩ ﺑﮑﺎﺭﮔﺮﻓﺘﻦ ﺍﻳﺪﻩ ﻫﺎ
ﺑﻪ ﺁﻧﭽﻪ ﻫﺴﺖ ﺟﻮﺭ ﺩﻳﮕﺮﯼ ﻧﮕﺎﻩ ﮐﻨﻴﻢ، ﺍﺯ ﻋﻨﻮﺍﻥ ﻫﺎ ﻓﺮﺍﺗﺮ ﺑﺮﻭﻳﻢ. ﺩﮔﻢ ﻭ ﺑﺴﺘﻪ ﻧﺒﺎﺷﻴﻢ، ﭘﻴﺶ ﻓﺮﺽ ﻫﺎ ﻭ ﺗﺼﻮﺭﺍﺕ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﮐﻨﺎﺭ ﺑﮕﺬﺍﺭﻳﻢ ﻭ ﺑﻪ ﺧﻮﺩﻣﺎﻥ ﺍﻳﻦ ﻓﺮﺻﺖ ﺭﺍ ﺑﺪﻫﻴﻢ ﺗﺎ ﺑﺎ ﮐﺸﻒ ﺁﻧﭽﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﻢ، ﺑﺎ ﻧﮕﺎﻫﯽ ﺧﻼﻗﺎﻧﻪ ﻭ ﻣﺘﻔﺎﻭﺕ ﺑﻪ ﺷﮑﻞ ﺩﻳﮕﺮﯼ ﺁﻥ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﮔﻴﺮﻳﻢ.

ﻳﮏ ﻣﺜﺎﻝ ﺳﺎﺩﻩ ﻭ ﺭﻭﺯﻣﺮﻩ ﺍﺵ ﺍﺳﺘﻔﺎﺩﻩ ﺩﻭﺑﺎﺭﻩ ﻭ ﮐﺎﺭﺑﺮﺩﯼ ﺍﺯ ﺯﺑﺎﻟﻪ ﻫﺎﺳﺖ ﮐﻪ ﻋﻤﻮﻣﺎ ﻫﻢ ﺑﻪ ﺳﺨﺘﯽ ﺑﺎﺯﻳﺎﻓﺖ ﻣﯽ ﺷﻮﻧﺪ ﻭ ﺑﻪ ﭼﺮﺧﻪ ﻃﺒﻴﻌﺖ ﺑﺎﺯﻣﯽ ﮔﺮﺩﻧﺪ.

ﺯﺑﺎﻟﻪ ﻫﺎﯼ ﺍﻳﻨﭽﻨﻴﻨﯽ ﻓﻀﺎﯼ ﺯﻳﺎﺩﯼ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﺍﺷﻐﺎﻝ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺩﻭﺭ ﺍﻧﺪﺍﺧﺘﻦ ﺁﻧﻬﺎ ﺑﺎ ﺯﺑﺎﻟﻪ ﻫﺎﯼ ﻣﻌﻤﻮﻟﯽ ﺑﻪ ﻃﺒﻴﻌﺖ ﺁﺳﻴﺐ ﺟﺪﯼ ﻭﺍﺭﺩ ﻣﯽ ﮐﻨﺪ ﻭ ﺑﺎﺯﻳﺎﻓﺖ ﺁﻧﻬﺎ ﻫﻢ ﻫﺰﻳﻨﻪ ﺑﺮ ﺍﺳﺖ. ﻭﻟﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩ ﺧﻼﻕ ﻭ ﺣﺎﻣﯽ ﻣﺤﻴﻂ ﺯﻳﺴﺖ ﻭ ﺁﻳﻨﺪﻩ ﺯﻣﻴﻦ، ﺍﺯ ﺯﺑﺎﻟﻪ ﺑﻪ ﺷﮑﻞ ﮐﺎﺭﺑﺮﺩﯼ ﺩﻭﺑﺎﺭﻩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩﻩ ﺍﻧﺪ ﮐﻪ ﺍﺗﻔﺎﻗﺎ ﺳﺎﺩﻩ ﻭ ﻗﺎﺑﻞ ﺍﻧﺠﺎﻡ ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮ ﻣﺎ ﻫﻢ ﻫﺴﺖ.

 

ﺗﻐﻴﻴﺮ ﺟﻬﺖ
ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﺑﺪﻳﻊ ﻭ ﻧﻮ ﻭﻗﺘﯽ ﺑﺮﻭﺯ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﺗﻮﺟﻪ ﻭ ﺗﻤﺮﮐﺰ ﺍﺯ ﻳﮏ ﺳﻮﯼ ﻣﺸﮑﻞ ﺑﻪ ﺑﺨﺶ ﻭ ﺯﻭﺍﻳﺎﯼ ﺩﻳﮕﺮﯼ ﺗﻐﻴﻴﺮ ﻣﯽ ﮐﻨﺪ.

ﻳﮏ ﻣﺜﺎﻝ ﮐﻼﺳﻴﮏ ﺑﺮﺍﯼ ﺍﻳﻦ ﺭﻭﺵ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﺭﻭﺷﻨﯽ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ ﭼﮕﻮﻧﻪ ﺑﺎ ﺗﻐﻴﻴﺮ ﺟﻬﺖ ﻧﮕﺎﻩ ﻣﺎﻥ ﺑﻪ ﻣﺸﮑﻞ، ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﻧﻮ ﺑﻪ ﺳﺮﺍﻍ ﻣﺎﻥ ﻣﯽ ﺁﻳﻨﺪ.

ﺩﺭ ﻳﮑﯽ ﺍﺯ ﺷﻬﺮﻫﺎﯼ ﺑﺰﺭﮒ، ﺷﻬﺮﺩﺍﺭﯼ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﺯ ﺍﺳﮑﻴﺖ ﺳﻮﺍﺭﯼ ﻧﻮﺟﻮﺍﻧﺎﻥ ﺩﺭ ﺍﻧﺤﻨﺎﯼ ﻳﮏ ﺍﺗﻮﺑﺎﻥ ﺟﻠﻮﮔﻴﺮﯼ ﮐﻨﺪ، ﺩﭼﺎﺭ ﻣﺸﮑﻞ ﺑﻮﺩ. ﺍﻭﻝ ﺗﻮﺭﯼ ﻓﻠﺰﯼ ﺩﺭ ﻣﺤﻞ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﺍﻳﻦ ﺍﻧﺤﻨﺎ ﮔﺬﺍﺷﺘﻨﺪ، ﻧﻮﺟﻮﺍﻧﺎﻥ ﺍﺯ ﺭﻭﯼ ﺗﻮﺭ ﻓﻠﺰﯼ ﻣﯽ ﭘﺮﻳﺪﻧﺪ ﻭ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺑﻪ ﻣﺤﻞ ﻣﻮﺭﺩ ﻧﻈﺮ ﮐﻪ ﺑﺴﻴﺎﺭ ﻣﻨﺎﺳﺐ ﺍﺳﮑﻴﺖ ﺑﺎﺯﯼ ﺑﻮﺩ ﻣﯽ ﺭﺳﺎﻧﺪﻧﺪ. ﺩﺭ ﻣﺮﺣﻠﻪ ﺑﻌﺪﯼ ﺣﺼﺎﺭ ﻓﻠﺰﯼ ﺭﺍ ﺑﻠﻨﺪﺗﺮ ﮐﺮﺩﻧﺪ، ﺑﺎﺯ ﻫﻢ ﺍﻓﺎﻗﻪ ﻧﮑﺮﺩ. ﺣﺼﺎﺭ ﻣﻘﺎﻭﻡ ﺗﺮ ﻫﻢ ﺭﺍﻩ ﺣﻞ ﺧﻮﺑﯽ ﻧﺒﻮﺩ ﭼﻮﻥ ﻧﻮﺟﻮﺍﻧﺎﻥ ﺁﻥ ﺭﺍ ﻫﻢ ﺑﺮﻳﺪﻧﺪ ﻭ ﺑﻪ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺑﻪ ﻣﺤﻞ ﻣﻮﺭﺩ ﻧﻈﺮﺷﺎﻥ ﺭﺳﺎﻧﺪﻧﺪ.

ﺑﺎﻻﺧﺮﻩ ﺷﺨﺼﯽ ﺩﺭ ﺁﻥ ﺍﺩﺍﺭﻩ ﭘﻴﺪﺍ ﺷﺪ ﮐﻪ ﺑﭙﺮﺳﺪ ﻣﺸﮑﻞ ﻭﺍﻗﻌﺎ ﭼﻴﺴﺖ؟ ﻣﺤﺪﻭﺩﻳﺖ ﺩﺳﺘﺮﺳﯽ ﻧﻮﺟﻮﺍﻧﺎﻥ ﺑﻪ ﻗﻮﺱ ﺍﺗﻮﺑﺎﻥ ﮐﻪ ﻣﺤﻞ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﻧﻮﺟﻮﺍﻧﺎﻥ ﺑﺮﺍﯼ ﺍﺳﮑﻴﺖ ﺑﺎﺯﯼ ﺍﺳﺖ ﻳﺎ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﺗﻤﺎﻳﻞ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﺧﻄﺮﻧﺎﮎ.

ﺍﻳﻦ ﺑﺎﺭ ﻗﻮﺱ ﺩﺍﺧﻞ ﺍﺗﻮﺑﺎﻥ ﺭﺍ ﺑﺎ ﺑﺘﻮﻥ ﻃﻮﺭﯼ ﭘﺮ ﮐﺮﺩﻧﺪ ﮐﻪ ﺑﻪ ﺟﺎﯼ ﺍﻧﺤﻨﺎ ﺑﻪ ﺯﺍﻭﻳﻪ ﺍﯼ ﺑﺎ ﺷﻴﺐ ﺗﻨﺪ ﺗﺒﺪﻳﻞ ﺷﺪ ﮐﻪ ﺩﻳﮕﺮ ﺟﺎﯼ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺍﺳﮑﻴﺖ ﺑﺎﺯﯼ ﻧﺒﻮﺩ. ﺩﻳﮕﺮ ﮐﺴﯽ ﻣﺎﻳﻞ ﻧﺒﻮﺩ ﺩﺭ ﺁﻥ ﻣﺤﻞ ﺍﺳﮑﻴﺖ ﺑﺎﺯﯼ ﮐﻨﺪ ﺗﺎ ﻻﺯﻡ ﺑﺎﺷﺪ ﺑﺮﺍﯼ ﺣﻔﻆ ﺟﺎﻧﺶ ﺍﺯ ﺍﻳﻦ ﮐﺎﺭﺵ ﺟﻠﻮﮔﻴﺮﯼ ﮐﺮﺩ!

ﺍﻳﻦ ﻣﺜﺎﻝ ﺑﻪ ﺭﻭﺷﻨﯽ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ:

ﻫﺪﻑ ﺣﻞ ﻣﺸﮑﻞ ﺍﺳﺖ، ﻧﻪ ﺍﺟﺮﺍﻳﯽ ﮐﺮﺩﻥ ﻳﮏ ﺭﺍﻩ ﺣﻞ

ﻭﻗﺘﯽ ﻳﮏ ﻧﻮﻉ ﺭﺍﻩ ﺣﻞ ﮐﺎﺭ ﻧﻤﯽ ﮐﻨﺪ ﺑﻪ ﺳﺮﺍﻍ ﺭﺍﻩ ﻫﺎﯼ ﺩﻳﮕﺮ ﺑﺮﻭﻳﺪ. ﻫﻴﭻ ﺍﺣﺴﺎﺱ ﺗﻌﻬﺪ ﻭ ﻭﻇﻴﻔﻪ ﺍﯼ ﺑﻪ ﻳﮏ ﺭﺍﻩ ﺣﻞ ﻣﺸﺨﺺ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻫﺪﻑ ﮐﻪ ﻫﻤﺎﻥ ﺣﻞ ﻣﺸﮑﻞ ﺍﺳﺖ ﻣﻬﻢ ﺍﺳﺖ.

ﺣﻮﺍﺱ ﻣﺎﻥ ﺑﺎﺷﺪ ﮔﺎﻫﯽ ﺑﻪ ﻳﮏ ﺭﺍﻩ ﺣﻞ ﮐﻪ ﻗﺒﻼ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﺍﻳﻢ ﻭ ﺍﻟﺰﺍﻣﺎ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ ﺣﻞ ﻧﻴﺴﺖ ﻭ ﻣﻮﺛﺮ ﻫﻢ ﻧﻴﺴﺖ ﺁﻧﻘﺪﺭ ﻧﭽﺴﺒﻴﻢ ﮐﻪ ﺍﺯ ﺭﺍﻩ ﻫﺎﯼ ﺩﻳﮕﺮ ﻏﺎﻓﻞ ﺷﻮﻳﻢ ﮐﻪ ﻧﺘﻴﺠﻪ ﺍﺵ ﻓﻘﻂ ﺧﺴﺘﮕﯽ ﻭ ﺑﯽ ﺣﻮﺻﻠﮕﯽ ﺍﺳﺖ.
