ﺩﺭﺱ ﺷﺸﻢ- ﻣﻮﺍﻧﻊ ﺗﻔﮑﺮ ﺧﻼﻕ

ﻧﺤﻮﻩ ﺑﺮﺧﻮﺭﺩ ﻭ ﻧﮕﺮﺵ ﻣﺎ ﻧﺴﺒﺖ ﺑﻪ ﻣﺸﮑﻼﺕ ﻭ ﭼﺎﻟﺶ ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺑﺎ ﺁﻧﻬﺎ ﺑﺮﺧﻮﺭﺩ ﻣﯽ ﮐﻨﻴﻢ، ﻣﯽ ﺗﻮﺍﻧﺪ ﺧﻮﺩ ﻣﺎﻧﻌﯽ ﺑﺮﺍﯼ ﺧﻼﻕ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﺎﺷﺪ.

 

ﺍﻧﮑﺎﺭ ﻣﺸﮑﻼﺕ
ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﺑﻪ ﻣﺸﮑﻼﺕ ﻣﻌﻤﻮﻻ ﺑﺰﺭﮒ ﺗﺮ ﺍﺯ ﺧﻮﺩ ﻣﺸﮑﻞ ﻫﺴﺘﻨﺪ. ﺑﻴﺸﺘﺮ ﻣﺮﺩﻡ ﻭﺟﻮﺩ ﻣﺸﮑﻞ ﺭﺍ ﺗﺎ ﺟﺎﻳﯽ ﮐﻪ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﭼﺎﺭﻩ ﺟﻮﻳﯽ ﻭ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﺧﻴﻠﯽ ﺩﻳﺮ ﺷﺪﻩ، ﻧﺎﺩﻳﺪﻩ ﻣﯽ ﮔﻴﺮﻧﺪ ﻳﺎ ﺍﻧﮑﺎﺭ ﻣﯽ ﮐﻨﻨﺪ. ﻭﻟﯽ ﻋﺪﻩ ﺍﯼ ﻫﻢ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﺘﻔﺎﻭﺕ، ﻧﮕﺎﻩ ﻭ ﻓﮑﺮ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺑﻪ ﻣﺸﮑﻼﺕ ﺑﻪ ﻋﻨﻮﺍﻥ ﭼﺎﻟﺶ ﻭ ﻓﺮﺻﺘﯽ ﺑﺮﺍﯼ ﺑﻬﺘﺮ ﮐﺮﺩﻥ ﮐﺎﺭﻫﺎ ﻧﮕﺎﻩ ﻣﯽ ﮐﻨﻨﺪ.

ﺑﺮﺍﯼ ﮐﺴﯽ ﮐﻪ ﺩﻧﺒﺎﻝ ﺭﻭﺵ ﻫﺎ ﻭ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﻧﻮ ﻭ ﺧﻼﻗﺎﻧﻪ ﺍﺳﺖ، ﺗﻌﺮﻳﻒ ﻣﺸﮑﻞ ﺍﻳﻦ ﺍﺳﺖ:

۱- ﺩﻳﺪﻥ ﺗﻔﺎﻭﺕ ﺑﻴﻦ ﺁﻧﭽﻪ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﺩﺍﺭﺩ ﻭ ﺁﻧﭽﻪ ﻣﯽ ﺧﻮﺍﻫﺪ.

۲- ﭘﺬﻳﺮﺵ ﻭ ﺑﺎﻭﺭ ﺍﻳﻨﮑﻪ ﺷﺮﺍﻳﻄﯽ ﺑﻬﺘﺮ ﺍﺯ ﺷﺮﺍﻳﻂ ﻣﻮﺟﻮﺩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

۳- ﻓﺮﺻﺘﯽ ﺑﺮﺍﯼ ﻋﮑﺲ ﺍﻟﻌﻤﻠﯽ ﻣﺜﺒﺖ

 

ﻧﻤﯽ ﺷﻪ ﮐﺎﺭﯼ ﮐﺮﺩ
ﺍﻳﻦ ﻧﻮﻉ ﻧﮕﺎﻩ ﮐﻪ ﺑﺮﺍﯼ ﻣﺸﮑﻞ، ﻗﺪﺭﺕ ﺑﯽ ﺣﺪﯼ (ﮐﻪ ﺣﻘﻴﻘﺖ ﻫﻢ ﻧﺪﺍﺭﺩ)  ﻗﺎﺋﻞ ﺷﻮﻳﻢ ﮐﻪ ﺩﻳﮕﺮ ﻫﻴﭻ ﺭﺍﻩ ﺣﻠﯽ ﺑﺮ ﺁﻥ ﮐﺎﺭﮔﺮ ﻧﻴﺴﺖ، ﺟﺎﻳﯽ ﺑﺮﺍﯼ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﺷﻴﻮﻩ ﻫﺎ ﻭ ﺭﻭﺵ ﻫﺎﯼ ﻧﻮ ﻭ ﺟﺪﻳﺪ ﻧﻤﯽ ﮔﺬﺍﺭﺩ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺍﮔﺮ ﺑﺸﺮ ﺍﺯ ﺍﺑﺘﺪﺍﯼ ﺗﺎﺭﻳﺦ ﺍﻳﻦ ﻃﻮﺭ ﺑﻪ ﻣﺴﺎﺋﻞ ﻧﮕﺎﻩ ﻣﯽ ﮐﺮﺩ، ﻫﻴﭻ ﺩﺭﻣﺎﻧﯽ ﺑﺮﺍﯼ ﻫﻴﭻ ﺑﻴﻤﺎﺭﯼ ﻻﻋﻼﺟﯽ ﭘﻴﺪﺍ ﻧﻤﯽ ﺷﺪ ﻭ ﻫﻤﻪ ﺑﻴﻤﺎﺭﯼ ﻫﺎ ﻻﻋﻼﺝ ﺑﺎﻗﯽ ﻣﯽ ﻣﺎﻧﺪﻧﺪ، ﺍﻧﺴﺎﻥ ﻫﻴﭻ ﻭﻗﺖ ﭘﺮﻭﺍﺯ ﻧﻤﯽ ﮐﺮﺩ ﻭ... .

 

ﺍﺯ ﺩﺳﺖ ﻣﻦ ﮐﻪ ﮐﺎﺭﯼ ﺑﺮ ﻧﻤﯽ ﺁﻳﺪ
ﺑﻌﻀﯽ ﺍﺯ ﺁﺩﻡ ﻫﺎ ﻓﮑﺮ ﻣﯽ ﮐﻨﻨﺪ، ﺑﺮﺍﯼ ﺣﻞ ﻳﮏ ﻣﺸﮑﻞ ﺣﺘﻤﺎ ﺑﺎﻳﺪ ﻳﮏ ﻣﺘﺨﺼﺺ، ﻳﮏ ﻧﺨﺒﻪ ﻭ... ﺑﺎﺷﻨﺪ. ﺍﻳﻨﺠﺎ ﺑﺎﺯ ﻫﻢ ﺭﺟﻮﻉ ﺑﻪ ﺗﺎﺭﻳﺦ ﭘﻴﺸﺮﻓﺖ ﺍﺧﺘﺮﺍﻋﺎﺕ ﺑﺸﺮﯼ ﻻﺯﻡ ﺍﺳﺖ.

ﺑﺮﺍﺩﺭﺍﻥ ﺭﺍﻳﺖ ﮐﻪ ﻋﮑﺲ ﺍﻭﻟﻴﻦ ﻣﺎﺷﻴﻦ ﭘﺮﻧﺪﻩ ﺍﯼ ﺭﺍ ﮐﻪ ﺳﺎﺧﺘﻨﺪ ﺩﺭ ﺑﺎﻻ ﺩﻳﺪﻳﺪ، ﻣﻬﻨﺪﺱ ﻃﺮﺍﺣﯽ ﻭ ﺳﺎﺧﺖ ﻫﻮﺍﭘﻴﻤﺎ ﺑﻮﺩﻧﺪ؟ ﻧﻪ ﺁﻧﻬﺎ ﺗﻌﻤﻴﺮﮐﺎﺭ ﺩﻭﭼﺮﺧﻪ ﺑﻮﺩﻧﺪ. ﺳﺎﺯﻧﺪﻩ ﺧﻮﺩﮐﺎﺭ ﻻﺩﻳﺴﻼﺋﻮ ﺑﻴﺮﻭ، ﻣﻬﻨﺪﺱ ﻣﮑﺎﻧﻴﮏ ﻧﺒﻮﺩ، ﺍﻭ ﻧﺴﺨﻪ ﺧﻮﺍﻥ ﺑﻮﺩ. ﭘﺲ ﺍﻳﻦ ﺩﻭ ﺗﺼﻮﻳﺮ ﻭ ﻣﺜﺎﻝ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﺍﮔﺮ ﺍﻳﻦ ﺁﺩﻡ ﻫﺎ ﻫﻢ ﺍﻳﻨﻄﻮﺭ ﻓﮑﺮ ﻣﯽ ﮐﺮﺩﻧﺪ، ﻧﻪ ﺧﻮﺩﮐﺎﺭﯼ ﺑﺮﺍﯼ ﻧﻮﺷﺘﻦ ﺩﺍﺷﺘﻴﻢ ﻭ ﻧﻪ ﻫﻴﭻ ﻭﻗﺖ ﭘﺮﻭﺍﺯ ﺑﺎ ﻫﻮﺍﭘﻴﻤﺎ ﺭﺍ ﺗﺠﺮﺑﻪ ﻣﯽ ﮐﺮﺩﻳﻢ.

ﺗﺎﺭﻳﺦ ﺍﺧﺘﺮﺍﻋﺎﺕ ﻭ ﻧﻮﺁﻭﺭﯼ ﻫﺎﯼ ﺍﻧﺴﺎﻥ ﭘﺮ ﺍﺳﺖ ﺍﺯ ﺍﻳﻦ ﺩﺍﺳﺘﺎﻥ ﻫﺎ.

 

ﻣﻦ ﺧﻼﻕ ﻧﻴﺴﺘﻢ
ﺍﻟﺒﺘﻪ ﺷﻤﺎ ﭘﺲ ﺍﺯ ﮔﺬﺭﺍﻧﺪﻥ ﺩﺭﺱ ﻫﺎﯼ ﻗﺒﻠﯽ ﻭ ﻳﺎﺩﮔﺮﻓﺘﻦ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺭﻭﺵ ﻫﺎ، ﺩﻳﮕﺮ ﺍﻳﻦ ﻃﻮﺭ ﻓﮑﺮ ﻧﻤﯽ ﮐﻨﻴﺪ (ﺍﻣﻴﺪﻭﺍﺭﻡ ﮐﻪ ﺍﻳﻦ ﻃﻮﺭ ﻓﮑﺮ ﻧﮑﻨﻴﺪ)، ﻭﻟﯽ ﻃﺮﺡ ﺍﻳﻦ ﻧﻮﻉ ﻧﮕﺮﺵ ﻫﺎ ﮐﻪ ﻣﺎﻧﻊ ﺑﺎﺭﻭﺭ ﺷﺪﻥ ﻭ ﺑﺮﻭﺯ ﺧﻼﻗﻴﺖ ﻫﺴﺘﻨﺪ، ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﺍﮔﺮ ﺟﺎﻳﯽ ﺍﺣﺴﺎﺱ ﮐﺮﺩﻳﻢ ﺗﮑﻨﻴﮏ ﻫﺎ ﻭ ﺷﻴﻮﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ ﺑﺮﺍﯼ ﻣﺎ ﮐﺎﺭ ﻧﻤﯽ ﮐﻨﺪ، ﺭﻳﺸﻪ ﺁﻥ ﺭﺍ ﺑﻬﺘﺮ ﺑﺸﻨﺎﺳﻴﻢ.

ﻣﺎ ﺗﺎ ﺍﻳﻨﺠﺎ ﺑﺎﺭﻫﺎ ﻭ ﺑﺎﺭﻫﺎ ﺧﻮﺍﻧﺪﻳﻢ ﮐﻪ ﺗﻤﺎﻡ ﻣﺎ ﺑﺎ ﺗﻤﺮﻳﻦ ﻭ ﺗﮑﺮﺍﺭ ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﻔﮑﺮ ﺧﻼﻕ ﻭ ﺑﺎ ﺗﻐﻴﻴﺮ ﻧﻮﻉ ﻧﮕﺎﻩ ﻣﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﻣﺸﮑﻼﺕ ﻭ ﭼﺎﻟﺶ ﻫﺎﯼ ﺍﻃﺮﺍﻑ ﻣﺎﻥ، ﻗﺎﺩﺭ ﺧﻮﺍﻫﻴﻢ ﺑﻮﺩ «ﻃﺮﺣﯽ ﻧﻮ ﺩﺭﺍﻧﺪﺍﺯﻳﻢ». ﻫﻤﭽﻨﺎﻥ ﻓﻬﻤﻴﺪﻳﻢ، ﺧﻼﻗﻴﺖ ﻣﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﺷﻴﻮﻩ ﻫﺎﯼ ﺗﻔﮑﺮ ﺳﻨﺘﯽ، ﻋﺎﺩﺕ ﻫﺎ ﻭ ﺍﻟﮕﻮﻫﺎﯼ ﻣﺎ ﺑﺮﺍﯼ ﻓﮑﺮ ﮐﺮﺩﻥ، ﺁﻣﻮﺯﻩ ﻫﺎﯼ ﺁﻣﻮﺯﺵ ﺳﻨﺘﯽ ﺳﺮﮐﻮﺏ ﺷﺪﻩ ﺑﺎﺷﺪ.

ﺗﻤﺎﻡ ﺁﻧﭽﻪ ﺍﻧﺘﻈﺎﺭ ﻣﯽ ﺭﻭﺩ، ﭘﺲ ﺍﺯ ﮔﺬﺭﺍﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﻳﻢ، ﺍﻳﺠﺎﺩ ﻓﻀﺎﻳﯽ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺑﺮﻭﺯ ﺧﻼﻗﻴﺖ ﻭ ﻳﺎﺩ ﮔﺮﻓﺘﻦ ﺷﻴﻮﻩ ﻫﺎ ﻭ ﺗﮑﻨﻴﮏ ﻫﺎﯼ ﺍﺣﻴﺎﯼ ﺁﻥ ﺍﺳﺖ.

 

ﺗﺮﺱ ﺍﺯ ﺷﮑﺴﺖ
ﺗﻮﻣﺎﺱ ﺍﺩﻳﺴﻮﻥ، ﺑﺮﺍﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻣﻨﺎﺳﺐ ﺗﺮﻳﻦ ﺭﺷﺘﻪ، ﺑﺮﺍﯼ ﺳﺎﺧﺖ ﻻﻣﭗ ﺭﺷﺘﻪ ﺍﯼ، ﻫﺮ ﭼﻴﺰﯼ ﮐﻪ ﻓﮑﺮ ﮐﻨﻴﺪ ﺭﺍ ﺍﻣﺘﺤﺎﻥ ﮐﺮﺩ، ﺣﺘﯽ ﺗﺎﺭﻫﺎﻳﯽ ﺍﺯ ﺭﻳﺶ ﺩﻭﺳﺘﺶ ﺭﺍ. ﺍﻭ ﺩﺭ ﻧﻬﺎﻳﺖ ۱۸۰۰ ﺑﺎﺭ ﺗﻼﺵ ﮐﺮﺩ ﭘﺲ ﺍﺯ ﻣﺮﺗﺒﻪ ﻫﺰﺍﺭﻡ ﮐﺴﯽ ﺍﺯ ﺍﻭ ﭘﺮﺳﻴﺪ، ﺧﺴﺘﻪ ﻧﺸﺪﯼ ﺍﺯ ﺍﻳﻦ ﻫﻤﻪ ﺗﻼﺵ ﺑﯽ ﺛﻤﺮ؟ ﺟﻮﺍﺏ ﺍﺩﻳﺴﻮﻥ ﺍﻳﻦ ﺑﻮﺩ «ﮐﻠﯽ ﺍﻃﻼﻋﺎﺕ ﻭ ﺩﺍﻧﺶ ﮐﺴﺐ ﮐﺮﺩﻡ، ﻓﻬﻤﻴﺪﻡ ۱۰۰۰ ﻣﺎﺩﻩ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﻣﻨﺎﺳﺐ ﻧﻴﺴﺘﻨﺪ.»

ﺗﺮﺱ ﺍﺯ ﺷﮑﺴﺖ ﻭ ﻋﺪﻡ ﻣﻮﻓﻘﻴﺖ ﻳﮑﯽ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻳﻦ ﻣﻮﺍﻧﻊ ﺑﺮﻭﺯ ﺧﻼﻗﻴﺖ ﺑﺮﺍﯼ ﺣﻞ ﻣﺸﮑﻼﺕ ﺍﺳﺖ. ﺭﺍﻩ ﺣﻞ ﺁﻥ ﻫﻢ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺩﺭ ﺩﺭﺱ ﻫﺎﯼ ﻗﺒﻠﯽ ﻫﻢ ﺧﻮﺍﻧﺪﻳﻢ ﺗﻐﻴﻴﺮ ﻧﻮﻉ ﻧﮕﺎﻩ ﻭ ﻃﺮﺯ ﺗﻔﮑﺮﻣﺎﻥ ﺍﺳﺖ.

ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ، ﻋﺪﻡ ﻣﻮﻓﻘﻴﺖ ﻫﻴﭻ ﺍﺷﮑﺎﻟﯽ ﮐﻪ ﻧﺪﺍﺭﺩ، ﺩﺭ ﻋﻮﺽ ﺑﺎﻋﺚ ﺣﺮﮐﺖ، ﻣﺒﺎﺭﺯﻩ ﻭ ﺗﻼﺵ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﺑﺴﻴﺎﺭ ﺑﺴﻴﺎﺭ ﺍﺯ ﻣﻨﻔﻌﻞ ﺑﻮﺩﻥ ﺑﻬﺘﺮ ﺍﺳﺖ.

ﻏﻴﺮ ﺍﺯ ﻧﻤﻮﻧﻪ ﻫﺎﯼ ﺑﺎﻻ، ﺑﺎﻭﺭﻫﺎﯼ ﺟﻤﻌﯽ ﻭ ﺳﻨﺘﯽ ﻭ ﭘﻴﺶ ﻓﺮﺽ ﻫﺎﯼ ﻏﺎﻟﺐ ﺩﻳﮕﺮﯼ ﻫﻢ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﺎﻧﻊ ﻣﺘﻔﺎﻭﺕ ﺍﻧﺪﻳﺸﻴﺪﻥ ﻭ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﺭﺍﻩ ﺣﻞ ﻫﺎﯼ ﺑﺪﻳﻊ ﻭ ﺧﻼﻗﺎﻧﻪ ﻫﺴﺘﻨﺪ ﻭ ﺍﺳﺎﺳﺎ ﺭﺍﻩ ﺑﺮﻭﺯ ﺧﻼﻗﻴﺖ ﺭﺍ ﺳﺪ ﻣﯽ ﮐﻨﻨﺪ.

 

ﭘﻴﺶ ﻓﺮﺽ ﻫﺎ
ﻋﻤﻮﻣﺎ، ﻫﺮ ﭼﻪ ﺳﻦ ﻣﺎ ﺑﺎﻻﺗﺮ ﻣﯽ ﺭﻭﺩ، ﺍﻋﺘﻤﺎﺩﻣﺎﻥ ﺑﻪ ﺁﻧﭽﻪ ﺍﺯ ﻗﺒﻞ ﻣﯽ ﺩﺍﻧﻴﻢ ﺑﻴﺸﺘﺮ ﻣﯽ ﺷﻮﺩ. ﺍﻳﻦ ﭘﻴﺶ ﻓﺮﺽ ﻫﺎ ﻣﻌﻤﻮﻻ ﻣﺎ ﺭﺍ ﺍﺯ ﺩﺭﮎ ﺁﻧﭽﻪ ﻓﺮﺍﺗﺮ ﺍﺯ ﺩﺍﻧﺴﺘﻪ ﻫﺎ ﻭ ﺍﺣﺘﻤﺎﻻﺕ ﻭ ﺍﻣﮑﺎﻥ ﻫﺎﻳﯽ ﮐﻪ ﻣﯽ ﺷﻨﺎﺳﻴﻢ ﺑﺎﺷﺪ، ﻋﺎﺟﺰ ﻣﯽ ﮐﻨﺪ. ﺁﻧﻬﺎ ﻣﺎ ﺭﺍ ﺩﺭ ﻣﻘﺎﺑﻞ ﭘﺬﻳﺮﻓﺘﻦ ﺗﻐﻴﻴﺮ ﻭ ﻣﻮﻓﻘﻴﺖ، ﺍﻧﻌﻄﺎﻑ ﻧﺎﭘﺬﻳﺮ ﻣﯽ ﮐﻨﻨﺪ.

ﻣﺜﺎﻝ: ﺑﺮﺍﯼ ﻭﺻﻞ ﮐﺮﺩﻥ ﻗﻄﻌﺎﺕ ﻫﻮﺍﭘﻴﻤﺎ ﭼﻪ ﭼﻴﺰﯼ ﺭﺍﺣﺖ ﺗﺮ ﻭ ﻣﻘﺎﻭﻡ ﺗﺮ ﺍﺯ ﭘﺮﭺ ﻣﯽ ﺗﻮﺍﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ؟ ﭘﺎﺳﺦ ﻋﻠﻢ ﺍﻣﺮﻭﺯ ﺑﻪ ﺍﻳﻦ ﻣﺸﮑﻞ ﭼﺴﺐ ﺍﺳﺖ. ﺧﻮﺏ، ﻣﺎ ﺍﺣﺘﻤﺎﻻ ﻫﻴﭻ ﻭﻗﺖ ﺑﻪ ﭼﺴﺐ ﺑﻪ ﻋﻨﻮﺍﻥ ﮔﺰﻳﻨﻪ ﻣﻨﺎﺳﺒﯽ ﻓﮑﺮ ﻧﻤﯽ ﮐﻨﻴﻢ؛ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﭘﻴﺸﻔﺮﺽ ﻫﺎ ﻭ ﺗﺼﻮﺭ ﻏﺎﻟﺐ ﺍﺯ ﭼﺴﺐ، ﺑﺪﻭﻥ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺍﻳﻨﮑﻪ ﻫﺰﺍﺭﺍﻥ ﻧﻮﻉ ﭼﺴﺐ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻗﻄﻌﺎﺕ ﻓﻠﺰﯼ ﻫﻮﺍﭘﻴﻤﺎ ﺭﺍ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻫﻢ ﻭﺻﻞ ﮐﻨﺪ.

ﻳﮏ ﻣﺜﺎﻝ ﺩﻳﮕﺮ: ﭼﮕﻮﻧﻪ ﻣﯽ ﺗﻮﺍﻥ ﺷﻴﺸﻪ ﻫﺎﯼ ﺿﺪ ﮔﻠﻮﻟﻪ ﺳﺒﮏ ﺗﺮﯼ ﺳﺎﺧﺖ، ﺳﺒﮏ ﺗﺮ ﻭ ﻣﻘﺎﻭﻡ ﺗﺮ.

ﺿﺨﻴﻢ ﮐﺮﺩﻥ ﻗﻄﺮ ﺷﻴﺸﻪ ﺑﻪ ﺷﺪﺕ ﺑﻪ ﻭﺯﻥ ﺷﻴﺸﻪ ﺍﺿﺎﻓﻪ ﻣﯽ ﮐﻨﺪ، ﺭﺍﻩ ﺣﻞ ﺍﻣﺮﻭﺯﯼ ﺍﻳﻦ ﻣﺴﺌﻠﻪ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭘﻼﺳﺘﻴﮏ ﺍﺳﺖ. ﺑﺎﺯ ﻫﻢ ﻣﻤﮑﻦ ﺍﺳﺖ ﺏi ﻧﻈﺮ ﺑﻴﺸﺘﺮ ﻣﺎ ﺑﻪ ﺩﻟﻴﻞ ﭘﻴﺸﻔﺮﺽ ﻫﺎﯼ ﻗﺒﻠﯽ ﻭ ﺗﻌﺮﻳﻒ ﻏﺎﻟﺐ ﺍﺯ ﭘﻼﺳﺘﻴﮏ ﻭ ﻣﻘﺎﻭﻣﺖ ﺁﻥ، ﭘﻼﺳﺘﻴﮏ ﮔﺰﻳﻨﻪ ﻣﻨﺎﺳﺒﯽ ﻧﺒﺎﺷﺪ!

ﻳﮏ ﻣﺜﺎﻝ ﺳﺎﺩﻩ ﺗﺮ: ﭼﻄﻮﺭ ﮐﻴﮑﯽ ﺭﺍ ﺑﻴﻦ ﺩﻭ ﺑﭽﻪ ﻣﺴﺎﻭﯼ ﺗﻘﺴﻴﻢ ﮐﻨﻴﻢ ﮐﻪ ﻫﺮ ﺩﻭ ﺍﺯ ﺳﻬﻢ ﺷﺎﻥ ﺭﺍﺿﯽ ﺑﺎﺷﻨﺪ ﻭ ﻫﺮﮐﺪﺍﻡ ﺑﺮﺵ ﺩﻳﮕﺮﯼ ﺭﺍ ﻧﺨﻮﺍﻫﻨﺪ. ﻳﮏ ﺭﺍﻩ ﺣﻞ ﺷﺎﻳﺪ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﺷﻤﺎ ﺗﮑﻪ ﺑﺰﺭﮒ ﺗﺮﯼ ﺑﻪ ﺑﭽﻪ ﺍﯼ ﺑﺪﻫﻴﺪ ﮐﻪ ﻣﯽ ﺩﺍﻧﻴﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﺳﻬﻤﺶ ﺭﺍﺿﯽ ﻧﺒﺎﺷﺪ!

ﭼﺮﺍ ﻭﻇﻴﻔﻪ ﺗﻘﺴﻴﻢ ﮐﻴﮏ ﺭﺍ ﺑﻪ ﻫﻤﺎﻥ ﺑﭽﻪ ﻧﻤﯽ ﺳﭙﺎﺭﻳﻢ؟ ﺍﻳﻦ ﭘﻴﺶ ﻓﺮﺽ ﮐﻪ ﺑﭽﻪ ﺍﯼ ﮐﻪ ﺯﻳﺎﺩﻩ ﺧﻮﺍﻩ ﺗﺮ ﺍﺳﺖ، ﻧﻤﯽ ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. ﺍﻳﻦ ﻗﻀﺎﻭﺕ ﻭ ﭘﻴﺶ ﻓﺮﺽ ﻗﺒﻠﯽ ﺑﺎﻋﺚ ﻣﯽ ﺷﻮﺩ ﻣﺎ ﺍﺣﺘﻤﺎﻻﺕ ﻭ ﺷﻴﻮﻩ ﻫﺎﯼ ﺩﻳﮕﺮ ﺭﺍ ﻧﺒﻴﻨﻴﻢ.

 

ﭼﺴﺒﻴﺪﻥ ﺑﻪ ﺗﻌﺮﻳﻒ ﻏﺎﻟﺐ
ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎ، ﺑﻪ ﺍﺷﻴﺎﯼ ﺍﻃﺮﺍﻑ ﻣﺎﻥ ﻓﻘﻂ ﺑﺮﺍﺳﺎﺱ ﻧﺎﻡ ﺁﻧﻬﺎ ﻧﮕﺎﻩ ﻣﯽ ﮐﻨﻴﻢ، ﻧﻪ ﮐﺎﺭﯼ ﮐﻪ ﺍﺯ ﺁﻧﻬﺎ ﺑﺮ ﻣﯽ ﺁﻳﺪ. ﻣﺜﻼ ﺍﻳﻨﮑﻪ ﺗﻤﻴﺰﮐﺎﺭﯼ ﻳﺎ ﺑﺎﺯ ﻭ ﺑﺴﺘﻪ ﮐﺮﺩﻥ ﺩﺭ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﺣﺮﮐﺎﺕ ﻭﺭﺯﺷﯽ ﺍﻧﺠﺎﻡ ﺷﻮﺩ.

ﺩﺭ ﺻﻨﻌﺖ ﻭ ﺑﺎﺯﺍﺭ ﮐﺎﺭ ﻫﻢ ﻫﻤﻴﻦ ﻃﻮﺭ ﺍﺳﺖ. ﺩﺭ ﺍﻭﺧﺮ ﻗﺮﻥ ﻧﻮﺯﺩﻫﻢ ﻭ ﺍﻭﺍﻳﻞ ﺑﻴﺴﺘﻢ، ﺭﺍﻩ ﺁﻫﻦ ﺧﻮﺩ ﺭﺍ ﻓﻘﻂ ﺭﺍﻩ ﺁﻫﻦ ﻣﻴﺪﻳﺪ. ﺑﺎﺏ ﺷﺪﻥ ﺍﺗﻮﻣﺒﻴﻞ ﻭ ﺑﻌﺪ ﺍﺯ ﺁﻥ ﻫﻢ ﻫﻮﺍﭘﻴﻤﺎ، ﺭﺑﻄﯽ ﺑﻪ ﺁﻧﻬﺎ ﻧﺪﺍﺷﺖ. ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺑﺨﺸﯽ ﺍﺯ ﺳﻴﺴﺘﻢ ﺣﻤﻞ ﻭ ﻧﻘﻞ ﻋﻤﻮﻣﯽ ﻧﻤﯽ ﺩﻳﺪﻧﺪ، ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺭﻫﺎ ﮐﺮﺩﻥ ﺗﻌﺎﺭﻳﻒ ﻏﺎﻟﺐ، ﻣﯽ ﺗﻮﺍﻧﺴﺖ ﻓﺮﺻﺖ ﻃﻼﻳﯽ ﻭ ﺳﻮﺩﺁﻭﺭﯼ ﺑﺮﺍﯼ ﺻﻨﻌﺖ ﺭﺍﻩ ﺁﻫﻦ ﺩﺭ ﺁﻥ ﺑﺎﺯﻩ ﺯﻣﺎﻧﯽ ﺑﺎﺷﺪ.

ﻣﺜﺎﻝ ﺩﻳﮕﺮ: ﺻﻨﻌﺖ ﻭ ﺑﺎﺯﺍﺭ، ﺗﻠﮕﺮﺍﻑ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﻧﮕﺮﻓﺖ. ﺑﺎ ﺑﺎﺏ ﺷﺪﻥ ﺗﻠﻔﻦ، ﺻﻨﻌﺖ ﺗﻠﮕﺮﺍﻑ ﺧﻮﺩ ﺭﺍ ﺑﺨﺸﯽ ﺍﺯ ﺑﺎﺯﺍﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ ﻧﺪﻳﺪ ﻭ ﺩﺭ ﻗﺎﻟﺐ ﺗﻌﺮﻳﻒ ﻭ ﺭﺍﻳﺞ ﺍﺯ ﺗﻠﮕﺮﺍﻑ ﻭ ﮐﺎﺭﮐﺮﺩ ﺁﻥ ﻣﺎﻧﺪ. ﻧﺘﻴﺠﻪ ﺍﻳﻦ ﺷﺪ ﮐﻪ ﺑﻪ ﺟﺎﯼ ﺭﺷﺪ، ﺍﻳﻦ ﺑﺎﺯﺍﺭ ﺍﺯ ﺑﻴﻦ ﺭﻓﺖ ﻭ ﻣﺮﺩ.

ﻣﺸﺎﺑﻪ ﻫﻤﻴﻦ ﺍﺗﻔﺎﻕ ﺩﺭ ﻣﻮﺭﺩ ﺁﺩﻡ ﻫﺎ ﻫﻢ ﻣﯽ ﺍﻓﺘﺪ، ﺑﻪ ﻟﺤﻈﺎﺗﯽ ﻓﮑﺮ ﮐﻨﻴﺪ، ﭼﻄﻮﺭ ﺍﻃﺮﺍﻓﻴﺎﻥ ﺗﺎﻥ ﺭﺍ ﺑﺎ ﻧﻘﺶ ﻫﺎ ﻭ ﺗﻌﺎﺭﻳﻒ ﺭﺍﻳﺞ ﺳﻨﺘﯽ ﻭ ﻏﺎﻟﺐ ﺩﺭ ﺟﺎﻣﻌﻪ ﻗﻀﺎﻭﺕ ﮐﺮﺩﻩ ﺍﻳﺪ ﻭ ﺗﻮﺍﻧﺎﻳﯽ ﻫﺎﯼ ﺑﺎﻟﻘﻮﻩ ﺁﻧﻬﺎ ﺭﺍ ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺘﻪ ﺍﻳﺪ.

ﺩﺭ ﺧﺎﻧﻪ ﻭ ﻣﺤﻞ ﮐﺎﺭ، ﺁﻳﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﺩﻋﺎ ﮐﻨﻴﺪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻧﺸﺪﻩ، ﺗﻮﺍﻧﺎﻳﯽ ﻫﺎﯼ ﮐﻢ ﺳﻦ ﺗﺮ ﺭﺍ ﻧﺒﻴﻨﻴﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﻧﺎﺩﻳﺪﻩ ﺑﮕﻴﺮﻳﺪ، ﭼﻮﻥ ﺩﺭ ﺑﺎﻭﺭ ﺳﻨﺘﯽ ﻭ ﺗﻌﺎﺭﻳﻒ ﻭ ﻧﻘﺶ ﻫﺎﯼ ﺳﻨﺘﯽ ﻏﺎﻟﺐ ﺁﻧﻬﺎ «ﺑﭽﻪ ﺍﻧﺪ»، «ﭘﺨﺘﻪ ﻧﻴﺴﺘﻨﺪ»، «ﻋﻘﻞ ﺷﺎﻥ ﻧﻤﯽ ﺭﺳﺪ» ﻭ... .

ﻳﺎ ﺗﻮﺍﻧﺎﻳﯽ ﻫﺎﯼ ﺑﺎﻟﻘﻮﻩ ﺍﻓﺮﺍﺩ ﺭﺍ ﺍﺯ ﺭﻭﯼ ﻧﻘﺶ ﻫﺎﯼ ﺟﻨﺴﻴﺘﯽ ﻏﺎﻟﺒﯽ ﮐﻪ ﺟﺎﻣﻌﻪ ﺑﻪ ﻃﻮﺭ ﺳﻨﺘﯽ ﺑﻪ ﺍﻓﺮﺍﺩ ﺗﺤﻤﻴﻞ ﻣﯽ ﮐﻨﺪ، ﻗﻀﺎﻭﺕ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ، ﺑﺮﺍﯼ ﺍﺳﺘﺨﺪﺍﻡ ﻳﺎ ﺗﻘﺴﻴﻢ ﮐﺎﺭ ﻭ ﻃﺮﻑ ﻣﺸﻮﺭﺕ ﻗﺮﺍﺭ ﺩﺍﺩﻥ «ﺯﻥ ﺑﺮﺍﯼ ﻧﺎﻣﻪ ﺭﺳﺎﻧﯽ؟»، «ﺯﻥ ﻫﺎ ﻣﺪﻳﺮﻫﺎﯼ ﺧﻮﺑﯽ ﻧﻴﺴﺘﻨﺪ»، «ﻣﺮﺩﺍ! ﻣﺮﺩ ﮐﻪ ﺍﺣﺴﺎﺱ ﻧﺪﺍﺭﻩ، ﻧﻤﯽ ﺷﻪ ﻣﺮﺑﯽ ﺑﭽﻪ ﻣﺮﺩ ﺑﺎﺷﻪ» ﻭ... .

ﻫﺮﮔﻮﻧﻪ ﭘﻴﺶ ﺩﺍﻭﺭﯼ، ﻣﻮﺿﻊ ﮔﻴﺮﯼ، ﺗﻌﻤﻴﻢ ﺩﺍﺩﻥ ﻭ ﻋﻤﻮﻣﯽ ﮐﺮﺩﻥ ﻭ ﻳﺎ ﻗﻀﺎﻭﺕ ﺍﺯ ﺭﻭﯼ ﺗﺠﺮﺑﻪ ﻫﺎﯼ ﺷﺨﺼﯽ ﺩﺭ ﻣﻮﺭﺩ ﺍﻓﺮﺍﺩ، ﻣﺎ ﺭﺍ ﺍﺯ ﺩﻳﺪﻥ ﻭ ﺩﺭﮎ ﺗﻮﺍﻧﺎﻳﯽ ﻫﺎﯼ ﺍﻃﺮﺍﻓﻴﺎﻥ ﻣﺎﻥ ﺑﺎﺯ ﻣﯽ ﺩﺍﺭﺩ؛ ﺩﺭ ﻫﺮ ﺣﻮﺯﻩ ﺍﯼ ﭼﻪ ﺷﺨﺼﯽ ﻳﺎ ﻋﻤﻮﻣﯽ، ﺍﺟﺎﺯﻩ ﻧﻤﯽ ﺩﻫﺪ ﻣﺎ ﻓﺮﺍﺗﺮ ﺍﺯ ﺑﺎﻭﺭﻫﺎﯼ ﺳﻨﺘﯽ ﻏﺎﻟﺐ ﻭ ﻧﻘﺶ ﻫﺎﯼ ﺭﺍﻳﺞ ﺟﺎﻣﻌﻪ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺗﻌﺎﻣﻞ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

 

ﻭﺍﺑﺴﺘﮕﯽ ﺑﻪ ﺩﻳﮕﺮﺍﻥ
ﺍﻳﻦ ﺣﺲ ﮐﻪ ﻫﻴﭻ ﺍﺑﺰﺍﺭ ﻭ ﺍﻣﮑﺎﻧﺎﺗﯽ ﺩﺭ ﺩﺳﺖ ﻧﺪﺍﺭﻳﻢ، ﭘﺲ ﺑﻬﺘﺮ ﺍﺳﺖ ﺳﻌﯽ ﻫﻢ ﻧﮑﻨﻴﻢ! ﺩﺭ ﺁﻣﻮﺯﺵ ﺳﻨﺘﯽ ﻣﺎ ﺑﻴﺸﺘﺮ ﻳﺎﺩ ﮔﺮﻓﺘﻪ ﺍﻳﻢ ﺗﻘﺮﻳﺒﺎ ﺑﺮﺍﯼ ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻭﺍﺑﺴﺘﻪ ﺑﺎﺷﻴﻢ ﺗﺎ ﺧﻮﺩﻣﺎﻥ. ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﻣﺤﺪﻭﺩ ﻭ ﻧﺎﺗﻮﺍﻥ ﺑﺪﺍﻧﻴﻢ. ﺑﻪ ﻃﻮﺭ ﺳﻨﺘﯽ ﺑﻴﺸﺘﺮ ﺩﺭ ﻣﻮﺭﺩ ﻣﺤﺪﻭﺩﻳﺖ ﻫﺎ ﻭ ﻧﺎﺗﻮﺍﻧﯽ ﻫﺎ ﺧﻮﺍﻧﺪﻩ ﻭ ﺷﻨﻴﺪﻩ ﺍﻳﻢ.

ﺑﻠﻪ ﻣﺤﺪﻭﺩﻳﺖ ﻭ ﻧﺎﺗﻮﺍﻧﯽ ﻫﺴﺖ، ﻭﻟﯽ ﺍﺑﺰﺍﺭ ﻫﻢ ﮐﻢ ﻧﻴﺴﺖ؛ ﺍﮔﺮ ﺑﻪ ﺍﻃﻼﻋﺎﺗﯽ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻳﺪ، ﺍﺯ ﺍﻳﻨﺘﺮﻧﺖ، ﮐﺘﺎﺏ ﻫﺎ، ﺩﻭﺳﺘﺎﻥ ﻭ... ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﮔﺮ ﮐﺎﺭﯼ ﺭﺍ ﺑﻠﺪ ﻧﻴﺴﺘﻴﺪ، ﻳﺎﺩ ﺑﮕﻴﺮﻳﺪ؛ ﺁﺷﭙﺰﯼ، ﺣﺴﺎﺑﺪﺍﺭﯼ، ﺑﺎﻏﺒﺎﻧﯽ، ﮐﺎﺭ ﺑﺎ ﺍﻳﻨﺘﺮﻧﺖ، ﺧﻼﻗﻴﺖ ﻭ... .

ﻣﻮﺍﻧﻊ ﺭﻭﺍﻧﺸﻨﺎﺧﺘﯽ
ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺭﺍﻩ ﺣﻞ ﻫﺎ ﺻﺮﻓﺎ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ «ﻣﻨﺎﺳﺐ» ﻳﺎ «ﺧﻮﺏ» ﺑﻪ ﻧﻈﺮ ﻧﻤﯽ ﺭﺳﻨﺪ، ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺘﻪ ﻣﯽ ﺷﻮﻧﺪ ﻳﺎ ﺭﺩ ﻣﯽ ﺷﻮﻧﺪ. ﻭﻟﯽ ﻫﻤﻴﻦ ﺭﺍﻩ ﺣﻞ ﻫﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻴﻠﯽ ﻫﻢ ﮐﺎﺭﺁﻣﺪ ﺑﺎﺷﻨﺪ. ﺧﻮﺭﺩﻥ ﻣﺎﺭﻣﻮﻟﮏ ﺷﺎﻳﺪ ﺑﻪ ﻧﻈﺮ ﺟﺎﻟﺐ ﻧﺮﺳﺪ، ﻭﻟﯽ ﺩﺭ ﺟﺎﻳﯽ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺍﺯ ﺧﻄﺮ ﻣﺮﮒ ﻭ ﮔﺮﺳﻨﮕﯽ ﻧﺠﺎﺕ ﻣﯽ ﺩﻫﺪ، ﻗﻄﻌﺎ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ ﺣﻞ ﺍﺳﺖ.

ﻣﻮﺍﻧﻊ ﺭﻭﺍﻧﺸﻨﺎﺧﺘﯽ، ﺷﻤﺎ ﺭﺍ ﺍﺯ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻦ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺭﺍﻩ ﺣﻞ ﻫﺎ ﺑﻪ ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ «ﺧﻮﺏ» ﻧﻴﺴﺘﻨﺪ ﻳﺎ ﺍﻳﻨﮑﻪ «ﻣﻨﺎﺳﺐ» ﻧﻴﺴﺘﻨﺪ ﻳﺎ «ﺧﻨﺪﻩ ﺩﺍﺭﻧﺪ» ﺑﺎﺯ ﻣﯽ ﺩﺍﺭﺩ ﻭ ﺷﻤﺎ ﺭﺍ ﻣﺤﺪﻭﺩ ﻣﯽ ﮐﻨﻨﺪ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺩﺭ ﺧﻼﻗﺎﻧﻪ ﻓﮑﺮ ﮐﺮﺩﻥ ﻭ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺷﻴﻮﻩ ﻫﺎﯼ ﻧﻮ، ﻣﺤﺪﻭﺩ ﻧﮑﺮﺩﻥ ﻓﮑﺮ ﻭ ﺫﻫﻦ ﻣﺎﻥ ﺍﺳﺖ.

ﺩﺭ ﺯﻣﺎﻥ ﺟﻨﮓ ﻭﻳﺘﻨﺎﻡ (ﺩﺭ ﺩﻫﻪ ﭘﻨﺠﺎﻩ ﻣﻴﻼﺩﯼ ﻭ ﺩﻫﻪ ﺳﯽ ﺷﻤﺴﯽ) ﻗﻄﻌﺎ ﭘﻮﺷﻴﺪﻥ ﺟﻮﺭﺍﺏ ﺷﻠﻮﺍﺭﯼ ﺯﻧﺎﻧﻪ ﺑﺮﺍﯼ ﻣﺮﺩﺍﻥ «ﺑﺪ» ﺑﻮﺩﻩ، ﻭﻟﯽ ﮐﻤﺎﻧﺪﻭﻫﺎ ﺩﺭ ﺁﻥ ﺟﻨﮓ، ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﺠﺒﻮﺭ ﺑﻮﺩﻧﺪ ﺍﺯ ﺑﺎﺗﻼﻕ ﻫﺎ ﻭ ﺟﻨﮕﻞ ﻫﺎﯼ ﺍﻧﺒﻮﻩ ﺭﺩ ﺷﻮﻧﺪ، ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺍﺯ ﺧﻄﺮ ﺯﺍﻟﻮ ﻭ ﺯﺧﻤﯽ ﺷﺪﻥ ﺑﺎ ﺷﺎﺧﻪ ﻫﺎ ﻭ ﮔﻴﺎﻫﺎﻥ ﺩﺭ ﺍﻣﺎﻥ ﺑﺎﺷﻨﺪ، ﺯﻳﺮ ﺷﻠﻮﺭ ﺟﻴﻦ ﻫﺎﯼ ﺷﺎﻥ، ﺟﻮﺭﺍﺏ ﺷﻠﻮﺍﺭﯼ ﺯﻧﺎﻧﻪ ﭘﺎ ﮐﺮﺩﻧﺪ، ﺑﺎ ﺍﻳﻨﮑﻪ ﺷﺎﻳﺪ ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺧﺘﯽ (ﺩﺭ ﺁﻥ ﺩﻭﺭﺍﻥ) «ﺧﻮﺏ» ﻳﺎ «ﺩﺭﺳﺖ» ﻧﺒﻮﺩﻩ، ﺁﻧﻬﺎ ﺑﺮ ﺍﻳﻦ ﻣﺎﻧﻊ ﻏﻠﺒﻪ ﮐﺮﺩﻧﺪ.
